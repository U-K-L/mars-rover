64 bool NCP5623::hw_init(void)
65 {
66  uint8_t addrs[] = { NCP5623_LED_I2C_ADDR, NCP5623_C_LED_I2C_ADDR };
67  for (uint8_t i=0; i<ARRAY_SIZE(addrs); i++) {
68  // first look for led on external bus
69  _dev = std::move(hal.i2c_mgr->get_device(_bus, addrs[i]));
70  if (!_dev) {
71  continue;
72  }
73 
74  _dev->get_semaphore()->take_blocking();
75  _dev->set_retries(10);
76 
77  // enable the led
78  bool ret = write(NCP5623_LED_ENABLE, 0x1f);
79  if (!ret) {
80  _dev->get_semaphore()->give();
81  continue;
82  }
83 
84  // update the red, green and blue values to zero
85  uint8_t off[3] = { _led_off, _led_off, _led_off };
86  ret = write_pwm(off);
87 
88  _dev->set_retries(1);
89 
90  // give back i2c semaphore
91  _dev->get_semaphore()->give();
92 
93  if (ret) {
94  _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&NCP5623::_timer, void));
95  }
96  return true;
97  }
98 
99  return false;
100 }
