37 // update the state of the sensor
38 void AP_Proximity_MorseSITL::update(void)
39 {
40  SITL::vector3f_array &points = sitl->state.scanner.points;
41  SITL::float_array &ranges = sitl->state.scanner.ranges;
42  if (points.length != ranges.length ||
43  points.length == 0) {
44  set_status(AP_Proximity::Proximity_NoData);
45  return;
46  }
47 
48  set_status(AP_Proximity::Proximity_Good);
49 
50  memset(_distance_valid, 0, sizeof(_distance_valid));
51  memset(_angle, 0, sizeof(_angle));
52  memset(_distance, 0, sizeof(_distance));
53 
54  // only use 8 sectors to match RPLidar
55  const uint8_t nsectors = MIN(8, PROXIMITY_SECTORS_MAX);
56  const uint16_t degrees_per_sector = 360 / nsectors;
57 
58  for (uint16_t i=0; i<points.length; i++) {
59  Vector3f &point = points.data[i];
60  float &range = ranges.data[i];
61  distance_maximum = MAX(distance_maximum, range);
62  if (point.is_zero()) {
63  continue;
64  }
65  float angle_deg = wrap_360(degrees(atan2f(-point.y, point.x)));
66  uint16_t angle_rounded = uint16_t(angle_deg+0.5);
67  uint8_t sector = wrap_360(angle_rounded + 22.5f) / degrees_per_sector;
68  if (!_distance_valid[sector] || range < _distance[sector]) {
69  _distance_valid[sector] = true;
70  _distance[sector] = range;
71  _angle[sector] = angle_deg;
72  update_boundary_for_sector(sector);
73  }
74  }
75 
76 #if 0
77  printf("npoints=%u\n", points.length);
78  for (uint16_t i=0; i<nsectors; i++) {
79  printf("sector[%u] ang=%.1f dist=%.1f\n", i, _angle[i], _distance[i]);
80  }
81 #endif
82 }
