42 // interrupt handler for reading pwm value
43 void AP_RangeFinder_PWM::irq_handler(uint8_t pin, bool pin_high, uint32_t timestamp_us)
44 {
45  if (pin_high) {
46  irq_pulse_start_us = timestamp_us;
47  } else {
48  if (irq_pulse_start_us != 0) {
49  irq_value_us += timestamp_us - irq_pulse_start_us;
50  irq_pulse_start_us = 0;

56 // read - return last value measured by sensor
57 bool AP_RangeFinder_PWM::get_reading(uint16_t &reading_cm)
58 {
59  // disable interrupts and grab state
60  void *irqstate = hal.scheduler->disable_interrupts_save();
61  const uint32_t value_us = irq_value_us;
62  const uint16_t sample_count = irq_sample_count;
63  irq_value_us = 0;
64  irq_sample_count = 0;
65  hal.scheduler->restore_interrupts(irqstate);
66 
67  if (value_us == 0 || sample_count == 0) {
68  return false;
69  }
70  reading_cm = value_us/(sample_count * 10); // correct for LidarLite. Parameter needed? Converts from decimetres -> cm here
71  return true;
72 }

74 void AP_RangeFinder_PWM::check_pin()
75 {
76  if (params.pin == last_pin) {
77  return;
78  }
79 
80  // detach last one
81  if (last_pin > 0) {
82  if (!hal.gpio->detach_interrupt(last_pin)) {
83  gcs().send_text(MAV_SEVERITY_WARNING,
84  "RangeFinder_PWM: Failed to detach from pin %u",
85  last_pin);
86  // ignore this failure or the user may be stuck
87  }
88  }
89 
90  // set last pin to params.pin so we don't continually try to attach
91  // to it if the attach is failing
92  last_pin = params.pin;
93 
94  if (params.pin <= 0) {
95  // don't need to install handler
96  return;
97  }
98 
99  // install interrupt handler on rising and falling edge
100  hal.gpio->pinMode(params.pin, HAL_GPIO_INPUT);
101  if (!hal.gpio->attach_interrupt(
102  params.pin,
103  FUNCTOR_BIND_MEMBER(&AP_RangeFinder_PWM::irq_handler,
104  void,
105  uint8_t,
106  bool,
107  uint32_t),
108  AP_HAL::GPIO::INTERRUPT_BOTH)) {
109  // failed to attach interrupt
110  gcs().send_text(MAV_SEVERITY_WARNING,
111  "RangeFinder_PWM: Failed to attach to pin %u",
112  params.pin);
113  return;
114  }
115 }

135 /*
136  update the state of the sensor
137 */
138 void AP_RangeFinder_PWM::update(void)
139 {
140  // check if pin has changed and configure interrupt handlers if required:
141  check_pins();
142 
143  if (last_pin <= 0) {
144  // disabled (by configuration)
145  return;
146  }
147 
148  if (params.stop_pin != -1) {
149  const bool oor = out_of_range();
150  if (oor) {
151  if (!was_out_of_range) {
152  // we are above the power saving range. Disable the sensor
153  hal.gpio->write(params.stop_pin, false);
154  set_status(RangeFinder::RangeFinder_NoData);
155  state.distance_cm = 0;
156  state.voltage_mv = 0;
157  was_out_of_range = oor;
158  }
159  return;
160  }
161  // re-enable the sensor:
162  if (!oor && was_out_of_range) {
163  hal.gpio->write(params.stop_pin, true);
164  was_out_of_range = oor;
165  }
166  }
167 
168  if (!get_reading(state.distance_cm)) {
169  // failure; consider changing our state
170  if (AP_HAL::millis() - state.last_reading_ms > 200) {
171  set_status(RangeFinder::RangeFinder_NoData);
172  }
173  return;
174  }
175 
176  // update range_valid state based on distance measured
177  state.last_reading_ms = AP_HAL::millis();
178  update_status();
179 }
