160 void AC_Loiter::set_pilot_desired_acceleration(float euler_roll_angle_cd, float euler_pitch_angle_cd, float dt)
161 {
162  // Convert from centidegrees on public interface to radians
163  const float euler_roll_angle = radians(euler_roll_angle_cd*0.01f);
164  const float euler_pitch_angle = radians(euler_pitch_angle_cd*0.01f);
165 
166  // convert our desired attitude to an acceleration vector assuming we are hovering
167  const float pilot_cos_pitch_target = constrain_float(cosf(euler_pitch_angle), 0.5f, 1.0f);
168  const float pilot_accel_rgt_cms = GRAVITY_MSS*100.0f * tanf(euler_roll_angle)/pilot_cos_pitch_target;
169  const float pilot_accel_fwd_cms = -GRAVITY_MSS*100.0f * tanf(euler_pitch_angle);
170 
171  // rotate acceleration vectors input to lat/lon frame
172  _desired_accel.x = (pilot_accel_fwd_cms*_ahrs.cos_yaw() - pilot_accel_rgt_cms*_ahrs.sin_yaw());
173  _desired_accel.y = (pilot_accel_fwd_cms*_ahrs.sin_yaw() + pilot_accel_rgt_cms*_ahrs.cos_yaw());
174 
175  // difference between where we think we should be and where we want to be
176  Vector2f angle_error(wrap_PI(euler_roll_angle - _predicted_euler_angle.x), wrap_PI(euler_pitch_angle - _predicted_euler_angle.y));
177 
178  // calculate the angular velocity that we would expect given our desired and predicted attitude
179  _attitude_control.input_shaping_rate_predictor(angle_error, _predicted_euler_rate, dt);
180 
181  // update our predicted attitude based on our predicted angular velocity
182  _predicted_euler_angle += _predicted_euler_rate * dt;
183 
184  // convert our predicted attitude to an acceleration vector assuming we are hovering
185  const float pilot_predicted_cos_pitch_target = cosf(_predicted_euler_angle.y);
186  const float pilot_predicted_accel_rgt_cms = GRAVITY_MSS*100.0f * tanf(_predicted_euler_angle.x)/pilot_predicted_cos_pitch_target;
187  const float pilot_predicted_accel_fwd_cms = -GRAVITY_MSS*100.0f * tanf(_predicted_euler_angle.y);
188 
189  // rotate acceleration vectors input to lat/lon frame
190  _predicted_accel.x = (pilot_predicted_accel_fwd_cms*_ahrs.cos_yaw() - pilot_predicted_accel_rgt_cms*_ahrs.sin_yaw());
191  _predicted_accel.y = (pilot_predicted_accel_fwd_cms*_ahrs.sin_yaw() + pilot_predicted_accel_rgt_cms*_ahrs.cos_yaw());
192 }

235 void AC_Loiter::calc_desired_velocity(float nav_dt)
236 {
237  float ekfGndSpdLimit, ekfNavVelGainScaler;
238  AP::ahrs_navekf().getEkfControlLimits(ekfGndSpdLimit, ekfNavVelGainScaler);
239 
240  // calculate a loiter speed limit which is the minimum of the value set by the LOITER_SPEED
241  // parameter and the value set by the EKF to observe optical flow limits
242  float gnd_speed_limit_cms = MIN(_speed_cms, ekfGndSpdLimit*100.0f);
243  gnd_speed_limit_cms = MAX(gnd_speed_limit_cms, LOITER_SPEED_MIN);
244 
245  float pilot_acceleration_max = GRAVITY_MSS*100.0f * tanf(radians(get_angle_max_cd()*0.01f));
246 
247  // range check nav_dt
248  if (nav_dt < 0) {
249  return;
250  }
251 
252  _pos_control.set_max_speed_xy(gnd_speed_limit_cms);
253  _pos_control.set_max_accel_xy(_accel_cmss);
254  _pos_control.set_leash_length_xy(LOITER_POS_CORRECTION_MAX);
255 
256  // get loiters desired velocity from the position controller where it is being stored.
257  const Vector3f &desired_vel_3d = _pos_control.get_desired_velocity();
258  Vector2f desired_vel(desired_vel_3d.x,desired_vel_3d.y);
259 
260  // update the desired velocity using our predicted acceleration
261  desired_vel.x += _predicted_accel.x * nav_dt;
262  desired_vel.y += _predicted_accel.y * nav_dt;
263 
264  Vector2f loiter_accel_brake;
265  float desired_speed = desired_vel.length();
266  if (!is_zero(desired_speed)) {
267  Vector2f desired_vel_norm = desired_vel/desired_speed;
268 
269  // TODO: consider using a velocity squared relationship like
270  // pilot_acceleration_max*(desired_speed/gnd_speed_limit_cms)^2;
271  // the drag characteristic of a multirotor should be examined to generate a curve
272  // we could add a expo function here to fine tune it
273 
274  // calculate a drag acceleration based on the desired speed.
275  float drag_decel = pilot_acceleration_max*desired_speed/gnd_speed_limit_cms;
276 
277  // calculate a braking acceleration if sticks are at zero
278  float loiter_brake_accel = 0.0f;
279  if (_desired_accel.is_zero()) {
280  if ((AP_HAL::millis()-_brake_timer) > _brake_delay * 1000.0f) {
281  float brake_gain = _pos_control.get_vel_xy_pid().kP() * 0.5f;
282  loiter_brake_accel = constrain_float(AC_AttitudeControl::sqrt_controller(desired_speed, brake_gain, _brake_jerk_max_cmsss, nav_dt), 0.0f, _brake_accel_cmss);
283  }
284  } else {
285  loiter_brake_accel = 0.0f;
286  _brake_timer = AP_HAL::millis();
287  }
288  _brake_accel += constrain_float(loiter_brake_accel-_brake_accel, -_brake_jerk_max_cmsss*nav_dt, _brake_jerk_max_cmsss*nav_dt);
289  loiter_accel_brake = desired_vel_norm*_brake_accel;
290 
291  // update the desired velocity using the drag and braking accelerations
292  desired_speed = MAX(desired_speed-(drag_decel+_brake_accel)*nav_dt,0.0f);
293  desired_vel = desired_vel_norm*desired_speed;
294  }
295 
296  // add braking to the desired acceleration
297  _desired_accel -= loiter_accel_brake;
298 
299  // Apply EKF limit to desired velocity - this limit is calculated by the EKF and adjusted as required to ensure certain sensor limits are respected (eg optical flow sensing)
300  float horizSpdDem = desired_vel.length();
301  if (horizSpdDem > gnd_speed_limit_cms) {
302  desired_vel.x = desired_vel.x * gnd_speed_limit_cms / horizSpdDem;
303  desired_vel.y = desired_vel.y * gnd_speed_limit_cms / horizSpdDem;
304  }
305 
306  // Limit the velocity to prevent fence violations
307  // TODO: We need to also limit the _desired_accel
308  AC_Avoid *_avoid = AP::ac_avoid();
309  if (_avoid != nullptr) {
310  _avoid->adjust_velocity(_pos_control.get_pos_xy_p().kP(), _accel_cmss, desired_vel, nav_dt);
311  }
312 
313  // send adjusted feed forward acceleration and velocity back to the Position Controller
314  _pos_control.set_desired_accel_xy(_desired_accel.x, _desired_accel.y);
315  _pos_control.set_desired_velocity_xy(desired_vel.x, desired_vel.y);
316 }
