175 void AP_InertialSensor_Invensense::start()
176 {
177  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
178  return;
179  }
180 
181  // initially run the bus at low speed
182  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
183 
184  // only used for wake-up in accelerometer only low power mode
185  _register_write(MPUREG_PWR_MGMT_2, 0x00);
186  hal.scheduler->delay(1);
187 
188  // always use FIFO
189  _fifo_reset();
190 
191  // grab the used instances
192  enum DevTypes gdev, adev;
193  switch (_mpu_type) {
194  case Invensense_MPU9250:
195  gdev = DEVTYPE_GYR_MPU9250;
196  adev = DEVTYPE_ACC_MPU9250;
197  break;
198  case Invensense_ICM20602:
199  gdev = DEVTYPE_INS_ICM20602;
200  adev = DEVTYPE_INS_ICM20602;
201  break;
202  case Invensense_ICM20601:
203  gdev = DEVTYPE_INS_ICM20601;
204  adev = DEVTYPE_INS_ICM20601;
205  break;
206  case Invensense_MPU6000:
207  case Invensense_MPU6500:
208  case Invensense_ICM20608:
209  default:
210  gdev = DEVTYPE_GYR_MPU6000;
211  adev = DEVTYPE_ACC_MPU6000;
212  break;
213  case Invensense_ICM20789:
214  gdev = DEVTYPE_INS_ICM20789;
215  adev = DEVTYPE_INS_ICM20789;
216  break;
217  case Invensense_ICM20689:
218  gdev = DEVTYPE_INS_ICM20689;
219  adev = DEVTYPE_INS_ICM20689;
220  break;
221  }
222 
223  /*
224  setup temperature sensitivity and offset. This varies
225  considerably between parts
226  */
227  switch (_mpu_type) {
228  case Invensense_MPU9250:
229  temp_zero = 21.0f;
230  temp_sensitivity = 1.0f/340;
231  break;
232 
233  case Invensense_MPU6000:
234  case Invensense_MPU6500:
235  temp_zero = 36.53f;
236  temp_sensitivity = 1.0f/340;
237  break;
238 
239  case Invensense_ICM20608:
240  case Invensense_ICM20602:
241  case Invensense_ICM20601:
242  temp_zero = 25.0f;
243  temp_sensitivity = 1.0f/326.8f;
244  break;
245 
246  case Invensense_ICM20789:
247  temp_zero = 25.0f;
248  temp_sensitivity = 0.003f;
249  break;
250  case Invensense_ICM20689:
251  temp_zero = 25.0f;
252  temp_sensitivity = 0.003f;
253  break;
254  }
255 
256  _gyro_instance = _imu.register_gyro(1000, _dev->get_bus_id_devtype(gdev));
257  _accel_instance = _imu.register_accel(1000, _dev->get_bus_id_devtype(adev));
258 
259  // setup ODR and on-sensor filtering
260  _set_filter_register();
261 
262  // update backend sample rate
263  _set_accel_raw_sample_rate(_accel_instance, _backend_rate_hz);
264  _set_gyro_raw_sample_rate(_gyro_instance, _backend_rate_hz);
265 
266  // indicate what multiplier is appropriate for the sensors'
267  // readings to fit them into an int16_t:
268  _set_raw_sample_accel_multiplier(_accel_instance, multiplier_accel);
269 
270  if (_fast_sampling) {
271  hal.console->printf("MPU[%u]: enabled fast sampling rate %uHz/%uHz\n",
272  _accel_instance, _backend_rate_hz*_fifo_downsample_rate, _backend_rate_hz);
273  }
274 
275  // set sample rate to 1000Hz and apply a software filter
276  // In this configuration, the gyro sample rate is 8kHz
277  _register_write(MPUREG_SMPLRT_DIV, 0, true);
278  hal.scheduler->delay(1);
279 
280  // Gyro scale 2000º/s
281  _register_write(MPUREG_GYRO_CONFIG, BITS_GYRO_FS_2000DPS, true);
282  hal.scheduler->delay(1);
283 
284  // read the product ID rev c has 1/2 the sensitivity of rev d
285  uint8_t product_id = _register_read(MPUREG_PRODUCT_ID);
286 
287  if (_mpu_type == Invensense_MPU6000 &&
288  ((product_id == MPU6000ES_REV_C4) ||
289  (product_id == MPU6000ES_REV_C5) ||
290  (product_id == MPU6000_REV_C4) ||
291  (product_id == MPU6000_REV_C5))) {
292  // Accel scale 8g (4096 LSB/g)
293  // Rev C has different scaling than rev D
294  _register_write(MPUREG_ACCEL_CONFIG,1<<3, true);
295  _accel_scale = GRAVITY_MSS / 4096.f;
296  _gyro_scale = (radians(1) / 16.4f);
297  } else if (_mpu_type == Invensense_ICM20601) {
298  // Accel scale 32g (4096 LSB/g)
299  _register_write(MPUREG_ACCEL_CONFIG,1<<3, true);
300  _accel_scale = GRAVITY_MSS / 4096.f;
301  _gyro_scale = (radians(1) / 8.2f);
302  _clip_limit = 29.5f * GRAVITY_MSS;
303  } else {
304  // Accel scale 16g (2048 LSB/g)
305  _register_write(MPUREG_ACCEL_CONFIG,3<<3, true);
306  _accel_scale = GRAVITY_MSS / 2048.f;
307  _gyro_scale = (radians(1) / 16.4f);
308  }
309  hal.scheduler->delay(1);
310 
311  if (_mpu_type == Invensense_ICM20608 ||
312  _mpu_type == Invensense_ICM20602 ||
313  _mpu_type == Invensense_ICM20601) {
314  // this avoids a sensor bug, see description above
315  _register_write(MPUREG_ICM_UNDOC1, MPUREG_ICM_UNDOC1_VALUE, true);
316  }
317 
318  // configure interrupt to fire when new data arrives
319  _register_write(MPUREG_INT_ENABLE, BIT_RAW_RDY_EN);
320  hal.scheduler->delay(1);
321 
322  // clear interrupt on any read, and hold the data ready pin high
323  // until we clear the interrupt. We don't do this for the 20789 as
324  // that sensor has already setup the appropriate config inside the
325  // baro driver.
326  if (_mpu_type != Invensense_ICM20789) {
327  uint8_t v = _register_read(MPUREG_INT_PIN_CFG) | BIT_INT_RD_CLEAR | BIT_LATCH_INT_EN;
328  v &= BIT_BYPASS_EN;
329  _register_write(MPUREG_INT_PIN_CFG, v);
330  }
331 
332  // now that we have initialised, we set the bus speed to high
333  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
334 
335  _dev->get_semaphore()->give();
336 
337  // setup sensor rotations from probe()
338  set_gyro_orientation(_gyro_instance, _rotation);
339  set_accel_orientation(_accel_instance, _rotation);
340 
341  // setup scale factors for fifo data after downsampling
342  _fifo_accel_scale = _accel_scale / (MAX(_fifo_downsample_rate,2)/2);
343  _fifo_gyro_scale = _gyro_scale / _fifo_downsample_rate;
344 
345  // allocate fifo buffer
346  _fifo_buffer = (uint8_t *)hal.util->malloc_type(MPU_FIFO_BUFFER_LEN * MPU_SAMPLE_SIZE, AP_HAL::Util::MEM_DMA_SAFE);
347  if (_fifo_buffer == nullptr) {
348  AP_HAL::panic("Invensense: Unable to allocate FIFO buffer");
349  }
350 
351  // start the timer process to read samples
352  _dev->register_periodic_callback(1000000UL / _backend_rate_hz, FUNCTOR_BIND_MEMBER(&AP_InertialSensor_Invensense::_poll_data, void));
353 }

413 bool AP_InertialSensor_Invensense::_accumulate(uint8_t *samples, uint8_t n_samples)
414 {
415  for (uint8_t i = 0; i < n_samples; i++) {
416  const uint8_t *data = samples + MPU_SAMPLE_SIZE * i;
417  Vector3f accel, gyro;
418  bool fsync_set = false;
419 
420 #if INVENSENSE_EXT_SYNC_ENABLE
421  fsync_set = (int16_val(data, 2) & 1U) != 0;
422 #endif
423 
424  accel = Vector3f(int16_val(data, 1),
425  int16_val(data, 0),
426  -int16_val(data, 2));
427  accel *= _accel_scale;
428 
429  int16_t t2 = int16_val(data, 3);
430  if (!_check_raw_temp(t2)) {
431  debug("temp reset IMU[%u] %d %d", _accel_instance, _raw_temp, t2);
432  _fifo_reset();
433  return false;
434  }
435  float temp = t2 * temp_sensitivity + temp_zero;
436 
437  gyro = Vector3f(int16_val(data, 5),
438  int16_val(data, 4),
439  -int16_val(data, 6));
440  gyro *= _gyro_scale;
441 
442  _rotate_and_correct_accel(_accel_instance, accel);
443  _rotate_and_correct_gyro(_gyro_instance, gyro);
444 
445  _notify_new_accel_raw_sample(_accel_instance, accel, 0, fsync_set);
446  _notify_new_gyro_raw_sample(_gyro_instance, gyro);
447 
448  _temp_filtered = _temp_filter.apply(temp);
449  }
450  return true;
451 }

453 /*
454  when doing fast sampling the sensor gives us 8k samples/second. Every 2nd accel sample is a duplicate.
455 
456  To filter this we first apply a 1p low pass filter at 188Hz, then we
457  average over 8 samples to bring the data rate down to 1kHz. This
458  gives very good aliasing rejection at frequencies well above what
459  can be handled with 1kHz sample rates.
460  */
461 bool AP_InertialSensor_Invensense::_accumulate_sensor_rate_sampling(uint8_t *samples, uint8_t n_samples)
462 {
463  int32_t tsum = 0;
464  const int32_t unscaled_clip_limit = _clip_limit / _accel_scale;
465  bool clipped = false;
466  bool ret = true;
467 
468  for (uint8_t i = 0; i < n_samples; i++) {
469  const uint8_t *data = samples + MPU_SAMPLE_SIZE * i;
470 
471  // use temperatue to detect FIFO corruption
472  int16_t t2 = int16_val(data, 3);
473  if (!_check_raw_temp(t2)) {
474  debug("temp reset IMU[%u] %d %d", _accel_instance, _raw_temp, t2);
475  _fifo_reset();
476  ret = false;
477  break;
478  }
479  tsum += t2;
480 
481  if ((_accum.count & 1) == 0) {
482  // accel data is at 4kHz
483  Vector3f a(int16_val(data, 1),
484  int16_val(data, 0),
485  -int16_val(data, 2));
486  if (fabsf(a.x) > unscaled_clip_limit ||
487  fabsf(a.y) > unscaled_clip_limit ||
488  fabsf(a.z) > unscaled_clip_limit) {
489  clipped = true;
490  }
491  _accum.accel += _accum.accel_filter.apply(a);
492  Vector3f a2 = a * _accel_scale;
493  _notify_new_accel_sensor_rate_sample(_accel_instance, a2);
494  }
495 
496  Vector3f g(int16_val(data, 5),
497  int16_val(data, 4),
498  -int16_val(data, 6));
499 
500  Vector3f g2 = g * _gyro_scale;
501  _notify_new_gyro_sensor_rate_sample(_gyro_instance, g2);
502 
503  _accum.gyro += _accum.gyro_filter.apply(g);
504  _accum.count++;
505 
506  if (_accum.count == _fifo_downsample_rate) {
507 
508  _accum.accel *= _fifo_accel_scale;
509  _accum.gyro *= _fifo_gyro_scale;
510 
511  _rotate_and_correct_accel(_accel_instance, _accum.accel);
512  _rotate_and_correct_gyro(_gyro_instance, _accum.gyro);
513 
514  _notify_new_accel_raw_sample(_accel_instance, _accum.accel, 0, false);
515  _notify_new_gyro_raw_sample(_gyro_instance, _accum.gyro);
516 
517  _accum.accel.zero();
518  _accum.gyro.zero();
519  _accum.count = 0;
520  }
521  }
522 
523  if (clipped) {
524  increment_clip_count(_accel_instance);
525  }
526 
527  if (ret) {
528  float temp = (static_cast<float>(tsum)/n_samples)*temp_sensitivity + temp_zero;
529  _temp_filtered = _temp_filter.apply(temp);
530  }
531 
532  return ret;
533 }

535 void AP_InertialSensor_Invensense::_read_fifo()
536 {
537  uint8_t n_samples;
538  uint16_t bytes_read;
539  uint8_t *rx = _fifo_buffer;
540  bool need_reset = false;
541 
542  if (!_block_read(MPUREG_FIFO_COUNTH, rx, 2)) {
543  goto check_registers;
544  }
545 
546  bytes_read = uint16_val(rx, 0);
547  n_samples = bytes_read / MPU_SAMPLE_SIZE;
548 
549  if (n_samples == 0) {
550  /* Not enough data in FIFO */
551  goto check_registers;
552  }
553 
554  /*
555  testing has shown that if we have more than 32 samples in the
556  FIFO then some of those samples will be corrupt. It always is
557  the ones at the end of the FIFO, so clear those with a reset
558  once we've read the first 24. Reading 24 gives us the normal
559  number of samples for fast sampling at 400Hz
560 
561  On I2C with the much lower clock rates we need a lower threshold
562  or we may never catch up
563  */
564  if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C) {
565  if (n_samples > 4) {
566  need_reset = true;
567  n_samples = 4;
568  }
569  } else {
570  if (n_samples > 32) {
571  need_reset = true;
572  n_samples = 24;
573  }
574  }
575 
576  while (n_samples > 0) {
577  uint8_t n = MIN(n_samples, MPU_FIFO_BUFFER_LEN);
578  if (!_dev->set_chip_select(true)) {
579  if (!_block_read(MPUREG_FIFO_R_W, rx, n * MPU_SAMPLE_SIZE)) {
580  goto check_registers;
581  }
582  } else {
583  // this ensures we keep things nicely setup for DMA
584  uint8_t reg = MPUREG_FIFO_R_W | 0x80;
585  if (!_dev->transfer(&reg, 1, nullptr, 0)) {
586  _dev->set_chip_select(false);
587  goto check_registers;
588  }
589  memset(rx, 0, n * MPU_SAMPLE_SIZE);
590  if (!_dev->transfer(rx, n * MPU_SAMPLE_SIZE, rx, n * MPU_SAMPLE_SIZE)) {
591  hal.console->printf("MPU60x0: error in fifo read %u bytes\n", n * MPU_SAMPLE_SIZE);
592  _dev->set_chip_select(false);
593  goto check_registers;
594  }
595  _dev->set_chip_select(false);
596  }
597 
598  if (_fast_sampling) {
599  if (!_accumulate_sensor_rate_sampling(rx, n)) {
600  debug("IMU[%u] stop at %u of %u", _accel_instance, n_samples, bytes_read/MPU_SAMPLE_SIZE);
601  break;

770 bool AP_InertialSensor_Invensense::_hardware_init(void)
771 {
772  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
773  return false;
774  }
775 
776  // setup for register checking. We check much less often on I2C
777  // where the cost of the checks is higher
778  _dev->setup_checked_registers(7, _dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C?200:20);
779 
780  // initially run the bus at low speed
781  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
782 
783  if (!_check_whoami()) {
784  _dev->get_semaphore()->give();
785  return false;
786  }
787 
788  // Chip reset
789  uint8_t tries;
790  for (tries = 0; tries < 5; tries++) {
791  _last_stat_user_ctrl = _register_read(MPUREG_USER_CTRL);
792 
793  /* First disable the master I2C to avoid hanging the slaves on the
794  * aulixiliar I2C bus - it will be enabled again if the AuxiliaryBus
795  * is used */
796  if (_last_stat_user_ctrl & BIT_USER_CTRL_I2C_MST_EN) {
797  _last_stat_user_ctrl &= ~BIT_USER_CTRL_I2C_MST_EN;
798  _register_write(MPUREG_USER_CTRL, _last_stat_user_ctrl);
799  hal.scheduler->delay(10);
800  }
801 
802  /* reset device */
803  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_DEVICE_RESET);
804  hal.scheduler->delay(100);
805 
806  /* bus-dependent initialization */
807  if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI) {
808  /* Disable I2C bus if SPI selected (Recommended in Datasheet to be
809  * done just after the device is reset) */
810  _last_stat_user_ctrl |= BIT_USER_CTRL_I2C_IF_DIS;
811  _register_write(MPUREG_USER_CTRL, _last_stat_user_ctrl);
812  }
813 
814  /* bus-dependent initialization */
815  if ((_dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C) && (_mpu_type == Invensense_MPU9250 || _mpu_type == Invensense_ICM20789)) {
816  /* Enable I2C bypass to access internal device */
817  _register_write(MPUREG_INT_PIN_CFG, BIT_BYPASS_EN);
818  }
819 
820 
821  // Wake up device and select GyroZ clock. Note that the
822  // Invensense starts up in sleep mode, and it can take some time
823  // for it to come out of sleep
824  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_ZGYRO);
825  hal.scheduler->delay(5);
826 
827  // check it has woken up
828  if (_register_read(MPUREG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_ZGYRO) {
829  break;
830  }
831 
832  hal.scheduler->delay(10);
833  if (_data_ready()) {
834  break;
835  }
836  }
837 
838  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
839 
840  if (tries == 5) {
841  hal.console->printf("Failed to boot Invensense 5 times\n");
842  _dev->get_semaphore()->give();
843  return false;
844  }
845 
846  if (_mpu_type == Invensense_ICM20608 ||
847  _mpu_type == Invensense_ICM20602 ||
848  _mpu_type == Invensense_ICM20601) {
849  // this avoids a sensor bug, see description above
850  _register_write(MPUREG_ICM_UNDOC1, MPUREG_ICM_UNDOC1_VALUE, true);
851  }
