41 /*
42  load *.bin font file, in MAX7456 format
43  */
44 void AP_OSD_SITL::load_font(void)
45 {
46  uint32_t font_size;
47  char fontname[] = "font0.bin";
48  last_font = get_font_num();
49  fontname[4] = last_font + '0';
50  uint8_t *font_data = AP_ROMFS::find_decompress(fontname, font_size);
51  if (font_data == nullptr && last_font != 0) {
52  last_font = 0;
53  fontname[4] = last_font + '0';
54  font_data = AP_ROMFS::find_decompress(fontname, font_size);
55  }
56  if (font_data == nullptr || font_size != 54 * 256) {
57  AP_HAL::panic("Bad font file");
58  }
59  for (uint16_t i=0; i<256; i++) {
60  const uint8_t *c = &font_data[i*54];
61  // each pixel is 4 bytes, RGBA
62  sf::Uint8 *pixels = new sf::Uint8[char_width * char_height * 4];
63  if (!font[i].create(char_width, char_height)) {
64  AP_HAL::panic("Failed to create texture");
65  }
66  for (uint16_t y=0; y<char_height; y++) {
67  for (uint16_t x=0; x<char_width; x++) {
68  // 2 bits per pixel
69  uint16_t bitoffset = (y*char_width+x)*2;
70  uint8_t byteoffset = bitoffset / 8;
71  uint8_t bitshift = 6-(bitoffset % 8);
72  uint8_t v = (c[byteoffset] >> bitshift) & 3;
73  sf::Uint8 *p = &pixels[(y*char_width+x)*4];
74  switch (v) {
75  case 0:
76  p[0] = 0;
77  p[1] = 0;
78  p[2] = 0;
79  p[3] = 255;
80  break;
81  case 1:
82  case 3:
83  p[0] = 0;
84  p[1] = 0;
85  p[2] = 0;
86  p[3] = 0;
87  break;
88  case 2:
89  p[0] = 255;
90  p[1] = 255;
91  p[2] = 255;
92  p[3] = 255;
93  break;
94  }
95 
96  }
97  }
98  font[i].update(pixels);
99  }
100  free(font_data);
101 }

129 // main loop of graphics thread
130 void AP_OSD_SITL::update_thread(void)
131 {
132  load_font();
133  {
134  WITH_SEMAPHORE(AP::notify().sf_window_mutex);
135  w = new sf::RenderWindow(sf::VideoMode(video_cols*(char_width+char_spacing)*char_scale,
136  video_lines*(char_height+char_spacing)*char_scale),
137  "OSD");
138  }
139  if (!w) {
140  AP_HAL::panic("Unable to create OSD window");
141  }
142 
143  while (true) {
144  {
145  WITH_SEMAPHORE(AP::notify().sf_window_mutex);
146  sf::Event event;
147  while (w->pollEvent(event)) {
148  if (event.type == sf::Event::Closed) {
149  w->close();
150  }
151  }
152  if (!w->isOpen()) {
153  break;
154  }
155  if (counter != last_counter) {
156  last_counter = counter;
157 
158  uint8_t buffer2[video_lines][video_cols];
159  {
160  WITH_SEMAPHORE(mutex);
161  memcpy(buffer2, buffer, sizeof(buffer2));
162  }
163  w->clear();
164 
165  for (uint8_t y=0; y<video_lines; y++) {
166  for (uint8_t x=0; x<video_cols; x++) {
167  uint16_t px = x * (char_width+char_spacing) * char_scale;
168  uint16_t py = y * (char_height+char_spacing) * char_scale;
169  sf::Sprite s;
170  uint8_t c = buffer2[y][x];
171  s.setTexture(font[c]);
172  s.setPosition(sf::Vector2f(px, py));
173  s.scale(sf::Vector2f(char_scale,char_scale));
174  w->draw(s);
175  }
176  }
177 
178  w->display();
179  if (last_font != get_font_num()) {
180  load_font();
181  }
182  }
183  }
184  usleep(10000);
185  }
186 }
