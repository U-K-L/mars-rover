35 /*
36  open a SBUS UART
37  */
38 int RCInput_RCProtocol::open_sbus(const char *path)
39 {
40  int fd = open(path, O_RDWR | O_NONBLOCK | O_CLOEXEC);
41  if (fd == -1) {
42  return -1;
43  }
44  struct termios2 tio {};
45 
46  if (ioctl(fd, TCGETS2, &tio) != 0) {
47  close(fd);
48  fd = -1;
49  return -1;
50  }
51  tio.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR
52  | IGNCR | ICRNL | IXON);
53  tio.c_iflag |= (INPCK | IGNPAR);
54  tio.c_oflag &= ~OPOST;
55  tio.c_lflag &= ~(ECHO | ECHONL | ICANON | ISIG | IEXTEN);
56  tio.c_cflag &= ~(CSIZE | CRTSCTS | PARODD | CBAUD);
57  // use BOTHER to specify speed directly in c_[io]speed member
58  tio.c_cflag |= (CS8 | CSTOPB | CLOCAL | PARENB | BOTHER | CREAD);
59  tio.c_ispeed = 100000;
60  tio.c_ospeed = 100000;
61  if (ioctl(fd, TCSETS2, &tio) != 0) {
62  close(fd);
63  fd = -1;
64  return -1;
65  }
66 
67  return fd;
68 }

126 void RCInput_RCProtocol::_timer_tick(void)
127 {
128  uint8_t b[40];
129 
130  if (fd_sbus != -1) {
131  ssize_t n = ::read(fd_sbus, &b[0], sizeof(b));
132  if (n > 0) {
133  for (uint8_t i=0; i<n; i++) {
134  rcp.process_byte(b[i], 100000);
135  }
136  }
137  }
138  if (fd_115200 != -1) {
139  ssize_t n = ::read(fd_115200, &b[0], sizeof(b));
140  if (n > 0) {
141  for (uint8_t i=0; i<n; i++) {
142  rcp.process_byte(b[i], 115200);
143  }
144  }
145  }
146 
147  if (rcp.new_input()) {
148  uint8_t n = rcp.num_channels();
149  for (uint8_t i=0; i<n; i++) {
150  _pwm_values[i] = rcp.read(i);
151  }
152  _num_channels = n;
153  rc_input_count++;
154  }
