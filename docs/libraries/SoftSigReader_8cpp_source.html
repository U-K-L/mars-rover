28 bool SoftSigReader::attach_capture_timer(ICUDriver* icu_drv, icuchannel_t chan, uint8_t dma_stream, uint32_t dma_channel)
29 {
30  if (chan > ICU_CHANNEL_2) {
31  return false;
32  }
33  signal = (uint32_t*)hal.util->malloc_type(sizeof(uint32_t)*SOFTSIG_BOUNCE_BUF_SIZE, AP_HAL::Util::MEM_DMA_SAFE);
34  if (signal == nullptr) {
35  return false;
36  }
37  _icu_drv = icu_drv;
38  //Setup Burst transfer of period and width measurement
39  osalDbgAssert(dma == nullptr, "double DMA allocation");
40  chSysLock();
41  dma = dmaStreamAllocI(dma_stream,
42  12, //IRQ Priority
43  (stm32_dmaisr_t)_irq_handler,
44  (void *)this);
45  osalDbgAssert(dma, "stream allocation failed");
46  chSysUnlock();
47 #if STM32_DMA_SUPPORTS_DMAMUX
48  dmaSetRequestSource(dma, dma_channel);
49 #endif
50  //setup address for full word transfer from Timer
51  dmaStreamSetPeripheral(dma, &icu_drv->tim->DMAR);
52 
53  dmamode = STM32_DMA_CR_DMEIE | STM32_DMA_CR_TEIE;
54  dmamode |= STM32_DMA_CR_CHSEL(dma_channel);
55  dmamode |= STM32_DMA_CR_PL(0);
56  dmamode |= STM32_DMA_CR_DIR_P2M | STM32_DMA_CR_PSIZE_WORD |
57  STM32_DMA_CR_MSIZE_WORD | STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE;
58  dmaStreamSetMemory0(dma, signal);
59  dmaStreamSetTransactionSize(dma, SOFTSIG_BOUNCE_BUF_SIZE);
60  dmaStreamSetMode(dma, dmamode);
61 
62  icucfg.frequency = INPUT_CAPTURE_FREQUENCY;
63  icucfg.channel = chan;
64  icucfg.width_cb = NULL;
65  icucfg.period_cb = NULL;
66  icucfg.overflow_cb = NULL;
67 
68  if (chan == ICU_CHANNEL_1) {
69  icucfg.dier = STM32_TIM_DIER_CC1DE;
70  icucfg.mode = ICU_INPUT_ACTIVE_HIGH;
71  need_swap = true;
72  } else {
73  icucfg.mode = ICU_INPUT_ACTIVE_LOW;
74  icucfg.dier = STM32_TIM_DIER_CC2DE;
75  }
76 #ifdef HAL_RCIN_IS_INVERTED
77  icucfg.mode = (icucfg.mode==ICU_INPUT_ACTIVE_LOW)?ICU_INPUT_ACTIVE_HIGH:ICU_INPUT_ACTIVE_LOW;
78 #endif
79  icuStart(_icu_drv, &icucfg);
80  //Extended Timer Setup to enable DMA transfer
81  //selected offset for TIM_CCR1 and for two words
82  _icu_drv->tim->DCR = STM32_TIM_DCR_DBA(0x0D) | STM32_TIM_DCR_DBL(1);
83  //Enable DMA
84  dmaStreamEnable(dma);
85 
86  //sets input filtering to 4 timer clock
87  stm32_timer_set_input_filter(_icu_drv->tim, chan, 2);
88 
89  //Start Timer
90  icuStartCapture(_icu_drv);
91  return true;
92 }

94 void SoftSigReader::_irq_handler(void* self, uint32_t flags)
95 {
96  SoftSigReader* sig_reader = (SoftSigReader*)self;
97  // we need to restart the DMA as quickly as possible to prevent losing pulses, so we
98  // make a fixed length copy to a 2nd buffer. On the F100 this reduces the time with DMA
99  // disabled from 20us to under 1us
100  cacheBufferInvalidate(sig_reader->signal, SOFTSIG_BOUNCE_BUF_SIZE*4);
101  memcpy(sig_reader->signal2, sig_reader->signal, SOFTSIG_BOUNCE_BUF_SIZE*4);
102  //restart the DMA transfers
103  dmaStreamDisable(sig_reader->dma);
104  dmaStreamSetPeripheral(sig_reader->dma, &sig_reader->_icu_drv->tim->DMAR);
105  dmaStreamSetMemory0(sig_reader->dma, sig_reader->signal);
106  dmaStreamSetTransactionSize(sig_reader->dma, SOFTSIG_BOUNCE_BUF_SIZE);
107  dmaStreamSetMode(sig_reader->dma, sig_reader->dmamode);
108  dmaStreamEnable(sig_reader->dma);
109  sig_reader->sigbuf.push((const pulse_t *)sig_reader->signal2, SOFTSIG_BOUNCE_BUF_SIZE/2);
110 }
