266 void AC_PrecLand::run_estimator(float rangefinder_alt_m, bool rangefinder_alt_valid)
267 {
268  const struct inertial_data_frame_s *inertial_data_delayed = (*_inertial_history)[0];
269 
270  switch (_estimator_type) {
271  case ESTIMATOR_TYPE_RAW_SENSOR: {
272  // Return if there's any invalid velocity data
273  for (uint8_t i=0; i<_inertial_history->available(); i++) {
274  const struct inertial_data_frame_s *inertial_data = (*_inertial_history)[i];
275  if (!inertial_data->inertialNavVelocityValid) {
276  _target_acquired = false;
277  return;
278  }
279  }
280 
281  // Predict
282  if (target_acquired()) {
283  _target_pos_rel_est_NE.x -= inertial_data_delayed->inertialNavVelocity.x * inertial_data_delayed->dt;
284  _target_pos_rel_est_NE.y -= inertial_data_delayed->inertialNavVelocity.y * inertial_data_delayed->dt;
285  _target_vel_rel_est_NE.x = -inertial_data_delayed->inertialNavVelocity.x;
286  _target_vel_rel_est_NE.y = -inertial_data_delayed->inertialNavVelocity.y;
287  }
288 
289  // Update if a new Line-Of-Sight measurement is available
290  if (construct_pos_meas_using_rangefinder(rangefinder_alt_m, rangefinder_alt_valid)) {
291  _target_pos_rel_est_NE.x = _target_pos_rel_meas_NED.x;
292  _target_pos_rel_est_NE.y = _target_pos_rel_meas_NED.y;
293  _target_vel_rel_est_NE.x = -inertial_data_delayed->inertialNavVelocity.x;
294  _target_vel_rel_est_NE.y = -inertial_data_delayed->inertialNavVelocity.y;
295 
296  _last_update_ms = AP_HAL::millis();
297  _target_acquired = true;
298  }
299 
300  // Output prediction
301  if (target_acquired()) {
302  run_output_prediction();
303  }
304  break;
305  }
306  case ESTIMATOR_TYPE_KALMAN_FILTER: {
307  // Predict
308  if (target_acquired()) {
309  const float& dt = inertial_data_delayed->dt;
310  const Vector3f& vehicleDelVel = inertial_data_delayed->correctedVehicleDeltaVelocityNED;
311 
312  _ekf_x.predict(dt, -vehicleDelVel.x, _accel_noise*dt);
313  _ekf_y.predict(dt, -vehicleDelVel.y, _accel_noise*dt);
314  }
315 
316  // Update if a new Line-Of-Sight measurement is available
317  if (construct_pos_meas_using_rangefinder(rangefinder_alt_m, rangefinder_alt_valid)) {
318  float xy_pos_var = sq(_target_pos_rel_meas_NED.z*(0.01f + 0.01f*AP::ahrs().get_gyro().length()) + 0.02f);
319  if (!target_acquired()) {
320  // reset filter state
321  if (inertial_data_delayed->inertialNavVelocityValid) {
322  _ekf_x.init(_target_pos_rel_meas_NED.x, xy_pos_var, -inertial_data_delayed->inertialNavVelocity.x, sq(2.0f));
323  _ekf_y.init(_target_pos_rel_meas_NED.y, xy_pos_var, -inertial_data_delayed->inertialNavVelocity.y, sq(2.0f));
324  } else {
325  _ekf_x.init(_target_pos_rel_meas_NED.x, xy_pos_var, 0.0f, sq(10.0f));
326  _ekf_y.init(_target_pos_rel_meas_NED.y, xy_pos_var, 0.0f, sq(10.0f));
327  }
328  _last_update_ms = AP_HAL::millis();
329  _target_acquired = true;
330  } else {
331  float NIS_x = _ekf_x.getPosNIS(_target_pos_rel_meas_NED.x, xy_pos_var);
332  float NIS_y = _ekf_y.getPosNIS(_target_pos_rel_meas_NED.y, xy_pos_var);
333  if (MAX(NIS_x, NIS_y) < 3.0f || _outlier_reject_count >= 3) {
334  _outlier_reject_count = 0;
335  _ekf_x.fusePos(_target_pos_rel_meas_NED.x, xy_pos_var);
336  _ekf_y.fusePos(_target_pos_rel_meas_NED.y, xy_pos_var);
337  _last_update_ms = AP_HAL::millis();
338  _target_acquired = true;
339  } else {
340  _outlier_reject_count++;
341  }
342  }
343  }
344 
345  // Output prediction
346  if (target_acquired()) {
347  _target_pos_rel_est_NE.x = _ekf_x.getPos();
348  _target_pos_rel_est_NE.y = _ekf_y.getPos();
349  _target_vel_rel_est_NE.x = _ekf_x.getVel();
350  _target_vel_rel_est_NE.y = _ekf_y.getVel();
351 
352  run_output_prediction();
353  }
354  break;
355  }
356  }

381 bool AC_PrecLand::construct_pos_meas_using_rangefinder(float rangefinder_alt_m, bool rangefinder_alt_valid)
382 {
383  Vector3f target_vec_unit_body;
384  if (retrieve_los_meas(target_vec_unit_body)) {
385  const struct inertial_data_frame_s *inertial_data_delayed = (*_inertial_history)[0];
386 
387  Vector3f target_vec_unit_ned = inertial_data_delayed->Tbn * target_vec_unit_body;
388  bool target_vec_valid = target_vec_unit_ned.z > 0.0f;
389  bool alt_valid = (rangefinder_alt_valid && rangefinder_alt_m > 0.0f) || (_backend->distance_to_target() > 0.0f);
390  if (target_vec_valid && alt_valid) {
391  float dist, alt;
392  if (_backend->distance_to_target() > 0.0f) {
393  dist = _backend->distance_to_target();
394  alt = dist * target_vec_unit_ned.z;
395  } else {
396  alt = MAX(rangefinder_alt_m, 0.0f);
397  dist = alt / target_vec_unit_ned.z;
398  }
399 
400  // Compute camera position relative to IMU
401  Vector3f accel_body_offset = AP::ins().get_imu_pos_offset(AP::ahrs().get_primary_accel_index());
402  Vector3f cam_pos_ned = inertial_data_delayed->Tbn * (_cam_offset.get() - accel_body_offset);
403 
404  // Compute target position relative to IMU
405  _target_pos_rel_meas_NED = Vector3f(target_vec_unit_ned.x*dist, target_vec_unit_ned.y*dist, alt) + cam_pos_ned;

408  }

412 void AC_PrecLand::run_output_prediction()
413 {
414  _target_pos_rel_out_NE = _target_pos_rel_est_NE;
415  _target_vel_rel_out_NE = _target_vel_rel_est_NE;
416 
417  // Predict forward from delayed time horizon
418  for (uint8_t i=1; i<_inertial_history->available(); i++) {
419  const struct inertial_data_frame_s *inertial_data = (*_inertial_history)[i];
420  _target_vel_rel_out_NE.x -= inertial_data->correctedVehicleDeltaVelocityNED.x;
421  _target_vel_rel_out_NE.y -= inertial_data->correctedVehicleDeltaVelocityNED.y;
422  _target_pos_rel_out_NE.x += _target_vel_rel_out_NE.x * inertial_data->dt;
423  _target_pos_rel_out_NE.y += _target_vel_rel_out_NE.y * inertial_data->dt;
424  }
425 
426  const AP_AHRS &_ahrs = AP::ahrs();
427 
428  const Matrix3f& Tbn = (*_inertial_history)[_inertial_history->available()-1]->Tbn;
429  Vector3f accel_body_offset = AP::ins().get_imu_pos_offset(_ahrs.get_primary_accel_index());
430 
431  // Apply position correction for CG offset from IMU
432  Vector3f imu_pos_ned = Tbn * accel_body_offset;
433  _target_pos_rel_out_NE.x += imu_pos_ned.x;
434  _target_pos_rel_out_NE.y += imu_pos_ned.y;
435 
436  // Apply position correction for body-frame horizontal camera offset from CG, so that vehicle lands lens-to-target
437  Vector3f cam_pos_horizontal_ned = Tbn * Vector3f(_cam_offset.get().x, _cam_offset.get().y, 0);
438  _target_pos_rel_out_NE.x -= cam_pos_horizontal_ned.x;
439  _target_pos_rel_out_NE.y -= cam_pos_horizontal_ned.y;
440 
441  // Apply velocity correction for IMU offset from CG
442  Vector3f vel_ned_rel_imu = Tbn * (_ahrs.get_gyro() % (-accel_body_offset));
443  _target_vel_rel_out_NE.x -= vel_ned_rel_imu.x;
444  _target_vel_rel_out_NE.y -= vel_ned_rel_imu.y;
445 
446  // Apply land offset
447  Vector3f land_ofs_ned_m = _ahrs.get_rotation_body_to_ned() * Vector3f(_land_ofs_cm_x,_land_ofs_cm_y,0) * 0.01f;
448  _target_pos_rel_out_NE.x += land_ofs_ned_m.x;
449  _target_pos_rel_out_NE.y += land_ofs_ned_m.y;
