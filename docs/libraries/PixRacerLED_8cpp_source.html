37 bool PixRacerLED::hw_init(void)
38 {
39  // when HAL_GPIO_LED_ON is 0 then we must not use pinMode()
40  // as it could remove the OPENDRAIN attribute on the pin
41 #if HAL_GPIO_LED_ON != 0
42  hal.gpio->pinMode(HAL_GPIO_A_LED_PIN, HAL_GPIO_OUTPUT);
43  hal.gpio->pinMode(HAL_GPIO_B_LED_PIN, HAL_GPIO_OUTPUT);
44  hal.gpio->pinMode(HAL_GPIO_C_LED_PIN, HAL_GPIO_OUTPUT);
45 #endif
46  hal.gpio->write(HAL_GPIO_A_LED_PIN, HAL_GPIO_LED_OFF);
47  hal.gpio->write(HAL_GPIO_B_LED_PIN, HAL_GPIO_LED_OFF);
48  hal.gpio->write(HAL_GPIO_C_LED_PIN, HAL_GPIO_LED_OFF);
49  return true;
