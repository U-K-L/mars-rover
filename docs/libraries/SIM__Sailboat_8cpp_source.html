41 // calculate the lift and drag as values from 0 to 1
42 // given an apparent wind speed in m/s and angle-of-attack in degrees
43 void Sailboat::calc_lift_and_drag(float wind_speed, float angle_of_attack_deg, float& lift, float& drag) const
44 {
45  const uint16_t index_width_deg = 10;
46  const uint8_t index_max = ARRAY_SIZE(lift_curve) - 1;
47 
48  // check extremes
49  if (angle_of_attack_deg <= 0.0f) {
50  lift = lift_curve[0];
51  drag = drag_curve[0];
52  } else if (angle_of_attack_deg >= index_max * index_width_deg) {
53  lift = lift_curve[index_max];
54  drag = drag_curve[index_max];
55  } else {
56  uint8_t index = constrain_int16(angle_of_attack_deg / index_width_deg, 0, index_max);
57  float remainder = angle_of_attack_deg - (index * index_width_deg);
58  lift = linear_interpolate(lift_curve[index], lift_curve[index+1], remainder, 0.0f, index_width_deg);
59  drag = linear_interpolate(drag_curve[index], drag_curve[index+1], remainder, 0.0f, index_width_deg);
60  }
61 
62  // apply scaling by wind speed
63  lift *= wind_speed;
64  drag *= wind_speed;
65 }

67 // return turning circle (diameter) in meters for steering angle proportion in the range -1 to +1
68 float Sailboat::get_turn_circle(float steering) const
69 {
70  if (is_zero(steering)) {
71  return 0;
72  }
73  return turning_circle * sinf(radians(steering_angle_max)) / sinf(radians(steering * steering_angle_max));
74 }

76 // return yaw rate in deg/sec given a steering input (in the range -1 to +1) and speed in m/s
77 float Sailboat::get_yaw_rate(float steering, float speed) const
78 {
79  if (is_zero(steering) || is_zero(speed)) {
80  return 0;
81  }
82  float d = get_turn_circle(steering);
83  float c = M_PI * d;
84  float t = c / speed;
85  float rate = 360.0f / t;
86  return rate;
87 }

89 // return lateral acceleration in m/s/s given a steering input (in the range -1 to +1) and speed in m/s
90 float Sailboat::get_lat_accel(float steering, float speed) const
91 {
92  float yaw_rate = get_yaw_rate(steering, speed);
93  float accel = radians(yaw_rate) * speed;
94  return accel;
95 }

97 /*
98  update the sailboat simulation by one time step
99  */
100 void Sailboat::update(const struct sitl_input &input)
101 {
102  // update wind
103  update_wind(input);
104 
105  // in sailboats the steering controls the rudder, the throttle controls the main sail position
106  float steering = 2*((input.servos[STEERING_SERVO_CH]-1000)/1000.0f - 0.5f);
107 
108  // calculate mainsail angle from servo output 4, 0 to 90 degrees
109  float mainsail_angle_bf = constrain_float((input.servos[MAINSAIL_SERVO_CH]-1000)/1000.0f * 90.0f, 0.0f, 90.0f);
110 
111  // calculate apparent wind in earth-frame (this is the direction the wind is coming from)
112  // Note than the SITL wind direction is defined as the direction the wind is travelling to
113  // This is accounted for in these calculations
114  Vector3f wind_apparent_ef = wind_ef + velocity_ef;
115  const float wind_apparent_dir_ef = degrees(atan2f(wind_apparent_ef.y, wind_apparent_ef.x));
116  const float wind_apparent_speed = safe_sqrt(sq(wind_apparent_ef.x)+sq(wind_apparent_ef.y));
117 
118  const float wind_apparent_dir_bf = wrap_180(wind_apparent_dir_ef - degrees(AP::ahrs().yaw));
119 
120  // set RPM and airspeed from wind speed, allows to test RPM and Airspeed wind vane back end in SITL
121  rpm1 = wind_apparent_speed;
122  airspeed_pitot = wind_apparent_speed;
123 
124  // calculate angle-of-attack from wind to mainsail
125  float aoa_deg = MAX(fabsf(wind_apparent_dir_bf) - mainsail_angle_bf, 0);
126 
127  // calculate Lift force (perpendicular to wind direction) and Drag force (parallel to wind direction)
128  float lift_wf, drag_wf;
129  calc_lift_and_drag(wind_apparent_speed, aoa_deg, lift_wf, drag_wf);
130 
131  // rotate lift and drag from wind frame into body frame
132  const float sin_rot_rad = sinf(radians(wind_apparent_dir_bf));
133  const float cos_rot_rad = cosf(radians(wind_apparent_dir_bf));
134  const float force_fwd = fabsf(lift_wf * sin_rot_rad) - (drag_wf * cos_rot_rad);
135 
136  // how much time has passed?
137  float delta_time = frame_time_us * 1.0e-6f;
138 
139  // speed in m/s in body frame
140  Vector3f velocity_body = dcm.transposed() * velocity_ef;
141 
142  // speed along x axis, +ve is forward
143  float speed = velocity_body.x;
144 
145  // yaw rate in degrees/s
146  float yaw_rate = get_yaw_rate(steering, speed);
147 
148  gyro = Vector3f(0,0,radians(yaw_rate));
149 
150  // update attitude
151  dcm.rotate(gyro * delta_time);
152  dcm.normalize();
153 
154  // hull drag
155  float hull_drag = sq(speed) * 0.5f;
156  if (!is_positive(speed)) {
157  hull_drag *= -1.0f;
158  }
159 
160  // accel in body frame due acceleration from sail and deceleration from hull friction
161  accel_body = Vector3f(force_fwd - hull_drag, 0, 0);
162  accel_body /= mass;
163 
164  // add in accel due to direction change
165  accel_body.y += radians(yaw_rate) * speed;
166 
167  // now in earth frame
168  Vector3f accel_earth = dcm * accel_body;
169 
170  // we are on the ground, so our vertical accel is zero
171  accel_earth.z = 0;
172 
173  // work out acceleration as seen by the accelerometers. It sees the kinematic
174  // acceleration (ie. real movement), plus gravity
175  accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
176 
177  // new velocity vector
178  velocity_ef += accel_earth * delta_time;
179 
180  // new position vector
181  position += velocity_ef * delta_time;
182 
183  // update lat/lon/altitude
184  update_position();
185  time_advance();
186 
187  // update magnetic field
188  update_mag_field_bf();
189 }
