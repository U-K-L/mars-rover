16 /*
17  * The intention of a magnetometer in a compass application is to measure
18  * Earth's magnetic field. Measurements other than those of Earth's magnetic
19  * field are considered errors. This algorithm computes a set of correction
20  * parameters that null out errors from various sources:
21  *
22  * - Sensor bias error
23  * - "Hard iron" error caused by materials fixed to the vehicle body that
24  * produce static magnetic fields.
25  * - Sensor scale-factor error
26  * - Sensor cross-axis sensitivity
27  * - "Soft iron" error caused by materials fixed to the vehicle body that
28  * distort magnetic fields.
29  *
30  * This is done by taking a set of samples that are assumed to be the product
31  * of rotation in earth's magnetic field and fitting an offset ellipsoid to
32  * them, determining the correction to be applied to adjust the samples into an
33  * origin-centered sphere.
34  *
35  * The state machine of this library is described entirely by the
36  * compass_cal_status_t enum, and all state transitions are managed by the
37  * set_status function. Normally, the library is in the NOT_STARTED state. When
38  * the start function is called, the state transitions to WAITING_TO_START,
39  * until two conditions are met: the delay as elapsed, and the memory for the
40  * sample buffer has been successfully allocated.
41  * Once these conditions are met, the state transitions to RUNNING_STEP_ONE, and
42  * samples are collected via calls to the new_sample function. These samples are
43  * accepted or rejected based on distance to the nearest sample. The samples are
44  * assumed to cover the surface of a sphere, and the radius of that sphere is
45  * initialized to a conservative value. Based on a circle-packing pattern, the
46  * minimum distance is set such that some percentage of the surface of that
47  * sphere must be covered by samples.
48  *
49  * Once the sample buffer is full, a sphere fitting algorithm is run, which
50  * computes a new sphere radius. The sample buffer is thinned of samples which
51  * no longer meet the acceptance criteria, and the state transitions to
52  * RUNNING_STEP_TWO. Samples continue to be collected until the buffer is full
53  * again, the full ellipsoid fit is run, and the state transitions to either
54  * SUCCESS or FAILED.
55  *
56  * The fitting algorithm used is Levenberg-Marquardt. See also:
57  * http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm
58  */

107 float CompassCalibrator::get_completion_percent() const {
108  // first sampling step is 1/3rd of the progress bar
109  // never return more than 99% unless _status is COMPASS_CAL_SUCCESS
110  switch(_status) {
111  case COMPASS_CAL_NOT_STARTED:
112  case COMPASS_CAL_WAITING_TO_START:
113  return 0.0f;
114  case COMPASS_CAL_RUNNING_STEP_ONE:
115  return 33.3f * _samples_collected/COMPASS_CAL_NUM_SAMPLES;
116  case COMPASS_CAL_RUNNING_STEP_TWO:
117  return 33.3f + 65.7f*((float)(_samples_collected-_samples_thinned)/(COMPASS_CAL_NUM_SAMPLES-_samples_thinned));
118  case COMPASS_CAL_SUCCESS:
119  return 100.0f;
120  case COMPASS_CAL_FAILED:
121  case COMPASS_CAL_BAD_ORIENTATION:
122  default:
123  return 0.0f;
124  };
125 }

127 void CompassCalibrator::update_completion_mask(const Vector3f& v)
128 {
129  Matrix3f softiron{
130  _params.diag.x, _params.offdiag.x, _params.offdiag.y,
131  _params.offdiag.x, _params.diag.y, _params.offdiag.z,
132  _params.offdiag.y, _params.offdiag.z, _params.diag.z
133  };
134  Vector3f corrected = softiron * (v + _params.offset);
135  int section = AP_GeodesicGrid::section(corrected, true);
136  if (section < 0) {
137  return;
138  }
139  _completion_mask[section / 8] |= 1 << (section % 8);
140 }

180 void CompassCalibrator::update(bool &failure) {
181  failure = false;
182 
183  if(!fitting()) {
184  return;
185  }
186 
187  if(_status == COMPASS_CAL_RUNNING_STEP_ONE) {
188  if (_fit_step >= 10) {
189  if(is_equal(_fitness,_initial_fitness) || isnan(_fitness)) { //if true, means that fitness is diverging instead of converging
190  set_status(COMPASS_CAL_FAILED);
191  failure = true;
192  }
193  set_status(COMPASS_CAL_RUNNING_STEP_TWO);
194  } else {
195  if (_fit_step == 0) {
196  calc_initial_offset();
197  }
198  run_sphere_fit();
199  _fit_step++;
200  }
201  } else if(_status == COMPASS_CAL_RUNNING_STEP_TWO) {
202  if (_fit_step >= 35) {
203  if(fit_acceptable() && calculate_orientation()) {
204  set_status(COMPASS_CAL_SUCCESS);
205  } else {
206  set_status(COMPASS_CAL_FAILED);
207  failure = true;
208  }
209  } else if (_fit_step < 15) {
210  run_sphere_fit();
211  _fit_step++;
212  } else {
213  run_ellipsoid_fit();
214  _fit_step++;
215  }
216  }
217 }

255 bool CompassCalibrator::set_status(compass_cal_status_t status) {
256  if (status != COMPASS_CAL_NOT_STARTED && _status == status) {
257  return true;
258  }
259 
260  switch(status) {
261  case COMPASS_CAL_NOT_STARTED:
262  reset_state();
263  _status = COMPASS_CAL_NOT_STARTED;
264 
265  if(_sample_buffer != nullptr) {
266  free(_sample_buffer);
267  _sample_buffer = nullptr;
268  }
269  return true;
270 
271  case COMPASS_CAL_WAITING_TO_START:
272  reset_state();
273  _status = COMPASS_CAL_WAITING_TO_START;
274 
275  set_status(COMPASS_CAL_RUNNING_STEP_ONE);
276  return true;
277 
278  case COMPASS_CAL_RUNNING_STEP_ONE:
279  if(_status != COMPASS_CAL_WAITING_TO_START) {
280  return false;
281  }
282 
283  if(_attempt == 1 && (AP_HAL::millis()-_start_time_ms)*1.0e-3f < _delay_start_sec) {
284  return false;
285  }
286 
287  if (_sample_buffer == nullptr) {
288  _sample_buffer =
289  (CompassSample*) calloc(COMPASS_CAL_NUM_SAMPLES, sizeof(CompassSample));
290  }
291 
292  if(_sample_buffer != nullptr) {
293  initialize_fit();
294  _status = COMPASS_CAL_RUNNING_STEP_ONE;
295  return true;
296  }
297 
298  return false;
299 
300  case COMPASS_CAL_RUNNING_STEP_TWO:
301  if(_status != COMPASS_CAL_RUNNING_STEP_ONE) {
302  return false;
303  }
304  thin_samples();
305  initialize_fit();
306  _status = COMPASS_CAL_RUNNING_STEP_TWO;
307  return true;
308 
309  case COMPASS_CAL_SUCCESS:
310  if(_status != COMPASS_CAL_RUNNING_STEP_TWO) {
311  return false;
312  }
313 
314  if(_sample_buffer != nullptr) {
315  free(_sample_buffer);
316  _sample_buffer = nullptr;
317  }
318 
319  _status = COMPASS_CAL_SUCCESS;
320  return true;
321 
322  case COMPASS_CAL_FAILED:
323  if (_status == COMPASS_CAL_BAD_ORIENTATION) {
324  // don't overwrite bad orientation status
325  return false;
326  }
327  FALLTHROUGH;
328 
329  case COMPASS_CAL_BAD_ORIENTATION:
330  if(_status == COMPASS_CAL_NOT_STARTED) {
331  return false;
332  }
333 
334  if(_retry && set_status(COMPASS_CAL_WAITING_TO_START)) {
335  _attempt++;
336  return true;
337  }
338 
339  if(_sample_buffer != nullptr) {
340  free(_sample_buffer);
341  _sample_buffer = nullptr;
342  }
343 
344  _status = status;
345  return true;
346 
347  default:
348  return false;
349  };
350 }

352 bool CompassCalibrator::fit_acceptable() {
353  if( !isnan(_fitness) &&
354  _params.radius > 150 && _params.radius < 950 && //Earth's magnetic field strength range: 250-850mG
355  fabsf(_params.offset.x) < _offset_max &&
356  fabsf(_params.offset.y) < _offset_max &&
357  fabsf(_params.offset.z) < _offset_max &&
358  _params.diag.x > 0.2f && _params.diag.x < 5.0f &&
359  _params.diag.y > 0.2f && _params.diag.y < 5.0f &&
360  _params.diag.z > 0.2f && _params.diag.z < 5.0f &&
361  fabsf(_params.offdiag.x) < 1.0f && //absolute of sine/cosine output cannot be greater than 1
362  fabsf(_params.offdiag.y) < 1.0f &&
363  fabsf(_params.offdiag.z) < 1.0f ){
364 
365  return _fitness <= sq(_tolerance);
366  }
367  return false;
368 }

370 void CompassCalibrator::thin_samples() {
371  if(_sample_buffer == nullptr) {
372  return;
373  }
374 
375  _samples_thinned = 0;
376  // shuffle the samples http://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle
377  // this is so that adjacent samples don't get sequentially eliminated
378  for(uint16_t i=_samples_collected-1; i>=1; i--) {
379  uint16_t j = get_random16() % (i+1);
380  CompassSample temp = _sample_buffer[i];
381  _sample_buffer[i] = _sample_buffer[j];
382  _sample_buffer[j] = temp;
383  }
384 
385  for(uint16_t i=0; i < _samples_collected; i++) {
386  if(!accept_sample(_sample_buffer[i])) {
387  _sample_buffer[i] = _sample_buffer[_samples_collected-1];
388  _samples_collected --;
389  _samples_thinned ++;
390  }
391  }
392 
393  update_completion_mask();
394 }

396 /*
397  * The sample acceptance distance is determined as follows:
398  * For any regular polyhedron with triangular faces, the angle theta subtended
399  * by two closest points is defined as
400  *
401  * theta = arccos(cos(A)/(1-cos(A)))
402  *
403  * Where:
404  * A = (4pi/F + pi)/3
405  * and
406  * F = 2V - 4 is the number of faces for the polyhedron in consideration,
407  * which depends on the number of vertices V
408  *
409  * The above equation was proved after solving for spherical triangular excess
410  * and related equations.
411  */
412 bool CompassCalibrator::accept_sample(const Vector3f& sample)
413 {
414  static const uint16_t faces = (2 * COMPASS_CAL_NUM_SAMPLES - 4);
415  static const float a = (4.0f * M_PI / (3.0f * faces)) + M_PI / 3.0f;
416  static const float theta = 0.5f * acosf(cosf(a) / (1.0f - cosf(a)));
417 
418  if(_sample_buffer == nullptr) {
419  return false;
420  }
421 
422  float min_distance = _params.radius * 2*sinf(theta/2);
423 
424  for (uint16_t i = 0; i<_samples_collected; i++){
425  float distance = (sample - _sample_buffer[i].get()).length();
426  if(distance < min_distance) {
427  return false;
428  }
429  }
430  return true;
431 }

451 float CompassCalibrator::calc_mean_squared_residuals(const param_t& params) const
452 {
453  if(_sample_buffer == nullptr || _samples_collected == 0) {
454  return 1.0e30f;
455  }
456  float sum = 0.0f;
457  for(uint16_t i=0; i < _samples_collected; i++){
458  Vector3f sample = _sample_buffer[i].get();
459  float resid = calc_residual(sample, params);
460  sum += sq(resid);
461  }
462  sum /= _samples_collected;
463  return sum;
464 }

466 void CompassCalibrator::calc_sphere_jacob(const Vector3f& sample, const param_t& params, float* ret) const{
467  const Vector3f &offset = params.offset;
468  const Vector3f &diag = params.diag;
469  const Vector3f &offdiag = params.offdiag;
470  Matrix3f softiron(
471  diag.x , offdiag.x , offdiag.y,
472  offdiag.x , diag.y , offdiag.z,
473  offdiag.y , offdiag.z , diag.z
474  );
475 
476  float A = (diag.x * (sample.x + offset.x)) + (offdiag.x * (sample.y + offset.y)) + (offdiag.y * (sample.z + offset.z));
477  float B = (offdiag.x * (sample.x + offset.x)) + (diag.y * (sample.y + offset.y)) + (offdiag.z * (sample.z + offset.z));
478  float C = (offdiag.y * (sample.x + offset.x)) + (offdiag.z * (sample.y + offset.y)) + (diag.z * (sample.z + offset.z));
479  float length = (softiron*(sample+offset)).length();
480 
481  // 0: partial derivative (radius wrt fitness fn) fn operated on sample
482  ret[0] = 1.0f;
483  // 1-3: partial derivative (offsets wrt fitness fn) fn operated on sample
484  ret[1] = -1.0f * (((diag.x * A) + (offdiag.x * B) + (offdiag.y * C))/length);
485  ret[2] = -1.0f * (((offdiag.x * A) + (diag.y * B) + (offdiag.z * C))/length);
486  ret[3] = -1.0f * (((offdiag.y * A) + (offdiag.z * B) + (diag.z * C))/length);
487 }

499 void CompassCalibrator::run_sphere_fit()
500 {
501  if(_sample_buffer == nullptr) {
502  return;
503  }
504 
505  const float lma_damping = 10.0f;
506 
507  float fitness = _fitness;
508  float fit1, fit2;
509  param_t fit1_params, fit2_params;
510  fit1_params = fit2_params = _params;
511 
512  float JTJ[COMPASS_CAL_NUM_SPHERE_PARAMS*COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
513  float JTJ2[COMPASS_CAL_NUM_SPHERE_PARAMS*COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
514  float JTFI[COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
515 
516  // Gauss Newton Part common for all kind of extensions including LM
517  for(uint16_t k = 0; k<_samples_collected; k++) {
518  Vector3f sample = _sample_buffer[k].get();
519 
520  float sphere_jacob[COMPASS_CAL_NUM_SPHERE_PARAMS];
521 
522  calc_sphere_jacob(sample, fit1_params, sphere_jacob);
523 
524  for(uint8_t i = 0;i < COMPASS_CAL_NUM_SPHERE_PARAMS; i++) {
525  // compute JTJ
526  for(uint8_t j = 0; j < COMPASS_CAL_NUM_SPHERE_PARAMS; j++) {
527  JTJ[i*COMPASS_CAL_NUM_SPHERE_PARAMS+j] += sphere_jacob[i] * sphere_jacob[j];
528  JTJ2[i*COMPASS_CAL_NUM_SPHERE_PARAMS+j] += sphere_jacob[i] * sphere_jacob[j]; //a backup JTJ for LM
529  }
530  // compute JTFI
531  JTFI[i] += sphere_jacob[i] * calc_residual(sample, fit1_params);
532  }
533  }
534 
535 
536  //------------------------Levenberg-Marquardt-part-starts-here---------------------------------//
537  //refer: http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm#Choice_of_damping_parameter
538  for(uint8_t i = 0; i < COMPASS_CAL_NUM_SPHERE_PARAMS; i++) {
539  JTJ[i*COMPASS_CAL_NUM_SPHERE_PARAMS+i] += _sphere_lambda;
540  JTJ2[i*COMPASS_CAL_NUM_SPHERE_PARAMS+i] += _sphere_lambda/lma_damping;
541  }
542 
543  if(!inverse(JTJ, JTJ, 4)) {
544  return;
545  }
546 
547  if(!inverse(JTJ2, JTJ2, 4)) {
548  return;
549  }
550 
551  for(uint8_t row=0; row < COMPASS_CAL_NUM_SPHERE_PARAMS; row++) {
552  for(uint8_t col=0; col < COMPASS_CAL_NUM_SPHERE_PARAMS; col++) {
553  fit1_params.get_sphere_params()[row] -= JTFI[col] * JTJ[row*COMPASS_CAL_NUM_SPHERE_PARAMS+col];
554  fit2_params.get_sphere_params()[row] -= JTFI[col] * JTJ2[row*COMPASS_CAL_NUM_SPHERE_PARAMS+col];
555  }
556  }
557 
558  fit1 = calc_mean_squared_residuals(fit1_params);
559  fit2 = calc_mean_squared_residuals(fit2_params);
560 
561  if(fit1 > _fitness && fit2 > _fitness){
562  _sphere_lambda *= lma_damping;
563  } else if(fit2 < _fitness && fit2 < fit1) {
564  _sphere_lambda /= lma_damping;
565  fit1_params = fit2_params;
566  fitness = fit2;
567  } else if(fit1 < _fitness){
568  fitness = fit1;
569  }
570  //--------------------Levenberg-Marquardt-part-ends-here--------------------------------//
571 
572  if(!isnan(fitness) && fitness < _fitness) {
573  _fitness = fitness;
574  _params = fit1_params;
575  update_completion_mask();
576  }
577 }

581 void CompassCalibrator::calc_ellipsoid_jacob(const Vector3f& sample, const param_t& params, float* ret) const{
582  const Vector3f &offset = params.offset;
583  const Vector3f &diag = params.diag;
584  const Vector3f &offdiag = params.offdiag;
585  Matrix3f softiron(
586  diag.x , offdiag.x , offdiag.y,
587  offdiag.x , diag.y , offdiag.z,
588  offdiag.y , offdiag.z , diag.z
589  );
590 
591  float A = (diag.x * (sample.x + offset.x)) + (offdiag.x * (sample.y + offset.y)) + (offdiag.y * (sample.z + offset.z));
592  float B = (offdiag.x * (sample.x + offset.x)) + (diag.y * (sample.y + offset.y)) + (offdiag.z * (sample.z + offset.z));
593  float C = (offdiag.y * (sample.x + offset.x)) + (offdiag.z * (sample.y + offset.y)) + (diag.z * (sample.z + offset.z));
594  float length = (softiron*(sample+offset)).length();
595 
596  // 0-2: partial derivative (offset wrt fitness fn) fn operated on sample
597  ret[0] = -1.0f * (((diag.x * A) + (offdiag.x * B) + (offdiag.y * C))/length);
598  ret[1] = -1.0f * (((offdiag.x * A) + (diag.y * B) + (offdiag.z * C))/length);
599  ret[2] = -1.0f * (((offdiag.y * A) + (offdiag.z * B) + (diag.z * C))/length);
600  // 3-5: partial derivative (diag offset wrt fitness fn) fn operated on sample
601  ret[3] = -1.0f * ((sample.x + offset.x) * A)/length;
602  ret[4] = -1.0f * ((sample.y + offset.y) * B)/length;
603  ret[5] = -1.0f * ((sample.z + offset.z) * C)/length;
604  // 6-8: partial derivative (off-diag offset wrt fitness fn) fn operated on sample
605  ret[6] = -1.0f * (((sample.y + offset.y) * A) + ((sample.x + offset.x) * B))/length;
606  ret[7] = -1.0f * (((sample.z + offset.z) * A) + ((sample.x + offset.x) * C))/length;
607  ret[8] = -1.0f * (((sample.z + offset.z) * B) + ((sample.y + offset.y) * C))/length;
608 }

610 void CompassCalibrator::run_ellipsoid_fit()
611 {
612  if(_sample_buffer == nullptr) {
613  return;
614  }
615 
616  const float lma_damping = 10.0f;
617 
618 
619  float fitness = _fitness;
620  float fit1, fit2;
621  param_t fit1_params, fit2_params;
622  fit1_params = fit2_params = _params;
623 
624 
625  float JTJ[COMPASS_CAL_NUM_ELLIPSOID_PARAMS*COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
626  float JTJ2[COMPASS_CAL_NUM_ELLIPSOID_PARAMS*COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
627  float JTFI[COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
628 
629  // Gauss Newton Part common for all kind of extensions including LM
630  for(uint16_t k = 0; k<_samples_collected; k++) {
631  Vector3f sample = _sample_buffer[k].get();
632 
633  float ellipsoid_jacob[COMPASS_CAL_NUM_ELLIPSOID_PARAMS];
634 
635  calc_ellipsoid_jacob(sample, fit1_params, ellipsoid_jacob);
636 
637  for(uint8_t i = 0;i < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; i++) {
638  // compute JTJ
639  for(uint8_t j = 0; j < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; j++) {
640  JTJ [i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+j] += ellipsoid_jacob[i] * ellipsoid_jacob[j];
641  JTJ2[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+j] += ellipsoid_jacob[i] * ellipsoid_jacob[j];
642  }
643  // compute JTFI
644  JTFI[i] += ellipsoid_jacob[i] * calc_residual(sample, fit1_params);
645  }
646  }
647 
648 
649 
650  //------------------------Levenberg-Marquardt-part-starts-here---------------------------------//
651  //refer: http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm#Choice_of_damping_parameter
652  for(uint8_t i = 0; i < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; i++) {
653  JTJ[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+i] += _ellipsoid_lambda;
654  JTJ2[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+i] += _ellipsoid_lambda/lma_damping;
655  }
656 
657  if(!inverse(JTJ, JTJ, 9)) {
658  return;
659  }
660 
661  if(!inverse(JTJ2, JTJ2, 9)) {
662  return;
663  }
664 
665  for(uint8_t row=0; row < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; row++) {
666  for(uint8_t col=0; col < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; col++) {
667  fit1_params.get_ellipsoid_params()[row] -= JTFI[col] * JTJ[row*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+col];
668  fit2_params.get_ellipsoid_params()[row] -= JTFI[col] * JTJ2[row*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+col];
669  }
670  }
671 
672  fit1 = calc_mean_squared_residuals(fit1_params);
673  fit2 = calc_mean_squared_residuals(fit2_params);
674 
675  if(fit1 > _fitness && fit2 > _fitness){
676  _ellipsoid_lambda *= lma_damping;
677  } else if(fit2 < _fitness && fit2 < fit1) {
678  _ellipsoid_lambda /= lma_damping;
679  fit1_params = fit2_params;
680  fitness = fit2;
681  } else if(fit1 < _fitness){
682  fitness = fit1;
683  }
684  //--------------------Levenberg-part-ends-here--------------------------------//
685 
686  if(fitness < _fitness) {
687  _fitness = fitness;
688  _params = fit1_params;
689  update_completion_mask();
690  }
691 }

733 /*
734  calculate the implied earth field for a compass sample and compass
735  rotation. This is used to check for consistency between
736  samples.
737 
738  If the orientation is correct then when rotated the same (or
739  similar) earth field should be given for all samples.
740 
741  Note that this earth field uses an arbitrary north reference, so it
742  may not match the true earth field.
743  */
744 Vector3f CompassCalibrator::calculate_earth_field(CompassSample &sample, enum Rotation r)
745 {
746  Vector3f v = sample.get();
747 
748  // convert the sample back to sensor frame
749  v.rotate_inverse(_orientation);
750 
751  // rotate to body frame for this rotation
752  v.rotate(r);
753 
754  // apply offsets, rotating them for the orientation we are testing
755  Vector3f rot_offsets = _params.offset;
756  rot_offsets.rotate_inverse(_orientation);
757 
758  rot_offsets.rotate(r);
759 
760  v += rot_offsets;
761 
762  // rotate the sample from body frame back to earth frame
763  Matrix3f rot = sample.att.get_rotmat();
764 
765  Vector3f efield = rot * v;
766 
767  // earth field is the mag sample in earth frame
768  return efield;
769 }

771 /*
772  calculate compass orientation using the attitude estimate associated
773  with each sample, and fix orientation on external compasses if
774  the feature is enabled
775  */
776 bool CompassCalibrator::calculate_orientation(void)
777 {
778  if (!_check_orientation) {
779  // we are not checking orientation
780  return true;
781  }
782 
783  // this function is very slow
784  EXPECT_DELAY_MS(1000);
785 
786  float variance[ROTATION_MAX] {};
787 
788  for (enum Rotation r = ROTATION_NONE; r<ROTATION_MAX; r = (enum Rotation)(r+1)) {
789  // calculate the average implied earth field across all samples
790  Vector3f total_ef {};
791  for (uint32_t i=0; i<_samples_collected; i++) {
792  Vector3f efield = calculate_earth_field(_sample_buffer[i], r);
793  total_ef += efield;
794  }
795  Vector3f avg_efield = total_ef / _samples_collected;
796 
797  // now calculate the square error for this rotation against the average earth field
798  for (uint32_t i=0; i<_samples_collected; i++) {
799  Vector3f efield = calculate_earth_field(_sample_buffer[i], r);
800  float err = (efield - avg_efield).length_squared();
801  // divide by number of samples collected to get the variance
802  variance[r] += err / _samples_collected;
803  }
804  }
805 
806  // find the rotation with the lowest variance
807  enum Rotation besti = ROTATION_NONE;
808  float bestv = variance[0];
809  for (enum Rotation r = ROTATION_NONE; r<ROTATION_MAX; r = (enum Rotation)(r+1)) {
810  if (variance[r] < bestv) {
811  bestv = variance[r];
812  besti = r;
813  }
814  }
815 
816  // consider this a pass if the best orientation is 2x better
817  // variance than 2nd best
818  const float variance_threshold = 2.0;
819 
820  float second_best = besti==ROTATION_NONE?variance[1]:variance[0];
821  enum Rotation besti2 = ROTATION_NONE;
822  for (enum Rotation r = ROTATION_NONE; r<ROTATION_MAX; r = (enum Rotation)(r+1)) {
823  if (!rotation_equal(besti, r)) {
824  if (variance[r] < second_best) {
825  second_best = variance[r];
826  besti2 = r;
827  }
828  }
829  }
830 
831  _orientation_confidence = second_best/bestv;
832 
833  bool pass;
834  if (besti == _orientation) {
835  // if the orientation matched then allow for a low threshold
836  pass = true;
837  } else {
838  pass = _orientation_confidence > variance_threshold;
839  }
840  if (!pass) {
841  gcs().send_text(MAV_SEVERITY_CRITICAL, "Mag(%u) bad orientation: %u/%u %.1f", _compass_idx,
842  besti, besti2, (double)_orientation_confidence);
843  } else if (besti == _orientation) {
844  // no orientation change
845  gcs().send_text(MAV_SEVERITY_INFO, "Mag(%u) good orientation: %u %.1f", _compass_idx, besti, (double)_orientation_confidence);
846  } else if (!_is_external || !_fix_orientation) {
847  gcs().send_text(MAV_SEVERITY_CRITICAL, "Mag(%u) internal bad orientation: %u %.1f", _compass_idx, besti, (double)_orientation_confidence);
848  } else {
849  gcs().send_text(MAV_SEVERITY_INFO, "Mag(%u) new orientation: %u was %u %.1f", _compass_idx, besti, _orientation, (double)_orientation_confidence);
850  }
851 
852  if (!pass) {
853  set_status(COMPASS_CAL_BAD_ORIENTATION);
854  return false;
855  }
856 
857  if (_orientation == besti) {
858  // no orientation change
859  return true;
860  }
861 
862  if (!_is_external || !_fix_orientation) {
863  // we won't change the orientation, but we set _orientation
864  // for reporting purposes
865  _orientation = besti;
866  set_status(COMPASS_CAL_BAD_ORIENTATION);
867  return false;
868  }
869 
870  // correct the offsets for the new orientation
871  Vector3f rot_offsets = _params.offset;
872  rot_offsets.rotate_inverse(_orientation);
873  rot_offsets.rotate(besti);
874  _params.offset = rot_offsets;
875 
876  // rotate the samples for the new orientation
877  for (uint32_t i=0; i<_samples_collected; i++) {
878  Vector3f s = _sample_buffer[i].get();
879  s.rotate_inverse(_orientation);
880  s.rotate(besti);
881  _sample_buffer[i].set(s);
882  }
883 
884  _orientation = besti;
885 
886  // re-run the fit to get the diagonals and off-diagonals for the
887  // new orientation
888  initialize_fit();
889  run_sphere_fit();
890  run_ellipsoid_fit();
891 
892  return fit_acceptable();
893 }
