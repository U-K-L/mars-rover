218 /*
219  * init gyro
220  */
221 bool AP_InertialSensor_RST::_init_gyro(void)
222 {
223  uint8_t whoami;
224 
225  if (!_dev_gyro->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
226  return false;
227  }
228 
229  // set flag for reading registers
230  _dev_gyro->set_read_flag(0x80);
231 
232  _dev_gyro->set_speed(AP_HAL::Device::SPEED_HIGH);
233 
234  _dev_gyro->read_registers(GYRO_WHO_AM_I, &whoami, sizeof(whoami));
235  if (whoami != GYRO_WHO_I_AM) {
236  hal.console->printf("RST: unexpected gyro WHOAMI 0x%x\n", (unsigned)whoami);
237  printf("RST: unexpected gyro WHOAMI 0x%x\n", (unsigned)whoami);
238  goto fail_whoami;
239  }
240 
241  printf("detect i3g4250d\n");
242 
243  //enter power-down mode first
244  _dev_gyro->write_register(GYRO_CTRL_REG1, 0);
245 
246  hal.scheduler->delay(100);
247 
248  _dev_gyro->write_register(GYRO_CTRL_REG1,
249  GYRO_REG1_POWER_NORMAL |
250  GYRO_REG1_Z_ENABLE | GYRO_REG1_X_ENABLE | GYRO_REG1_Y_ENABLE |
251  RATE_800HZ_LP_50HZ);
252 
253  /* disable high-pass filters */
254  _dev_gyro->write_register(GYRO_CTRL_REG2, 0);
255 
256  /* DRDY disable */
257  _dev_gyro->write_register(GYRO_CTRL_REG3, 0x0);
258  _dev_gyro->write_register(GYRO_CTRL_REG4, RANGE_2000DPS);
259 
260  /* disable wake-on-interrupt */
261  _dev_gyro->write_register(GYRO_CTRL_REG5, GYRO_REG5_FIFO_ENABLE);
262 
263  /* disable FIFO. This makes things simpler and ensures we
264  * aren't getting stale data. It means we must run the hrt
265  * callback fast enough to not miss data. */
266  _dev_gyro->write_register(GYRO_FIFO_CTRL_REG, GYRO_FIFO_CTRL_BYPASS_MODE);
267 
268  _gyro_scale = 70e-3f / 180.0f * M_PI;
269 
270  hal.scheduler->delay(100);
271 
272  _dev_gyro->get_semaphore()->give();
273 
274  return true;
275 
276 fail_whoami:
277  _dev_gyro->get_semaphore()->give();
278  return false;
279 }

341 void AP_InertialSensor_RST::start(void)
342 {
343  _gyro_instance = _imu.register_gyro(800, _dev_gyro->get_bus_id_devtype(DEVTYPE_GYR_I3G4250D));
344  _accel_instance = _imu.register_accel(1000, _dev_accel->get_bus_id_devtype(DEVTYPE_ACC_IIS328DQ));

347  set_accel_orientation(_accel_instance, _rotation_a);
348 
349  // start the timer process to read samples

352 }

365 // Accumulate values from gyros
366 void AP_InertialSensor_RST::gyro_measure(void)
367 {
368  Vector3f gyro;
369  uint8_t status = 0;
370  int16_t raw_data[3];
371 
372  _dev_gyro->read_registers(GYRO_STATUS_REG, &status, sizeof(status));
373 
374  if ((status & GYRO_STATUS_ZYXDA) == 0) {
375  return;
376  }
377 
378  if (_dev_gyro->read_registers(GYRO_OUT_X_L | ADDR_INCREMENT, (uint8_t *)raw_data, sizeof(raw_data))) {
379  gyro = Vector3f(raw_data[0], raw_data[1], raw_data[2]);
380  gyro *= _gyro_scale;
381  _rotate_and_correct_gyro(_gyro_instance, gyro);
382  _notify_new_gyro_raw_sample(_gyro_instance, gyro);
383  }
384 }
