29 /*
30  * Polygon_outside(): test for a point in a polygon
31  * Input: P = a point,
32  * V[] = vertex points of a polygon V[n+1] with V[n]=V[0]
33  * Return: true if P is outside the polygon
34  *
35  * This does not take account of the curvature of the earth, but we
36  * expect that to be very small over the distances involved in the
37  * fence boundary
38  */
39 template <typename T>
40 bool Polygon_outside(const Vector2<T> &P, const Vector2<T> *V, unsigned n)
41 {
42  const bool complete = Polygon_complete(V, n);
43  if (complete) {
44  // the last point is the same as the first point; treat as if
45  // the last point wasn't passed in
46  n--;
47  }
48 
49  unsigned i, j;
50  // step through each edge pair-wise looking for crossings:
51  bool outside = true;
52  for (i=0; i<n; i++) {
53  j = i+1;
54  if (j >= n) {
55  j = 0;
56  }
57  if ((V[i].y > P.y) == (V[j].y > P.y)) {
58  continue;
59  }
60  const T dx1 = P.x - V[i].x;
61  const T dx2 = V[j].x - V[i].x;
62  const T dy1 = P.y - V[i].y;
63  const T dy2 = V[j].y - V[i].y;
64  const int8_t dx1s = (dx1 < 0) ? -1 : 1;
65  const int8_t dx2s = (dx2 < 0) ? -1 : 1;
66  const int8_t dy1s = (dy1 < 0) ? -1 : 1;
67  const int8_t dy2s = (dy2 < 0) ? -1 : 1;
68  const int8_t m1 = dx1s * dy2s;
69  const int8_t m2 = dx2s * dy1s;
70  // we avoid the 64 bit multiplies if we can based on sign checks.
71  if (dy2 < 0) {
72  if (m1 > m2) {
73  outside = !outside;
74  } else if (m1 < m2) {
75  continue;
76  } else {
77  if (std::is_floating_point<T>::value) {
78  if ( dx1 * dy2 > dx2 * dy1 ) {
79  outside = !outside;
80  }
81  } else {
82  if ( dx1 * (int64_t)dy2 > dx2 * (int64_t)dy1 ) {
83  outside = !outside;
84  }
85  }
86  }
87  } else {
88  if (m1 < m2) {
89  outside = !outside;
90  } else if (m1 > m2) {
91  continue;
92  } else {
93  if (std::is_floating_point<T>::value) {
94  if ( dx1 * dy2 < dx2 * dy1 ) {
95  outside = !outside;
96  }
97  } else {
98  if ( dx1 * (int64_t)dy2 < dx2 * (int64_t)dy1 ) {
99  outside = !outside;
100  }
101  }
102  }
103  }
104  }
105  return outside;
106 }

108 /*
109  * check if a polygon is complete.
110  *
111  * We consider a polygon to be complete if we have at least 4 points,
112  * and the first point is the same as the last point. That is the
113  * minimum requirement for the Polygon_outside function to work
114  */
115 template <typename T>
116 bool Polygon_complete(const Vector2<T> *V, unsigned n)
117 {
118  return (n >= 4 && V[n-1] == V[0]);
119 }

128 /*
129  determine if the polygon of N verticies defined by points V is
130  intersected by a line from point p1 to point p2
131  intersection argument returns the intersection closest to p1
132  */
133 bool Polygon_intersects(const Vector2f *V, unsigned N, const Vector2f &p1, const Vector2f &p2, Vector2f &intersection)
134 {
135  const bool complete = Polygon_complete(V, N);
136  if (complete) {
137  // if the last point is the same as the first point
138  // treat as if the last point wasn't passed in
139  N--;
140  }
141 
142  float intersect_dist_sq = FLT_MAX;
143  for (uint8_t i=0; i<N; i++) {
144  uint8_t j = i+1;
145  if (j >= N) {
146  j = 0;
147  }
148  const Vector2f &v1 = V[i];
149  const Vector2f &v2 = V[j];
150  // optimisations for common cases
151  if (v1.x > p1.x && v2.x > p1.x && v1.x > p2.x && v2.x > p2.x) {
152  continue;
153  }
154  if (v1.y > p1.y && v2.y > p1.y && v1.y > p2.y && v2.y > p2.y) {
155  continue;
156  }
157  if (v1.x < p1.x && v2.x < p1.x && v1.x < p2.x && v2.x < p2.x) {
158  continue;
159  }
160  if (v1.y < p1.y && v2.y < p1.y && v1.y < p2.y && v2.y < p2.y) {
161  continue;
162  }
163  Vector2f intersect_tmp;
164  if (Vector2f::segment_intersection(v1,v2,p1,p2,intersect_tmp)) {
165  float dist_sq = sq(intersect_tmp.x - p1.x) + sq(intersect_tmp.y - p1.y);
166  if (dist_sq < intersect_dist_sq) {
167  intersect_dist_sq = dist_sq;
168  intersection = intersect_tmp;
169  }
170  }
171  }
172  return (intersect_dist_sq < FLT_MAX);
173 }

175 /*
176  return the closest distance that a line from p1 to p2 comes to an
177  edge of closed polygon V, defined by N points
178  negative numbers indicate the line cross into the polygon with the negative size being the distance from p2 to the intersection point closest to p1
179  */
180 float Polygon_closest_distance_line(const Vector2f *V, unsigned N, const Vector2f &p1, const Vector2f &p2)
181 {
182  Vector2f intersection;
183  if (Polygon_intersects(V,N,p1,p2,intersection)) {
184  return -sqrtf(sq(intersection.x - p2.x) + sq(intersection.y - p2.y));
185  }
186  float closest_sq = FLT_MAX;
187  for (uint8_t i=0; i<N-1; i++) {
188  const Vector2f &v1 = V[i];
189  const Vector2f &v2 = V[i+1];
190 
191  float dist_sq = Vector2f::closest_distance_between_lines_squared(v1, v2, p1, p2);
192  if (dist_sq < closest_sq) {
193  closest_sq = dist_sq;
194  }
195  }
196  return sqrtf(closest_sq);
197 }
