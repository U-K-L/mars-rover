124  event_failed(IOEVENT_INIT);

126  }

349 /*
350  read count 16 bit registers
351 */
352 bool AP_IOMCU::read_registers(uint8_t page, uint8_t offset, uint8_t count, uint16_t *regs)
353 {
354  while (count > PKT_MAX_REGS) {
355  if (!read_registers(page, offset, PKT_MAX_REGS, regs)) {
356  return false;
357  }
358  offset += PKT_MAX_REGS;
359  count -= PKT_MAX_REGS;
360  regs += PKT_MAX_REGS;
361  }
362 
363  IOPacket pkt;
364 
365  discard_input();
366 
367  memset(&pkt.regs[0], 0, count*2);
368 
369  pkt.code = CODE_READ;
370  pkt.count = count;
371  pkt.page = page;
372  pkt.offset = offset;
373  pkt.crc = 0;
374 
375  uint8_t pkt_size = pkt.get_size();
376  if (is_chibios_backend) {
377  // save bandwidth on reads
378  pkt_size = 4;
379  }
380 
381  /*
382  the protocol is a bit strange, as it unnecessarily sends the
383  same size packet that it expects to receive. This means reading
384  a large number of registers wastes a lot of serial bandwidth
385  */
386  pkt.crc = crc_crc8((const uint8_t *)&pkt, pkt_size);
387  if (uart.write((uint8_t *)&pkt, pkt_size) != pkt_size) {
388  protocol_fail_count++;
389  return false;
390  }
391 
392  // wait for the expected number of reply bytes or timeout
393  if (!uart.wait_timeout(count*2+4, 10)) {
394  return false;
395  }
396 
397  uint8_t *b = (uint8_t *)&pkt;
398  uint8_t n = uart.available();
399  for (uint8_t i=0; i<n; i++) {
400  if (i < sizeof(pkt)) {
401  b[i] = uart.read();
402  }
403  }
404 
405  uint8_t got_crc = pkt.crc;
406  pkt.crc = 0;
407  uint8_t expected_crc = crc_crc8((const uint8_t *)&pkt, pkt.get_size());
408  if (got_crc != expected_crc) {
409  debug("bad crc %02x should be %02x n=%u %u/%u/%u\n",
410  got_crc, expected_crc,
411  n, page, offset, count);
412  protocol_fail_count++;
413  return false;
414  }
415 
416  if (pkt.code != CODE_SUCCESS) {
417  debug("bad code %02x read %u/%u/%u\n", pkt.code, page, offset, count);
418  protocol_fail_count++;
419  return false;
420  }
421  if (pkt.count < count) {
422  debug("bad count %u read %u/%u/%u n=%u\n", pkt.count, page, offset, count, n);
423  protocol_fail_count++;
424  return false;
425  }
426  memcpy(regs, pkt.regs, count*2);
427  if (protocol_fail_count > IOMCU_MAX_REPEATED_FAILURES) {
428  handle_repeated_failures();
429  }
430  protocol_fail_count = 0;
431  protocol_count++;
432  return true;
433 }

435 /*
436  write count 16 bit registers
437 */
438 bool AP_IOMCU::write_registers(uint8_t page, uint8_t offset, uint8_t count, const uint16_t *regs)
439 {
440  while (count > PKT_MAX_REGS) {
441  if (!write_registers(page, offset, PKT_MAX_REGS, regs)) {
442  return false;
443  }
444  offset += PKT_MAX_REGS;
445  count -= PKT_MAX_REGS;
446  regs += PKT_MAX_REGS;
447  }
448  IOPacket pkt;
449 
450  discard_input();
451 
452  memset(&pkt.regs[0], 0, count*2);
453 
454  pkt.code = CODE_WRITE;
455  pkt.count = count;
456  pkt.page = page;
457  pkt.offset = offset;
458  pkt.crc = 0;
459  memcpy(pkt.regs, regs, 2*count);
460  pkt.crc = crc_crc8((const uint8_t *)&pkt, pkt.get_size());
461  if (uart.write((uint8_t *)&pkt, pkt.get_size()) != pkt.get_size()) {
462  protocol_fail_count++;
463  return false;
464  }
465 
466  // wait for the expected number of reply bytes or timeout
467  if (!uart.wait_timeout(4, 10)) {
468  //debug("no reply for %u/%u/%u\n", page, offset, count);
469  protocol_fail_count++;
470  return false;
471  }
472 
473  uint8_t *b = (uint8_t *)&pkt;
474  uint8_t n = uart.available();
475  for (uint8_t i=0; i<n; i++) {
476  if (i < sizeof(pkt)) {
477  b[i] = uart.read();
478  }
479  }
480 
481  if (pkt.code != CODE_SUCCESS) {
482  debug("bad code %02x write %u/%u/%u %02x/%02x n=%u\n",
483  pkt.code, page, offset, count,
484  pkt.page, pkt.offset, n);
485  protocol_fail_count++;
486  return false;
487  }
488  uint8_t got_crc = pkt.crc;
489  pkt.crc = 0;

491  if (got_crc != expected_crc) {
492  debug("bad crc %02x should be %02x\n", got_crc, expected_crc);
493  protocol_fail_count++;

505 bool AP_IOMCU::modify_register(uint8_t page, uint8_t offset, uint16_t clearbits, uint16_t setbits)
506 {

696 /*
697  check ROMFS firmware against CRC on IOMCU, and if incorrect then upload new firmware
698  */
699 bool AP_IOMCU::check_crc(void)
700 {
701  // flash size minus 4k bootloader
702  const uint32_t flash_size = 0x10000 - 0x1000;
703 
704  fw = AP_ROMFS::find_decompress(fw_name, fw_size);
705  if (!fw) {
706  hal.console->printf("failed to find %s\n", fw_name);
707  return false;
708  }
709  uint32_t crc = crc_crc32(0, fw, fw_size);
710 
711  // pad CRC to max size
712  for (uint32_t i=0; i<flash_size-fw_size; i++) {
713  uint8_t b = 0xff;
714  crc = crc_crc32(crc, &b, 1);
715  }
716 
717  uint32_t io_crc = 0;
718  uint8_t tries = 32;
719  while (tries--) {
720  if (read_registers(PAGE_SETUP, PAGE_REG_SETUP_CRC, 2, (uint16_t *)&io_crc)) {
721  break;
722  }
723  }
724  if (io_crc == crc) {
725  hal.console->printf("IOMCU: CRC ok\n");
726  crc_is_ok = true;
727  free(fw);
728  fw = nullptr;
729  return true;
730  } else {
731  hal.console->printf("IOMCU: CRC mismatch expected: 0x%X got: 0x%X\n", (unsigned)crc, (unsigned)io_crc);
732  }
733 
734  const uint16_t magic = REBOOT_BL_MAGIC;
735  write_registers(PAGE_SETUP, PAGE_REG_SETUP_REBOOT_BL, 1, &magic);
736 
737  if (!upload_fw()) {
738  free(fw);
739  fw = nullptr;
740  AP_BoardConfig::sensor_config_error("Failed to update IO firmware");
741  }
742 
743  free(fw);
744  fw = nullptr;
745  return false;
746 }

828 /*
829  setup for mixing. This allows fixed wing aircraft to fly in manual
830  mode if the FMU dies
831  */
832 bool AP_IOMCU::setup_mixing(RCMapper *rcmap, int8_t override_chan,
833  float mixing_gain, uint16_t manual_rc_mask)
834 {
835  if (!is_chibios_backend) {
836  return false;
837  }
838  bool changed = false;
839 #define MIX_UPDATE(a,b) do { if ((a) != (b)) { a = b; changed = true; }} while (0)
840 
841  // update mixing structure, checking for changes
842  for (uint8_t i=0; i<IOMCU_MAX_CHANNELS; i++) {
843  const SRV_Channel *c = SRV_Channels::srv_channel(i);
844  if (!c) {
845  continue;
846  }
847  MIX_UPDATE(mixing.servo_trim[i], c->get_trim());
848  MIX_UPDATE(mixing.servo_min[i], c->get_output_min());
849  MIX_UPDATE(mixing.servo_max[i], c->get_output_max());
850  MIX_UPDATE(mixing.servo_function[i], c->get_function());
851  MIX_UPDATE(mixing.servo_reversed[i], c->get_reversed());
852  }
853  // update RCMap
854  MIX_UPDATE(mixing.rc_channel[0], rcmap->roll());
855  MIX_UPDATE(mixing.rc_channel[1], rcmap->pitch());
856  MIX_UPDATE(mixing.rc_channel[2], rcmap->throttle());
857  MIX_UPDATE(mixing.rc_channel[3], rcmap->yaw());
858  for (uint8_t i=0; i<4; i++) {
859  const RC_Channel *c = RC_Channels::rc_channel(mixing.rc_channel[i]-1);
860  if (!c) {
861  continue;
862  }
863  MIX_UPDATE(mixing.rc_min[i], c->get_radio_min());
864  MIX_UPDATE(mixing.rc_max[i], c->get_radio_max());
865  MIX_UPDATE(mixing.rc_trim[i], c->get_radio_trim());
866  MIX_UPDATE(mixing.rc_reversed[i], c->get_reverse());
867 
868  // cope with reversible throttle
869  if (i == 2 && c->get_type() == RC_Channel::RC_CHANNEL_TYPE_ANGLE) {
870  MIX_UPDATE(mixing.throttle_is_angle, 1);
871  } else {
872  MIX_UPDATE(mixing.throttle_is_angle, 0);
873  }
874  }
875 
876  MIX_UPDATE(mixing.rc_chan_override, override_chan);
877  MIX_UPDATE(mixing.mixing_gain, (uint16_t)(mixing_gain*1000));
878  MIX_UPDATE(mixing.manual_rc_mask, manual_rc_mask);
879 
880  // and enable
881  MIX_UPDATE(mixing.enabled, 1);
882  if (changed) {
883  trigger_event(IOEVENT_MIXING);
884  }
885  return true;
886 }
