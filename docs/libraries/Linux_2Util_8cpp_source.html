104 {
105  char *cbuf = (char *)buf;
106 
107  // try first to use machine-id file. Most systems will have this
108  const char *paths[] = { "/etc/machine-id", "/var/lib/dbus/machine-id" };
109  for (uint8_t i=0; i<ARRAY_SIZE(paths); i++) {
110  int fd = open(paths[i], O_RDONLY);
111  if (fd == -1) {
112  continue;
113  }
114  ssize_t ret = read(fd, buf, len);
115  close(fd);
116  if (ret <= 0) {
117  continue;
118  }
119  len = ret;
120  char *p = strchr(cbuf, '\n');
121  if (p) {
122  *p = 0;
123  }
124  len = strnlen(cbuf, len);
125  return true;
126  }
127 
128  // fallback to hostname
129  if (gethostname(cbuf, len) != 0) {
130  // use a default name so this always succeeds. Without it we can't
131  // implement some features (such as UAVCAN)
132  strncpy(cbuf, HAL_LINUX_DEFAULT_SYSTEM_ID, len);
133  }
134  len = strnlen(cbuf, len);
135  return true;
136 }
