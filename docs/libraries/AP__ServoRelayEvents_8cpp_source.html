29 bool AP_ServoRelayEvents::do_set_servo(uint8_t _channel, uint16_t pwm)
30 {
31  SRV_Channel *c = SRV_Channels::srv_channel(_channel-1);
32  if (c == nullptr) {
33  return false;
34  }
35  switch(c->get_function())
36  {
37  case SRV_Channel::k_none:
38  case SRV_Channel::k_manual:
39  case SRV_Channel::k_rcin1 ... SRV_Channel::k_rcin16: // rc pass-thru
40  break;
41  default:
42  gcs().send_text(MAV_SEVERITY_INFO, "ServoRelayEvent: Channel %d is already in use", _channel);
43  return false;
44  }
45  if (type == EVENT_TYPE_SERVO &&
46  channel == _channel) {
47  // cancel previous repeat
48  repeat = 0;
49  }
50  c->set_output_pwm(pwm);
51  c->ignore_small_rcin_changes();
52  return true;

55 bool AP_ServoRelayEvents::do_set_relay(uint8_t relay_num, uint8_t state)
56 {
57  AP_Relay *relay = AP::relay();
58  if (relay == nullptr) {
59  return false;
60  }
61 
62  if (!relay->enabled(relay_num)) {
63  return false;
64  }
65  if (type == EVENT_TYPE_RELAY &&
66  channel == relay_num) {
67  // cancel previous repeat
68  repeat = 0;
69  }
70  if (state == 1) {
71  relay->on(relay_num);
72  } else if (state == 0) {
73  relay->off(relay_num);
74  } else {
75  relay->toggle(relay_num);
76  }
77  return true;
78 }

127 /*
128  update state for MAV_CMD_DO_REPEAT_SERVO and MAV_CMD_DO_REPEAT_RELAY
129 */
130 void AP_ServoRelayEvents::update_events(void)
131 {
132  if (repeat == 0 || (AP_HAL::millis() - start_time_ms) < delay_ms) {
133  return;
134  }
135 
136  start_time_ms = AP_HAL::millis();
137 
138  switch (type) {
139  case EVENT_TYPE_SERVO: {
140  SRV_Channel *c = SRV_Channels::srv_channel(channel-1);
141  if (c != nullptr) {
142  if (repeat & 1) {
143  c->set_output_pwm(c->get_trim());
144  } else {
145  c->set_output_pwm(servo_value);
146  c->ignore_small_rcin_changes();
147  }
148  }
149  break;
150  }
151 
152  case EVENT_TYPE_RELAY: {
153  AP_Relay *relay = AP::relay();
154  if (relay != nullptr) {
155  relay->toggle(channel);
156  }
157  break;
158  }
159  }
160  if (repeat > 0) {
161  repeat--;
162  } else {
163  // toggle bottom bit so servos flip in value
164  repeat ^= 1;
165  }
166 }
