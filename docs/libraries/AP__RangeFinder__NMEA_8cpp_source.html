59 // return last value measured by sensor
60 bool AP_RangeFinder_NMEA::get_reading(uint16_t &reading_cm)
61 {
62  if (uart == nullptr) {
63  return false;
64  }
65 
66  // read any available lines from the lidar
67  float sum = 0.0f;
68  uint16_t count = 0;
69  int16_t nbytes = uart->available();
70  while (nbytes-- > 0) {
71  char c = uart->read();
72  if (decode(c)) {
73  sum += _distance_m;
74  count++;
75  }
76  }
77 
78  // return false on failure
79  if (count == 0) {
80  return false;
81  }
82 
83  // return average of all measurements
84  reading_cm = 100.0f * sum / count;
85  return true;
86 }

88 // add a single character to the buffer and attempt to decode
89 // returns true if a complete sentence was successfully decoded
90 bool AP_RangeFinder_NMEA::decode(char c)
91 {
92  switch (c) {
93  case ',':
94  // end of a term, add to checksum
95  _checksum ^= c;
96  FALLTHROUGH;
97  case '\r':
98  case '\n':
99  case '*':
100  {
101  // null terminate and decode latest term
102  _term[_term_offset] = 0;
103  bool valid_sentence = decode_latest_term();
104 
105  // move onto next term
106  _term_number++;
107  _term_offset = 0;
108  _term_is_checksum = (c == '*');
109  return valid_sentence;
110  }
111 
112  case '$': // sentence begin
113  _sentence_type = SONAR_UNKNOWN;
114  _term_number = 0;
115  _term_offset = 0;
116  _checksum = 0;
117  _term_is_checksum = false;
118  _distance_m = -1.0f;
119  return false;
120  }
121 
122  // ordinary characters are added to term
123  if (_term_offset < sizeof(_term) - 1)
124  _term[_term_offset++] = c;
125  if (!_term_is_checksum)
126  _checksum ^= c;
127 
128  return false;
129 }

131 // decode the most recently consumed term
132 // returns true if new sentence has just passed checksum test and is validated
133 bool AP_RangeFinder_NMEA::decode_latest_term()
134 {
135  // handle the last term in a message
136  if (_term_is_checksum) {
137  uint8_t checksum = 16 * char_to_hex(_term[0]) + char_to_hex(_term[1]);
138  return ((checksum == _checksum) &&
139  !is_negative(_distance_m) &&
140  (_sentence_type == SONAR_DBT || _sentence_type == SONAR_DPT));
141  }
142 
143  // the first term determines the sentence type
144  if (_term_number == 0) {
145  // the first two letters of the NMEA term are the talker ID.
146  // we accept any two characters here.
147  if (_term[0] < 'A' || _term[0] > 'Z' ||
148  _term[1] < 'A' || _term[1] > 'Z') {
149  _sentence_type = SONAR_UNKNOWN;
150  return false;
151  }
152  const char *term_type = &_term[2];
153  if (strcmp(term_type, "DBT") == 0) {
154  _sentence_type = SONAR_DBT;
155  } else if (strcmp(term_type, "DPT") == 0) {
156  _sentence_type = SONAR_DPT;
157  } else {
158  _sentence_type = SONAR_UNKNOWN;
159  }
160  return false;
161  }
162 
163  if (_sentence_type == SONAR_DBT) {
164  // parse DBT messages
165  if (_term_number == 3) {
166  _distance_m = atof(_term);
167  }
168  } else if (_sentence_type == SONAR_DPT) {
169  // parse DPT messages
170  if (_term_number == 1) {
171  _distance_m = atof(_term);
172  }
173  }
174 
175  return false;
176 }
