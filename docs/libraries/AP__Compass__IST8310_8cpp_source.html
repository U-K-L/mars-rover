104 bool AP_Compass_IST8310::init()
105 {
106  uint8_t reset_count = 0;
107 
108  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
109  return false;
110  }
111 
112  // high retries for init
113  _dev->set_retries(10);
114 
115  uint8_t whoami;
116  if (!_dev->read_registers(WAI_REG, &whoami, 1) ||
117  whoami != DEVICE_ID) {
118  // not an IST8310
119  goto fail;
120  }
121 
122  for (; reset_count < 5; reset_count++) {
123  if (!_dev->write_register(CNTL2_REG, CNTL2_VAL_SRST)) {
124  hal.scheduler->delay(10);
125  continue;
126  }
127 
128  hal.scheduler->delay(10);
129 
130  uint8_t cntl2 = 0xFF;
131  if (_dev->read_registers(CNTL2_REG, &cntl2, 1) &&
132  (cntl2 & 0x01) == 0) {
133  break;
134  }
135  }
136 
137  if (reset_count == 5) {
138  printf("IST8310: failed to reset device\n");
139  goto fail;
140  }
141 
142  if (!_dev->write_register(AVGCNTL_REG, AVGCNTL_VAL_Y_16 | AVGCNTL_VAL_XZ_16) ||
143  !_dev->write_register(PDCNTL_REG, PDCNTL_VAL_PULSE_DURATION_NORMAL)) {
144  printf("IST8310: found device but could not set it up\n");
145  goto fail;
146  }
147 
148  // lower retries for run
149  _dev->set_retries(3);
150 
151  // start state machine: request a sample
152  start_conversion();
153 
154  _dev->get_semaphore()->give();
155 
156  _instance = register_compass();
157 
158  printf("%s found on bus %u id %u address 0x%02x\n", name,
159  _dev->bus_num(), _dev->get_bus_id(), _dev->get_bus_address());
160 
161  set_rotation(_instance, _rotation);
162 
163  _dev->set_device_type(DEVTYPE_IST8310);
164  set_dev_id(_instance, _dev->get_bus_id());
165 
166  if (_force_external) {
167  set_external(_instance, true);
168  }
169 
170  _periodic_handle = _dev->register_periodic_callback(SAMPLING_PERIOD_USEC,
171  FUNCTOR_BIND_MEMBER(&AP_Compass_IST8310::timer, void));
172 
173  _perf_xfer_err = hal.util->perf_alloc(AP_HAL::Util::PC_COUNT, "IST8310_xfer_err");
174  _perf_bad_data = hal.util->perf_alloc(AP_HAL::Util::PC_COUNT, "IST8310_bad_data");
175 
176  return true;
177 
178 fail:
179  _dev->get_semaphore()->give();
180  return false;
181 }

191 void AP_Compass_IST8310::timer()
192 {
193  if (_ignore_next_sample) {
194  _ignore_next_sample = false;
195  start_conversion();
196  return;
197  }
198 
199  struct PACKED {
200  le16_t rx;
201  le16_t ry;
202  le16_t rz;
203  } buffer;
204 
205  bool ret = _dev->read_registers(OUTPUT_X_L_REG, (uint8_t *) &buffer, sizeof(buffer));
206  if (!ret) {
207  hal.util->perf_count(_perf_xfer_err);
208  return;
209  }
210 
211  start_conversion();
212 
213  /* same period, but start counting from now */
214  _dev->adjust_periodic_callback(_periodic_handle, SAMPLING_PERIOD_USEC);
215 
216  auto x = static_cast<int16_t>(le16toh(buffer.rx));
217  auto y = static_cast<int16_t>(le16toh(buffer.ry));
218  auto z = static_cast<int16_t>(le16toh(buffer.rz));
219 
220  /*
221  * Check if value makes sense according to the FSR and Resolution of
222  * this sensor, discarding outliers
223  */
224  if (x > IST8310_MAX_VAL_XY || x < IST8310_MIN_VAL_XY ||
225  y > IST8310_MAX_VAL_XY || y < IST8310_MIN_VAL_XY ||
226  z > IST8310_MAX_VAL_Z || z < IST8310_MIN_VAL_Z) {
227  hal.util->perf_count(_perf_bad_data);
228  return;
229  }
230 
231  // flip Z to conform to right-hand rule convention
232  z = -z;
233 
234  /* Resolution: 0.3 µT/LSB - already convert to milligauss */
235  Vector3f field = Vector3f{x * 3.0f, y * 3.0f, z * 3.0f};
236 
237  accumulate_sample(field, _instance);
238 }
