90 static void *malloc_flags(size_t size, uint32_t flags)
91 {
92  if (size == 0) {
93  return NULL;
94  }
95  const uint8_t dma_flags = (MEM_REGION_FLAG_DMA_OK | MEM_REGION_FLAG_SDCARD);
96  const uint8_t alignment = (flags&dma_flags?DMA_ALIGNMENT:MIN_ALIGNMENT);
97  void *p = NULL;
98  uint8_t i;
99 
100  if (flags & dma_flags) {
101  // allocate multiple of DMA alignment
102  size = (size + (DMA_ALIGNMENT-1)) & ~(DMA_ALIGNMENT-1);
103  }
104 
105  // if no flags are set or this is a DMA request and default heap
106  // is DMA safe then start with default heap
107  if (flags == 0 || (flags == MEM_REGION_FLAG_DMA_OK &&
108  (memory_regions[0].flags & MEM_REGION_FLAG_DMA_OK))) {
109  p = chHeapAllocAligned(NULL, size, alignment);
110  if (p) {
111  goto found;
112  }
113  }
114 
115  // try with matching flags
116  for (i=1; i<NUM_MEMORY_REGIONS; i++) {
117  if ((flags & MEM_REGION_FLAG_DMA_OK) &&
118  !(memory_regions[i].flags & MEM_REGION_FLAG_DMA_OK)) {
119  continue;
120  }
121  if ((flags & MEM_REGION_FLAG_SDCARD) &&
122  !(memory_regions[i].flags & MEM_REGION_FLAG_SDCARD)) {
123  continue;
124  }
125  if ((flags & MEM_REGION_FLAG_FAST) &&
126  !(memory_regions[i].flags & MEM_REGION_FLAG_FAST)) {
127  continue;
128  }
129  p = chHeapAllocAligned(&heaps[i], size, alignment);
130  if (p) {
131  goto found;
132  }
133  }
134 
135  // if this is a not a DMA request then we can fall back to any heap
136  if (!(flags & dma_flags)) {
137  for (i=1; i<NUM_MEMORY_REGIONS; i++) {
138  p = chHeapAllocAligned(&heaps[i], size, alignment);
139  if (p) {
140  goto found;
141  }
142  }
143  // try default heap
144  p = chHeapAllocAligned(NULL, size, alignment);
145  if (p) {
146  goto found;
147  }
148  }
149 
150 #if DMA_RESERVE_SIZE != 0
151  // fall back to DMA reserve
152  p = chHeapAllocAligned(&dma_reserve_heap, size, alignment);
153  if (p) {
154  memset(p, 0, size);
155  return p;
156  }
157 #endif
158 
159  // failed
160  return NULL;
161 
162 found:
163  memset(p, 0, size);
164  return p;
165 }

215 /*
216  return total available memory in bytes
217  */
218 size_t mem_available(void)
219 {
220  size_t totalp = 0;
221  uint8_t i;
222 
223  // get memory available on main heap
224  chHeapStatus(NULL, &totalp, NULL);
225 
226  // we also need to add in memory that is not yet allocated to the heap
227  totalp += chCoreGetStatusX();
228 
229  // now our own heaps
230  for (i=1; i<NUM_MEMORY_REGIONS; i++) {
231  size_t available = 0;
232  chHeapStatus(&heaps[i], &available, NULL);
233  totalp += available;
234  }
235 
236 #if DMA_RESERVE_SIZE != 0
237  // and reserve DMA heap
238  size_t available = 0;
239  chHeapStatus(&dma_reserve_heap, &available, NULL);
240  totalp += available;
241 #endif
242 
243  return totalp;
244 }
