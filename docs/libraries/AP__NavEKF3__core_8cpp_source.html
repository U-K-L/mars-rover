637 /*
638  * Update the quaternion, velocity and position states using delayed IMU measurements
639  * because the EKF is running on a delayed time horizon. Note that the quaternion is
640  * not used by the EKF equations, which instead estimate the error in the attitude of
641  * the vehicle when each observation is fused. This attitude error is then used to correct
642  * the quaternion.
643 */
644 void NavEKF3_core::UpdateStrapdownEquationsNED()
645 {
646  // update the quaternion states by rotating from the previous attitude through
647  // the delta angle rotation quaternion and normalise
648  // apply correction for earth's rotation rate
649  // % * - and + operators have been overloaded
650  stateStruct.quat.rotate(delAngCorrected - prevTnb * earthRateNED*imuDataDelayed.delAngDT);
651  stateStruct.quat.normalize();
652 
653  // transform body delta velocities to delta velocities in the nav frame
654  // use the nav frame from previous time step as the delta velocities
655  // have been rotated into that frame
656  // * and + operators have been overloaded
657  Vector3f delVelNav; // delta velocity vector in earth axes
658  delVelNav = prevTnb.mul_transpose(delVelCorrected);
659  delVelNav.z += GRAVITY_MSS*imuDataDelayed.delVelDT;
660 
661  // calculate the body to nav cosine matrix
662  stateStruct.quat.inverse().rotation_matrix(prevTnb);
663 
664  // calculate the rate of change of velocity (used for launch detect and other functions)
665  velDotNED = delVelNav / imuDataDelayed.delVelDT;
666 
667  // apply a first order lowpass filter
668  velDotNEDfilt = velDotNED * 0.05f + velDotNEDfilt * 0.95f;
669 
670  // calculate a magnitude of the filtered nav acceleration (required for GPS
671  // variance estimation)
672  accNavMag = velDotNEDfilt.length();
673  accNavMagHoriz = norm(velDotNEDfilt.x , velDotNEDfilt.y);
674 
675  // if we are not aiding, then limit the horizontal magnitude of acceleration
676  // to prevent large manoeuvre transients disturbing the attitude
677  if ((PV_AidingMode == AID_NONE) && (accNavMagHoriz > 5.0f)) {
678  float gain = 5.0f/accNavMagHoriz;
679  delVelNav.x *= gain;
680  delVelNav.y *= gain;
681  }
682 
683  // save velocity for use in trapezoidal integration for position calcuation
684  Vector3f lastVelocity = stateStruct.velocity;
685 
686  // sum delta velocities to get velocity
687  stateStruct.velocity += delVelNav;
688 
689  // apply a trapezoidal integration to velocities to calculate position
690  stateStruct.position += (stateStruct.velocity + lastVelocity) * (imuDataDelayed.delVelDT*0.5f);
691 
692  // accumulate the bias delta angle and time since last reset by an OF measurement arrival
693  delAngBodyOF += delAngCorrected;
694  delTimeOF += imuDataDelayed.delAngDT;
695 
696  // limit states to protect against divergence
697  ConstrainStates();
698 
699  // If main filter velocity states are valid, update the range beacon receiver position states
700  if (filterStatus.flags.horiz_vel) {
701  receiverPos += (stateStruct.velocity + lastVelocity) * (imuDataDelayed.delVelDT*0.5f);
702  }
703 }

705 /*
706  * Propagate PVA solution forward from the fusion time horizon to the current time horizon
707  * using simple observer which performs two functions:
708  * 1) Corrects for the delayed time horizon used by the EKF.
709  * 2) Applies a LPF to state corrections to prevent 'stepping' in states due to measurement
710  * fusion introducing unwanted noise into the control loops.
711  * The inspiration for using a complementary filter to correct for time delays in the EKF
712  * is based on the work by A Khosravian.
713  *
714  * "Recursive Attitude Estimation in the Presence of Multi-rate and Multi-delay Vector Measurements"
715  * A Khosravian, J Trumpf, R Mahony, T Hamel, Australian National University
716 */
717 void NavEKF3_core::calcOutputStates()
718 {
719  // apply corrections to the IMU data
720  Vector3f delAngNewCorrected = imuDataNew.delAng;
721  Vector3f delVelNewCorrected = imuDataNew.delVel;
722  correctDeltaAngle(delAngNewCorrected, imuDataNew.delAngDT, imuDataNew.gyro_index);
723  correctDeltaVelocity(delVelNewCorrected, imuDataNew.delVelDT, imuDataNew.accel_index);
724 
725  // apply corrections to track EKF solution
726  Vector3f delAng = delAngNewCorrected + delAngCorrection;
727 
728  // convert the rotation vector to its equivalent quaternion
729  Quaternion deltaQuat;
730  deltaQuat.from_axis_angle(delAng);
731 
732  // update the quaternion states by rotating from the previous attitude through
733  // the delta angle rotation quaternion and normalise
734  outputDataNew.quat *= deltaQuat;
735  outputDataNew.quat.normalize();
736 
737  // calculate the body to nav cosine matrix
738  Matrix3f Tbn_temp;
739  outputDataNew.quat.rotation_matrix(Tbn_temp);
740 
741  // transform body delta velocities to delta velocities in the nav frame
742  Vector3f delVelNav = Tbn_temp*delVelNewCorrected;
743  delVelNav.z += GRAVITY_MSS*imuDataNew.delVelDT;
744 
745  // save velocity for use in trapezoidal integration for position calcuation
746  Vector3f lastVelocity = outputDataNew.velocity;
747 
748  // sum delta velocities to get velocity
749  outputDataNew.velocity += delVelNav;
750 
751  // apply a trapezoidal integration to velocities to calculate position
752  outputDataNew.position += (outputDataNew.velocity + lastVelocity) * (imuDataNew.delVelDT*0.5f);
753 
754  // If the IMU accelerometer is offset from the body frame origin, then calculate corrections
755  // that can be added to the EKF velocity and position outputs so that they represent the velocity
756  // and position of the body frame origin.
757  // Note the * operator has been overloaded to operate as a dot product
758  if (!accelPosOffset.is_zero()) {
759  // calculate the average angular rate across the last IMU update
760  // note delAngDT is prevented from being zero in readIMUData()
761  Vector3f angRate = imuDataNew.delAng * (1.0f/imuDataNew.delAngDT);
762 
763  // Calculate the velocity of the body frame origin relative to the IMU in body frame
764  // and rotate into earth frame. Note % operator has been overloaded to perform a cross product
765  Vector3f velBodyRelIMU = angRate % (- accelPosOffset);
766  velOffsetNED = Tbn_temp * velBodyRelIMU;
767 
768  // calculate the earth frame position of the body frame origin relative to the IMU
769  posOffsetNED = Tbn_temp * (- accelPosOffset);
770  } else {
771  velOffsetNED.zero();
772  posOffsetNED.zero();
773  }
774 
775  // store INS states in a ring buffer that with the same length and time coordinates as the IMU data buffer
776  if (runUpdates) {
777  // store the states at the output time horizon
778  storedOutput[storedIMU.get_youngest_index()] = outputDataNew;
779 
780  // recall the states from the fusion time horizon
781  outputDataDelayed = storedOutput[storedIMU.get_oldest_index()];
782 
783  // compare quaternion data with EKF quaternion at the fusion time horizon and calculate correction
784 
785  // divide the demanded quaternion by the estimated to get the error
786  Quaternion quatErr = stateStruct.quat / outputDataDelayed.quat;
787 
788  // Convert to a delta rotation using a small angle approximation
789  quatErr.normalize();
790  Vector3f deltaAngErr;
791  float scaler;
792  if (quatErr[0] >= 0.0f) {
793  scaler = 2.0f;
794  } else {
795  scaler = -2.0f;
796  }
797  deltaAngErr.x = scaler * quatErr[1];
798  deltaAngErr.y = scaler * quatErr[2];
799  deltaAngErr.z = scaler * quatErr[3];
800 
801  // calculate a gain that provides tight tracking of the estimator states and
802  // adjust for changes in time delay to maintain consistent damping ratio of ~0.7
803  float timeDelay = 1e-3f * (float)(imuDataNew.time_ms - imuDataDelayed.time_ms);
804  timeDelay = MAX(timeDelay, dtIMUavg);
805  float errorGain = 0.5f / timeDelay;
806 
807  // calculate a correction to the delta angle
808  // that will cause the INS to track the EKF quaternions
809  delAngCorrection = deltaAngErr * errorGain * dtIMUavg;
810 
811  // calculate velocity and position tracking errors
812  Vector3f velErr = (stateStruct.velocity - outputDataDelayed.velocity);
813  Vector3f posErr = (stateStruct.position - outputDataDelayed.position);
814 
815  // collect magnitude tracking error for diagnostics
816  outputTrackError.x = deltaAngErr.length();
817  outputTrackError.y = velErr.length();
818  outputTrackError.z = posErr.length();
819 
820  // convert user specified time constant from centi-seconds to seconds
821  float tauPosVel = constrain_float(0.01f*(float)frontend->_tauVelPosOutput, 0.1f, 0.5f);
822 
823  // calculate a gain to track the EKF position states with the specified time constant
824  float velPosGain = dtEkfAvg / constrain_float(tauPosVel, dtEkfAvg, 10.0f);
825 
826  // use a PI feedback to calculate a correction that will be applied to the output state history
827  posErrintegral += posErr;
828  velErrintegral += velErr;
829  Vector3f velCorrection = velErr * velPosGain + velErrintegral * sq(velPosGain) * 0.1f;
830  Vector3f posCorrection = posErr * velPosGain + posErrintegral * sq(velPosGain) * 0.1f;
831 
832  // loop through the output filter state history and apply the corrections to the velocity and position states
833  // this method is too expensive to use for the attitude states due to the quaternion operations required
834  // but does not introduce a time delay in the 'correction loop' and allows smaller tracking time constants
835  // to be used
836  output_elements outputStates;
837  for (unsigned index=0; index < imu_buffer_length; index++) {
838  outputStates = storedOutput[index];
839 
840  // a constant velocity correction is applied
841  outputStates.velocity += velCorrection;
842 
843  // a constant position correction is applied
844  outputStates.position += posCorrection;
845 
846  // push the updated data to the buffer
847  storedOutput[index] = outputStates;
848  }
849 
850  // update output state to corrected values
851  outputDataNew = storedOutput[storedIMU.get_youngest_index()];
852 
853  }
854 }

856 /*
857  * Calculate the predicted state covariance matrix using algebraic equations generated with Matlab symbolic toolbox.
858  * The script file used to generate these and other equations in this filter can be found here:
859  * https://github.com/PX4/ecl/blob/master/matlab/scripts/Inertial%20Nav%20EKF/GenerateNavFilterEquations.m
860 */
861 void NavEKF3_core::CovariancePrediction()
862 {
863  hal.util->perf_begin(_perf_CovariancePrediction);
864  float daxVar; // X axis delta angle noise variance rad^2
865  float dayVar; // Y axis delta angle noise variance rad^2
866  float dazVar; // Z axis delta angle noise variance rad^2
867  float dvxVar; // X axis delta velocity variance noise (m/s)^2
868  float dvyVar; // Y axis delta velocity variance noise (m/s)^2
869  float dvzVar; // Z axis delta velocity variance noise (m/s)^2
870  float dvx; // X axis delta velocity (m/s)
871  float dvy; // Y axis delta velocity (m/s)
872  float dvz; // Z axis delta velocity (m/s)
873  float dax; // X axis delta angle (rad)
874  float day; // Y axis delta angle (rad)
875  float daz; // Z axis delta angle (rad)
876  float q0; // attitude quaternion
877  float q1; // attitude quaternion
878  float q2; // attitude quaternion
879  float q3; // attitude quaternion
880  float dax_b; // X axis delta angle measurement bias (rad)
881  float day_b; // Y axis delta angle measurement bias (rad)
882  float daz_b; // Z axis delta angle measurement bias (rad)
883  float dvx_b; // X axis delta velocity measurement bias (rad)
884  float dvy_b; // Y axis delta velocity measurement bias (rad)
885  float dvz_b; // Z axis delta velocity measurement bias (rad)
886 
887  // Calculate the time step used by the covariance prediction as an average of the gyro and accel integration period
888  // Constrain to prevent bad timing jitter causing numerical conditioning problems with the covariance prediction
889  dt = constrain_float(0.5f*(imuDataDelayed.delAngDT+imuDataDelayed.delVelDT),0.5f * dtEkfAvg, 2.0f * dtEkfAvg);
890 
891  // use filtered height rate to increase wind process noise when climbing or descending
892  // this allows for wind gradient effects.Filter height rate using a 10 second time constant filter
893  float alpha = 0.1f * dt;
894  hgtRate = hgtRate * (1.0f - alpha) - stateStruct.velocity.z * alpha;
895 
896  // calculate covariance prediction process noise added to diagonals of predicted covariance matrix
897  // error growth of first 10 kinematic states is built into auto-code for covariance prediction and driven by IMU noise parameters
898  Vector14 processNoiseVariance = {};
899 
900  if (!inhibitDelAngBiasStates) {
901  float dAngBiasVar = sq(sq(dt) * constrain_float(frontend->_gyroBiasProcessNoise, 0.0f, 1.0f));
902  for (uint8_t i=0; i<=2; i++) processNoiseVariance[i] = dAngBiasVar;
903  }
904 
905  if (!inhibitDelVelBiasStates) {
906  float dVelBiasVar = sq(sq(dt) * constrain_float(frontend->_accelBiasProcessNoise, 0.0f, 1.0f));
907  for (uint8_t i=3; i<=5; i++) {
908  uint8_t stateIndex = i + 10;
909  if (P[stateIndex][stateIndex] > 1E-8f) {
910  processNoiseVariance[i] = dVelBiasVar;
911  } else {
912  // increase the process noise variance up to a maximum of 100 x the nominal value if the variance is below the target minimum
913  processNoiseVariance[i] = 10.0f * dVelBiasVar * (1e-8f / fmaxf(P[stateIndex][stateIndex],1e-9f));
914  }
915  }
916  }
917 
918  if (!inhibitMagStates) {
919  float magEarthVar = sq(dt * constrain_float(frontend->_magEarthProcessNoise, 0.0f, 1.0f));
920  float magBodyVar = sq(dt * constrain_float(frontend->_magBodyProcessNoise, 0.0f, 1.0f));
921  for (uint8_t i=6; i<=8; i++) processNoiseVariance[i] = magEarthVar;
922  for (uint8_t i=9; i<=11; i++) processNoiseVariance[i] = magBodyVar;
923  }
924 
925  if (!inhibitWindStates) {
926  float windVelVar = sq(dt * constrain_float(frontend->_windVelProcessNoise, 0.0f, 1.0f) * (1.0f + constrain_float(frontend->_wndVarHgtRateScale, 0.0f, 1.0f) * fabsf(hgtRate)));
927  for (uint8_t i=12; i<=13; i++) processNoiseVariance[i] = windVelVar;
928  }
929 
930  // set variables used to calculate covariance growth
931  dvx = imuDataDelayed.delVel.x;
932  dvy = imuDataDelayed.delVel.y;
933  dvz = imuDataDelayed.delVel.z;
934  dax = imuDataDelayed.delAng.x;
935  day = imuDataDelayed.delAng.y;
936  daz = imuDataDelayed.delAng.z;
937  q0 = stateStruct.quat[0];
938  q1 = stateStruct.quat[1];
939  q2 = stateStruct.quat[2];
940  q3 = stateStruct.quat[3];
941  dax_b = stateStruct.gyro_bias.x;
942  day_b = stateStruct.gyro_bias.y;
943  daz_b = stateStruct.gyro_bias.z;
944  dvx_b = stateStruct.accel_bias.x;
945  dvy_b = stateStruct.accel_bias.y;
946  dvz_b = stateStruct.accel_bias.z;
947  float _gyrNoise = constrain_float(frontend->_gyrNoise, 0.0f, 1.0f);
948  daxVar = dayVar = dazVar = sq(dt*_gyrNoise);
949  float _accNoise = constrain_float(frontend->_accNoise, 0.0f, 10.0f);
950  dvxVar = dvyVar = dvzVar = sq(dt*_accNoise);
951 
952  // calculate the predicted covariance due to inertial sensor error propagation
953  // we calculate the lower diagonal and copy to take advantage of symmetry
954 
955  // intermediate calculations
956  Vector21 SF;
957  SF[0] = dvz - dvz_b;
958  SF[1] = dvy - dvy_b;
959  SF[2] = dvx - dvx_b;
960  SF[3] = 2*q1*SF[2] + 2*q2*SF[1] + 2*q3*SF[0];
961  SF[4] = 2*q0*SF[1] - 2*q1*SF[0] + 2*q3*SF[2];
962  SF[5] = 2*q0*SF[2] + 2*q2*SF[0] - 2*q3*SF[1];
963  SF[6] = day/2 - day_b/2;
964  SF[7] = daz/2 - daz_b/2;
965  SF[8] = dax/2 - dax_b/2;
966  SF[9] = dax_b/2 - dax/2;
967  SF[10] = daz_b/2 - daz/2;
968  SF[11] = day_b/2 - day/2;
969  SF[12] = 2*q1*SF[1];
970  SF[13] = 2*q0*SF[0];
971  SF[14] = q1/2;
972  SF[15] = q2/2;
973  SF[16] = q3/2;
974  SF[17] = sq(q3);
975  SF[18] = sq(q2);
976  SF[19] = sq(q1);
977  SF[20] = sq(q0);
978 
979  Vector8 SG;
980  SG[0] = q0/2;
981  SG[1] = sq(q3);
982  SG[2] = sq(q2);
983  SG[3] = sq(q1);
984  SG[4] = sq(q0);
985  SG[5] = 2*q2*q3;
986  SG[6] = 2*q1*q3;
987  SG[7] = 2*q1*q2;
988 
989  Vector11 SQ;
990  SQ[0] = dvzVar*(SG[5] - 2*q0*q1)*(SG[1] - SG[2] - SG[3] + SG[4]) - dvyVar*(SG[5] + 2*q0*q1)*(SG[1] - SG[2] + SG[3] - SG[4]) + dvxVar*(SG[6] - 2*q0*q2)*(SG[7] + 2*q0*q3);
991  SQ[1] = dvzVar*(SG[6] + 2*q0*q2)*(SG[1] - SG[2] - SG[3] + SG[4]) - dvxVar*(SG[6] - 2*q0*q2)*(SG[1] + SG[2] - SG[3] - SG[4]) + dvyVar*(SG[5] + 2*q0*q1)*(SG[7] - 2*q0*q3);
992  SQ[2] = dvzVar*(SG[5] - 2*q0*q1)*(SG[6] + 2*q0*q2) - dvyVar*(SG[7] - 2*q0*q3)*(SG[1] - SG[2] + SG[3] - SG[4]) - dvxVar*(SG[7] + 2*q0*q3)*(SG[1] + SG[2] - SG[3] - SG[4]);
993  SQ[3] = (dayVar*q1*SG[0])/2 - (dazVar*q1*SG[0])/2 - (daxVar*q2*q3)/4;
994  SQ[4] = (dazVar*q2*SG[0])/2 - (daxVar*q2*SG[0])/2 - (dayVar*q1*q3)/4;
995  SQ[5] = (daxVar*q3*SG[0])/2 - (dayVar*q3*SG[0])/2 - (dazVar*q1*q2)/4;
996  SQ[6] = (daxVar*q1*q2)/4 - (dazVar*q3*SG[0])/2 - (dayVar*q1*q2)/4;
997  SQ[7] = (dazVar*q1*q3)/4 - (daxVar*q1*q3)/4 - (dayVar*q2*SG[0])/2;
998  SQ[8] = (dayVar*q2*q3)/4 - (daxVar*q1*SG[0])/2 - (dazVar*q2*q3)/4;
999  SQ[9] = sq(SG[0]);
1000  SQ[10] = sq(q1);
1001 
1002  Vector11 SPP;
1003  SPP[0] = SF[12] + SF[13] - 2*q2*SF[2];
1004  SPP[1] = SF[17] - SF[18] - SF[19] + SF[20];
1005  SPP[2] = SF[17] - SF[18] + SF[19] - SF[20];
1006  SPP[3] = SF[17] + SF[18] - SF[19] - SF[20];
1007  SPP[4] = 2*q0*q2 - 2*q1*q3;
1008  SPP[5] = 2*q0*q1 - 2*q2*q3;
1009  SPP[6] = 2*q0*q3 - 2*q1*q2;
1010  SPP[7] = 2*q0*q1 + 2*q2*q3;
1011  SPP[8] = 2*q0*q3 + 2*q1*q2;
1012  SPP[9] = 2*q0*q2 + 2*q1*q3;
1013  SPP[10] = SF[16];
1014
