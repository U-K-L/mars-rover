54 SilentWings::SilentWings(const char *home_str, const char *frame_str) :
55  Aircraft(home_str, frame_str),
56  last_data_time_ms(0),
57  first_pkt_timestamp_ms(0),
58  time_base_us(0),
59  sock(true),
60  home_initialized(false),
61  inited_first_pkt_timestamp(false)
62 {
63  // Force ArduPlane to use sensor data from SilentWings as the actual state,
64  // without using EKF, i.e., using "fake EKF (type 10)". Disable gyro calibration.
65  // Set a few other parameters to specific values to keep the calibration checks happy.
66  // TO DO: fix this. Setting parameters in this way doesn't appear to have any effect.
67  for (uint8_t i = 0; i < ARRAY_SIZE(sim_defaults); i++) {
68  AP_Param::set_default_by_name(sim_defaults[i].name, sim_defaults[i].value);
69  if (sim_defaults[i].save) {
70  enum ap_var_type ptype;
71  AP_Param *p = AP_Param::find(sim_defaults[i].name, &ptype);
72  if (!p->configured()) {
73  p->save();
74  }
75  }
76  }

84 {
85  // Ignore the specified port_in.

90  fprintf(stderr, "SITL: socket in bind failed on sim in : %d - %s\n", _port_in, strerror(errno));
91  fprintf(stderr, "Aborting launch...\n");
92  exit(1);
93  }
94  printf("Bind %s:%d for SITL in\n", "127.0.0.1", _port_in);
95  sock.reuseaddress();

97

99  // Ignore the specified port_out.
100  printf("Setting Silent Wings interface to %s:%d \n", _sw_address, _sw_port);

104 /*
105  decode and send servos
106 */
107 void SilentWings::send_servos(const struct sitl_input &input)
108 {
109  char *buf = nullptr;
110  // Turn off direct joystick input to the simulator. All joystick commands
111  // should go through Mission Planner and get properly fused with ArduPlane's
112  // control inputs when in automatic flight modes.
113  float joystick = 0.0f;
114  float aileron = filtered_servo_angle(input, 0);
115  float elevator = filtered_servo_angle(input, 1);
116  float throttle = filtered_servo_range(input, 2);
117  float rudder = filtered_servo_angle(input, 3);
118 
119  ssize_t buflen = asprintf(&buf,
120  "JOY %f\n"
121  "AIL %f\n"
122  "ELE %f\n"
123  "RUD %f\n"
124  "THR %f\n",
125  joystick, aileron, elevator, rudder, throttle) - 1;
126 
127  if (buflen < 0) {
128  fprintf(stderr, "Fatal: Failed to allocate enough space for data\n"),
129  exit(1);
130  }
131 
132  ssize_t sent = sock.sendto(buf, buflen, _sw_address, _sw_port);
133  free(buf);
134 
135  if (sent < 0) {
136  fprintf(stderr, "Fatal: Failed to send on control socket\n"),
137  exit(1);
138  }
139 
140  if (sent < buflen) {
141  fprintf(stderr, "Failed to send all bytes on control socket\n");
142  }
143 }

146 /*
147  Receive an update from the FDM
148  This is a blocking function
149  */
150 bool SilentWings::recv_fdm(void)
151 {
152  fdm_packet tmp_pkt;
153  memset(&pkt, 0, sizeof(pkt));
154 
155  ssize_t nread = sock.recv(&tmp_pkt, sizeof(pkt), 0);
156 
157  // nread == -1 (255) means no data has arrived
158  if (nread != sizeof(pkt)) {
159  return false;
160  }
161 
162  memcpy(&pkt, &tmp_pkt, sizeof(pkt));
163 
164  // data received successfully
165  return true;
166 }

169 void SilentWings::process_packet()
170 {
171  // pkt.timestamp is the time of day in SilentWings, measured in ms
172  // since midnight.
173  // TO DO: check what happens when a flight in SW crosses midnight
174  if (inited_first_pkt_timestamp) {
175  uint64_t tus = (pkt.timestamp - first_pkt_timestamp_ms) * 1.0e3f;
176  time_now_us = time_base_us + tus;
177  }
178  else {
179  first_pkt_timestamp_ms = pkt.timestamp;
180  time_base_us = time_now_us;
181  inited_first_pkt_timestamp = true;
182  }
183 
184  dcm.from_euler(radians(pkt.roll), radians(pkt.pitch), radians(pkt.yaw));
185  accel_body = Vector3f(pkt.ax * GRAVITY_MSS, pkt.ay * GRAVITY_MSS, pkt.az * GRAVITY_MSS); // This is g-load.
186  gyro = Vector3f(radians(pkt.d_roll), radians(pkt.d_pitch), radians(pkt.d_yaw));
187  // SilentWings provides velocity in body frame.
188  velocity_ef = dcm * Vector3f(pkt.vx, pkt.vy, pkt.vz);
189  // SilentWings also provides velocity in body frame w.r.t. the wind, from which we can infer the wind.
190  wind_ef = dcm * (Vector3f(pkt.vx, pkt.vy, pkt.vz) - Vector3f(pkt.vx_wind, pkt.vy_wind, pkt.vz_wind));
191  airspeed = pkt.v_eas;
192  airspeed_pitot = pkt.v_eas;
193  Location curr_location;
194  curr_location.lat = pkt.position_latitude * 1.0e7;
195  curr_location.lng = pkt.position_longitude * 1.0e7;
196  curr_location.alt = pkt.altitude_msl * 100.0f;
197  ground_level = curr_location.alt * 0.01f - pkt.altitude_ground;
198  Vector3f posdelta = home.get_distance_NED(curr_location);
199  position.x = posdelta.x;
200  position.y = posdelta.y;
201  position.z = posdelta.z;
202  update_position();
203 
204  // In case Silent Wings' reported location and our location calculated using an offset from the home location diverge, we need
205  // to reset the home location.
206  if (curr_location.get_distance(location) > 4 || abs(curr_location.alt - location.alt)*0.01f > 2.0f || !home_initialized) {
207  printf("SilentWings home reset dist=%f alt=%.1f/%.1f\n",
208  curr_location.get_distance(location), curr_location.alt*0.01f, location.alt*0.01f);
209  // reset home location
210  home.lat = curr_location.lat;
211  home.lng = curr_location.lng;
212  // Resetting altitude reference point in flight can throw off a bunch
213  // of important calculations, so let the home altitude always be 0m MSL
214  home.alt = 0;
215  position.x = 0;
216  position.y = 0;
217  position.z = -curr_location.alt;
218  home_initialized = true;
219  update_position();
220  }
221 
222  // Auto-adjust to Silent Wings' frame rate
223  // This affects the data rate (without this adjustment, the data rate is
224  // low no matter what the output_udp_rate in SW's options.dat file is).
225  double deltat = (AP_HAL::millis() - last_data_time_ms) / 1000.0f;
226 
227  if (deltat < 0.01 && deltat > 0) {
228  adjust_frame_time(1.0/deltat);
229  }
230 
231  last_data_time_ms = AP_HAL::millis();
232 
233  report.data_count++;
234  report.frame_count++;
235 
236  if (0) {
237  printf("Delta: %f Time: %" PRIu64 "\n", deltat, time_now_us);
238  printf("Accel.x %f\n", accel_body.x);
239  printf("Accel.y %f\n", accel_body.y);
240  printf("Accel.z %f\n", accel_body.z);
241  printf("Gyro.x %f\n", gyro.x);
242  printf("Gyro.y %f\n", gyro.y);
243  printf("Gyro.z %f\n", gyro.z);
244  printf("Pos.x %f\n", position.x);
245  printf("Pos.y %f\n", position.y);
246  printf("Pos.z %f\n", position.z);
247  printf("Roll %f\n", pkt.roll);
248  printf("Pitch %f\n", pkt.pitch);
249  printf("Yaw %f\n", pkt.yaw);
250  }
251 }

253 /*
254  Extrapolates sensor data if Silent Wings hasn't sent us a data packet in a while.
255  */
256 bool SilentWings::interim_update()
257 {
258  if (AP_HAL::millis() - last_data_time_ms > 200) {
259  // don't extrapolate beyond 0.2s
260  return false;
261  }
262 
263  float delta_time = frame_time_us * 1e-6f;
264  time_now_us += frame_time_us;
265  extrapolate_sensors(delta_time);
266  update_position();
267  report.frame_count++;
268  return true;
269 }

272 /*
273  Update the Silent Wings simulation by one time step.
274  */
275 void SilentWings::update(const struct sitl_input &input)
276 {
277  if (recv_fdm()) {
278  process_packet();
279  // Time has been advanced by process_packet(.)
280  send_servos(input);
281  }
282  else if (interim_update()) {
283  // This clause is triggered only if we previously
284  // received at least one data packet.
285  // Time has been advanced by interim_update(.)
286  send_servos(input);
287  }
288 
289  // This clause is triggered if and only if we haven't received
290  // any data packets yet (and therefore didn't attempt
291  // extrapolating data via interim_update(.) either).
292  if (!inited_first_pkt_timestamp){
293  time_advance();
294  }
295  else {
296  if (use_time_sync) {
297  sync_frame_time();
298  }
299  }
300 
301  update_mag_field_bf();
302 
303  uint32_t now = AP_HAL::millis();
304 
305  if (report.last_report_ms == 0) {
306  report.last_report_ms = now;
307  printf("Resetting last report time to now\n");
308  }
309 
310  if (now - report.last_report_ms > 5000) {
311  float dt = (now - report.last_report_ms) * 1.0e-3f;
312  printf("Data rate: %.1f FPS Frame rate: %.1f FPS\n",
313  report.data_count/dt, report.frame_count/dt);
314  report.last_report_ms = now;
315  report.data_count = 0;
316  report.frame_count = 0;
317  }
318 }
