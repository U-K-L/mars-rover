52 void UARTDriver::begin(uint32_t baud, uint16_t rxSpace, uint16_t txSpace)
53 {
54  if (_portNumber >= ARRAY_SIZE(_sitlState->_uart_path)) {
55  AP_HAL::panic("port number out of range; you may need to extend _sitlState->_uart_path");
56  }
57 
58  const char *path = _sitlState->_uart_path[_portNumber];
59 
60  if (baud != 0) {
61  _uart_baudrate = baud;
62  }
63 
64  if (strcmp(path, "GPS1") == 0) {
65  /* gps */
66  _connected = true;
67  _fd = _sitlState->gps_pipe();
68  } else if (strcmp(path, "GPS2") == 0) {
69  /* 2nd gps */
70  _connected = true;
71  _fd = _sitlState->gps2_pipe();
72  } else {
73  /* parse type:args:flags string for path.
74  For example:
75  tcp:5760:wait // tcp listen on port 5760
76  tcp:0:wait // tcp listen on use base_port + 0
77  tcpclient:192.168.2.15:5762
78  udpclient:127.0.0.1
79  udpclient:127.0.0.1:14550
80  mcast:
81  mcast:239.255.145.50:14550
82  uart:/dev/ttyUSB0:57600
83  sim:ParticleSensor_SDS021:
84  */
85  char *saveptr = nullptr;
86  char *s = strdup(path);
87  char *devtype = strtok_r(s, ":", &saveptr);
88  char *args1 = strtok_r(nullptr, ":", &saveptr);
89  char *args2 = strtok_r(nullptr, ":", &saveptr);
90  if (strcmp(devtype, "tcp") == 0) {
91  uint16_t port = atoi(args1);
92  bool wait = (args2 && strcmp(args2, "wait") == 0);
93  _tcp_start_connection(port, wait);
94  } else if (strcmp(devtype, "tcpclient") == 0) {
95  if (args2 == nullptr) {
96  AP_HAL::panic("Invalid tcp client path: %s", path);
97  }
98  uint16_t port = atoi(args2);
99  _tcp_start_client(args1, port);
100  } else if (strcmp(devtype, "uart") == 0) {
101  uint32_t baudrate = args2? atoi(args2) : baud;
102  ::printf("UART connection %s:%u\n", args1, baudrate);
103  _uart_path = strdup(args1);
104  _uart_baudrate = baudrate;
105  _uart_start_connection();
106  } else if (strcmp(devtype, "sim") == 0) {
107  ::printf("SIM connection %s:%s on port %u\n", args1, args2, _portNumber);
108  if (!_connected) {
109  _connected = true;
110  _fd = _sitlState->sim_fd(args1, args2);
111  }
112  } else if (strcmp(devtype, "udpclient") == 0) {
113  // udp client connection
114  const char *ip = args1;
115  uint16_t port = args2?atoi(args2):14550;
116  if (!_connected) {
117  ::printf("UDP connection %s:%u\n", ip, port);
118  _udp_start_client(ip, port);
119  }
120  } else if (strcmp(devtype, "mcast") == 0) {
121  // udp multicast connection
122  const char *ip = args1 && *args1?args1:mcast_ip_default;
123  uint16_t port = args2?atoi(args2):mcast_port_default;
124  if (!_connected) {
125  ::printf("UDP multicast connection %s:%u\n", ip, port);
126  _udp_start_multicast(ip, port);
127  }
128  } else {
129  AP_HAL::panic("Invalid device path: %s", path);
130  }
131  free(s);
132  }
133 
134  if (hal.console != this) { // don't clear USB buffers (allows early startup messages to escape)
135  _readbuffer.clear();
136  _writebuffer.clear();
137  }
138 
139  _set_nonblocking(_fd);
140 }

189 size_t UARTDriver::write(const uint8_t *buffer, size_t size)
190 {
191  if (txspace() <= size) {
192  size = txspace();
193  }
194  if (size <= 0) {
195  return 0;
196  }
197  if (_unbuffered_writes) {
198  // write buffer straight to the file descriptor
199  const ssize_t nwritten = ::write(_fd, buffer, size);
200  if (nwritten == -1 && errno != EAGAIN && _uart_path) {
201  close(_fd);
202  _fd = -1;
203  _connected = false;
204  }
205  // these have no effect
206  tcdrain(_fd);
207  } else {
208  _writebuffer.write(buffer, size);
209  }
210  return size;
211 }

214 /*
215  start a TCP connection for the serial port. If wait_for_connection
216  is true then block until a client connects
217  */
218 void UARTDriver::_tcp_start_connection(uint16_t port, bool wait_for_connection)
219 {
220  int one=1;
221  struct sockaddr_in sockaddr;
222  int ret;
223 
224  if (_connected) {
225  return;
226  }
227 
228  _use_send_recv = true;
229 
230  if (_console) {
231  // hack for console access
232  _connected = true;
233  _use_send_recv = false;
234  _listen_fd = -1;
235  _fd = 1;
236  return;
237  }
238 
239  if (_fd != -1) {
240  close(_fd);
241  }
242 
243  if (_listen_fd == -1) {
244  memset(&sockaddr,0,sizeof(sockaddr));
245 
246 #ifdef HAVE_SOCK_SIN_LEN
247  sockaddr.sin_len = sizeof(sockaddr);
248 #endif
249  if (port > 1000) {
250  sockaddr.sin_port = htons(port);
251  } else {
252  sockaddr.sin_port = htons(_sitlState->base_port() + port);
253  }
254  sockaddr.sin_family = AF_INET;
255 
256  _listen_fd = socket(AF_INET, SOCK_STREAM, 0);
257  if (_listen_fd == -1) {
258  fprintf(stderr, "socket failed - %s\n", strerror(errno));
259  exit(1);
260  }
261  ret = fcntl(_listen_fd, F_SETFD, FD_CLOEXEC);
262  if (ret == -1) {
263  fprintf(stderr, "fcntl failed on setting FD_CLOEXEC - %s\n", strerror(errno));
264  exit(1);
265  }
266 
267  /* we want to be able to re-use ports quickly */
268  if (setsockopt(_listen_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one)) == -1) {
269  fprintf(stderr, "setsockopt failed: %s\n", strerror(errno));
270  exit(1);
271  }
272 
273  fprintf(stderr, "bind port %u for %u\n",
274  (unsigned)ntohs(sockaddr.sin_port),
275  (unsigned)_portNumber);
276 
277  ret = bind(_listen_fd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
278  if (ret == -1) {
279  fprintf(stderr, "bind failed on port %u - %s\n",
280  (unsigned)ntohs(sockaddr.sin_port),
281  strerror(errno));
282  exit(1);
283  }
284 
285  ret = listen(_listen_fd, 5);
286  if (ret == -1) {
287  fprintf(stderr, "listen failed - %s\n", strerror(errno));
288  exit(1);
289  }
290 
291  fprintf(stderr, "Serial port %u on TCP port %u\n", _portNumber,
292  _sitlState->base_port() + _portNumber);
293  fflush(stdout);
294  }
295 
296  if (wait_for_connection) {
297  fprintf(stdout, "Waiting for connection ....\n");
298  fflush(stdout);
299  _fd = accept(_listen_fd, nullptr, nullptr);
300  if (_fd == -1) {
301  fprintf(stderr, "accept() error - %s", strerror(errno));
302  exit(1);
303  }
304  setsockopt(_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
305  setsockopt(_fd, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one));
306  fcntl(_fd, F_SETFD, FD_CLOEXEC);
307  _connected = true;
308  fprintf(stdout, "Connection on serial port %u\n", _portNumber);
309  }
310 }

426  start a UDP multicast connection
427  */
428 void UARTDriver::_udp_start_multicast(const char *address, uint16_t port)
429 {
430  if (_connected) {
431  return;
432  }
433 
434  // establish the listening port
435  struct sockaddr_in sockaddr;
436  int ret;
437 
438  memset(&sockaddr,0,sizeof(sockaddr));
439 
440 #ifdef HAVE_SOCK_SIN_LEN
441  sockaddr.sin_len = sizeof(sockaddr);
442 #endif
443  sockaddr.sin_port = htons(port);
444  sockaddr.sin_family = AF_INET;
445  sockaddr.sin_addr.s_addr = inet_addr(address);
446 
447  _mc_fd = socket(AF_INET, SOCK_DGRAM, 0);
448  if (_mc_fd == -1) {
449  fprintf(stderr, "socket failed - %s\n", strerror(errno));
450  exit(1);
451  }
452  ret = fcntl(_mc_fd, F_SETFD, FD_CLOEXEC);
453  if (ret == -1) {
454  fprintf(stderr, "fcntl failed on setting FD_CLOEXEC - %s\n", strerror(errno));
455  exit(1);
456  }
457  int one = 1;
458  if (setsockopt(_mc_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one)) == -1) {
459  fprintf(stderr, "setsockopt failed: %s\n", strerror(errno));
460  exit(1);
461  }
462 
463  // close on exec, to allow reboot
464  fcntl(_mc_fd, F_SETFD, FD_CLOEXEC);
465 
466  ret = bind(_mc_fd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
467  if (ret == -1) {
468  fprintf(stderr, "multicast bind failed on port %u - %s\n",
469  (unsigned)ntohs(sockaddr.sin_port),
470  strerror(errno));
471  exit(1);
472  }
473 
474  struct ip_mreq mreq {};
475  mreq.imr_multiaddr.s_addr = inet_addr(address);
476  mreq.imr_interface.s_addr = htonl(INADDR_ANY);
477 
478  ret = setsockopt(_mc_fd, IPPROTO_IP, IP_ADD_MEMBERSHIP, &mreq, sizeof(mreq));
479  if (ret == -1) {
480  fprintf(stderr, "multicast membership add failed on port %u - %s\n",
481  (unsigned)ntohs(sockaddr.sin_port),
482  strerror(errno));
483  exit(1);
484  }
485 
486  // now start the outgoing connection as an ordinary UDP connection
487  _udp_start_client(address, port);
488 }

491 /*
492  start a UART connection for the serial port
493  */
494 void UARTDriver::_uart_start_connection(void)
495 {
496  struct termios t {};
497  if (!_connected) {
498  _fd = ::open(_uart_path, O_RDWR | O_CLOEXEC);
499  if (_fd == -1) {
500  return;
501  }
502  // use much smaller buffer sizes on real UARTs
503  _writebuffer.set_size(1024);
504  _readbuffer.set_size(512);
505  ::printf("Opened %s\n", _uart_path);
506  }
507 
508  if (_fd == -1) {
509  AP_HAL::panic("Unable to open UART %s", _uart_path);
510  }
511 
512  // set non-blocking
513  int flags = fcntl(_fd, F_GETFL, 0);
514  flags = flags | O_NONBLOCK;
515  fcntl(_fd, F_SETFL, flags);
516 
517  // disable LF -> CR/LF
518  tcgetattr(_fd, &t);
519  t.c_iflag &= ~(BRKINT | ICRNL | IMAXBEL | IXON | IXOFF);
520  t.c_oflag &= ~(OPOST | ONLCR);
521  t.c_lflag &= ~(ISIG | ICANON | IEXTEN | ECHO | ECHOE | ECHOK | ECHOCTL | ECHOKE);
522  t.c_cc[VMIN] = 0;
523  if (_sitlState->use_rtscts()) {
524  t.c_cflag |= CRTSCTS;
525  }
526  tcsetattr(_fd, TCSANOW, &t);
527 
528  // set baudrate
529  set_speed(_uart_baudrate);
530 
531  _connected = true;
532  _use_send_recv = false;
533 }

557 /*
558  use select() to see if something is pending
559  */
560 bool UARTDriver::_select_check(int fd)
561 {
562  if (fd == -1) {
563  return false;
564  }
565  fd_set fds;
566  struct timeval tv;
567 
568  FD_ZERO(&fds);
569  FD_SET(fd, &fds);
570 
571  // zero time means immediate return from select()
572  tv.tv_sec = 0;
573  tv.tv_usec = 0;
574 
575  if (select(fd+1, &fds, nullptr, nullptr, &tv) == 1) {
576  return true;
577  }
578  return false;
579 }

612 void UARTDriver::_timer_tick(void)
613 {
614  if (!_connected) {
615  _check_reconnect();
616  return;
617  }
618  ssize_t nwritten;
619  uint32_t max_bytes = 10000;
620  SITL::SITL *_sitl = AP::sitl();
621  if (_sitl && _sitl->telem_baudlimit_enable) {
622  // limit byte rate to configured baudrate
623  uint32_t now = AP_HAL::micros();
624  float dt = 1.0e-6 * (now - last_tick_us);
625  max_bytes = _uart_baudrate * dt / 10;
626  if (max_bytes == 0) {
627  return;
628  }
629  last_tick_us = now;
630  }
631 
632  if (_packetise) {
633  uint16_t n = _writebuffer.available();
634  n = MIN(n, max_bytes);
635  if (n > 0) {
636  n = mavlink_packetise(_writebuffer, n);
637  }
638  if (n > 0) {
639  // keep as a single UDP packet
640  uint8_t tmpbuf[n];
641  _writebuffer.peekbytes(tmpbuf, n);
642  ssize_t ret = send(_fd, tmpbuf, n, MSG_DONTWAIT);
643  if (ret > 0) {
644  _writebuffer.advance(ret);
645  }
646  }
647  } else {
648  uint32_t navail;
649  const uint8_t *readptr = _writebuffer.readptr(navail);
650  if (readptr && navail > 0) {
651  navail = MIN(navail, max_bytes);
652  if (!_use_send_recv) {
653  nwritten = ::write(_fd, readptr, navail);
654  if (nwritten == -1 && errno != EAGAIN && _uart_path) {
655  close(_fd);
656  _fd = -1;
657  _connected = false;
658  }
659  } else {
660  nwritten = send(_fd, readptr, navail, MSG_DONTWAIT);
661  }
662  if (nwritten > 0) {
663  _writebuffer.advance(nwritten);
664  }
665  }
666  }
667 
668  uint32_t space = _readbuffer.space();
669  if (space == 0) {
670  return;
671  }
672  space = MIN(space, max_bytes);
673 
674  char buf[space];
675  ssize_t nread = 0;
676  if (_mc_fd >= 0) {
677  if (_select_check(_mc_fd)) {
678  struct sockaddr_in from;
679  socklen_t fromlen = sizeof(from);
680  nread = recvfrom(_mc_fd, buf, space, MSG_DONTWAIT, (struct sockaddr *)&from, &fromlen);
681  uint16_t port = ntohs(from.sin_port);
682  if (_mc_myport == 0) {
683  // get our own address, so we can recognise packets from ourself
684  struct sockaddr_in myaddr;
685  socklen_t myaddrlen;
686  if (getsockname(_fd, (struct sockaddr *)&myaddr, &myaddrlen) == 0) {
687  _mc_myport = ntohs(myaddr.sin_port);
688  }
689  }
690  if (_mc_myport == port) {
691  // assume this is a packet from ourselves. This is not
692  // entirely accurate, as it could be a packet from
693  // another machine that has assigned the same port,
694  // unfortunately we don't have a better way to detect
695  // packets from ourselves
696  nread = 0;
697  }

699  } else if (!_use_send_recv) {

701  return;
702  }

706  close(_fd);

709  }

712  if (nread <= 0 && !_is_udp) {
713  // the socket has reached EOF
714  close(_fd);
715  _fd = -1;
716  _connected = false;
717  fprintf(stdout, "Closed connection on serial port %u\n", _portNumber);
718  fflush(stdout);
719  return;
720  }
721  }
722  if (nread > 0) {
723  _readbuffer.write((uint8_t *)buf, nread);
724  _receive_timestamp = AP_HAL::micros64();
725  }
726 }

728 /*
729  return timestamp estimate in microseconds for when the start of
730  a nbytes packet arrived on the uart. This should be treated as a
731  time constraint, not an exact time. It is guaranteed that the
732  packet did not start being received after this time, but it
733  could have been in a system buffer before the returned time.
734 
735  This takes account of the baudrate of the link. For transports
736  that have no baudrate (such as USB) the time estimate may be
737  less accurate.
738 
739  A return value of zero means the HAL does not support this API
740 */
741 uint64_t UARTDriver::receive_time_constraint_us(uint16_t nbytes)
742 {
743  uint64_t last_receive_us = _receive_timestamp;
744  if (_uart_baudrate > 0) {
745  // assume 10 bits per byte.
746  uint32_t transport_time_us = (1000000UL * 10UL / _uart_baudrate) * (nbytes+available());
747  last_receive_us -= transport_time_us;
748  }
749  return last_receive_us;
