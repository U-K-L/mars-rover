209 void Frame::init(float _mass, float hover_throttle, float _terminal_velocity, float _terminal_rotation_rate)
210 {
211  /*
212  scaling from total motor power to Newtons. Allows the copter
213  to hover against gravity when each motor is at hover_throttle
214  */
215  thrust_scale = (_mass * GRAVITY_MSS) / (num_motors * hover_throttle);
216 
217  terminal_velocity = _terminal_velocity;
218  terminal_rotation_rate = _terminal_rotation_rate;
219 }

222  find a frame by name
223  */
224 Frame *Frame::find_frame(const char *name)
225 {
226  for (uint8_t i=0; i < ARRAY_SIZE(supported_frames); i++) {
227  // do partial name matching to allow for frame variants
228  if (strncasecmp(name, supported_frames[i].name, strlen(supported_frames[i].name)) == 0) {
229  return &supported_frames[i];
230  }
231  }
232  return nullptr;
233 }

235 // calculate rotational and linear accelerations
236 void Frame::calculate_forces(const Aircraft &aircraft,
237  const struct sitl_input &input,
238  Vector3f &rot_accel,
239  Vector3f &body_accel)
240 {
241  Vector3f thrust; // newtons
242 
243  for (uint8_t i=0; i<num_motors; i++) {
244  Vector3f mraccel, mthrust;
245  motors[i].calculate_forces(input, thrust_scale, motor_offset, mraccel, mthrust);
246  rot_accel += mraccel;
247  thrust += mthrust;
248  }
249 
250  body_accel = thrust/aircraft.gross_mass();
251 
252  if (terminal_rotation_rate > 0) {
253  // rotational air resistance
254  const Vector3f &gyro = aircraft.get_gyro();
255  rot_accel.x -= gyro.x * radians(400.0) / terminal_rotation_rate;
256  rot_accel.y -= gyro.y * radians(400.0) / terminal_rotation_rate;
257  rot_accel.z -= gyro.z * radians(400.0) / terminal_rotation_rate;
258  }
259 
260  if (terminal_velocity > 0) {
261  // air resistance
262  Vector3f air_resistance = -aircraft.get_velocity_air_ef() * (GRAVITY_MSS/terminal_velocity);
263  body_accel += aircraft.get_dcm().transposed() * air_resistance;
264  }
265 
266  // add some noise
267  const float gyro_noise = radians(0.1);
268  const float accel_noise = 0.3;
269  const float noise_scale = thrust.length() / (thrust_scale * num_motors);
270  rot_accel += Vector3f(aircraft.rand_normal(0, 1),
271  aircraft.rand_normal(0, 1),
272  aircraft.rand_normal(0, 1)) * gyro_noise * noise_scale;
273  body_accel += Vector3f(aircraft.rand_normal(0, 1),
274  aircraft.rand_normal(0, 1),
275  aircraft.rand_normal(0, 1)) * accel_noise * noise_scale;
276 }

279 // calculate current and voltage
280 void Frame::current_and_voltage(const struct sitl_input &input, float &voltage, float &current)
281 {
282  voltage = 0;
283  current = 0;
284  for (uint8_t i=0; i<num_motors; i++) {
285  float c, v;
286  motors[i].current_and_voltage(input, v, c, motor_offset);
287  current += c;
288  voltage += v;
289  }
290  // use average for voltage, total for current
291  voltage /= num_motors;
292 }
