82 /* By looking at the datasheet, the accel range i (as defined by the macros
83  * BMI160_ACC_RANGE_*G) maps to the range bits by the function f defined:
84  * f(0) = 3; f(i) = f(i - 1) + i + 1
85  * Which can be written as the closed formula:
86  * f(i) = (i * (i + 3)) / 2 + 3 */
87 #define BMI160_ACC_RANGE_BITS \
88  (BMI160_ACC_RANGE * (BMI160_ACC_RANGE + 3) / 2 + 3)
89 
90 /* The rate in Hz based on the ODR bits can be calculated with
91  * 100 / (2 ^ (8 - odr) */
92 #define BMI160_ODR_TO_HZ(odr_) \
93  (uint16_t)(odr_ > 8 ? 100 << (odr_ - 8) : 100 >> (8 - odr_))

214 bool AP_InertialSensor_BMI160::_configure_accel()
215 {
216  bool r;
217 
218  r = _dev->write_register(BMI160_REG_ACC_CONF, BMI160_OSR | BMI160_ODR);
219  if (!r) {
220  return false;
221  }
222  hal.scheduler->delay(1);
223 
224  _check_err_reg();
225 
226  r = _dev->write_register(BMI160_REG_ACC_RANGE, BMI160_ACC_RANGE_BITS);
227  if (!r) {
228  return false;
229  }
230  hal.scheduler->delay(1);
231 
232  /* The sensitivity in LSb/g for an accel range i (as defined by the macros
233  * BMI160_ACC_RANGE_*G) can be calculated with:
234  * 2 ^ 16 / (2 * 2 ^ (i + 1)) = 2 ^(14 - i)
235  * That matches the typical values in the datasheet. */
236  _accel_scale = GRAVITY_MSS / (1 << (14 - BMI160_ACC_RANGE));
237 
238  return true;
239 }

241 bool AP_InertialSensor_BMI160::_configure_gyro()
242 {
243  bool r;
244 
245  r = _dev->write_register(BMI160_REG_GYR_CONF, BMI160_OSR | BMI160_ODR);
246  if (!r) {
247  return false;
248  }
249  hal.scheduler->delay(1);
250 
251  _check_err_reg();
252 
253  r = _dev->write_register(BMI160_REG_GYR_RANGE, BMI160_GYR_RANGE);
254  if (!r) {
255  return false;
256  }
257  hal.scheduler->delay(1);
258 
259  /* The sensitivity in LSb/degrees/s a gyro range i can be calculated with:
260  * 2 ^ 16 / (2 * 2000 / 2 ^ i) = 2 ^ (14 + i) / 1000
261  * The scale is the inverse of that. */
262  _gyro_scale = radians(1000.f / (1 << (14 + BMI160_GYR_RANGE)));
263 
264  return true;
265 }

303 bool AP_InertialSensor_BMI160::_configure_fifo()
304 {
305  bool r;
306 
307  /* The unit for the FIFO watermark is 4 bytes. */
308  r = _dev->write_register(BMI160_REG_FIFO_CONFIG_0,
309  sizeof(struct RawData) / 4);
310  if (!r) {
311  hal.console->printf("BMI160: Unable to configure FIFO watermark level\n");
312  return false;
313  }
314  hal.scheduler->delay(1);
315 
316  r = _dev->write_register(BMI160_REG_FIFO_CONFIG_1,
317  BMI160_FIFO_ACC_EN | BMI160_FIFO_GYR_EN);
318  if (!r) {
319  hal.console->printf("BMI160: Unable to enable FIFO\n");
320  return false;
321  }
322  hal.scheduler->delay(1);
323 
324  _check_err_reg();
325 
326  r = _dev->write_register(BMI160_REG_CMD, BMI160_CMD_FIFO_FLUSH);
327  if (!r) {
328  hal.console->printf("BMI160: Unable to flush FIFO\n");
329  return false;
330  }
331 
332  return true;

335 void AP_InertialSensor_BMI160::_read_fifo()
336 {
337  struct RawData raw_data[BMI160_MAX_FIFO_SAMPLES];
338  uint16_t num_bytes;
339  uint16_t excess;
340  uint8_t num_samples = 0;
341  bool r = true;
342 
343  static_assert(sizeof(raw_data) <= 100, "Too big to keep on stack");
344 
345  /* If FIFO watermark not surpassed. */
346  if (_int1_pin && _int1_pin->read() == 0) {
347  goto read_fifo_end;
348  }
349 
350  r = _dev->read_registers(BMI160_REG_FIFO_LENGTH,
351  (uint8_t *)&num_bytes,
352  sizeof(num_bytes));
353  if (!r) {
354  goto read_fifo_end;
355  }
356 
357  num_bytes = le16toh(num_bytes);
358  if (!num_bytes) {
359  goto read_fifo_end;
360  }
361 
362 read_fifo_read_data:
363  if (num_bytes > sizeof(raw_data)) {
364  excess = num_bytes - sizeof(raw_data);
365  num_bytes = sizeof(raw_data);
366  } else {
367  excess = 0;
368  }
369 
370  r = _dev->read_registers(BMI160_REG_FIFO_DATA,
371  (uint8_t *)raw_data,
372  num_bytes);
373  if (!r) {
374  goto read_fifo_end;
375  }
376 
377  /* Read again just once */
378  if (excess && num_samples) {
379  hal.console->printf("BMI160: dropping %u samples from fifo\n",
380  (uint8_t)(excess / sizeof(struct RawData)));
381  _dev->write_register(BMI160_REG_CMD, BMI160_CMD_FIFO_FLUSH);
382  excess = 0;
383  }
384 
385  num_samples = num_bytes / sizeof(struct RawData);
386  for (uint8_t i = 0; i < num_samples; i++) {
387  Vector3f accel{(float)(int16_t)le16toh(raw_data[i].accel.x),
388  (float)(int16_t)le16toh(raw_data[i].accel.y),
389  (float)(int16_t)le16toh(raw_data[i].accel.z)};
390  Vector3f gyro{(float)(int16_t)le16toh(raw_data[i].gyro.x),
391  (float)(int16_t)le16toh(raw_data[i].gyro.y),
392  (float)(int16_t)le16toh(raw_data[i].gyro.z)};
393 
394 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_AERO
395  accel.rotate(ROTATION_ROLL_180);
396  gyro.rotate(ROTATION_ROLL_180);
397 #endif
398 
399  accel *= _accel_scale;
400  gyro *= _gyro_scale;
401 
402  _rotate_and_correct_accel(_accel_instance, accel);
403  _rotate_and_correct_gyro(_gyro_instance, gyro);
404 
405  _notify_new_accel_raw_sample(_accel_instance, accel);
406  _notify_new_gyro_raw_sample(_gyro_instance, gyro);
407  }
408 
409  if (excess) {
410  num_bytes = excess;
411  goto read_fifo_read_data;
412  }
413 
414 read_fifo_end:
415  if (!r) {
416  hal.console->printf("BMI160: error on reading FIFO\n");
417  }
418 }

425 bool AP_InertialSensor_BMI160::_hardware_init()
426 {
427  bool ret = false;
428 
429  hal.scheduler->delay(BMI160_POWERUP_DELAY_MSEC);
430 
431  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
432  return false;
433  }
434 
435  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
436 
437  for (unsigned i = 0; i < BMI160_HARDWARE_INIT_MAX_TRIES; i++) {
438  uint8_t v;
439  ret = _dev->write_register(BMI160_REG_CMD,
440  BMI160_CMD_SOFTRESET);
441 
442  if (!ret) {
443  continue;
444  }
445  hal.scheduler->delay(BMI160_SOFTRESET_DELAY_MSEC);
446 
447  /* The datasheet recommends doing a read operation on the register 0x7F
448  * in order to guarantee the sensor works using the SPI protocol. This
449  * shouldn't have side effects for I2C. */
450  ret = _dev->read_registers(0x7F, &v, 1);
451  if (!ret) {
452  continue;
453  }
454 
455  ret = _dev->read_registers(BMI160_REG_CHIPID, &v, 1);
456  if (!ret) {
457  continue;
458  }
459  if (v != BMI160_CHIPID) {
460  ret = false;
461  continue;
462  }
463 
464  ret = _dev->write_register(BMI160_REG_CMD,
465  BMI160_CMD_ACCEL_NORMAL_POWER_MODE);
466  if (!ret) {
467  continue;
468  }
469  hal.scheduler->delay(BMI160_ACCEL_NORMAL_POWER_MODE_DELAY_MSEC);
470 
471  ret = _dev->write_register(BMI160_REG_CMD,
472  BMI160_CMD_GYRO_NORMAL_POWER_MODE);
473  if (!ret) {
474  continue;
475  }
476  hal.scheduler->delay(BMI160_GYRO_NORMAL_POWER_MODE_DELAY_MSEC);
477 
478  break;
479  }
480 
481  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
482 
483  _dev->get_semaphore()->give();
484  return ret;
485 }
