128 // calculate_scalars
129 void AP_MotorsHeli_Quad::calculate_scalars()
130 {
131  // range check collective min, max and mid
132  if( _collective_min >= _collective_max ) {
133  _collective_min = AP_MOTORS_HELI_COLLECTIVE_MIN;
134  _collective_max = AP_MOTORS_HELI_COLLECTIVE_MAX;
135  }
136 
137  _collective_mid = constrain_int16(_collective_mid, _collective_min, _collective_max);
138 
139  // calculate collective mid point as a number from 0 to 1000
140  _collective_mid_pct = ((float)(_collective_mid-_collective_min))/((float)(_collective_max-_collective_min));
141 
142  // calculate factors based on swash type and servo position
143  calculate_roll_pitch_collective_factors();
144 
145  // set mode of main rotor controller and trigger recalculation of scalars
146  _rotor.set_control_mode(static_cast<RotorControlMode>(_rsc_mode.get()));
147  enable_rsc_parameters();
148  calculate_armed_scalars();
149 }

151 // calculate_swash_factors - calculate factors based on swash type and servo position
152 void AP_MotorsHeli_Quad::calculate_roll_pitch_collective_factors()
153 {
154  // assume X quad layout, with motors at 45, 135, 225 and 315 degrees
155  // order FrontRight, RearLeft, FrontLeft, RearLeft
156  const float angles[AP_MOTORS_HELI_QUAD_NUM_MOTORS] = { 45, 225, 315, 135 };
157  const bool x_clockwise[AP_MOTORS_HELI_QUAD_NUM_MOTORS] = { false, false, true, true };
158  const float cos45 = cosf(radians(45));
159 
160  for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
161  bool clockwise = x_clockwise[i];
162  if (_frame_type == MOTOR_FRAME_TYPE_H) {
163  // reverse yaw for H frame
164  clockwise = !clockwise;
165  }
166  _rollFactor[CH_1+i] = -0.5*sinf(radians(angles[i]))/cos45;
167  _pitchFactor[CH_1+i] = 0.5*cosf(radians(angles[i]))/cos45;
168  _yawFactor[CH_1+i] = clockwise?-0.5:0.5;
169  _collectiveFactor[CH_1+i] = 1;
170  }
171 }

203 //
204 // move_actuators - moves swash plate to attitude of parameters passed in
205 // - expected ranges:
206 // roll : -1 ~ +1
207 // pitch: -1 ~ +1
208 // collective: 0 ~ 1
209 // yaw: -1 ~ +1
210 //
211 void AP_MotorsHeli_Quad::move_actuators(float roll_out, float pitch_out, float collective_in, float yaw_out)
212 {
213  // initialize limits flag
214  limit.roll_pitch = false;
215  limit.yaw = false;
216  limit.throttle_lower = false;
217  limit.throttle_upper = false;
218 
219  // constrain collective input
220  float collective_out = collective_in;
221  if (collective_out <= 0.0f) {
222  collective_out = 0.0f;
223  limit.throttle_lower = true;
224  }
225  if (collective_out >= 1.0f) {
226  collective_out = 1.0f;
227  limit.throttle_upper = true;
228  }
229 
230  // ensure not below landed/landing collective
231  if (_heliflags.landing_collective && collective_out < _collective_mid_pct) {
232  collective_out = _collective_mid_pct;
233  limit.throttle_lower = true;
234  }
235 
236  float collective_range = (_collective_max - _collective_min)*0.001f;
237 
238  if (_heliflags.inverted_flight) {
239  collective_out = 1 - collective_out;
240  }
241 
242  // feed power estimate into main rotor controller
243  _rotor.set_collective(fabsf(collective_out));
244 
245  // scale collective to -1 to 1
246  collective_out = collective_out*2-1;
247 
248  // reserve some collective for attitude control
249  collective_out *= collective_range;
250 
251  for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
252  _out[i] =
253  _rollFactor[CH_1+i] * roll_out +
254  _pitchFactor[CH_1+i] * pitch_out +
255  _collectiveFactor[CH_1+i] * collective_out;
256  }
257 
258  // see if we need to scale down yaw_out
259  for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
260  float y = _yawFactor[CH_1+i] * yaw_out;
261  if (_out[i] < 0) {
262  // the slope of the yaw effect changes at zero collective
263  y = -y;
264  }
265  if (_out[i] * (_out[i] + y) < 0) {
266  // applying this yaw demand would change the sign of the
267  // collective, which means the yaw would not be applied
268  // evenly. We scale down the overall yaw demand to prevent
269  // it crossing over zero
270  float s = -(_out[i] / y);
271  yaw_out *= s;
272  }
273  }
274 
275  // now apply the yaw correction
276  for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
277  float y = _yawFactor[CH_1+i] * yaw_out;
278  if (_out[i] < 0) {
279  // the slope of the yaw effect changes at zero collective
280  y = -y;
281  }
282  _out[i] += y;
283  }
284 
285 }

287 void AP_MotorsHeli_Quad::output_to_motors()
288 {
289  if (!_flags.initialised_ok) {
290  return;
291  }
292 
293  // move the servos
294  for (uint8_t i=0; i<AP_MOTORS_HELI_QUAD_NUM_MOTORS; i++) {
295  rc_write_angle(AP_MOTORS_MOT_1+i, _out[i] * QUAD_SERVO_MAX_ANGLE);
296  }
297 
298  switch (_spool_state) {
299  case SpoolState::SHUT_DOWN:
300  // sends minimum values out to the motors
301  update_motor_control(ROTOR_CONTROL_STOP);
302  break;
303  case SpoolState::GROUND_IDLE:
304  // sends idle output to motors when armed. rotor could be static or turning (autorotation)
305  update_motor_control(ROTOR_CONTROL_IDLE);
306  break;
307  case SpoolState::SPOOLING_UP:
308  case SpoolState::THROTTLE_UNLIMITED:
309  // set motor output based on thrust requests
310  update_motor_control(ROTOR_CONTROL_ACTIVE);
311  break;
312  case SpoolState::SPOOLING_DOWN:
313  // sends idle output to motors and wait for rotor to stop
314  update_motor_control(ROTOR_CONTROL_IDLE);
315  break;
316  }
317 }
