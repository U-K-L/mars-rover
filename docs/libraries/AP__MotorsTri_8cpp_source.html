29 void AP_MotorsTri::init(motor_frame_class frame_class, motor_frame_type frame_type)
30 {
31  add_motor_num(AP_MOTORS_MOT_1);
32  add_motor_num(AP_MOTORS_MOT_2);
33  add_motor_num(AP_MOTORS_MOT_4);
34 
35  // set update rate for the 3 motors (but not the servo on channel 7)
36  set_update_rate(_speed_hz);
37 
38  // set the motor_enabled flag so that the ESCs can be calibrated like other frame types
39  motor_enabled[AP_MOTORS_MOT_1] = true;
40  motor_enabled[AP_MOTORS_MOT_2] = true;
41  motor_enabled[AP_MOTORS_MOT_4] = true;
42 
43  // find the yaw servo
44  _yaw_servo = SRV_Channels::get_channel_for(SRV_Channel::k_motor7, AP_MOTORS_CH_TRI_YAW);
45  if (!_yaw_servo) {
46  gcs().send_text(MAV_SEVERITY_ERROR, "MotorsTri: unable to setup yaw channel");
47  // don't set initialised_ok
48  return;
49  }
50 
51  // allow mapping of motor7
52  add_motor_num(AP_MOTORS_CH_TRI_YAW);
53 
54  // record successful initialisation if what we setup was the desired frame_class
55  _flags.initialised_ok = (frame_class == MOTOR_FRAME_TRI);
56 }

78 void AP_MotorsTri::output_to_motors()
79 {
80  switch (_spool_state) {
81  case SpoolState::SHUT_DOWN:
82  // sends minimum values out to the motors
83  rc_write(AP_MOTORS_MOT_1, output_to_pwm(0));
84  rc_write(AP_MOTORS_MOT_2, output_to_pwm(0));
85  rc_write(AP_MOTORS_MOT_4, output_to_pwm(0));
86  rc_write(AP_MOTORS_CH_TRI_YAW, _yaw_servo->get_trim());
87  break;
88  case SpoolState::GROUND_IDLE:
89  // sends output to motors when armed but not flying
90  set_actuator_with_slew(_actuator[1], actuator_spin_up_to_ground_idle());
91  set_actuator_with_slew(_actuator[2], actuator_spin_up_to_ground_idle());
92  set_actuator_with_slew(_actuator[4], actuator_spin_up_to_ground_idle());
93  rc_write(AP_MOTORS_MOT_1, output_to_pwm(_actuator[1]));
94  rc_write(AP_MOTORS_MOT_2, output_to_pwm(_actuator[2]));
95  rc_write(AP_MOTORS_MOT_4, output_to_pwm(_actuator[4]));
96  rc_write(AP_MOTORS_CH_TRI_YAW, _yaw_servo->get_trim());
97  break;
98  case SpoolState::SPOOLING_UP:
99  case SpoolState::THROTTLE_UNLIMITED:
100  case SpoolState::SPOOLING_DOWN:
101  // set motor output based on thrust requests
102  set_actuator_with_slew(_actuator[1], thrust_to_actuator(_thrust_right));
103  set_actuator_with_slew(_actuator[2], thrust_to_actuator(_thrust_left));
104  set_actuator_with_slew(_actuator[4], thrust_to_actuator(_thrust_rear));
105  rc_write(AP_MOTORS_MOT_1, output_to_pwm(_actuator[1]));
106  rc_write(AP_MOTORS_MOT_2, output_to_pwm(_actuator[2]));
107  rc_write(AP_MOTORS_MOT_4, output_to_pwm(_actuator[4]));
108  rc_write(AP_MOTORS_CH_TRI_YAW, calc_yaw_radio_output(_pivot_angle, radians(_yaw_servo_angle_max_deg)));
109  break;
110  }
111 }

130 // output_armed - sends commands to the motors
131 // includes new scaling stability patch
132 void AP_MotorsTri::output_armed_stabilizing()
133 {
134  float roll_thrust; // roll thrust input value, +/- 1.0
135  float pitch_thrust; // pitch thrust input value, +/- 1.0
136  float yaw_thrust; // yaw thrust input value, +/- 1.0
137  float throttle_thrust; // throttle thrust input value, 0.0 - 1.0
138  float throttle_avg_max; // throttle thrust average maximum value, 0.0 - 1.0
139  float throttle_thrust_best_rpy; // throttle providing maximum roll, pitch and yaw range without climbing
140  float rpy_scale = 1.0f; // this is used to scale the roll, pitch and yaw to fit within the motor limits
141  float rpy_low = 0.0f; // lowest motor value
142  float rpy_high = 0.0f; // highest motor value
143  float thr_adj; // the difference between the pilot's desired throttle and throttle_thrust_best_rpy
144 
145  // sanity check YAW_SV_ANGLE parameter value to avoid divide by zero
146  _yaw_servo_angle_max_deg = constrain_float(_yaw_servo_angle_max_deg, AP_MOTORS_TRI_SERVO_RANGE_DEG_MIN, AP_MOTORS_TRI_SERVO_RANGE_DEG_MAX);
147 
148  // apply voltage and air pressure compensation
149  const float compensation_gain = get_compensation_gain();
150  roll_thrust = _roll_in * compensation_gain;
151  pitch_thrust = _pitch_in * compensation_gain;
152  yaw_thrust = _yaw_in * compensation_gain * sinf(radians(_yaw_servo_angle_max_deg)); // we scale this so a thrust request of 1.0f will ask for full servo deflection at full rear throttle
153  throttle_thrust = get_throttle() * compensation_gain;
154  throttle_avg_max = _throttle_avg_max * compensation_gain;
155 
156  // calculate angle of yaw pivot
157  _pivot_angle = safe_asin(yaw_thrust);
158  if (fabsf(_pivot_angle) > radians(_yaw_servo_angle_max_deg)) {
159  limit.yaw = true;
160  _pivot_angle = constrain_float(_pivot_angle, -radians(_yaw_servo_angle_max_deg), radians(_yaw_servo_angle_max_deg));
161  }
162 
163  float pivot_thrust_max = cosf(_pivot_angle);
164  float thrust_max = 1.0f;
165 
166  // sanity check throttle is above zero and below current limited throttle
167  if (throttle_thrust <= 0.0f) {
168  throttle_thrust = 0.0f;
169  limit.throttle_lower = true;
170  }
171  if (throttle_thrust >= _throttle_thrust_max) {
172  throttle_thrust = _throttle_thrust_max;
173  limit.throttle_upper = true;
174  }
175 
176  throttle_avg_max = constrain_float(throttle_avg_max, throttle_thrust, _throttle_thrust_max);
177 
178  // The following mix may be offer less coupling between axis but needs testing
179  //_thrust_right = roll_thrust * -0.5f + pitch_thrust * 1.0f;
180  //_thrust_left = roll_thrust * 0.5f + pitch_thrust * 1.0f;
181  //_thrust_rear = 0;
182 
183  _thrust_right = roll_thrust * -0.5f + pitch_thrust * 0.5f;
184  _thrust_left = roll_thrust * 0.5f + pitch_thrust * 0.5f;
185  _thrust_rear = pitch_thrust * -0.5f;
186 
187  // calculate roll and pitch for each motor
188  // set rpy_low and rpy_high to the lowest and highest values of the motors
189 
190  // record lowest roll pitch command
191  rpy_low = MIN(_thrust_right,_thrust_left);
192  rpy_high = MAX(_thrust_right,_thrust_left);
193  if (rpy_low > _thrust_rear){
194  rpy_low = _thrust_rear;
195  }
196  // check to see if the rear motor will reach maximum thrust before the front two motors
197  if ((1.0f - rpy_high) > (pivot_thrust_max - _thrust_rear)){
198  thrust_max = pivot_thrust_max;
199  rpy_high = _thrust_rear;
200  }
201 
202  // calculate throttle that gives most possible room for yaw (range 1000 ~ 2000) which is the lower of:
203  // 1. 0.5f - (rpy_low+rpy_high)/2.0 - this would give the maximum possible room margin above the highest motor and below the lowest
204  // 2. the higher of:
205  // a) the pilot's throttle input
206  // b) the point _throttle_rpy_mix between the pilot's input throttle and hover-throttle
207  // Situation #2 ensure we never increase the throttle above hover throttle unless the pilot has commanded this.
208  // Situation #2b allows us to raise the throttle above what the pilot commanded but not so far that it would actually cause the copter to rise.
209  // We will choose #1 (the best throttle for yaw control) if that means reducing throttle to the motors (i.e. we favor reducing throttle *because* it provides better yaw control)
210  // We will choose #2 (a mix of pilot and hover throttle) only when the throttle is quite low. We favor reducing throttle instead of better yaw control because the pilot has commanded it
211 
212  // check everything fits
213  throttle_thrust_best_rpy = MIN(0.5f*thrust_max - (rpy_low+rpy_high)/2.0, throttle_avg_max);
214  if(is_zero(rpy_low)){
215  rpy_scale = 1.0f;
216  } else {
217  rpy_scale = constrain_float(-throttle_thrust_best_rpy/rpy_low, 0.0f, 1.0f);
218  }
219 
220  // calculate how close the motors can come to the desired throttle
221  thr_adj = throttle_thrust - throttle_thrust_best_rpy;
222  if(rpy_scale < 1.0f){
223  // Full range is being used by roll, pitch, and yaw.
224  limit.roll_pitch = true;
225  if (thr_adj > 0.0f){
226  limit.throttle_upper = true;
227  }
228  thr_adj = 0.0f;
229  }else{
230  if(thr_adj < -(throttle_thrust_best_rpy+rpy_low)){
231  // Throttle can't be reduced to desired value
232  thr_adj = -(throttle_thrust_best_rpy+rpy_low);
233  }else if(thr_adj > thrust_max - (throttle_thrust_best_rpy+rpy_high)){
234  // Throttle can't be increased to desired value
235  thr_adj = thrust_max - (throttle_thrust_best_rpy+rpy_high);
236  limit.throttle_upper = true;
237  }
238  }
239 
240  // add scaled roll, pitch, constrained yaw and throttle for each motor
241  _thrust_right = throttle_thrust_best_rpy+thr_adj + rpy_scale*_thrust_right;
242  _thrust_left = throttle_thrust_best_rpy+thr_adj + rpy_scale*_thrust_left;
243  _thrust_rear = throttle_thrust_best_rpy+thr_adj + rpy_scale*_thrust_rear;
244 
245  // scale pivot thrust to account for pivot angle
246  // we should not need to check for divide by zero as _pivot_angle is constrained to the 5deg ~ 80 deg range
247  _thrust_rear = _thrust_rear/cosf(_pivot_angle);
248 
249  // constrain all outputs to 0.0f to 1.0f
250  // test code should be run with these lines commented out as they should not do anything
251  _thrust_right = constrain_float(_thrust_right, 0.0f, 1.0f);
252  _thrust_left = constrain_float(_thrust_left, 0.0f, 1.0f);
253  _thrust_rear = constrain_float(_thrust_rear, 0.0f, 1.0f);
254 }

256 // output_test_seq - spin a motor at the pwm value specified
257 // motor_seq is the motor's sequence number from 1 to the number of motors on the frame
258 // pwm value is an actual pwm value that will be output, normally in the range of 1000 ~ 2000
259 void AP_MotorsTri::output_test_seq(uint8_t motor_seq, int16_t pwm)
260 {
261  // exit immediately if not armed
262  if (!armed()) {
263  return;
264  }
265 
266  // output to motors and servos
267  switch (motor_seq) {
268  case 1:
269  // front right motor
270  rc_write(AP_MOTORS_MOT_1, pwm);
271  break;
272  case 2:
273  // back motor
274  rc_write(AP_MOTORS_MOT_4, pwm);
275  break;
276  case 3:
277  // back servo
278  rc_write(AP_MOTORS_CH_TRI_YAW, pwm);
279  break;
280  case 4:
281  // front left motor
282  rc_write(AP_MOTORS_MOT_2, pwm);
283  break;
284  default:
285  // do nothing
286  break;
287  }

290 // calc_yaw_radio_output - calculate final radio output for yaw channel
291 int16_t AP_MotorsTri::calc_yaw_radio_output(float yaw_input, float yaw_input_max)
292 {
293  int16_t ret;
294 
295  if (_yaw_servo->get_reversed()) {
296  yaw_input = -yaw_input;
297  }
298 
299  if (yaw_input >= 0){
300  ret = (_yaw_servo->get_trim() + (yaw_input/yaw_input_max * (_yaw_servo->get_output_max() - _yaw_servo->get_trim())));
301  } else {
302  ret = (_yaw_servo->get_trim() + (yaw_input/yaw_input_max * (_yaw_servo->get_trim() - _yaw_servo->get_output_min())));
303  }
304 
305  return ret;
306 }

313 void AP_MotorsTri::thrust_compensation(void)
314 {

317  float thrust[4] { _thrust_right, _thrust_left, 0, _thrust_rear };
318 
319  // apply vehicle supplied compensation function

321 
322  // extract compensated thrust values
323  _thrust_right = thrust[0];
324  _thrust_left = thrust[1];
325  _thrust_rear = thrust[3];
326  }
327 }
