42 /*
43  find a compressed file and uncompress it. Space for decompressed
44  data comes from malloc. Caller must be careful to free the resulting
45  data after use. The next byte after the file data is guaranteed to
46  be null
47 */
48 uint8_t *AP_ROMFS::find_decompress(const char *name, uint32_t &size)
49 {
50  uint32_t compressed_size;
51  const uint8_t *compressed_data = find_file(name, compressed_size);
52  if (!compressed_data) {
53  return nullptr;
54  }
55 
56  // last 4 bytes of gzip file are length of decompressed data
57  const uint8_t *p = &compressed_data[compressed_size-4];
58  uint32_t decompressed_size = p[0] | p[1] << 8 | p[2] << 16 | p[3] << 24;
59 
60  uint8_t *decompressed_data = (uint8_t *)malloc(decompressed_size + 1);
61  if (!decompressed_data) {
62  return nullptr;
63  }
64 
65  // explicitly null terimnate the data
66  decompressed_data[decompressed_size] = 0;
67 
68  TINF_DATA *d = (TINF_DATA *)malloc(sizeof(TINF_DATA));
69  if (!d) {
70  free(decompressed_data);
71  return nullptr;
72  }
73  uzlib_uncompress_init(d, NULL, 0);
74 
75  d->source = compressed_data;
76  d->source_limit = compressed_data + compressed_size - 4;
77 
78  // assume gzip format
79  int res = uzlib_gzip_parse_header(d);
80  if (res != TINF_OK) {
81  free(decompressed_data);
82  free(d);
83  return nullptr;
84  }
85 
86  d->dest = decompressed_data;
87  d->destSize = decompressed_size;
88 
89  // we don't check CRC, as it just wastes flash space for constant
90  // ROMFS data
91  res = uzlib_uncompress(d);
92 
93  free(d);
94 
95  if (res != TINF_OK) {
96  free(decompressed_data);
97  return nullptr;
98  }
99 
100  size = decompressed_size;
101  return decompressed_data;
102 }
