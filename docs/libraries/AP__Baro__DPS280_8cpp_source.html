83 /*
84  read calibration data
85  */
86 bool AP_Baro_DPS280::read_calibration(void)
87 {
88  uint8_t buf[18];
89 
90  if (!dev->read_registers(DPS280_REG_COEF, buf, 18)) {
91  return false;
92  }
93 
94  calibration.C0 = (buf[0] << 4) + ((buf[1] >>4) & 0x0F);
95  calibration.C1 = (buf[2] + ((buf[1] & 0x0F)<<8));
96  calibration.C00 = ((buf[4]<<4) + (buf[3]<<12)) + ((buf[5]>>4) & 0x0F);
97  calibration.C10 = ((buf[5] & 0x0F)<<16) + buf[7] + (buf[6]<<8);
98  calibration.C01 = (buf[9] + (buf[8]<<8));
99  calibration.C11 = (buf[11] + (buf[10]<<8));
100  calibration.C20 = (buf[13] + (buf[12]<<8));
101  calibration.C21 = (buf[15] + (buf[14]<<8));
102  calibration.C30 = (buf[17] + (buf[16]<<8));
103 
104  fix_config_bits16(calibration.C0, 12);
105  fix_config_bits16(calibration.C1, 12);
106  fix_config_bits32(calibration.C00, 20);
107  fix_config_bits32(calibration.C10, 20);
108  fix_config_bits16(calibration.C01, 16);
109  fix_config_bits16(calibration.C11, 16);
110  fix_config_bits16(calibration.C20, 16);
111  fix_config_bits16(calibration.C21, 16);
112  fix_config_bits16(calibration.C30, 16);
113 
114  /* get calibration source */
115  if (!dev->read_registers(DPS280_REG_CSRC, &calibration.temp_source, 1)) {
116  return false;
117  }
118  calibration.temp_source &= 0x80;
119 
120  return true;
121 }

123 bool AP_Baro_DPS280::init()
124 {
125  if (!dev || !dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
126  return false;
127  }
128 
129  dev->set_read_flag(0x80);
130  dev->set_speed(AP_HAL::Device::SPEED_HIGH);
131 
132  uint8_t whoami=0;
133  if (!dev->read_registers(DPS280_REG_PID, &whoami, 1) ||
134  whoami != DPS280_WHOAMI) {
135  // not a DPS280
136  printf("DPS280 whoami=0x%x\n", whoami);
137  dev->get_semaphore()->give();
138  return false;
139  }
140 
141  if (!read_calibration()) {
142  dev->get_semaphore()->give();
143  return false;
144  }
145 
146  dev->write_register(DPS280_REG_CREG, 0x0C); // shift for 16x oversampling
147  dev->write_register(DPS280_REG_PCONF, 0x54); // 32 Hz, 16x oversample
148  dev->write_register(DPS280_REG_TCONF, 0x54 | calibration.temp_source); // 32 Hz, 16x oversample
149  dev->write_register(DPS280_REG_MCONF, 0x07); // continuous temp and pressure
150 
151  instance = _frontend.register_sensor();
152 
153  dev->get_semaphore()->give();
154 
155  // request 64Hz update. New data will be available at 32Hz
156  dev->register_periodic_callback((1000 / 64) * AP_USEC_PER_MSEC, FUNCTOR_BIND_MEMBER(&AP_Baro_DPS280::timer, void));
157 
158  return true;
159 }

161 /*
162  calculate corrected pressure and temperature
163  */
164 void AP_Baro_DPS280::calculate_PT(int32_t UT, int32_t UP, float &pressure, float &temperature)
165 {
166  const struct dps280_cal &cal = calibration;
167  // scaling for 16x oversampling
168  const float scaling_16 = 1.0f/253952;
169 
170  float temp_scaled;
171  float press_scaled;
172 
173  temp_scaled = float(UT) * scaling_16;
174  temperature = cal.C0 * 0.5f + cal.C1 * temp_scaled;
175 
176  press_scaled = float(UP) * scaling_16;
177 
178  pressure = cal.C00;
179  pressure += press_scaled * (cal.C10 + press_scaled * (cal.C20 + press_scaled * cal.C30));
180  pressure += temp_scaled * cal.C01;
181  pressure += temp_scaled * press_scaled * (cal.C11 + press_scaled * cal.C21);
182 }

184 // acumulate a new sensor reading
185 void AP_Baro_DPS280::timer(void)
186 {
187  uint8_t buf[6];
188  uint8_t ready;
189 
190  if (!dev->read_registers(DPS280_REG_MCONF, &ready, 1) || !(ready & (1U<<4))) {
191  // pressure not ready
192  return;
193  }
194  if (!dev->read_registers(DPS280_REG_PRESS, buf, 3)) {
195  return;
196  }
197  if (!dev->read_registers(DPS280_REG_TEMP, &buf[3], 3)) {
198  return;
199  }
200 
201  int32_t press = (buf[2]) + (buf[1]<<8) + (buf[0]<<16);
202  int32_t temp = (buf[5]) + (buf[4]<<8) + (buf[3]<<16);
203  fix_config_bits32(press, 24);
204  fix_config_bits32(temp, 24);
205 
206  float pressure, temperature;
207 
208  calculate_PT(temp, press, pressure, temperature);
209 
210  if (!pressure_ok(pressure)) {
211  return;
212  }
213 
214  WITH_SEMAPHORE(_sem);
215 
216  pressure_sum += pressure;
217  temperature_sum += temperature;
218  count++;
219 }
