104 void SIM_Precland::update(const Location &loc, const Vector3f &position)
105 {
106  const uint32_t now = AP_HAL::millis();
107 
108  if (!_enable) {
109  _healthy = false;
110  return;
111  }
112  if (is_zero(_alt_limit) || _dist_limit < 1.0f) {
113  _healthy = false;
114  return;
115  }
116 
117  if (now - _last_update_ms < 1000.0f * (1.0f / _rate)) {
118  return;
119  }
120  _last_update_ms = now;
121 
122  const float distance_z = -position.z;
123  const float origin_height_m = _origin_height * 0.01f;
124  if (distance_z > _alt_limit + origin_height_m) {
125  _healthy = false;
126  return;
127  }
128 
129  const Location origin_center(static_cast<int32_t>(_origin_lat * 1.0e7f),
130  static_cast<int32_t>(_origin_lon * 1.0e7f),
131  static_cast<int32_t>(_origin_height),
132  Location::AltFrame::ABOVE_HOME);
133  Vector2f center;
134  if (!origin_center.get_vector_xy_from_origin_NE(center)) {
135  _healthy = false;
136  return;
137  }
138  center = center * 0.01f; // cm to m
139 
140  switch (_type) {
141  case PRECLAND_TYPE_CONE: {
142  const float in_radius = distance_z * _dist_limit / (_alt_limit + origin_height_m);
143  if (norm(position.x - center.x, position.y - center.y) > in_radius) {
144  _healthy = false;
145  return;
146  }
147  break;
148  }
149  case PRECLAND_TYPE_SPHERE: {
150  if (norm(position.x - center.x, position.y - center.y, distance_z - _origin_height) > (_alt_limit + origin_height_m)) {
151  _healthy = false;
152  return;
153  }
154  break;
155  }
156  default:
157  case PRECLAND_TYPE_CYLINDER: {
158  if (norm(position.x - center.x, position.y - center.y) > _dist_limit) {
159  _healthy = false;
160  return;
161  }
162  break;
163  }
164  }
165  _target_pos = position - Vector3f(center.x, center.y, origin_height_m);
166  _healthy = true;
167 }
