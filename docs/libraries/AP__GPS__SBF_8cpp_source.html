64 bool
65 AP_GPS_SBF::read(void)
66 {
67  bool ret = false;
68  uint32_t available_bytes = port->available();
69  for (uint32_t i = 0; i < available_bytes; i++) {
70  uint8_t temp = port->read();
71  ret |= parse(temp);
72  }
73 
74  if (gps._auto_config != AP_GPS::GPS_AUTO_CONFIG_DISABLE) {
75  if (_init_blob_index < ARRAY_SIZE(_initialisation_blob)) {
76  uint32_t now = AP_HAL::millis();
77  const char *init_str = _initialisation_blob[_init_blob_index];
78 
79  if (now > _init_blob_time) {
80  if (now > _config_last_ack_time + 2500) {
81  // try to enable input on the GPS port if we have not made progress on configuring it
82  Debug("SBF Sending port enable");
83  port->write((const uint8_t*)_port_enable, strlen(_port_enable));
84  _config_last_ack_time = now;
85  } else {
86  Debug("SBF sending init string: %s", init_str);
87  port->write((const uint8_t*)init_str, strlen(init_str));
88  }
89  _init_blob_time = now + 1000;
90  }
91  } else if (gps._raw_data == 2) { // only manage disarm/rearms when the user opts into it
92  if (hal.util->get_soft_armed()) {
93  _has_been_armed = true;
94  } else if (_has_been_armed && (RxState & SBF_DISK_MOUNTED)) {
95  // since init is done at this point and unmounting should be rate limited,
96  // take over the _init_blob_time variable
97  uint32_t now = AP_HAL::millis();
98  if (now > _init_blob_time) {
99  unmount_disk();
100  _init_blob_time = now + 1000;
101  }
102  }
103  }
104  }
105 
106  return ret;
107 }

109 bool
110 AP_GPS_SBF::parse(uint8_t temp)
111 {
112  switch (sbf_msg.sbf_state)
113  {
114  default:
115  case sbf_msg_parser_t::PREAMBLE1:
116  if (temp == SBF_PREAMBLE1) {
117  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE2;
118  sbf_msg.read = 0;
119  }
120  break;
121  case sbf_msg_parser_t::PREAMBLE2:
122  if (temp == SBF_PREAMBLE2) {
123  sbf_msg.sbf_state = sbf_msg_parser_t::CRC1;
124  } else if (temp == 'R') {
125  Debug("SBF got a response\n");
126  sbf_msg.sbf_state = sbf_msg_parser_t::COMMAND_LINE;
127  }
128  else
129  {
130  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
131  }
132  break;
133  case sbf_msg_parser_t::CRC1:
134  sbf_msg.crc = temp;
135  sbf_msg.sbf_state = sbf_msg_parser_t::CRC2;
136  break;
137  case sbf_msg_parser_t::CRC2:
138  sbf_msg.crc += (uint16_t)(temp << 8);
139  sbf_msg.sbf_state = sbf_msg_parser_t::BLOCKID1;
140  break;
141  case sbf_msg_parser_t::BLOCKID1:
142  sbf_msg.blockid = temp;
143  sbf_msg.sbf_state = sbf_msg_parser_t::BLOCKID2;
144  break;
145  case sbf_msg_parser_t::BLOCKID2:
146  sbf_msg.blockid += (uint16_t)(temp << 8);
147  sbf_msg.sbf_state = sbf_msg_parser_t::LENGTH1;
148  break;
149  case sbf_msg_parser_t::LENGTH1:
150  sbf_msg.length = temp;
151  sbf_msg.sbf_state = sbf_msg_parser_t::LENGTH2;
152  break;
153  case sbf_msg_parser_t::LENGTH2:
154  sbf_msg.length += (uint16_t)(temp << 8);
155  sbf_msg.sbf_state = sbf_msg_parser_t::DATA;
156  if (sbf_msg.length % 4 != 0) {
157  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
158  Debug("bad packet length=%u\n", (unsigned)sbf_msg.length);
159  }
160  if (sbf_msg.length < 8) {
161  Debug("bad packet length=%u\n", (unsigned)sbf_msg.length);
162  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
163  crc_error_counter++; // this is a probable buffer overflow, but this
164  // indicates not enough bytes to do a crc
165  break;
166  }
167  break;
168  case sbf_msg_parser_t::DATA:
169  if (sbf_msg.read < sizeof(sbf_msg.data)) {
170  sbf_msg.data.bytes[sbf_msg.read] = temp;
171  }
172  sbf_msg.read++;
173  if (sbf_msg.read >= (sbf_msg.length - 8)) {
174  if (sbf_msg.read > sizeof(sbf_msg.data)) {
175  // not interested in these large messages
176  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
177  break;
178  }
179  uint16_t crc = crc16_ccitt((uint8_t*)&sbf_msg.blockid, 2, 0);
180  crc = crc16_ccitt((uint8_t*)&sbf_msg.length, 2, crc);
181  crc = crc16_ccitt((uint8_t*)&sbf_msg.data, sbf_msg.length - 8, crc);
182 
183  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
184 
185  if (sbf_msg.crc == crc) {
186  return process_message();
187  } else {
188  Debug("crc fail\n");
189  crc_error_counter++;
190  }
191  }
192  break;
193  case sbf_msg_parser_t::COMMAND_LINE:
194  if (sbf_msg.read < (sizeof(sbf_msg.data) - 1)) {
195  sbf_msg.data.bytes[sbf_msg.read] = temp;
196  } else {
197  // we don't have enough buffer to compare the commands
198  // most probable cause is that a user injected a longer command then
199  // we have buffer for, or it could be a corruption, either way we
200  // simply ignore the result
201  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
202  break;
203  }
204  sbf_msg.read++;
205  if (temp == '\n') {
206  sbf_msg.data.bytes[sbf_msg.read] = 0;
207 
208  // received the result, lets assess it
209  if (sbf_msg.data.bytes[0] == ':') {
210  // valid command, determine if it was the one we were trying
211  // to send in the configuration sequence
212  if (_init_blob_index < ARRAY_SIZE(_initialisation_blob)) {
213  if (!strncmp(_initialisation_blob[_init_blob_index], (char *)(sbf_msg.data.bytes + 2),
214  sbf_msg.read - SBF_EXCESS_COMMAND_BYTES)) {
215  Debug("SBF Ack Command: %s\n", sbf_msg.data.bytes);
216  _init_blob_index++;
217  _config_last_ack_time = AP_HAL::millis();
218  } else {
219  Debug("SBF Ack command (unexpected): %s\n", sbf_msg.data.bytes);
220  }
221  }
222  } else {
223  // rejected command, send it out as a debug
224  Debug("SBF NACK Command: %s\n", sbf_msg.data.bytes);
225  }
226  // resume normal parsing
227  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
228  break;
229  }
230  break;
231  }
232 
233  return false;

236 bool
237 AP_GPS_SBF::process_message(void)
238 {
239  uint16_t blockid = (sbf_msg.blockid & 8191u);
240 
241  Debug("BlockID %d", blockid);
242 
243  switch (blockid) {
244  case PVTGeodetic:
245  {
246  const msg4007 &temp = sbf_msg.data.msg4007u;
247 
248  // Update time state
249  if (temp.WNc != 65535) {
250  state.time_week = temp.WNc;
251  state.time_week_ms = (uint32_t)(temp.TOW);
252  }
253 
254  check_new_itow(temp.TOW, sbf_msg.length);
255  state.last_gps_time_ms = AP_HAL::millis();
256 
257  // Update velocity state (don't use −2·10^10)
258  if (temp.Vn > -200000) {
259  state.velocity.x = (float)(temp.Vn);
260  state.velocity.y = (float)(temp.Ve);
261  state.velocity.z = (float)(-temp.Vu);
262 
263  state.have_vertical_velocity = true;
264 
265  float ground_vector_sq = state.velocity[0] * state.velocity[0] + state.velocity[1] * state.velocity[1];
266  state.ground_speed = (float)safe_sqrt(ground_vector_sq);
267 
268  state.ground_course = wrap_360(degrees(atan2f(state.velocity[1], state.velocity[0])));
269  state.rtk_age_ms = temp.MeanCorrAge * 10;
270 
271  // value is expressed as twice the rms error = int16 * 0.01/2
272  state.horizontal_accuracy = (float)temp.HAccuracy * 0.005f;
273  state.vertical_accuracy = (float)temp.VAccuracy * 0.005f;
274  state.have_horizontal_accuracy = true;
275  state.have_vertical_accuracy = true;
276  }
277 
278  // Update position state (don't use -2·10^10)
279  if (temp.Latitude > -200000) {
280  state.location.lat = (int32_t)(temp.Latitude * RAD_TO_DEG_DOUBLE * (double)1e7);
281  state.location.lng = (int32_t)(temp.Longitude * RAD_TO_DEG_DOUBLE * (double)1e7);
282  state.location.alt = (int32_t)(((float)temp.Height - temp.Undulation) * 1e2f);
283  }
284 
285  if (temp.NrSV != 255) {
286  state.num_sats = temp.NrSV;
287  }
288 
289  Debug("temp.Mode=0x%02x\n", (unsigned)temp.Mode);
290  switch (temp.Mode & 15) {
291  case 0: // no pvt
292  state.status = AP_GPS::NO_FIX;
293  break;
294  case 1: // standalone
295  state.status = AP_GPS::GPS_OK_FIX_3D;
296  break;
297  case 2: // dgps
298  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
299  break;
300  case 3: // fixed location
301  state.status = AP_GPS::GPS_OK_FIX_3D;
302  break;
303  case 4: // rtk fixed
304  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
305  break;
306  case 5: // rtk float
307  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
308  break;
309  case 6: // sbas
310  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
311  break;
312  case 7: // moving rtk fixed
313  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
314  break;
315  case 8: // moving rtk float
316  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
317  break;
318  }
319 
320  if ((temp.Mode & 64) > 0) { // gps is in base mode
321  state.status = AP_GPS::NO_FIX;
322  } else if ((temp.Mode & 128) > 0) { // gps only has 2d fix
323  state.status = AP_GPS::GPS_OK_FIX_2D;
324  }
325 
326  return true;
327  }
328  case DOP:
329  {
330  const msg4001 &temp = sbf_msg.data.msg4001u;
331  check_new_itow(temp.TOW, sbf_msg.length);
332 
333  state.hdop = temp.HDOP;
334  state.vdop = temp.VDOP;
335  break;
336  }
337  case ReceiverStatus:
338  {
339  const msg4014 &temp = sbf_msg.data.msg4014u;
340  check_new_itow(temp.TOW, sbf_msg.length);
341  RxState = temp.RxState;
342  if ((RxError & RX_ERROR_MASK) != (temp.RxError & RX_ERROR_MASK)) {
343  gcs().send_text(MAV_SEVERITY_INFO, "GPS %d: SBF error changed (0x%08x/0x%08x)", state.instance + 1,
344  RxError & RX_ERROR_MASK, temp.RxError & RX_ERROR_MASK);
345  }
346  RxError = temp.RxError;
347  break;
348  }
349  case VelCovGeodetic:
350  {
351  const msg5908 &temp = sbf_msg.data.msg5908u;
352 
353  check_new_itow(temp.TOW, sbf_msg.length);
354  // select the maximum variance, as the EKF will apply it to all the columns in it's estimate
355  // FIXME: Support returning the covariance matrix to the EKF
356  float max_variance_squared = MAX(temp.Cov_VnVn, MAX(temp.Cov_VeVe, temp.Cov_VuVu));
357  if (is_positive(max_variance_squared)) {
358  state.have_speed_accuracy = true;
359  state.speed_accuracy = sqrt(max_variance_squared);
360  } else {
361  state.have_speed_accuracy = false;
362  }
363  break;
364  }
365  }
366 
367  return false;
368 }

400 bool AP_GPS_SBF::prepare_for_arming(void) {
401  bool is_logging = true; // assume that its logging until proven otherwise
402  if (gps._raw_data) {
403  if (!(RxState & SBF_DISK_MOUNTED)){
404  is_logging = false;
405  gcs().send_text(MAV_SEVERITY_INFO, "GPS %d: SBF disk is not mounted", state.instance + 1);
406 
407  // simply attempt to mount the disk, no need to check if the command was
408  // ACK/NACK'd as we don't continuously attempt to remount the disk
409  gcs().send_text(MAV_SEVERITY_INFO, "GPS %d: Attempting to mount disk", state.instance + 1);
410  mount_disk();
411  // reset the flag to indicate if we should be logging
412  _has_been_armed = false;
413  }
414  else if (RxState & SBF_DISK_FULL) {
415  is_logging = false;
416  gcs().send_text(MAV_SEVERITY_INFO, "GPS %d: SBF disk is full", state.instance + 1);
417  }
418  else if (!(RxState & SBF_DISK_ACTIVITY)) {
419  is_logging = false;
420  gcs().send_text(MAV_SEVERITY_INFO, "GPS %d: SBF is not currently logging", state.instance + 1);
421  }
422  }
423 
424  return is_logging;
