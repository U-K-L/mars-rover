216  static bool point_on_segment(const Vector2<T>& point,
217  const Vector2<T>& seg_start,
218  const Vector2<T>& seg_end) {
219  const float expected_run = seg_end.x-seg_start.x;
220  const float intersection_run = point.x-seg_start.x;
221  // check slopes are identical:
222  if (fabsf(expected_run) < FLT_EPSILON) {
223  if (fabsf(intersection_run) > FLT_EPSILON) {
224  return false;
225  }
226  } else {
227  const float expected_slope = (seg_end.y-seg_start.y)/expected_run;
228  const float intersection_slope = (point.y-seg_start.y)/intersection_run;
229  if (fabsf(expected_slope - intersection_slope) > FLT_EPSILON) {
230  return false;
231  }
232  }
233  // check for presence in bounding box
234  if (seg_start.x < seg_end.x) {
235  if (point.x < seg_start.x || point.x > seg_end.x) {
236  return false;
237  }
238  } else {
239  if (point.x < seg_end.x || point.x > seg_start.x) {
240  return false;
241  }
242  }
243  if (seg_start.y < seg_end.y) {
244  if (point.y < seg_start.y || point.y > seg_end.y) {
245  return false;
246  }
247  } else {
248  if (point.y < seg_end.y || point.y > seg_start.y) {
249  return false;
250  }
251  }
252  return true;
253  }
