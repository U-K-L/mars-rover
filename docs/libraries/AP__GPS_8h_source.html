492  /*
493  buffer for re-assembling RTCM data for GPS injection.
494  The 8 bit flags field in GPS_RTCM_DATA is interpreted as:
495  1 bit for "is fragmented"
496  2 bits for fragment number
497  5 bits for sequence number
498 
499  The rtcm_buffer is allocated on first use. Once a block of data
500  is successfully reassembled it is injected into all active GPS
501  backends. This assumes we don't want more than 4*180=720 bytes
502  in a RTCM data block
503  */
504  struct rtcm_buffer {
505  uint8_t fragments_received;
506  uint8_t sequence;
507  uint8_t fragment_count;
508  uint16_t total_length;
509  uint8_t buffer[MAVLINK_MSG_GPS_RTCM_DATA_FIELD_DATA_LEN*4];
510  } *rtcm_buffer;
