121 void Scheduler::delay_microseconds(uint16_t usec)
122 {
123  if (usec == 0) { //chibios faults with 0us sleep
124  return;
125  }
126  uint32_t ticks;
127  ticks = chTimeUS2I(usec);
128  if (ticks == 0) {
129  // calling with ticks == 0 causes a hard fault on ChibiOS
130  ticks = 1;
131  }
132  chThdSleep(ticks); //Suspends Thread for desired microseconds
133 }

308 {
309  Scheduler *sched = (Scheduler *)arg;
310  chRegSetThreadName("apm_timer");
311 
312  while (!sched->_hal_initialized) {
313  sched->delay_microseconds(1000);
314  }

333 void Scheduler::_monitor_thread(void *arg)
334 {
335  Scheduler *sched = (Scheduler *)arg;
336  chRegSetThreadName("apm_monitor");
337 
338  while (!sched->_initialized) {
339  sched->delay(100);
340  }
341  bool using_watchdog = AP_BoardConfig::watchdog_enabled();
342 
343  while (true) {
344  sched->delay(100);
345  if (using_watchdog) {
346  stm32_watchdog_save((uint32_t *)&hal.util->persistent_data, (sizeof(hal.util->persistent_data)+3)/4);
347  }
348  uint32_t now = AP_HAL::millis();
349  uint32_t loop_delay = now - sched->last_watchdog_pat_ms;
350  if (loop_delay >= 200) {
351  // the main loop has been stuck for at least
352  // 200ms. Starting logging the main loop state
353  const AP_HAL::Util::PersistentData &pd = hal.util->persistent_data;
354  AP::logger().Write("MON", "TimeUS,LDelay,Task,IErr,IErrCnt,MavMsg,MavCmd,SemLine,SPICnt,I2CCnt", "QIbIIHHHII",
355  AP_HAL::micros64(),
356  loop_delay,
357  pd.scheduler_task,
358  pd.internal_errors,
359  pd.internal_error_count,
360  pd.last_mavlink_msgid,
361  pd.last_mavlink_cmd,
362  pd.semaphore_line,
363  pd.spi_count,
364  pd.i2c_count);
365  }

485 /*
486  create a new thread
487 */
488 bool Scheduler::thread_create(AP_HAL::MemberProc proc, const char *name, uint32_t stack_size, priority_base base, int8_t priority)
489 {
490  // take a copy of the MemberProc, it is freed after thread exits
491  AP_HAL::MemberProc *tproc = (AP_HAL::MemberProc *)malloc(sizeof(proc));
492  if (!tproc) {
493  return false;
494  }
495  *tproc = proc;
496 
497  uint8_t thread_priority = APM_IO_PRIORITY;
498  static const struct {
499  priority_base base;
500  uint8_t p;
501  } priority_map[] = {
502  { PRIORITY_BOOST, APM_MAIN_PRIORITY_BOOST},
503  { PRIORITY_MAIN, APM_MAIN_PRIORITY},
504  { PRIORITY_SPI, APM_SPI_PRIORITY},
505  { PRIORITY_I2C, APM_I2C_PRIORITY},
506  { PRIORITY_CAN, APM_CAN_PRIORITY},
507  { PRIORITY_TIMER, APM_TIMER_PRIORITY},
508  { PRIORITY_RCIN, APM_RCIN_PRIORITY},
509  { PRIORITY_IO, APM_IO_PRIORITY},
510  { PRIORITY_UART, APM_UART_PRIORITY},
511  { PRIORITY_STORAGE, APM_STORAGE_PRIORITY},
512  { PRIORITY_SCRIPTING, APM_SCRIPTING_PRIORITY},
513  };
514  for (uint8_t i=0; i<ARRAY_SIZE(priority_map); i++) {
515  if (priority_map[i].base == base) {
516  thread_priority = constrain_int16(priority_map[i].p + priority, LOWPRIO, HIGHPRIO);
517  break;
518  }
519  }
520  thread_t *thread_ctx = thread_create_alloc(THD_WORKING_AREA_SIZE(stack_size),
521  name,
522  thread_priority,
523  thread_create_trampoline,
524  tproc);
525  if (thread_ctx == nullptr) {
526  free(tproc);
527  return false;
528  }
529  return true;
530 }

532 /*
533  inform the scheduler that we are calling an operation from the
534  main thread that may take an extended amount of time. This can
535  be used to prevent watchdog reset during expected long delays
536  A value of zero cancels the previous expected delay
537 */
538 void Scheduler::expect_delay_ms(uint32_t ms)
539 {
540  if (!in_main_thread()) {
541  // only for main thread
542  return;
543  }
544  if (ms == 0) {
545  if (expect_delay_nesting > 0) {
546  expect_delay_nesting--;
547  }
548  if (expect_delay_nesting == 0) {
549  expect_delay_start = 0;
550  }
551  } else {
552  uint32_t now = AP_HAL::millis();
553  if (expect_delay_start != 0) {
554  // we already have a delay running, possibly extend it
555  uint32_t done = now - expect_delay_start;
556  if (expect_delay_length > done) {
557  ms = MAX(ms, expect_delay_length - done);
558  }
559  }
560  expect_delay_start = now;
561  expect_delay_length = ms;
562  expect_delay_nesting++;
563 
564  // also put our priority below timer thread if we are boosted
565  boost_end();
566  }
567 }
