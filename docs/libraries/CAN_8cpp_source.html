51 static can_frame makeSocketCanFrame(const uavcan::CanFrame& uavcan_frame)
52 {
53  can_frame sockcan_frame { uavcan_frame.id& uavcan::CanFrame::MaskExtID, uavcan_frame.dlc, { } };
54  std::copy(uavcan_frame.data, uavcan_frame.data + uavcan_frame.dlc, sockcan_frame.data);
55  if (uavcan_frame.isExtended()) {
56  sockcan_frame.can_id |= CAN_EFF_FLAG;
57  }
58  if (uavcan_frame.isErrorFrame()) {
59  sockcan_frame.can_id |= CAN_ERR_FLAG;
60  }
61  if (uavcan_frame.isRemoteTransmissionRequest()) {
62  sockcan_frame.can_id |= CAN_RTR_FLAG;
63  }
64  return sockcan_frame;
65 }

136 int CAN::openSocket(const std::string& iface_name)
137 {
138  errno = 0;
139 
140  int s = socket(PF_CAN, SOCK_RAW, CAN_RAW);
141  if (s < 0) {
142  return s;
143  }
144 
145  std::shared_ptr<void> defer(&s, [](int* fd) { if (*fd >= 0) close(*fd); });
146  const int ret = s;
147 
148  // Detect the iface index
149  auto ifr = ifreq();
150  if (iface_name.length() >= IFNAMSIZ) {
151  errno = ENAMETOOLONG;
152  return -1;
153  }
154  std::strncpy(ifr.ifr_name, iface_name.c_str(), iface_name.length());
155  if (ioctl(s, SIOCGIFINDEX, &ifr) < 0 || ifr.ifr_ifindex < 0) {
156  return -1;
157  }
158 
159  // Bind to the specified CAN iface
160  {
161  auto addr = sockaddr_can();
162  addr.can_family = AF_CAN;
163  addr.can_ifindex = ifr.ifr_ifindex;
164  if (bind(s, reinterpret_cast<sockaddr*>(&addr), sizeof(addr)) < 0) {
165  return -1;
166  }
167  }
168 
169  // Configure
170  {
171  const int on = 1;
172  // Timestamping
173  if (setsockopt(s, SOL_SOCKET, SO_TIMESTAMP, &on, sizeof(on)) < 0) {
174  return -1;
175  }
176  // Socket loopback
177  if (setsockopt(s, SOL_CAN_RAW, CAN_RAW_RECV_OWN_MSGS, &on, sizeof(on)) < 0) {
178  return -1;
179  }
180  // Non-blocking
181  if (fcntl(s, F_SETFL, O_NONBLOCK) < 0) {
182  return -1;
183  }
184  }
185 
186  // Validate the resulting socket
187  {
188  int socket_error = 0;
189  socklen_t errlen = sizeof(socket_error);
190  getsockopt(s, SOL_SOCKET, SO_ERROR, reinterpret_cast<void*>(&socket_error), &errlen);
191  if (socket_error != 0) {
192  errno = socket_error;
193  return -1;
194  }
195  }
196  s = -1;
197  return ret;
198 }

294 void CAN::_pollWrite()
295 {
296  while (hasReadyTx()) {
297  const TxItem tx = _tx_queue.top();
298 
299  if (tx.deadline >= getMonotonic()) {
300  const int res = _write(tx.frame);
301  if (res == 1) { // Transmitted successfully
302  _incrementNumFramesInSocketTxQueue();
303  if (tx.flags & uavcan::CanIOFlagLoopback) {
304  _pending_loopback_ids.insert(tx.frame.id);
305  }
306  } else if (res == 0) { // Not transmitted, nor is it an error
307  break; // Leaving the loop, the frame remains enqueued for the next retry
308  } else { // Transmission error
309  _registerError(SocketCanError::SocketWriteFailure);
310  }
311  } else {
312  _registerError(SocketCanError::TxTimeout);
313  }
314 
315  // Removing the frame from the queue even if transmission failed
316  _tx_queue.pop();
317  }
318 }

320 void CAN::_pollRead()
321 {
322  uint8_t iterations_count = 0;
323  while (iterations_count < CAN_MAX_POLL_ITERATIONS_COUNT)
324  {
325  iterations_count++;
326  RxItem rx;
327  rx.ts_mono = getMonotonic(); // Monotonic timestamp is not required to be precise (unlike UTC)
328  bool loopback = false;
329  const int res = _read(rx.frame, rx.ts_utc, loopback);
330  if (res == 1) {
331  bool accept = true;
332  if (loopback) { // We receive loopback for all CAN frames
333  _confirmSentFrame();
334  rx.flags |= uavcan::CanIOFlagLoopback;
335  accept = _wasInPendingLoopbackSet(rx.frame);
336  }
337  if (accept) {
338  _rx_queue.push(rx);
339  }
340  } else if (res == 0) {
341  break;
342  } else {
343  _registerError(SocketCanError::SocketReadFailure);
344  break;
345  }
346  }
347 }

350 {

354 
355  const int res = write(_fd, &sockcan_frame, sizeof(sockcan_frame));
356  if (res <= 0) {
357  if (errno == ENOBUFS || errno == EAGAIN) { // Writing is not possible atm, not an error
358  return 0;
359  }
360  return res;
361  }
362  if (res != sizeof(sockcan_frame)) {
363  return -1;
364  }
365  return 1;

369 int CAN::_read(uavcan::CanFrame& frame, uavcan::UtcTime& ts_utc, bool& loopback) const
370 {
371  auto iov = iovec();
372  auto sockcan_frame = can_frame();
373  iov.iov_base = &sockcan_frame;
374  iov.iov_len = sizeof(sockcan_frame);
375  union {
376  uint8_t data[CMSG_SPACE(sizeof(::timeval))];
377  struct cmsghdr align;
378  } control;
379 
380  auto msg = msghdr();
381  msg.msg_iov = &iov;
382  msg.msg_iovlen = 1;
383  msg.msg_control = control.data;
384  msg.msg_controllen = sizeof(control.data);
385 
386  const int res = recvmsg(_fd, &msg, MSG_DONTWAIT);
387  if (res <= 0) {
388  return (res < 0 && errno == EWOULDBLOCK) ? 0 : res;
389  }
390  /*
391  * Flags
392  */
393  loopback = (msg.msg_flags & static_cast<int>(MSG_CONFIRM)) != 0;
394 
395  if (!loopback && !_checkHWFilters(sockcan_frame)) {
396  return 0;
397  }
398 
399  frame = makeUavcanFrame(sockcan_frame);
400  /*
401  * Timestamp
402  */
403  const cmsghdr* const cmsg = CMSG_FIRSTHDR(&msg);
404  if (cmsg->cmsg_level == SOL_SOCKET && cmsg->cmsg_type == SO_TIMESTAMP) {
405  auto tv = timeval();
406  std::memcpy(&tv, CMSG_DATA(cmsg), sizeof(tv)); // Copy to avoid alignment problems
407  ts_utc = uavcan::UtcTime::fromUSec(std::uint64_t(tv.tv_sec) * 1000000ULL + tv.tv_usec);
408  } else {
409  return -1;
410  }
411  return 1;
412 }

499 int16_t CANManager::select(uavcan::CanSelectMasks& inout_masks,
500  const uavcan::CanFrame* (&)[uavcan::MaxCanIfaces],
501  uavcan::MonotonicTime blocking_deadline)
502 {
503  // Detecting whether we need to block at all
504  bool need_block = (inout_masks.write == 0); // Write queue is infinite
505 
506  for (unsigned i = 0; need_block && (i < _ifaces.size()); i++) {
507  const bool need_read = inout_masks.read & (1 << i);
508  if (need_read && _ifaces[i]->hasReadyRx()) {
509  need_block = false;
510  }
511  }
512 
513  if (need_block) {
514  // Poll FD set setup
515  pollfd pollfds[uavcan::MaxCanIfaces] = {};
516  unsigned num_pollfds = 0;
517  IfaceWrapper* pollfd_index_to_iface[uavcan::MaxCanIfaces] = {};
518 
519  for (unsigned i = 0; i < _ifaces.size(); i++) {
520  if (_ifaces[i]->isDown()) {
521  continue;
522  }
523  pollfds[num_pollfds].fd = _ifaces[i]->getFileDescriptor();
524  pollfds[num_pollfds].events = POLLIN;
525  if (_ifaces[i]->hasReadyTx() || (inout_masks.write & (1U << i))) {
526  pollfds[num_pollfds].events |= POLLOUT;
527  }
528  pollfd_index_to_iface[num_pollfds] = _ifaces[i].get();
529  num_pollfds++;
530  }
531 
532  if (num_pollfds == 0) {
533  return 0;
534  }
535 
536  // Timeout conversion
537  const std::int64_t timeout_usec = (blocking_deadline - getMonotonic()).toUSec();
538  auto ts = timespec();
539  if (timeout_usec > 0) {
540  ts.tv_sec = timeout_usec / 1000000LL;
541  ts.tv_nsec = (timeout_usec % 1000000LL) * 1000;
542  }
543 
544  // Blocking here
545  const int res = ppoll(pollfds, num_pollfds, &ts, nullptr);
546  if (res < 0) {
547  return res;
548  }
549 
550  // Handling poll output
551  for (unsigned i = 0; i < num_pollfds; i++) {
552  pollfd_index_to_iface[i]->updateDownStatusFromPollResult(pollfds[i]);
553 
554  const bool poll_read = pollfds[i].revents & POLLIN;
555  const bool poll_write = pollfds[i].revents & POLLOUT;
556  pollfd_index_to_iface[i]->poll(poll_read, poll_write);
557  }
558  }
559 
560  // Writing the output masks
561  inout_masks = uavcan::CanSelectMasks();
562  for (unsigned i = 0; i < _ifaces.size(); i++) {
563  if (!_ifaces[i]->isDown()) {
564  inout_masks.write |= std::uint8_t(1U << i); // Always ready to write if not down
565  }
566  if (_ifaces[i]->hasReadyRx()) {
567  inout_masks.read |= std::uint8_t(1U << i); // Readability depends only on RX buf, even if down
568  }
569  }
570 
571  // Return value is irrelevant as long as it's non-negative
572  return _ifaces.size();
573 }
