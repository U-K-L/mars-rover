59  generate an accelerometer sample
60  */
61 void AP_InertialSensor_SITL::generate_accel(uint8_t instance)
62 {
63  // minimum noise levels are 2 bits, but averaged over many
64  // samples, giving around 0.01 m/s/s
65  float accel_noise = 0.01f;
66 
67  if (sitl->motors_on) {
68  // add extra noise when the motors are on
69  accel_noise += instance==0?sitl->accel_noise:sitl->accel2_noise;
70  }
71 
72  // add accel bias and noise
73  Vector3f accel_bias = instance==0?sitl->accel_bias.get():sitl->accel2_bias.get();
74  float xAccel = sitl->state.xAccel + accel_bias.x;
75  float yAccel = sitl->state.yAccel + accel_bias.y;
76  float zAccel = sitl->state.zAccel + accel_bias.z;
77  const Vector3f &vibe_freq = sitl->vibe_freq;
78  if (vibe_freq.is_zero()) {
79  xAccel += accel_noise * rand_float();
80  yAccel += accel_noise * rand_float();
81  zAccel += accel_noise * rand_float();
82  } else {
83  float t = AP_HAL::micros() * 1.0e-6f;
84  xAccel += sinf(t * 2 * M_PI * vibe_freq.x) * accel_noise;
85  yAccel += sinf(t * 2 * M_PI * vibe_freq.y) * accel_noise;
86  zAccel += sinf(t * 2 * M_PI * vibe_freq.z) * accel_noise;
87  }
88 
89  // correct for the acceleration due to the IMU position offset and angular acceleration
90  // correct for the centripetal acceleration
91  // only apply corrections to first accelerometer
92  Vector3f pos_offset = sitl->imu_pos_offset;
93  if (!pos_offset.is_zero()) {
94  // calculate sensed acceleration due to lever arm effect
95  // Note: the % operator has been overloaded to provide a cross product
96  Vector3f angular_accel = Vector3f(radians(sitl->state.angAccel.x) , radians(sitl->state.angAccel.y) , radians(sitl->state.angAccel.z));
97  Vector3f lever_arm_accel = angular_accel % pos_offset;
98 
99  // calculate sensed acceleration due to centripetal acceleration
100  Vector3f angular_rate = Vector3f(radians(sitl->state.rollRate), radians(sitl->state.pitchRate), radians(sitl->state.yawRate));
101  Vector3f centripetal_accel = angular_rate % (angular_rate % pos_offset);
102 
103  // apply corrections
104  xAccel += lever_arm_accel.x + centripetal_accel.x;
105  yAccel += lever_arm_accel.y + centripetal_accel.y;
106  zAccel += lever_arm_accel.z + centripetal_accel.z;
107  }
108 
109  if (fabsf(sitl->accel_fail) > 1.0e-6f) {
110  xAccel = sitl->accel_fail;
111  yAccel = sitl->accel_fail;
112  zAccel = sitl->accel_fail;
113  }
114 
115  Vector3f accel = Vector3f(xAccel, yAccel, zAccel);
116 
117  _rotate_and_correct_accel(accel_instance[instance], accel);
118 
119  uint8_t nsamples = enable_fast_sampling(accel_instance[instance])?4:1;
120  for (uint8_t i=0; i<nsamples; i++) {
121  _notify_new_accel_raw_sample(accel_instance[instance], accel);
122  }
123 }

128 void AP_InertialSensor_SITL::generate_gyro(uint8_t instance)
129 {
130  // minimum gyro noise is less than 1 bit
131  float gyro_noise = ToRad(0.04f);
132 
133  if (sitl->motors_on) {
134  // add extra noise when the motors are on
135  gyro_noise += ToRad(sitl->gyro_noise);
136  }
137 
138  float p = radians(sitl->state.rollRate) + gyro_drift();
139  float q = radians(sitl->state.pitchRate) + gyro_drift();
140  float r = radians(sitl->state.yawRate) + gyro_drift();
141 
142  const Vector3f &vibe_freq = sitl->vibe_freq;
143  if (vibe_freq.is_zero()) {
144  p += gyro_noise * rand_float();
145  q += gyro_noise * rand_float();
146  r += gyro_noise * rand_float();
147  } else {
148  float t = AP_HAL::micros() * 1.0e-6f;
149  p += sinf(t * 2 * M_PI * vibe_freq.x) * gyro_noise;
150  q += sinf(t * 2 * M_PI * vibe_freq.y) * gyro_noise;
151  r += sinf(t * 2 * M_PI * vibe_freq.z) * gyro_noise;
152  }
153 
154  Vector3f gyro = Vector3f(p, q, r);
155 
156  // add in gyro scaling
157  Vector3f scale = sitl->gyro_scale;
158  gyro.x *= (1 + scale.x*0.01f);
159  gyro.y *= (1 + scale.y*0.01f);
160  gyro.z *= (1 + scale.z*0.01f);
161 
162  _rotate_and_correct_gyro(gyro_instance[instance], gyro);
163 
164  uint8_t nsamples = enable_fast_sampling(gyro_instance[instance])?8:1;
165  for (uint8_t i=0; i<nsamples; i++) {
166  _notify_new_gyro_raw_sample(gyro_instance[instance], gyro);
167  }
168 }
