20 /*
21  process a PPM-sum pulse of the given width
22  */
23 void AP_RCProtocol_PPMSum::process_pulse(uint32_t width_s0, uint32_t width_s1)
24 {
25  if (width_s0 == 0 || width_s1 == 0) {
26  //invalid data: reset frame
27  ppm_state._channel_counter = -1;
28  return;
29  }
30  uint32_t width_usec = width_s0 + width_s1;
31  if (width_usec >= 2700) {
32  // a long pulse indicates the end of a frame. Reset the
33  // channel counter so next pulse is channel 0
34  if (ppm_state._channel_counter >= MIN_RCIN_CHANNELS) {
35  add_input(ppm_state._channel_counter, ppm_state._pulse_capt, false);
36  }
37  ppm_state._channel_counter = 0;
38  return;
39  }
40  if (ppm_state._channel_counter == -1) {
41  // we are not synchronised
42  return;
43  }
44 
45  /*
46  we limit inputs to between 700usec and 2300usec. This allows us
47  to decode SBUS on the same pin, as SBUS will have a maximum
48  pulse width of 100usec
49  */
50  if (width_usec > 700 && width_usec < 2300) {
51  // take a reading for the current channel
52  // buffer these
53  ppm_state._pulse_capt[ppm_state._channel_counter] = width_usec;
54 
55  // move to next channel
56  ppm_state._channel_counter++;
57  }
58 
59  // if we have reached the maximum supported channels then
60  // mark as unsynchronised, so we wait for a wide pulse
61  if (ppm_state._channel_counter >= MAX_RCIN_CHANNELS) {
62  add_input(ppm_state._channel_counter, ppm_state._pulse_capt, false);
63  ppm_state._channel_counter = -1;
64  }
65 }
