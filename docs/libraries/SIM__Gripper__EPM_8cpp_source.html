47  update gripper state

49 void Gripper_EPM::update_servobased(int16_t gripper_pwm)
50 {
51  if (!servo_based) {
52  return;
53  }
54  if (gripper_pwm >= 0) {
55  demand = (gripper_pwm - 1000) * 0.001f; // 0.0 - 1.0
56  if (is_negative(demand)) { // never updated
57  demand = 0.0f;
58  }
59  }
60 }

63 void Gripper_EPM::update_from_demand()
64 {
65  const uint64_t now = AP_HAL::micros64();
66  const float dt = (now - last_update_us) * 1.0e-6f;
67 
68  // decay the field
69  field_strength = field_strength * (100.0f - field_decay_rate * dt) / 100.0f;
70 
71  // note that "demand" here is just an on/off switch; we only care
72  // about which range it falls into
73  if (demand > 0.6f) {
74  // we are instructed to grip harder
75  field_strength = field_strength + (100.0f - field_strength) * field_strength_slew_rate / 100.0f * dt;
76  } else if (demand < 0.4f) {
77  // we are instructed to loosen grip
78  field_strength = field_strength * (100.0f - field_degauss_rate * dt) / 100.0f;
79  } else {
80  // neutral; no demanded change
81  }
82 
83  if (should_report()) {
84  ::fprintf(stderr, "demand=%f\n", demand);
85  printf("Field strength: %f%%\n", field_strength);
86  printf("Field strength: %f Tesla\n", tesla());
87  last_report_us = now;
88  reported_field_strength = field_strength;
89  }
90 
91  last_update_us = now;
92 }

94 void Gripper_EPM::update(const struct sitl_input &input)
95 {
96  const int16_t gripper_pwm = gripper_emp_servo_pin >= 1 ? input.servos[gripper_emp_servo_pin-1] : -1;

104 bool Gripper_EPM::should_report()
105 {
106  if (AP_HAL::micros64() - last_report_us < report_interval) {
107  return false;
108  }
109 
110  if (abs(reported_field_strength - field_strength) > 10.0) {

112  }
113 
114  return false;

117 float Gripper_EPM::tesla()
118 {
119  // https://en.wikipedia.org/wiki/Orders_of_magnitude_(magnetic_field)
120  // 200N lifting capacity ~= 2.5T
121  const float percentage_to_tesla = 0.25f;
122  return static_cast<float>(percentage_to_tesla * field_strength / 100.0f);
