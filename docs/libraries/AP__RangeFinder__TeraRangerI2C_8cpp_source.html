70 /*
71  initialise sensor
72  */
73 bool AP_RangeFinder_TeraRangerI2C::init(void)
74 {
75  if (!dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
76  return false;
77  }
78 
79  dev->set_retries(10);
80 
81  // check WHOAMI
82  uint8_t whoami;
83  if (!dev->read_registers(TR_WHOAMI, &whoami, 1) ||
84  whoami != TR_WHOAMI_VALUE) {
85  dev->get_semaphore()->give();
86  return false;
87  }
88 
89  if (!measure()) {
90  dev->get_semaphore()->give();
91  return false;
92  }
93 
94  // give time for the sensor to process the request
95  hal.scheduler->delay(70);
96 
97  uint16_t _distance_cm;
98  if (!collect_raw(_distance_cm)) {
99  dev->get_semaphore()->give();
100  return false;
101  }
102 
103  dev->get_semaphore()->give();
104 
105  dev->set_retries(1);
106 
107  dev->register_periodic_callback(10000,
108  FUNCTOR_BIND_MEMBER(&AP_RangeFinder_TeraRangerI2C::timer, void));
109 
110  return true;
111 }

120 // collect_raw() - return last value measured by sensor
121 bool AP_RangeFinder_TeraRangerI2C::collect_raw(uint16_t &raw_distance)
122 {
123  uint8_t d[3];
124 
125  // Take range reading
126  if (!dev->transfer(nullptr, 0, d, sizeof(d))) {
127  return false;
128  }
129 
130  // Check for CRC
131  if (d[2] != crc_crc8(d, 2)) {
132  return false;
133  } else {
134  raw_distance = ((uint16_t(d[0]) << 8) | d[1]);
135  return true;
136  }
137 }

139 // Checks for error code and if correct converts to cm
140 bool AP_RangeFinder_TeraRangerI2C::process_raw_measure(uint16_t raw_distance, uint16_t &output_distance_cm)
141 {
142  // Check for error codes
143  if (raw_distance == 0xFFFF) {
144  // Too far away is unreliable so we dont enforce max range here
145  return false;
146  } else if (raw_distance == 0x0000) {
147  // Too close
148  output_distance_cm = params.min_distance_cm;
149  return true;
150  } else if (raw_distance == 0x0001) {
151  // Unable to measure
152  return false;
153  } else {
154  output_distance_cm = raw_distance/10; // Conversion to centimeters
155  return true;
156  }
157 }

159 /*
160  timer called at 100Hz, EVO sensors max freq is 100..240Hz
161 */
162 void AP_RangeFinder_TeraRangerI2C::timer(void)
163 {
164  // Take a reading
165  uint16_t _raw_distance = 0;
166  uint16_t _distance_cm = 0;
167 
168  if (collect_raw(_raw_distance)) {
169  WITH_SEMAPHORE(_sem);
170 
171  if (process_raw_measure(_raw_distance, _distance_cm)){
172  accum.sum += _distance_cm;
173  accum.count++;
174  }
175  }
176  // and immediately ask for a new reading
177  measure();
178 }

180 /*

183 void AP_RangeFinder_TeraRangerI2C::update(void)
184 {
185  WITH_SEMAPHORE(_sem);
186 
187  if (accum.count > 0) {
188  state.distance_cm = accum.sum / accum.count;
189  state.last_reading_ms = AP_HAL::millis();
190  accum.sum = 0;
191  accum.count = 0;
192  update_status();
193  } else if (AP_HAL::millis() - state.last_reading_ms > 200) {
194  set_status(RangeFinder::RangeFinder_NoData);
195  }
196 }
