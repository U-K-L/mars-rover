88 // update visual odometry sensor
89 void AP_VisualOdom::update()
90 {
91  if (!enabled()) {
92  return;
93  }
94 
95  // check for updates
96  if (_state.last_processed_sensor_update_ms == _state.last_sensor_update_ms) {
97  // This reading has already been processed
98  return;
99  }
100  _state.last_processed_sensor_update_ms = _state.last_sensor_update_ms;
101 
102  const float time_delta_sec = get_time_delta_usec() / 1000000.0f;
103 
104  AP::ahrs_navekf().writeBodyFrameOdom(get_confidence(),
105  get_position_delta(),
106  get_angle_delta(),
107  time_delta_sec,
108  get_last_update_ms(),
109  get_pos_offset());
110  // log sensor data
111  AP::logger().Write_VisualOdom(time_delta_sec,
112  get_angle_delta(),
113  get_position_delta(),
114  get_confidence());
115 }

117 // return true if sensor is basically healthy (we are receiving data)
118 bool AP_VisualOdom::healthy() const
119 {
120  if (!enabled()) {
121  return false;
122  }
123 
124  // healthy if we have received sensor messages within the past 300ms
125  return ((AP_HAL::millis() - _state.last_sensor_update_ms) < AP_VISUALODOM_TIMEOUT_MS);
126 }
