54 void AP_WindVane_NMEA::update()
55 {
56  if (uart == nullptr) {
57  return;
58  }
59 
60  // read any available lines from the windvane
61  int16_t nbytes = uart->available();
62  while (nbytes-- > 0) {
63  char c = uart->read();
64  if (decode(c)) {
65  // user may not have NMEA selected for both speed and direction
66  if (_frontend._direction_type.get() == _frontend.WindVaneType::WINDVANE_NMEA) {
67  direction_update_frontend(wrap_PI(radians(_wind_dir_deg + _frontend._dir_analog_bearing_offset.get()) + AP::ahrs().yaw));
68  }
69  if (_frontend._speed_sensor_type.get() == _frontend.Speed_type::WINDSPEED_NMEA) {
70  speed_update_frontend(_speed_ms);
71  }
72  }
73  }
74 }

76 // add a single character to the buffer and attempt to decode
77 // returns true if a complete sentence was successfully decoded
78 bool AP_WindVane_NMEA::decode(char c)
79 {
80  switch (c) {
81  case ',':
82  // end of a term, add to checksum
83  _checksum ^= c;
84  FALLTHROUGH;
85  case '\r':
86  case '\n':
87  case '*':
88  {
89  // null terminate and decode latest term
90  _term[_term_offset] = 0;
91  bool valid_sentence = decode_latest_term();
92 
93  // move onto next term
94  _term_number++;
95  _term_offset = 0;
96  _term_is_checksum = (c == '*');
97  return valid_sentence;
98  }
99 
100  case '$': // sentence begin
101  _sentence_valid = false;
102  _term_number = 0;
103  _term_offset = 0;
104  _checksum = 0;
105  _term_is_checksum = false;
106  _wind_dir_deg = -1.0f;
107  _speed_ms = -1.0f;
108  return false;
109  }
110 
111  // ordinary characters are added to term
112  if (_term_offset < sizeof(_term) - 1) {
113  _term[_term_offset++] = c;
114  }
115  if (!_term_is_checksum) {
116  _checksum ^= c;
117  }
118 
119  return false;
120 }

122 // decode the most recently consumed term
123 // returns true if new sentence has just passed checksum test and is validated
124 bool AP_WindVane_NMEA::decode_latest_term()
125 {
126  // handle the last term in a message
127  if (_term_is_checksum) {
128  uint8_t checksum = 16 * char_to_hex(_term[0]) + char_to_hex(_term[1]);
129  return ((checksum == _checksum) && _sentence_valid);
130  }
131 
132  // the first term determines the sentence type
133  if (_term_number == 0) {
134  // the first two letters of the NMEA term are the talker ID.
135  // we accept any two characters here.
136  if (_term[0] < 'A' || _term[0] > 'Z' ||
137  _term[1] < 'A' || _term[1] > 'Z') {
138  // unknown ID (we are actually expecting II)
139  return false;
140  }
141  const char *term_type = &_term[2];
142  if (strcmp(term_type, "MWV") == 0) {
143  // we found the sentence type for wind
144  _sentence_valid = true;
145  }
146  return false;
147  }
148 
149  // if this is not the sentence we want then wait for another
150  if (!_sentence_valid) {
151  return false;
152  }
153 
154  switch (_term_number) {
155  case 1:
156  _wind_dir_deg = atof(_term);
157  // check for sensible value
158  if (is_negative(_wind_dir_deg) || _wind_dir_deg > 360.0f) {
159  _sentence_valid = false;
160  }
161  break;
162 
163  case 2:
164  // we are expecting R for relative wind
165  // (could be T for true wind, maybe add in the future...)
166  if (_term[0] != 'R') {
167  _sentence_valid = false;
168  }
169  break;
170 
171  case 3:
172  _speed_ms = atof(_term);
173  break;
174 
175  case 4:
176  if (_term[0] == 'K') {
177  // convert from km/h to m/s
178  _speed_ms *= KM_PER_HOUR_TO_M_PER_SEC;
179  } else if (_term[0] == 'N') {
180  // convert from Knots to m/s
181  _speed_ms *= KNOTS_TO_M_PER_SEC;
182  }
183  // could also be M for m/s, but we want that anyway so nothing to do
184  // check for sensible value
185  if (is_negative(_speed_ms) || _speed_ms > 100.0f) {
186  _sentence_valid = false;
187  }
188  break;
189 
190  case 5:
191  // expecting A for data valid
192  if (_term[0] != 'A') {
193  _sentence_valid = false;
194  }
195  break;
196 
197  }
198  return false;
199 }

201 // return the numeric value of an ascii hex character
202 int16_t AP_WindVane_NMEA::char_to_hex(char a)
203 {
204  if (a >= 'A' && a <= 'F')
205  return a - 'A' + 10;
206  else if (a >= 'a' && a <= 'f')
207  return a - 'a' + 10;
208  else
209  return a - '0';
