42  * init - perform required initialisation
43  */
44 bool AP_Frsky_Telem::init()
45 {
46  const AP_SerialManager &serial_manager = AP::serialmanager();
47 
48  // check for protocol configured for a serial port - only the first serial port with one of these protocols will then run (cannot have FrSky on multiple serial ports)
49  if ((_port = serial_manager.find_serial(AP_SerialManager::SerialProtocol_FrSky_D, 0))) {
50  _protocol = AP_SerialManager::SerialProtocol_FrSky_D; // FrSky D protocol (D-receivers)
51  } else if ((_port = serial_manager.find_serial(AP_SerialManager::SerialProtocol_FrSky_SPort, 0))) {
52  _protocol = AP_SerialManager::SerialProtocol_FrSky_SPort; // FrSky SPort protocol (X-receivers)
53  } else if ((_port = serial_manager.find_serial(AP_SerialManager::SerialProtocol_FrSky_SPort_Passthrough, 0))) {
54  _protocol = AP_SerialManager::SerialProtocol_FrSky_SPort_Passthrough; // FrSky SPort and SPort Passthrough (OpenTX) protocols (X-receivers)
55  // make frsky_telemetry available to GCS_MAVLINK (used to queue statustext messages from GCS_MAVLINK)
56  // add firmware and frame info to message queue
57  const char* _frame_string = gcs().frame_string();
58  if (_frame_string == nullptr) {
59  queue_message(MAV_SEVERITY_INFO, AP::fwversion().fw_string);
60  } else {
61  char firmware_buf[MAVLINK_MSG_STATUSTEXT_FIELD_TEXT_LEN+1];
62  snprintf(firmware_buf, sizeof(firmware_buf), "%s %s", AP::fwversion().fw_string, _frame_string);
63  queue_message(MAV_SEVERITY_INFO, firmware_buf);
64  }
65  }
66 
67  if (_port != nullptr) {
68  hal.scheduler->register_io_process(FUNCTOR_BIND_MEMBER(&AP_Frsky_Telem::tick, void));
69  // we don't want flow control for either protocol
70  _port->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
71 
72  return true;
73  }
74 
75  return false;
76 }

79 /*
80  * send telemetry data
81  * for FrSky SPort Passthrough (OpenTX) protocol (X-receivers)
82  */
83 void AP_Frsky_Telem::send_SPort_Passthrough(void)
84 {
85  int16_t numc;
86  numc = _port->available();
87 
88  // check if available is negative
89  if (numc < 0) {
90  return;
91  }
92 
93  // this is the constant for hub data frame
94  if (_port->txspace() < 19) {
95  return;
96  }
97 
98  // keep only the last two bytes of the data found in the serial buffer, as we shouldn't respond to old poll requests
99  uint8_t prev_byte = 0;
100  for (int16_t i = 0; i < numc; i++) {
101  prev_byte = _passthrough.new_byte;
102  _passthrough.new_byte = _port->read();
103  }
104 
105  if ((prev_byte == START_STOP_SPORT) && (_passthrough.new_byte == SENSOR_ID_28)) { // byte 0x7E is the header of each poll request
106  if (_passthrough.send_chunk) { // skip other data and send a message chunk during this iteration
107  _passthrough.send_chunk = false;
108  if (get_next_msg_chunk()) {
109  send_uint32(DIY_FIRST_ID, _msg_chunk.chunk);
110  }
111  } else {
112  // build message queue for sensor_status_flags
113  check_sensor_status_flags();
114  // build message queue for ekf_status
115  check_ekf_status();
116  // if there are pending messages in the queue, send them during next iteration
117  if (!_statustext_queue.empty()) {
118  _passthrough.send_chunk = true;
119  }
120  if (_passthrough.send_attiandrng) { // skip other data, send attitude (roll, pitch) and range only this iteration
121  _passthrough.send_attiandrng = false; // next iteration, check if we should send something other
122  } else { // send other sensor data if it's time for them, and reset the corresponding timer if sent
123  _passthrough.send_attiandrng = true; // next iteration, send attitude b/c it needs frequent updates to remain smooth
124  uint32_t now = AP_HAL::millis();
125  if ((now - _passthrough.params_timer) >= 1000) {
126  send_uint32(DIY_FIRST_ID+7, calc_param());
127  _passthrough.params_timer = AP_HAL::millis();
128  return;
129  }
130  if ((now - _passthrough.ap_status_timer) >= 500) {
131  if (gcs().vehicle_initialised()) { // send ap status only once vehicle has been initialised
132  send_uint32(DIY_FIRST_ID+1, calc_ap_status());
133  _passthrough.ap_status_timer = AP_HAL::millis();
134  return;
135  }
136  }
137  if ((now - _passthrough.batt_timer) >= 1000) {
138  send_uint32(DIY_FIRST_ID+3, calc_batt(0));
139  _passthrough.batt_timer = AP_HAL::millis();
140  return;
141  }
142  if (AP::battery().num_instances() > 1) {
143  if ((now - _passthrough.batt_timer2) >= 1000) {
144  send_uint32(DIY_FIRST_ID+8, calc_batt(1));
145  _passthrough.batt_timer2 = AP_HAL::millis();
146  return;
147  }
148  }
149  if ((now - _passthrough.gps_status_timer) >= 1000) {
150  send_uint32(DIY_FIRST_ID+2, calc_gps_status());
151  _passthrough.gps_status_timer = AP_HAL::millis();
152  return;
153  }
154  if ((now - _passthrough.home_timer) >= 500) {
155  send_uint32(DIY_FIRST_ID+4, calc_home());
156  _passthrough.home_timer = AP_HAL::millis();
157  return;
158  }
159  if ((now - _passthrough.velandyaw_timer) >= 500) {
160  send_uint32(DIY_FIRST_ID+5, calc_velandyaw());
161  _passthrough.velandyaw_timer = AP_HAL::millis();
162  return;
163  }
164  if ((now - _passthrough.gps_latlng_timer) >= 1000) {
165  send_uint32(GPS_LONG_LATI_FIRST_ID, calc_gps_latlng(&_passthrough.send_latitude)); // gps latitude or longitude
166  if (!_passthrough.send_latitude) { // we've cycled and sent one each of longitude then latitude, so reset the timer
167  _passthrough.gps_latlng_timer = AP_HAL::millis();
168  }
169  return;
170  }
171  }
172  // if nothing else needed to be sent, send attitude (roll, pitch) and range data
173  send_uint32(DIY_FIRST_ID+6, calc_attiandrng());
174  }
175  }
176 }

182 void AP_Frsky_Telem::send_SPort(void)
183 {
184  const AP_AHRS &_ahrs = AP::ahrs();
185 
186  int16_t numc;
187  numc = _port->available();
188 
189  // check if available is negative
190  if (numc < 0) {
191  return;
192  }
193 
194  // this is the constant for hub data frame
195  if (_port->txspace() < 19) {
196  return;
197  }
198 
199  for (int16_t i = 0; i < numc; i++) {
200  int16_t readbyte = _port->read();
201  if (_SPort.sport_status == false) {
202  if (readbyte == START_STOP_SPORT) {
203  _SPort.sport_status = true;
204  }
205  } else {
206  const AP_BattMonitor &_battery = AP::battery();
207  switch(readbyte) {
208  case SENSOR_ID_FAS:
209  switch (_SPort.fas_call) {
210  case 0:
211  send_uint32(DATA_ID_FUEL, (uint16_t)roundf(_battery.capacity_remaining_pct())); // send battery remaining
212  break;
213  case 1:
214  send_uint32(DATA_ID_VFAS, (uint16_t)roundf(_battery.voltage() * 10.0f)); // send battery voltage
215  break;
216  case 2:
217  send_uint32(DATA_ID_CURRENT, (uint16_t)roundf(_battery.current_amps() * 10.0f)); // send current consumption
218  break;
219  }
220  if (_SPort.fas_call++ > 2) _SPort.fas_call = 0;
221  break;
222  case SENSOR_ID_GPS:
223  switch (_SPort.gps_call) {
224  case 0:
225  calc_gps_position(); // gps data is not recalculated until all of it has been sent
226  send_uint32(DATA_ID_GPS_LAT_BP, _gps.latdddmm); // send gps lattitude degree and minute integer part
227  break;
228  case 1:
229  send_uint32(DATA_ID_GPS_LAT_AP, _gps.latmmmm); // send gps lattitude minutes decimal part
230  break;
231  case 2:
232  send_uint32(DATA_ID_GPS_LAT_NS, _gps.lat_ns); // send gps North / South information
233  break;
234  case 3:
235  send_uint32(DATA_ID_GPS_LONG_BP, _gps.londddmm); // send gps longitude degree and minute integer part
236  break;
237  case 4:
238  send_uint32(DATA_ID_GPS_LONG_AP, _gps.lonmmmm); // send gps longitude minutes decimal part
239  break;
240  case 5:
241  send_uint32(DATA_ID_GPS_LONG_EW, _gps.lon_ew); // send gps East / West information
242  break;
243  case 6:
244  send_uint32(DATA_ID_GPS_SPEED_BP, _gps.speed_in_meter); // send gps speed integer part
245  break;
246  case 7:
247  send_uint32(DATA_ID_GPS_SPEED_AP, _gps.speed_in_centimeter); // send gps speed decimal part
248  break;
249  case 8:
250  send_uint32(DATA_ID_GPS_ALT_BP, _gps.alt_gps_meters); // send gps altitude integer part
251  break;
252  case 9:
253  send_uint32(DATA_ID_GPS_ALT_AP, _gps.alt_gps_cm); // send gps altitude decimals
254  break;
255  case 10:
256  send_uint32(DATA_ID_GPS_COURS_BP, (uint16_t)((_ahrs.yaw_sensor / 100) % 360)); // send heading in degree based on AHRS and not GPS
257  break;
258  }
259  if (_SPort.gps_call++ > 10) _SPort.gps_call = 0;
260  break;
261  case SENSOR_ID_VARIO:
262  switch (_SPort.vario_call) {
263  case 0 :
264  calc_nav_alt(); // nav altitude is not recalculated until all of it has been sent
265  send_uint32(DATA_ID_BARO_ALT_BP, _gps.alt_nav_meters); // send altitude integer part
266  break;
267  case 1:
268  send_uint32(DATA_ID_BARO_ALT_AP, _gps.alt_nav_cm); // send altitude decimal part
269  break;
270  }
271  if (_SPort.vario_call++ > 1) _SPort.vario_call = 0;
272  break;
273  case SENSOR_ID_SP2UR:
274  switch (_SPort.various_call) {
275  case 0 :
276  send_uint32(DATA_ID_TEMP2, (uint16_t)(AP::gps().num_sats() * 10 + AP::gps().status())); // send GPS status and number of satellites as num_sats*10 + status (to fit into a uint8_t)
277  break;
278  case 1:
279  send_uint32(DATA_ID_TEMP1, gcs().custom_mode()); // send flight mode
280  break;
281  }
282  if (_SPort.various_call++ > 1) _SPort.various_call = 0;
283  break;
284  }
285  _SPort.sport_status = false;
286  }
287  }

334 /*
335  * tick - main call to send data to the receiver (called by scheduler at 1kHz)
336  */
337 void AP_Frsky_Telem::tick(void)
338 {
339  // check UART has been initialised
340  if (!_initialised_uart) {
341  // initialise uart (this must be called from within tick b/c the UART begin must be called from the same thread as it is used from)
342  if (_protocol == AP_SerialManager::SerialProtocol_FrSky_D) { // FrSky D protocol (D-receivers)
343  _port->begin(AP_SERIALMANAGER_FRSKY_D_BAUD, AP_SERIALMANAGER_FRSKY_BUFSIZE_RX, AP_SERIALMANAGER_FRSKY_BUFSIZE_TX);
344  } else { // FrSky SPort and SPort Passthrough (OpenTX) protocols (X-receivers)
345  _port->begin(AP_SERIALMANAGER_FRSKY_SPORT_BAUD, AP_SERIALMANAGER_FRSKY_BUFSIZE_RX, AP_SERIALMANAGER_FRSKY_BUFSIZE_TX);
346  }
347  _initialised_uart = true;// true when we have detected the protocol and UART has been initialised
348  }
349 
350  if (_protocol == AP_SerialManager::SerialProtocol_FrSky_D) { // FrSky D protocol (D-receivers)
351  send_D();
352  } else if (_protocol == AP_SerialManager::SerialProtocol_FrSky_SPort) { // FrSky SPort protocol (X-receivers)
353  send_SPort();
354  } else if (_protocol == AP_SerialManager::SerialProtocol_FrSky_SPort_Passthrough) { // FrSky SPort Passthrough (OpenTX) protocol (X-receivers)
355  send_SPort_Passthrough();
356  }

394  _port->write(byte);

397  if (byte == START_STOP_SPORT) {
398  _port->write(0x7D);
399  _port->write(0x5E);
400  } else if (byte == BYTESTUFF_SPORT) {
401  _port->write(0x7D);
402  _port->write(0x5D);
403  } else {
404  _port->write(byte);
405  }

440 /*
441  * grabs one "chunk" (4 bytes) of the queued message to be transmitted
442  * for FrSky SPort Passthrough (OpenTX) protocol (X-receivers)
443  */
444 bool AP_Frsky_Telem::get_next_msg_chunk(void)
445 {
446  if (_statustext_queue.empty()) {
447  return false;
448  }
449 
450  if (_msg_chunk.repeats == 0) { // if it's the first time get_next_msg_chunk is called for a given chunk
451  uint8_t character = 0;
452  _msg_chunk.chunk = 0; // clear the 4 bytes of the chunk buffer
453 
454  for (int i = 3; i > -1 && _msg_chunk.char_index < sizeof(_statustext_queue[0]->text); i--) {
455  character = _statustext_queue[0]->text[_msg_chunk.char_index++];
456 
457  if (!character) {
458  break;
459  }
460 
461  _msg_chunk.chunk |= character << i * 8;
462  }
463 
464  if (!character || (_msg_chunk.char_index == sizeof(_statustext_queue[0]->text))) { // we've reached the end of the message (string terminated by '\0' or last character of the string has been processed)
465  _msg_chunk.char_index = 0; // reset index to get ready to process the next message
466  // add severity which is sent as the MSB of the last three bytes of the last chunk (bits 24, 16, and 8) since a character is on 7 bits
467  _msg_chunk.chunk |= (_statustext_queue[0]->severity & 0x4)<<21;
468  _msg_chunk.chunk |= (_statustext_queue[0]->severity & 0x2)<<14;
469  _msg_chunk.chunk |= (_statustext_queue[0]->severity & 0x1)<<7;
470  }
471  }
472 
473  if (_msg_chunk.repeats++ > 2) { // repeat each message chunk 3 times to ensure transmission
474  _msg_chunk.repeats = 0;
475  if (_msg_chunk.char_index == 0) { // if we're ready for the next message
476  _statustext_queue.remove(0);
477  }
478  }
479  return true;
480 }

499 /*
500  * add sensor_status_flags information to message cue, normally passed as sys_status mavlink messages to the GCS, for transmission through FrSky link
501  * for FrSky SPort Passthrough (OpenTX) protocol (X-receivers)
502  */
503 void AP_Frsky_Telem::check_sensor_status_flags(void)
504 {
505  uint32_t now = AP_HAL::millis();
506 
507  const uint32_t _sensor_status_flags = sensor_status_flags();
508 
509  if ((now - check_sensor_status_timer) >= 5000) { // prevent repeating any system_status messages unless 5 seconds have passed
510  // only one error is reported at a time (in order of preference). Same setup and displayed messages as Mission Planner.
511  if ((_sensor_status_flags & MAV_SYS_STATUS_SENSOR_GPS) > 0) {
512  queue_message(MAV_SEVERITY_CRITICAL, "Bad GPS Health");

811 uint16_t AP_Frsky_Telem::prep_number(int32_t number, uint8_t digits, uint8_t power)
812 {
813  uint16_t res = 0;
814  uint32_t abs_number = abs(number);
815 
816  if ((digits == 2) && (power == 1)) { // number encoded on 8 bits: 7 bits for digits + 1 for 10^power
817  if (abs_number < 100) {
818  res = abs_number<<1;
819  } else if (abs_number < 1270) {
820  res = ((uint8_t)roundf(abs_number * 0.1f)<<1)|0x1;
821  } else { // transmit max possible value (0x7F x 10^1 = 1270)
822  res = 0xFF;
823  }
824  if (number < 0) { // if number is negative, add sign bit in front
825  res |= 0x1<<8;
826  }
827  } else if ((digits == 2) && (power == 2)) { // number encoded on 9 bits: 7 bits for digits + 2 for 10^power
828  if (abs_number < 100) {
829  res = abs_number<<2;
830  } else if (abs_number < 1000) {
831  res = ((uint8_t)roundf(abs_number * 0.1f)<<2)|0x1;
832  } else if (abs_number < 10000) {
833  res = ((uint8_t)roundf(abs_number * 0.01f)<<2)|0x2;
834  } else if (abs_number < 127000) {
835  res = ((uint8_t)roundf(abs_number * 0.001f)<<2)|0x3;
836  } else { // transmit max possible value (0x7F x 10^3 = 127000)
837  res = 0x1FF;
838  }
839  if (number < 0) { // if number is negative, add sign bit in front
840  res |= 0x1<<9;
841  }
842  } else if ((digits == 3) && (power == 1)) { // number encoded on 11 bits: 10 bits for digits + 1 for 10^power
843  if (abs_number < 1000) {
844  res = abs_number<<1;
845  } else if (abs_number < 10240) {
846  res = ((uint16_t)roundf(abs_number * 0.1f)<<1)|0x1;
847  } else { // transmit max possible value (0x3FF x 10^1 = 10240)
848  res = 0x7FF;
849  }

852  }
853  } else if ((digits == 3) && (power == 2)) { // number encoded on 12 bits: 10 bits for digits + 2 for 10^power
854  if (abs_number < 1000) {
855  res = abs_number<<2;
856  } else if (abs_number < 10000) {

864  }
865  if (number < 0) { // if number is negative, add sign bit in front
866  res |= 0x1<<12;
867  }
868  }
