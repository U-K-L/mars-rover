51 uint8_t AP_RCProtocol_ST24::st24_crc8(uint8_t *ptr, uint8_t len)
52 {
53  uint8_t i, crc ;
54  crc = 0;
55 
56  while (len--) {
57  for (i = 0x80; i != 0; i >>= 1) {
58  if ((crc & 0x80) != 0) {
59  crc <<= 1;
60  crc ^= 0x07;
61 
62  } else {
63  crc <<= 1;
64  }
65 
66  if ((*ptr & i) != 0) {
67  crc ^= 0x07;
68  }
69  }
70 
71  ptr++;
72  }
73 
74  return (crc);
75 }

86 void AP_RCProtocol_ST24::_process_byte(uint8_t byte)
87 {
88  switch (_decode_state) {
89  case ST24_DECODE_STATE_UNSYNCED:
90  if (byte == ST24_STX1) {
91  _decode_state = ST24_DECODE_STATE_GOT_STX1;
92 
93  }
94 
95  break;
96 
97  case ST24_DECODE_STATE_GOT_STX1:
98  if (byte == ST24_STX2) {
99  _decode_state = ST24_DECODE_STATE_GOT_STX2;
100 
101  } else {
102  _decode_state = ST24_DECODE_STATE_UNSYNCED;
103  }
104 
105  break;
106 
107  case ST24_DECODE_STATE_GOT_STX2:
108 
109  /* ensure no data overflow failure or hack is possible */
110  if ((unsigned)byte <= sizeof(_rxpacket.length) + sizeof(_rxpacket.type) + sizeof(_rxpacket.st24_data)) {
111  _rxpacket.length = byte;
112  _rxlen = 0;
113  _decode_state = ST24_DECODE_STATE_GOT_LEN;
114 
115  } else {
116  _decode_state = ST24_DECODE_STATE_UNSYNCED;
117  }
118 
119  break;
120 
121  case ST24_DECODE_STATE_GOT_LEN:
122  _rxpacket.type = byte;
123  _rxlen++;
124  _decode_state = ST24_DECODE_STATE_GOT_TYPE;
125  break;
126 
127  case ST24_DECODE_STATE_GOT_TYPE:
128  _rxpacket.st24_data[_rxlen - 1] = byte;
129  _rxlen++;
130 
131  if (_rxlen == (_rxpacket.length - 1)) {
132  _decode_state = ST24_DECODE_STATE_GOT_DATA;
133  }
134 
135  break;
136 
137  case ST24_DECODE_STATE_GOT_DATA:
138  _rxpacket.crc8 = byte;
139  _rxlen++;
140 
141  if (st24_crc8((uint8_t *) & (_rxpacket.length), _rxlen) == _rxpacket.crc8) {
142 
143  /* decode the actual packet */
144 
145  switch (_rxpacket.type) {
146 
147  case ST24_PACKET_TYPE_CHANNELDATA12: {
148  uint16_t values[12];
149  uint8_t num_values;
150  ChannelData12 *d = (ChannelData12 *)_rxpacket.st24_data;
151  //TBD: add support for RSSI
152  // *rssi = d->rssi;
153  //*rx_count = d->packet_count;
154 
155  /* this can lead to rounding of the strides */
156  num_values = (MAX_RCIN_CHANNELS < 12) ? MAX_RCIN_CHANNELS : 12;
157 
158  unsigned stride_count = (num_values * 3) / 2;
159  unsigned chan_index = 0;
160 
161  for (unsigned i = 0; i < stride_count; i += 3) {
162  values[chan_index] = ((uint16_t)d->channel[i] << 4);
163  values[chan_index] |= ((uint16_t)(0xF0 & d->channel[i + 1]) >> 4);
164  /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
165  values[chan_index] = (uint16_t)(values[chan_index] * ST24_SCALE_FACTOR + .5f) + ST24_SCALE_OFFSET;
166  chan_index++;
167 
168  values[chan_index] = ((uint16_t)d->channel[i + 2]);
169  values[chan_index] |= (((uint16_t)(0x0F & d->channel[i + 1])) << 8);
170  /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
171  values[chan_index] = (uint16_t)(values[chan_index] * ST24_SCALE_FACTOR + .5f) + ST24_SCALE_OFFSET;
172  chan_index++;
173  }
174  }
175  break;
176 
177  case ST24_PACKET_TYPE_CHANNELDATA24: {
178  uint16_t values[24];
179  uint8_t num_values;
180  ChannelData24 *d = (ChannelData24 *)&_rxpacket.st24_data;
181 
182  //*rssi = d->rssi;
183  //*rx_count = d->packet_count;
184 
185  /* this can lead to rounding of the strides */
186  num_values = (MAX_RCIN_CHANNELS < 24) ? MAX_RCIN_CHANNELS : 24;
187 
188  unsigned stride_count = (num_values * 3) / 2;
189  unsigned chan_index = 0;
190 
191  for (unsigned i = 0; i < stride_count; i += 3) {
192  values[chan_index] = ((uint16_t)d->channel[i] << 4);
193  values[chan_index] |= ((uint16_t)(0xF0 & d->channel[i + 1]) >> 4);
194  /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
195  values[chan_index] = (uint16_t)(values[chan_index] * ST24_SCALE_FACTOR + .5f) + ST24_SCALE_OFFSET;
196  chan_index++;
197 
198  values[chan_index] = ((uint16_t)d->channel[i + 2]);
199  values[chan_index] |= (((uint16_t)(0x0F & d->channel[i + 1])) << 8);
200  /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
201  values[chan_index] = (uint16_t)(values[chan_index] * ST24_SCALE_FACTOR + .5f) + ST24_SCALE_OFFSET;
202  chan_index++;
203  }
204  }
205  break;
206 
207  case ST24_PACKET_TYPE_TRANSMITTERGPSDATA: {
208 
209  // ReceiverFcPacket* d = (ReceiverFcPacket*)&_rxpacket.st24_data;
210  /* we silently ignore this data for now, as it is unused */
211  }
212  break;
213 
214  default:
215  break;
216  }
217 
218  } else {
219  /* decoding failed */
220  }
221 
222  _decode_state = ST24_DECODE_STATE_UNSYNCED;
223  break;
224  }
