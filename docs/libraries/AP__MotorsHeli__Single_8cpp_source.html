111 // init_outputs - initialise Servo/PWM ranges and endpoints
112 bool AP_MotorsHeli_Single::init_outputs()
113 {
114  if (!_flags.initialised_ok) {
115  // map primary swash servos
116  for (uint8_t i=0; i<AP_MOTORS_HELI_SINGLE_NUM_SWASHPLATE_SERVOS; i++) {
117  add_motor_num(CH_1+i);
118  }
119  if (_swashplate.get_swash_type() == SWASHPLATE_TYPE_H4_90 || _swashplate.get_swash_type() == SWASHPLATE_TYPE_H4_45) {
120  add_motor_num(CH_5);
121  }
122 
123  // yaw servo
124  add_motor_num(CH_4);
125 
126  // initialize main rotor servo
127  _main_rotor.init_servo();
128 
129  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_DIRECTDRIVE_VARPITCH) {
130  _tail_rotor.init_servo();
131  } else if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_SERVO_EXTGYRO) {
132  // external gyro output
133  add_motor_num(AP_MOTORS_HELI_SINGLE_EXTGYRO);
134  }
135  }
136 
137  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_SERVO_EXTGYRO) {
138  // External Gyro uses PWM output thus servo endpoints are forced
139  SRV_Channels::set_output_min_max(SRV_Channels::get_motor_function(AP_MOTORS_HELI_SINGLE_EXTGYRO), 1000, 2000);
140  }
141 
142  // reset swash servo range and endpoints
143  for (uint8_t i=0; i<AP_MOTORS_HELI_SINGLE_NUM_SWASHPLATE_SERVOS; i++) {
144  reset_swash_servo(SRV_Channels::get_motor_function(i));
145  }
146  if (_swashplate.get_swash_type() == SWASHPLATE_TYPE_H4_90 || _swashplate.get_swash_type() == SWASHPLATE_TYPE_H4_45) {
147  reset_swash_servo(SRV_Channels::get_motor_function(4));
148  }
149 
150  // yaw servo is an angle from -4500 to 4500
151  SRV_Channels::set_angle(SRV_Channel::k_motor4, YAW_SERVO_MAX_ANGLE);
152 
153  _flags.initialised_ok = true;
154 
155  return true;
156 }

158 // output_test_seq - spin a motor at the pwm value specified
159 // motor_seq is the motor's sequence number from 1 to the number of motors on the frame
160 // pwm value is an actual pwm value that will be output, normally in the range of 1000 ~ 2000
161 void AP_MotorsHeli_Single::output_test_seq(uint8_t motor_seq, int16_t pwm)
162 {
163  // exit immediately if not armed
164  if (!armed()) {
165  return;
166  }
167 
168  // output to motors and servos
169  switch (motor_seq) {
170  case 1:
171  // swash servo 1
172  rc_write(AP_MOTORS_MOT_1, pwm);
173  break;
174  case 2:
175  // swash servo 2
176  rc_write(AP_MOTORS_MOT_2, pwm);
177  break;
178  case 3:
179  // swash servo 3
180  rc_write(AP_MOTORS_MOT_3, pwm);
181  break;
182  case 4:
183  // external gyro & tail servo
184  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_SERVO_EXTGYRO) {

186  rc_write(AP_MOTORS_HELI_SINGLE_EXTGYRO, _ext_gyro_gain_acro);
187  } else {
188  rc_write(AP_MOTORS_HELI_SINGLE_EXTGYRO, _ext_gyro_gain_std);
189  }
190  }
191  rc_write(AP_MOTORS_MOT_4, pwm);
192  break;
193  case 5:
194  // main rotor
195  rc_write(AP_MOTORS_HELI_SINGLE_RSC, pwm);
196  break;
197  default:
198  // do nothing
199  break;
200  }

241 // calculate_scalars - recalculates various scalers used.
242 void AP_MotorsHeli_Single::calculate_scalars()
243 {
244  // range check collective min, max and mid
245  if( _collective_min >= _collective_max ) {
246  _collective_min = AP_MOTORS_HELI_COLLECTIVE_MIN;
247  _collective_max = AP_MOTORS_HELI_COLLECTIVE_MAX;
248  }
249  _collective_mid = constrain_int16(_collective_mid, _collective_min, _collective_max);
250 
251  // calculate collective mid point as a number from 0 to 1
252  _collective_mid_pct = ((float)(_collective_mid-_collective_min))/((float)(_collective_max-_collective_min));
253 
254  // configure swashplate and update scalars
255  _swashplate.configure();
256  _swashplate.calculate_roll_pitch_collective_factors();
257 
258  // send setpoints to main rotor controller and trigger recalculation of scalars
259  _main_rotor.set_control_mode(static_cast<RotorControlMode>(_rsc_mode.get()));
260  enable_rsc_parameters();
261  calculate_armed_scalars();
262 
263  // send setpoints to DDVP rotor controller and trigger recalculation of scalars
264  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_DIRECTDRIVE_VARPITCH) {
265  _tail_rotor.set_control_mode(ROTOR_CONTROL_MODE_SPEED_SETPOINT);
266  _tail_rotor.set_ramp_time(_rsc_ramp_time);
267  _tail_rotor.set_runup_time(_rsc_runup_time);
268  _tail_rotor.set_critical_speed(_rsc_critical*0.001f);
269  _tail_rotor.set_idle_output(_rsc_idle_output*0.001f);
270  } else {
271  _tail_rotor.set_control_mode(ROTOR_CONTROL_MODE_DISABLED);
272  _tail_rotor.set_ramp_time(0);
273  _tail_rotor.set_runup_time(0);
274  _tail_rotor.set_critical_speed(0);
275  _tail_rotor.set_idle_output(0);
276  }
277 }

321 //
322 // move_actuators - moves swash plate and tail rotor
323 // - expected ranges:
324 // roll : -1 ~ +1
325 // pitch: -1 ~ +1
326 // collective: 0 ~ 1
327 // yaw: -1 ~ +1
328 //
329 void AP_MotorsHeli_Single::move_actuators(float roll_out, float pitch_out, float coll_in, float yaw_out)
330 {
331  float yaw_offset = 0.0f;
332 
333  // initialize limits flag
334  limit.roll_pitch = false;
335  limit.yaw = false;
336  limit.throttle_lower = false;
337  limit.throttle_upper = false;
338 
339  if (_heliflags.inverted_flight) {
340  coll_in = 1 - coll_in;
341  }
342 
343  // rescale roll_out and pitch_out into the min and max ranges to provide linear motion
344  // across the input range instead of stopping when the input hits the constrain value
345  // these calculations are based on an assumption of the user specified cyclic_max
346  // coming into this equation at 4500 or less
347  float total_out = norm(pitch_out, roll_out);
348 
349  if (total_out > (_cyclic_max/4500.0f)) {
350  float ratio = (float)(_cyclic_max/4500.0f) / total_out;
351  roll_out *= ratio;
352  pitch_out *= ratio;
353  limit.roll_pitch = true;
354  }
355 
356  // constrain collective input
357  float collective_out = coll_in;
358  if (collective_out <= 0.0f) {
359  collective_out = 0.0f;
360  limit.throttle_lower = true;
361  }
362  if (collective_out >= 1.0f) {
363  collective_out = 1.0f;
364  limit.throttle_upper = true;
365  }
366 
367  // ensure not below landed/landing collective
368  if (_heliflags.landing_collective && collective_out < _collective_mid_pct) {
369  collective_out = _collective_mid_pct;
370  limit.throttle_lower = true;
371  }
372 
373  // if servo output not in manual mode, process pre-compensation factors
374  if (_servo_mode == SERVO_CONTROL_MODE_AUTOMATED) {
375  // rudder feed forward based on collective
376  // the feed-forward is not required when the motor is stopped or at idle, and thus not creating torque
377  // also not required if we are using external gyro
378  if ((_main_rotor.get_control_output() > _main_rotor.get_idle_output()) && _tail_type != AP_MOTORS_HELI_SINGLE_TAILTYPE_SERVO_EXTGYRO) {
379  // sanity check collective_yaw_effect
380  _collective_yaw_effect = constrain_float(_collective_yaw_effect, -AP_MOTORS_HELI_SINGLE_COLYAW_RANGE, AP_MOTORS_HELI_SINGLE_COLYAW_RANGE);
381  // the 4.5 scaling factor is to bring the values in line with previous releases
382  yaw_offset = _collective_yaw_effect * fabsf(collective_out - _collective_mid_pct) / 4.5f;
383  }
384  } else {
385  yaw_offset = 0.0f;
386  }
387 
388  // feed power estimate into main rotor controller
389  // ToDo: include tail rotor power?
390  // ToDo: add main rotor cyclic power?
391  _main_rotor.set_collective(fabsf(collective_out));
392 
393  // scale collective pitch for swashplate servos
394  float collective_scalar = ((float)(_collective_max-_collective_min))*0.001f;
395  float collective_out_scaled = collective_out * collective_scalar + (_collective_min - 1000)*0.001f;
396 
397  // get servo positions from swashplate library
398  _servo1_out = _swashplate.get_servo_out(CH_1,pitch_out,roll_out,collective_out_scaled);
399  _servo2_out = _swashplate.get_servo_out(CH_2,pitch_out,roll_out,collective_out_scaled);
400  _servo3_out = _swashplate.get_servo_out(CH_3,pitch_out,roll_out,collective_out_scaled);
401  if (_swashplate.get_swash_type() == SWASHPLATE_TYPE_H4_90 || _swashplate.get_swash_type() == SWASHPLATE_TYPE_H4_45) {
402  _servo5_out = _swashplate.get_servo_out(CH_4,pitch_out,roll_out,collective_out_scaled);
403  }
404 
405  // update the yaw rate using the tail rotor/servo
406  move_yaw(yaw_out + yaw_offset);
407 }

425 void AP_MotorsHeli_Single::output_to_motors()
426 {
427  if (!_flags.initialised_ok) {
428  return;
429  }
430 
431  // actually move the servos. PWM is sent based on nominal 1500 center. servo output shifts center based on trim value.
432  rc_write_swash(AP_MOTORS_MOT_1, _servo1_out);
433  rc_write_swash(AP_MOTORS_MOT_2, _servo2_out);
434  rc_write_swash(AP_MOTORS_MOT_3, _servo3_out);
435  // get servo positions from swashplate library and write to servo for 4 servo of 4 servo swashplate
436  if (_swashplate.get_swash_type() == SWASHPLATE_TYPE_H4_90 || _swashplate.get_swash_type() == SWASHPLATE_TYPE_H4_45) {
437  rc_write_swash(AP_MOTORS_MOT_5, _servo5_out);
438  }
439  if (_tail_type != AP_MOTORS_HELI_SINGLE_TAILTYPE_DIRECTDRIVE_FIXEDPITCH){
440  rc_write_angle(AP_MOTORS_MOT_4, _servo4_out * YAW_SERVO_MAX_ANGLE);
441  }
442  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_SERVO_EXTGYRO) {
443  // output gain to exernal gyro
444  if (_acro_tail && _ext_gyro_gain_acro > 0) {
445  rc_write(AP_MOTORS_HELI_SINGLE_EXTGYRO, 1000 + _ext_gyro_gain_acro);
446  } else {
447  rc_write(AP_MOTORS_HELI_SINGLE_EXTGYRO, 1000 + _ext_gyro_gain_std);
448  }
449  }
450 
451  switch (_spool_state) {
452  case SpoolState::SHUT_DOWN:
453  // sends minimum values out to the motors
454  update_motor_control(ROTOR_CONTROL_STOP);
455  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_DIRECTDRIVE_FIXEDPITCH){
456  rc_write_angle(AP_MOTORS_MOT_4, -YAW_SERVO_MAX_ANGLE);
457  }
458  break;
459  case SpoolState::GROUND_IDLE:
460  // sends idle output to motors when armed. rotor could be static or turning (autorotation)
461  update_motor_control(ROTOR_CONTROL_IDLE);
462  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_DIRECTDRIVE_FIXEDPITCH){
463  rc_write_angle(AP_MOTORS_MOT_4, -YAW_SERVO_MAX_ANGLE);
464  }
465  break;
466  case SpoolState::SPOOLING_UP:
467  case SpoolState::THROTTLE_UNLIMITED:
468  // set motor output based on thrust requests
469  update_motor_control(ROTOR_CONTROL_ACTIVE);
470  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_DIRECTDRIVE_FIXEDPITCH){
471  // constrain output so that motor never fully stops
472  _servo4_out = constrain_float(_servo4_out, -0.9f, 1.0f);
473  // output yaw servo to tail rsc
474  rc_write_angle(AP_MOTORS_MOT_4, _servo4_out * YAW_SERVO_MAX_ANGLE);
475  }
476  break;
477  case SpoolState::SPOOLING_DOWN:
478  // sends idle output to motors and wait for rotor to stop
479  update_motor_control(ROTOR_CONTROL_IDLE);
480  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_DIRECTDRIVE_FIXEDPITCH){
481  rc_write_angle(AP_MOTORS_MOT_4, -YAW_SERVO_MAX_ANGLE);
482  }
483  break;
484 
485  }

488 // servo_test - move servos through full range of movement
489 void AP_MotorsHeli_Single::servo_test()
490 {
491  _servo_test_cycle_time += 1.0f / _loop_rate;
492 
493  if ((_servo_test_cycle_time >= 0.0f && _servo_test_cycle_time < 0.5f)|| // Tilt swash back
494  (_servo_test_cycle_time >= 6.0f && _servo_test_cycle_time < 6.5f)){
495  _pitch_test += (1.0f / (_loop_rate / 2.0f));
496  _oscillate_angle += 8 * M_PI / _loop_rate;
497  _yaw_test = 0.5f * sinf(_oscillate_angle);
