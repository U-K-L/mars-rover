67 lua_scripts::script_info *lua_scripts::load_script(lua_State *L, char *filename) {
68  if (int error = luaL_loadfile(L, filename)) {
69  switch (error) {
70  case LUA_ERRSYNTAX:
71  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: Syntax error in %s", filename);
72  return nullptr;
73  case LUA_ERRMEM:
74  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: Insufficent memory loading %s", filename);
75  return nullptr;
76  case LUA_ERRFILE:
77  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: Unable to load the file: %s", lua_tostring(L, -1));
78  hal.console->printf("Lua: File error: %s\n", lua_tostring(L, -1));
79  return nullptr;
80  default:
81  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: Unknown error (%d) loading %s", error, filename);
82  return nullptr;
83  }
84  }
85 
86  script_info *new_script = (script_info *)hal.util->heap_realloc(_heap, nullptr, sizeof(script_info));
87  if (new_script == nullptr) {
88  // No memory, shouldn't happen, we even attempted to do a GC
89  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: Insufficent memory loading %s", filename);
90  lua_pop(L, 1); // we can't use the function we just loaded, so ditch it
91  return nullptr;
92  }
93 
94  new_script->name = filename;
95  new_script->next = nullptr;
96 
97 
98  // find and create a sandbox for the new chunk
99  lua_getglobal(L, "get_sandbox_env");
100  if (lua_pcall(L, 0, LUA_MULTRET, 0)) {
101  gcs().send_text(MAV_SEVERITY_CRITICAL, "Scripting: Could not create sandbox: %s", lua_tostring(L, -1));
102  hal.console->printf("Lua: Scripting: Could not create sandbox: %s", lua_tostring(L, -1));
103  return nullptr;
104  }
105  load_generated_sandbox(L);
106  lua_setupvalue(L, -2, 1);
107 
108  new_script->lua_ref = luaL_ref(L, LUA_REGISTRYINDEX); // cache the reference
109  new_script->next_run_ms = AP_HAL::millis64() - 1; // force the script to be stale
110 
111  return new_script;
112 }

114 void lua_scripts::load_all_scripts_in_dir(lua_State *L, const char *dirname) {
115  if (dirname == nullptr) {
116  return;
117  }
118 
119  DIR *d = opendir(dirname);
120  if (d == nullptr) {
121  gcs().send_text(MAV_SEVERITY_INFO, "Lua: Could not find a scripts directory");
122  return;
123  }
124 
125  // load anything that ends in .lua
126  for (struct dirent *de=readdir(d); de; de=readdir(d)) {
127  uint8_t length = strlen(de->d_name);
128  if (length < 5) {
129  // not long enough
130  continue;
131  }
132 
133  if (strncmp(&de->d_name[length-4], ".lua", 4)) {
134  // doesn't end in .lua
135  continue;
136  }
137 
138  // FIXME: because chunk name fetching is not working we are allocating and storing an extra string we shouldn't need to
139  size_t size = strlen(dirname) + strlen(de->d_name) + 2;
140  char * filename = (char *) hal.util->heap_realloc(_heap, nullptr, size);
141  if (filename == nullptr) {
142  continue;
143  }
144  snprintf(filename, size, "%s/%s", dirname, de->d_name);
145 
146  // we have something that looks like a lua file, attempt to load it
147  script_info * script = load_script(L, filename);
148  if (script == nullptr) {
149  hal.util->heap_realloc(_heap, filename, 0);
150  continue;
151  }
152  reschedule_script(script);
153 
154  }
155  closedir(d);
156 }

158 void lua_scripts::run_next_script(lua_State *L) {
159  if (scripts == nullptr) {
160 #if defined(AP_SCRIPTING_CHECKS) && AP_SCRIPTING_CHECKS >= 1
161  AP_HAL::panic("Lua: Attempted to run a script without any scripts queued");
162 #endif // defined(AP_SCRIPTING_CHECKS) && AP_SCRIPTING_CHECKS >= 1
163  return;
164  }
165 
166  // reset the current script tracking information
167  overtime = false;
168 
169  // strip the selected script out of the list
170  script_info *script = scripts;
171  scripts = script->next;
172 
173  // reset the hook to clear the counter
174  const int32_t vm_steps = MAX(_vm_steps, 1000);
175  lua_sethook(L, hook, LUA_MASKCOUNT, vm_steps);
176 
177  // store top of stack so we can calculate the number of return values
178  int stack_top = lua_gettop(L);
179 
180  // pop the function to the top of the stack
181  lua_rawgeti(L, LUA_REGISTRYINDEX, script->lua_ref);
182 
183  if(lua_pcall(L, 0, LUA_MULTRET, 0)) {
184  if (overtime) {
185  // script has consumed an excessive amount of CPU time
186  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: %s exceeded time limit (%d)", script->name, vm_steps);
187  remove_script(L, script);
188  } else {
189  gcs().send_text(MAV_SEVERITY_INFO, "Lua: %s", lua_tostring(L, -1));
190  hal.console->printf("Lua: Error: %s\n", lua_tostring(L, -1));
191  remove_script(L, script);
192  }
193  lua_pop(L, 1);
194  return;
195  } else {
196  int returned = lua_gettop(L) - stack_top;
197  switch (returned) {
198  case 0:
199  // no time to reschedule so bail out
200  remove_script(L, script);
201  break;
202  case 2:
203  {
204  // sanity check the return types
205  if (lua_type(L, -1) != LUA_TNUMBER) {
206  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: %s did not return a delay (0x%d)", script->name, lua_type(L, -1));
207  lua_pop(L, 2);
208  remove_script(L, script);
209  return;
210  }
211  if (lua_type(L, -2) != LUA_TFUNCTION) {
212  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: %s did not return a function (0x%d)", script->name, lua_type(L, -2));
213  lua_pop(L, 2);
214  remove_script(L, script);
215  return;
216  }
217 
218  // types match the expectations, go ahead and reschedule
219  script->next_run_ms = AP_HAL::millis64() + (uint64_t)luaL_checknumber(L, -1);
220  lua_pop(L, 1);
221  int old_ref = script->lua_ref;
222  script->lua_ref = luaL_ref(L, LUA_REGISTRYINDEX);
223  luaL_unref(L, LUA_REGISTRYINDEX, old_ref);
224  reschedule_script(script);
225  break;
226  }
227  default:
228  {
229  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: %s returned bad result count (%d)", script->name, returned);
230  remove_script(L, script);
231  // pop all the results we got that we didn't expect
232  lua_pop(L, returned);
233  break;
234  }
235  }
236  }
237 }

239 void lua_scripts::remove_script(lua_State *L, script_info *script) {
240  if (script == nullptr) {
241  return;
242  }
243 
244  // ensure that the script isn't in the loaded list for any reason
245  if (scripts == nullptr) {
246  // nothing to do, already not in the list
247  } else if (scripts == script) {
248  scripts = script->next;
249  } else {
250  for(script_info * current = scripts; current->next != nullptr; current = current->next) {
251  if (current->next == script) {
252  current->next = script->next;
253  break;
254  }
255  }
256  }
257 
258  if (L != nullptr) {
259  // state could be null if we are force killing all scripts
260  luaL_unref(L, LUA_REGISTRYINDEX, script->lua_ref);
261  }
262  hal.util->heap_realloc(_heap, script->name, 0);
263  hal.util->heap_realloc(_heap, script, 0);
264 }

266 void lua_scripts::reschedule_script(script_info *script) {
267  if (script == nullptr) {
268 #if defined(AP_SCRIPTING_CHECKS) && AP_SCRIPTING_CHECKS >= 1
269  AP_HAL::panic("Lua: Attempted to schedule a null pointer");
270 #endif // defined(AP_SCRIPTING_CHECKS) && AP_SCRIPTING_CHECKS >= 1
271  return;
272  }
273 
274  script->next = nullptr;
275  if (scripts == nullptr) {
276  scripts = script;
277  return;
278  }
279 
280  uint64_t next_run_ms = script->next_run_ms;
281 
282  if (scripts->next_run_ms > next_run_ms) {
283  script->next = scripts;
284  scripts = script;
285  return;
286  }
287 
288  script_info *previous = scripts;
289  while (previous->next != nullptr) {
290  if (previous->next->next_run_ms > next_run_ms) {
291  script->next = previous->next;
292  previous->next = script;
293  return;
294  }
295  previous = previous->next;
296  }
297 
298  previous->next = script;
299 }

308 void lua_scripts::run(void) {
309  if (_heap == nullptr) {
310  gcs().send_text(MAV_SEVERITY_INFO, "Lua: Unable to allocate a heap");
311  return;
312  }
313 
314  // panic should be hooked first
315  if (setjmp(panic_jmp)) {
316  if (lua_state != nullptr) {
317  lua_close(lua_state); // shutdown the old state
318  }
319  // remove all the old scheduled scripts
320  for (script_info *script = scripts; script != nullptr; script = scripts) {
321  remove_script(nullptr, script);
322  }
323  scripts = nullptr;
324  overtime = false;
325  }
326 
327  lua_state = lua_newstate(alloc, NULL);
328  lua_State *L = lua_state;
329  if (L == nullptr) {
330  gcs().send_text(MAV_SEVERITY_CRITICAL, "Lua: Couldn't allocate a lua state");
331  return;
332  }
333  lua_atpanic(L, atpanic);
334  luaL_openlibs(L);
335  load_lua_bindings(L);
336 
337  // load the sandbox creation function
338  uint32_t sandbox_size;
339  char *sandbox_data = (char *)AP_ROMFS::find_decompress("sandbox.lua", sandbox_size);
340  if (sandbox_data == nullptr) {
341  gcs().send_text(MAV_SEVERITY_CRITICAL, "Scripting: Could not find sandbox");
342  return;
343  }
344 
345  if (luaL_dostring(L, sandbox_data)) {
346  gcs().send_text(MAV_SEVERITY_CRITICAL, "Scripting: Loading sandbox: %s", lua_tostring(L, -1));
347  return;
348  }
349  free(sandbox_data);
350 
351  // Scan the filesystem in an appropriate manner and autostart scripts
352  load_all_scripts_in_dir(L, SCRIPTING_DIRECTORY);
353 
354  while (true) {
355 #if defined(AP_SCRIPTING_CHECKS) && AP_SCRIPTING_CHECKS >= 1
356  if (lua_gettop(L) != 0) {
357  AP_HAL::panic("Lua: Stack should be empty before running scripts");
358  }
359 #endif // defined(AP_SCRIPTING_CHECKS) && AP_SCRIPTING_CHECKS >= 1
360 
361  if (scripts != nullptr) {
362 #if defined(AP_SCRIPTING_CHECKS) && AP_SCRIPTING_CHECKS >= 1
363  // Sanity check that the scripts list is ordered correctly
364  script_info *sanity = scripts;
365  while (sanity->next != nullptr) {
366  if (sanity->next_run_ms > sanity->next->next_run_ms) {
367  AP_HAL::panic("Lua: Script tasking order has been violated");
368  }
369  sanity = sanity->next;
370  }
371 #endif // defined(AP_SCRIPTING_CHECKS) && AP_SCRIPTING_CHECKS >= 1
372 
373  // compute delay time
374  uint64_t now_ms = AP_HAL::millis64();
375  if (now_ms < scripts->next_run_ms) {
376  hal.scheduler->delay(scripts->next_run_ms - now_ms);
377  }
378 
379  if (_debug_level > 1) {
380  gcs().send_text(MAV_SEVERITY_DEBUG, "Lua: Running %s", scripts->name);
381  }
382 
383  const uint32_t startMem = lua_gc(L, LUA_GCCOUNT, 0) * 1024 + lua_gc(L, LUA_GCCOUNTB, 0);
384  const uint32_t loadEnd = AP_HAL::micros();
385 
386  run_next_script(L);
387 
388  const uint32_t runEnd = AP_HAL::micros();
389  const uint32_t endMem = lua_gc(L, LUA_GCCOUNT, 0) * 1024 + lua_gc(L, LUA_GCCOUNTB, 0);
390  if (_debug_level > 1) {
391  gcs().send_text(MAV_SEVERITY_DEBUG, "Lua: Time: %d Mem: %d", runEnd - loadEnd, endMem - startMem);
392  }
393 
394  } else {
395  gcs().send_text(MAV_SEVERITY_DEBUG, "Lua: No scripts to run");
396  hal.scheduler->delay(10000);
397  }
398 
399  }
