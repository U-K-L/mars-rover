49 void drive(uint16_t hz_speed) {
50  hal.rcout->set_freq(0xFF, hz_speed);
51 
52  while (1) {
53  for (uint8_t i = 0; i < 14; i++) {
54  hal.rcout->write(i, pwm);
55  pwm += delta;
56  if (delta > 0 && pwm >= 2000) {
57  delta = -1;
58  hal.console->printf("reversing\n");
59  } else if (delta < 0 && pwm <= 1000) {
60  delta = 1;
61  hal.console->printf("normalizing\n");
62  }
63  }
64  hal.scheduler->delay(5);
65  if (hal.console->available()) {
66  break;
67  }
68  }
