115  if (_enabled <= 0) {
116  return;
117  }
118  // check if the plane is sinking too fast for more than a second and release parachute
119  uint32_t time = AP_HAL::millis();
120  if((_critical_sink > 0) && (_sink_rate > _critical_sink) && !_release_initiated && _is_flying) {
121  if(_sink_time == 0) {
122  _sink_time = AP_HAL::millis();
123  }
124  if((time - _sink_time) >= 1000) {
125  release();
126  }
127  } else {
128  _sink_time = 0;
129  }
130 
131  // calc time since release
132  uint32_t time_diff = AP_HAL::millis() - _release_time;
133  uint32_t delay_ms = _delay_ms<=0 ? 0: (uint32_t)_delay_ms;
134 
135  // check if we should release parachute
136  if ((_release_time != 0) && !_release_in_progress) {
137  if (time_diff >= delay_ms) {
138  if (_release_type == AP_PARACHUTE_TRIGGER_TYPE_SERVO) {
139  // move servo
140  SRV_Channels::set_output_pwm(SRV_Channel::k_parachute_release, _servo_on_pwm);
141  }else if (_release_type <= AP_PARACHUTE_TRIGGER_TYPE_RELAY_3) {
142  // set relay
143  _relay.on(_release_type);
144  }
145  _release_in_progress = true;
146  _released = true;
147  }
148  }else if ((_release_time == 0) || time_diff >= delay_ms + AP_PARACHUTE_RELEASE_DURATION_MS) {
149  if (_release_type == AP_PARACHUTE_TRIGGER_TYPE_SERVO) {
150  // move servo back to off position

152  }else if (_release_type <= AP_PARACHUTE_TRIGGER_TYPE_RELAY_3) {
