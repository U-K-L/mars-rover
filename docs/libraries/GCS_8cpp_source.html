48 void GCS::send_to_active_channels(uint32_t msgid, const char *pkt)
49 {
50  const mavlink_msg_entry_t *entry = mavlink_get_msg_entry(msgid);
51  if (entry == nullptr) {
52  return;
53  }
54  for (uint8_t i=0; i<num_gcs(); i++) {
55  GCS_MAVLINK &c = chan(i);
56  if (!c.initialised) {
57  continue;
58  }
59  if (!c.is_active()) {
60  continue;
61  }
62  if (entry->max_msg_len + c.packet_overhead() > c.get_uart()->txspace()) {
63  // no room on this channel
64  continue;
65  }
66  c.send_message(pkt, entry);
67  }
68 }
