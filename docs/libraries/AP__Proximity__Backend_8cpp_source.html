90 // get distances in PROXIMITY_MAX_DIRECTION directions. used for sending distances to ground station
91 bool AP_Proximity_Backend::get_horizontal_distances(AP_Proximity::Proximity_Distance_Array &prx_dist_array) const
92 {
93  // exit immediately if we have no good ranges
94  bool valid_distances = false;
95  for (uint8_t i=0; i<_num_sectors; i++) {
96  if (_distance_valid[i]) {
97  valid_distances = true;
98  break;
99  }
100  }
101  if (!valid_distances) {
102  return false;
103  }
104 
105  // initialise orientations and directions
106  // see MAV_SENSOR_ORIENTATION for orientations (0 = forward, 1 = 45 degree clockwise from north, etc)
107  // distances initialised to maximum distances
108  bool dist_set[PROXIMITY_MAX_DIRECTION]{};
109  for (uint8_t i=0; i<PROXIMITY_MAX_DIRECTION; i++) {
110  prx_dist_array.orientation[i] = i;
111  prx_dist_array.distance[i] = distance_max();
112  }
113 
114  // cycle through all sectors filling in distances
115  for (uint8_t i=0; i<_num_sectors; i++) {
116  if (_distance_valid[i]) {
117  // convert angle to orientation
118  int16_t orientation = static_cast<int16_t>(_angle[i] * (PROXIMITY_MAX_DIRECTION / 360.0f));
119  if ((orientation >= 0) && (orientation < PROXIMITY_MAX_DIRECTION) && (_distance[i] < prx_dist_array.distance[orientation])) {
120  prx_dist_array.distance[orientation] = _distance[i];
121  dist_set[orientation] = true;
122  }
123  }
124  }
125 
126  // fill in missing orientations with average of adjacent orientations if necessary and possible
127  for (uint8_t i=0; i<PROXIMITY_MAX_DIRECTION; i++) {
128  if (!dist_set[i]) {
129  uint8_t orient_before = (i==0) ? (PROXIMITY_MAX_DIRECTION - 1) : (i-1);
130  uint8_t orient_after = (i==(PROXIMITY_MAX_DIRECTION - 1)) ? 0 : (i+1);
131  if (dist_set[orient_before] && dist_set[orient_after]) {
132  prx_dist_array.distance[i] = (prx_dist_array.distance[orient_before] + prx_dist_array.distance[orient_after]) / 2.0f;
133  }
134  }
135  }
136  return true;
137 }

167 // initialise the boundary and sector_edge_vector array used for object avoidance
168 // should be called if the sector_middle_deg or _setor_width_deg arrays are changed
169 void AP_Proximity_Backend::init_boundary()
170 {
171  for (uint8_t sector=0; sector < _num_sectors; sector++) {
172  float angle_rad = radians((float)_sector_middle_deg[sector]+(float)_sector_width_deg[sector]/2.0f);
173  _sector_edge_vector[sector].x = cosf(angle_rad) * 100.0f;
174  _sector_edge_vector[sector].y = sinf(angle_rad) * 100.0f;
175  _boundary_point[sector] = _sector_edge_vector[sector] * PROXIMITY_BOUNDARY_DIST_DEFAULT;
176  }
177  update_locations();
178 }

180 // update boundary points used for object avoidance based on a single sector's distance changing
181 // the boundary points lie on the line between sectors meaning two boundary points may be updated based on a single sector's distance changing
182 // the boundary point is set to the shortest distance found in the two adjacent sectors, this is a conservative boundary around the vehicle
183 void AP_Proximity_Backend::update_boundary_for_sector(uint8_t sector)
184 {
185  // sanity check
186  if (sector >= _num_sectors) {
187  return;
188  }
189 
190  // find adjacent sector (clockwise)
191  uint8_t next_sector = sector + 1;
192  if (next_sector >= _num_sectors) {
193  next_sector = 0;
194  }
195 
196  // boundary point lies on the line between the two sectors at the shorter distance found in the two sectors
197  float shortest_distance = PROXIMITY_BOUNDARY_DIST_DEFAULT;
198  if (_distance_valid[sector] && _distance_valid[next_sector]) {
199  shortest_distance = MIN(_distance[sector], _distance[next_sector]);
200  } else if (_distance_valid[sector]) {
201  shortest_distance = _distance[sector];
202  } else if (_distance_valid[next_sector]) {
203  shortest_distance = _distance[next_sector];
204  }
205  if (shortest_distance < PROXIMITY_BOUNDARY_DIST_MIN) {
206  shortest_distance = PROXIMITY_BOUNDARY_DIST_MIN;
207  }
208  _boundary_point[sector] = _sector_edge_vector[sector] * shortest_distance;
209 
210  // if the next sector (clockwise) has an invalid distance, set boundary to create a cup like boundary
211  if (!_distance_valid[next_sector]) {
212  _boundary_point[next_sector] = _sector_edge_vector[next_sector] * shortest_distance;
213  }
214 
215  // repeat for edge between sector and previous sector
216  uint8_t prev_sector = (sector == 0) ? _num_sectors-1 : sector-1;
217  shortest_distance = PROXIMITY_BOUNDARY_DIST_DEFAULT;
218  if (_distance_valid[prev_sector] && _distance_valid[sector]) {
219  shortest_distance = MIN(_distance[prev_sector], _distance[sector]);
220  } else if (_distance_valid[prev_sector]) {
221  shortest_distance = _distance[prev_sector];
222  } else if (_distance_valid[sector]) {
223  shortest_distance = _distance[sector];
224  }
225  _boundary_point[prev_sector] = _sector_edge_vector[prev_sector] * shortest_distance;
226 
227  // if the sector counter-clockwise from the previous sector has an invalid distance, set boundary to create a cup like boundary
228  uint8_t prev_sector_ccw = (prev_sector == 0) ? _num_sectors-1 : prev_sector-1;
229  if (!_distance_valid[prev_sector_ccw]) {
230  _boundary_point[prev_sector_ccw] = _sector_edge_vector[prev_sector_ccw] * shortest_distance;
231  }
232  update_locations();
233 }

261 bool AP_Proximity_Backend::convert_angle_to_sector(float angle_degrees, uint8_t &sector) const
262 {
263  // sanity check angle
264  if (angle_degrees > 360.0f || angle_degrees < -180.0f) {
265  return false;
266  }
267 
268  // convert to 0 ~ 360
269  if (angle_degrees < 0.0f) {
270  angle_degrees += 360.0f;
271  }
272 
273  bool closest_found = false;
274  uint8_t closest_sector;
275  float closest_angle;
276 
277  // search for which sector angle_degrees falls into
278  for (uint8_t i = 0; i < _num_sectors; i++) {
279  float angle_diff = fabsf(wrap_180(_sector_middle_deg[i] - angle_degrees));
280 
281  // record if closest
282  if (!closest_found || angle_diff < closest_angle) {
283  closest_found = true;
284  closest_sector = i;
285  closest_angle = angle_diff;
286  }
287 
288  if (fabsf(angle_diff) <= _sector_width_deg[i] / 2.0f) {
289  sector = i;
290  return true;
291  }
292  }
293 
294  // angle_degrees might have been within a gap between sectors
295  if (closest_found) {
296  sector = closest_sector;
297  return true;
298  }
299 
300  return false;
301 }

327 // retrieve start or end angle of next ignore area (i.e. closest ignore area higher than the start_angle)
328 // start_or_end = 0 to get start, 1 to retrieve end
329 bool AP_Proximity_Backend::get_next_ignore_start_or_end(uint8_t start_or_end, int16_t start_angle, int16_t &ignore_start) const
330 {
331  bool found = false;
332  int16_t smallest_angle_diff = 0;
333  int16_t smallest_angle_start = 0;
334 
335  for (uint8_t i=0; i < PROXIMITY_MAX_IGNORE; i++) {
336  if (frontend._ignore_width_deg[i] != 0) {
337  int16_t offset = start_or_end == 0 ? -frontend._ignore_width_deg[i] : +frontend._ignore_width_deg[i];
338  int16_t ignore_start_angle = wrap_360(frontend._ignore_angle_deg[i] + offset/2.0f);
339  int16_t ang_diff = wrap_360(ignore_start_angle - start_angle);
340  if (!found || ang_diff < smallest_angle_diff) {
341  smallest_angle_diff = ang_diff;
342  smallest_angle_start = ignore_start_angle;
343  found = true;
344  }
345  }
346  }
347 
348  if (found) {
349  ignore_start = smallest_angle_start;
350  }
351  return found;
352 }

354 void AP_Proximity_Backend::update_locations()
355 {
356  WITH_SEMAPHORE(_rsem);
357 
358  // get number of objects, return early if none
359  const uint8_t num_objects = get_object_count();
360  if (num_objects == 0) {
361  _location_count = 0;
362  return;
363  }
364 
365  // get vehicle location and heading in degrees
366  float veh_bearing;
367  Location my_loc;
368  {
369  WITH_SEMAPHORE(AP::ahrs().get_semaphore());
370  if (!AP::ahrs().get_position(my_loc)) {
371  return;
372  }
373  veh_bearing = AP::ahrs().yaw_sensor * 0.01f;
374  }
375 
376  // convert offset from vehicle position to earth frame location
377  const uint32_t now_ms = AP_HAL::millis();
378  _location_count = 0;
379  for (uint8_t i=0; i<num_objects; i++) {
380  float ang_deg, dist_m;
381  if (get_object_angle_and_distance(i, ang_deg, dist_m)) {
382  Location temp_loc = my_loc;
383  temp_loc.offset_bearing(wrap_180(veh_bearing + ang_deg), dist_m);
384  _locations[_location_count].loc = temp_loc;
385  _locations[_location_count].radius_m = 1;
386  _locations[_location_count].last_update_ms = now_ms;
387  _location_count++;
388  }
389  }
