23 // create a rotation matrix given some euler angles
24 // this is based on http://gentlenav.googlecode.com/files/EulerAngles.pdf
25 template <typename T>
26 void Matrix3<T>::from_euler(float roll, float pitch, float yaw)
27 {
28  const float cp = cosf(pitch);
29  const float sp = sinf(pitch);
30  const float sr = sinf(roll);
31  const float cr = cosf(roll);
32  const float sy = sinf(yaw);
33  const float cy = cosf(yaw);
34 
35  a.x = cp * cy;
36  a.y = (sr * sp * cy) - (cr * sy);
37  a.z = (cr * sp * cy) + (sr * sy);
38  b.x = cp * sy;
39  b.y = (sr * sp * sy) + (cr * cy);
40  b.z = (cr * sp * sy) - (sr * cy);
41  c.x = -sp;
42  c.y = sr * cp;
43  c.z = cr * cp;
44 }

48 template <typename T>
49 void Matrix3<T>::to_euler(float *roll, float *pitch, float *yaw) const
50 {

88 /*
89  fill the matrix from Euler angles in radians in 312 convention
90 */
91 template <typename T>
92 void Matrix3<T>::from_euler312(float roll, float pitch, float yaw)
93 {
94  const float c3 = cosf(pitch);
95  const float s3 = sinf(pitch);
96  const float s2 = sinf(roll);
97  const float c2 = cosf(roll);
98  const float s1 = sinf(yaw);
99  const float c1 = cosf(yaw);
100 
101  a.x = c1 * c3 - s1 * s2 * s3;
102  b.y = c1 * c2;
103  c.z = c3 * c2;
104  a.y = -c2*s1;
105  a.z = s3*c1 + c3*s2*s1;
106  b.x = c3*s1 + s3*s2*c1;
107  b.z = s1*s3 - s2*c1*c3;
108  c.x = -s3*c2;
109  c.y = s2;
110 }

112 // apply an additional rotation from a body frame gyro vector
113 // to a rotation matrix.
114 template <typename T>
115 void Matrix3<T>::rotate(const Vector3<T> &g)
116 {
117  (*this) += Matrix3<T>{
118  a.y * g.z - a.z * g.y, a.z * g.x - a.x * g.z, a.x * g.y - a.y * g.x,
119  b.y * g.z - b.z * g.y, b.z * g.x - b.x * g.z, b.x * g.y - b.y * g.x,
120  c.y * g.z - c.z * g.y, c.z * g.x - c.x * g.z, c.x * g.y - c.y * g.x
121  };
122 }

124 /*
125  re-normalise a rotation matrix
126 */
127 template <typename T>
128 void Matrix3<T>::normalize(void)
129 {
130  const float error = a * b;
131  const Vector3<T> t0 = a - (b * (0.5f * error));
132  const Vector3<T> t1 = b - (a * (0.5f * error));
133  const Vector3<T> t2 = t0 % t1;
134  a = t0 * (1.0f / t0.length());
135  b = t1 * (1.0f / t1.length());
136  c = t2 * (1.0f / t2.length());
137 }

190 T Matrix3<T>::det() const
191 {
192  return a.x * (b.y * c.z - b.z * c.y) +
193  a.y * (b.z * c.x - b.x * c.z) +

197 template <typename T>
198 bool Matrix3<T>::inverse(Matrix3<T>& inv) const
199 {
200  const T d = det();
201 
202  if (is_zero(d)) {
203  return false;
204  }
205 
206  inv.a.x = (b.y * c.z - c.y * b.z) / d;
207  inv.a.y = (a.z * c.y - a.y * c.z) / d;
208  inv.a.z = (a.y * b.z - a.z * b.y) / d;
209  inv.b.x = (b.z * c.x - b.x * c.z) / d;
210  inv.b.y = (a.x * c.z - a.z * c.x) / d;
211  inv.b.z = (b.x * a.z - a.x * b.z) / d;
212  inv.c.x = (b.x * c.y - c.x * b.y) / d;
213  inv.c.y = (c.x * a.y - a.x * c.y) / d;
214  inv.c.z = (a.x * b.y - b.x * a.y) / d;
215 
216  return true;
217 }

238 // create rotation matrix for rotation about the vector v by angle theta
239 // See: http://www.euclideanspace.com/maths/geometry/rotations/conversions/angleToMatrix/
240 template <typename T>
241 void Matrix3<T>::from_axis_angle(const Vector3<T> &v, float theta)
242 {
243  const float C = cosf(theta);
244  const float S = sinf(theta);
245  const float t = 1.0f - C;
246  const Vector3f normv = v.normalized();
247  const float x = normv.x;
248  const float y = normv.y;
249  const float z = normv.z;
250 
251  a.x = t*x*x + C;
252  a.y = t*x*y - z*S;
253  a.z = t*x*z + y*S;
254  b.x = t*x*y + z*S;
255  b.y = t*y*y + C;
256  b.z = t*y*z - x*S;
257  c.x = t*x*z - y*S;
258  c.y = t*y*z + x*S;
259  c.z = t*z*z + C;
260 }
