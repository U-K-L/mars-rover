113 class AP_UAVCAN_FileStorageBackend : public uavcan::dynamic_node_id_server::IStorageBackend
114 {
118  enum { MaxPathLength = 128 };
119 
120  enum { FilePermissions = 438 };
121 
122  enum { MaxNumOpens = 100 };
126  typedef uavcan::MakeString<MaxPathLength>::Type PathString;
127 
128  PathString base_path;
129 
130  static uint8_t num_opens;
131 protected:
132  virtual String get(const String& key) const
133  {
134  using namespace std;
135  PathString path = base_path.c_str();
136  path += key;
137  String value;
138  //This is to deter frequent inflight opening and closing of files during an event
139  //where the device is misbehaving
140  if (num_opens >= MaxNumOpens) {
141  return value;
142  }
143  num_opens++;
144  int fd = open(path.c_str(), O_RDONLY);
145  if (fd >= 0)
146  {
147  char buffer[MaxStringLength + 1];
148  (void)memset(buffer, 0, sizeof(buffer));
149  ssize_t remaining = MaxStringLength;
150  ssize_t total_read = 0;
151  ssize_t nread = 0;
152  do
153  {
154  nread = ::read(fd, &buffer[total_read], remaining);
155  if (nread > 0)
156  {
157  remaining -= nread,
158  total_read += nread;
159  }
160  }
161  while (nread > 0 && remaining > 0);
162  (void)close(fd);
163  if (total_read > 0)
164  {
165  for (int i = 0; i < total_read; i++)
166  {
167  if (buffer[i] == ' ' || buffer[i] == '\n' || buffer[i] == '\r')
168  {
169  buffer[i] = '\0';
170  break;
171  }
172  }
173  value = buffer;
174  }
175  }
176  return value;
177  }
178 
179  virtual void set(const String& key, const String& value)
180  {
181  using namespace std;
182  PathString path = base_path.c_str();
183  path += key;
184  //This is to deter frequent inflight opening and closing of files during an event
185  //where the device is misbehaving
186  if (num_opens >= MaxNumOpens) {
187  return;
188  }
189  num_opens++;
190 #if HAL_OS_POSIX_IO
191  int fd = open(path.c_str(), O_WRONLY | O_CREAT | O_TRUNC, FilePermissions);
192 #else
193  int fd = open(path.c_str(), O_WRONLY | O_CREAT | O_TRUNC);
194 #endif
195  if (fd >= 0)
196  {
197  ssize_t remaining = value.size();
198  ssize_t total_written = 0;
199  ssize_t written = 0;
200  do
201  {
202  written = write(fd, &value.c_str()[total_written], remaining);
203  if (written > 0)
204  {
205  total_written += written;
206  remaining -= written;
207  }
208  }
209  while (written > 0 && remaining > 0);
210 
211  (void)fsync(fd);
212  (void)close(fd);
213  }
214  }

224  int init(const PathString& path)
225  {
226  using namespace std;
227 
228  int rv = -uavcan::ErrInvalidParam;
229 
230  if (path.size() > 0)
231  {
232  base_path = path.c_str();
233 
234  if (base_path.back() == '/')
235  {
236  base_path.pop_back();
237  }
238 
239  rv = 0;

242  {
243  // coverity[toctou]
244  rv = mkdir(base_path.c_str(), S_IRWXU | S_IRWXG | S_IRWXO);
245  }
246  if (rv >= 0)
247  {
248  base_path.push_back('/');
249  if ((base_path.size() + MaxStringLength) > MaxPathLength)
250  {
251  rv = -uavcan::ErrInvalidConfiguration;
252  }
253  }
254  }
255  return rv;
256  }

261 bool AP_UAVCAN_Servers::init(uavcan::Node<0> &node)
262 {
263  if (_server_instance != nullptr) {
264  return true;
265  }
266 
267  int ret = 0;
268 
269  _storage_backend = new AP_UAVCAN_FileStorageBackend;
270  if (_storage_backend == nullptr) {
271  debug_uavcan("UAVCAN_Servers: Failed to Allocate FileStorageBackend\n");
272  goto failed;
273  }
274 
275  ret = _storage_backend->init(UAVCAN_NODE_DB_PATH);
276  if (ret < 0) {
277  debug_uavcan("UAVCAN_Servers: FileStorageBackend init: %d, errno: %d\n", ret, errno);
278  goto failed;
279  }
280 
281  _tracer = new AP_UAVCAN_FileEventTracer;
282  if (_tracer == nullptr) {
283  debug_uavcan("UAVCAN_Servers: Failed to Allocate FileEventTracer\n");
284  goto failed;
285  }
286 
287  _server_instance = new AP_UAVCAN_CentralizedServer(node, *_storage_backend, *_tracer);
288  if (_server_instance == nullptr) {
289  debug_uavcan("UAVCAN_Servers: Failed to Allocate Server\n");
290  goto failed;
291  }
292 
293  {
294  uavcan::dynamic_node_id_server::centralized::Storage storage(*_storage_backend);
295  if (storage.getNodeIDForUniqueID(node.getHardwareVersion().unique_id) != node.getNodeID()) {
296  //Node ID was changed, reseting database
297  reset();
298  }
299  }
300 
301  if (_restart_request_handler == nullptr) {
302  _restart_request_handler = new AP_UAVCAN_RestartRequestHandler();
303  if (_restart_request_handler == nullptr) {
304  goto failed;
305  }
306  }
307  node.setRestartRequestHandler(_restart_request_handler);
308 
309  //Start Dynamic Node Server
310  ret = _server_instance->init(node.getHardwareVersion().unique_id);
311  if (ret < 0) {
312  debug_uavcan("UAVCAN_Server init: %d, errno: %d\n", ret, errno);
313  goto failed;
314  }
315 
316  return true;
317 
318 failed:
319  delete _restart_request_handler;
320  delete _storage_backend;
321  delete _tracer;
322  delete _server_instance;
323  return false;
324 }
