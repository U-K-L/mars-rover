35 void SITL_SFML_LED::update_thread(void)
36 {
37  sf::RenderWindow *w;
38  {
39  WITH_SEMAPHORE(AP::notify().sf_window_mutex);
40  w = new sf::RenderWindow(sf::VideoMode(width, height), "LED");
41  }
42 
43  if (!w) {
44  AP_HAL::panic("Unable to create RGBLed window");
45  }
46 
47  while (true) {
48  {
49  WITH_SEMAPHORE(AP::notify().sf_window_mutex);
50  sf::Event event;
51  while (w->pollEvent(event)) {
52  if (event.type == sf::Event::Closed) {
53  w->close();
54  break;
55  }
56  }
57  if (!w->isOpen()) {
58  break;
59  }
60  const uint32_t colour = red<<16 | green<<8 | blue;
61  if (colour != last_colour) {
62  last_colour = colour;
63 
64  w->clear(sf::Color(red, green, blue, 255));
65  w->display();
66  }
67  }
68  usleep(10000);
69  }
70 }
