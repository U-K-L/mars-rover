489 {

537 bool AC_Fence::load_polygon_from_eeprom()
538 {
539  // check if we need to create array
540  if (!_boundary_create_attempted) {
541  _boundary = (Vector2f *)_poly_loader.create_point_array(sizeof(Vector2f));
542  _boundary_create_attempted = true;
543  }
544 
545  // exit if we could not allocate RAM for the boundary
546  if (_boundary == nullptr) {
547  return false;
548  }
549 
550  // get current location from EKF
551  Location temp_loc;
552  if (!AP::ahrs_navekf().get_location(temp_loc)) {
553  return false;
554  }
555  struct Location ekf_origin {};
556  if (!AP::ahrs().get_origin(ekf_origin)) {
557  return false;
558  }
559 
560  // sanity check total
561  _total = constrain_int16(_total, 0, _poly_loader.max_points());
562 
563  // load each point from eeprom
564  Vector2l temp_latlon;
565  for (uint16_t index=0; index<_total; index++) {
566  // load boundary point as lat/lon point
567  if (!_poly_loader.load_point_from_eeprom(index, temp_latlon)) {
568  return false;
569  }
570  // move into location structure and convert to offset from ekf origin
571  temp_loc.lat = temp_latlon.x;
572  temp_loc.lng = temp_latlon.y;
573  _boundary[index] = ekf_origin.get_distance_NE(temp_loc) * 100.0f;
574  }
575  _boundary_num_points = _total;
576  _boundary_update_ms = AP_HAL::millis();
577 
578  // update validity of polygon
579  _boundary_valid = _poly_loader.boundary_valid(_boundary_num_points, _boundary);
580 
581  return true;
582 }
