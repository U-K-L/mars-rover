30 #if CH_CFG_ST_RESOLUTION == 16
31 static uint32_t get_systime_us32(void)
32 {
33  systime_t now = chVTGetSystemTimeX();
34 #if CH_CFG_ST_FREQUENCY != 1000000
35  now *= (1000000UL / CH_CFG_ST_FREQUENCY);
36 #endif
37  if (now < last_systime) {
38  uint32_t last_u32 = timer_base_us32;
39  timer_base_us32 += (uint32_t)TIME_MAX_SYSTIME;
40  if (timer_base_us32 < last_u32) {
41  timer_base_us64 += ((uint32_t)-1);
42  timer_base_ms += ((uint32_t)-1)/1000;
43  }
44  }
45  last_systime = now;
46  return timer_base_us32 + (uint32_t)now;
47 }
48 
49 #elif CH_CFG_ST_RESOLUTION == 32
50 static uint32_t get_systime_us32(void)
51 {
52  systime_t now = chVTGetSystemTimeX();
53 #if CH_CFG_ST_FREQUENCY != 1000000
54  now *= (1000000UL / CH_CFG_ST_FREQUENCY);
55 #endif
56  if (now < last_systime) {
57  timer_base_us64 += TIME_MAX_SYSTIME;
58  timer_base_ms += TIME_MAX_SYSTIME/1000;
59  }
60  last_systime = now;
61  return now;
62 }

72 uint64_t hrt_micros64()
73 {
74  syssts_t sts = chSysGetStatusAndLockX();
75  uint32_t now = get_systime_us32();
