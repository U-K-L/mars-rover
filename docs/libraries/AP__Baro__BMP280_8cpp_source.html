67 bool AP_Baro_BMP280::_init()
68 {
69  if (!_dev) {
70  return false;
71  }
72  WITH_SEMAPHORE(_dev->get_semaphore());
73 
74  _has_sample = false;
75 
76  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
77 
78  uint8_t whoami;
79  if (!_dev->read_registers(BMP280_REG_ID, &whoami, 1) ||
80  whoami != BMP280_ID) {
81  // not a BMP280
82  return false;
83  }
84 
85  // read the calibration data
86  uint8_t buf[24];
87  _dev->read_registers(BMP280_REG_CALIB, buf, sizeof(buf));
88 
89  _t1 = ((int16_t)buf[1] << 8) | buf[0];
90  _t2 = ((int16_t)buf[3] << 8) | buf[2];
91  _t3 = ((int16_t)buf[5] << 8) | buf[4];
92  _p1 = ((int16_t)buf[7] << 8) | buf[6];
93  _p2 = ((int16_t)buf[9] << 8) | buf[8];
94  _p3 = ((int16_t)buf[11] << 8) | buf[10];
95  _p4 = ((int16_t)buf[13] << 8) | buf[12];
96  _p5 = ((int16_t)buf[15] << 8) | buf[14];
97  _p6 = ((int16_t)buf[17] << 8) | buf[16];
98  _p7 = ((int16_t)buf[19] << 8) | buf[18];
99  _p8 = ((int16_t)buf[21] << 8) | buf[20];
100  _p9 = ((int16_t)buf[23] << 8) | buf[22];
101 
102  // SPI write needs bit mask
103  uint8_t mask = 0xFF;
104  if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI) {
105  mask = 0x7F;
106  }
107 
108  _dev->setup_checked_registers(2, 20);
109 
110  _dev->write_register((BMP280_REG_CTRL_MEAS & mask), (BMP280_OVERSAMPLING_T << 5) |
111  (BMP280_OVERSAMPLING_P << 2) | BMP280_MODE, true);
112 
113  _dev->write_register((BMP280_REG_CONFIG & mask), BMP280_FILTER_COEFFICIENT << 2, true);
114 
115  _instance = _frontend.register_sensor();
116 
117  // request 50Hz update
118  _dev->register_periodic_callback(20 * AP_USEC_PER_MSEC, FUNCTOR_BIND_MEMBER(&AP_Baro_BMP280::_timer, void));
119 
120  return true;
121 }

151 // calculate temperature
152 void AP_Baro_BMP280::_update_temperature(int32_t temp_raw)
153 {
154  int32_t var1, var2, t;
155 
156  // according to datasheet page 22
157  var1 = ((((temp_raw >> 3) - ((int32_t)_t1 << 1))) * ((int32_t)_t2)) >> 11;
158  var2 = (((((temp_raw >> 4) - ((int32_t)_t1)) * ((temp_raw >> 4) - ((int32_t)_t1))) >> 12) * ((int32_t)_t3)) >> 14;
159  _t_fine = var1 + var2;
160  t = (_t_fine * 5 + 128) >> 8;
161 
162  const float temp = ((float)t) / 100.0f;
163 
164  WITH_SEMAPHORE(_sem);
165 
166  _temperature = temp;
167 }

169 // calculate pressure
170 void AP_Baro_BMP280::_update_pressure(int32_t press_raw)
171 {
172  int64_t var1, var2, p;
173 
174  // according to datasheet page 22
175  var1 = ((int64_t)_t_fine) - 128000;
176  var2 = var1 * var1 * (int64_t)_p6;
177  var2 = var2 + ((var1 * (int64_t)_p5) << 17);
178  var2 = var2 + (((int64_t)_p4) << 35);
179  var1 = ((var1 * var1 * (int64_t)_p3) >> 8) + ((var1 * (int64_t)_p2) << 12);
180  var1 = (((((int64_t)1) << 47) + var1)) * ((int64_t)_p1) >> 33;
181 
182  if (var1 == 0) {
183  return;
184  }
185 
186  p = 1048576 - press_raw;
187  p = (((p << 31) - var2) * 3125) / var1;
188  var1 = (((int64_t)_p9) * (p >> 13) * (p >> 13)) >> 25;
189  var2 = (((int64_t)_p8) * p) >> 19;
190  p = ((p + var1 + var2) >> 8) + (((int64_t)_p7) << 4);
191 
192 
193  const float press = (float)p / 256.0f;
194  if (!pressure_ok(press)) {
195  return;
196  }
197 
198  WITH_SEMAPHORE(_sem);
199 
200  _pressure = press;
201  _has_sample = true;
202 }
