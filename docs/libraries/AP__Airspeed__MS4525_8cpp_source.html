37 // probe and initialise the sensor
38 bool AP_Airspeed_MS4525::init()
39 {
40  const struct {
41  uint8_t bus;
42  uint8_t addr;
43  } addresses[] = {
44  { 1, MS4525D0_I2C_ADDR },
45  { 0, MS4525D0_I2C_ADDR },
46  { 2, MS4525D0_I2C_ADDR },
47  { 3, MS4525D0_I2C_ADDR },
48  };
49  bool found = false;
50  for (uint8_t i=0; i<ARRAY_SIZE(addresses); i++) {
51  _dev = hal.i2c_mgr->get_device(addresses[i].bus, addresses[i].addr);
52  if (!_dev) {
53  continue;
54  }
55  WITH_SEMAPHORE(_dev->get_semaphore());
56 
57  // lots of retries during probe
58  _dev->set_retries(10);
59 
60  _measure();
61  hal.scheduler->delay(10);
62  _collect();
63 
64  found = (_last_sample_time_ms != 0);
65  if (found) {
66  printf("MS4525: Found sensor on bus %u address 0x%02x\n", addresses[i].bus, addresses[i].addr);
67  break;
68  }
69  }
70  if (!found) {
71  printf("MS4525: no sensor found\n");
72  return false;
73  }
74 
75  // drop to 2 retries for runtime
76  _dev->set_retries(2);
77 
78  _dev->register_periodic_callback(20000,
79  FUNCTOR_BIND_MEMBER(&AP_Airspeed_MS4525::_timer, void));
80  return true;
81 }

93 /*
94  this equation is an inversion of the equation in the
95  pressure transfer function figure on page 4 of the datasheet
96 
97  We negate the result so that positive differential pressures
98  are generated when the bottom port is used as the static
99  port on the pitot and top port is used as the dynamic port
100 */
101 float AP_Airspeed_MS4525::_get_pressure(int16_t dp_raw) const
102 {
103  const float P_max = get_psi_range();
104  const float P_min = - P_max;
105  const float PSI_to_Pa = 6894.757f;
106 
107  float diff_press_PSI = -((dp_raw - 0.1f*16383) * (P_max-P_min)/(0.8f*16383) + P_min);
108  float press = diff_press_PSI * PSI_to_Pa;
109  return press;
110 }

121 // read the values from the sensor
122 void AP_Airspeed_MS4525::_collect()
123 {
124  uint8_t data[4];
125  uint8_t data2[4];
126 
127  _measurement_started_ms = 0;
128 
129  if (!_dev->transfer(nullptr, 0, data, sizeof(data))) {
130  return;
131  }
132  // reread the data, so we can attempt to detect bad inputs
133  if (!_dev->transfer(nullptr, 0, data2, sizeof(data2))) {
134  return;
135  }
136 
137  uint8_t status = (data[0] & 0xC0) >> 6;
138  // only check the status on the first read, the second read is expected to be stale
139  if (status == 2 || status == 3) {
140  return;
141  }
142 
143  int16_t dp_raw, dT_raw;
144  dp_raw = (data[0] << 8) + data[1];
145  dp_raw = 0x3FFF & dp_raw;
146  dT_raw = (data[2] << 8) + data[3];
147  dT_raw = (0xFFE0 & dT_raw) >> 5;
148 
149  int16_t dp_raw2, dT_raw2;
150  dp_raw2 = (data2[0] << 8) + data2[1];
151  dp_raw2 = 0x3FFF & dp_raw2;
152  dT_raw2 = (data2[2] << 8) + data2[3];
153  dT_raw2 = (0xFFE0 & dT_raw2) >> 5;
154 
155  // reject any values that are the absolute minimum or maximums these
156  // can happen due to gnd lifts or communication errors on the bus
157  if (dp_raw == 0x3FFF || dp_raw == 0 || dT_raw == 0x7FF || dT_raw == 0 ||
158  dp_raw2 == 0x3FFF || dp_raw2 == 0 || dT_raw2 == 0x7FF || dT_raw2 == 0) {
159  return;
160  }
161 
162  // reject any double reads where the value has shifted in the upper more than
163  // 0xFF
164  if (abs(dp_raw - dp_raw2) > 0xFF || abs(dT_raw - dT_raw2) > 0xFF) {
165  return;
166  }
167 
168  float press = _get_pressure(dp_raw);
169  float press2 = _get_pressure(dp_raw2);
170  float temp = _get_temperature(dT_raw);
171  float temp2 = _get_temperature(dT_raw2);
172 
173  _voltage_correction(press, temp);
174  _voltage_correction(press2, temp2);
175 
176  WITH_SEMAPHORE(sem);
177 
178  _press_sum += press + press2;
179  _temp_sum += temp + temp2;
180  _press_count += 2;
181  _temp_count += 2;
182 
183  _last_sample_time_ms = AP_HAL::millis();
184 }

193 void AP_Airspeed_MS4525::_voltage_correction(float &diff_press_pa, float &temperature)
194 {
195  const float slope = 65.0f;
196  const float temp_slope = 0.887f;
197 
198  /*
199  apply a piecewise linear correction within range given by above graph
200  */
201  float voltage_diff = hal.analogin->board_voltage() - 5.0f;
202 
203  voltage_diff = constrain_float(voltage_diff, -0.7f, 0.5f);
204 
205  diff_press_pa -= voltage_diff * slope;
206  temperature -= voltage_diff * temp_slope;
207 }

209 // 50Hz timer

211 {

223 // return the current differential_pressure in Pascal
224 bool AP_Airspeed_MS4525::get_differential_pressure(float &pressure)
225 {
226  if ((AP_HAL::millis() - _last_sample_time_ms) > 100) {
227  return false;
228  }
229 
230  WITH_SEMAPHORE(sem);
231 
232  if (_press_count > 0) {

234  _press_count = 0;
