20 static float rad_diff(float rad1, float rad2)
21 {
22  float diff = rad1 - rad2;
23  if (diff > M_PI) {
24  diff -= 2 * M_PI;
25  }
26  if (diff < -M_PI) {
27  diff += 2 * M_PI;
28  }
29  return fabsf(diff);
30 }

52
