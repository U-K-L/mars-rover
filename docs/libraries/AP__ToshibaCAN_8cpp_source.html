116 // loop to send output to ESCs in background thread
117 void AP_ToshibaCAN::loop()
118 {
119  uavcan::MonotonicTime timeout;
120  const uint32_t timeout_us = MIN(AP::scheduler().get_loop_period_us(), TOSHIBACAN_SEND_TIMEOUT_US);
121 
122  while (true) {
123  if (!_initialized) {
124  // if not initialised wait 2ms
125  debug_can(2, "ToshibaCAN: not initialized\n\r");
126  hal.scheduler->delay_microseconds(2000);
127  continue;
128  }
129 
130  // check for updates
131  if (update_count == update_count_sent) {
132  hal.scheduler->delay_microseconds(50);
133  continue;
134  }
135 
136  // prepare commands and frames
137  if (send_stage == 0) {
138  motor_lock_cmd_t unlock_cmd = {};
139  motor_rotation_cmd_t mot_rot_cmd1;
140  motor_rotation_cmd_t mot_rot_cmd2;
141  motor_rotation_cmd_t mot_rot_cmd3;
142  {
143  // take semaphore to read scaled motor outputs
144  WITH_SEMAPHORE(_rc_out_sem);
145 
146  // prepare command to lock or unlock motors
147  unlock_cmd.motor1 = (_scaled_output[0] == 0) ? 2 : 1;
148  unlock_cmd.motor2 = (_scaled_output[1] == 0) ? 2 : 1;
149  unlock_cmd.motor3 = (_scaled_output[2] == 0) ? 2 : 1;
150  unlock_cmd.motor4 = (_scaled_output[3] == 0) ? 2 : 1;
151  unlock_cmd.motor5 = (_scaled_output[4] == 0) ? 2 : 1;
152  unlock_cmd.motor6 = (_scaled_output[5] == 0) ? 2 : 1;
153  unlock_cmd.motor7 = (_scaled_output[6] == 0) ? 2 : 1;
154  unlock_cmd.motor8 = (_scaled_output[7] == 0) ? 2 : 1;
155  unlock_cmd.motor9 = (_scaled_output[8] == 0) ? 2 : 1;
156  unlock_cmd.motor10 = (_scaled_output[9] == 0) ? 2 : 1;
157  unlock_cmd.motor11 = (_scaled_output[10] == 0) ? 2 : 1;
158  unlock_cmd.motor12 = (_scaled_output[11] == 0) ? 2 : 1;
159 
160  // prepare command to spin motors in bank1
161  mot_rot_cmd1.motor1 = htobe16(_scaled_output[0]);
162  mot_rot_cmd1.motor2 = htobe16(_scaled_output[1]);
163  mot_rot_cmd1.motor3 = htobe16(_scaled_output[2]);
164  mot_rot_cmd1.motor4 = htobe16(_scaled_output[3]);
165 
166  // prepare message to spin motors in bank2
167  mot_rot_cmd2.motor1 = htobe16(_scaled_output[4]);
168  mot_rot_cmd2.motor2 = htobe16(_scaled_output[5]);
169  mot_rot_cmd2.motor3 = htobe16(_scaled_output[6]);
170  mot_rot_cmd2.motor4 = htobe16(_scaled_output[7]);
171 
172  // prepare message to spin motors in bank3
173  mot_rot_cmd3.motor1 = htobe16(_scaled_output[8]);
174  mot_rot_cmd3.motor2 = htobe16(_scaled_output[9]);
175  mot_rot_cmd3.motor3 = htobe16(_scaled_output[10]);
176  mot_rot_cmd3.motor4 = htobe16(_scaled_output[11]);
177 
178  // copy update time
179  update_count_buffered = update_count;
180  }
181  unlock_frame = {(uint8_t)COMMAND_LOCK, unlock_cmd.data, sizeof(unlock_cmd.data)};
182  mot_rot_frame1 = {((uint8_t)COMMAND_MOTOR1 & uavcan::CanFrame::MaskStdID), mot_rot_cmd1.data, sizeof(mot_rot_cmd1.data)};
183  mot_rot_frame2 = {((uint8_t)COMMAND_MOTOR2 & uavcan::CanFrame::MaskStdID), mot_rot_cmd2.data, sizeof(mot_rot_cmd2.data)};
184  mot_rot_frame3 = {((uint8_t)COMMAND_MOTOR3 & uavcan::CanFrame::MaskStdID), mot_rot_cmd3.data, sizeof(mot_rot_cmd3.data)};
185 
186  // advance to next stage
187  send_stage++;
188  }
189 
190  // send unlock command
191  if (send_stage == 1) {
192  timeout = uavcan::MonotonicTime::fromUSec(AP_HAL::micros64() + timeout_us);
193  if (!write_frame(unlock_frame, timeout)) {
194  continue;
195  }
196  send_stage++;
197  }
198 
199  // send output to motor bank3
200  if (send_stage == 2) {
201  timeout = uavcan::MonotonicTime::fromUSec(AP_HAL::micros64() + timeout_us);
202  if (!write_frame(mot_rot_frame3, timeout)) {
203  continue;
204  }
205  send_stage++;
206  }
207 
208  // send output to motor bank2
209  if (send_stage == 3) {
210  timeout = uavcan::MonotonicTime::fromUSec(AP_HAL::micros64() + timeout_us);
211  if (!write_frame(mot_rot_frame2, timeout)) {
212  continue;
213  }
214  send_stage++;
215  }
216 
217  // send output to motor bank1
218  if (send_stage == 4) {
219  timeout = uavcan::MonotonicTime::fromUSec(AP_HAL::micros64() + timeout_us);
220  if (!write_frame(mot_rot_frame1, timeout)) {
221  continue;
222  }
223  send_stage++;
224  }
225 
226  // check if we should request update from ESCs
227  if (send_stage == 5) {
228  uint32_t now_ms = AP_HAL::millis();
229  uint32_t diff_ms = now_ms - _telemetry_req_ms;
230 
231  // check if 100ms has passed since last update request
232  if (diff_ms >= TOSHIBA_CAN_ESC_UPDATE_MS) {
233  // set _telem_req_ms to time we ideally should have requested update
234  if (diff_ms >= 2 * TOSHIBA_CAN_ESC_UPDATE_MS) {
235  _telemetry_req_ms = now_ms;
236  } else {
237  _telemetry_req_ms += TOSHIBA_CAN_ESC_UPDATE_MS;
238  }
239 
240  // prepare command to request data1 (rpm and voltage) from all ESCs
241  motor_request_data_cmd_t request_data_cmd = {};
242  request_data_cmd.motor1 = 1;
243  request_data_cmd.motor2 = 1;
244  request_data_cmd.motor3 = 1;
245  request_data_cmd.motor4 = 1;
246  request_data_cmd.motor5 = 1;
247  request_data_cmd.motor6 = 1;
248  request_data_cmd.motor7 = 1;
249  request_data_cmd.motor8 = 1;
250  request_data_cmd.motor9 = 1;
251  request_data_cmd.motor10 = 1;
252  request_data_cmd.motor11 = 1;
253  request_data_cmd.motor12 = 1;
254  uavcan::CanFrame request_data_frame;
255  request_data_frame = {(uint8_t)COMMAND_REQUEST_DATA, request_data_cmd.data, sizeof(request_data_cmd.data)};
256 
257  // send request data command
258  timeout = uavcan::MonotonicTime::fromUSec(AP_HAL::micros64() + timeout_us);
259  if (!write_frame(request_data_frame, timeout)) {
260  continue;
261  }
262 
263  // increment count to request temperature
264  _telemetry_temp_req_counter++;
265  }
266 
267  send_stage++;
268  }
269 
270  // check if we should request temperature from ESCs
271  if (send_stage == 6) {
272  if (_telemetry_temp_req_counter > 10) {
273  _telemetry_temp_req_counter = 0;
274 
275  // prepare command to request data2 (temperature) from all ESCs
276  motor_request_data_cmd_t request_data_cmd = {};
277  request_data_cmd.motor1 = 2;
278  request_data_cmd.motor2 = 2;
279  request_data_cmd.motor3 = 2;
280  request_data_cmd.motor4 = 2;
281  request_data_cmd.motor5 = 2;
282  request_data_cmd.motor6 = 2;
283  request_data_cmd.motor7 = 2;
284  request_data_cmd.motor8 = 2;
285  request_data_cmd.motor9 = 2;
286  request_data_cmd.motor10 = 2;
287  request_data_cmd.motor11 = 2;
288  request_data_cmd.motor12 = 2;
289  uavcan::CanFrame request_data_frame;
290  request_data_frame = {(uint8_t)COMMAND_REQUEST_DATA, request_data_cmd.data, sizeof(request_data_cmd.data)};
291 
292  // send request data command
293  timeout = uavcan::MonotonicTime::fromUSec(AP_HAL::micros64() + timeout_us);
294  if (!write_frame(request_data_frame, timeout)) {
295  continue;
296  }
297  }
298 
299  send_stage++;
300  }
301 
302  // check for replies from ESCs
303  if (send_stage == 7) {
304  uavcan::CanFrame recv_frame;
305  while (read_frame(recv_frame, timeout)) {
306  // decode rpm and voltage data
307  if ((recv_frame.id >= MOTOR_DATA1) && (recv_frame.id <= MOTOR_DATA1 + 12)) {
308  // copy contents to our structure
309  motor_reply_data1_t reply_data;
310  memcpy(reply_data.data, recv_frame.data, sizeof(reply_data.data));
311  // store response in telemetry array
312  uint8_t esc_id = recv_frame.id - MOTOR_DATA1;
313  if (esc_id < TOSHIBACAN_MAX_NUM_ESCS) {
314  WITH_SEMAPHORE(_telem_sem);
315  _telemetry[esc_id].rpm = be16toh(reply_data.rpm);
316  _telemetry[esc_id].millivolts = be16toh(reply_data.millivolts);
317  _telemetry[esc_id].count++;
318  _telemetry[esc_id].new_data = true;
319  _esc_present_bitmask |= ((uint32_t)1 << esc_id);
320  }
321  }
322 
323  // decode temperature data
324  if ((recv_frame.id >= MOTOR_DATA2) && (recv_frame.id <= MOTOR_DATA2 + 12)) {
325  // motor data2 data format is 8 bytes (64 bits)
326  // 10 bits: U temperature
327  // 10 bits: V temperature
328  // 10 bits: W temperature
329  // 10 bits: motor temperature
330  // remaining 24 bits: reserved
331  uint16_t u_temp = ((uint16_t)recv_frame.data[0] << 2) | ((uint16_t)recv_frame.data[1] >> 6);
332  uint16_t v_temp = (((uint16_t)recv_frame.data[1] & (uint16_t)0x3F) << 4) | (((uint16_t)recv_frame.data[2] & (uint16_t)0xF0) >> 4);
333  uint16_t w_temp = (((uint16_t)recv_frame.data[2] & (uint16_t)0x0F) << 6) | (((uint16_t)recv_frame.data[3] & (uint16_t)0xFC) >> 2);
334  uint16_t temp_max = MAX(u_temp, MAX(v_temp, w_temp));
335 
336  // store repose in telemetry array
337  uint8_t esc_id = recv_frame.id - MOTOR_DATA2;
338  if (esc_id < TOSHIBACAN_MAX_NUM_ESCS) {
339  WITH_SEMAPHORE(_telem_sem);
340  _telemetry[esc_id].temperature = temp_max < 20 ? 0 : temp_max / 5 - 20;
341  _esc_present_bitmask |= ((uint32_t)1 << esc_id);
342  }
343  }
344  }
345  }
346 
347  // success!
348  send_stage = 0;
349 
350  // record success so we don't send this frame again
351  update_count_sent = update_count_buffered;
352  }
353 }

355 // write frame on CAN bus
356 bool AP_ToshibaCAN::write_frame(uavcan::CanFrame &out_frame, uavcan::MonotonicTime timeout)
357 {
358  // wait for space in buffer to send command
359  uavcan::CanSelectMasks inout_mask;
360  do {
361  inout_mask.read = 0;
362  inout_mask.write = 1 << CAN_IFACE_INDEX;
363  _select_frames[CAN_IFACE_INDEX] = &out_frame;
364  _can_driver->select(inout_mask, _select_frames, timeout);
365 
366  // delay if no space is available to send
367  if (!inout_mask.write) {
368  hal.scheduler->delay_microseconds(50);
369  }
370  } while (!inout_mask.write);
371 
372  // send frame and return success
373  return (_can_driver->getIface(CAN_IFACE_INDEX)->send(out_frame, timeout, uavcan::CanIOFlagAbortOnError) == 1);
374 }

376 // read frame on CAN bus, returns true on success
377 bool AP_ToshibaCAN::read_frame(uavcan::CanFrame &recv_frame, uavcan::MonotonicTime timeout)
378 {
379  // wait for space in buffer to read
380  uavcan::CanSelectMasks inout_mask;
381  inout_mask.read = 1 << CAN_IFACE_INDEX;
382  inout_mask.write = 0;
383  _select_frames[CAN_IFACE_INDEX] = &recv_frame;
384  _can_driver->select(inout_mask, _select_frames, timeout);
385 
386  // return false if no data is available to read
387  if (!inout_mask.read) {
388  return false;
389  }
390  uavcan::MonotonicTime time;
391  uavcan::UtcTime utc_time;
392  uavcan::CanIOFlags flags {};
393 
394  // read frame and return success
395  return (_can_driver->getIface(CAN_IFACE_INDEX)->receive(recv_frame, time, utc_time, flags) == 1);
396 }

398 // called from SRV_Channels
399 void AP_ToshibaCAN::update()
400 {
401  // take semaphore and update outputs
402  {
403  WITH_SEMAPHORE(_rc_out_sem);
404  for (uint8_t i = 0; i < MIN(TOSHIBACAN_MAX_NUM_ESCS, 16); i++) {
405  SRV_Channel *c = SRV_Channels::srv_channel(i);
406  if (c == nullptr) {
407  _scaled_output[i] = 0;
408  } else {
409  uint16_t pwm_out = c->get_output_pwm();
410  if (pwm_out <= 1000) {
411  _scaled_output[i] = 0;
412  } else if (pwm_out >= 2000) {
413  _scaled_output[i] = TOSHIBACAN_OUTPUT_MAX;
414  } else {
415  _scaled_output[i] = TOSHIBACAN_OUTPUT_MIN + (pwm_out - 1000) * 0.001f * (TOSHIBACAN_OUTPUT_MAX - TOSHIBACAN_OUTPUT_MIN);
416  }
417  }
418  }
419  update_count++;
420  }
421 
422  // log ESCs telemetry info
423  AP_Logger *logger = AP_Logger::get_singleton();
424  if (logger && logger->logging_enabled()) {
425  WITH_SEMAPHORE(_telem_sem);
426 
427  // log if any new data received. Logging only supports up to 8 ESCs
428  uint64_t time_us = AP_HAL::micros64();
429  for (uint8_t i = 0; i < MIN(TOSHIBACAN_MAX_NUM_ESCS, 8); i++) {
430  if (_telemetry[i].new_data) {
431  logger->Write_ESC(i, time_us,
432  _telemetry[i].rpm * 100U,
433  _telemetry[i].millivolts * 0.1f,
434  0,
435  _telemetry[i].temperature * 100.0f,
436  0);
437  _telemetry[i].new_data = false;
438  }
439  }
440  }

443 // send ESC telemetry messages over MAVLink
444 void AP_ToshibaCAN::send_esc_telemetry_mavlink(uint8_t mav_chan)
445 {
446  // compile time check this method handles the correct number of motors
447  static_assert(TOSHIBACAN_MAX_NUM_ESCS == 12, "update AP_ToshibaCAN::send_esc_telemetry_mavlink only handles 12 motors");
448 
449  // return immediately if no ESCs have been found
450  if (_esc_present_bitmask == 0) {
451  return;
452  }
453 
454  // return if no space in output buffer to send mavlink messages
455  if (!HAVE_PAYLOAD_SPACE((mavlink_channel_t)mav_chan, ESC_TELEMETRY_1_TO_4)) {
456  return;
457  }
458 
459  // output telemetry messages
460  {
461  // take semaphore to access telemetry data
462  WITH_SEMAPHORE(_telem_sem);
463 
464  // loop through 3 groups of 4 ESCs
465  for (uint8_t i = 0; i < 3; i++) {
466 
467  // skip this group of ESCs if no data to send
468  if ((_esc_present_bitmask & ((uint32_t)0x0F << i*4)) == 0) {
469  continue;
470  }
471 
472  // arrays to hold output
473  uint8_t temperature[4] {};
474  uint16_t voltage[4] {};
475  uint16_t rpm[4] {};
476  uint16_t count[4] {};
477  uint16_t nosup[4] {}; // single empty array for unsupported current and current_tot
478 
479  // fill in output arrays
480  for (uint8_t j = 0; j < 4; j++) {
481  uint8_t esc_id = i * 4 + j;
482  temperature[j] = _telemetry[esc_id].temperature;
483  voltage[j] = _telemetry[esc_id].millivolts * 0.1f;
484  rpm[j] = _telemetry[esc_id].rpm;
485  count[j] = _telemetry[esc_id].count;
486  }
487 
488  // send messages
489  switch (i) {
490  case 0:
491  mavlink_msg_esc_telemetry_1_to_4_send((mavlink_channel_t)mav_chan, temperature, voltage, nosup, nosup, rpm, count);
492  break;
493  case 1:
494  mavlink_msg_esc_telemetry_5_to_8_send((mavlink_channel_t)mav_chan, temperature, voltage, nosup, nosup, rpm, count);
495  break;
496  case 2:
497  mavlink_msg_esc_telemetry_9_to_12_send((mavlink_channel_t)mav_chan, temperature, voltage, nosup, nosup, rpm, count);
498  break;
499  default:
500  break;
501  }
502  }
503  }
