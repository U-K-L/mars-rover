42 // Process bytes available from the stream
43 //
44 // The stream is assumed to contain only messages we recognise. If it
45 // contains other messages, and those messages contain the preamble
46 // bytes, it is possible for this code to fail to synchronise to the
47 // stream immediately. Without buffering the entire message and
48 // re-processing it from the top, this is unavoidable. The parser
49 // attempts to avoid this when possible.
50 //
51 bool
52 AP_GPS_SIRF::read(void)
53 {
54  uint8_t data;
55  int16_t numc;
56  bool parsed = false;
57 
58  numc = port->available();
59  while(numc--) {
60 
61  // read the next byte
62  data = port->read();
63 
64  switch(_step) {
65 
66  // Message preamble detection
67  //
68  // If we fail to match any of the expected bytes, we reset
69  // the state machine and re-consider the failed byte as
70  // the first byte of the preamble. This improves our
71  // chances of recovering from a mismatch and makes it less
72  // likely that we will be fooled by the preamble appearing
73  // as data in some other message.
74  //
75  case 1:
76  if (PREAMBLE2 == data) {
77  _step++;
78  break;
79  }
80  _step = 0;
81  FALLTHROUGH;
82  case 0:
83  if(PREAMBLE1 == data)
84  _step++;
85  break;
86 
87  // Message length
88  //
89  // We always collect the length so that we can avoid being
90  // fooled by preamble bytes in messages.
91  //
92  case 2:
93  _step++;
94  _payload_length = (uint16_t)data << 8;
95  break;
96  case 3:
97  _step++;
98  _payload_length |= data;
99  _payload_counter = 0;
100  _checksum = 0;
101  break;
102 
103  // Message header processing
104  //
105  // We sniff the message ID to determine whether we are going
106  // to gather the message bytes or just discard them.
107  //
108  case 4:
109  _step++;
110  _accumulate(data);
111  _payload_length--;
112  _gather = false;
113  switch(data) {
114  case MSG_GEONAV:
115  if (_payload_length == sizeof(sirf_geonav)) {
116  _gather = true;
117  _msg_id = data;
118  }
119  break;
120  }
121  break;
122 
123  // Receive message data
124  //
125  // Note that we are effectively guaranteed by the protocol
126  // that the checksum and postamble cannot be mistaken for
127  // the preamble, so if we are discarding bytes in this
128  // message when the payload is done we return directly
129  // to the preamble detector rather than bothering with
130  // the checksum logic.
131  //
132  case 5:
133  if (_gather) { // gather data if requested
134  _accumulate(data);
135  _buffer[_payload_counter] = data;
136  if (++_payload_counter == _payload_length)
137  _step++;
138  } else {
139  if (++_payload_counter == _payload_length)
140  _step = 0;
141  }
142  break;
143 
144  // Checksum and message processing
145  //
146  case 6:
147  _step++;
148  if ((_checksum >> 8) != data) {
149  _step = 0;
150  }
151  break;
152  case 7:
153  _step = 0;
154  if ((_checksum & 0xff) != data) {
155  break;
156  }
157  if (_gather) {
158  parsed = _parse_gps(); // Parse the new GPS packet
159  }
160  }
161  }
162  return(parsed);
163 }

165 bool
166 AP_GPS_SIRF::_parse_gps(void)
167 {
168  switch(_msg_id) {
169  case MSG_GEONAV:
170  //time = _swapl(&_buffer.nav.time);
171  // parse fix type
172  if (_buffer.nav.fix_invalid) {
173  state.status = AP_GPS::NO_FIX;
174  }else if ((_buffer.nav.fix_type & FIX_MASK) == FIX_3D) {
175  state.status = AP_GPS::GPS_OK_FIX_3D;
176  }else{
177  state.status = AP_GPS::GPS_OK_FIX_2D;
178  }
179  state.location.lat = swap_int32(_buffer.nav.latitude);
180  state.location.lng = swap_int32(_buffer.nav.longitude);
181  state.location.alt = swap_int32(_buffer.nav.altitude_msl);
182  state.ground_speed = swap_int32(_buffer.nav.ground_speed)*0.01f;
183  state.ground_course = wrap_360(swap_int16(_buffer.nav.ground_course)*0.01f);
184  state.num_sats = _buffer.nav.satellites;
185  fill_3d_velocity();
186  return true;
187  }
188  return false;
189 }

199 /*
200  detect a SIRF GPS
201  */
202 bool AP_GPS_SIRF::_detect(struct SIRF_detect_state &state, uint8_t data)
203 {
204  switch (state.step) {
205  case 1:
206  if (PREAMBLE2 == data) {
207  state.step++;
208  break;
209  }
210  state.step = 0;
211  FALLTHROUGH;
212  case 0:
213  state.payload_length = state.payload_counter = state.checksum = 0;
214  if (PREAMBLE1 == data)
215  state.step++;
216  break;
217  case 2:
218  state.step++;
219  if (data != 0) {
220  // only look for short messages
221  state.step = 0;
222  }
223  break;
224  case 3:
225  state.step++;
226  state.payload_length = data;
227  break;
228  case 4:
229  state.checksum = (state.checksum + data) & 0x7fff;
230  if (++state.payload_counter == state.payload_length) {
231  state.step++;
232  }
233  break;
234  case 5:
235  state.step++;
236  if ((state.checksum >> 8) != data) {
237  state.step = 0;
238  }
239  break;
240  case 6:
241  state.step = 0;
242  if ((state.checksum & 0xff) == data) {
243  return true;
244  }
245  }
246  return false;
247 }
