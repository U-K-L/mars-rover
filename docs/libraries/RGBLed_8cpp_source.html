104 // _scheduled_update - updates _red, _green, _blue according to notify flags
105 uint32_t RGBLed::get_colour_sequence(void) const
106 {
107  // initialising pattern
108  if (AP_Notify::flags.initialising) {
109  return sequence_initialising;
110  }
111 
112  // save trim and esc calibration pattern
113  if (AP_Notify::flags.save_trim || AP_Notify::flags.esc_calibration) {
114  return sequence_trim_or_esc;
115  }
116 
117  // radio and battery failsafe patter: flash yellow
118  // gps failsafe pattern : flashing yellow and blue
119  // ekf_bad pattern : flashing yellow and red
120  if (AP_Notify::flags.failsafe_radio ||
121  AP_Notify::flags.failsafe_battery ||
122  AP_Notify::flags.ekf_bad ||
123  AP_Notify::flags.gps_glitching ||
124  AP_Notify::flags.leak_detected) {
125 
126  if (AP_Notify::flags.leak_detected) {
127  // purple if leak detected
128  return sequence_failsafe_leak;
129  } else if (AP_Notify::flags.ekf_bad) {
130  // red on if ekf bad
131  return sequence_failsafe_ekf;
132  } else if (AP_Notify::flags.gps_glitching) {
133  // blue on gps glitch
134  return sequence_failsafe_gps_glitching;
135  }
136  // all off for radio or battery failsafe
137  return sequence_failsafe_radio_or_battery;
138  }
139 
140  // solid green or blue if armed
141  if (AP_Notify::flags.armed) {
142  // solid green if armed with GPS 3d lock
143  if (AP_Notify::flags.gps_status >= AP_GPS::GPS_OK_FIX_3D) {
144  return sequence_armed;
145  }
146  // solid blue if armed with no GPS lock
147  return sequence_armed_nogps;
148  }
149 
150  // double flash yellow if failing pre-arm checks
151  if (!AP_Notify::flags.pre_arm_check) {
152  return sequence_prearm_failing;
153  }
154  if (AP_Notify::flags.gps_status >= AP_GPS::GPS_OK_FIX_3D_DGPS && AP_Notify::flags.pre_arm_gps_check) {
155  return sequence_disarmed_good_dgps;
156  }
157 
158  if (AP_Notify::flags.gps_status >= AP_GPS::GPS_OK_FIX_3D && AP_Notify::flags.pre_arm_gps_check) {
159  return sequence_disarmed_good_gps;
160  }
161 
162  return sequence_disarmed_bad_gps;
163 }

165 // update - updates led according to timed_updated. Should be called
166 // at 50Hz
167 void RGBLed::update()
168 {
169  uint32_t current_colour_sequence = 0;
170 
171  switch (rgb_source()) {
172  case mavlink:
173  update_override();
174  return; // note this is a return not a break!
175  case standard:
176  current_colour_sequence = get_colour_sequence();
177  break;
178  case obc:
179  current_colour_sequence = get_colour_sequence_obc();
180  break;
181  }
182 
183  const uint8_t brightness = get_brightness();
184 
185  uint8_t step = (AP_HAL::millis()/100) % 10;
186 
187  // ensure we can't skip a step even with awful timing
188  if (step != last_step) {
189  step = (last_step+1) % 10;
190  last_step = step;
191  }
192 
193  const uint8_t colour = (current_colour_sequence >> (step*3)) & 7;
194 
195  _red_des = (colour & RED) ? brightness : 0;
196  _green_des = (colour & GREEN) ? brightness : 0;
197  _blue_des = (colour & BLUE) ? brightness : 0;
198 
199  set_rgb(_red_des, _green_des, _blue_des);
200 }

202 /*
203  handle LED control, only used when LED_OVERRIDE=1
204 */
205 void RGBLed::handle_led_control(mavlink_message_t *msg)
206 {
207  if (rgb_source() != mavlink) {
208  // ignore LED_CONTROL commands if not in LED_OVERRIDE mode
209  return;
210  }
211 
212  // decode mavlink message
213  mavlink_led_control_t packet;
214  mavlink_msg_led_control_decode(msg, &packet);
215 
216  _led_override.start_ms = AP_HAL::millis();
217 
218  switch (packet.custom_len) {
219  case 3:
220  _led_override.rate_hz = 0;
221  _led_override.r = packet.custom_bytes[0];
222  _led_override.g = packet.custom_bytes[1];
223  _led_override.b = packet.custom_bytes[2];
224  break;
225  case 4:
226  _led_override.rate_hz = packet.custom_bytes[3];
227  _led_override.r = packet.custom_bytes[0];
228  _led_override.g = packet.custom_bytes[1];
229  _led_override.b = packet.custom_bytes[2];
230  break;
231  default:
232  // not understood
233  break;
234  }
235 }

237 /*
238  update LED when in override mode
239  */
240 void RGBLed::update_override(void)
241 {
242  if (_led_override.rate_hz == 0) {
243  // solid colour
244  _set_rgb(_led_override.r, _led_override.g, _led_override.b);
245  return;
246  }
247  // blinking
248  uint32_t ms_per_cycle = 1000 / _led_override.rate_hz;
249  uint32_t cycle = (AP_HAL::millis() - _led_override.start_ms) % ms_per_cycle;
250  if (cycle > ms_per_cycle / 2) {
251  // on
252  _set_rgb(_led_override.r, _led_override.g, _led_override.b);
253  } else {
254  _set_rgb(0, 0, 0);
255  }
256 }
