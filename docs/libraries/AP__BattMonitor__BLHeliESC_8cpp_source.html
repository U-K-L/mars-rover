42  uint32_t now = AP_HAL::millis();

44 
45  for (uint8_t i=0; i<AP_BLHELI_MAX_ESCS; i++) {

50 
51  // accumulate consumed_sum regardless of age, to cope with ESC

53  consumed_sum += td.consumption;
54 
55  if (now - td.timestamp_ms > 1000) {
56  // don't use old data
57  continue;
58  }
59 
60  num_escs++;
61  voltage_sum += td.voltage;
62  current_sum += td.current;
63  temperature_sum += td.temperature;
64  if (td.timestamp_ms > highest_ms) {
65  highest_ms = td.timestamp_ms;
66  }
67  }
68 
69  if (num_escs > 0) {
70  _state.voltage = (voltage_sum / num_escs) * 0.01;
71  _state.temperature = temperature_sum / num_escs;
72  _state.healthy = true;
73  } else {
74  _state.voltage = 0;
75  _state.temperature = 0;
76  _state.healthy = false;

79  _state.consumed_mah = consumed_sum;
