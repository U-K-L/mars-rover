533  innovVelSumSq += sq(velInnov[i]);
534  varVelSum += varInnovVelPos[i];
535  }
536  // apply an innovation consistency threshold test, but don't fail if bad IMU data
537  // calculate the test ratio
538  velTestRatio = innovVelSumSq / (varVelSum * sq(MAX(0.01f * (float)frontend->_gpsVelInnovGate, 1.0f)));
539  // fail if the ratio is greater than 1
540  velHealth = ((velTestRatio < 1.0f) || badIMUdata);
541  // use velocity data if healthy, timed out, or in constant position mode
542  if (velHealth || velTimeout) {
543  velHealth = true;
544  // restart the timeout count
545  lastVelPassTime_ms = imuSampleTime_ms;
546  // If we are doing full aiding and velocity fusion times out, reset to the GPS velocity
547  if (PV_AidingMode == AID_ABSOLUTE && velTimeout) {
548  // reset the velocity to the GPS velocity
549  ResetVelocity();
550  // don't fuse GPS velocity data on this time step
551  fuseVelData = false;
552  // Reset the normalised innovation to avoid failing the bad fusion tests
553  velTestRatio = 0.0f;
554  }
555  } else {
556  velHealth = false;
557  }
558  }
559 
560  // test height measurements
561  if (fuseHgtData) {
562  // calculate height innovations
563  innovVelPos[5] = stateStruct.position.z - observation[5];
564  varInnovVelPos[5] = P[9][9] + R_OBS_DATA_CHECKS[5];
565  // calculate the innovation consistency test ratio
566  hgtTestRatio = sq(innovVelPos[5]) / (sq(MAX(0.01f * (float)frontend->_hgtInnovGate, 1.0f)) * varInnovVelPos[5]);
567  // fail if the ratio is > 1, but don't fail if bad IMU data
568  hgtHealth = ((hgtTestRatio < 1.0f) || badIMUdata);
569  // Fuse height data if healthy or timed out or in constant position mode
570  if (hgtHealth || hgtTimeout || (PV_AidingMode == AID_NONE && onGround)) {
571  // Calculate a filtered value to be used by pre-flight health checks
572  // We need to filter because wind gusts can generate significant baro noise and we want to be able to detect bias errors in the inertial solution
573  if (onGround) {
574  float dtBaro = (imuSampleTime_ms - lastHgtPassTime_ms)*1.0e-3f;
575  const float hgtInnovFiltTC = 2.0f;
576  float alpha = constrain_float(dtBaro/(dtBaro+hgtInnovFiltTC),0.0f,1.0f);
577  hgtInnovFiltState += (innovVelPos[5]-hgtInnovFiltState)*alpha;
578  } else {
579  hgtInnovFiltState = 0.0f;
580  }
581 
582  // if timed out, reset the height
583  if (hgtTimeout) {
584  ResetHeight();
585  }
586 
587  // If we have got this far then declare the height data as healthy and reset the timeout counter
588  hgtHealth = true;
589  lastHgtPassTime_ms = imuSampleTime_ms;
590  }
591  }
592 
593  // set range for sequential fusion of velocity and position measurements depending on which data is available and its health
594  if (fuseVelData && velHealth) {
595  fuseData[0] = true;
596  fuseData[1] = true;
597  if (useGpsVertVel) {
598  fuseData[2] = true;
599  }
600  }
601  if (fusePosData && posHealth) {
602  fuseData[3] = true;
603  fuseData[4] = true;
604  }
605  if (fuseHgtData && hgtHealth) {
606  fuseData[5] = true;
607  }
608 
609  // fuse measurements sequentially
610  for (obsIndex=0; obsIndex<=5; obsIndex++) {
611  if (fuseData[obsIndex]) {
612  stateIndex = 4 + obsIndex;
613  // calculate the measurement innovation, using states from a different time coordinate if fusing height data
614  // adjust scaling on GPS measurement noise variances if not enough satellites
615  if (obsIndex <= 2)
616  {
617  innovVelPos[obsIndex] = stateStruct.velocity[obsIndex] - observation[obsIndex];
618  R_OBS[obsIndex] *= sq(gpsNoiseScaler);
619  }
620  else if (obsIndex == 3 || obsIndex == 4) {
621  innovVelPos[obsIndex] = stateStruct.position[obsIndex-3] - observation[obsIndex];
622  R_OBS[obsIndex] *= sq(gpsNoiseScaler);
623  } else if (obsIndex == 5) {
624  innovVelPos[obsIndex] = stateStruct.position[obsIndex-3] - observation[obsIndex];
625  const float gndMaxBaroErr = 4.0f;
626  const float gndBaroInnovFloor = -0.5f;
627 
628  if(getTouchdownExpected() && activeHgtSource == HGT_SOURCE_BARO) {
629  // when a touchdown is expected, floor the barometer innovation at gndBaroInnovFloor
630  // constrain the correction between 0 and gndBaroInnovFloor+gndMaxBaroErr
631  // this function looks like this:
632  // |/
633  //---------|---------
634  // ____/|
635  // / |
636  // / |
637  innovVelPos[5] += constrain_float(-innovVelPos[5]+gndBaroInnovFloor, 0.0f, gndBaroInnovFloor+gndMaxBaroErr);
638  }
639  }
640 
641  // calculate the Kalman gain and calculate innovation variances
642  varInnovVelPos[obsIndex] = P[stateIndex][stateIndex] + R_OBS[obsIndex];
643  SK = 1.0f/varInnovVelPos[obsIndex];
644  for (uint8_t i= 0; i<=9; i++) {
645  Kfusion[i] = P[i][stateIndex]*SK;
646  }
647 
648  // inhibit delta angle bias state estmation by setting Kalman gains to zero
649  if (!inhibitDelAngBiasStates) {
650  for (uint8_t i = 10; i<=12; i++) {
651  Kfusion[i] = P[i][stateIndex]*SK;
652  }
653  } else {
654  // zero indexes 10 to 12 = 3*4 bytes
655  memset(&Kfusion[10], 0, 12);
656  }
657 
658  // inhibit delta velocity bias state estimation by setting Kalman gains to zero
659  if (!inhibitDelVelBiasStates) {
660  for (uint8_t i = 13; i<=15; i++) {
661  Kfusion[i] = P[i][stateIndex]*SK;
662  }
663  } else {
664  // zero indexes 13 to 15 = 3*4 bytes
665  memset(&Kfusion[13], 0, 12);
666  }
667 
668  // inhibit magnetic field state estimation by setting Kalman gains to zero
669  if (!inhibitMagStates) {
670  for (uint8_t i = 16; i<=21; i++) {
671  Kfusion[i] = P[i][stateIndex]*SK;
672  }
673  } else {
674  // zero indexes 16 to 21 = 6*4 bytes
675  memset(&Kfusion[16], 0, 24);
676  }
677 
678  // inhibit wind state estimation by setting Kalman gains to zero
679  if (!inhibitWindStates) {
680  Kfusion[22] = P[22][stateIndex]*SK;
681  Kfusion[23] = P[23][stateIndex]*SK;
682  } else {
683  // zero indexes 22 to 23 = 2*4 bytes
684  memset(&Kfusion[22], 0, 8);
685  }
686 
687  // update the covariance - take advantage of direct observation of a single state at index = stateIndex to reduce computations
688  // this is a numerically optimised implementation of standard equation P = (I - K*H)*P;
689  for (uint8_t i= 0; i<=stateIndexLim; i++) {
690  for (uint8_t j= 0; j<=stateIndexLim; j++)
691  {
692  KHP[i][j] = Kfusion[i] * P[stateIndex][j];
693  }
694  }
695  // Check that we are not going to drive any variances negative and skip the update if so
696  bool healthyFusion = true;
697  for (uint8_t i= 0; i<=stateIndexLim; i++) {
698  if (KHP[i][i] > P[i][i]) {
699  healthyFusion = false;
700  }
701  }
702  if (healthyFusion) {
703  // update the covariance matrix
704  for (uint8_t i= 0; i<=stateIndexLim; i++) {
705  for (uint8_t j= 0; j<=stateIndexLim; j++) {
706  P[i][j] = P[i][j] - KHP[i][j];
707  }
708  }
709 
710  // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
711  ForceSymmetry();
712  ConstrainVariances();
713 
714  // update states and renormalise the quaternions
715  for (uint8_t i = 0; i<=stateIndexLim; i++) {
716  statesArray[i] = statesArray[i] - Kfusion[i] * innovVelPos[obsIndex];
717  }
718  stateStruct.quat.normalize();
719 
720  // record good fusion status
721  if (obsIndex == 0) {
722  faultStatus.bad_nvel = false;
723  } else if (obsIndex == 1) {
724  faultStatus.bad_evel = false;
725  } else if (obsIndex == 2) {
726  faultStatus.bad_dvel = false;
727  } else if (obsIndex == 3) {
728  faultStatus.bad_npos = false;
729  } else if (obsIndex == 4) {
730  faultStatus.bad_epos = false;
731  } else if (obsIndex == 5) {
732  faultStatus.bad_dpos = false;
733  }
734  } else {
735  // record bad fusion status
736  if (obsIndex == 0) {
737  faultStatus.bad_nvel = true;
738  } else if (obsIndex == 1) {
739  faultStatus.bad_evel = true;
740  } else if (obsIndex == 2) {
741  faultStatus.bad_dvel = true;
742  } else if (obsIndex == 3) {
743  faultStatus.bad_npos = true;
744  } else if (obsIndex == 4) {
745  faultStatus.bad_epos = true;
746  } else if (obsIndex == 5) {
747  faultStatus.bad_dpos = true;
748  }
749  }
750  }
751  }
752  }
753 
754  // stop performance timer
755  hal.util->perf_end(_perf_FuseVelPosNED);
756 }

787 
788  // select height source
789  if (((frontend->_useRngSwHgt > 0) || (frontend->_altSource == 1)) && (imuSampleTime_ms - rngValidMeaTime_ms < 500)) {
790  if (frontend->_altSource == 1) {
791  // always use range finder
792  activeHgtSource = HGT_SOURCE_RNG;
793  } else {

795  float rangeMaxUse = 1e-4f * (float)frontend->_rng.max_distance_cm_orient(ROTATION_PITCH_270) * (float)frontend->_useRngSwHgt;
796  bool aboveUpperSwHgt = (terrainState - stateStruct.position.z) > rangeMaxUse;
797  bool belowLowerSwHgt = (terrainState - stateStruct.position.z) < 0.7f * rangeMaxUse;
798 
799  // If the terrain height is consistent and we are moving slowly, then it can be
800  // used as a height reference in combination with a range finder
801  // apply a hysteresis to the speed check to prevent rapid switching
802  bool dontTrustTerrain, trustTerrain;
803  if (filterStatus.flags.horiz_vel) {
804  // We can use the velocity estimate
805  float horizSpeed = norm(stateStruct.velocity.x, stateStruct.velocity.y);
806  dontTrustTerrain = (horizSpeed > frontend->_useRngSwSpd) || !terrainHgtStable;
807  float trust_spd_trigger = MAX((frontend->_useRngSwSpd - 1.0f),(frontend->_useRngSwSpd * 0.5f));
808  trustTerrain = (horizSpeed < trust_spd_trigger) && terrainHgtStable;
809  } else {
810  // We can't use the velocity estimate
811  dontTrustTerrain = !terrainHgtStable;
812  trustTerrain = terrainHgtStable;
813  }
814

816  * Switch between range finder and primary height source using height above ground and speed thresholds with
817  * hysteresis to avoid rapid switching. Using range finder for height requires a consistent terrain height
818  * which cannot be assumed if the vehicle is moving horizontally.
819  */

822  if (frontend->_altSource == 0) {
823  activeHgtSource = HGT_SOURCE_BARO;
824  } else if (frontend->_altSource == 2) {
825  activeHgtSource = HGT_SOURCE_GPS;
826  }
827  } else if (belowLowerSwHgt && trustTerrain && (activeHgtSource != HGT_SOURCE_RNG)) {
828  // reliable terrain and range finder so start using range finder height
829  activeHgtSource = HGT_SOURCE_RNG;
830  }
831  }
832  } else if ((frontend->_altSource == 2) && ((imuSampleTime_ms - lastTimeGpsReceived_ms) < 500) && validOrigin && gpsAccuracyGood) {
833  activeHgtSource = HGT_SOURCE_GPS;
834  } else if ((frontend->_altSource == 3) && validOrigin && rngBcnGoodToAlign) {
835  activeHgtSource = HGT_SOURCE_BCN;
836  } else {
837  activeHgtSource = HGT_SOURCE_BARO;
838  }

841  bool lostRngHgt = ((activeHgtSource == HGT_SOURCE_RNG) && ((imuSampleTime_ms - rngValidMeaTime_ms) > 500));
842  bool lostGpsHgt = ((activeHgtSource == HGT_SOURCE_GPS) && ((imuSampleTime_ms - lastTimeGpsReceived_ms) > 2000));
843  if (lostRngHgt || lostGpsHgt) {
844  activeHgtSource = HGT_SOURCE_BARO;
845  }
846 
847  // if there is new baro data to fuse, calculate filtered baro data required by other processes
848  if (baroDataToFuse) {
849  // calculate offset to baro data that enables us to switch to Baro height use during operation
850  if (activeHgtSource != HGT_SOURCE_BARO) {
851  calcFiltBaroOffset();
852  }
853  // filtered baro data used to provide a reference for takeoff
854  // it is is reset to last height measurement on disarming in performArmingChecks()
855  if (!getTakeoffExpected()) {
856  const float gndHgtFiltTC = 0.5f;
857  const float dtBaro = frontend->hgtAvg_ms*1.0e-3f;

861  }
862

864  // combined local NED position height and origin height remains consistent with the GPS altitude
865  // This also enables the GPS height to be used as a backup height source
866  if (gpsDataToFuse &&
867  (((frontend->_originHgtMode & (1 << 0)) && (activeHgtSource == HGT_SOURCE_BARO)) ||
868  ((frontend->_originHgtMode & (1 << 1)) && (activeHgtSource == HGT_SOURCE_RNG)))
869  ) {
870  correctEkfOriginHeight();
871  }

874  if (rangeDataToFuse && (activeHgtSource == HGT_SOURCE_RNG)) {

876  // correct for tilt using a flat earth model

880  // correct for terrain position relative to datum

886  // add uncertainty created by terrain gradient and vehicle tilt
887  posDownObsNoise += sq(rangeDataDelayed.rng * frontend->_terrGradMax) * MAX(0.0f , (1.0f - sq(prevTnb.c.z)));
888  } else {
889  // disable fusion if tilted too far
890  fuseHgtData = false;
891  }
892  } else if (gpsDataToFuse && (activeHgtSource == HGT_SOURCE_GPS)) {
893  // using GPS data
894  hgtMea = gpsDataDelayed.hgt;
895  // enable fusion
896  fuseHgtData = true;
897  // set the observation noise using receiver reported accuracy or the horizontal noise scaled for typical VDOP/HDOP ratio
898  if (gpsHgtAccuracy > 0.0f) {
899  posDownObsNoise = sq(constrain_float(gpsHgtAccuracy, 1.5f * frontend->_gpsHorizPosNoise, 100.0f));
900  } else {
901  posDownObsNoise = sq(constrain_float(1.5f * frontend->_gpsHorizPosNoise, 0.1f, 10.0f));
902  }
903  } else if (baroDataToFuse && (activeHgtSource == HGT_SOURCE_BARO)) {
904  // using Baro data

907  fuseHgtData = true;

913  }

917  hgtMea = MAX(hgtMea, meaHgtAtTakeOff);
918  }
919  } else {
920  fuseHgtData = false;
921  }

924  // set timeout period based on whether we have vertical GPS velocity available to constrain drift

927  hgtTimeout = true;
928  } else {
929  hgtTimeout = false;
930  }

954

962  // note - % operator overloaded for cross product

1004  float t27 = q1*ve*2.0f;

1008  float t28 = q3*vn*2.0f;

1031  float t46 = P[1][0]*t18;

1033  float t84 = P[6][0]*t11;
