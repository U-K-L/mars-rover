62 bool VideoIn::open_device(const char *device_path, uint32_t memtype)
63 {
64  struct v4l2_capability cap;
65  int ret;
66 
67  _fd = -1;
68  _buffers = nullptr;
69  _fd = open(device_path, O_RDWR|O_CLOEXEC);
70  _memtype = memtype;
71  if (_fd < 0) {
72  hal.console->printf("Error opening device %s: %s (%d).\n",
73  device_path,
74  strerror(errno), errno);
75  return false;
76  }
77 
78  memset(&cap, 0, sizeof cap);
79  ret = ioctl(_fd, VIDIOC_QUERYCAP, &cap);
80  if (ret < 0) {
81  hal.console->printf("Error querying caps\n");
82  return false;
83  }
84 
85  if (!(cap.capabilities & V4L2_CAP_VIDEO_CAPTURE)) {
86  hal.console->printf("Error opening device %s: is not a video capture device\n",
87  device_path);
88  close(_fd);
89  _fd = -1;
90  return false;
91  }
92 
93  return true;
94 }

96 bool VideoIn::allocate_buffers(uint32_t nbufs)
97 {
98  struct v4l2_requestbuffers rb;
99  struct v4l2_buffer buf;
100  struct buffer *buffers;
101  unsigned int i;
102  int ret;
103 
104  memset(&rb, 0, sizeof rb);
105  rb.count = nbufs;
106  rb.type = (v4l2_buf_type) V4L2_CAP_VIDEO_CAPTURE;
107  rb.memory = (v4l2_memory) _memtype;
108 
109  ret = ioctl(_fd, VIDIOC_REQBUFS, &rb);
110  if (ret < 0) {
111  printf("Unable to request buffers: %s (%d).\n", strerror(errno), errno);
112  return ret;
113  }
114 
115  buffers = (struct buffer *)calloc(rb.count, sizeof buffers[0]);
116  if (buffers == nullptr) {
117  hal.console->printf("Unable to allocate buffers\n");
118  return false;
119  }
120 
121  for (i=0; i < rb.count; i++) {
122  memset(&buf, 0, sizeof(buf));
123  buf.index = i;
124  buf.type = rb.type;
125  buf.memory = rb.memory;
126  ret = ioctl(_fd, VIDIOC_QUERYBUF, &buf);
127  if (ret < 0) {
128  hal.console->printf("Unable to query buffer %u: %s(%d).\n", i,
129  strerror(errno), errno);
130  return false;
131  }
132 
133  switch (_memtype) {
134  case V4L2_MEMORY_MMAP:
135  buffers[i].mem = mmap(0, buf.length, PROT_READ | PROT_WRITE,
136  MAP_SHARED, _fd, buf.m.offset);
137  if (buffers[i].mem == MAP_FAILED) {
138  hal.console->printf("Unable to map buffer %u: %s (%d)\n", i,
139  strerror(errno), errno);
140  return false;
141  }
142  buffers[i].size = buf.length;
143  break;
144  case V4L2_MEMORY_USERPTR:
145  ret = posix_memalign(&buffers[i].mem, getpagesize(), buf.length);
146  if (ret < 0) {
147  hal.console->printf("Unable to allocate buffer %u (%d)\n", i,
148  ret);
149  return false;
150  }
151  buffers[i].size = buf.length;
152  break;
153  default:
154  return false;
155  }
156  }
157  _nbufs = rb.count;
158  _buffers = buffers;
159  return true;
160 }

176 bool VideoIn::set_format(uint32_t *width, uint32_t *height, uint32_t *format,
177  uint32_t *bytesperline, uint32_t *sizeimage)
178 {
179  struct v4l2_format fmt;
180  int ret;
181 
182  memset(&fmt, 0, sizeof fmt);
183  fmt.type = (v4l2_buf_type) V4L2_CAP_VIDEO_CAPTURE;
184  fmt.fmt.pix.width = *width;
185  fmt.fmt.pix.height = *height;
186  fmt.fmt.pix.pixelformat = *format;
187  fmt.fmt.pix.colorspace = V4L2_COLORSPACE_REC709;
188 
189  ret = ioctl(_fd, VIDIOC_S_FMT, &fmt);
190  if (ret < 0) {
191  hal.console->printf("VideoIn: unable to set format: %s (%d).\n",
192  strerror(errno), errno);
193  return false;
194  }
195 
196  /* warn if format different from the one that was supposed
197  * to be set
198  */
199  if ((fmt.fmt.pix.pixelformat != *format) ||
200  (fmt.fmt.pix.width != *width) ||
201  (fmt.fmt.pix.height != *height)) {
202  hal.console->printf("format set to (%08x)"
203  "%ux%u buffer size %u field : %d\n",
204  fmt.fmt.pix.pixelformat, fmt.fmt.pix.width,
205  fmt.fmt.pix.height, fmt.fmt.pix.sizeimage,
206  fmt.fmt.pix.field);
207  }
208 
209  *width = fmt.fmt.pix.width;
210  *height = fmt.fmt.pix.height;
211  *format = fmt.fmt.pix.pixelformat;
212  *bytesperline = fmt.fmt.pix.bytesperline;
213  *sizeimage = fmt.fmt.pix.sizeimage;
214 
215  return true;
216 }

252 void VideoIn::shrink_8bpp(uint8_t *buffer, uint8_t *new_buffer,
253  uint32_t width, uint32_t height, uint32_t left,
254  uint32_t selection_width, uint32_t top,
255  uint32_t selection_height, uint32_t fx, uint32_t fy)
256 {
257  uint32_t i, j, k, kk, px, block_x, block_y, block_position;
258  uint32_t out_width = selection_width / fx;
259  uint32_t out_height = selection_height / fy;
260  uint32_t width_per_fy = width * fy;
261  uint32_t fx_fy = fx * fy;
262  uint32_t width_sum, out_width_sum = 0;
263 
264  /* selection offset */
265  block_y = top * width;
266  block_position = left + block_y;
267 
268  for (i = 0; i < out_height; i++) {
269  block_x = left;
270  for (j = 0; j < out_width; j++) {
271  px = 0;
272 
273  width_sum = 0;
274  for(k = 0; k < fy; k++) {
275  for(kk = 0; kk < fx; kk++) {
276  px += buffer[block_position + kk + width_sum];
277  }
278  width_sum += width;
279  }
280 
281  new_buffer[j + out_width_sum] = px / (fx_fy);
282 
283  block_x += fx;
284  block_position = block_x + block_y;
285  }
286  block_y += width_per_fy;
287  out_width_sum += out_width;
288  }
289 }

361 bool VideoIn::_dequeue_frame(Frame &frame)
362 {
363  struct v4l2_buffer buf;
364  int ret;
365 
366  /* Dequeue a buffer. */
367  memset(&buf, 0, sizeof buf);
368  buf.type = (v4l2_buf_type) V4L2_CAP_VIDEO_CAPTURE;
369  buf.memory = (v4l2_memory) _memtype;
370  ret = ioctl(_fd, VIDIOC_DQBUF, &buf);
371  if (ret < 0) {
372  if (errno != EIO) {
373  hal.console->printf("Unable to dequeue buffer: %s (%d).\n",
374  strerror(errno), errno);
375  return false;
376  }
377  buf.type = (v4l2_buf_type) V4L2_CAP_VIDEO_CAPTURE;
378  buf.memory = (v4l2_memory) _memtype;
379  if (_memtype == V4L2_MEMORY_USERPTR) {
380  buf.m.userptr = (unsigned long)_buffers[buf.index].mem;
381  }
382  }
383 
384  frame.data = _buffers[buf.index].mem;
385  frame.buf_index = buf.index;
386  frame.timestamp = _timeval_to_us(buf.timestamp);
387  frame.sequence = buf.sequence;
388 
389  return true;
