160 {

164  (FLASH->SR2 & (FLASH_SR_BSY|FLASH_SR_QW|FLASH_SR_WBNE))) {

184 static void stm32_flash_unlock(void)
185 {
186  if (flash_keep_unlocked) {
187  return;
188  }
189  stm32_flash_wait_idle();
190 
191 #if defined(STM32H7)
192  if (FLASH->CR1 & FLASH_CR_LOCK) {
193  /* Unlock sequence */
194  FLASH->KEYR1 = FLASH_KEY1;
195  FLASH->KEYR1 = FLASH_KEY2;
196  }
197  if (FLASH->CR2 & FLASH_CR_LOCK) {
198  /* Unlock sequence */
199  FLASH->KEYR2 = FLASH_KEY1;
200  FLASH->KEYR2 = FLASH_KEY2;
201  }
202 #else
203  if (FLASH->CR & FLASH_CR_LOCK) {
204  /* Unlock sequence */
205  FLASH->KEYR = FLASH_KEY1;
206  FLASH->KEYR = FLASH_KEY2;
207  }
208 #endif
209 
210 #ifdef FLASH_ACR_DCEN
211  // disable the data cache - see stm32 errata 2.1.11
212  FLASH->ACR &= ~FLASH_ACR_DCEN;
213 #endif

217 {

219  return;
220  }
221 #if defined(STM32H7)

249 uint32_t stm32_flash_getpageaddr(uint32_t page)
250 {
251  if (page >= STM32_FLASH_NPAGES) {
252  return 0;

254 #if defined(STM32_FLASH_FIXED_PAGE_SIZE)
255  return STM32_FLASH_BASE + page * STM32_FLASH_FIXED_PAGE_SIZE * 1024;
256 #else

258  uint32_t address = STM32_FLASH_BASE;
259  uint8_t i;
260

262  flash_pageaddr[i] = address;
263  address += stm32_flash_getpagesize(i);
264  }
265  flash_pageaddr_initialised = true;
266  }

293 bool stm32_flash_ispageerased(uint32_t page)
294 {
295  uint32_t addr;
296  uint32_t count;
297 
298  if (page >= STM32_FLASH_NPAGES) {
299  return false;
300  }
301 
302  for (addr = stm32_flash_getpageaddr(page), count = stm32_flash_getpagesize(page);
303  count; count -= 4, addr += 4) {
304  uint32_t v = getreg32(addr);
305  if (v != 0xffffffff) {
306  return false;
307  }
308  }
309 
310  return true;
311 }

313 /*
314  erase a page
315  */
316 bool stm32_flash_erasepage(uint32_t page)
317 {
318  if (page >= STM32_FLASH_NPAGES) {
319  return false;
320  }
321 
322 #if STM32_FLASH_DISABLE_ISR
323  syssts_t sts = chSysGetStatusAndLockX();
324 #endif
325  stm32_flash_wait_idle();
326  stm32_flash_unlock();
327 
328  // clear any previous errors
329  stm32_flash_clear_errors();
330 
331 #if defined(STM32H7)
332  if (page < 8) {
333  // first bank
334  FLASH->SR1 = ~0;
335 
336  stm32_flash_wait_idle();
337 
338  uint32_t snb = page << 8;
339 
340  // use 32 bit operations
341  FLASH->CR1 = FLASH_CR_PSIZE_1 | snb | FLASH_CR_SER;
342  FLASH->CR1 |= FLASH_CR_START;
343  while (FLASH->SR1 & FLASH_SR_QW) ;
344  } else {
345  // second bank
346  FLASH->SR2 = ~0;
347 
348  stm32_flash_wait_idle();
349 
350  uint32_t snb = (page-8) << 8;
351 
352  // use 32 bit operations
353  FLASH->CR2 = FLASH_CR_PSIZE_1 | snb | FLASH_CR_SER;
354  FLASH->CR2 |= FLASH_CR_START;
355  while (FLASH->SR2 & FLASH_SR_QW) ;
356  }
357 #else
358  stm32_flash_wait_idle();
359 
360  // the snb mask is not contiguous, calculate the mask for the page
361  uint8_t snb = (((page % 12) << 3) | ((page / 12) << 7));
362 
363  // use 32 bit operations
364  FLASH->CR = FLASH_CR_PSIZE_1 | snb | FLASH_CR_SER;
365  FLASH->CR |= FLASH_CR_STRT;
366 #endif
367 
368  stm32_flash_wait_idle();
369 
370  cacheBufferInvalidate(stm32_flash_getpageaddr(page), stm32_flash_getpagesize(page));
371 
372  stm32_flash_lock();
373 #if STM32_FLASH_DISABLE_ISR
374  chSysRestoreStatusX(sts);
375 #endif
376  return stm32_flash_ispageerased(page);
377 }

382  the H7 needs special handling, and only writes 32 bytes at a time

384 static bool stm32h7_flash_write32(uint32_t addr, const void *buf)
385 {
386  volatile uint32_t *CR, *CCR, *SR;
387  if (addr - STM32_FLASH_BASE < 8 * STM32_FLASH_FIXED_PAGE_SIZE * 1024) {
388  CR = &FLASH->CR1;
389  CCR = &FLASH->CCR1;
390  SR = &FLASH->SR1;
391  } else {
392  CR = &FLASH->CR2;
393  CCR = &FLASH->CCR2;
394  SR = &FLASH->SR2;
395  }
396  stm32_flash_wait_idle();
397  *CCR = ~0;
398  *CR |= FLASH_CR_PG;
399 
400  const uint32_t *v = (const uint32_t *)buf;
401  uint8_t i;
402  for (i=0; i<8; i++) {
403  while (*SR & (FLASH_SR_BSY|FLASH_SR_QW)) ;
404  putreg32(*v, addr);
405  v++;
406  addr += 4;
407  }
408  __DSB();
409  stm32_flash_wait_idle();
410  *CCR = ~0;
411  *CR &= ~FLASH_CR_PG;
412  return true;
413 }

415 bool stm32_flash_write(uint32_t addr, const void *buf, uint32_t count)
416 {
417  uint8_t *b = (uint8_t *)buf;
418 
419  if ((count & 0x1F) || (addr & 0x1F)) {
420  // only allow 256 bit aligned writes
421  return false;
422  }
423  stm32_flash_unlock();
424  while (count >= 32) {
425  if (!stm32h7_flash_write32(addr, b)) {
426  return false;
427  }
428  // check contents
429  if (memcmp((void *)addr, b, 32) != 0) {
430  stm32_flash_lock();
431  return false;
432  }
433  addr += 32;
434  count -= 32;
435  b += 32;
436  }
437  stm32_flash_lock();
438  return true;
439 }

443 bool stm32_flash_write(uint32_t addr, const void *buf, uint32_t count)
444 {
445  uint8_t *b = (uint8_t *)buf;
446 
447  /* STM32 requires half-word access */
448  if (count & 1) {
449  return false;
450  }
451 
452  if ((addr+count) >= STM32_FLASH_BASE+STM32_FLASH_SIZE) {
453  return false;
454  }
455 
456  /* Get flash ready and begin flashing */
457 
458  if (!(RCC->CR & RCC_CR_HSION)) {
459  return false;
460  }
461 
462 #if STM32_FLASH_DISABLE_ISR
463  syssts_t sts = chSysGetStatusAndLockX();
464 #endif
465 
466  stm32_flash_unlock();
467 
468  // clear previous errors
469  stm32_flash_clear_errors();
470 
471  stm32_flash_wait_idle();
472 
473  // do as much as possible with 32 bit writes
474  while (count >= 4 && (addr & 3) == 0) {
475  FLASH->CR &= ~(FLASH_CR_PSIZE);
476  FLASH->CR |= FLASH_CR_PSIZE_1 | FLASH_CR_PG;
477  const uint32_t v1 = *(uint32_t *)b;
478 
479  putreg32(v1, addr);
480 
481  // ensure write ordering with cache
482  __DSB();
483 
484  stm32_flash_wait_idle();
485 
486  const uint32_t v2 = getreg32(addr);
487  if (v2 != v1) {
488  FLASH->CR &= ~(FLASH_CR_PG);
489  goto failed;
490  }
491 
492  count -= 4;
493  b += 4;
494  addr += 4;
495  }
496 
497  // the rest as 16 bit
498  while (count >= 2) {
499  FLASH->CR &= ~(FLASH_CR_PSIZE);
500  FLASH->CR |= FLASH_CR_PSIZE_0 | FLASH_CR_PG;
501 
502  putreg16(*(uint16_t *)b, addr);
503 
504  // ensure write ordering with cache
505  __DSB();
506 
507  stm32_flash_wait_idle();
508 
509  if (getreg16(addr) != *(uint16_t *)b) {
510  FLASH->CR &= ~(FLASH_CR_PG);
511  goto failed;
512  }
513 
514  count -= 2;
515  b += 2;
516  addr += 2;
517  }
518 
519  FLASH->CR &= ~(FLASH_CR_PG);
520 
521  stm32_flash_lock();
522 #if STM32_FLASH_DISABLE_ISR
523  chSysRestoreStatusX(sts);
524 #endif
525  return true;
526 
527 failed:
528  stm32_flash_lock();
529 #if STM32_FLASH_DISABLE_ISR
530  chSysRestoreStatusX(sts);
531 #endif
532  return false;
533 }
534 #endif // not STM32H7
