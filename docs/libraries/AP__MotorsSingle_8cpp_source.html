69 void AP_MotorsSingle::output_to_motors()
70 {
71  if (!_flags.initialised_ok) {
72  return;
73  }
74  switch (_spool_state) {
75  case SpoolState::SHUT_DOWN:
76  // sends minimum values out to the motors
77  rc_write_angle(AP_MOTORS_MOT_1, _roll_radio_passthrough * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
78  rc_write_angle(AP_MOTORS_MOT_2, _pitch_radio_passthrough * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
79  rc_write_angle(AP_MOTORS_MOT_3, -_roll_radio_passthrough * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
80  rc_write_angle(AP_MOTORS_MOT_4, -_pitch_radio_passthrough * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
81  rc_write(AP_MOTORS_MOT_5, output_to_pwm(0));
82  rc_write(AP_MOTORS_MOT_6, output_to_pwm(0));
83  break;
84  case SpoolState::GROUND_IDLE:
85  // sends output to motors when armed but not flying
86  for (uint8_t i=0; i<NUM_ACTUATORS; i++) {
87  rc_write_angle(AP_MOTORS_MOT_1+i, _spin_up_ratio * _actuator_out[i] * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
88  }
89  set_actuator_with_slew(_actuator[5], actuator_spin_up_to_ground_idle());
90  set_actuator_with_slew(_actuator[6], actuator_spin_up_to_ground_idle());
91  rc_write(AP_MOTORS_MOT_5, output_to_pwm(_actuator[5]));
92  rc_write(AP_MOTORS_MOT_6, output_to_pwm(_actuator[6]));
93  break;
94  case SpoolState::SPOOLING_UP:
95  case SpoolState::THROTTLE_UNLIMITED:
96  case SpoolState::SPOOLING_DOWN:
97  // set motor output based on thrust requests
98  for (uint8_t i=0; i<NUM_ACTUATORS; i++) {
99  rc_write_angle(AP_MOTORS_MOT_1+i, _actuator_out[i] * AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
100  }
101  set_actuator_with_slew(_actuator[5], thrust_to_actuator(_thrust_out));
102  set_actuator_with_slew(_actuator[6], thrust_to_actuator(_thrust_out));
103  rc_write(AP_MOTORS_MOT_5, output_to_pwm(_actuator[5]));
104  rc_write(AP_MOTORS_MOT_6, output_to_pwm(_actuator[6]));
105  break;
106  }
107 }

129 // sends commands to the motors
130 void AP_MotorsSingle::output_armed_stabilizing()
131 {
132  float roll_thrust; // roll thrust input value, +/- 1.0
133  float pitch_thrust; // pitch thrust input value, +/- 1.0
134  float yaw_thrust; // yaw thrust input value, +/- 1.0
135  float throttle_thrust; // throttle thrust input value, 0.0 - 1.0
136  float throttle_avg_max; // throttle thrust average maximum value, 0.0 - 1.0
137  float thrust_min_rpy; // the minimum throttle setting that will not limit the roll and pitch output
138  float thr_adj; // the difference between the pilot's desired throttle and throttle_thrust_best_rpy
139  float rp_scale = 1.0f; // this is used to scale the roll, pitch and yaw to fit within the motor limits
140  float actuator_allowed = 0.0f; // amount of yaw we can fit in
141  float actuator[NUM_ACTUATORS]; // combined roll, pitch and yaw thrusts for each actuator
142  float actuator_max = 0.0f; // maximum actuator value
143 
144  // apply voltage and air pressure compensation
145  const float compensation_gain = get_compensation_gain();
146  roll_thrust = _roll_in * compensation_gain;
147  pitch_thrust = _pitch_in * compensation_gain;
148  yaw_thrust = _yaw_in * compensation_gain;
149  throttle_thrust = get_throttle() * compensation_gain;
150  throttle_avg_max = _throttle_avg_max * compensation_gain;
151 
152  // sanity check throttle is above zero and below current limited throttle
153  if (throttle_thrust <= 0.0f) {
154  throttle_thrust = 0.0f;
155  limit.throttle_lower = true;
156  }
157  if (throttle_thrust >= _throttle_thrust_max) {
158  throttle_thrust = _throttle_thrust_max;
159  limit.throttle_upper = true;
160  }
161 
162  throttle_avg_max = constrain_float(throttle_avg_max, throttle_thrust, _throttle_thrust_max);
163 
164  float rp_thrust_max = MAX(fabsf(roll_thrust), fabsf(pitch_thrust));
165 
166  // calculate how much roll and pitch must be scaled to leave enough range for the minimum yaw
167  if (is_zero(rp_thrust_max)) {
168  rp_scale = 1.0f;
169  } else {
170  rp_scale = constrain_float((1.0f - MIN(fabsf(yaw_thrust), (float)_yaw_headroom/1000.0f)) / rp_thrust_max, 0.0f, 1.0f);
171  if (rp_scale < 1.0f) {
172  limit.roll_pitch = true;
173  }
174  }
175 
176  actuator_allowed = 1.0f - rp_scale * rp_thrust_max;
177  if (fabsf(yaw_thrust) > actuator_allowed) {
178  yaw_thrust = constrain_float(yaw_thrust, -actuator_allowed, actuator_allowed);
179  limit.yaw = true;
180  }
181 
182  // combine roll, pitch and yaw on each actuator
183  // front servo
184 
185  actuator[0] = rp_scale * roll_thrust - yaw_thrust;
186  // right servo
187  actuator[1] = rp_scale * pitch_thrust - yaw_thrust;
188  // rear servo
189  actuator[2] = -rp_scale * roll_thrust - yaw_thrust;
190  // left servo
191  actuator[3] = -rp_scale * pitch_thrust - yaw_thrust;
192 
193  // calculate the minimum thrust that doesn't limit the roll, pitch and yaw forces
194  thrust_min_rpy = MAX(MAX(fabsf(actuator[0]), fabsf(actuator[1])), MAX(fabsf(actuator[2]), fabsf(actuator[3])));
195 
196  thr_adj = throttle_thrust - throttle_avg_max;
197  if (thr_adj < (thrust_min_rpy - throttle_avg_max)) {
198  // Throttle can't be reduced to the desired level because this would mean roll or pitch control
199  // would not be able to reach the desired level because of lack of thrust.
200  thr_adj = MIN(thrust_min_rpy, throttle_avg_max) - throttle_avg_max;
201  }
202 
203  // calculate the throttle setting for the lift fan
204  _thrust_out = throttle_avg_max + thr_adj;
205 
206  if (is_zero(_thrust_out)) {
207  limit.roll_pitch = true;
208  limit.yaw = true;
209  }
210 
211  // limit thrust out for calculation of actuator gains
212  float thrust_out_actuator = constrain_float(MAX(_throttle_hover*0.5f,_thrust_out), 0.5f, 1.0f);
213 
214  // calculate the maximum allowed actuator output and maximum requested actuator output
215  for (uint8_t i=0; i<NUM_ACTUATORS; i++) {
216  if (actuator_max > fabsf(actuator[i])) {
217  actuator_max = fabsf(actuator[i]);
218  }
219  }
220  if (actuator_max > thrust_out_actuator && !is_zero(actuator_max)) {
221  // roll, pitch and yaw request can not be achieved at full servo defection
222  // reduce roll, pitch and yaw to reduce the requested defection to maximum
223  limit.roll_pitch = true;
224  limit.yaw = true;
225  rp_scale = thrust_out_actuator/actuator_max;
226  } else {
227  rp_scale = 1.0f;
228  }
229 
230  // force of a lifting surface is approximately equal to the angle of attack times the airflow velocity squared
231  // static thrust is proportional to the airflow velocity squared
232  // therefore the torque of the roll and pitch actuators should be approximately proportional to
233  // the angle of attack multiplied by the static thrust.
234  for (uint8_t i=0; i<NUM_ACTUATORS; i++) {
235  _actuator_out[i] = constrain_float(rp_scale*actuator[i]/thrust_out_actuator, -1.0f, 1.0f);
236  }
237 }

239 // output_test_seq - spin a motor at the pwm value specified
240 // motor_seq is the motor's sequence number from 1 to the number of motors on the frame
241 // pwm value is an actual pwm value that will be output, normally in the range of 1000 ~ 2000
242 void AP_MotorsSingle::output_test_seq(uint8_t motor_seq, int16_t pwm)
243 {
244  // exit immediately if not armed
245  if (!armed()) {
246  return;
247  }
248 
249  // output to motors and servos
250  switch (motor_seq) {
251  case 1:
252  // flap servo 1
253  rc_write(AP_MOTORS_MOT_1, pwm);
254  break;
255  case 2:
256  // flap servo 2
257  rc_write(AP_MOTORS_MOT_2, pwm);
258  break;
259  case 3:
260  // flap servo 3
261  rc_write(AP_MOTORS_MOT_3, pwm);
262  break;
263  case 4:
264  // flap servo 4
265  rc_write(AP_MOTORS_MOT_4, pwm);
266  break;
267  case 5:
268  // spin motor 1
269  rc_write(AP_MOTORS_MOT_5, pwm);
270  break;
271  case 6:
272  // spin motor 2
273  rc_write(AP_MOTORS_MOT_6, pwm);
274  break;
275  default:
276  // do nothing
277  break;
