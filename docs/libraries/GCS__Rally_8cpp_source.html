22 void GCS_MAVLINK::handle_rally_point(mavlink_message_t *msg)
23 {
24  AP_Rally *r = AP::rally();
25  if (r == nullptr) {
26  return;
27  }
28 
29  mavlink_rally_point_t packet;
30  mavlink_msg_rally_point_decode(msg, &packet);
31 
32  if (packet.idx >= r->get_rally_total() ||
33  packet.idx >= r->get_rally_max()) {
34  send_text(MAV_SEVERITY_WARNING,"Bad rally point ID");
35  return;
36  }
37 
38  if (packet.count != r->get_rally_total()) {
39  send_text(MAV_SEVERITY_WARNING,"Bad rally point count");
40  return;
41  }
42 
43  // sanity check location
44  if (!check_latlng(packet.lat, packet.lng)) {
45  return;
46  }
47 
48  RallyLocation rally_point;
49  rally_point.lat = packet.lat;
50  rally_point.lng = packet.lng;
51  rally_point.alt = packet.alt;
52  rally_point.break_alt = packet.break_alt;
53  rally_point.land_dir = packet.land_dir;
54  rally_point.flags = packet.flags;
55 
56  if (!r->set_rally_point_with_index(packet.idx, rally_point)) {
57  send_text(MAV_SEVERITY_CRITICAL, "Error setting rally point");
58  }
59 }

61 void GCS_MAVLINK::handle_rally_fetch_point(mavlink_message_t *msg)
62 {
63  AP_Rally *r = AP::rally();
64  if (r == nullptr) {
65  return;
66  }
67 
68  mavlink_rally_fetch_point_t packet;
69  mavlink_msg_rally_fetch_point_decode(msg, &packet);
70 
71  if (packet.idx > r->get_rally_total()) {
72  send_text(MAV_SEVERITY_WARNING, "Bad rally point ID");
73  return;
74  }
75 
76  RallyLocation rally_point;
77  if (!r->get_rally_point_with_index(packet.idx, rally_point)) {
78  send_text(MAV_SEVERITY_WARNING, "Failed to get rally point");
79  return;
80  }
81 
82  mavlink_msg_rally_point_send(chan, msg->sysid, msg->compid, packet.idx,
83  r->get_rally_total(), rally_point.lat, rally_point.lng,
84  rally_point.alt, rally_point.break_alt, rally_point.land_dir,
85  rally_point.flags);
86 }

88 void GCS_MAVLINK::handle_common_rally_message(mavlink_message_t *msg)
89 {
90  switch (msg->msgid) {
91  case MAVLINK_MSG_ID_RALLY_POINT:
92  handle_rally_point(msg);
93  break;
94  case MAVLINK_MSG_ID_RALLY_FETCH_POINT:
95  handle_rally_fetch_point(msg);
96  break;
97  default:
98 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
99  AP_HAL::panic("Unhandled common rally message");
100 #endif
101  break;
102  }
103 }

134 MAV_MISSION_RESULT MissionItemProtocol_Rally::convert_MISSION_ITEM_INT_to_RallyLocation(const mavlink_mission_item_int_t &cmd, RallyLocation &ret)
135 {
136  if (cmd.frame != MAV_FRAME_GLOBAL_RELATIVE_ALT_INT &&
137  cmd.frame != MAV_FRAME_GLOBAL_RELATIVE_ALT) {
138  return MAV_MISSION_UNSUPPORTED_FRAME;
139  }
140  if (!check_lat(cmd.x)) {
141  return MAV_MISSION_INVALID_PARAM5_X;
142  }
143  if (!check_lng(cmd.y)) {
144  return MAV_MISSION_INVALID_PARAM6_Y;
145  }
146  if (cmd.z < INT16_MIN || cmd.z > INT16_MAX) {
147  return MAV_MISSION_INVALID_PARAM7;
148  }
149  ret.lat = cmd.x;
150  ret.lng = cmd.y;
151  ret.alt = cmd.z;
152  return MAV_MISSION_ACCEPTED;
153 }
