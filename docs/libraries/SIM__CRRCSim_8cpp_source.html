44 /*
45  decode and send servos for heli
46 */
47 void CRRCSim::send_servos_heli(const struct sitl_input &input)
48 {
49  float swash1 = (input.servos[0]-1000) / 1000.0f;
50  float swash2 = (input.servos[1]-1000) / 1000.0f;
51  float swash3 = (input.servos[2]-1000) / 1000.0f;
52  float tail_rotor = (input.servos[3]-1000) / 1000.0f;
53  float rsc = (input.servos[7]-1000) / 1000.0f;
54 
55  float col_pitch = (swash1+swash2+swash3)/3.0 - 0.5f;
56  float roll_rate = (swash1 - swash2)/2;
57  float pitch_rate = -((swash1 + swash2)/2.0 - swash3)/2;
58  float yaw_rate = -(tail_rotor - 0.5);
59 
60  servo_packet pkt;
61  pkt.roll_rate = constrain_float(roll_rate, -0.5, 0.5);
62  pkt.pitch_rate = constrain_float(pitch_rate, -0.5, 0.5);
63  pkt.throttle = constrain_float(rsc, 0, 1);
64  pkt.yaw_rate = constrain_float(yaw_rate, -0.5, 0.5);
65  pkt.col_pitch = constrain_float(col_pitch, -0.5, 0.5);
66 
67  sock.sendto(&pkt, sizeof(pkt), "127.0.0.1", 9002);
68 }

70 /*
71  decode and send servos for fixed wing
72 */
73 void CRRCSim::send_servos_fixed_wing(const struct sitl_input &input)
74 {
75  float roll_rate = ((input.servos[0]-1000)/1000.0) - 0.5;
76  float pitch_rate = ((input.servos[1]-1000)/1000.0) - 0.5;
77  float yaw_rate = ((input.servos[3]-1000)/1000.0) - 0.5;
78  float throttle = ((input.servos[2]-1000)/1000.0);
79 
80  servo_packet pkt;
81  pkt.roll_rate = constrain_float(roll_rate, -0.5, 0.5);
82  pkt.pitch_rate = constrain_float(pitch_rate, -0.5, 0.5);
83  pkt.throttle = constrain_float(throttle, 0, 1);
84  pkt.yaw_rate = constrain_float(yaw_rate, -0.5, 0.5);
85  pkt.col_pitch = 0;
86 
87  sock.sendto(&pkt, sizeof(pkt), "127.0.0.1", 9002);
88 }

102 /*
103  receive an update from the FDM
104  This is a blocking function
105  */
106 void CRRCSim::recv_fdm(const struct sitl_input &input)
107 {
108  fdm_packet pkt;
109 
110  /*
111  we re-send the servo packet every 0.1 seconds until we get a
112  reply. This allows us to cope with some packet loss to the FDM
113  */
114  while (sock.recv(&pkt, sizeof(pkt), 100) != sizeof(pkt)) {
115  send_servos(input);
116  }
117 
118  accel_body = Vector3f(pkt.xAccel, pkt.yAccel, pkt.zAccel);
119  gyro = Vector3f(pkt.rollRate, pkt.pitchRate, pkt.yawRate);
120  velocity_ef = Vector3f(pkt.speedN, pkt.speedE, pkt.speedD);
121 
122  Location loc1, loc2;
123  loc2.lat = pkt.latitude * 1.0e7;
124  loc2.lng = pkt.longitude * 1.0e7;
125  const Vector2f posdelta = loc1.get_distance_NE(loc2);
126  position.x = posdelta.x;
127  position.y = posdelta.y;
128  position.z = -pkt.altitude;
129 
130  airspeed = pkt.airspeed;
131  airspeed_pitot = pkt.airspeed;
132 
133  dcm.from_euler(pkt.roll, pkt.pitch, pkt.yaw);
134 
135  // auto-adjust to crrcsim frame rate
136  double deltat = pkt.timestamp - last_timestamp;
137  time_now_us += deltat * 1.0e6;
138 
139  if (deltat < 0.01 && deltat > 0) {
140  adjust_frame_time(1.0/deltat);
141  }
142  last_timestamp = pkt.timestamp;
143 }
