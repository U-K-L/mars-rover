142 bool AP_Compass_QMC5883L::_check_whoami()
143 {
144  uint8_t whoami;
145  //Affected by other devices,must read registers 0x00 once or reset,after can read the ID registers reliably
146  _dev->read_registers(0x00,&whoami,1);
147  if (!_dev->read_registers(0x0C, &whoami,1)||
148  whoami != 0x01){
149  return false;
150  }
151  if (!_dev->read_registers(QMC5883L_REG_ID, &whoami,1)||
152  whoami != QMC5883_ID_VAL){
153  return false;
154  }
155  return true;
156 }

158 void AP_Compass_QMC5883L::timer()
159 {
160  struct PACKED {
161  le16_t rx;
162  le16_t ry;
163  le16_t rz;
164  } buffer;
165 
166  const float range_scale = 1000.0f / 3000.0f;
167 
168  uint8_t status;
169  if(!_dev->read_registers(QMC5883L_REG_STATUS,&status,1)){
170  return;
171  }
172  //new data is ready
173  if (!(status & 0x04)) {
174  return;
175  }
176 
177  if(!_dev->read_registers(QMC5883L_REG_DATA_OUTPUT_X, (uint8_t *) &buffer, sizeof(buffer))){
178  return ;
179  }
180 
181  auto x = -static_cast<int16_t>(le16toh(buffer.rx));
182  auto y = static_cast<int16_t>(le16toh(buffer.ry));
183  auto z = -static_cast<int16_t>(le16toh(buffer.rz));
184 
185 #if 0
186  printf("mag.x:%d\n",x);
187  printf("mag.y:%d\n",y);
188  printf("mag.z:%d\n",z);
189 #endif
190 
191  Vector3f field = Vector3f{x * range_scale , y * range_scale, z * range_scale };
192 
193  // rotate to the desired orientation
194  if (is_external(_instance)) {
195  field.rotate(ROTATION_YAW_90);
196  }
197 
198  accumulate_sample(field, _instance, 20);
199 }
