49 bool GCS_MAVLINK::signing_key_load(struct SigningKey &key)
50 {
51  if (_signing_storage.size() < sizeof(key)) {
52  return false;
53  }
54  if (!_signing_storage.read_block(&key, 0, sizeof(key))) {
55  return false;
56  }
57  if (key.magic != SIGNING_KEY_MAGIC) {
58  return false;
59  }
60  return true;
61 }

63 /*
64  handle a setup_signing message
65  */
66 void GCS_MAVLINK::handle_setup_signing(const mavlink_message_t *msg)
67 {
68  // setting up signing key when armed generally not useful /
69  // possibly not a good idea
70  if (hal.util->get_soft_armed()) {
71  send_text(MAV_SEVERITY_WARNING, "ERROR: Won't setup signing when armed");
72  return;
73  }
74 
75  // decode
76  mavlink_setup_signing_t packet;
77  mavlink_msg_setup_signing_decode(msg, &packet);
78 
79  struct SigningKey key;
80  key.magic = SIGNING_KEY_MAGIC;
81  key.timestamp = packet.initial_timestamp;
82  memcpy(key.secret_key, packet.secret_key, 32);
83 
84  if (!signing_key_save(key)) {
85  send_text(MAV_SEVERITY_WARNING, "ERROR: Failed to save signing key");
86  return;
87  }
88 
89  // activate it immediately
90  load_signing_key();
91 }

99 static const uint32_t accept_list[] = {
100  MAVLINK_MSG_ID_RADIO_STATUS,

104 static bool accept_unsigned_callback(const mavlink_status_t *status, uint32_t msgId)
105 {
106  if (status == mavlink_get_channel_status(MAVLINK_COMM_0)) {
107  // always accept channel 0, assumed to be secure channel. This
108  // is USB on PX4 boards
109  return true;
110  }
111  for (uint8_t i=0; i<ARRAY_SIZE(accept_list); i++) {
112  if (accept_list[i] == msgId) {
113  return true;
114  }
115  }
116  return false;
117 }

120 /*
121  load signing key
122  */
123 void GCS_MAVLINK::load_signing_key(void)
124 {
125  struct SigningKey key;
126  if (!signing_key_load(key)) {
127  return;
128  }
129  mavlink_status_t *status = mavlink_get_channel_status(chan);
130  if (status == nullptr) {
131  hal.console->printf("Failed to load signing key - no status");
132  return;
133  }
134  memcpy(signing.secret_key, key.secret_key, 32);
135  signing.link_id = (uint8_t)chan;
136  // use a timestamp 1 minute past the last recorded
137  // timestamp. Combined with saving the key once every 30s this
138  // prevents a window for replay attacks
139  signing.timestamp = key.timestamp + 60UL * 100UL * 1000UL;
140  signing.flags = MAVLINK_SIGNING_FLAG_SIGN_OUTGOING;
141  signing.accept_unsigned_callback = accept_unsigned_callback;
142 
143  // if timestamp and key are all zero then we disable signing
144  bool all_zero = (key.timestamp == 0);
145  for (uint8_t i=0; i<sizeof(key.secret_key); i++) {
146  if (signing.secret_key[i] != 0) {
147  all_zero = false;
148  }
149  }
150 
151  // enable signing on all channels
152  for (uint8_t i=0; i<MAVLINK_COMM_NUM_BUFFERS; i++) {
153  mavlink_status_t *cstatus = mavlink_get_channel_status((mavlink_channel_t)(MAVLINK_COMM_0 + i));
154  if (cstatus != nullptr) {
155  if (all_zero) {
156  // disable signing
157  cstatus->signing = nullptr;
158  cstatus->signing_streams = nullptr;
159  } else {
160  cstatus->signing = &signing;
161  cstatus->signing_streams = &signing_streams;
162  }
163  }
164  }
165 }

167 /*
168  update signing timestamp. This is called when we get GPS lock
169  timestamp_usec is since 1/1/1970 (the epoch)
170  */
171 void GCS_MAVLINK::update_signing_timestamp(uint64_t timestamp_usec)
172 {
173  uint64_t signing_timestamp = (timestamp_usec / (1000*1000ULL));
174  // this is the offset from 1/1/1970 to 1/1/2015
175  const uint64_t epoch_offset = 1420070400;
176  if (signing_timestamp > epoch_offset) {
177  signing_timestamp -= epoch_offset;
178  }
179 
180  // convert to 10usec units
181  signing_timestamp *= 100 * 1000ULL;
182 
183  // increase signing timestamp on any links that have signing
184  for (uint8_t i=0; i<MAVLINK_COMM_NUM_BUFFERS; i++) {
185  mavlink_channel_t chan = (mavlink_channel_t)(MAVLINK_COMM_0 + i);
186  mavlink_status_t *status = mavlink_get_channel_status(chan);
187  if (status && status->signing && status->signing->timestamp < signing_timestamp) {
188  status->signing->timestamp = signing_timestamp;
189  }
190  }
191 
192  // save to stable storage
193  save_signing_timestamp(true);
194 }

197 /*
198  save the signing timestamp periodically
199  */
200 void GCS_MAVLINK::save_signing_timestamp(bool force_save_now)
201 {
202  uint32_t now = AP_HAL::millis();
203  // we save the timestamp every 30s, unless forced by a GPS update
204  if (!force_save_now && now - last_signing_save_ms < 30*1000UL) {
205  return;
206  }
207 
208  last_signing_save_ms = now;
209 
210  struct SigningKey key;
211  if (!signing_key_load(key)) {
212  return;
213  }
214  bool need_save = false;
215 
216  for (uint8_t i=0; i<MAVLINK_COMM_NUM_BUFFERS; i++) {
217  mavlink_channel_t chan = (mavlink_channel_t)(MAVLINK_COMM_0 + i);
218  const mavlink_status_t *status = mavlink_get_channel_status(chan);
219  if (status && status->signing && status->signing->timestamp > key.timestamp) {
220  key.timestamp = status->signing->timestamp;
221  need_save = true;
222  }
223  }
224  if (need_save) {
225  // save updated key
226  signing_key_save(key);
227  }
228 }

242 /*
243  return packet overhead in bytes for a channel
244  */
245 uint8_t GCS_MAVLINK::packet_overhead_chan(mavlink_channel_t chan)
246 {
247  /*
248  reserve 100 bytes for parameters when a GCS fails to fetch a
249  parameter due to lack of buffer space. The reservation lasts 2
250  seconds
251  */
252  uint8_t reserved_space = 0;
253  if (reserve_param_space_start_ms != 0 &&
254  AP_HAL::millis() - reserve_param_space_start_ms < 2000) {
255  reserved_space = 100;
256  } else {
257  reserve_param_space_start_ms = 0;
258  }
259 
260  const mavlink_status_t *status = mavlink_get_channel_status(chan);
261  if (status->signing && (status->signing->flags & MAVLINK_SIGNING_FLAG_SIGN_OUTGOING)) {
262  return MAVLINK_NUM_NON_PAYLOAD_BYTES + MAVLINK_SIGNATURE_BLOCK_LEN + reserved_space;
263  }
264  return MAVLINK_NUM_NON_PAYLOAD_BYTES + reserved_space;
265 }
