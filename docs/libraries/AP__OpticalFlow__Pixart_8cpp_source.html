103 // setup the device
104 bool AP_OpticalFlow_Pixart::setup_sensor(void)
105 {
106  if (!_dev) {
107  return false;
108  }
109  WITH_SEMAPHORE(_dev->get_semaphore());
110 
111  uint8_t id;
112  uint16_t crc;
113 
114  // power-up sequence
115  reg_write(PIXART_REG_POWER_RST, 0x5A);
116  hal.scheduler->delay(50);
117 
118  // check product ID
119  uint8_t id1 = reg_read(PIXART_REG_PRODUCT_ID);
120  uint8_t id2;
121  if (id1 == 0x3f) {
122  id2 = reg_read(PIXART_REG_INV_PROD_ID);
123  } else {
124  id2 = reg_read(PIXART_REG_INV_PROD_ID2);
125  }
126  debug("id1=0x%02x id2=0x%02x ~id1=0x%02x\n", id1, id2, uint8_t(~id1));
127  if (id1 == 0x3F && id2 == uint8_t(~id1)) {
128  model = PIXART_3900;
129  } else if (id1 == 0x49 && id2 == uint8_t(~id1)) {
130  model = PIXART_3901;
131  } else {
132  debug("Not a recognised device\n");
133  return false;
134  }
135 
136  if (model == PIXART_3900) {
137  srom_download();
138 
139  id = reg_read(PIXART_REG_SROM_ID);
140  if (id != srom_id) {
141  debug("Pixart: bad SROM ID: 0x%02x\n", id);
142  return false;
143  }
144 
145  reg_write(PIXART_REG_SROM_EN, 0x15);
146  hal.scheduler->delay(10);
147 
148  crc = reg_read16u(PIXART_REG_DOUT_L);
149  if (crc != 0xBEEF) {
150  debug("Pixart: bad SROM CRC: 0x%04x\n", crc);
151  return false;
152  }
153  }
154 
155  if (model == PIXART_3900) {
156  load_configuration(init_data_3900, ARRAY_SIZE(init_data_3900));
157  } else {
158  load_configuration(init_data_3901_1, ARRAY_SIZE(init_data_3901_1));
159  hal.scheduler->delay(100);
160  load_configuration(init_data_3901_2, ARRAY_SIZE(init_data_3901_2));
161  }
162 
163  hal.scheduler->delay(50);
164 
165  debug("Pixart %s ready\n", model==PIXART_3900?"3900":"3901");
166 
167  integral.last_frame_us = AP_HAL::micros();
168 
169  _dev->register_periodic_callback(2000, FUNCTOR_BIND_MEMBER(&AP_OpticalFlow_Pixart::timer, void));
170  return true;
171 }

173 // write an 8 bit register
174 void AP_OpticalFlow_Pixart::reg_write(uint8_t reg, uint8_t value)
175 {
176  _dev->set_chip_select(true);
177  reg |= PIXART_WRITE_FLAG;
178  _dev->transfer(&reg, 1, nullptr, 0);
179  hal.scheduler->delay_microseconds(PIXART_Tsrad);
180  _dev->transfer(&value, 1, nullptr, 0);
181  _dev->set_chip_select(false);
182  hal.scheduler->delay_microseconds(120);
183 }

187 {

189  _dev->set_chip_select(true);

193  _dev->set_chip_select(false);
194  hal.scheduler->delay_microseconds(200);

199 uint16_t AP_OpticalFlow_Pixart::reg_read16u(uint8_t reg)
200 {
201  uint16_t low = reg_read(reg);

212 void AP_OpticalFlow_Pixart::srom_download(void)
213 {
214  reg_write(0x39, 0x02);
215  hal.scheduler->delay(1);
216  reg_write(PIXART_REG_SROM_EN, 0x1D);
217  hal.scheduler->delay(10);
218  reg_write(PIXART_REG_SROM_EN, 0x18);
219 
220  if (!_dev->set_chip_select(true)) {
221  debug("Failed to force CS\n");
222  }
223  hal.scheduler->delay_microseconds(1);
224  uint8_t reg = PIXART_REG_SROM_BURST | PIXART_WRITE_FLAG;
225  _dev->transfer(&reg, 1, nullptr, 0);
226 
227  for (uint16_t i = 0; i < ARRAY_SIZE(srom_data); i++) {
228  hal.scheduler->delay_microseconds(15);
229  _dev->transfer(&srom_data[i], 1, nullptr, 0);
230  }
231 
232  hal.scheduler->delay_microseconds(125);
233  if (!_dev->set_chip_select(false)) {
234  debug("Failed to force CS off\n");
235  }
236  hal.scheduler->delay_microseconds(160);
237 }

239 void AP_OpticalFlow_Pixart::load_configuration(const RegData *init_data, uint16_t n)
240 {
241  for (uint16_t i = 0; i < n; i++) {
242  // writing a config register can fail - retry up to 5 times
243  for (uint8_t tries=0; tries<5; tries++) {
244  reg_write(init_data[i].reg, init_data[i].value);
245  uint8_t v = reg_read(init_data[i].reg);
246  if (v == init_data[i].value) {
247  break;
248  }
249  //debug("reg[%u:%02x] 0x%02x 0x%02x\n", (unsigned)i, (unsigned)init_data[i].reg, (unsigned)init_data[i].value, (unsigned)v);
250  }
251  }

254 void AP_OpticalFlow_Pixart::motion_burst(void)
255 {
256  uint8_t *b = (uint8_t *)&burst;
257 
258  burst.delta_x = 0;
259  burst.delta_y = 0;
260 
261  _dev->set_chip_select(true);
262  uint8_t reg = model==PIXART_3900?PIXART_REG_MOT_BURST:PIXART_REG_MOT_BURST2;
263 
264  _dev->transfer(&reg, 1, nullptr, 0);
265  hal.scheduler->delay_microseconds(150);
266 
267  for (uint8_t i=0; i<sizeof(burst); i++) {
268  _dev->transfer(nullptr, 0, &b[i], 1);
269  if (i == 0 && (burst.motion & 0x80) == 0) {
270  // no motion, save some bus bandwidth
271  _dev->set_chip_select(false);
272  return;
273  }
274  }
275  _dev->set_chip_select(false);
276 }

323 // update - read latest values from sensor and fill in x,y and totals.
324 void AP_OpticalFlow_Pixart::update(void)
325 {
326  uint32_t now = AP_HAL::millis();
327  if (now - last_update_ms < 100) {
328  return;
329  }
330  last_update_ms = now;
331 
332  struct OpticalFlow::OpticalFlow_state state;
333  state.surface_quality = burst.squal;
334 
335  if (integral.sum_us > 0) {
336  WITH_SEMAPHORE(_sem);
337 
338  const Vector2f flowScaler = _flowScaler();
339  float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
340  float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
341  float dt = integral.sum_us * 1.0e-6;
342 
343  state.flowRate = Vector2f(integral.sum.x * flowScaleFactorX,
344  integral.sum.y * flowScaleFactorY);
345  state.flowRate *= flow_pixel_scaling / dt;
346 
347  // we only apply yaw to flowRate as body rate comes from AHRS
348  _applyYaw(state.flowRate);
349 
350  state.bodyRate = integral.gyro / dt;
351 
352  integral.sum.zero();
353  integral.sum_us = 0;
354  integral.gyro.zero();
355  } else {
356  state.flowRate.zero();
357  state.bodyRate.zero();
358  }
359 
360  // copy results to front end
361  _update_frontend(state);
362 }
