125 void RCInput::_process_sbus_pulse(uint16_t width_s0, uint16_t width_s1)
126 {
127  // convert to bit widths, allowing for up to 1usec error, assuming 100000 bps
128  uint16_t bits_s0 = (width_s0+1) / 10;
129  uint16_t bits_s1 = (width_s1+1) / 10;
130  uint16_t nlow;
131 
132  uint8_t byte_ofs = sbus_state.bit_ofs/12;
133  uint8_t bit_ofs = sbus_state.bit_ofs%12;
134 
135  if (bits_s0 == 0 || bits_s1 == 0) {
136  // invalid data
137  goto reset;
138  }
139 
140  if (bits_s0+bit_ofs > 10) {
141  // invalid data as last two bits must be stop bits
142  goto reset;
143  }
144 
145  // pull in the high bits
146  sbus_state.bytes[byte_ofs] |= ((1U<<bits_s0)-1) << bit_ofs;
147  sbus_state.bit_ofs += bits_s0;
148  bit_ofs += bits_s0;
149 
150  // pull in the low bits
151  nlow = bits_s1;
152  if (nlow + bit_ofs > 12) {
153  nlow = 12 - bit_ofs;
154  }
155  bits_s1 -= nlow;
156  sbus_state.bit_ofs += nlow;
157 
158  if (sbus_state.bit_ofs == 25*12 && bits_s1 > 12) {
159  // we have a full frame
160  uint8_t bytes[25];
161  uint8_t i;
162  for (i=0; i<25; i++) {
163  // get inverted data
164  uint16_t v = ~sbus_state.bytes[i];
165  // check start bit
166  if ((v & 1) != 0) {
167  goto reset;
168  }
169  // check stop bits
170  if ((v & 0xC00) != 0xC00) {
171  goto reset;
172  }
173  // check parity
174  uint8_t parity = 0, j;
175  for (j=1; j<=8; j++) {
176  parity ^= (v & (1U<<j))?1:0;
177  }
178  if (parity != (v&0x200)>>9) {
179  goto reset;
180  }
181  bytes[i] = ((v>>1) & 0xFF);
182  }
183  uint16_t values[LINUX_RC_INPUT_NUM_CHANNELS];
184  uint16_t num_values=0;
185  bool sbus_failsafe=false, sbus_frame_drop=false;
186  if (sbus_decode(bytes, values, &num_values,
187  &sbus_failsafe, &sbus_frame_drop,
188  LINUX_RC_INPUT_NUM_CHANNELS) &&
189  num_values >= MIN_NUM_CHANNELS) {
190  for (i=0; i<num_values; i++) {
191  _pwm_values[i] = values[i];
192  }
193  _num_channels = num_values;
194  if (!sbus_failsafe) {
195  rc_input_count++;
196  }
197  }
198  goto reset;
199  } else if (bits_s1 > 12) {
200  // break
201  goto reset;
202  }
203  return;
204 reset:
205  memset(&sbus_state, 0, sizeof(sbus_state));
206 }
