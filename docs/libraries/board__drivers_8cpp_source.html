61 void AP_BoardConfig::board_setup_drivers(void)
62 {
63 #if defined(CONFIG_ARCH_BOARD_PX4FMU_V4)
64  /*
65  this works around an issue with some FMUv4 hardware (eg. copies
66  of the Pixracer) which have incorrect components leading to
67  sensor brownout on boot
68  */
69  if (px4_start_driver(fmu_main, "fmu", "sensor_reset 20")) {
70  printf("FMUv4 sensor reset complete\n");
71  }
72 #endif
73 
74  if (state.board_type == PX4_BOARD_OLDDRIVERS) {
75  printf("Old drivers no longer supported\n");
76  state.board_type = PX4_BOARD_AUTO;
77  }
78 
79  // run board auto-detection
80  board_autodetect();
81 
82  if (state.board_type == PX4_BOARD_PH2SLIM ||
83  state.board_type == PX4_BOARD_PIXHAWK2) {
84  _imu_target_temperature.set_default(45);
85  if (_imu_target_temperature.get() < 0) {
86  // don't allow a value of -1 on the cube, or it could cook
87  // the IMU
88  _imu_target_temperature.set(45);
89  }
90  }
91 
92  px4_configured_board = (enum px4_board_type)state.board_type.get();
93 
94  switch (px4_configured_board) {
95  case PX4_BOARD_PX4V1:
96  case PX4_BOARD_PIXHAWK:
97  case PX4_BOARD_PIXHAWK2:
98  case PX4_BOARD_FMUV5:
99  case PX4_BOARD_FMUV6:
100  case PX4_BOARD_SP01:
101  case PX4_BOARD_PIXRACER:
102  case PX4_BOARD_PHMINI:
103  case PX4_BOARD_AUAV21:
104  case PX4_BOARD_PH2SLIM:
105  case VRX_BOARD_BRAIN51:
106  case VRX_BOARD_BRAIN52:
107  case VRX_BOARD_BRAIN52E:
108  case VRX_BOARD_UBRAIN51:
109  case VRX_BOARD_UBRAIN52:
110  case VRX_BOARD_CORE10:
111  case VRX_BOARD_BRAIN54:
112  case PX4_BOARD_AEROFC:
113  case PX4_BOARD_PIXHAWK_PRO:
114  case PX4_BOARD_PCNC1:
115  case PX4_BOARD_MINDPXV2:
116  break;
117  default:
118  sensor_config_error("Unknown board type");
119  break;
120  }

125 /*
126  check a SPI device for a register value
127  */
128 bool AP_BoardConfig::spi_check_register(const char *devname, uint8_t regnum, uint8_t value, uint8_t read_flag)
129 {
130  auto dev = hal.spi->get_device(devname);
131  if (!dev) {
132 #if SPI_PROBE_DEBUG
133  hal.console->printf("%s: no device\n", devname);
134 #endif
135  return false;
136  }
137  dev->set_read_flag(read_flag);
138  if (!dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
139  return false;
140  }
141  uint8_t v;
142  if (!dev->read_registers(regnum, &v, 1)) {
143 #if SPI_PROBE_DEBUG
144  hal.console->printf("%s: reg %02x read fail\n", devname, (unsigned)regnum);
145 #endif
146  dev->get_semaphore()->give();
147  return false;
148  }
149  dev->get_semaphore()->give();
150 #if SPI_PROBE_DEBUG
151  hal.console->printf("%s: reg %02x expected:%02x got:%02x\n", devname, (unsigned)regnum, (unsigned)value, (unsigned)v);
152 #endif
153  return v == value;
154 }

156 #if defined(HAL_CHIBIOS_ARCH_CUBEBLACK)
157 static bool check_ms5611(const char* devname) {
158  auto dev = hal.spi->get_device(devname);
159  if (!dev) {
160 #if SPI_PROBE_DEBUG
161  hal.console->printf("%s: no device\n", devname);
162 #endif
163  return false;
164  }
165 
166  AP_HAL::Semaphore *dev_sem = dev->get_semaphore();
167 
168  if (!dev_sem || !dev_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
169  return false;
170  }
171 
172  static const uint8_t CMD_MS56XX_RESET = 0x1E;
173  static const uint8_t CMD_MS56XX_PROM = 0xA0;
174 
175  dev->transfer(&CMD_MS56XX_RESET, 1, nullptr, 0);
176  hal.scheduler->delay(4);
177 
178  uint16_t prom[8];
179  bool all_zero = true;
180  for (uint8_t i = 0; i < 8; i++) {
181  const uint8_t reg = CMD_MS56XX_PROM + (i << 1);
182  uint8_t val[2];
183  if (!dev->transfer(&reg, 1, val, sizeof(val))) {
184  dev_sem->give();
185 #if SPI_PROBE_DEBUG
186  hal.console->printf("%s: transfer fail\n", devname);
187 #endif
188  return false;
189  }
190  prom[i] = (val[0] << 8) | val[1];
191 
192  if (prom[i] != 0) {
193  all_zero = false;
194  }
195  }
196  dev_sem->give();
197 
198  uint16_t crc_read = prom[7]&0xf;
199  prom[7] &= 0xff00;
200 
201  if (crc_read != crc_crc4(prom) || all_zero) {
202 #if SPI_PROBE_DEBUG
203  hal.console->printf("%s: crc fail\n", devname);
204 #endif
205  return false;
206  }
207 
208 #if SPI_PROBE_DEBUG
209  hal.console->printf("%s: found successfully\n", devname);
210 #endif
211 
212  return true;
213 }
214 #endif

232 void AP_BoardConfig::validate_board_type(void)
233 {
234  /* some boards can be damaged by the user setting the wrong board
235  type. The key one is the cube which has a heater which can
236  cook the IMUs if the user uses an old paramater file. We
237  override the board type for that specific case
238  */
239 #if defined(CONFIG_ARCH_BOARD_PX4FMU_V2) || defined(HAL_CHIBIOS_ARCH_FMUV3)
240  if (state.board_type == PX4_BOARD_PIXHAWK &&
241  (spi_check_register("mpu6000_ext", MPUREG_WHOAMI, MPU_WHOAMI_MPU60X0) ||
242  spi_check_register("mpu9250_ext", MPUREG_WHOAMI, MPU_WHOAMI_MPU9250) ||
243  spi_check_register("icm20608", MPUREG_WHOAMI, MPU_WHOAMI_ICM20608) ||
244  spi_check_register("icm20608_ext", MPUREG_WHOAMI, MPU_WHOAMI_ICM20602) ||
245  spi_check_register("icm20602_ext", MPUREG_WHOAMI, MPU_WHOAMI_ICM20602)) &&
246  (spi_check_register("lsm9ds0_ext_am", LSMREG_WHOAMI, LSM_WHOAMI_LSM303D) ||
247  spi_check_register("icm20948_ext", INV2REG_WHOAMI, INV2_WHOAMI_ICM20948))) {
248  // Pixhawk2 has LSM303D and MPUxxxx on external bus. If we
249  // detect those, then force PIXHAWK2, even if the user has
250  // configured for PIXHAWK1
251 #if !defined(CONFIG_ARCH_BOARD_PX4FMU_V3) && !defined(HAL_CHIBIOS_ARCH_FMUV3)
252  // force user to load the right firmware
253  sensor_config_error("Pixhawk2 requires FMUv3 firmware");
254 #endif
255  state.board_type.set(PX4_BOARD_PIXHAWK2);
256  hal.console->printf("Forced PIXHAWK2\n");
257  }
258 #endif
259 
260 #if defined(CONFIG_ARCH_BOARD_PX4FMU_V4PRO)
261  // Nothing to do for the moment
262 #endif
263 }

266 void AP_BoardConfig::check_cubeblack(void)
267 {
268 #if defined(HAL_CHIBIOS_ARCH_CUBEBLACK)
269  if (state.board_type != PX4_BOARD_PIXHAWK2) {
270  state.board_type.set(PX4_BOARD_PIXHAWK2);
271  }
272 
273  bool success = true;
274  if (!spi_check_register("mpu9250", MPUREG_WHOAMI, MPU_WHOAMI_MPU9250)) { success = false; }
275  if (!spi_check_register("mpu9250_ext", MPUREG_WHOAMI, MPU_WHOAMI_MPU9250)) { success = false; }
276  if (!spi_check_register("lsm9ds0_ext_g", LSMREG_WHOAMI, LSM_WHOAMI_L3GD20)) { success = false; }
277  if (!spi_check_register("lsm9ds0_ext_am", LSMREG_WHOAMI, LSM_WHOAMI_LSM303D)) { success = false; }
278  if (!check_ms5611("ms5611")) { success = false; }
279  if (!check_ms5611("ms5611_ext")) { success = false; }
280 
281  if (!success) {
282  sensor_config_error("Failed to init CubeBlack - sensor mismatch");
283  }
284 #endif
285 }

291 void AP_BoardConfig::board_autodetect(void)
292 {
293 #if defined(HAL_CHIBIOS_ARCH_CUBEBLACK)
294  check_cubeblack();
295  return;
296 #endif
297 
298  if (state.board_type != PX4_BOARD_AUTO) {
299  validate_board_type();
300  // user has chosen a board type
301  return;
302  }
303 
304 #if defined(CONFIG_ARCH_BOARD_PX4FMU_V1)
305  // only one choice
306  state.board_type.set(PX4_BOARD_PX4V1);
307  hal.console->printf("Detected PX4v1\n");
308 
309 #elif defined(CONFIG_ARCH_BOARD_PX4FMU_V2) || defined(HAL_CHIBIOS_ARCH_FMUV3)
310  if ((spi_check_register("mpu6000_ext", MPUREG_WHOAMI, MPU_WHOAMI_MPU60X0) ||
311  spi_check_register("mpu6000_ext", MPUREG_WHOAMI, MPU_WHOAMI_MPU60X0) ||
312  spi_check_register("mpu9250_ext", MPUREG_WHOAMI, MPU_WHOAMI_MPU60X0) ||
313  spi_check_register("mpu9250_ext", MPUREG_WHOAMI, MPU_WHOAMI_MPU9250) ||
314  spi_check_register("icm20608_ext", MPUREG_WHOAMI, MPU_WHOAMI_ICM20608) ||
315  spi_check_register("icm20608_ext", MPUREG_WHOAMI, MPU_WHOAMI_ICM20602) ||
316  spi_check_register("icm20602_ext", MPUREG_WHOAMI, MPU_WHOAMI_ICM20602)) &&
317  (spi_check_register("lsm9ds0_ext_am", LSMREG_WHOAMI, LSM_WHOAMI_LSM303D) ||
318  spi_check_register("icm20948_ext", INV2REG_WHOAMI, INV2_WHOAMI_ICM20948))) {
319  // Pixhawk2 has LSM303D and MPUxxxx on external bus
320  state.board_type.set(PX4_BOARD_PIXHAWK2);
321  hal.console->printf("Detected PIXHAWK2\n");
322  } else if ((spi_check_register("icm20608-am", MPUREG_WHOAMI, MPU_WHOAMI_ICM20608) ||
323  spi_check_register("icm20608-am", MPUREG_WHOAMI, MPU_WHOAMI_ICM20602)) &&
324  spi_check_register("mpu9250", MPUREG_WHOAMI, MPU_WHOAMI_MPU9250)) {
325  // PHMINI has an ICM20608 and MPU9250 on sensor bus
326  state.board_type.set(PX4_BOARD_PHMINI);
327  hal.console->printf("Detected PixhawkMini\n");
328  } else if (spi_check_register("lsm9ds0_am", LSMREG_WHOAMI, LSM_WHOAMI_LSM303D) &&
329  (spi_check_register("mpu6000", MPUREG_WHOAMI, MPU_WHOAMI_MPU60X0) ||
330  spi_check_register("icm20608", MPUREG_WHOAMI, MPU_WHOAMI_ICM20608) ||
331  spi_check_register("icm20608", MPUREG_WHOAMI, MPU_WHOAMI_ICM20602) ||
332  spi_check_register("mpu9250", MPUREG_WHOAMI, MPU_WHOAMI_MPU9250))) {
333 
334  // classic or upgraded Pixhawk1
335  state.board_type.set(PX4_BOARD_PIXHAWK);
336  hal.console->printf("Detected Pixhawk\n");
337  } else {
338  sensor_config_error("Unable to detect board type");
339  }
340 #elif defined(CONFIG_ARCH_BOARD_PX4FMU_V4) || defined(HAL_CHIBIOS_ARCH_FMUV4)
341  // only one choice
342  state.board_type.set_and_notify(PX4_BOARD_PIXRACER);

344 #elif defined(HAL_CHIBIOS_ARCH_MINDPXV2)

348 #elif defined(CONFIG_ARCH_BOARD_PX4FMU_V4PRO) || defined(HAL_CHIBIOS_ARCH_FMUV4PRO)
349  // only one choice

352 #elif defined(CONFIG_ARCH_BOARD_AEROFC_V1)
353  state.board_type.set_and_notify(PX4_BOARD_AEROFC);
354  hal.console->printf("Detected Aero FC\n");
355 #elif defined(HAL_CHIBIOS_ARCH_FMUV5)
356  state.board_type.set_and_notify(PX4_BOARD_FMUV5);
