24 #if defined(STM32H7)
25 // always use a bouncebuffer on H7, to ensure alignment and padding
26 #define IS_DMA_SAFE(addr) false
27 #elif defined(STM32F7)
28 // on F76x we only consider first half of DTCM memory as DMA safe, 2nd half is used as fast memory for EKF
29 // on F74x we only have 64k of DTCM
30 #define IS_DMA_SAFE(addr) ((((uint32_t)(addr)) & ((0xFFFFFFFF & ~(64*1024U-1)) | 1U)) == 0x20000000)
31 #else
32 // this checks an address is in main memory and 16 bit aligned
33 #define IS_DMA_SAFE(addr) ((((uint32_t)(addr)) & 0xF0000001) == 0x20000000)
34 #endif

37  initialise a bouncebuffer
38  */
39 void bouncebuffer_init(struct bouncebuffer_t **bouncebuffer, uint32_t prealloc_bytes, bool sdcard)
40 {
41  (*bouncebuffer) = calloc(1, sizeof(struct bouncebuffer_t));
42  osalDbgAssert(((*bouncebuffer) != NULL), "bouncebuffer init");
43  (*bouncebuffer)->is_sdcard = sdcard;
44  if (prealloc_bytes) {
45  (*bouncebuffer)->dma_buf = sdcard?malloc_sdcard_dma(prealloc_bytes):malloc_dma(prealloc_bytes);
46  osalDbgAssert(((*bouncebuffer)->dma_buf != NULL), "bouncebuffer preallocate");
47  (*bouncebuffer)->size = prealloc_bytes;
48  }
49 }

51 /*
52  setup for reading from a device into memory, allocating a bouncebuffer if needed
53  Note that *buf can be NULL, in which case we allocate DMA capable memory, but don't
54  copy to it in bouncebuffer_finish_read(). This avoids DMA failures in dummyrx in the SPI LLD
55  */
56 void bouncebuffer_setup_read(struct bouncebuffer_t *bouncebuffer, uint8_t **buf, uint32_t size)
57 {
58  if (!bouncebuffer || IS_DMA_SAFE(*buf)) {
59  // nothing needs to be done
60  return;
61  }
62  osalDbgAssert((bouncebuffer->busy == false), "bouncebuffer read");
63  bouncebuffer->orig_buf = *buf;
64  if (bouncebuffer->size < size) {
65  if (bouncebuffer->size > 0) {
66  free(bouncebuffer->dma_buf);
67  }
68  bouncebuffer->dma_buf = bouncebuffer->is_sdcard?malloc_sdcard_dma(size):malloc_dma(size);
69  osalDbgAssert((bouncebuffer->dma_buf != NULL), "bouncebuffer read allocate");
70  bouncebuffer->size = size;
71  }
72  *buf = bouncebuffer->dma_buf;
73 #if defined(STM32H7)
74  osalDbgAssert((((uint32_t)*buf)&31) == 0, "bouncebuffer read align");
75  cacheBufferInvalidate(*buf, (size+31)&~31);
76 #endif
77  bouncebuffer->busy = true;
78 }

83 void bouncebuffer_finish_read(struct bouncebuffer_t *bouncebuffer, const uint8_t *buf, uint32_t size)
84 {
85  if (bouncebuffer && buf == bouncebuffer->dma_buf) {
86  osalDbgAssert((bouncebuffer->busy == true), "bouncebuffer finish_read");
87  if (bouncebuffer->orig_buf) {
88  memcpy(bouncebuffer->orig_buf, buf, size);
89  }
90  bouncebuffer->busy = false;
91  }
