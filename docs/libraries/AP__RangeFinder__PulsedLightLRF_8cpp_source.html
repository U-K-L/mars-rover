77 /*
78  called at 50Hz
79  */
80 void AP_RangeFinder_PulsedLightLRF::timer(void)
81 {
82  if (check_reg_counter++ == 10) {
83  check_reg_counter = 0;
84  if (!_dev->check_next_register()) {
85  // re-send the acquire. this handles the case of power
86  // cycling while running in continuous mode
87  _dev->write_register(LL40LS_MEASURE_REG, LL40LS_MSRREG_ACQUIRE);
88  }
89  }
90 
91  switch (phase) {
92  case PHASE_COLLECT: {
93  be16_t val;
94  // read the high and low byte distance registers
95  if (_dev->read_registers(LL40LS_DISTHIGH_REG | LL40LS_AUTO_INCREMENT, (uint8_t*)&val, sizeof(val))) {
96  uint16_t _distance_cm = be16toh(val);
97  // remove momentary spikes
98  if (abs(_distance_cm - last_distance_cm) < 100) {
99  state.distance_cm = _distance_cm;
100  state.last_reading_ms = AP_HAL::millis();
101  update_status();
102  }
103  last_distance_cm = _distance_cm;
104  } else {
105  set_status(RangeFinder::RangeFinder_NoData);
106  }
107  if (!v2_hardware) {
108  // for v2 hw we use continuous mode
109  phase = PHASE_MEASURE;
110  }
111  if (!v3hp_hardware) {
112  // for v3hp hw we start PHASE_MEASURE immediately after PHASE_COLLECT
113  break;
114  }
115  }
116  FALLTHROUGH;
117  case PHASE_MEASURE:
118  if (_dev->write_register(LL40LS_MEASURE_REG, LL40LS_MSRREG_ACQUIRE)) {
119  phase = PHASE_COLLECT;
120  }
121  break;
122  }
123 }

157 /*
158  initialise the sensor to required settings
159  */
160 bool AP_RangeFinder_PulsedLightLRF::init(void)
161 {
162  if (!_dev || !_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
163  return false;
164  }
165  _dev->set_retries(3);
166 
167  // LidarLite needs split transfers
168  _dev->set_split_transfers(true);
169 
170  if (rftype == RangeFinder::RangeFinder_TYPE_PLI2CV3) {
171  v2_hardware = true;
172  } else if (rftype == RangeFinder::RangeFinder_TYPE_PLI2CV3HP) {
173  v3hp_hardware = true;
174  } else {
175  // auto-detect v1 vs v2
176  if (!(_dev->read_registers(LL40LS_HW_VERSION, &hw_version, 1) &&
177  hw_version > 0 &&
178  _dev->read_registers(LL40LS_SW_VERSION, &sw_version, 1) &&
179  sw_version > 0)) {
180  printf("PulsedLightI2C: bad version 0x%02x 0x%02x\n", (unsigned)hw_version, (unsigned)sw_version);
181  // invalid version information
182  goto failed;
183  }
184  v2_hardware = (hw_version >= 0x15);
185  }
186 
187  const struct settings_table *table;
188  uint8_t num_settings;
189 
190  if (v2_hardware) {
191  table = settings_v2;
192  num_settings = sizeof(settings_v2) / sizeof(settings_table);
193  phase = PHASE_COLLECT;
194  } else if (v3hp_hardware) {
195  table = settings_v3hp;
196  num_settings = sizeof(settings_v3hp) / sizeof(settings_table);
197  phase = PHASE_MEASURE;
198  } else {
199  table = settings_v1;
200  num_settings = sizeof(settings_v1) / sizeof(settings_table);
201  phase = PHASE_MEASURE;
202  }
203 
204  _dev->setup_checked_registers(num_settings);
205 
206  for (uint8_t i = 0; i < num_settings; i++) {
207  if (!_dev->write_register(table[i].reg, table[i].value, true)) {
208  goto failed;
209  }
210  }
211 
212  printf("Found LidarLite device=0x%x v2=%d v3hp=%d\n", _dev->get_bus_id(), (int)v2_hardware, (int)v3hp_hardware);
213 
214  _dev->get_semaphore()->give();
215 
216  _dev->register_periodic_callback(20000,
217  FUNCTOR_BIND_MEMBER(&AP_RangeFinder_PulsedLightLRF::timer, void));
218  return true;
219 
220 failed:
221  _dev->get_semaphore()->give();
222  return false;
223 }
