74 void AP_KDECAN::init(uint8_t driver_index, bool enable_filters)
75 {
76  _driver_index = driver_index;
77 
78  debug_can(2, "KDECAN: starting init\n\r");
79 
80  if (_initialized) {
81  debug_can(1, "KDECAN: already initialized\n\r");
82  return;
83  }
84 
85  // get CAN manager instance
86  AP_HAL::CANManager* can_mgr = hal.can_mgr[driver_index];
87 
88  if (can_mgr == nullptr) {
89  debug_can(1, "KDECAN: no mgr for this driver\n\r");
90  return;
91  }
92 
93  if (!can_mgr->is_initialized()) {
94  debug_can(1, "KDECAN: mgr not initialized\n\r");
95  return;
96  }
97 
98  // store pointer to CAN driver
99  _can_driver = can_mgr->get_driver();
100 
101  if (_can_driver == nullptr) {
102  debug_can(1, "KDECAN: no CAN driver\n\r");
103  return;
104  }
105 
106  // find available KDE ESCs
107  frame_id_t id = { { .object_address = ESC_INFO_OBJ_ADDR,
108  .destination_id = BROADCAST_NODE_ID,
109  .source_id = AUTOPILOT_NODE_ID,
110  .priority = 0,
111  .unused = 0 } };
112 
113  uavcan::CanFrame frame { (id.value | uavcan::CanFrame::FlagEFF), nullptr, 0 };
114 
115  if(!_can_driver->getIface(CAN_IFACE_INDEX)->send(frame, uavcan::MonotonicTime::fromMSec(AP_HAL::millis() + 1000), 0)) {
116  debug_can(1, "KDECAN: couldn't send discovery message\n\r");
117  return;
118  }
119 
120  debug_can(2, "KDECAN: discovery message sent\n\r");
121 
122  uint32_t start = AP_HAL::millis();
123 
124  // wait 1 second for answers
125  while (AP_HAL::millis() - start < 1000) {
126  uavcan::CanFrame esc_id_frame {};
127  uavcan::MonotonicTime time {};
128  uavcan::UtcTime utc_time {};
129  uavcan::CanIOFlags flags {};
130 
131  int16_t n = _can_driver->getIface(CAN_IFACE_INDEX)->receive(esc_id_frame, time, utc_time, flags);
132 
133  if (n != 1) {
134  continue;
135  }
136 
137  if (!esc_id_frame.isExtended()) {
138  continue;
139  }
140 
141  if (esc_id_frame.dlc != 5) {
142  continue;
143  }
144 
145  id.value = esc_id_frame.id & uavcan::CanFrame::MaskExtID;
146 
147  if (id.source_id == BROADCAST_NODE_ID ||
148  id.source_id >= (KDECAN_MAX_NUM_ESCS + ESC_NODE_ID_FIRST) ||
149  id.destination_id != AUTOPILOT_NODE_ID ||
150  id.object_address != ESC_INFO_OBJ_ADDR) {
151  continue;
152  }
153 
154  _esc_present_bitmask |= (1 << (id.source_id - ESC_NODE_ID_FIRST));
155  _esc_max_node_id = id.source_id - ESC_NODE_ID_FIRST + 1;
156 
157  debug_can(2, "KDECAN: found ESC id %u\n\r", id.source_id);
158  }
159 
160  snprintf(_thread_name, sizeof(_thread_name), "kdecan_%u", driver_index);
161 
162  // start thread for receiving and sending CAN frames
163  if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_KDECAN::loop, void), _thread_name, 4096, AP_HAL::Scheduler::PRIORITY_CAN, 0)) {
164  debug_can(1, "KDECAN: couldn't create thread\n\r");
165  return;
166  }
167 
168  _initialized = true;
169 
170  debug_can(2, "KDECAN: init done\n\r");
171 
172  return;
173 }

175 void AP_KDECAN::loop()
176 {
177  uavcan::MonotonicTime timeout;
178  uavcan::CanFrame empty_frame { (0 | uavcan::CanFrame::FlagEFF), nullptr, 0 };
179  const uavcan::CanFrame* select_frames[uavcan::MaxCanIfaces] { };
180  select_frames[CAN_IFACE_INDEX] = &empty_frame;
181 
182  uint16_t output_buffer[KDECAN_MAX_NUM_ESCS] {};
183 
184  enumeration_state_t enumeration_state = _enumeration_state;
185  uint64_t enumeration_start = 0;
186  uint8_t enumeration_esc_num = 0;
187 
188  const uint32_t LOOP_INTERVAL_US = MIN(AP::scheduler().get_loop_period_us(), SET_PWM_MIN_INTERVAL_US);
189  uint64_t pwm_last_sent = 0;
190  uint8_t sending_esc_num = 0;
191 
192  uint64_t telemetry_last_request = 0;
193 
194  while (true) {
195  if (!_initialized) {
196  debug_can(2, "KDECAN: not initialized\n\r");
197  hal.scheduler->delay_microseconds(2000);
198  continue;
199  }
200 
201  uavcan::CanSelectMasks inout_mask;
202  uint64_t now = AP_HAL::micros64();
203 
204  // get latest enumeration state set from GCS
205  if (_enum_sem.take(1)) {
206  enumeration_state = _enumeration_state;
207  _enum_sem.give();
208  } else {
209  debug_can(2, "KDECAN: failed to get enumeration semaphore on loop\n\r");
210  }
211 
212  if (enumeration_state != ENUMERATION_STOPPED) {
213  // check if enumeration timed out
214  if (enumeration_start != 0 && now - enumeration_start >= ENUMERATION_TIMEOUT_MS * 1000) {
215  enumeration_start = 0;
216 
217  WITH_SEMAPHORE(_enum_sem);
218 
219  // check if enumeration state didn't change or was set to stop
220  if (enumeration_state == _enumeration_state || _enumeration_state == ENUMERATION_STOP) {
221  enumeration_state = _enumeration_state = ENUMERATION_STOPPED;
222  }
223 
224  continue;
225  }
226 
227  timeout = uavcan::MonotonicTime::fromUSec(now + 1000);
228 
229  switch (enumeration_state) {
230  case ENUMERATION_START: {
231  inout_mask.write = 1 << CAN_IFACE_INDEX;
232 
233  // send broadcast frame to start enumeration
234  frame_id_t id = { { .object_address = ENUM_OBJ_ADDR,
235  .destination_id = BROADCAST_NODE_ID,
236  .source_id = AUTOPILOT_NODE_ID,
237  .priority = 0,
238  .unused = 0 } };
239  be16_t data = htobe16((uint16_t) ENUMERATION_TIMEOUT_MS);
240  uavcan::CanFrame frame { (id.value | uavcan::CanFrame::FlagEFF), (uint8_t*) &data, sizeof(data) };
241 
242  uavcan::CanSelectMasks in_mask = inout_mask;
243  select_frames[CAN_IFACE_INDEX] = &frame;
244 
245  // wait for write space to be available
246  _can_driver->select(inout_mask, select_frames, timeout);
247  select_frames[CAN_IFACE_INDEX] = &empty_frame;
248 
249  if (in_mask.write & inout_mask.write) {
250  now = AP_HAL::micros64();
251  timeout = uavcan::MonotonicTime::fromUSec(now + ENUMERATION_TIMEOUT_MS * 1000);
252 
253  int8_t res = _can_driver->getIface(CAN_IFACE_INDEX)->send(frame, timeout, 0);
254 
255  if (res == 1) {
256  enumeration_start = now;
257  enumeration_esc_num = 0;
258  _esc_present_bitmask = 0;
259  _esc_max_node_id = 0;
260 
261  WITH_SEMAPHORE(_enum_sem);
262 
263  if (enumeration_state == _enumeration_state) {
264  enumeration_state = _enumeration_state = ENUMERATION_RUNNING;
265  }
266  } else if (res == 0) {
267  debug_can(1, "KDECAN: strange buffer full when starting ESC enumeration\n\r");
268  break;
269  } else {
270  debug_can(1, "KDECAN: error sending message to start ESC enumeration, result %d\n\r", res);
271  break;
272  }
273  } else {
274  break;
275  }
276  FALLTHROUGH;
277  }
278  case ENUMERATION_RUNNING: {
279  inout_mask.read = 1 << CAN_IFACE_INDEX;
280  inout_mask.write = 0;
281 
282  // wait for enumeration messages from ESCs
283  uavcan::CanSelectMasks in_mask = inout_mask;
284  _can_driver->select(inout_mask, select_frames, timeout);
285 
286  if (in_mask.read & inout_mask.read) {
287  uavcan::CanFrame recv_frame;
288  uavcan::MonotonicTime time;
289  uavcan::UtcTime utc_time;
290  uavcan::CanIOFlags flags {};
291 
292  int16_t res = _can_driver->getIface(CAN_IFACE_INDEX)->receive(recv_frame, time, utc_time, flags);
293 
294  if (res == 1) {
295  if (time.toUSec() < enumeration_start) {
296  // old message
297  break;
298  }
299 
300  frame_id_t id { .value = recv_frame.id & uavcan::CanFrame::MaskExtID };
301 
302  if (id.object_address == UPDATE_NODE_ID_OBJ_ADDR) {
303  // reply from setting new node ID
304  _esc_present_bitmask |= 1 << (id.source_id - ESC_NODE_ID_FIRST);
305  _esc_max_node_id = MAX(_esc_max_node_id, id.source_id - ESC_NODE_ID_FIRST + 1);
306  break;
307  } else if (id.object_address != ENUM_OBJ_ADDR) {
308  // discardable frame, only looking for enumeration
309  break;
310  }
311 
312  // try to set node ID for the received ESC
313  while (AP_HAL::micros64() - enumeration_start < ENUMERATION_TIMEOUT_MS * 1000) {
314  inout_mask.read = 0;
315  inout_mask.write = 1 << CAN_IFACE_INDEX;
316 
317  // wait for write space to be available
318  in_mask = inout_mask;
319  _can_driver->select(inout_mask, select_frames, timeout);
320 
321  if (in_mask.write & inout_mask.write) {
322  id = { { .object_address = UPDATE_NODE_ID_OBJ_ADDR,
323  .destination_id = uint8_t(enumeration_esc_num + ESC_NODE_ID_FIRST),
324  .source_id = AUTOPILOT_NODE_ID,
325  .priority = 0,
326  .unused = 0 } };
327  uavcan::CanFrame send_frame { (id.value | uavcan::CanFrame::FlagEFF), (uint8_t*) &recv_frame.data, recv_frame.dlc };
328  timeout = uavcan::MonotonicTime::fromUSec(enumeration_start + ENUMERATION_TIMEOUT_MS * 1000);
329 
330  res = _can_driver->getIface(CAN_IFACE_INDEX)->send(send_frame, timeout, 0);
331 
332  if (res == 1) {
333  enumeration_esc_num++;
334  break;
335  } else if (res == 0) {
336  debug_can(1, "KDECAN: strange buffer full when setting ESC node ID\n\r");
337  } else {
338  debug_can(1, "KDECAN: error sending message to set ESC node ID, result %d\n\r", res);
339  }
340  }
341  }
342  } else if (res == 0) {
343  debug_can(1, "KDECAN: strange failed read when getting ESC enumeration message\n\r");
344  } else {
345  debug_can(1, "KDECAN: error receiving ESC enumeration message, result %d\n\r", res);
346  }
347  }
348  break;
349  }
350  case ENUMERATION_STOP: {
351  inout_mask.write = 1 << CAN_IFACE_INDEX;
352 
353  // send broadcast frame to stop enumeration
354  frame_id_t id = { { .object_address = ENUM_OBJ_ADDR,
355  .destination_id = BROADCAST_NODE_ID,
356  .source_id = AUTOPILOT_NODE_ID,
357  .priority = 0,
358  .unused = 0 } };
359  le16_t data = htole16((uint16_t) ENUMERATION_TIMEOUT_MS);
360  uavcan::CanFrame frame { (id.value | uavcan::CanFrame::FlagEFF), (uint8_t*) &data, sizeof(data) };
361 
362  uavcan::CanSelectMasks in_mask = inout_mask;
363  select_frames[CAN_IFACE_INDEX] = &frame;
364 
365  // wait for write space to be available
366  _can_driver->select(inout_mask, select_frames, timeout);
367  select_frames[CAN_IFACE_INDEX] = &empty_frame;
368 
369  if (in_mask.write & inout_mask.write) {
370  timeout = uavcan::MonotonicTime::fromUSec(enumeration_start + ENUMERATION_TIMEOUT_MS * 1000);
371 
372  int8_t res = _can_driver->getIface(CAN_IFACE_INDEX)->send(frame, timeout, 0);
373 
374  if (res == 1) {
375  enumeration_start = 0;
376 
377  WITH_SEMAPHORE(_enum_sem);
378 
379  if (enumeration_state == _enumeration_state) {
380  enumeration_state = _enumeration_state = ENUMERATION_STOPPED;
381  }
382  } else if (res == 0) {
383  debug_can(1, "KDECAN: strange buffer full when stop ESC enumeration\n\r");
384  } else {
385  debug_can(1, "KDECAN: error sending message to stop ESC enumeration, result %d\n\r", res);
386  }
387  }
388  break;
389  }
390  case ENUMERATION_STOPPED:
391  default:
392  debug_can(2, "KDECAN: something wrong happened, shouldn't be here, enumeration state: %u\n\r", enumeration_state);
393  break;
394  }
395 
396  continue;
397  }
398 
399  if (!_esc_present_bitmask) {
400  debug_can(1, "KDECAN: no valid ESC present");
401  hal.scheduler->delay(1000);
402  continue;
403  }
404 
405  // always look for received frames
406  inout_mask.read = 1 << CAN_IFACE_INDEX;
407  timeout = uavcan::MonotonicTime::fromUSec(now + LOOP_INTERVAL_US);
408 
409  // check if:
410  // - is currently sending throttle frames, OR
411  // - there are new output values and, a throttle frame was never sent or it's no longer in CAN queue, OR
412  // - it is time to send throttle frames again, regardless of new output values, OR
413  // - it is time to ask for telemetry information
414  if (sending_esc_num > 0 ||
415  (_new_output.load(std::memory_order_acquire) && (pwm_last_sent == 0 || now - pwm_last_sent > SET_PWM_TIMEOUT_US)) ||
416  (pwm_last_sent != 0 && (now - pwm_last_sent > SET_PWM_MIN_INTERVAL_US)) ||
417  (now - telemetry_last_request > TELEMETRY_INTERVAL_US)) {
418 
419  inout_mask.write = 1 << CAN_IFACE_INDEX;
420  } else { // don't need to send frame, choose the maximum time we'll wait for receiving a frame
421  uint64_t next_action = MIN(now + LOOP_INTERVAL_US, telemetry_last_request + TELEMETRY_INTERVAL_US);
422 
423  if (pwm_last_sent != 0) {
424  next_action = MIN(next_action, pwm_last_sent + SET_PWM_MIN_INTERVAL_US);
425  }
426 
427  timeout = uavcan::MonotonicTime::fromUSec(next_action);
428  }
429 
430  // wait for write space or receive frame
431  uavcan::CanSelectMasks in_mask = inout_mask;
432  _can_driver->select(inout_mask, select_frames, timeout);
433 
434  if (in_mask.read & inout_mask.read) {
435  uavcan::CanFrame frame;
436  uavcan::MonotonicTime time;
437  uavcan::UtcTime utc_time;
438  uavcan::CanIOFlags flags {};
439 
440  int16_t res = _can_driver->getIface(CAN_IFACE_INDEX)->receive(frame, time, utc_time, flags);
441 
442  if (res == 1) {
443  frame_id_t id { .value = frame.id & uavcan::CanFrame::MaskExtID };
444 
445  // check if frame is valid: directed at autopilot, doesn't come from broadcast and ESC was detected before
446  if (id.destination_id == AUTOPILOT_NODE_ID &&
447  id.source_id != BROADCAST_NODE_ID &&
448  (1 << (id.source_id - ESC_NODE_ID_FIRST) & _esc_present_bitmask)) {
449  switch (id.object_address) {
450  case TELEMETRY_OBJ_ADDR: {
451  if (frame.dlc != 8) {
452  break;
453  }
454 
455  if (!_telem_sem.take(1)) {
456  debug_can(2, "KDECAN: failed to get telemetry semaphore on write\n\r");
457  break;
458  }
459 
460  _telemetry[id.source_id - ESC_NODE_ID_FIRST].time = time.toUSec();
461  _telemetry[id.source_id - ESC_NODE_ID_FIRST].voltage = frame.data[0] << 8 | frame.data[1];
462  _telemetry[id.source_id - ESC_NODE_ID_FIRST].current = frame.data[2] << 8 | frame.data[3];
463  _telemetry[id.source_id - ESC_NODE_ID_FIRST].rpm = frame.data[4] << 8 | frame.data[5];
464  _telemetry[id.source_id - ESC_NODE_ID_FIRST].temp = frame.data[6];
465  _telemetry[id.source_id - ESC_NODE_ID_FIRST].new_data = true;
466  _telem_sem.give();
467  break;
468  }
469  default:
470  // discard frame
471  break;
472  }
473  }
474  }
475  }
476 
477  if (in_mask.write & inout_mask.write) {
478  now = AP_HAL::micros64();
479 
480  bool new_output = _new_output.load(std::memory_order_acquire);
481 
482  if (sending_esc_num > 0) {
483  // currently sending throttle frames, check it didn't timeout
484  if (now - pwm_last_sent > SET_PWM_TIMEOUT_US) {
485  debug_can(2, "KDECAN: timed-out after sending frame to ESC with ID %d\n\r", sending_esc_num - 1);
486  sending_esc_num = 0;
487  }
488  }
489 
490  if (sending_esc_num == 0 && new_output) {
491  if (!_rc_out_sem.take(1)) {
492  debug_can(2, "KDECAN: failed to get PWM semaphore on read\n\r");
493  continue;
494  }
495 
496  memcpy(output_buffer, _scaled_output, KDECAN_MAX_NUM_ESCS * sizeof(uint16_t));
497 
498  _rc_out_sem.give();
499  }
500 
501  // check if:
502  // - is currently sending throttle frames, OR
503  // - there are new output values and, a throttle frame was never sent or it's no longer in CAN queue, OR
504  // - it is time to send throttle frames again, regardless of new output values
505  if (sending_esc_num > 0 ||
506  (new_output && (pwm_last_sent == 0 || now - pwm_last_sent > SET_PWM_TIMEOUT_US)) ||
507  (pwm_last_sent != 0 && (now - pwm_last_sent > SET_PWM_MIN_INTERVAL_US))) {
508 
509  for (uint8_t esc_num = sending_esc_num; esc_num < _esc_max_node_id; esc_num++) {
510 
511  if ((_esc_present_bitmask & (1 << esc_num)) == 0) {
512  continue;
513  }
514 
515  be16_t kde_pwm = htobe16(output_buffer[esc_num]);
516 
517  if (hal.util->safety_switch_state() == AP_HAL::Util::SAFETY_DISARMED) {
518  kde_pwm = 0;
519  }
520 
521  frame_id_t id = { { .object_address = SET_PWM_OBJ_ADDR,
522  .destination_id = uint8_t(esc_num + ESC_NODE_ID_FIRST),
523  .source_id = AUTOPILOT_NODE_ID,
524  .priority = 0,
525  .unused = 0 } };
526 
527  uavcan::CanFrame frame { (id.value | uavcan::CanFrame::FlagEFF), (uint8_t*) &kde_pwm, sizeof(kde_pwm) };
528 
529  if (esc_num == 0) {
530  timeout = uavcan::MonotonicTime::fromUSec(now + SET_PWM_TIMEOUT_US);
531  } else {
532  timeout = uavcan::MonotonicTime::fromUSec(pwm_last_sent + SET_PWM_TIMEOUT_US);
533  }
534 
535  int8_t res = _can_driver->getIface(CAN_IFACE_INDEX)->send(frame, timeout, 0);
536 
537  if (res == 1) {
538  if (esc_num == 0) {
539  pwm_last_sent = now;
540 
541  if (new_output) {
542  _new_output.store(false, std::memory_order_release);
543  }
544  }
545 
546  sending_esc_num = (esc_num + 1) % _esc_max_node_id;
547  } else if (res == 0) {
548  debug_can(1, "KDECAN: strange buffer full when sending message to ESC with ID %d\n\r", esc_num + ESC_NODE_ID_FIRST);
549  } else {
550  debug_can(1, "KDECAN: error sending message to ESC with ID %d, result %d\n\r", esc_num + ESC_NODE_ID_FIRST, res);
551  }
552 
553  break;
554  }
555  } else if (now - telemetry_last_request > TELEMETRY_INTERVAL_US) {
556  // broadcast telemetry request frame
557  frame_id_t id = { { .object_address = TELEMETRY_OBJ_ADDR,
558  .destination_id = BROADCAST_NODE_ID,
559  .source_id = AUTOPILOT_NODE_ID,
560  .priority = 0,
561  .unused = 0 } };
562 
563  uavcan::CanFrame frame { (id.value | uavcan::CanFrame::FlagEFF), nullptr, 0 };
564  timeout = uavcan::MonotonicTime::fromUSec(now + TELEMETRY_TIMEOUT_US);
565 
566  int8_t res = _can_driver->getIface(CAN_IFACE_INDEX)->send(frame, timeout, 0);
567 
568  if (res == 1) {
569  telemetry_last_request = now;
570  } else if (res == 0) {
571  debug_can(1, "KDECAN: strange buffer full when sending message requesting telemetry\n\r");
572  } else {
573  debug_can(1, "KDECAN: error sending message requesting telemetry, result %d\n\r", res);
574  }
575  }
576  }
577  }

580 void AP_KDECAN::update()
581 {
582  if (_rc_out_sem.take(1)) {
583  for (uint8_t i = 0; i < KDECAN_MAX_NUM_ESCS; i++) {

595  }
596  }
597 
598  _rc_out_sem.give();

607  return;
608  }

612  return;
613  }
614 
615  telemetry_info_t telem_buffer[KDECAN_MAX_NUM_ESCS] {};
616 
617  for (uint8_t i = 0; i < _esc_max_node_id; i++) {

619  telem_buffer[i] = _telemetry[i];
620  _telemetry[i].new_data = false;
621  }
622  }
623 
624  _telem_sem.give();
625 
626  uint8_t num_poles = _num_poles > 0 ? _num_poles : DEFAULT_NUM_POLES;
627

629  for (uint8_t i = 0; i < _esc_max_node_id; i++) {

631  logger->Write_ESC(i, telem_buffer[i].time,

634  telem_buffer[i].current,

636  }

669  }

672  reason = "Too many KDECAN ESCs detected";
673  return false;
674  }
675

677  reason = "Wrong KDECAN node IDs, run enumeration";

684 void AP_KDECAN::send_mavlink(uint8_t chan)
685 {
686  if (!_telem_sem.take(1)) {
687  debug_can(2, "KDECAN: failed to get telemetry semaphore on MAVLink read\n\r");
688  return;
689  }
690 
691  telemetry_info_t telem_buffer[KDECAN_MAX_NUM_ESCS];
692  memcpy(telem_buffer, _telemetry, sizeof(telemetry_info_t) * KDECAN_MAX_NUM_ESCS);
693  _telem_sem.give();
694 
695  uint16_t voltage[4] {};
696  uint16_t current[4] {};
697  uint16_t rpm[4] {};
698  uint8_t temperature[4] {};
699  uint16_t totalcurrent[4] {};
700  uint16_t count[4] {};
701  uint8_t num_poles = _num_poles > 0 ? _num_poles : DEFAULT_NUM_POLES;
702  uint64_t now = AP_HAL::micros64();
703 
704  for (uint8_t i = 0; i < _esc_max_node_id && i < 8; i++) {
705  uint8_t idx = i % 4;
706  if (telem_buffer[i].time && (now - telem_buffer[i].time < 1000000)) {
707  voltage[idx] = telem_buffer[i].voltage;
708  current[idx] = telem_buffer[i].current;
709  rpm[idx] = uint16_t(telem_buffer[i].rpm * 60UL * 2 / num_poles);
710  temperature[idx] = telem_buffer[i].temp;
711  } else {
712  voltage[idx] = 0;
713  current[idx] = 0;
714  rpm[idx] = 0;
715  temperature[idx] = 0;
716  }
717 
718  if (idx == 3 || i == _esc_max_node_id - 1) {
719  if (!HAVE_PAYLOAD_SPACE((mavlink_channel_t)chan, ESC_TELEMETRY_1_TO_4)) {
720  return;
721  }
722 
723  if (i < 4) {
724  mavlink_msg_esc_telemetry_1_to_4_send((mavlink_channel_t)chan, temperature, voltage, current, totalcurrent, rpm, count);
725  } else {
726  mavlink_msg_esc_telemetry_5_to_8_send((mavlink_channel_t)chan, temperature, voltage, current, totalcurrent, rpm, count);
727  }
728  }
729  }
