32 // calculate a destination to avoid the polygon fence
33 // returns DIJKSTRA_STATE_SUCCESS and populates origin_new and destination_new if avoidance is required
34 AP_OADijkstra::AP_OADijkstra_State AP_OADijkstra::update(const Location &current_loc, const Location &destination, Location& origin_new, Location& destination_new)
35 {
36  // require ekf origin to have been set
37  struct Location ekf_origin {};
38  if (!AP::ahrs().get_origin(ekf_origin)) {
39  return DIJKSTRA_STATE_NOT_REQUIRED;
40  }
41 
42  // no avoidance required if fence is disabled
43  if (!polygon_fence_enabled()) {
44  return DIJKSTRA_STATE_NOT_REQUIRED;
45  }
46 
47  // check for fence updates
48  if (check_polygon_fence_updated()) {
49  _polyfence_with_margin_ok = false;
50  }
51 
52  // create inner polygon fence
53  if (!_polyfence_with_margin_ok) {
54  _polyfence_with_margin_ok = create_polygon_fence_with_margin(_polyfence_margin * 100.0f);
55  if (!_polyfence_with_margin_ok) {
56  _polyfence_visgraph_ok = false;
57  _shortest_path_ok = false;
58  return DIJKSTRA_STATE_ERROR;
59  }
60  }
61 
62  // create visgraph for inner polygon fence
63  if (!_polyfence_visgraph_ok) {
64  _polyfence_visgraph_ok = create_polygon_fence_visgraph();
65  if (!_polyfence_visgraph_ok) {
66  _shortest_path_ok = false;
67  return DIJKSTRA_STATE_ERROR;
68  }
69  }
70 
71  // rebuild path if destination has changed
72  if (!destination.same_latlon_as(_destination_prev)) {
73  _destination_prev = destination;
74  _shortest_path_ok = false;
75  }
76 
77  // calculate shortest path from current_loc to destination
78  if (!_shortest_path_ok) {
79  _shortest_path_ok = calc_shortest_path(current_loc, destination);
80  if (!_shortest_path_ok) {
81  return DIJKSTRA_STATE_ERROR;
82  }
83  // start from 2nd point on path (first is the original origin)
84  _path_idx_returned = 1;
85  }
86 
87  // path has been created, return latest point
88  Vector2f dest_pos;
89  if (get_shortest_path_point(_path_idx_returned, dest_pos)) {
90 
91  // for the first point return origin as current_loc
92  Vector2f origin_pos;
93  if ((_path_idx_returned > 0) && get_shortest_path_point(_path_idx_returned-1, origin_pos)) {
94  // convert offset from ekf origin to Location
95  Location temp_loc(Vector3f(origin_pos.x, origin_pos.y, 0.0f));
96  origin_new = temp_loc;
97  } else {
98  // for first point use current loc as origin
99  origin_new = current_loc;
100  }
101 
102  // convert offset from ekf origin to Location
103  Location temp_loc(Vector3f(dest_pos.x, dest_pos.y, 0.0f));
104  destination_new = destination;
105  destination_new.lat = temp_loc.lat;
106  destination_new.lng = temp_loc.lng;
107 
108  // check if we should advance to next point for next iteration
109  const bool near_oa_wp = current_loc.get_distance(destination_new) <= 2.0f;
110  const bool past_oa_wp = current_loc.past_interval_finish_line(origin_new, destination_new);
111  if (near_oa_wp || past_oa_wp) {
112  _path_idx_returned++;
113  }
114  return DIJKSTRA_STATE_SUCCESS;
115  }
116 
117  return DIJKSTRA_STATE_ERROR;
118 }

144 // create a smaller polygon fence within the existing polygon fence
145 // returns true on success
146 bool AP_OADijkstra::create_polygon_fence_with_margin(float margin_cm)
147 {
148  // exit immediately if polygon fence is not enabled
149  const AC_Fence *fence = AC_Fence::get_singleton();
150  if (fence == nullptr) {
151  return false;
152  }
153 
154  // get polygon boundary
155  uint16_t num_points;
156  const Vector2f* boundary = fence->get_boundary_points(num_points);
157  if ((boundary == nullptr) || (num_points < 3)) {
158  return false;
159  }
160 
161  // expand fence point array if required
162  if (!_polyfence_pts.expand_to_hold(num_points)) {
163  return false;
164  }
165 
166  // for each point on polygon fence
167  // Note: boundary is "unclosed" meaning the last point is *not* the same as the first
168  for (uint8_t i=0; i<num_points; i++) {
169 
170  // find points before and after current point (relative to current point)
171  const uint8_t before_idx = (i == 0) ? num_points-1 : i-1;
172  const uint8_t after_idx = (i == num_points-1) ? 0 : i+1;
173  Vector2f before_pt = boundary[before_idx] - boundary[i];
174  Vector2f after_pt = boundary[after_idx] - boundary[i];
175 
176  // if points are overlapping fail
177  if (before_pt.is_zero() || after_pt.is_zero() || (before_pt == after_pt)) {
178  return false;
179  }
180 
181  // scale points to be unit vectors
182  before_pt.normalize();
183  after_pt.normalize();
184 
185  // calculate intermediate point and scale to margin
186  Vector2f intermediate_pt = (after_pt + before_pt) * 0.5f;
187  float intermediate_len = intermediate_pt.length();
188  intermediate_pt *= (margin_cm / intermediate_len);
189 
190  // find final point which is inside the original polygon
191  _polyfence_pts[i] = boundary[i] + intermediate_pt;
192  if (Polygon_outside(_polyfence_pts[i], boundary, num_points)) {
193  _polyfence_pts[i] = boundary[i] - intermediate_pt;
194  if (Polygon_outside(_polyfence_pts[i], boundary, num_points)) {
195  // could not find a point on either side that was within the fence so fail
196  // this can happen if fence lines are closer than margin_cm
197  return false;
198  }
199  }
200  }
201 
202  // update number of fence points
203  _polyfence_numpoints = num_points;
204 
205  // record fence update time so we don't process this exact fence again
206  _polyfence_update_ms = fence->get_boundary_update_ms();
207 
208  return true;
209 }

211 // create a visibility graph of the polygon fence
212 // returns true on success
213 // requires create_polygon_fence_with_margin to have been run
214 bool AP_OADijkstra::create_polygon_fence_visgraph()
215 {
216  // exit immediately if no polygon fence (with margin)
217  if (_polyfence_numpoints == 0) {
218  return false;
219  }
220 
221  // exit immediately if polygon fence is not enabled
222  const AC_Fence *fence = AC_Fence::get_singleton();
223  if (fence == nullptr) {
224  return false;
225  }
226 
227  // get polygon boundary
228  uint16_t num_points;
229  const Vector2f* boundary = fence->get_boundary_points(num_points);
230  if ((boundary == nullptr) || (num_points < 3)) {
231  return false;
232  }
233 
234  // fail if more than number of polygon points algorithm can handle
235  if (num_points >= OA_DIJKSTRA_POLYGON_SHORTPATH_NOTSET_IDX) {
236  return false;
237  }
238 
239  // clear polygon visibility graph
240  _polyfence_visgraph.clear();
241 
242  // calculate distance from each polygon fence point to all other points
243  for (uint8_t i=0; i<_polyfence_numpoints-1; i++) {
244  const Vector2f &start1 = _polyfence_pts[i];
245  for (uint8_t j=i+1; j<_polyfence_numpoints; j++) {
246  const Vector2f &end1 = _polyfence_pts[j];
247  Vector2f intersection;
248  // ToDo: calculation below could be sped up by removing unused intersection and
249  // skipping segments we know are parallel to our fence-with-margin segments
250  if (!Polygon_intersects(boundary, num_points, start1, end1, intersection)) {
251  // line segment does not intersect with original fence so add to visgraph
252  _polyfence_visgraph.add_item({AP_OAVisGraph::OATYPE_FENCE_POINT, i},
253  {AP_OAVisGraph::OATYPE_FENCE_POINT, j},
254  (_polyfence_pts[i] - _polyfence_pts[j]).length());
255  }
256  }
257  }
258 
259  return true;
260 }

262 // updates visibility graph for a given position which is an offset (in cm) from the ekf origin
263 // to add an additional position (i.e. the destination) set add_extra_position = true and provide the position in the extra_position argument
264 // requires create_polygon_fence_with_margin to have been run
265 // returns true on success
266 bool AP_OADijkstra::update_visgraph(AP_OAVisGraph& visgraph, const AP_OAVisGraph::OAItemID& oaid, const Vector2f &position, bool add_extra_position, Vector2f extra_position)
267 {
268  // exit immediately if no polygon fence (with margin)
269  if (_polyfence_numpoints == 0) {
270  return false;
271  }
272 
273  // exit immediately if polygon fence is not enabled
274  const AC_Fence *fence = AC_Fence::get_singleton();
275  if (fence == nullptr) {
276  return false;
277  }
278 
279  // get polygon boundary
280  uint16_t num_points;
281  const Vector2f* boundary = fence->get_boundary_points(num_points);
282  if ((boundary == nullptr) || (num_points < 3)) {
283  return false;
284  }
285 
286  // clear visibility graph
287  visgraph.clear();
288 
289  // calculate distance from extra_position to all fence points
290  for (uint8_t i=0; i<_polyfence_numpoints; i++) {
291  Vector2f intersection;
292  if (!Polygon_intersects(boundary, num_points, position, _polyfence_pts[i], intersection)) {
293  // line segment does not intersect with original fence so add to visgraph
294  visgraph.add_item(oaid, {AP_OAVisGraph::OATYPE_FENCE_POINT, i}, (position - _polyfence_pts[i]).length());
295  }
296  // ToDo: store infinity when there is no clear path between points to allow faster search later
297  }
298 
299  // add extra point to visibility graph if it doesn't intersect with polygon fence
300  if (add_extra_position) {
301  Vector2f intersection;
302  if (!Polygon_intersects(boundary, num_points, position, extra_position, intersection)) {
303  visgraph.add_item(oaid, {AP_OAVisGraph::OATYPE_DESTINATION, 0}, (position - extra_position).length());
304  }
305  }
306 
307  return true;
308 }

310 // update total distance for all nodes visible from current node
311 // curr_node_idx is an index into the _short_path_data array
312 void AP_OADijkstra::update_visible_node_distances(node_index curr_node_idx)
313 {
314  // sanity check
315  if (curr_node_idx > _short_path_data_numpoints) {
316  return;
317  }
318 
319  // get current node for convenience
320  const ShortPathNode &curr_node = _short_path_data[curr_node_idx];
321 
322  // for each visibility graph
323  const AP_OAVisGraph* visgraphs[] = {&_polyfence_visgraph, &_destination_visgraph};
324  for (uint8_t v=0; v<ARRAY_SIZE(visgraphs); v++) {
325 
326  // skip if empty
327  const AP_OAVisGraph &curr_visgraph = *visgraphs[v];
328  if (curr_visgraph.num_items() == 0) {
329  continue;
330  }
331 
332  // search visibility graph for items visible from current_node
333  for (uint8_t i=0; i<curr_visgraph.num_items(); i++) {
334  const AP_OAVisGraph::VisGraphItem &item = curr_visgraph[i];
335  // match if current node's id matches either of the id's in the graph (i.e. either end of the vector)
336  if ((curr_node.id == item.id1) || (curr_node.id == item.id2)) {
337  AP_OAVisGraph::OAItemID matching_id = (curr_node.id == item.id1) ? item.id2 : item.id1;
338  // find item's id in node array
339  node_index item_node_idx;
340  if (find_node_from_id(matching_id, item_node_idx)) {
341  // if current node's distance + distance to item is less than item's current distance, update item's distance
342  const float dist_to_item_via_current_node = _short_path_data[curr_node_idx].distance_cm + item.distance_cm;
343  if (dist_to_item_via_current_node < _short_path_data[item_node_idx].distance_cm) {
344  // update item's distance and set "distance_from_idx" to current node's index
345  _short_path_data[item_node_idx].distance_cm = dist_to_item_via_current_node;
346  _short_path_data[item_node_idx].distance_from_idx = curr_node_idx;
347  }
348  }
349  }
350  }
351  }
352 }

354 // find a node's index into _short_path_data array from it's id (i.e. id type and id number)
355 // returns true if successful and node_idx is updated
356 bool AP_OADijkstra::find_node_from_id(const AP_OAVisGraph::OAItemID &id, node_index &node_idx) const
357 {
358  switch (id.id_type) {
359  case AP_OAVisGraph::OATYPE_SOURCE:
360  // source node is always the first node
361  if (_short_path_data_numpoints > 0) {
362  node_idx = 0;
363  return true;
364  }
365  break;
366  case AP_OAVisGraph::OATYPE_DESTINATION:
367  // destination is always the 2nd node
368  if (_short_path_data_numpoints > 1) {
369  node_idx = 1;
370  return true;
371  }
372  break;
373  case AP_OAVisGraph::OATYPE_FENCE_POINT:
374  // must be a fence node which start from 3rd node
375  if (_short_path_data_numpoints > id.id_num + 2) {
376  node_idx = id.id_num + 2;
377  return true;
378  }
379  break;
380  }
381 
382  // could not find node
383  return false;
384 }

409 // calculate shortest path from origin to destination
410 // returns true on success
411 // requires create_polygon_fence_with_margin and create_polygon_fence_visgraph to have been run
412 // resulting path is stored in _shortest_path array as vector offsets from EKF origin
413 bool AP_OADijkstra::calc_shortest_path(const Location &origin, const Location &destination)
414 {
415  // convert origin and destination to offsets from EKF origin
416  Vector2f origin_NE, destination_NE;
417  if (!origin.get_vector_xy_from_origin_NE(origin_NE) || !destination.get_vector_xy_from_origin_NE(destination_NE)) {
418  return false;
419  }
420 
421  // create origin and destination visgraphs of polygon points
422  update_visgraph(_source_visgraph, {AP_OAVisGraph::OATYPE_SOURCE, 0}, origin_NE, true, destination_NE);
423  update_visgraph(_destination_visgraph, {AP_OAVisGraph::OATYPE_DESTINATION, 0}, destination_NE);
424 
425  // expand _short_path_data if necessary
426  if (!_short_path_data.expand_to_hold(2 + _polyfence_numpoints)) {
427  return false;
428  }
429 
430  // add origin and destination (node_type, id, visited, distance_from_idx, distance_cm) to short_path_data array
431  _short_path_data[0] = {{AP_OAVisGraph::OATYPE_SOURCE, 0}, false, 0, 0};
432  _short_path_data[1] = {{AP_OAVisGraph::OATYPE_DESTINATION, 0}, false, OA_DIJKSTRA_POLYGON_SHORTPATH_NOTSET_IDX, FLT_MAX};
433  _short_path_data_numpoints = 2;
434 
435  // add fence points to short_path_data array (node_type, id, visited, distance_from_idx, distance_cm)
436  for (uint8_t i=0; i<_polyfence_numpoints; i++) {
437  _short_path_data[_short_path_data_numpoints++] = {{AP_OAVisGraph::OATYPE_FENCE_POINT, i}, false, OA_DIJKSTRA_POLYGON_SHORTPATH_NOTSET_IDX, FLT_MAX};
438  }
439 
440  // start algorithm from source point
441  node_index current_node_idx = 0;
442 
443  // update nodes visible from source point
444  for (uint8_t i=0; i<_source_visgraph.num_items(); i++) {
445  node_index node_idx;
446  if (find_node_from_id(_source_visgraph[i].id2, node_idx)) {
447  _short_path_data[node_idx].distance_cm = _source_visgraph[i].distance_cm;
448  _short_path_data[node_idx].distance_from_idx = current_node_idx;
449  } else {
450  return false;
451  }
452  }
453  // mark source node as visited
454  _short_path_data[current_node_idx].visited = true;
455 
456  // move current_node_idx to node with lowest distance
457  while (find_closest_node_idx(current_node_idx)) {
458  // update distances to all neighbours of current node
459  update_visible_node_distances(current_node_idx);
460 
461  // mark current node as visited
462  _short_path_data[current_node_idx].visited = true;
463  }
464 
465  // extract path starting from destination
466  bool success = false;
467  node_index nidx;
468  if (!find_node_from_id({AP_OAVisGraph::OATYPE_DESTINATION,0}, nidx)) {
469  return false;
470  }
471  _path_numpoints = 0;
472  while (true) {
473  // fail if out of space
474  if (_path_numpoints >= _path.max_items()) {
475  if (!_path.expand()) {
476  break;
477  }
478  }
479  // fail if newest node has invalid distance_from_index
480  if ((_short_path_data[nidx].distance_from_idx == OA_DIJKSTRA_POLYGON_SHORTPATH_NOTSET_IDX) ||
481  (_short_path_data[nidx].distance_cm >= FLT_MAX)) {
482  break;
483  } else {
484  // add node's id to path array
485  _path[_path_numpoints] = _short_path_data[nidx].id;
486  _path_numpoints++;
487 
488  // we are done if node is the source
489  if (_short_path_data[nidx].id.id_type == AP_OAVisGraph::OATYPE_SOURCE) {
490  success = true;
491  break;
492  } else {
493  // follow node's "distance_from_idx" to previous node on path
494  nidx = _short_path_data[nidx].distance_from_idx;
495  }
496  }
497  }
498  // update source and destination for by get_shortest_path_point
499  if (success) {
500  _path_source = origin_NE;
501  _path_destination = destination_NE;
502  }
503 
504  return success;
505 }
