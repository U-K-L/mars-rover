101 void AP_BoardConfig_CAN::init()
102 {
103  // Create all drivers that we need
104  bool initret = true;
105  #if !HAL_MINIMIZE_FEATURES
106  reset_slcan_serial();
107 #endif
108  for (uint8_t i = 0; i < MAX_NUMBER_OF_CAN_INTERFACES; i++) {
109  // Check the driver number assigned to this physical interface
110  uint8_t drv_num = _interfaces[i]._driver_number_cache = _interfaces[i]._driver_number;
111  if (drv_num != 0 && drv_num <= MAX_NUMBER_OF_CAN_DRIVERS) {
112  if (hal.can_mgr[drv_num - 1] == nullptr) {
113  // CAN Manager is the driver
114  // So if this driver was not created before for other physical interface - do it
115  #if CONFIG_HAL_BOARD == HAL_BOARD_LINUX
116  const_cast <AP_HAL::HAL&> (hal).can_mgr[drv_num - 1] = new Linux::CANManager;
117  #elif CONFIG_HAL_BOARD == HAL_BOARD_CHIBIOS
118  const_cast <AP_HAL::HAL&> (hal).can_mgr[drv_num - 1] = new ChibiOS::CANManager;
119  #endif
120  }
121 
122  // For this now existing driver (manager), start the physical interface
123  if (hal.can_mgr[drv_num - 1] != nullptr) {
124  initret = initret && hal.can_mgr[drv_num - 1]->begin(_interfaces[i]._bitrate, i);
125  #if CONFIG_HAL_BOARD == HAL_BOARD_CHIBIOS && !HAL_MINIMIZE_FEATURES
126  if (_slcan._can_port == (i+1) && hal.can_mgr[drv_num - 1] != nullptr ) {
127  ChibiOS_CAN::CanDriver* drv = (ChibiOS_CAN::CanDriver*)hal.can_mgr[drv_num - 1]->get_driver();
128  ChibiOS_CAN::CanIface::slcan_router().init(drv->getIface(i), drv->getUpdateEvent());
129  }
130  #endif
131  } else {
132  printf("Failed to initialize can interface %d\n\r", i + 1);
133  }
134  }
135  }
136 
137  if (initret) {
138  for (uint8_t i = 0; i < MAX_NUMBER_OF_CAN_DRIVERS; i++) {
139  Protocol_Type prot_type = _drivers[i]._protocol_type_cache = (Protocol_Type) _drivers[i]._protocol_type.get();
140 
141  if (hal.can_mgr[i] == nullptr) {
142  continue;
143  }
144 
145  _num_drivers = i + 1;
146  hal.can_mgr[i]->initialized(true);
147  printf("can_mgr %d initialized well\n\r", i + 1);
148 
149  if (prot_type == Protocol_Type_UAVCAN) {
150  _drivers[i]._driver = _drivers[i]._uavcan = new AP_UAVCAN;
151 
152  if (_drivers[i]._driver == nullptr) {
153  AP_HAL::panic("Failed to allocate uavcan %d\n\r", i + 1);
154  continue;
155  }
156 
157  AP_Param::load_object_from_eeprom(_drivers[i]._uavcan, AP_UAVCAN::var_info);
158  } else if (prot_type == Protocol_Type_KDECAN) {
159 // To be replaced with macro saying if KDECAN library is included
160 #if APM_BUILD_TYPE(APM_BUILD_ArduCopter) || APM_BUILD_TYPE(APM_BUILD_ArduPlane) || APM_BUILD_TYPE(APM_BUILD_ArduSub)
161  _drivers[i]._driver = _drivers[i]._kdecan = new AP_KDECAN;
162 
163  if (_drivers[i]._driver == nullptr) {
164  AP_HAL::panic("Failed to allocate KDECAN %d\n\r", i + 1);
165  continue;
166  }
167 
168  AP_Param::load_object_from_eeprom(_drivers[i]._kdecan, AP_KDECAN::var_info);
169 #endif
170  } else if (prot_type == Protocol_Type_ToshibaCAN) {
171  _drivers[i]._driver = _drivers[i]._tcan = new AP_ToshibaCAN;
172 
173  if (_drivers[i]._driver == nullptr) {
174  AP_BoardConfig::sensor_config_error("ToshibaCAN init failed");
175  continue;
176  }
177  } else {
178  continue;
179  }
180 #if !HAL_MINIMIZE_FEATURES
181  if (_slcan._can_port == 0) {
182  _drivers[i]._driver->init(i, true);
183  } else {
184  _drivers[i]._driver->init(i, false);
185  }
186 #endif
187  }
188  }
189 }
