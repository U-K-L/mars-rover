450 bool AP_InertialSensor_LSM9DS0::_hardware_init()
451 {
452  if (!_spi_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
453  return false;
454  }
455 
456  uint8_t tries, whoami;
457 
458  // set flag for reading registers
459  _dev_gyro->set_read_flag(0x80);
460  _dev_accel->set_read_flag(0x80);
461 
462  whoami_g = _register_read_g(WHO_AM_I_G);
463  if (whoami_g != LSM9DS0_G_WHOAMI && whoami_g != LSM9DS0_G_WHOAMI_H) {
464  goto fail_whoami;
465  }
466 
467  whoami = _register_read_xm(WHO_AM_I_XM);
468  if (whoami != LSM9DS0_XM_WHOAMI) {
469  goto fail_whoami;
470  }
471 
472  // setup for register checking
473  _dev_gyro->setup_checked_registers(5, 20);
474  _dev_accel->setup_checked_registers(4, 20);
475 
476  for (tries = 0; tries < 5; tries++) {
477  _dev_gyro->set_speed(AP_HAL::Device::SPEED_LOW);
478  _dev_accel->set_speed(AP_HAL::Device::SPEED_LOW);
479 
480  _gyro_init();
481  _accel_init();
482 
483  _dev_gyro->set_speed(AP_HAL::Device::SPEED_HIGH);
484  _dev_accel->set_speed(AP_HAL::Device::SPEED_HIGH);
485 
486  hal.scheduler->delay(10);
487  if (_accel_data_ready() && _gyro_data_ready()) {
488  break;
489  }
490 
491 #if LSM9DS0_DEBUG
492  _dump_registers();
493 #endif
494  }

496  hal.console->printf("Failed to boot LSM9DS0 5 times\n\n");
497  goto fail_tries;
498  }

500  _spi_sem->give();
501 
502  return true;
503 
504 fail_tries:

667 void AP_InertialSensor_LSM9DS0::_set_accel_scale(accel_scale scale)
668 {
669  /*
670  * Possible accelerometer scales (and their register bit settings) are:
671  * 2 g (000), 4g (001), 6g (010) 8g (011), 16g (100). Here's a bit of an
672  * algorithm to calculate g/(ADC tick) based on that 3-bit value:
673  */
674  _accel_scale = (((float) scale + 1.0f) * 2.0f) / 32768.0f;
675  if (scale == A_SCALE_16G) {
676  /* the datasheet shows an exception for +-16G */
677  _accel_scale = 0.000732f;
678  }
679  /* convert to G/LSB to (m/s/s)/LSB */
680  _accel_scale *= GRAVITY_MSS;
