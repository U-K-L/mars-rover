27 void AP_WindVane_Analog::update_direction()
28 {
29  _dir_analog_source->set_pin(_frontend._dir_analog_pin);
30  _current_analog_voltage = _dir_analog_source->voltage_average_ratiometric();
31 
32  const float voltage_ratio = linear_interpolate(0.0f, 1.0f, _current_analog_voltage, _frontend._dir_analog_volt_min, _frontend._dir_analog_volt_max);
33  const float direction = (voltage_ratio * radians(360 - _frontend._dir_analog_deadzone)) + radians(_frontend._dir_analog_bearing_offset);
34 
35  direction_update_frontend(wrap_PI(direction + AP::ahrs().yaw));
36 }

38 void AP_WindVane_Analog::calibrate()
39 {
40 
41  // start calibration
42  if (_cal_start_ms == 0) {
43  _cal_start_ms = AP_HAL::millis();
44  _cal_volt_max = _current_analog_voltage;
45  _cal_volt_min = _current_analog_voltage;
46  gcs().send_text(MAV_SEVERITY_INFO, "WindVane: Calibration started, rotate wind vane");
47  }
48 
49  // record min and max voltage
50  _cal_volt_max = MAX(_cal_volt_max, _current_analog_voltage);
51  _cal_volt_min = MIN(_cal_volt_min, _current_analog_voltage);
52 
53  // calibrate for 30 seconds
54  if ((AP_HAL::millis() - _cal_start_ms) > 30000) {
55  // check for required voltage difference
56  const float volt_diff = _cal_volt_max - _cal_volt_min;
57  if (volt_diff >= WINDVANE_CALIBRATION_VOLT_DIFF_MIN) {
58  // save min and max voltage
59  _frontend._dir_analog_volt_max.set_and_save(_cal_volt_max);
60  _frontend._dir_analog_volt_min.set_and_save(_cal_volt_min);
61  gcs().send_text(MAV_SEVERITY_INFO, "WindVane: Calibration complete (volt min:%.1f max:%1.f)",
62  (double)_cal_volt_min,
63  (double)_cal_volt_max);
64  } else {
65  gcs().send_text(MAV_SEVERITY_INFO, "WindVane: Calibration failed (volt diff %.1f below %.1f)",
66  (double)volt_diff,
67  (double)WINDVANE_CALIBRATION_VOLT_DIFF_MIN);
68  }
69  _frontend._calibration.set_and_save(0);
70  _cal_start_ms = 0;
71  }
72 }
