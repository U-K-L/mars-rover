135  // autotune modes (high level states)
136  enum TuneMode {
137  UNINITIALISED = 0, // autotune has never been run
138  TUNING = 1, // autotune is testing gains
139  SUCCESS = 2, // tuning has completed, user is flight testing the new gains
140  FAILED = 3, // tuning has failed, user is flying on original gains
141  };
142 
143  // steps performed while in the tuning mode
144  enum StepType {
145  WAITING_FOR_LEVEL = 0, // autotune is waiting for vehicle to return to level before beginning the next twitch
146  TWITCHING = 1, // autotune has begun a twitch and is watching the resulting vehicle movement
147  UPDATE_GAINS = 2 // autotune has completed a twitch and is updating the gains based on the results
148  };
149 
150  // things that can be tuned
151  enum AxisType {
152  ROLL = 0, // roll axis is being tuned (either angle or rate)
153  PITCH = 1, // pitch axis is being tuned (either angle or rate)
154  YAW = 2, // pitch axis is being tuned (either angle or rate)
155  };
156 
157  // mini steps performed while in Tuning mode, Testing step
158  enum TuneType {
159  RD_UP = 0, // rate D is being tuned up
160  RD_DOWN = 1, // rate D is being tuned down
161  RP_UP = 2, // rate P is being tuned up
162  SP_DOWN = 3, // angle P is being tuned down
163  SP_UP = 4 // angle P is being tuned up
164  };
