56 // probe and initialise the sensor
57 bool AP_Airspeed_SDP3X::init()
58 {
59  const uint8_t addresses[3] = { SDP3XD0_I2C_ADDR,
60  SDP3XD1_I2C_ADDR,
61  SDP3XD2_I2C_ADDR
62  };
63  bool found = false;
64  bool ret = false;
65 
66  for (uint8_t i=0; i<ARRAY_SIZE(addresses) && !found; i++) {
67  _dev = hal.i2c_mgr->get_device(get_bus(), addresses[i]);
68  if (!_dev) {
69  continue;
70  }
71  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
72  continue;
73  }
74 
75  // lots of retries during probe
76  _dev->set_retries(10);
77 
78  // stop continuous average mode
79  if (!_send_command(SDP3X_CONT_MEAS_STOP)) {
80  _dev->get_semaphore()->give();
81  continue;
82  }
83 
84  // these delays are needed for reliable operation
85  _dev->get_semaphore()->give();
86  hal.scheduler->delay_microseconds(20000);
87  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
88  continue;
89  }
90 
91  // start continuous average mode
92  if (!_send_command(SDP3X_CONT_MEAS_AVG_MODE)) {
93  _dev->get_semaphore()->give();
94  continue;
95  }
96 
97  // these delays are needed for reliable operation
98  _dev->get_semaphore()->give();
99  hal.scheduler->delay_microseconds(20000);
100  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
101  continue;
102  }
103 
104  // step 3 - get scale
105  uint8_t val[9];
106  ret = _dev->transfer(nullptr, 0, &val[0], sizeof(val));
107  if (!ret) {
108  _dev->get_semaphore()->give();
109  continue;
110  }
111 
112  // Check the CRC
113  if (!_crc(&val[0], 2, val[2]) || !_crc(&val[3], 2, val[5]) || !_crc(&val[6], 2, val[8])) {
114  _dev->get_semaphore()->give();
115  continue;
116  }
117 
118  _scale = (((uint16_t)val[6]) << 8) | val[7];
119 
120  _dev->get_semaphore()->give();
121 
122  found = true;
123 
124  char c = 'X';
125  switch (_scale) {
126  case SDP3X_SCALE_PRESSURE_SDP31:
127  c = '1';
128  break;
129  case SDP3X_SCALE_PRESSURE_SDP32:
130  c = '2';
131  break;
132  case SDP3X_SCALE_PRESSURE_SDP33:
133  c = '3';
134  break;
135  }
136  hal.console->printf("SDP3%c: Found on bus %u address 0x%02x scale=%u\n",
137  c, get_bus(), addresses[i], _scale);
138  }
139 
140  if (!found) {
141  return false;
142  }
143 
144  /*
145  this sensor uses zero offset and skips cal
146  */
147  set_use_zero_offset();
148  set_skip_cal();
149  set_offset(0);
150 
151  // drop to 2 retries for runtime
152  _dev->set_retries(2);
153 
154  _dev->register_periodic_callback(20000,
155  FUNCTOR_BIND_MEMBER(&AP_Airspeed_SDP3X::_timer, void));
156  return true;
157 }

159 // read the values from the sensor. Called at 50Hz
160 void AP_Airspeed_SDP3X::_timer()
161 {
162  // read 6 bytes from the sensor
163  uint8_t val[6];
164  int ret = _dev->transfer(nullptr, 0, &val[0], sizeof(val));
165  uint32_t now = AP_HAL::millis();
166  if (!ret) {
167  if (now - _last_sample_time_ms > 200) {
168  // try and re-connect
169  _send_command(SDP3X_CONT_MEAS_AVG_MODE);
170  }
171  return;
172  }
173  // Check the CRC
174  if (!_crc(&val[0], 2, val[2]) || !_crc(&val[3], 2, val[5])) {
175  return;
176  }
177 
178  int16_t P = (((int16_t)val[0]) << 8) | val[1];
179  int16_t temp = (((int16_t)val[3]) << 8) | val[4];
180 
181  float diff_press_pa = float(P) / float(_scale);
182  float temperature = float(temp) / SDP3X_SCALE_TEMPERATURE;
183 
184  WITH_SEMAPHORE(sem);
185 
186  _press_sum += diff_press_pa;
187  _temp_sum += temperature;
188  _press_count++;
189  _temp_count++;
190  _last_sample_time_ms = now;
191 }

193 /*
194  correct pressure for barometric height
195  With thanks to:
196  https://github.com/PX4/Firmware/blob/master/Tools/models/sdp3x_pitot_model.py
197  */
198 float AP_Airspeed_SDP3X::_correct_pressure(float press)
199 {
200  float sign = 1.0f;
201 
202  // fix for tube order
203  AP_Airspeed::pitot_tube_order tube_order = get_tube_order();
204  switch (tube_order) {
205  case AP_Airspeed::PITOT_TUBE_ORDER_NEGATIVE:
206  press = -press;
207  sign = -1.0f;
208  //FALLTHROUGH;
209  case AP_Airspeed::PITOT_TUBE_ORDER_POSITIVE:
210  break;
211  case AP_Airspeed::PITOT_TUBE_ORDER_AUTO:
212  default:
213  if (press < 0.0f) {
214  sign = -1.0f;
215  press = -press;
216  }
217  break;
218  }
219 
220  if (press <= 0.0f) {
221  return 0.0f;
222  }
223 
224  AP_Baro *baro = AP_Baro::get_singleton();
225 
226  if (baro == nullptr) {
227  return press;
228  }
229 
230  float temperature;
231  if (!get_temperature(temperature)) {
232  return press;
233  }
234 
235  float rho_air = baro->get_pressure() / (ISA_GAS_CONSTANT * (temperature + C_TO_KELVIN));
236 
237  /*
238  the constants in the code below come from a calibrated test of
239  the drotek pitot tube by Sensiron. They are specific to the droktek pitot tube
240 
241  At 25m/s, the rough proportions of each pressure correction are:
242 
243  - dp_pitot: 5%
244  - press_correction: 14%
245  - press: 81%
246 
247  dp_tube has been removed from the Sensiron model as it is
248  insignificant (less than 0.02% over the supported speed ranges)
249  */
250 
251  // flow through sensor
252  float flow_SDP3X = (300.805f - 300.878f / (0.00344205f * (float)powf(press, 0.68698f) + 1.0f)) * 1.29f / rho_air;
253  if (flow_SDP3X < 0.0f) {
254  flow_SDP3X = 0.0f;
255  }
256 
257  // diffential pressure through pitot tube
258  float dp_pitot = 28557670.0f * (1.0f - 1.0f / (1.0f + (float)powf((flow_SDP3X / 5027611.0f), 1.227924f)));
259 
260  // uncorrected pressure
261  float press_uncorrected = (press + dp_pitot) / SSL_AIR_DENSITY;
262 
263  // correction for speed at pitot-tube tip due to flow through sensor
264  float dv = 0.0331582f * flow_SDP3X;
265 
266  // airspeed ratio
267  float ratio = get_airspeed_ratio();
268 
269  // calculate equivalent pressure correction. This formula comes
270  // from turning the dv correction above into an equivalent
271  // pressure correction. We need to do this so the airspeed ratio
272  // calibrator can work, as it operates on pressure values
273  float press_correction = sq(sqrtf(press_uncorrected*ratio)+dv)/ratio - press_uncorrected;
274 
275  return (press_uncorrected + press_correction) * sign;
276 }

278 // return the current differential_pressure in Pascal
279 bool AP_Airspeed_SDP3X::get_differential_pressure(float &pressure)
280 {
281  uint32_t now = AP_HAL::millis();
282  if (now - _last_sample_time_ms > 100) {
283  return false;
284  }
285 
286  {
287  WITH_SEMAPHORE(sem);
288  if (_press_count > 0) {
289  _press = _press_sum / _press_count;
290  _press_count = 0;
291  _press_sum = 0;
292  }
293  }
294 
295  pressure = _correct_pressure(_press);
296  return true;
297 }

318 /*
319  check CRC for a set of bytes
320  */
321 bool AP_Airspeed_SDP3X::_crc(const uint8_t data[], unsigned size, uint8_t checksum)
322 {
323  uint8_t crc_value = 0xff;
324 
325  // calculate 8-bit checksum with polynomial 0x31 (x^8 + x^5 + x^4 + 1)
326  for (uint8_t i = 0; i < size; i++) {
327  crc_value ^= data[i];
328  for (uint8_t bit = 8; bit > 0; --bit) {
329  if (crc_value & 0x80) {
330  crc_value = (crc_value << 1) ^ 0x31;
331  } else {
332  crc_value = (crc_value << 1);
333  }
334  }
335  }
336  // verify checksum
337  return (crc_value == checksum);
338 }
