41 /*
42  process a pulse made up of a width of values at high voltage
43  followed by a width at low voltage
44  */
45 bool SoftSerial::process_pulse(uint32_t width_high, uint32_t width_low, uint8_t &byte)
46 {
47  // convert to bit widths, allowing for a half bit error
48  uint16_t bits_high = ((width_high+half_bit)*baudrate) / 1000000;
49  uint16_t bits_low = ((width_low+half_bit)*baudrate) / 1000000;
50 
51  byte_timestamp_us = timestamp_us;
52  timestamp_us += (width_high + width_low);
53 
54  if (bits_high == 0 || bits_low == 0) {
55  // invalid data
56  goto reset;
57  }
58 
59  if (bits_high >= byte_width) {
60  // if we have a start bit and a stop bit then we can have at
61  // most 9 bits in high state for data. The rest must be idle
62  // bits
63  bits_high = byte_width-1;
64  }
65 
66  if (state.bit_ofs == 0) {
67  // we are in idle state, waiting for first low bit. swallow
68  // the high bits
69  bits_high = 0;
70  }
71 
72  state.byte |= ((1U<<bits_high)-1) << state.bit_ofs;
73 
74  state.bit_ofs += bits_high + bits_low;
75 
76  if (state.bit_ofs >= byte_width) {
77  // check start bit
78  if ((state.byte & 1) != 0) {
79  goto reset;
80  }
81  // check stop bits
82  if ((state.byte & stop_mask) != stop_mask) {
83  goto reset;
84  }
85  if (config == SERIAL_CONFIG_8E2I) {
86  // check parity
87  if (__builtin_parity((state.byte>>1)&0xFF) != (state.byte&0x200)>>9) {
88  goto reset;
89  }
90  }
91 
92  byte = ((state.byte>>1) & 0xFF);
93  state.byte >>= byte_width;
94  state.bit_ofs -= byte_width;
95  if (state.bit_ofs > byte_width) {
96  state.byte = 0;
97  state.bit_ofs = bits_low;
98  }
99  // swallow idle bits
100  while (state.bit_ofs > 0 && (state.byte & 1)) {
101  state.bit_ofs--;
102  state.byte >>= 1;
103  }
104  return true;
105  }
106  return false;
107 
108 reset:
109  state.byte = 0;
110  state.bit_ofs = 0;
111 
112  return false;
113 }
