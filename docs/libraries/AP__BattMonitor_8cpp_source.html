351 void AP_BattMonitor::check_failsafes(void)
352 {
353  if (hal.util->get_soft_armed()) {
354  for (uint8_t i = 0; i < _num_instances; i++) {
355  if (drivers[i] == nullptr) {
356  continue;
357  }
358 
359  const BatteryFailsafe type = drivers[i]->update_failsafes();
360  if (type <= state[i].failsafe) {
361  continue;
362  }
363 
364  int8_t action = 0;
365  const char *type_str = nullptr;
366  switch (type) {
367  case AP_BattMonitor::BatteryFailsafe_None:
368  continue; // should not have been called in this case
369  case AP_BattMonitor::BatteryFailsafe_Low:
370  action = _params[i]._failsafe_low_action;
371  type_str = "low";
372  break;
373  case AP_BattMonitor::BatteryFailsafe_Critical:
374  action = _params[i]._failsafe_critical_action;
375  type_str = "critical";
376  break;
377  }
378 
379  gcs().send_text(MAV_SEVERITY_WARNING, "Battery %d is %s %.2fV used %.0f mAh", i + 1, type_str,
380  (double)voltage(i), (double)consumed_mah(i));
381  _has_triggered_failsafe = true;
382  AP_Notify::flags.failsafe_battery = true;
383  state[i].failsafe = type;
384 
385  // map the desired failsafe action to a prioritiy level
386  int8_t priority = 0;
387  if (_failsafe_priorities != nullptr) {
388  while (_failsafe_priorities[priority] != -1) {
389  if (_failsafe_priorities[priority] == action) {
390  break;
391  }
392  priority++;
393  }
394 
395  }
396 
397  // trigger failsafe if the action was equal or higher priority
398  // It's valid to retrigger the same action if a different battery provoked the event
399  if (priority <= _highest_failsafe_priority) {
400  _battery_failsafe_handler_fn(type_str, action);
401  _highest_failsafe_priority = priority;
402  }
403  }
404  }
405 }

501 bool AP_BattMonitor::reset_remaining(uint16_t battery_mask, float percentage)
502 {
503  bool ret = true;
504  BatteryFailsafe highest_failsafe = BatteryFailsafe_None;
505  for (uint8_t i = 0; i < _num_instances; i++) {
506  if ((1U<<i) & battery_mask) {
507  ret &= drivers[i]->reset_remaining(percentage);
508  }
509  if (state[i].failsafe > highest_failsafe) {
510  highest_failsafe = state[i].failsafe;
511  }
512  }
513 
514  // If all backends are not in failsafe then set overall failsafe state
515  if (highest_failsafe == BatteryFailsafe_None) {
516  _highest_failsafe_priority = INT8_MAX;
517  _has_triggered_failsafe = false;
518  // and reset notify flag
519  AP_Notify::flags.failsafe_battery = false;
520  }
521  return ret;
522 }
