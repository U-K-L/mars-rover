52 void AP_GPS_UAVCAN::subscribe_msgs(AP_UAVCAN* ap_uavcan)
53 {
54  if (ap_uavcan == nullptr) {
55  return;
56  }
57 
58  auto* node = ap_uavcan->get_node();
59 
60  uavcan::Subscriber<uavcan::equipment::gnss::Fix, FixCb> *gnss_fix;
61  gnss_fix = new uavcan::Subscriber<uavcan::equipment::gnss::Fix, FixCb>(*node);
62  const int gnss_fix_start_res = gnss_fix->start(FixCb(ap_uavcan, &handle_fix_msg_trampoline));
63  if (gnss_fix_start_res < 0) {
64  AP_HAL::panic("UAVCAN GNSS subscriber start problem\n\r");
65  return;
66  }
67 
68  uavcan::Subscriber<uavcan::equipment::gnss::Auxiliary, AuxCb> *gnss_aux;
69  gnss_aux = new uavcan::Subscriber<uavcan::equipment::gnss::Auxiliary, AuxCb>(*node);
70  const int gnss_aux_start_res = gnss_aux->start(AuxCb(ap_uavcan, &handle_aux_msg_trampoline));
71  if (gnss_aux_start_res < 0) {
72  AP_HAL::panic("UAVCAN GNSS subscriber start problem\n\r");
73  return;

77 AP_GPS_Backend* AP_GPS_UAVCAN::probe(AP_GPS &_gps, AP_GPS::GPS_State &_state)
78 {
79  WITH_SEMAPHORE(_sem_registry);
80 
81  AP_GPS_UAVCAN* backend = nullptr;
82  for (uint8_t i = 0; i < GPS_MAX_RECEIVERS; i++) {
83  if (_detected_modules[i].driver == nullptr && _detected_modules[i].ap_uavcan != nullptr) {
84  backend = new AP_GPS_UAVCAN(_gps, _state);
85  if (backend == nullptr) {
86  debug_gps_uavcan(2,
87  _detected_modules[i].ap_uavcan->get_driver_index(),
88  "Failed to register UAVCAN GPS Node %d on Bus %d\n",
89  _detected_modules[i].node_id,
90  _detected_modules[i].ap_uavcan->get_driver_index());
91  } else {
92  _detected_modules[i].driver = backend;
93  backend->_detected_module = i;
94  debug_gps_uavcan(2,
95  _detected_modules[i].ap_uavcan->get_driver_index(),
96  "Registered UAVCAN GPS Node %d on Bus %d\n",
97  _detected_modules[i].node_id,
98  _detected_modules[i].ap_uavcan->get_driver_index());
99  }
100  break;
101  }
102  }
103  return backend;
104 }

106 AP_GPS_UAVCAN* AP_GPS_UAVCAN::get_uavcan_backend(AP_UAVCAN* ap_uavcan, uint8_t node_id)
107 {
108  if (ap_uavcan == nullptr) {
109  return nullptr;
110  }
111 
112  for (uint8_t i = 0; i < GPS_MAX_RECEIVERS; i++) {
113  if (_detected_modules[i].driver != nullptr &&
114  _detected_modules[i].ap_uavcan == ap_uavcan &&
115  _detected_modules[i].node_id == node_id) {
116  return _detected_modules[i].driver;
117  }
118  }
119 
120  bool already_detected = false;
121  // Check if there's an empty spot for possible registeration
122  for (uint8_t i = 0; i < GPS_MAX_RECEIVERS; i++) {
123  if (_detected_modules[i].ap_uavcan == ap_uavcan && _detected_modules[i].node_id == node_id) {
124  // Already Detected
125  already_detected = true;
126  break;
127  }
128  }
129  if (!already_detected) {
130  for (uint8_t i = 0; i < GPS_MAX_RECEIVERS; i++) {
131  if (_detected_modules[i].ap_uavcan == nullptr) {
132  _detected_modules[i].ap_uavcan = ap_uavcan;
133  _detected_modules[i].node_id = node_id;
134  break;
135  }
136  }
137  }
138  return nullptr;
139 }

141 void AP_GPS_UAVCAN::handle_fix_msg(const FixCb &cb)
142 {
143  bool process = false;
144 
145  WITH_SEMAPHORE(sem);
146 
147  if (cb.msg->status == uavcan::equipment::gnss::Fix::STATUS_NO_FIX) {
148  interim_state.status = AP_GPS::GPS_Status::NO_FIX;
149  } else {
150  if (cb.msg->status == uavcan::equipment::gnss::Fix::STATUS_TIME_ONLY) {
151  interim_state.status = AP_GPS::GPS_Status::NO_FIX;
152  } else if (cb.msg->status == uavcan::equipment::gnss::Fix::STATUS_2D_FIX) {
153  interim_state.status = AP_GPS::GPS_Status::GPS_OK_FIX_2D;
154  process = true;
155  } else if (cb.msg->status == uavcan::equipment::gnss::Fix::STATUS_3D_FIX) {
156  interim_state.status = AP_GPS::GPS_Status::GPS_OK_FIX_3D;
157  process = true;
158  }
159 
160  if (cb.msg->gnss_time_standard == uavcan::equipment::gnss::Fix::GNSS_TIME_STANDARD_UTC) {
161  uint64_t epoch_ms = uavcan::UtcTime(cb.msg->gnss_timestamp).toUSec();
162  epoch_ms /= 1000;
163  uint64_t gps_ms = epoch_ms - UNIX_OFFSET_MSEC;
164  interim_state.time_week = (uint16_t)(gps_ms / AP_MSEC_PER_WEEK);
165  interim_state.time_week_ms = (uint32_t)(gps_ms - (interim_state.time_week) * AP_MSEC_PER_WEEK);
166  }
167  }
168 
169  if (process) {
170  Location loc = { };
171  loc.lat = cb.msg->latitude_deg_1e8 / 10;
172  loc.lng = cb.msg->longitude_deg_1e8 / 10;
173  loc.alt = cb.msg->height_msl_mm / 10;
174  interim_state.location = loc;
175 
176  if (!uavcan::isNaN(cb.msg->ned_velocity[0])) {
177  Vector3f vel(cb.msg->ned_velocity[0], cb.msg->ned_velocity[1], cb.msg->ned_velocity[2]);
178  interim_state.velocity = vel;
179  interim_state.ground_speed = norm(vel.x, vel.y);
180  interim_state.ground_course = wrap_360(degrees(atan2f(vel.y, vel.x)));
181  interim_state.have_vertical_velocity = true;
182  } else {
183  interim_state.have_vertical_velocity = false;
184  }
185 
186  float pos_cov[9];
187  cb.msg->position_covariance.unpackSquareMatrix(pos_cov);
188  if (!uavcan::isNaN(pos_cov[8])) {
189  if (pos_cov[8] > 0) {
190  interim_state.vertical_accuracy = sqrtf(pos_cov[8]);
191  interim_state.have_vertical_accuracy = true;
192  } else {
193  interim_state.have_vertical_accuracy = false;
194  }
195  } else {
196  interim_state.have_vertical_accuracy = false;
197  }
198 
199  const float horizontal_pos_variance = MAX(pos_cov[0], pos_cov[4]);
200  if (!uavcan::isNaN(horizontal_pos_variance)) {
201  if (horizontal_pos_variance > 0) {
202  interim_state.horizontal_accuracy = sqrtf(horizontal_pos_variance);
203  interim_state.have_horizontal_accuracy = true;
204  } else {
205  interim_state.have_horizontal_accuracy = false;
206  }
207  } else {
208  interim_state.have_horizontal_accuracy = false;
209  }
210 
211  float vel_cov[9];
212  cb.msg->velocity_covariance.unpackSquareMatrix(vel_cov);
213  if (!uavcan::isNaN(vel_cov[0])) {
214  interim_state.speed_accuracy = sqrtf((vel_cov[0] + vel_cov[4] + vel_cov[8]) / 3.0);
215  interim_state.have_speed_accuracy = true;
216  } else {
217  interim_state.have_speed_accuracy = false;
218  }
219 
220  interim_state.num_sats = cb.msg->sats_used;
221  } else {
222  interim_state.have_vertical_velocity = false;
223  interim_state.have_vertical_accuracy = false;
224  interim_state.have_horizontal_accuracy = false;
225  interim_state.have_speed_accuracy = false;
226  interim_state.num_sats = 0;
227  }
228 
229  interim_state.last_gps_time_ms = AP_HAL::millis();
230 
231  _new_data = true;
232 }
