665 // correct the height of the EKF origin to be consistent with GPS Data using a Bayes filter.
666 void NavEKF2_core::correctEkfOriginHeight()
667 {
668  // Estimate the WGS-84 height of the EKF's origin using a Bayes filter
669 
670  // calculate the variance of our a-priori estimate of the ekf origin height
671  float deltaTime = constrain_float(0.001f * (imuDataDelayed.time_ms - lastOriginHgtTime_ms), 0.0f, 1.0f);
672  if (activeHgtSource == HGT_SOURCE_BARO) {
673  // Use the baro drift rate
674  const float baroDriftRate = 0.05f;
675  ekfOriginHgtVar += sq(baroDriftRate * deltaTime);
676  } else if (activeHgtSource == HGT_SOURCE_RNG) {
677  // use the worse case expected terrain gradient and vehicle horizontal speed
678  const float maxTerrGrad = 0.25f;
679  ekfOriginHgtVar += sq(maxTerrGrad * norm(stateStruct.velocity.x , stateStruct.velocity.y) * deltaTime);
680  } else {
681  // by definition our height source is absolute so cannot run this filter
682  return;
683  }
684  lastOriginHgtTime_ms = imuDataDelayed.time_ms;
685 
686  // calculate the observation variance assuming EKF error relative to datum is independent of GPS observation error
687  // when not using GPS as height source
688  float originHgtObsVar = sq(gpsHgtAccuracy) + P[8][8];
689 
690  // calculate the correction gain
691  float gain = ekfOriginHgtVar / (ekfOriginHgtVar + originHgtObsVar);
692 
693  // calculate the innovation
694  float innovation = - stateStruct.position.z - gpsDataDelayed.hgt;
695 
696  // check the innovation variance ratio
697  float ratio = sq(innovation) / (ekfOriginHgtVar + originHgtObsVar);
698 
699  // correct the EKF origin and variance estimate if the innovation is less than 5-sigma
700  if (ratio < 25.0f && gpsAccuracyGood) {
701  ekfGpsRefHgt -= (double)(gain * innovation);
702  ekfOriginHgtVar -= MAX(gain * ekfOriginHgtVar , 0.0f);
703  }
704 }
