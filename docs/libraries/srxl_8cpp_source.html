109 static int srxl_channels_get_v1v2(uint16_t max_values, uint8_t *num_values, uint16_t *values, bool *failsafe_state)
110 {
111  uint8_t loop;
112  uint32_t channel_raw_value;
113 
114  *num_values = (uint8_t)((frame_len_full - 3U)/2U);
115  *failsafe_state = 0U; /* this protocol version does not support failsafe information */
116 
117  /* get data channel data from frame */
118  for (loop=0U; loop < *num_values; loop++) {
119  channel_raw_value = ((((uint32_t)buffer[loop*2U+1U])& 0x0000000FU) << 8U) | ((uint32_t)(buffer[loop*2U+2U])); /* get 12bit channel raw value from srxl datastream (mask out unused bits with 0x0000000F) */
120  channels[loop] = (uint16_t)(((channel_raw_value * (uint32_t)1400U) >> 12U) + (uint32_t)800U); /* convert raw value to servo/esc signal pulsewidth in us */
121  }
122 
123  /* provide channel data to FMU */
124  if ( (uint16_t)*num_values > max_values) {
125  *num_values = (uint8_t)max_values;
126  }
127  memcpy(values, channels, (*num_values)*2);
128 
129  return 0; /* for srxl protocol version 1 and 2 it is not expected, that any error happen during decode process */
130  }

155 static int srxl_channels_get_v5(uint16_t max_values, uint8_t *num_values, uint16_t *values, bool *failsafe_state)
156 {
157  // up to 7 channel values per packet. Each channel value is 16
158  // bits, with 11 bits of data and 4 bits of channel number. The
159  // top bit indicates a special X-Plus channel
160  for (uint8_t i=0; i<7; i++) {
161  uint16_t b = buffer[i*2+2] << 8 | buffer[i*2+3];
162  uint16_t c = b >> 11; // channel number
163  int32_t v = b & 0x7FF;
164  if (b & 0x8000) {
165  continue;
166  }
167  if (c == 12) {
168  // special handling for channel 12
169  // see http://www.deviationtx.com/forum/protocol-development/2088-18-channels-for-dsm2-dsmx?start=40
170  //printf("c12: 0x%x %02x %02x\n", (unsigned)(b>>9), (unsigned)buffer[0], (unsigned)buffer[1]);
171  v = (b & 0x1FF) << 2;
172  c = 10 + ((b >> 9) & 0x7);
173  if (buffer[1] & 1) {
174  c += 4;
175  }
176 #if 0
177  printf("b=0x%04x v=%u c=%u b[1]=0x%02x\n",
178  (unsigned)b, (unsigned)v, (unsigned)c, (unsigned)buffer[1]);
179 #endif
180  } else if (c > 12) {
181  // invalid
182  v = 0;
183  }
184 
185  // if channel number if greater than 16 then it is a X-Plus
186  // channel. We don't yet know how to decode those. There is some information here:
187  // http://www.deviationtx.com/forum/protocol-development/2088-18-channels-for-dsm2-dsmx?start=40
188  // but we really need some sample data to confirm
189  if (c < SRXL_MAX_CHANNELS) {
190  v = (((v - 0x400) * 500) / 876) + 1500;
191  channels[c] = v;
192  if (c >= max_channels) {
193  max_channels = c+1;
194  }
195  }
196 
197  //printf("%u:%u ", (unsigned)c, (unsigned)v);
198  }
199  //printf("\n");
200 
201  *num_values = max_channels;
202  if (*num_values > max_values) {
203  *num_values = max_values;
204  }
205  memcpy(values, channels, (*num_values)*2);
206 
207  // check failsafe bit, this goes low when connection to the
208  // transmitter is lost
209  *failsafe_state = ((buffer[1] & 2) == 0);
210 
211  // success
212  return 0;
213  }

238 int srxl_decode(uint64_t timestamp_us, uint8_t byte, uint8_t *num_values, uint16_t *values, uint16_t max_values, bool *failsafe_state)
239 {
240  int ret = 1;
241 
242  /*----------------------------------------distinguish different srxl variants at the beginning of each frame---------------------------------------------- */
243  /* Check if we have a new begin of a frame --> indicators: Time gap in datastream + SRXL header 0xA<VARIANT>*/
244  if ( (timestamp_us - last_data_us) >= SRXL_MIN_FRAMESPACE_US) {
245  /* Now detect SRXL variant based on header */
246  switch(byte) {
247  case SRXL_HEADER_V1:
248  frame_len_full = SRXL_FRAMELEN_V1;
249  frame_header = SRXL_HEADER_V1;
250  decode_state = STATE_NEW;
251  break;
252  case SRXL_HEADER_V2:
253  frame_len_full = SRXL_FRAMELEN_V2;
254  frame_header = SRXL_HEADER_V2;
255  decode_state = STATE_NEW;
256  break;
257  case SRXL_HEADER_V5:
258  frame_len_full = SRXL_FRAMELEN_V5;
259  frame_header = SRXL_HEADER_V5;
260  decode_state = STATE_NEW;
261  break;
262  default:
263  frame_len_full = 0U;
264  frame_header = SRXL_HEADER_NOT_IMPL;
265  decode_state = STATE_IDLE;
266  buflen = 0;
267  return 2; /* protocol version not implemented --> no channel data --> unknown packet */
268  }
269  }
270 
271 
272  /*--------------------------------------------collect all data from stream and decode-------------------------------------------------------*/
273  switch (decode_state) {
274  case STATE_NEW: /* buffer header byte and prepare for frame reception and decoding */
275  buffer[0U]=byte;
276  crc_fmu = crc_xmodem_update(0U,byte);
277  buflen = 1U;
278  decode_state_next = STATE_COLLECT;
279  break;
280 
281  case STATE_COLLECT: /* receive all bytes. After reception decode frame and provide rc channel information to FMU */
282  if (buflen >= frame_len_full) {
283  // a logic bug in the state machine, this shouldn't happen
284  decode_state = STATE_IDLE;
285  buflen = 0;
286  frame_len_full = 0;
287  frame_header = SRXL_HEADER_NOT_IMPL;
288  return 2;
289  }
290  buffer[buflen] = byte;
291  buflen++;
292  /* CRC not over last 2 frame bytes as these bytes inhabitate the crc */
293  if (buflen <= (frame_len_full-2)) {
294  crc_fmu = crc_xmodem_update(crc_fmu,byte);
295  }
296  if( buflen == frame_len_full ) {
297  /* CRC check here */
298  crc_receiver = ((uint16_t)buffer[buflen-2] << 8U) | ((uint16_t)buffer[buflen-1]);
299  if (crc_receiver == crc_fmu) {
300  /* at this point buffer contains all frame data and crc is valid --> extract channel info according to SRXL variant */
301  switch (frame_header) {
302  case SRXL_HEADER_V1:
303  ret = srxl_channels_get_v1v2(max_values, num_values, values, failsafe_state);
304  break;
305  case SRXL_HEADER_V2:
306  ret = srxl_channels_get_v1v2(max_values, num_values, values, failsafe_state);
307  break;
308  case SRXL_HEADER_V5:
309  ret = srxl_channels_get_v5(max_values, num_values, values, failsafe_state);
310  break;
311  default:
312  ret = 2; /* protocol version not implemented --> no channel data */
313  break;
314  }
315  }
316  else {
317  ret = 4; /* CRC fail --> no channel data */
318  }
319  decode_state_next = STATE_IDLE; /* frame data buffering and decoding finished --> statemachine not in use until new header drops is */
320  }
321  else {
322  /* frame not completely received --> frame data buffering still ongoing */
323  decode_state_next = STATE_COLLECT;
324  }
325  break;
326 
327  default:
328  ret = 1; /* STATE_IDLE --> do nothing */
329  break;
330  } /* switch (decode_state) */
331 
332  decode_state = decode_state_next;
333  last_data_us = timestamp_us;
334  return ret;
335 }
