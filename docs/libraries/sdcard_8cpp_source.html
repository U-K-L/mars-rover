44 /*
45  initialise microSD card if avaialble. This is called during
46  AP_BoardConfig initialisation. The parameter BRD_SD_SLOWDOWN
47  controls a scaling factor on the microSD clock
48  */
49 bool sdcard_init()
50 {
51 #ifdef USE_POSIX
52  WITH_SEMAPHORE(sem);
53 
54  uint8_t sd_slowdown = AP_BoardConfig::get_sdcard_slowdown();
55 #if HAL_USE_SDC
56 
57  if (SDCD1.bouncebuffer == nullptr) {
58  bouncebuffer_init(&SDCD1.bouncebuffer, 512, true);
59  }
60 
61  if (sdcard_running) {
62  sdcard_stop();
63  }
64 
65  const uint8_t tries = 3;
66  for (uint8_t i=0; i<tries; i++) {
67  sdcconfig.slowdown = sd_slowdown;
68  sdcStart(&SDCD1, &sdcconfig);
69  if(sdcConnect(&SDCD1) == HAL_FAILED) {
70  sdcStop(&SDCD1);
71  continue;
72  }
73  if (f_mount(&SDC_FS, "/", 1) != FR_OK) {
74  sdcDisconnect(&SDCD1);
75  sdcStop(&SDCD1);
76  continue;
77  }
78  printf("Successfully mounted SDCard (slowdown=%u)\n", (unsigned)sd_slowdown);
79 
80  // Create APM Directory if needed
81  mkdir("/APM", 0777);
82  sdcard_running = true;
83  return true;
84  }
85 #elif HAL_USE_MMC_SPI
86  if (sdcard_running) {
87  sdcard_stop();
88  }
89 
90  sdcard_running = true;
91 
92  device = AP_HAL::get_HAL().spi->get_device("sdcard");
93  if (!device) {
94  printf("No sdcard SPI device found\n");
95  sdcard_running = false;
96  return false;
97  }
98  device->set_slowdown(sd_slowdown);
99 
100  mmcObjectInit(&MMCD1);
101 
102  mmcconfig.spip =
103  static_cast<ChibiOS::SPIDevice*>(device.get())->get_driver();
104  mmcconfig.hscfg = &highspeed;
105  mmcconfig.lscfg = &lowspeed;
106 
107  /*
108  try up to 3 times to init microSD interface
109  */
110  const uint8_t tries = 3;
111  for (uint8_t i=0; i<tries; i++) {
112  mmcStart(&MMCD1, &mmcconfig);
113 
114  if (mmcConnect(&MMCD1) == HAL_FAILED) {
115  mmcStop(&MMCD1);
116  continue;
117  }
118  if (f_mount(&SDC_FS, "/", 1) != FR_OK) {
119  mmcDisconnect(&MMCD1);
120  mmcStop(&MMCD1);
121  continue;
122  }
123  printf("Successfully mounted SDCard (slowdown=%u)\n", (unsigned)sd_slowdown);
124 
125  // Create APM Directory if needed
126  mkdir("/APM", 0777);
127  return true;
128  }
129 #endif
130  sdcard_running = false;
131 #endif
132  return false;
133 }
