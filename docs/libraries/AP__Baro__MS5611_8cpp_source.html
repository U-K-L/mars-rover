173 bool AP_Baro_MS56XX::_read_prom_5611(uint16_t prom[8])
174 {
175  /*
176  * MS5611-01BA datasheet, CYCLIC REDUNDANCY CHECK (CRC): "MS5611-01BA
177  * contains a PROM memory with 128-Bit. A 4-bit CRC has been implemented
178  * to check the data validity in memory."
179  *
180  * CRC field must me removed for CRC-4 calculation.
181  */
182  bool all_zero = true;
183  for (uint8_t i = 0; i < 8; i++) {
184  prom[i] = _read_prom_word(i);
185  if (prom[i] != 0) {
186  all_zero = false;
187  }
188  }
189 
190  if (all_zero) {
191  return false;
192  }
193 
194  /* save the read crc */
195  const uint16_t crc_read = prom[7] & 0xf;
196 
197  /* remove CRC byte */
198  prom[7] &= 0xff00;
199 
200  return crc_read == crc_crc4(prom);
201 }

203 bool AP_Baro_MS56XX::_read_prom_5637(uint16_t prom[8])
204 {
205  /*
206  * MS5637-02BA03 datasheet, CYCLIC REDUNDANCY CHECK (CRC): "MS5637
207  * contains a PROM memory with 112-Bit. A 4-bit CRC has been implemented
208  * to check the data validity in memory."
209  *
210  * 8th PROM word must be zeroed and CRC field removed for CRC-4
211  * calculation.
212  */
213  bool all_zero = true;
214  for (uint8_t i = 0; i < 7; i++) {
215  prom[i] = _read_prom_word(i);
216  if (prom[i] != 0) {
217  all_zero = false;
218  }
219  }
220 
221  if (all_zero) {
222  return false;
223  }
224 
225  prom[7] = 0;
226 
227  /* save the read crc */
228  const uint16_t crc_read = (prom[0] & 0xf000) >> 12;
229 
230  /* remove CRC byte */
231  prom[0] &= ~0xf000;
232 
233  return crc_read == crc_crc4(prom);
234 }

236 /*
237  * Read the sensor with a state machine
238  * We read one time temperature (state=0) and then 4 times pressure (states 1-4)
239  *
240  * Temperature is used to calculate the compensated pressure and doesn't vary
241  * as fast as pressure. Hence we reuse the same temperature for 4 samples of
242  * pressure.
243 */
244 void AP_Baro_MS56XX::_timer(void)
245 {
246  uint8_t next_cmd;
247  uint8_t next_state;
248  uint32_t adc_val = _read_adc();
249 
250  /*
251  * If read fails, re-initiate a read command for current state or we are
252  * stuck
253  */
254  if (adc_val == 0) {
255  next_state = _state;
256  } else {
257  next_state = (_state + 1) % 5;
258  }
259 
260  next_cmd = next_state == 0 ? ADDR_CMD_CONVERT_TEMPERATURE
261  : ADDR_CMD_CONVERT_PRESSURE;
262  if (!_dev->transfer(&next_cmd, 1, nullptr, 0)) {
263  return;
264  }
265 
266  /* if we had a failed read we are all done */
267  if (adc_val == 0) {
268  // a failed read can mean the next returned value will be
269  // corrupt, we must discard it
270  _discard_next = true;
271  return;
272  }
273 
274  if (_discard_next) {
275  _discard_next = false;
276  _state = next_state;
277  return;
278  }
279 
280  WITH_SEMAPHORE(_sem);
281 
282  if (_state == 0) {
283  _update_and_wrap_accumulator(&_accum.s_D2, adc_val,
284  &_accum.d2_count, 32);
285  } else if (pressure_ok(adc_val)) {
286  _update_and_wrap_accumulator(&_accum.s_D1, adc_val,
287  &_accum.d1_count, 128);
288  }
289 
290  _state = next_state;
291 }

308

345 // Calculate Temperature and compensated Pressure in real units (Celsius degrees*100, mbar*100).
346 void AP_Baro_MS56XX::_calculate_5611()
347 {
348  float dT;
349  float TEMP;
350  float OFF;
351  float SENS;
352 
353  // Formulas from manufacturer datasheet
354  // sub -15c temperature compensation is not included
355 
356  // we do the calculations using floating point allows us to take advantage
357  // of the averaging of D1 and D1 over multiple samples, giving us more
358  // precision
359  dT = _D2-(((uint32_t)_cal_reg.c5)<<8);
360  TEMP = (dT * _cal_reg.c6)/8388608;
361  OFF = _cal_reg.c2 * 65536.0f + (_cal_reg.c4 * dT) / 128;
362  SENS = _cal_reg.c1 * 32768.0f + (_cal_reg.c3 * dT) / 256;
363 
364  if (TEMP < 0) {
365  // second order temperature compensation when under 20 degrees C
366  float T2 = (dT*dT) / 0x80000000;
367  float Aux = TEMP*TEMP;
368  float OFF2 = 2.5f*Aux;
369  float SENS2 = 1.25f*Aux;
370  TEMP = TEMP - T2;
371  OFF = OFF - OFF2;
372  SENS = SENS - SENS2;
373  }
374 
375  float pressure = (_D1*SENS/2097152 - OFF)/32768;
376  float temperature = (TEMP + 2000) * 0.01f;
377  _copy_to_frontend(_instance, pressure, temperature);
378 }

415 // Calculate Temperature and compensated Pressure in real units (Celsius degrees*100, mbar*100).
416 void AP_Baro_MS56XX::_calculate_5637()
417 {
418  int32_t dT, TEMP;
419  int64_t OFF, SENS;
420  int32_t raw_pressure = _D1;
421  int32_t raw_temperature = _D2;
422 
423  // Formulas from manufacturer datasheet
424  // sub -15c temperature compensation is not included
425 
426  dT = raw_temperature - (((uint32_t)_cal_reg.c5) << 8);
427  TEMP = 2000 + ((int64_t)dT * (int64_t)_cal_reg.c6) / 8388608;
428  OFF = (int64_t)_cal_reg.c2 * (int64_t)131072 + ((int64_t)_cal_reg.c4 * (int64_t)dT) / (int64_t)64;
429  SENS = (int64_t)_cal_reg.c1 * (int64_t)65536 + ((int64_t)_cal_reg.c3 * (int64_t)dT) / (int64_t)128;
430 
431  if (TEMP < 2000) {
432  // second order temperature compensation when under 20 degrees C
433  int32_t T2 = ((int64_t)3 * ((int64_t)dT * (int64_t)dT) / (int64_t)8589934592);
434  int64_t aux = (TEMP - 2000) * (TEMP - 2000);
435  int64_t OFF2 = 61 * aux / 16;
436  int64_t SENS2 = 29 * aux / 16;
437 
438  TEMP = TEMP - T2;
439  OFF = OFF - OFF2;
440  SENS = SENS - SENS2;
441  }
442 
443  int32_t pressure = ((int64_t)raw_pressure * SENS / (int64_t)2097152 - OFF) / (int64_t)32768;
444  float temperature = TEMP * 0.01f;
445  _copy_to_frontend(_instance, (float)pressure, temperature);
446 }
