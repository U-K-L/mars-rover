34 uint8_t AP_BattMonitor_Backend::capacity_remaining_pct() const
35 {

37  if ( _params._pack_capacity > 10 ) { // a very very small battery
38  return MIN(MAX((100 * (mah_remaining) / _params._pack_capacity), 0), UINT8_MAX);

40  return 0;

44 // update battery resistance estimate
45 // faster rates of change of the current and voltage readings cause faster updates to the resistance estimate
46 // the battery resistance is calculated by comparing the latest current and voltage readings to a low-pass filtered current and voltage
47 // high current steps are integrated into the resistance estimate by varying the time constant of the resistance filter
48 void AP_BattMonitor_Backend::update_resistance_estimate()
49 {
50  // return immediately if no current
51  if (!has_current() || !is_positive(_state.current_amps)) {
52  return;
53  }
54 
55  // update maximum current seen since startup and protect against divide by zero
56  _current_max_amps = MAX(_current_max_amps, _state.current_amps);
57  float current_delta = _state.current_amps - _current_filt_amps;
58  if (is_zero(current_delta)) {
59  return;
60  }
61 
62  // update reference voltage and current
63  if (_state.voltage > _resistance_voltage_ref) {
64  _resistance_voltage_ref = _state.voltage;
65  _resistance_current_ref = _state.current_amps;
66  }
67 
68  // calculate time since last update
69  uint32_t now = AP_HAL::millis();
70  float loop_interval = (now - _resistance_timer_ms) / 1000.0f;
71  _resistance_timer_ms = now;
72 
73  // estimate short-term resistance
74  float filt_alpha = constrain_float(loop_interval/(loop_interval + AP_BATT_MONITOR_RES_EST_TC_1), 0.0f, 0.5f);
75  float resistance_alpha = MIN(1, AP_BATT_MONITOR_RES_EST_TC_2*fabsf((_state.current_amps-_current_filt_amps)/_current_max_amps));
76  float resistance_estimate = -(_state.voltage-_voltage_filt)/current_delta;
77  if (is_positive(resistance_estimate)) {
78  _state.resistance = _state.resistance*(1-resistance_alpha) + resistance_estimate*resistance_alpha;
79  }
80 
81  // calculate maximum resistance
82  if ((_resistance_voltage_ref > _state.voltage) && (_state.current_amps > _resistance_current_ref)) {
83  float resistance_max = (_resistance_voltage_ref - _state.voltage) / (_state.current_amps - _resistance_current_ref);
84  _state.resistance = MIN(_state.resistance, resistance_max);
85  }
86 
87  // update the filtered voltage and currents
88  _voltage_filt = _voltage_filt*(1-filt_alpha) + _state.voltage*filt_alpha;
89  _current_filt_amps = _current_filt_amps*(1-filt_alpha) + _state.current_amps*filt_alpha;
90 
91  // update estimated voltage without sag
92  _state.voltage_resting_estimate = _state.voltage + _state.current_amps * _state.resistance;
93 }

101 AP_BattMonitor::BatteryFailsafe AP_BattMonitor_Backend::update_failsafes(void)
102 {
103  const uint32_t now = AP_HAL::millis();
104 
105  bool low_voltage, low_capacity, critical_voltage, critical_capacity;
106  check_failsafe_types(low_voltage, low_capacity, critical_voltage, critical_capacity);
107 
108  if (critical_voltage) {
109  // this is the first time our voltage has dropped below minimum so start timer
110  if (_state.critical_voltage_start_ms == 0) {
111  _state.critical_voltage_start_ms = now;
112  } else if (_params._low_voltage_timeout > 0 &&
113  now - _state.critical_voltage_start_ms > uint32_t(_params._low_voltage_timeout)*1000U) {
114  return AP_BattMonitor::BatteryFailsafe_Critical;
115  }
116  } else {
117  // acceptable voltage so reset timer
118  _state.critical_voltage_start_ms = 0;
119  }
120 
121  if (critical_capacity) {
122  return AP_BattMonitor::BatteryFailsafe_Critical;
123  }
124 
125  if (low_voltage) {
126  // this is the first time our voltage has dropped below minimum so start timer
127  if (_state.low_voltage_start_ms == 0) {
128  _state.low_voltage_start_ms = now;
129  } else if (_params._low_voltage_timeout > 0 &&
130  now - _state.low_voltage_start_ms > uint32_t(_params._low_voltage_timeout)*1000U) {
131  return AP_BattMonitor::BatteryFailsafe_Low;
132  }
133  } else {
134  // acceptable voltage so reset timer
135  _state.low_voltage_start_ms = 0;
136  }
137 
138  if (low_capacity) {
139  return AP_BattMonitor::BatteryFailsafe_Low;
140  }
141 
142  // if we've gotten this far then battery is ok
143  return AP_BattMonitor::BatteryFailsafe_None;
144 }

155 bool AP_BattMonitor_Backend::arming_checks(char * buffer, size_t buflen) const
156 {
157  bool low_voltage, low_capacity, critical_voltage, critical_capacity;
158  check_failsafe_types(low_voltage, low_capacity, critical_voltage, critical_capacity);
159 
160  bool below_arming_voltage = is_positive(_params._arming_minimum_voltage) &&
161  (_state.voltage < _params._arming_minimum_voltage);
162  bool below_arming_capacity = (_params._arming_minimum_capacity > 0) &&
163  ((_params._pack_capacity - _state.consumed_mah) < _params._arming_minimum_capacity);
164 
165  bool result = update_check(buflen, buffer, low_voltage, "low voltage failsafe");
166  result = result && update_check(buflen, buffer, low_capacity, "low capacity failsafe");
167  result = result && update_check(buflen, buffer, critical_voltage, "critical voltage failsafe");
168  result = result && update_check(buflen, buffer, critical_capacity, "critical capacity failsafe");
169  result = result && update_check(buflen, buffer, below_arming_voltage, "below minimum arming voltage");
170  result = result && update_check(buflen, buffer, below_arming_capacity, "below minimum arming capacity");
171 
172  return result;
173 }

175 void AP_BattMonitor_Backend::check_failsafe_types(bool &low_voltage, bool &low_capacity, bool &critical_voltage, bool &critical_capacity) const
176 {
177  // use voltage or sag compensated voltage
178  float voltage_used;
179  switch (_params.failsafe_voltage_source()) {
180  case AP_BattMonitor_Params::BattMonitor_LowVoltageSource_Raw:
181  default:
182  voltage_used = _state.voltage;
183  break;
184  case AP_BattMonitor_Params::BattMonitor_LowVoltageSource_SagCompensated:
185  voltage_used = voltage_resting_estimate();
186  break;
187  }
188 
189  // check critical battery levels
190  if ((voltage_used > 0) && (_params._critical_voltage > 0) && (voltage_used < _params._critical_voltage)) {
191  critical_voltage = true;
192  } else {
193  critical_voltage = false;
194  }
195 
196  // check capacity failsafe if current monitoring is enabled
197  if (has_current() && (_params._critical_capacity > 0) &&
198  ((_params._pack_capacity - _state.consumed_mah) < _params._critical_capacity)) {
199  critical_capacity = true;
200  } else {
201  critical_capacity = false;
202  }
203 
204  if ((voltage_used > 0) && (_params._low_voltage > 0) && (voltage_used < _params._low_voltage)) {
205  low_voltage = true;
206  } else {
207  low_voltage = false;
208  }
209 
210  // check capacity if current monitoring is enabled
211  if (has_current() && (_params._low_capacity > 0) &&
212  ((_params._pack_capacity - _state.consumed_mah) < _params._low_capacity)) {
213  low_capacity = true;
214  } else {
215  low_capacity = false;
216  }
217 }

219 /*
220  default implementation for reset_remaining(). This sets consumed_wh
221  and consumed_mah based on the given percentage. Use percentage=100
222  for a full battery
223 */
224 bool AP_BattMonitor_Backend::reset_remaining(float percentage)
225 {
226  percentage = constrain_float(percentage, 0, 100);
227  const float used_proportion = (100 - percentage) * 0.01;
228  _state.consumed_mah = used_proportion * _params._pack_capacity;
229  // without knowing the history we can't do consumed_wh
230  // accurately. Best estimate is based on current voltage. This
231  // will be good when resetting the battery to a value close to
232  // full charge
233  _state.consumed_wh = _state.consumed_mah * 1000 * _state.voltage;
234 
235  // reset failsafe state for this backend
236  _state.failsafe = update_failsafes();
237 
238  return true;
239 }
