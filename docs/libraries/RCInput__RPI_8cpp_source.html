122 Memory_table::Memory_table(uint32_t page_count, int version)
123 {
124  uint32_t i;
125  int fdMem, file;
126  // Cache coherent adresses depends on RPI's version
127  uint32_t bus = version == 1 ? 0x40000000 : 0xC0000000;
128  uint64_t pageInfo;
129  void *offset;
130 
131  _virt_pages = (void **)calloc(page_count, sizeof(void *));
132  _phys_pages = (void **)calloc(page_count, sizeof(void *));
133  _page_count = page_count;
134 
135  if ((fdMem = open("/dev/mem", O_RDWR | O_SYNC | O_CLOEXEC)) < 0) {
136  fprintf(stderr, "Failed to open /dev/mem\n");
137  exit(-1);
138  }
139 
140  if ((file = open("/proc/self/pagemap", O_RDWR | O_SYNC | O_CLOEXEC)) < 0) {
141  fprintf(stderr, "Failed to open /proc/self/pagemap\n");
142  exit(-1);
143  }
144 
145  // Magic to determine the physical address for this page:
146  offset = mmap(0, _page_count * PAGE_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS | MAP_NORESERVE | MAP_LOCKED, -1, 0);
147  lseek(file, ((uintptr_t)offset) / PAGE_SIZE * 8, SEEK_SET);
148 
149  // Get list of available cache coherent physical addresses
150  for (i = 0; i < _page_count; i++) {
151  _virt_pages[i] = mmap(0, PAGE_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_ANONYMOUS | MAP_NORESERVE | MAP_LOCKED, -1, 0);
152  if (::read(file, &pageInfo, 8) < 8) {
153  fprintf(stderr, "Failed to read pagemap\n");
154  exit(-1);
155  }
156  _phys_pages[i] = (void *)((uintptr_t)(pageInfo * PAGE_SIZE) | bus);
157  }
158 
159  // Map physical addresses to virtual memory
160  for (i = 0; i < _page_count; i++) {
161  munmap(_virt_pages[i], PAGE_SIZE);
162  _virt_pages[i] = mmap(_virt_pages[i], PAGE_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED | MAP_FIXED | MAP_NORESERVE | MAP_LOCKED, fdMem, ((uintptr_t)_phys_pages[i] & (version == 1 ? 0xFFFFFFFF : ~bus)));
163  memset(_virt_pages[i], 0xee, PAGE_SIZE);
164  }
165  close(file);

284 // This function is used to init DMA control blocks (setting sampling GPIO
285 // register, destination adresses, synchronization)
286 void RCInput_RPI::init_ctrl_data()
287 {
288  uint32_t phys_fifo_addr;
289  uint32_t dest = 0;
290  uint32_t cbp = 0;
291  dma_cb_t *cbp_curr;
292  // Set fifo addr (for delay)
293  phys_fifo_addr = ((pcm_base + 0x04) & 0x00FFFFFF) | 0x7e000000;
294 
295  // Init dma control blocks.
296  /* We are transferring 8 bytes of GPIO register. Every 7th iteration we are
297  sampling TIMER register, which length is 8 bytes. So, for every 7 samples of GPIO we need
298  7 * 8 + 8 = 64 bytes of buffer. Value 7 was selected specially to have a 64-byte "block"
299  TIMER - GPIO. So, we have integer count of such "blocks" at one virtual page. (4096 / 64 = 64
300  "blocks" per page. As minimum, we must have 2 virtual pages of buffer (to have integer count of
301  vitual pages for control blocks): for every 7 iterations (64 bytes of buffer) we need 7 control blocks for GPIO
302  sampling, 7 control blocks for setting frequency and 1 control block for sampling timer, so,
303  we need 7 + 7 + 1 = 15 control blocks. For integer value, we need 15 pages of control blocks.
304  Each control block length is 32 bytes. In 15 pages we will have (15 * 4096 / 32) = 15 * 128 control
305  blocks. 15 * 128 control blocks = 64 * 128 bytes of buffer = 2 pages of buffer.
306  So, for 7 * 64 * 2 iteration we init DMA for sampling GPIO
307  and timer to ((7 * 8 + 8) * 64 * 2) = 8192 bytes = 2 pages of buffer.
308  */
309 
310  for (uint32_t i = 0; i < 7 * 128 * RCIN_RPI_BUFFER_LENGTH; i++) {
311  // Transfer timer every 7th sample
312  if (i % 7 == 0) {
313  cbp_curr = (dma_cb_t*)con_blocks->get_page(con_blocks->_virt_pages, cbp);
314 
315  init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP | RCIN_RPI_DMA_DEST_INC | RCIN_RPI_DMA_SRC_INC, RCIN_RPI_TIMER_BASE,
316  (uintptr_t)circle_buffer->get_page(circle_buffer->_phys_pages, dest),
317  8,
318  0,
319  (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages,
320  cbp + sizeof(dma_cb_t)));
321  dest += 8;
322  cbp += sizeof(dma_cb_t);
323  }
324 
325  // Transfer GPIO (8 bytes)
326  cbp_curr = (dma_cb_t *)con_blocks->get_page(con_blocks->_virt_pages, cbp);
327  init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP, RCIN_RPI_GPIO_LEV0_ADDR,
328  (uintptr_t)circle_buffer->get_page(circle_buffer->_phys_pages, dest),
329  8,
330  0,
331  (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages,
332  cbp + sizeof(dma_cb_t)));
333 
334  dest += 8;
335  cbp += sizeof(dma_cb_t);
336 
337  // Delay (for setting sampling frequency)
338  /* DMA is waiting data request signal (DREQ) from PCM. PCM is set for 1.25 MhZ freqency, so,
339  each sample of GPIO is limited by writing to PCA queue.
340  */
341  cbp_curr = (dma_cb_t *)con_blocks->get_page(con_blocks->_virt_pages, cbp);
342  init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP | RCIN_RPI_DMA_D_DREQ | RCIN_RPI_DMA_PER_MAP(2),
343  RCIN_RPI_TIMER_BASE, phys_fifo_addr,
344  4,
345  0,
346  (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages,
347  cbp + sizeof(dma_cb_t)));
348 
349  cbp += sizeof(dma_cb_t);
350  }
351  //Make last control block point to the first (to make circle)
352  cbp -= sizeof(dma_cb_t);
353  ((dma_cb_t *)con_blocks->get_page(con_blocks->_virt_pages, cbp))->next = (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages, 0);
354 }

356 /*Initialise PCM
357  See BCM2835 documentation:
358  http://www.raspberrypi.org/wp-content/uploads/2012/02/BCM2835-ARM-Peripherals.pdf
359  */
360 void RCInput_RPI::init_PCM()
361 {
362  pcm_reg[RCIN_RPI_PCM_CS_A] = 1; // Disable Rx+Tx, Enable PCM block
363  hal.scheduler->delay_microseconds(100);
364  clk_reg[RCIN_RPI_PCMCLK_CNTL] = 0x5A000006; // Source=PLLD (500MHz)
365  hal.scheduler->delay_microseconds(100);
366  clk_reg[RCIN_RPI_PCMCLK_DIV] = 0x5A000000 | ((50000/RCIN_RPI_SAMPLE_FREQ)<<12); // Set pcm div. If we need to configure DMA frequency.
367  hal.scheduler->delay_microseconds(100);
368  clk_reg[RCIN_RPI_PCMCLK_CNTL] = 0x5A000016; // Source=PLLD and enable
369  hal.scheduler->delay_microseconds(100);
370  pcm_reg[RCIN_RPI_PCM_TXC_A] = 0<<31 | 1<<30 | 0<<20 | 0<<16; // 1 channel, 8 bits
371  hal.scheduler->delay_microseconds(100);
372  pcm_reg[RCIN_RPI_PCM_MODE_A] = (10 - 1) << 10; //PCM mode
373  hal.scheduler->delay_microseconds(100);
374  pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<4 | 1<<3; // Clear FIFOs
375  hal.scheduler->delay_microseconds(100);
376  pcm_reg[RCIN_RPI_PCM_DREQ_A] = 64<<24 | 64<<8; // DMA Req when one slot is free?
377  hal.scheduler->delay_microseconds(100);
378  pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<9; // Enable DMA
379  hal.scheduler->delay_microseconds(100);
380  pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<2; // Enable Tx
381  hal.scheduler->delay_microseconds(100);
382 }

458

463 #endif

465  // Init memory for buffer and for DMA control blocks.
466  // See comments in "init_ctrl_data()" to understand values "2" and "15"
467  circle_buffer = new Memory_table(RCIN_RPI_BUFFER_LENGTH * 2, version);
468  con_blocks = new Memory_table(RCIN_RPI_BUFFER_LENGTH * 15, version);

475  rc_channels[i].enable_pin->mode(HAL_GPIO_INPUT);
476  }
477 
478  // Configuration
479  set_sigaction();
480  init_ctrl_data();
481  init_PCM();

485  hal.scheduler->delay(300);
486

491  for (uint32_t i = 0; i < RCIN_RPI_CHN_NUM; ++i) {
492  rc_channels[i].prev_tick = curr_tick;
493  rc_channels[i].curr_signal = (signal_states & (1 << RcChnGpioTbl[i])) ? RCIN_RPI_SIG_HIGH
494  : RCIN_RPI_SIG_LOW;
495  rc_channels[i].last_signal = rc_channels[i].curr_signal;
496  }
497  curr_pointer += 8;

505 void RCInput_RPI::_timer_tick()
506 {
507  uint32_t counter = 0;
508  uint64_t signal_states(0);
509 
510  if (!_initialized) {
511  return;
512  }
513 
514  // Now we are getting address in which DMAC is writing at current moment
515  dma_cb_t *ad = (dma_cb_t *)con_blocks->get_virt_addr(dma_reg[RCIN_RPI_DMA_CONBLK_AD | RCIN_RPI_DMA_CHANNEL << 8]);
516 
517  if (!ad) {
518  debug("DMA sampling stopped, restarting...\n");
519  init_ctrl_data();
520  init_PCM();
521  init_DMA();
522  return;
523  }

527  if (x != nullptr) {
528  counter = circle_buffer->bytes_available(curr_pointer,
529  circle_buffer->get_offset(circle_buffer->_virt_pages, (uintptr_t)x));
530  break;
531  }
532  }
533 
534  if (counter == 0) {
535  return;
536  }
537

539  // We can't stay in method for a long time, because it may lead to delays
540  if (counter > RCIN_RPI_MAX_COUNTER) {
541  debug("%5d sample(s) dropped\n", (counter - RCIN_RPI_MAX_COUNTER) / 0x8);
542  counter = RCIN_RPI_MAX_COUNTER;
543  }
544 
545  // Processing ready bytes
546  for (;counter > 0x40;) {
547  // Is it timer sample?
548  if (curr_pointer % (64) == 0) {
549  curr_tick = *((uint64_t *)circle_buffer->get_page(circle_buffer->_virt_pages, curr_pointer));
550  curr_pointer += 8;
551  counter -= 8;
552  }
553  // Reading required bit
554  signal_states = *((uint64_t *)circle_buffer->get_page(circle_buffer->_virt_pages, curr_pointer));
555  for (uint32_t i = 0; i < RCIN_RPI_CHN_NUM; ++i) {
556  rc_channels[i].curr_signal = (signal_states & (1 << RcChnGpioTbl[i])) ? RCIN_RPI_SIG_HIGH
557  : RCIN_RPI_SIG_LOW;
558 
559  // If the signal changed
560  if (rc_channels[i].curr_signal != rc_channels[i].last_signal) {
561  rc_channels[i].delta_time = curr_tick - rc_channels[i].prev_tick;
562  rc_channels[i].prev_tick = curr_tick;
563  switch (rc_channels[i].state) {
564  case RCIN_RPI_INITIAL_STATE:
565  rc_channels[i].state = RCIN_RPI_ZERO_STATE;
566  break;
567  case RCIN_RPI_ZERO_STATE:
568  if (rc_channels[i].curr_signal == 0) {
569  rc_channels[i].width_s0 = (uint16_t)rc_channels[i].delta_time;
570  rc_channels[i].state = RCIN_RPI_ONE_STATE;
571  }
572  break;
573  case RCIN_RPI_ONE_STATE:
574  if (rc_channels[i].curr_signal == 1) {
575  rc_channels[i].width_s1 = (uint16_t)rc_channels[i].delta_time;
576  rc_channels[i].state = RCIN_RPI_ZERO_STATE;
577  if (1 == RCIN_RPI_CHN_NUM) {
578  _process_rc_pulse(rc_channels[i].width_s0,
579  rc_channels[i].width_s1);
580  }
581  else {
582  _process_pwm_pulse(i, rc_channels[i].width_s0,
583  rc_channels[i].width_s1);
584  }
585  }
586  break;
587  }
588  }
589  rc_channels[i].last_signal = rc_channels[i].curr_signal;
590  }
591  curr_pointer += 8;
592  counter -= 8;
593  if (curr_pointer >= circle_buffer->get_page_count() * PAGE_SIZE) {
594  curr_pointer = 0;
595  }
596  curr_tick += curr_tick_inc;
597  }
598 }
