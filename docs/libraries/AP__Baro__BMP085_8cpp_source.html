58 bool AP_Baro_BMP085::_init()
59 {
60  union {
61  uint8_t buff[22];
62  uint16_t wb[11];
63  } bb;
64 
65  // get pointer to i2c bus semaphore
66  AP_HAL::Semaphore *sem = _dev->get_semaphore();
67 
68  // take i2c bus sempahore
69  WITH_SEMAPHORE(sem);
70 
71  if (BMP085_EOC >= 0) {
72  _eoc = hal.gpio->channel(BMP085_EOC);
73  _eoc->mode(HAL_GPIO_INPUT);
74  }
75 
76 
77  uint8_t id;
78 
79  if (!_dev->read_registers(0xD0, &id, 1)) {
80  return false;
81  }
82 
83  if (id!=0x55) {
84  return false; // not BMP180
85  }
86 
87 
88  _dev->read_registers(0xD1, &_vers, 1);
89 
90  bool prom_ok=false;
91  _type=0;
92 
93  // We read the calibration data registers
94  if (_dev->read_registers(0xAA, bb.buff, sizeof(bb.buff))) {
95  prom_ok=true;
96 
97  }
98 
99  if (!prom_ok) {
100  if (_read_prom((uint16_t *)&bb.wb[0])) { // BMP180 requires reads by 2 bytes
101  prom_ok=true;
102  _type=1;
103  }
104  }
105  if (!prom_ok) {
106  return false;
107  }
108 
109  ac1 = ((int16_t)bb.buff[0] << 8) | bb.buff[1];
110  ac2 = ((int16_t)bb.buff[2] << 8) | bb.buff[3];
111  ac3 = ((int16_t)bb.buff[4] << 8) | bb.buff[5];
112  ac4 = ((int16_t)bb.buff[6] << 8) | bb.buff[7];
113  ac5 = ((int16_t)bb.buff[8] << 8) | bb.buff[9];
114  ac6 = ((int16_t)bb.buff[10]<< 8) | bb.buff[11];
115  b1 = ((int16_t)bb.buff[12] << 8) | bb.buff[13];
116  b2 = ((int16_t)bb.buff[14] << 8) | bb.buff[15];
117  mb = ((int16_t)bb.buff[16] << 8) | bb.buff[17];
118  mc = ((int16_t)bb.buff[18] << 8) | bb.buff[19];
119  md = ((int16_t)bb.buff[20] << 8) | bb.buff[21];
120 
121  if ((ac1==0 || ac1==-1) ||
122  (ac2==0 || ac2==-1) ||
123  (ac3==0 || ac3==-1) ||
124  (ac4==0 || ac4==0xFFFF) ||
125  (ac5==0 || ac5==0xFFFF) ||
126  (ac6==0 || ac6==0xFFFF)) {
127  return false;
128  }
129 
130  _last_press_read_command_time = 0;
131  _last_temp_read_command_time = 0;
132 
133  // Send a command to read temperature
134  _cmd_read_temp();
135 
136  _state = 0;
137 
138  _instance = _frontend.register_sensor();
139 
140  _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&AP_Baro_BMP085::_timer, void));
141  return true;
142 }

144 uint16_t AP_Baro_BMP085::_read_prom_word(uint8_t word)
145 {
146  const uint8_t reg = 0xAA + (word << 1);
147  uint8_t val[2];
148  if (!_dev->transfer(&reg, 1, val, sizeof(val))) {
149  return 0;
150  }
151  return (val[0] << 8) | val[1];

154 bool AP_Baro_BMP085::_read_prom(uint16_t *prom)
155 {
156  bool all_zero = true;
157  for (uint8_t i = 0; i < 11; i++) {
158  prom[i] = _read_prom_word(i);
159  if (prom[i] != 0) {
160  all_zero = false;

162  }
163 
164  if (all_zero) {
165  return false;
166  }
167 
168  return true;

174 void AP_Baro_BMP085::_timer(void)
175 {
176  if (!_data_ready()) {
177  return;
178  }
179 
180  if (_state == 0) {
181  _read_temp();
182  } else if (_read_pressure()) {
183  _calculate();
184  }
185 
186  _state++;
187  if (_state == 25) {
188  _state = 0;
189  _cmd_read_temp();
190  } else {
191  _cmd_read_pressure();
192  }
193 }

219 // Read raw pressure values
220 bool AP_Baro_BMP085::_read_pressure()
221 {
222  uint8_t buf[3];
223  if (_dev->read_registers(0xF6, buf, sizeof(buf))) {
224  _raw_pressure = (((uint32_t)buf[0] << 16)
225  | ((uint32_t)buf[1] << 8)
226  | ((uint32_t)buf[2])) >> (8 - OVERSAMPLING);
227  return true;
228  }
229 
230  uint8_t xlsb;
231  if (_dev->read_registers(0xF6, buf, 2) && _dev->read_registers(0xF8, &xlsb, 1)) {
232  _raw_pressure = (((uint32_t)buf[0] << 16)
233  | ((uint32_t)buf[1] << 8)
234  | ((uint32_t)xlsb)) >> (8 - OVERSAMPLING);
235  return true;
236  }
237 
238  _last_press_read_command_time = AP_HAL::millis() + 1000;
239  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
240  return false;
241 }

252 {
253  uint8_t buf[2];
254  int32_t _temp_sensor;
255 
256  if (!_dev->read_registers(0xF6, buf, sizeof(buf))) {
257  _dev->set_speed(AP_HAL::Device::SPEED_LOW);

266 // _calculate Temperature and Pressure in real units.
267 void AP_Baro_BMP085::_calculate()
268 {
269  int32_t x1, x2, x3, b3, b5, b6, p;
270  uint32_t b4, b7;
271  int32_t tmp;
272 
273  // See Datasheet page 13 for this formulas
274  // Based also on Jee Labs BMP085 example code. Thanks for share.
275  // Temperature calculations
276  x1 = ((int32_t)_raw_temp - ac6) * ac5 >> 15;
277  x2 = ((int32_t) mc << 11) / (x1 + md);
278  b5 = x1 + x2;
279  _temp = (b5 + 8) >> 4;
280 
281  // Pressure calculations
282  b6 = b5 - 4000;
283  x1 = (b2 * (b6 * b6 >> 12)) >> 11;
284  x2 = ac2 * b6 >> 11;
285  x3 = x1 + x2;
286  //b3 = (((int32_t) ac1 * 4 + x3)<<OVERSAMPLING + 2) >> 2; // BAD
287  //b3 = ((int32_t) ac1 * 4 + x3 + 2) >> 2; //OK for OVERSAMPLING=0
288  tmp = ac1;
289  tmp = (tmp*4 + x3)<<OVERSAMPLING;
290  b3 = (tmp+2)/4;
291  x1 = ac3 * b6 >> 13;
292  x2 = (b1 * (b6 * b6 >> 12)) >> 16;
293  x3 = ((x1 + x2) + 2) >> 2;
294  b4 = (ac4 * (uint32_t)(x3 + 32768)) >> 15;
295  b7 = ((uint32_t) _raw_pressure - b3) * (50000 >> OVERSAMPLING);
296  p = b7 < 0x80000000 ? (b7 * 2) / b4 : (b7 / b4) * 2;
297 
298  x1 = (p >> 8) * (p >> 8);
299  x1 = (x1 * 3038) >> 16;
300  x2 = (-7357 * p) >> 16;
301  p += ((x1 + x2 + 3791) >> 4);
302 
303  if (!pressure_ok(p)) {
304  return;
305  }
306 
307  WITH_SEMAPHORE(_sem);
308 
309  _pressure_filter.apply(p);
310  _has_sample = true;
311 }

313 bool AP_Baro_BMP085::_data_ready()
314 {
315  if (BMP085_EOC >= 0) {
316  return _eoc->read();
317  }
318 
319  // No EOC pin: use time from last read instead.
320  if (_state == 0) {
321  return AP_HAL::millis() > _last_temp_read_command_time + 5;
322  }
323 
324  uint32_t conversion_time_msec;
325 
326  switch (OVERSAMPLING) {
327  case BMP085_OVERSAMPLING_ULTRALOWPOWER:
328  conversion_time_msec = 5;
329  break;
330  case BMP085_OVERSAMPLING_STANDARD:
331  conversion_time_msec = 8;
332  break;
333  case BMP085_OVERSAMPLING_HIGHRES:
334  conversion_time_msec = 14;
335  break;
336  case BMP085_OVERSAMPLING_ULTRAHIGHRES:
337  conversion_time_msec = 26;
338  default:
339  break;
340  }
341 
342  return AP_HAL::millis() > _last_press_read_command_time + conversion_time_msec;
343 }
