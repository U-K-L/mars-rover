37 // update direction to frontend
38 void AP_WindVane_Backend::direction_update_frontend(float apparent_angle_ef)
39 {
40  // apply low pass filter if enabled
41  if (is_positive(_frontend._dir_filt_hz)) {
42  _dir_sin_filt.set_cutoff_frequency(_frontend._dir_filt_hz);
43  _dir_cos_filt.set_cutoff_frequency(_frontend._dir_filt_hz);
44  // https://en.wikipedia.org/wiki/Mean_of_circular_quantities
45  const float filtered_sin = _dir_sin_filt.apply(sinf(apparent_angle_ef), 0.05f);
46  const float filtered_cos = _dir_cos_filt.apply(cosf(apparent_angle_ef), 0.05f);
47  _frontend._direction_apparent_ef = atan2f(filtered_sin, filtered_cos);
48  } else {
49  _frontend._direction_apparent_ef = apparent_angle_ef;
50  }
51 
52  // make sure between -pi and pi
53  _frontend._direction_apparent_ef = wrap_PI(_frontend._direction_apparent_ef);
54 }
