779 void emit_checker(const struct type t, int arg_number, const char *indentation, const char *name) {
780  assert(indentation != NULL);
781 
782  if (arg_number > NULLABLE_ARG_COUNT_BASE) {
783  error(ERROR_INTERNAL, "Can't handle more then %d arguments to a function", NULLABLE_ARG_COUNT_BASE);
784  }
785 
786  if (t.flags & TYPE_FLAGS_NULLABLE) {
787  arg_number = arg_number + NULLABLE_ARG_COUNT_BASE;
788  switch (t.type) {
789  case TYPE_BOOLEAN:
790  fprintf(source, "%sbool data_%d = {};\n", indentation, arg_number);
791  break;
792  case TYPE_FLOAT:
793  fprintf(source, "%sfloat data_%d = {};\n", indentation, arg_number);
794  break;
795  case TYPE_INT8_T:
796  fprintf(source, "%sint8_t data_%d = {};\n", indentation, arg_number);
797  break;
798  case TYPE_INT16_T:
799  fprintf(source, "%sint16_t data_%d = {};\n", indentation, arg_number);
800  break;
801  case TYPE_INT32_T:
802  fprintf(source, "%sint32_t data_%d = {};\n", indentation, arg_number);
803  break;
804  case TYPE_UINT8_T:
805  fprintf(source, "%suint8_t data_%d = {};\n", indentation, arg_number);
806  break;
807  case TYPE_UINT16_T:
808  fprintf(source, "%suint16_t data_%d = {};\n", indentation, arg_number);
809  break;
810  case TYPE_UINT32_T:
811  fprintf(source, "%suint32_t data_%d = {};\n", indentation, arg_number);
812  break;
813  case TYPE_NONE:
814  return; // nothing to do here, this should potentially be checked outside of this, but it makes an easier implementation to accept it
815  case TYPE_STRING:
816  fprintf(source, "%schar * data_%d = {};\n", indentation, arg_number);
817  break;
818  case TYPE_ENUM:
819  fprintf(source, "%suint32_t data_%d = {};\n", indentation, arg_number);
820  break;
821  case TYPE_USERDATA:
822  fprintf(source, "%s%s data_%d = {};\n", indentation, t.data.userdata_name, arg_number);
823  break;
824  }
825  } else {
826  // handle this in four stages
827  // - figure out any relevant minimum values for range checking
828  // - emit a non down casted version
829  // - then run range checks
830  // - then cast down as appropriate
831 
832  // select minimums
833  char * forced_min;
834  char * forced_max;
835  switch (t.type) {
836  case TYPE_FLOAT:
837  forced_min = "-INFINITY";
838  forced_max = "INFINITY";
839  break;
840  case TYPE_INT8_T:
841  forced_min = "INT8_MIN";
842  forced_max = "INT8_MAX";
843  break;
844  case TYPE_INT16_T:
845  forced_min = "INT16_MIN";
846  forced_max = "INT16_MAX";
847  break;
848  case TYPE_INT32_T:
849  forced_min = "INT32_MIN";
850  forced_max = "INT32_MAX";
851  break;
852  case TYPE_UINT8_T:
853  forced_min = "0";
854  forced_max = "UINT8_MAX";
855  break;
856  case TYPE_UINT16_T:
857  forced_min = "0";
858  forced_max = "UINT16_MAX";
859  break;
860  case TYPE_UINT32_T:
861  forced_min = "0";
862  forced_max = "UINT16_MAX";
863  break;
864  case TYPE_ENUM:
865  forced_min = forced_max = NULL;
866  break;
867  case TYPE_NONE:
868  return; // nothing to do here, this should potentially be checked outside of this, but it makes an easier implementation to accept it
869  case TYPE_STRING:
870  case TYPE_BOOLEAN:
871  case TYPE_USERDATA:
872  // these don't get range checked, so skip the raw_data phase
873  assert(t.range == NULL); // we should have caught this during the parse phase
874  break;
875  }
876 
877  // non down cast
878  switch (t.type) {
879  case TYPE_FLOAT:
880  fprintf(source, "%sconst float raw_data_%d = luaL_checknumber(L, %d);\n", indentation, arg_number, arg_number);
881  break;
882  case TYPE_INT8_T:
883  case TYPE_INT16_T:
884  case TYPE_INT32_T:
885  case TYPE_UINT8_T:
886  case TYPE_UINT16_T:
887  case TYPE_ENUM:
888  fprintf(source, "%sconst lua_Integer raw_data_%d = luaL_checkinteger(L, %d);\n", indentation, arg_number, arg_number);
889  break;
890  case TYPE_UINT32_T:
891  fprintf(source, "%sconst uint32_t raw_data_%d = coerce_to_uint32_t(L, %d);\n", indentation, arg_number, arg_number);
892  break;
893  case TYPE_NONE:
894  case TYPE_STRING:
895  case TYPE_BOOLEAN:
896  case TYPE_USERDATA:
897  // these don't get range checked, so skip the raw_data phase
898  assert(t.range == NULL); // we should have caught this during the parse phase
899  break;
900  }
901 
902  // range check
903  if (t.range != NULL) {
904  if ((forced_min != NULL) && (forced_max != NULL)) {
905  fprintf(source, "%sluaL_argcheck(L, ((raw_data_%d >= MAX(%s, %s)) && (raw_data_%d <= MIN(%s, %s))), %d, \"%s out of range\");\n",
906  indentation,
907  arg_number, t.range->low, forced_min,
908  arg_number, t.range->high, forced_max,
909  arg_number, name);
910  } else {
911  fprintf(source, "%sluaL_argcheck(L, ((raw_data_%d >= %s) && (raw_data_%d <= %s)), %d, \"%s out of range\");\n",
912  indentation,
913  arg_number, t.range->low,
914  arg_number, t.range->high,
915  arg_number, name);
916  }
917  }
918 
919  // down cast
920  switch (t.type) {
921  case TYPE_FLOAT:
922  // this is a trivial transformation, trust the compiler to resolve it for us
923  fprintf(source, "%sconst float data_%d = raw_data_%d;\n", indentation, arg_number, arg_number);
924  break;
925  case TYPE_INT8_T:
926  fprintf(source, "%sconst int8_t data_%d = static_cast<int8_t>(raw_data_%d);\n", indentation, arg_number, arg_number);
927  break;
928  case TYPE_INT16_T:
929  fprintf(source, "%sconst int16_t data_%d = static_cast<int16_t>(raw_data_%d);\n", indentation, arg_number, arg_number);
930  break;
931  case TYPE_INT32_T:
932  fprintf(source, "%sconst int32_t data_%d = raw_data_%d;\n", indentation, arg_number, arg_number);
933  break;
934  case TYPE_UINT8_T:
935  fprintf(source, "%sconst uint8_t data_%d = static_cast<uint8_t>(raw_data_%d);\n", indentation, arg_number, arg_number);
936  break;
937  case TYPE_UINT16_T:
938  fprintf(source, "%sconst uint16_t data_%d = static_cast<uint16_t>(raw_data_%d);\n", indentation, arg_number, arg_number);
939  break;
940  case TYPE_UINT32_T:
941  fprintf(source, "%sconst uint32_t data_%d = static_cast<uint32_t>(raw_data_%d);\n", indentation, arg_number, arg_number);
942  break;
943  case TYPE_BOOLEAN:
944  fprintf(source, "%sconst bool data_%d = static_cast<bool>(lua_toboolean(L, %d));\n", indentation, arg_number, arg_number);
945  break;
946  case TYPE_STRING:
947  fprintf(source, "%sconst char * data_%d = luaL_checkstring(L, %d);\n", indentation, arg_number, arg_number);
948  break;
949  case TYPE_ENUM:
950  fprintf(source, "%sconst %s data_%d = static_cast<%s>(raw_data_%d);\n", indentation, t.data.enum_name, arg_number, t.data.enum_name, arg_number);
951  break;
952  case TYPE_USERDATA:
953  fprintf(source, "%s%s & data_%d = *check_%s(L, %d);\n", indentation, t.data.userdata_name, arg_number, t.data.userdata_name, arg_number);
954  break;
955  case TYPE_NONE:
956  // nothing to do, we've either already emitted a reasonable value, or returned
957  break;
958  }
959  }
