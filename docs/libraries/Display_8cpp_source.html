319 bool Display::init(void)
320 {
321  // exit immediately if already initialised
322  if (_driver != nullptr) {
323  return true;
324  }
325 
326  // initialise driver
327  FOREACH_I2C(i) {
328  switch (pNotify->_display_type) {
329  case DISPLAY_SSD1306: {
330  _driver = Display_SSD1306_I2C::probe(std::move(hal.i2c_mgr->get_device(i, NOTIFY_DISPLAY_I2C_ADDR)));
331  break;
332  }
333  case DISPLAY_SH1106: {
334  _driver = Display_SH1106_I2C::probe(std::move(hal.i2c_mgr->get_device(i, NOTIFY_DISPLAY_I2C_ADDR)));
335  break;
336  }
337  case DISPLAY_SITL: {
338 #ifdef WITH_SITL_OSD
339  _driver = Display_SITL::probe(); // never fails
340 #elif CONFIG_HAL_BOARD == HAL_BOARD_SITL
341  ::fprintf(stderr, "SITL Display ineffective without --osd\n");
342 #endif
343  break;
344  }
