151 bool AP_Compass_HMC5843::init()
152 {
153  AP_HAL::Semaphore *bus_sem = _bus->get_semaphore();
154 
155  if (!bus_sem || !bus_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
156  hal.console->printf("HMC5843: Unable to get bus semaphore\n");
157  return false;
158  }
159 
160  // high retries for init
161  _bus->set_retries(10);
162 
163  if (!_bus->configure()) {
164  hal.console->printf("HMC5843: Could not configure the bus\n");
165  goto errout;
166  }
167 
168  if (!_check_whoami()) {
169  goto errout;
170  }
171 
172  if (!_calibrate()) {
173  hal.console->printf("HMC5843: Could not calibrate sensor\n");
174  goto errout;
175  }
176 
177  if (!_setup_sampling_mode()) {
178  goto errout;
179  }
180 
181  if (!_bus->start_measurements()) {
182  hal.console->printf("HMC5843: Could not start measurements on bus\n");
183  goto errout;
184  }
185 
186  _initialised = true;
187 
188  // lower retries for run
189  _bus->set_retries(3);
190 
191  bus_sem->give();
192 
193  // perform an initial read
194  read();
195 
196  _compass_instance = register_compass();
197 
198  set_rotation(_compass_instance, _rotation);
199 
200  _bus->set_device_type(DEVTYPE_HMC5883);
201  set_dev_id(_compass_instance, _bus->get_bus_id());
202 
203  if (_force_external) {
204  set_external(_compass_instance, true);
205  }
206 
207  // read from sensor at 75Hz
208  _bus->register_periodic_callback(13333,
209  FUNCTOR_BIND_MEMBER(&AP_Compass_HMC5843::_timer, void));
210 
211  hal.console->printf("HMC5843 found on bus 0x%x\n", (unsigned)_bus->get_bus_id());
212 
213  return true;
214 
215 errout:
216  bus_sem->give();
217  return false;
218 }

220 /*
221  * take a reading from the magnetometer
222  *
223  * bus semaphore has been taken already by HAL
224  */
225 void AP_Compass_HMC5843::_timer()
226 {
227  bool result = _read_sample();
228 
229  // always ask for a new sample
230  _take_sample();
231 
232  if (!result) {
233  return;
234  }
235 
236  // get raw_field - sensor frame, uncorrected
237  Vector3f raw_field = Vector3f(_mag_x, _mag_y, _mag_z);
238  raw_field *= _gain_scale;
239 
240  // rotate to the desired orientation
241  if (is_external(_compass_instance)) {
242  raw_field.rotate(ROTATION_YAW_90);
243  }
244 
245  // We expect to do reads at 10Hz, and we get new data at most 75Hz, so we
246  // don't expect to accumulate more than 8 before a read; let's make it
247  // 14 to give more room for the initialization phase
248  accumulate_sample(raw_field, _compass_instance, 14);
249 }

269 bool AP_Compass_HMC5843::_setup_sampling_mode()
270 {
271  _gain_scale = (1.0f / 1090) * 1000;
272  if (!_bus->register_write(HMC5843_REG_CONFIG_A,
273  HMC5843_CONF_TEMP_ENABLE |
274  HMC5843_OSR_75HZ |
275  HMC5843_SAMPLE_AVERAGING_1) ||
276  !_bus->register_write(HMC5843_REG_CONFIG_B,
277  HMC5883L_GAIN_1_30_GA) ||
278  !_bus->register_write(HMC5843_REG_MODE,
279  HMC5843_MODE_SINGLE)) {
280  return false;
281  }
282  return true;

285 /*
286  * Read Sensor data - bus semaphore must be taken
287  */
288 bool AP_Compass_HMC5843::_read_sample()
289 {
290  struct PACKED {
291  be16_t rx;
292  be16_t ry;
293  be16_t rz;
294  } val;
295  int16_t rx, ry, rz;
296 
297  if (!_bus->block_read(HMC5843_REG_DATA_OUTPUT_X_MSB, (uint8_t *) &val, sizeof(val))){
298  return false;
299  }
300 
301  rx = be16toh(val.rx);
302  ry = be16toh(val.rz);
303  rz = be16toh(val.ry);
304 
305  if (rx == -4096 || ry == -4096 || rz == -4096) {
306  // no valid data available
307  return false;
308  }
309 
310  _mag_x = -rx;
311  _mag_y = ry;
312  _mag_z = -rz;
313 
314  return true;
315 }

327 bool AP_Compass_HMC5843::_check_whoami()
328 {
329  uint8_t id[3];
330  if (!_bus->block_read(HMC5843_REG_ID_A, id, 3)) {
331  // can't talk on bus
332  return false;
333  }
334  if (id[0] != 'H' ||
335  id[1] != '4' ||
336  id[2] != '3') {
337  // not a HMC5x83 device
338  return false;
339  }
340 
341  return true;

344 bool AP_Compass_HMC5843::_calibrate()
345 {
346  uint8_t calibration_gain;
347  int numAttempts = 0, good_count = 0;
348  bool success = false;
349 
350  calibration_gain = HMC5883L_GAIN_2_50_GA;
351 
352  /*
353  * the expected values are based on observation of real sensors
354  */
355  float expected[3] = { 1.16*600, 1.08*600, 1.16*600 };
356 
357  uint8_t base_config = HMC5843_OSR_15HZ;
358  uint8_t num_samples = 0;
359 
360  while (success == 0 && numAttempts < 25 && good_count < 5) {
361  numAttempts++;
362 
363  // force positiveBias (compass should return 715 for all channels)
364  if (!_bus->register_write(HMC5843_REG_CONFIG_A,
365  base_config | HMC5843_OPMODE_POSITIVE_BIAS)) {
366  // compass not responding on the bus
367  continue;
368  }
369 
370  hal.scheduler->delay(50);
371 
372  // set gains
373  if (!_bus->register_write(HMC5843_REG_CONFIG_B, calibration_gain) ||
374  !_bus->register_write(HMC5843_REG_MODE, HMC5843_MODE_SINGLE)) {
375  continue;
376  }
377 
378  // read values from the compass
379  hal.scheduler->delay(50);
380  if (!_read_sample()) {
381  // we didn't read valid values
382  continue;
383  }
384 
385  num_samples++;
386 
387  float cal[3];
388 
389  // hal.console->printf("mag %d %d %d\n", _mag_x, _mag_y, _mag_z);
390 
391  cal[0] = fabsf(expected[0] / _mag_x);
392  cal[1] = fabsf(expected[1] / _mag_y);
393  cal[2] = fabsf(expected[2] / _mag_z);
394 
395  // hal.console->printf("cal=%.2f %.2f %.2f\n", cal[0], cal[1], cal[2]);
396 
397  // we throw away the first two samples as the compass may
398  // still be changing its state from the application of the
399  // strap excitation. After that we accept values in a
400  // reasonable range
401  if (numAttempts <= 2) {
402  continue;
403  }
404 
405 #define IS_CALIBRATION_VALUE_VALID(val) (val > 0.7f && val < 1.35f)
406 
407  if (IS_CALIBRATION_VALUE_VALID(cal[0]) &&
408  IS_CALIBRATION_VALUE_VALID(cal[1]) &&
409  IS_CALIBRATION_VALUE_VALID(cal[2])) {
410  // hal.console->printf("car=%.2f %.2f %.2f good\n", cal[0], cal[1], cal[2]);
411  good_count++;
412 
413  _scaling[0] += cal[0];
414  _scaling[1] += cal[1];
415  _scaling[2] += cal[2];
416  }
417 
418 #undef IS_CALIBRATION_VALUE_VALID
419 
420 #if 0
421  /* useful for debugging */
422  hal.console->printf("MagX: %d MagY: %d MagZ: %d\n", (int)_mag_x, (int)_mag_y, (int)_mag_z);
423  hal.console->printf("CalX: %.2f CalY: %.2f CalZ: %.2f\n", cal[0], cal[1], cal[2]);
424 #endif
425  }
426 
427  _bus->register_write(HMC5843_REG_CONFIG_A, base_config);
428 
429  if (good_count >= 5) {
430  _scaling[0] = _scaling[0] / good_count;
431  _scaling[1] = _scaling[1] / good_count;
432  _scaling[2] = _scaling[2] / good_count;
433  success = true;
434  } else {
435  /* best guess */
436  _scaling[0] = 1.0;
437  _scaling[1] = 1.0;
438  _scaling[2] = 1.0;
439  if (num_samples > 5) {
440  // a sensor can be broken for calibration but still
441  // otherwise workable, accept it if we are reading samples
442  success = true;
443  }
444  }
445 
446 #if 0
447  printf("scaling: %.2f %.2f %.2f\n",
448  _scaling[0], _scaling[1], _scaling[2]);
449 #endif
450 
451  return success;
452 }
