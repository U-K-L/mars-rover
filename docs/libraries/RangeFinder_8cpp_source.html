288 /*
289  initialise the RangeFinder class. We do detection of attached range
290  finders here. For now we won't allow for hot-plugging of
291  rangefinders.
292  */
293 void RangeFinder::init(enum Rotation orientation_default)
294 {
295  if (num_instances != 0) {
296  // init called a 2nd time?
297  return;
298  }
299 
300  convert_params();
301 
302  // set orientation defaults
303  for (uint8_t i=0; i<RANGEFINDER_MAX_INSTANCES; i++) {
304  params[i].orientation.set_default(orientation_default);
305  }
306 
307  for (uint8_t i=0, serial_instance = 0; i<RANGEFINDER_MAX_INSTANCES; i++) {
308  // serial_instance will be increased inside detect_instance
309  // if a serial driver is loaded for this instance
310  detect_instance(i, serial_instance);
311  if (drivers[i] != nullptr) {
312  // we loaded a driver for this instance, so it must be
313  // present (although it may not be healthy)
314  num_instances = i+1;
315  }
316 
317  // initialise status
318  state[i].status = RangeFinder_NotConnected;
319  state[i].range_valid_count = 0;
320  }
321 }

323 /*
324  update RangeFinder state for all instances. This should be called at
325  around 10Hz by main loop
326  */
327 void RangeFinder::update(void)
328 {
329  for (uint8_t i=0; i<num_instances; i++) {
330  if (drivers[i] != nullptr) {
331  if (params[i].type == RangeFinder_TYPE_NONE) {
332  // allow user to disable a rangefinder at runtime
333  state[i].status = RangeFinder_NotConnected;
334  state[i].range_valid_count = 0;
335  continue;
336  }
337  drivers[i]->update();
338  }
339  }
340 
341  Log_RFND();
342 }

357 /*
358  detect if an instance of a rangefinder is connected.
359  */
360 void RangeFinder::detect_instance(uint8_t instance, uint8_t& serial_instance)
361 {
362  enum RangeFinder_Type _type = (enum RangeFinder_Type)params[instance].type.get();
363  switch (_type) {
364  case RangeFinder_TYPE_PLI2C:
365  case RangeFinder_TYPE_PLI2CV3:
366  case RangeFinder_TYPE_PLI2CV3HP:
367  FOREACH_I2C(i) {
368  if (_add_backend(AP_RangeFinder_PulsedLightLRF::detect(i, state[instance], params[instance], _type))) {
369  break;
370  }
371  }
372  break;
373  case RangeFinder_TYPE_MBI2C:
374  FOREACH_I2C(i) {
375  if (_add_backend(AP_RangeFinder_MaxsonarI2CXL::detect(state[instance], params[instance],
376  hal.i2c_mgr->get_device(i, AP_RANGE_FINDER_MAXSONARI2CXL_DEFAULT_ADDR)))) {
377  break;
378  }
379  }
380  break;
381  case RangeFinder_TYPE_LWI2C:
382  if (params[instance].address) {
383 #ifdef HAL_RANGEFINDER_LIGHTWARE_I2C_BUS
384  _add_backend(AP_RangeFinder_LightWareI2C::detect(state[instance], params[instance],
385  hal.i2c_mgr->get_device(HAL_RANGEFINDER_LIGHTWARE_I2C_BUS, params[instance].address)));
386 #else
387  FOREACH_I2C(i) {
388  if (_add_backend(AP_RangeFinder_LightWareI2C::detect(state[instance], params[instance],
389  hal.i2c_mgr->get_device(i, params[instance].address)))) {
390  break;
391  }
392  }
393 #endif
394  }
395  break;
396  case RangeFinder_TYPE_TRI2C:
397  if (params[instance].address) {
398  FOREACH_I2C(i) {
399  if (_add_backend(AP_RangeFinder_TeraRangerI2C::detect(state[instance], params[instance],
400  hal.i2c_mgr->get_device(i, params[instance].address)))) {
401  break;
402  }
403  }
404  }
405  break;
406  case RangeFinder_TYPE_VL53L0X:
407  FOREACH_I2C(i) {
408  if (_add_backend(AP_RangeFinder_VL53L0X::detect(state[instance], params[instance],
409  hal.i2c_mgr->get_device(i, params[instance].address)))) {
410  break;
411  }
412  if (_add_backend(AP_RangeFinder_VL53L1X::detect(state[instance], params[instance],
413  hal.i2c_mgr->get_device(i, params[instance].address)))) {
414  break;
415  }
416  }
417  break;
418  case RangeFinder_TYPE_BenewakeTFminiPlus:
419  FOREACH_I2C(i) {
420  if (_add_backend(AP_RangeFinder_Benewake_TFMiniPlus::detect(state[instance], params[instance],
421  hal.i2c_mgr->get_device(i, params[instance].address)))) {
422  break;
423  }
424  }
425  break;
426 #if CONFIG_HAL_BOARD == HAL_BOARD_CHIBIOS
427  case RangeFinder_TYPE_PX4_PWM:
428  // to ease moving from PX4 to ChibiOS we'll lie a little about
429  // the backend driver...
430  if (AP_RangeFinder_PWM::detect()) {
431  drivers[instance] = new AP_RangeFinder_PWM(state[instance], params[instance], estimated_terrain_height);
432  }
433  break;
434 #endif
435 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BBBMINI
436  case RangeFinder_TYPE_BBB_PRU:
437  if (AP_RangeFinder_BBB_PRU::detect()) {
438  drivers[instance] = new AP_RangeFinder_BBB_PRU(state[instance], params[instance]);
439  }
440  break;
441 #endif
442  case RangeFinder_TYPE_LWSER:
443  if (AP_RangeFinder_LightWareSerial::detect(serial_manager, serial_instance)) {
444  drivers[instance] = new AP_RangeFinder_LightWareSerial(state[instance], params[instance], serial_manager, serial_instance++);
445  }
446  break;
447  case RangeFinder_TYPE_LEDDARONE:
448  if (AP_RangeFinder_LeddarOne::detect(serial_manager, serial_instance)) {
449  drivers[instance] = new AP_RangeFinder_LeddarOne(state[instance], params[instance], serial_manager, serial_instance++);
450  }
451  break;
452  case RangeFinder_TYPE_ULANDING:
453  if (AP_RangeFinder_uLanding::detect(serial_manager, serial_instance)) {
454  drivers[instance] = new AP_RangeFinder_uLanding(state[instance], params[instance], serial_manager, serial_instance++);
455  }
456  break;
457 #if (CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BEBOP || \
458  CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_DISCO) && defined(HAVE_LIBIIO)
459  case RangeFinder_TYPE_BEBOP:
460  if (AP_RangeFinder_Bebop::detect()) {
461  drivers[instance] = new AP_RangeFinder_Bebop(state[instance], params[instance]);
462  }
463  break;
464 #endif
465  case RangeFinder_TYPE_MAVLink:
466  if (AP_RangeFinder_MAVLink::detect()) {
467  drivers[instance] = new AP_RangeFinder_MAVLink(state[instance], params[instance]);
468  }
469  break;
470  case RangeFinder_TYPE_MBSER:
471  if (AP_RangeFinder_MaxsonarSerialLV::detect(serial_manager, serial_instance)) {
472  drivers[instance] = new AP_RangeFinder_MaxsonarSerialLV(state[instance], params[instance], serial_manager, serial_instance++);
473  }
474  break;
475  case RangeFinder_TYPE_ANALOG:
476  // note that analog will always come back as present if the pin is valid
477  if (AP_RangeFinder_analog::detect(params[instance])) {
478  drivers[instance] = new AP_RangeFinder_analog(state[instance], params[instance]);
479  }
480  break;
481  case RangeFinder_TYPE_NMEA:
482  if (AP_RangeFinder_NMEA::detect(serial_manager, serial_instance)) {
483  drivers[instance] = new AP_RangeFinder_NMEA(state[instance], params[instance], serial_manager, serial_instance++);
484  }
485  break;
486  case RangeFinder_TYPE_WASP:
487  if (AP_RangeFinder_Wasp::detect(serial_manager, serial_instance)) {
488  drivers[instance] = new AP_RangeFinder_Wasp(state[instance], params[instance], serial_manager, serial_instance++);
489  }
490  break;
491  case RangeFinder_TYPE_BenewakeTF02:
492  if (AP_RangeFinder_Benewake::detect(serial_manager, serial_instance)) {
493  drivers[instance] = new AP_RangeFinder_Benewake(state[instance], params[instance], serial_manager, serial_instance++, AP_RangeFinder_Benewake::BENEWAKE_TF02);
494  }
495  break;
496  case RangeFinder_TYPE_BenewakeTFmini:
497  if (AP_RangeFinder_Benewake::detect(serial_manager, serial_instance)) {
498  drivers[instance] = new AP_RangeFinder_Benewake(state[instance], params[instance], serial_manager, serial_instance++, AP_RangeFinder_Benewake::BENEWAKE_TFmini);
499  }
500  break;
501  case RangeFinder_TYPE_PWM:
502  if (AP_RangeFinder_PWM::detect()) {
503  drivers[instance] = new AP_RangeFinder_PWM(state[instance], params[instance], estimated_terrain_height);
504  }
505  break;
506  case RangeFinder_TYPE_BLPing:
507  if (AP_RangeFinder_BLPing::detect(serial_manager, serial_instance)) {
508  drivers[instance] = new AP_RangeFinder_BLPing(state[instance], params[instance], serial_manager, serial_instance++);
509  }
510  break;
511  default:
512  break;
513  }
514 
515  // if the backend has some local parameters then make those available in the tree
516  if (drivers[instance] && state[instance].var_info) {
517  backend_var_info[instance] = state[instance].var_info;
518  AP_Param::load_object_from_eeprom(drivers[instance], backend_var_info[instance]);
519  }
520 }

666 // Write an RFND (rangefinder) packet
667 void RangeFinder::Log_RFND()
668 {
669  if (_log_rfnd_bit == uint32_t(-1)) {
670  return;
671  }
672 
673  AP_Logger &logger = AP::logger();
674  if (!logger.should_log(_log_rfnd_bit)) {
675  return;
676  }
677 
678  for (uint8_t i=0; i<RANGEFINDER_MAX_INSTANCES; i++) {
679  const AP_RangeFinder_Backend *s = get_backend(i);
680  if (s == nullptr) {
681  continue;
682  }
683 
684  const struct log_RFND pkt = {
685  LOG_PACKET_HEADER_INIT(LOG_RFND_MSG),
686  time_us : AP_HAL::micros64(),
687  instance : i,
688  dist : s->distance_cm(),
689  status : (uint8_t)s->status(),
690  orient : s->orientation(),
691  };
692  AP::logger().WriteBlock(&pkt, sizeof(pkt));
693  }
694 }
