77 // check for replies from sensor, returns true if at least one message was processed
78 bool AP_Proximity_TeraRangerTower::read_sensor_data()
79 {
80  if (uart == nullptr) {
81  return false;
82  }
83 
84  uint16_t message_count = 0;
85  int16_t nbytes = uart->available();
86 
87  while (nbytes-- > 0) {
88  char c = uart->read();
89  if (c == 'T' ) {
90  buffer_count = 0;
91  }
92 
93  buffer[buffer_count++] = c;
94 
95  // we should always read 19 bytes THxxxxxxxxxxxxxxxxC
96  if (buffer_count >= 19){
97  buffer_count = 0;
98 
99  // check if message has right CRC
100  if (crc_crc8(buffer, 18) == buffer[18]){
101  update_sector_data(0, UINT16_VALUE(buffer[2], buffer[3])); // d1
102  update_sector_data(45, UINT16_VALUE(buffer[16], buffer[17])); // d8
103  update_sector_data(90, UINT16_VALUE(buffer[14], buffer[15])); // d7
104  update_sector_data(135, UINT16_VALUE(buffer[12], buffer[13])); // d6
105  update_sector_data(180, UINT16_VALUE(buffer[10], buffer[11])); // d5
106  update_sector_data(225, UINT16_VALUE(buffer[8], buffer[9])); // d4
107  update_sector_data(270, UINT16_VALUE(buffer[6], buffer[7])); // d3
108  update_sector_data(315, UINT16_VALUE(buffer[4], buffer[5])); // d2
109 
110  message_count++;
111  }
112  }
113  }
114  return (message_count > 0);
115 }
