27 /*
28  return a RC input value scaled from -4500 to 4500
29  */
30 int16_t AP_IOMCU_FW::mix_input_angle(uint8_t channel, uint16_t radio_in) const
31 {
32  const uint16_t &rc_min = mixing.rc_min[channel];
33  const uint16_t &rc_max = mixing.rc_max[channel];
34  const uint16_t &rc_trim = mixing.rc_trim[channel];
35  const uint16_t &reversed = mixing.rc_reversed[channel];
36 
37  int16_t ret = 0;
38  if (radio_in > rc_trim && rc_max != rc_trim) {
39  ret = (ANGLE_SCALE * (int32_t)(radio_in - rc_trim)) / (int32_t)(rc_max - rc_trim);
40  } else if (radio_in < rc_trim && rc_trim != rc_min) {
41  ret = (ANGLE_SCALE * (int32_t)(radio_in - rc_trim)) / (int32_t)(rc_trim - rc_min);
42  }
43  if (reversed) {
44  ret = -ret;
45  }
46  return ret;
47 }

49 /*
50  return a RC input value scaled from 0 to 1000
51  */
52 int16_t AP_IOMCU_FW::mix_input_range(uint8_t channel, uint16_t radio_in) const
53 {
54  const uint16_t &rc_min = mixing.rc_min[channel];
55  const uint16_t &rc_max = mixing.rc_max[channel];
56  const uint16_t &reversed = mixing.rc_reversed[channel];
57 
58  int16_t ret = 0;
59  if (radio_in > rc_max) {
60  ret = RANGE_SCALE;
61  } else if (radio_in < rc_min) {
62  ret = -RANGE_SCALE;
63  } else {
64  ret = (RANGE_SCALE * (int32_t)(radio_in - rc_min)) / (int32_t)(rc_max - rc_min);
65  }
66  if (reversed) {
67  ret = -ret;
68  }
69  return ret;
70 }

72 /*
73  return an output pwm giving an angle for a servo channel
74  */
75 uint16_t AP_IOMCU_FW::mix_output_angle(uint8_t channel, int16_t angle) const
76 {
77  const uint16_t &srv_min = mixing.servo_min[channel];
78  const uint16_t &srv_max = mixing.servo_max[channel];
79  const uint16_t &srv_trim = mixing.servo_trim[channel];
80  const uint16_t &reversed = mixing.servo_reversed[channel];
81  if (reversed) {
82  angle = -angle;
83  }
84  angle = constrain_int16(angle, -ANGLE_SCALE, ANGLE_SCALE);
85  if (angle > 0) {
86  return srv_trim + ((int32_t)angle * (int32_t)(srv_max - srv_trim)) / ANGLE_SCALE;
87  }
88  return srv_trim - (-(int32_t)angle * (int32_t)(srv_trim - srv_min)) / ANGLE_SCALE;
89 }

91 /*
92  return an output pwm giving an range for a servo channel
93  */
94 uint16_t AP_IOMCU_FW::mix_output_range(uint8_t channel, int16_t value) const
95 {
96  const uint16_t &srv_min = mixing.servo_min[channel];
97  const uint16_t &srv_max = mixing.servo_max[channel];
98  const uint16_t &reversed = mixing.servo_reversed[channel];
99  value = constrain_int16(value, 0, RANGE_SCALE);
100  if (reversed) {
101  value = RANGE_SCALE - value;
102  }
103  return srv_min + ((int32_t)value * (int32_t)(srv_max - srv_min)) / RANGE_SCALE;
104 }

110 int16_t AP_IOMCU_FW::mix_elevon_vtail(int16_t angle1, int16_t angle2, bool first_output) const
111 {
112  if (first_output) {
113  return (angle2 - angle1) * mixing.mixing_gain / 1000;

118 /*
119  run mixer. This is used when FMU is not providing inputs, or when
120  the OVERRIDE_CHAN is high. It allows for manual fixed wing flight
121  */
122 void AP_IOMCU_FW::run_mixer(void)
123 {
124  int16_t rcin[4] = {0, 0, 0, 0};
125  int16_t &roll = rcin[0];
126  int16_t &pitch = rcin[1];
127  int16_t &throttle = rcin[2];
128  int16_t &rudder = rcin[3];
129 
130  // get RC input angles
131  if (rc_input.flags_rc_ok) {
132  for (uint8_t i=0;i<4; i++) {
133  if (mixing.rc_channel[i] > 0 && mixing.rc_channel[i] <= IOMCU_MAX_CHANNELS) {
134  uint8_t chan = mixing.rc_channel[i]-1;
135  if (i == 2 && !mixing.throttle_is_angle) {
136  rcin[i] = mix_input_range(i, rc_input.pwm[chan]);
137  } else {
138  rcin[i] = mix_input_angle(i, rc_input.pwm[chan]);
139  }
140  }
141  }
142  }
143 
144  for (uint8_t i=0; i<IOMCU_MAX_CHANNELS; i++) {
145  SRV_Channel::Aux_servo_function_t function = (SRV_Channel::Aux_servo_function_t)mixing.servo_function[i];
146  uint16_t &pwm = reg_direct_pwm.pwm[i];
147 
148  if (mixing.manual_rc_mask & (1U<<i)) {
149  // treat as pass-thru if this channel is set in MANUAL_RC_MASK
150  function = SRV_Channel::k_manual;
151  }
152 
153  switch (function) {
154  case SRV_Channel::k_manual:
155  pwm = rc_input.pwm[i];
156  break;
157 
158  case SRV_Channel::k_rcin1 ... SRV_Channel::k_rcin16:
159  pwm = rc_input.pwm[(uint8_t)(function - SRV_Channel::k_rcin1)];
160  break;
161 
162  case SRV_Channel::k_aileron:
163  case SRV_Channel::k_aileron_with_input:
164  case SRV_Channel::k_flaperon_left:
165  case SRV_Channel::k_flaperon_right:
166  pwm = mix_output_angle(i, roll);
167  break;
168 
169  case SRV_Channel::k_elevator:
170  case SRV_Channel::k_elevator_with_input:
171  pwm = mix_output_angle(i, pitch);
172  break;
173 
174  case SRV_Channel::k_rudder:
175  case SRV_Channel::k_steering:
176  pwm = mix_output_angle(i, rudder);
177  break;
178 
179  case SRV_Channel::k_throttle:
180  case SRV_Channel::k_throttleLeft:
181  case SRV_Channel::k_throttleRight:
182  if (mixing.throttle_is_angle) {
183  pwm = mix_output_angle(i, throttle);
184  } else {
185  pwm = mix_output_range(i, throttle);
186  }
187  break;
188 
189  case SRV_Channel::k_flap:
190  case SRV_Channel::k_flap_auto:
191  // zero flaps
192  pwm = mix_output_range(i, 0);
193  break;
194 
195  case SRV_Channel::k_elevon_left:
196  case SRV_Channel::k_dspoilerLeft1:
197  case SRV_Channel::k_dspoilerLeft2:
198  // treat differential spoilers as elevons
199  pwm = mix_output_angle(i, mix_elevon_vtail(roll, pitch, true));
200  break;
201 
202  case SRV_Channel::k_elevon_right:
203  case SRV_Channel::k_dspoilerRight1:
204  case SRV_Channel::k_dspoilerRight2:
205  // treat differential spoilers as elevons
206  pwm = mix_output_angle(i, mix_elevon_vtail(roll, pitch, false));
207  break;
208 
209  case SRV_Channel::k_vtail_left:
210  pwm = mix_output_angle(i, mix_elevon_vtail(rudder, pitch, false));
211  break;
212 
213  case SRV_Channel::k_vtail_right:
214  pwm = mix_output_angle(i, mix_elevon_vtail(rudder, pitch, true));
215  break;
216 
217  default:
218  break;
219  }
220  }
