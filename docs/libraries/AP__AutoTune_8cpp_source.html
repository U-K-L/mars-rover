110 /*
111  start an autotune session
112 */
113 void AP_AutoTune::start(void)
114 {
115  running = true;
116  state = DEMAND_UNSATURATED;
117  uint32_t now = AP_HAL::millis();
118 
119  state_enter_ms = now;
120  last_save_ms = now;
121 
122  last_save = current;
123  restore = current;
124 
125  uint8_t level = aparm.autotune_level;
126  if (level > ARRAY_SIZE(tuning_table)) {
127  level = ARRAY_SIZE(tuning_table);
128  }
129  if (level < 1) {
130  level = 1;
131  }
132 
133  current.rmax.set(tuning_table[level-1].rmax);
134  // D gain is scaled to a fixed ratio of P gain
135  current.D.set(tuning_table[level-1].Dratio * current.P);
136  current.tau.set(tuning_table[level-1].tau);
137 
138  current.imax = constrain_float(current.imax, AUTOTUNE_MIN_IMAX, AUTOTUNE_MAX_IMAX);
139 
140  // force a fixed ratio of I to D gain on the rate feedback path
141  current.I = 0.5f * current.D / current.tau;
142 
143  next_save = current;
144 
145  Debug("START P -> %.3f\n", current.P.get());
146 }

160 /*
161  one update cycle of the autotuner
162  */
163 void AP_AutoTune::update(float desired_rate, float achieved_rate, float servo_out)
164 {
165  if (!running) {
166  return;
167  }
168  check_save();
169 
170  // see what state we are in
171  ATState new_state;
172  float abs_desired_rate = fabsf(desired_rate);
173  uint32_t now = AP_HAL::millis();
174 
175  if (fabsf(servo_out) >= 45) {
176  // we have saturated the servo demand (not including
177  // integrator), we cannot get any information that would allow
178  // us to increase the gain
179  saturated_surfaces = true;
180  }
181 
182  if (abs_desired_rate < 0.8f * current.rmax) {
183  // we are not demanding max rate
184  new_state = DEMAND_UNSATURATED;
185  } else if (fabsf(achieved_rate) > abs_desired_rate) {
186  new_state = desired_rate > 0 ? DEMAND_OVER_POS : DEMAND_OVER_NEG;
187  } else {
188  new_state = desired_rate > 0 ? DEMAND_UNDER_POS : DEMAND_UNDER_NEG;
189  }
190  if (new_state != state) {
191  check_state_exit(now - state_enter_ms);
192  state = new_state;
193  state_enter_ms = now;
194  saturated_surfaces = false;
195  }
196  if (state != DEMAND_UNSATURATED) {
197  write_log(servo_out, desired_rate, achieved_rate);
198  }
199 }

201 /*
202  called when we change state to see if we should change gains
203  */
204 void AP_AutoTune::check_state_exit(uint32_t state_time_ms)
205 {
206  switch (state) {
207  case DEMAND_UNSATURATED:
208  break;
209  case DEMAND_UNDER_POS:
210  case DEMAND_UNDER_NEG:
211  // we increase P if we have not saturated the surfaces during
212  // this state, and we have
213  if (state_time_ms >= AUTOTUNE_UNDERSHOOT_TIME && !saturated_surfaces) {
214  current.P.set(current.P * (100+AUTOTUNE_INCREASE_STEP) * 0.01f);
215  if (current.P > AUTOTUNE_MAX_P) {
216  current.P = AUTOTUNE_MAX_P;
217  }
218  Debug("UNDER P -> %.3f\n", current.P.get());
219  }
220  current.D.set(tuning_table[aparm.autotune_level-1].Dratio * current.P);
221  break;
222  case DEMAND_OVER_POS:
223  case DEMAND_OVER_NEG:
224  if (state_time_ms >= AUTOTUNE_OVERSHOOT_TIME) {
225  current.P.set(current.P * (100-AUTOTUNE_DECREASE_STEP) * 0.01f);
226  if (current.P < AUTOTUNE_MIN_P) {
227  current.P = AUTOTUNE_MIN_P;
228  }
229  Debug("OVER P -> %.3f\n", current.P.get());
230  }
231  current.D.set(tuning_table[aparm.autotune_level-1].Dratio * current.P);
232  break;
233  }
234 }

236 /*
237  see if we should save new values
238  */
239 void AP_AutoTune::check_save(void)
240 {
241  if (AP_HAL::millis() - last_save_ms < AUTOTUNE_SAVE_PERIOD) {
242  return;
243  }
244 
245  // save the next_save values, which are the autotune value from
246  // the last save period. This means the pilot has
247  // AUTOTUNE_SAVE_PERIOD milliseconds to decide they don't like the
248  // gains and switch out of autotune
249  ATGains tmp = current;
250 
251  save_gains(next_save);
252  Debug("SAVE P -> %.3f\n", current.P.get());
253 
254  // restore our current gains
255  current = tmp;
256 
257  // if the pilot exits autotune they get these saved values
258  restore = next_save;
259 
260  // the next values to save will be the ones we are flying now
261  next_save = current;
262  last_save_ms = AP_HAL::millis();
263 }
