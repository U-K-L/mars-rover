51 void RCOutput::init()
52 {
53  uint8_t pwm_count = AP_BoardConfig::get_pwm_count();
54  for (uint8_t i = 0; i < NUM_GROUPS; i++ ) {
55  //Start Pwm groups
56  pwm_group &group = pwm_group_list[i];
57  group.current_mode = MODE_PWM_NORMAL;
58  for (uint8_t j = 0; j < 4; j++ ) {
59  uint8_t chan = group.chan[j];
60  if (chan >= pwm_count) {
61  group.chan[j] = CHAN_DISABLED;
62  }
63  if (group.chan[j] != CHAN_DISABLED) {
64  num_fmu_channels = MAX(num_fmu_channels, group.chan[j]+1);
65  group.ch_mask |= (1U<<group.chan[j]);
66  }
67  }
68  if (group.ch_mask != 0) {
69  pwmStart(group.pwm_drv, &group.pwm_cfg);
70  group.pwm_started = true;
71  }
72  chVTObjectInit(&group.dma_timeout);
73  }
74 
75 #if HAL_WITH_IO_MCU
76  if (AP_BoardConfig::io_enabled()) {
77  iomcu.init();
78  // with IOMCU the local (FMU) channels start at 8
79  chan_offset = 8;
80  }
81 #endif
82  chMtxObjectInit(&trigger_mutex);
83 
84  // setup default output rate of 50Hz
85  set_freq(0xFFFF ^ ((1U<<chan_offset)-1), 50);
86 
87 #ifdef HAL_GPIO_PIN_SAFETY_IN
88  safety_state = AP_HAL::Util::SAFETY_DISARMED;
89 #endif

92 /*
93  setup the output frequency for a group and start pwm output
94  */
95 void RCOutput::set_freq_group(pwm_group &group)
96 {
97  if (mode_requires_dma(group.current_mode)) {
98  // speed setup in DMA handler
99  return;
100  }
101 
102  uint16_t freq_set = group.rc_frequency;
103  uint32_t old_clock = group.pwm_cfg.frequency;
104  uint32_t old_period = group.pwm_cfg.period;
105 
106  if (freq_set > 400 || group.current_mode == MODE_PWM_ONESHOT125) {
107  // use a 8MHz clock for higher frequencies or for
108  // oneshot125. Using 8MHz for oneshot125 results in the full
109  // 1000 steps for smooth output
110  group.pwm_cfg.frequency = 8000000;
111  } else if (freq_set <= 400) {
112  // use a 1MHz clock
113  group.pwm_cfg.frequency = 1000000;
114  }
115 
116  // check if the frequency is possible, and keep halving
117  // down to 1MHz until it is OK with the hardware timer we
118  // are using. If we don't do this we'll hit an assert in
119  // the ChibiOS PWM driver on some timers
120  PWMDriver *pwmp = group.pwm_drv;
121  uint32_t psc = (pwmp->clock / pwmp->config->frequency) - 1;
122  while ((psc > 0xFFFF || ((psc + 1) * pwmp->config->frequency) != pwmp->clock) &&
123  group.pwm_cfg.frequency > 1000000) {
124  group.pwm_cfg.frequency /= 2;
125  psc = (pwmp->clock / pwmp->config->frequency) - 1;
126  }
127 
128  if (group.current_mode == MODE_PWM_ONESHOT ||
129  group.current_mode == MODE_PWM_ONESHOT125) {
130  // force a period of 0, meaning no pulses till we trigger
131  group.pwm_cfg.period = 0;
132  } else {
133  group.pwm_cfg.period = group.pwm_cfg.frequency/freq_set;
134  }
135 
136  bool force_reconfig = false;
137  for (uint8_t j=0; j<4; j++) {
138  if (group.pwm_cfg.channels[j].mode == PWM_OUTPUT_ACTIVE_LOW) {
139  group.pwm_cfg.channels[j].mode = PWM_OUTPUT_ACTIVE_HIGH;
140  force_reconfig = true;
141  }
142  if (group.pwm_cfg.channels[j].mode == PWM_COMPLEMENTARY_OUTPUT_ACTIVE_LOW) {
143  group.pwm_cfg.channels[j].mode = PWM_COMPLEMENTARY_OUTPUT_ACTIVE_HIGH;
144  force_reconfig = true;
145  }
146 
147  }
148 
149  if (old_clock != group.pwm_cfg.frequency ||
150  old_period != group.pwm_cfg.period ||
151  !group.pwm_started ||
152  force_reconfig) {
153  // we need to stop and start to setup the new clock
154  if (group.pwm_started) {
155  pwmStop(group.pwm_drv);
156  }
157  pwmStart(group.pwm_drv, &group.pwm_cfg);
158  group.pwm_started = true;
159  }
160  pwmChangePeriod(group.pwm_drv, group.pwm_cfg.period);
161 }

163 /*
164  set output frequency in HZ for a set of channels given by a mask
165  */
166 void RCOutput::set_freq(uint32_t chmask, uint16_t freq_hz)
167 {
168  //check if the request spans accross any of the channel groups
169  uint8_t update_mask = 0;
170 
171 #if HAL_WITH_IO_MCU
172  if (AP_BoardConfig::io_enabled()) {
173  // change frequency on IOMCU
174  uint16_t io_chmask = chmask & 0xFF;
175  if (freq_hz > 50) {
176  io_fast_channel_mask |= io_chmask;
177  } else {
178  io_fast_channel_mask &= ~io_chmask;
179  }
180  if (io_chmask) {
181  iomcu.set_freq(io_fast_channel_mask, freq_hz);
182  }
183  }
184 #endif
185 
186  // convert to a local (FMU) channel mask
187  chmask >>= chan_offset;
188  if (chmask == 0) {
189  return;
190  }
191 
192  /*
193  we enable the new frequency on all groups that have one
194  of the requested channels. This means we may enable high
195  speed on some channels that aren't requested, but that
196  is needed in order to fly a vehicle such a a hex
197  multicopter properly
198  */
199  for (uint8_t i = 0; i < NUM_GROUPS; i++ ) {
200  // greater than 400 doesn't give enough room at higher periods for
201  // the down pulse. This still allows for high rate with oneshot and dshot.
202  pwm_group &group = pwm_group_list[i];
203  uint16_t group_freq = freq_hz;
204  if (group_freq > 400 && group.current_mode != MODE_PWM_BRUSHED) {
205  group_freq = 400;
206  }
207  if ((group.ch_mask & chmask) != 0) {
208  group.rc_frequency = group_freq;
209  set_freq_group(group);
210  update_mask |= group.ch_mask;
211  }
212  if (group_freq > 50) {
213  fast_channel_mask |= group.ch_mask;
214  }
215  }
216 }

306 void RCOutput::write(uint8_t chan, uint16_t period_us)
307 {
308  if (chan >= max_channels) {
309  return;
310  }
311  last_sent[chan] = period_us;
312 
313 #if HAL_WITH_IO_MCU
314  // handle IO MCU channels
315  if (AP_BoardConfig::io_enabled()) {
316  uint16_t io_period_us = period_us;
317  if (iomcu_oneshot125 && ((1U<<chan) & io_fast_channel_mask)) {
318  // the iomcu only has one oneshot setting, so we need to scale by a factor
319  // of 8 here for oneshot125
320  io_period_us /= 8;
321  }
322  iomcu.write_channel(chan, io_period_us);
323  }
324 #endif
325  if (chan < chan_offset) {
326  return;
327  }
328 
329  if (safety_state == AP_HAL::Util::SAFETY_DISARMED && !(safety_mask & (1U<<chan))) {
330  // implement safety pwm value
331  period_us = safe_pwm[chan];
332  }
333 
334  chan -= chan_offset;
335 
336  period[chan] = period_us;
337 
338  if (chan < num_fmu_channels) {
339  active_fmu_channels = MAX(chan+1, active_fmu_channels);
340  if (!corked) {
341  push_local();
342  }
343  }

346 /*
347  push values to local channels from period[] array
348  */
349 void RCOutput::push_local(void)
350 {
351  if (active_fmu_channels == 0) {
352  return;
353  }
354  uint16_t outmask = (1U<<active_fmu_channels)-1;
355  outmask &= en_mask;
356 
357  uint16_t widest_pulse = 0;
358  uint8_t need_trigger = 0;
359  bool safety_on = hal.util->safety_switch_state() == AP_HAL::Util::SAFETY_DISARMED;
360 
361  for (uint8_t i = 0; i < NUM_GROUPS; i++ ) {
362  pwm_group &group = pwm_group_list[i];
363  if (serial_group) {
364  continue;
365  }
366  if (!group.pwm_started) {
367  continue;
368  }
369  for (uint8_t j = 0; j < 4; j++) {
370  uint8_t chan = group.chan[j];
371  if (chan == CHAN_DISABLED) {
372  continue;
373  }
374  if (outmask & (1UL<<chan)) {
375  uint32_t period_us = period[chan];
376 
377  if (safety_on && !(safety_mask & (1U<<(chan+chan_offset)))) {
378  // safety is on, overwride pwm
379  period_us = safe_pwm[chan+chan_offset];
380  }
381 
382  if (group.current_mode == MODE_PWM_BRUSHED) {
383  if (period_us <= _esc_pwm_min) {
384  period_us = 0;
385  } else if (period_us >= _esc_pwm_max) {
386  period_us = PWM_FRACTION_TO_WIDTH(group.pwm_drv, 1, 1);
387  } else {
388  period_us = PWM_FRACTION_TO_WIDTH(group.pwm_drv,\
389  (_esc_pwm_max - _esc_pwm_min), (period_us - _esc_pwm_min));
390  }
391  pwmEnableChannel(group.pwm_drv, j, period_us);
392  } else if (group.current_mode == MODE_PWM_ONESHOT125) {
393  // this gives us a width in 125 ns increments, giving 1000 steps over the 125 to 250 range
394  uint32_t width = ((group.pwm_cfg.frequency/1000000U) * period_us) / 8U;
395  pwmEnableChannel(group.pwm_drv, j, width);
396  }
397  else if (group.current_mode < MODE_PWM_DSHOT150) {
398  uint32_t width = (group.pwm_cfg.frequency/1000000U) * period_us;
399  pwmEnableChannel(group.pwm_drv, j, width);
400  }
401 #ifndef DISABLE_DSHOT
402  else if (group.current_mode >= MODE_PWM_DSHOT150 && group.current_mode <= MODE_PWM_DSHOT1200) {
403  // set period_us to time for pulse output, to enable very fast rates
404  period_us = dshot_pulse_time_us;
405  }
406 #endif //#ifndef DISABLE_DSHOT
407  if (period_us > widest_pulse) {
408  widest_pulse = period_us;
409  }
410  if (group.current_mode == MODE_PWM_ONESHOT ||
411  group.current_mode == MODE_PWM_ONESHOT125 ||
412  (group.current_mode >= MODE_PWM_DSHOT150 &&
413  group.current_mode <= MODE_PWM_DSHOT1200)) {
414  need_trigger |= (1U<<i);
415  }
416  }
417  }
418  }
419 
420  if (widest_pulse > 2300) {
421  widest_pulse = 2300;
422  }
423  trigger_widest_pulse = widest_pulse;
424 
425  trigger_groupmask = need_trigger;
426 
427  if (trigger_groupmask) {
428  trigger_groups();
429  }
430 }

502 /*
503  setup a group for DMA output at a given bitrate. The bit_width is
504  the value for a pulse width in the DMA buffer for a full bit.
505 
506  This is used for both DShot and serial output
507  */
508 bool RCOutput::setup_group_DMA(pwm_group &group, uint32_t bitrate, uint32_t bit_width, bool active_high)
509 {
510 #ifndef DISABLE_DSHOT
511  if (!group.dma_buffer) {
512  group.dma_buffer = (uint32_t *)hal.util->malloc_type(dshot_buffer_length, AP_HAL::Util::MEM_DMA_SAFE);
513  if (!group.dma_buffer) {
514  return false;
515  }
516  }
517  // for dshot we setup for DMAR based output
518  if (!group.dma_handle) {
519  group.dma_handle = new Shared_DMA(group.dma_up_stream_id, SHARED_DMA_NONE,
520  FUNCTOR_BIND_MEMBER(&RCOutput::dma_allocate, void, Shared_DMA *),
521  FUNCTOR_BIND_MEMBER(&RCOutput::dma_deallocate, void, Shared_DMA *));
522  if (!group.dma_handle) {
523  return false;
524  }
525  }
526 
527  // hold the lock during setup, to ensure there isn't a DMA operation ongoing
528  group.dma_handle->lock();
529 
530  // configure timer driver for DMAR at requested rate
531  if (group.pwm_started) {
532  pwmStop(group.pwm_drv);
533  group.pwm_started = false;
534  }
535 
536  // adjust frequency to give an allowed value given the
537  // clock. There is probably a better way to do this
538  uint32_t clock_hz = group.pwm_drv->clock;
539  uint32_t target_frequency = bitrate * bit_width;
540  uint32_t prescaler = clock_hz / target_frequency;
541  while ((clock_hz / prescaler) * prescaler != clock_hz && prescaler <= 0x8000) {
542  prescaler++;
543  }
544  uint32_t freq = clock_hz / prescaler;
545  if (prescaler > 0x8000) {
546  group.dma_handle->unlock();
547  return false;
548  }
549 
550  group.pwm_cfg.frequency = freq;
551  group.pwm_cfg.period = bit_width;
552  group.pwm_cfg.dier = TIM_DIER_UDE;
553  group.pwm_cfg.cr2 = 0;
554  group.bit_width_mul = (freq + (target_frequency/2)) / target_frequency;
555 
556  for (uint8_t j=0; j<4; j++) {
557  pwmmode_t mode = group.pwm_cfg.channels[j].mode;
558  if (mode != PWM_OUTPUT_DISABLED) {
559  if(mode == PWM_COMPLEMENTARY_OUTPUT_ACTIVE_LOW || mode == PWM_COMPLEMENTARY_OUTPUT_ACTIVE_HIGH) {
560  group.pwm_cfg.channels[j].mode = active_high?PWM_COMPLEMENTARY_OUTPUT_ACTIVE_HIGH:PWM_COMPLEMENTARY_OUTPUT_ACTIVE_LOW;
561  } else {
562  group.pwm_cfg.channels[j].mode = active_high?PWM_OUTPUT_ACTIVE_HIGH:PWM_OUTPUT_ACTIVE_LOW;
563  }
564  }
565  }
566 
567  pwmStart(group.pwm_drv, &group.pwm_cfg);
568  group.pwm_started = true;
569 
570  for (uint8_t j=0; j<4; j++) {
571  if (group.chan[j] != CHAN_DISABLED) {
572  pwmEnableChannel(group.pwm_drv, j, 0);
573  }
574  }
575  group.dma_handle->unlock();
576  return true;
577 #else
578  return false;
579 #endif //#ifndef DISABLE_DSHOT
580 }

583 /*
584  setup output mode for a group, using group.current_mode. Used to restore output
585  after serial operations
586  */
587 void RCOutput::set_group_mode(pwm_group &group)
588 {
589  if (group.pwm_started) {
590  pwmStop(group.pwm_drv);
591  group.pwm_started = false;
592  }
593 
594  switch (group.current_mode) {
595  case MODE_PWM_BRUSHED:
596  // force zero output initially
597  for (uint8_t i=0; i<4; i++) {
598  if (group.chan[i] == CHAN_DISABLED) {
599  continue;
600  }
601  uint8_t chan = chan_offset + group.chan[i];
602  write(chan, 0);
603  }
604  break;
605 
606  case MODE_PWM_DSHOT150 ... MODE_PWM_DSHOT1200: {
607  const uint16_t rates[(1 + MODE_PWM_DSHOT1200) - MODE_PWM_DSHOT150] = { 150, 300, 600, 1200 };
608  uint32_t rate = rates[uint8_t(group.current_mode - MODE_PWM_DSHOT150)] * 1000UL;
609  const uint32_t bit_period = 20;
610 
611  // configure timer driver for DMAR at requested rate
612  if (!setup_group_DMA(group, rate, bit_period, true)) {
613  group.current_mode = MODE_PWM_NONE;
614  break;
615  }
616 
617  // calculate min time between pulses
618  dshot_pulse_time_us = 1000000UL * dshot_bit_length / rate;
619  break;
620  }
621 
622  case MODE_PWM_ONESHOT:
623  case MODE_PWM_ONESHOT125:
624  // for oneshot we set a period of 0, which results in no pulses till we trigger
625  group.pwm_cfg.period = 0;
626  group.rc_frequency = 1;
627  if (group.pwm_started) {
628  pwmChangePeriod(group.pwm_drv, group.pwm_cfg.period);
629  }
630  break;
631 
632  case MODE_PWM_NORMAL:
633  case MODE_PWM_NONE:
634  // nothing needed
635  break;
636  }
637 
638  set_freq_group(group);
639 
640  if (group.current_mode != MODE_PWM_NONE &&
641  !group.pwm_started) {
642  pwmStart(group.pwm_drv, &group.pwm_cfg);
643  group.pwm_started = true;

733 /*

735  */

737 {

739  return;

743  // guarantee minimum pulse separation
744  hal.scheduler->delay_microseconds(min_pulse_trigger_us - now);

749  pwm_group &group = pwm_group_list[i];

752  continue;
753  }
754  if (group.current_mode == MODE_PWM_ONESHOT ||
755  group.current_mode == MODE_PWM_ONESHOT125) {

782 /*

784  safety switch update
785  */

787 {

851 /*
852  create a DSHOT 16 bit packet. Based on prepareDshotPacket from betaflight
853  */
854 uint16_t RCOutput::create_dshot_packet(const uint16_t value, bool telem_request)
855 {
856  uint16_t packet = (value << 1);
857 
858  if (telem_request) {
859  packet |= 1;
860  }
861 
862  // compute checksum
863  uint16_t csum = 0;
864  uint16_t csum_data = packet;
865  for (uint8_t i = 0; i < 3; i++) {
866  csum ^= csum_data;
867  csum_data >>= 4;
868  }
869  csum &= 0xf;
870  // append checksum
871  packet = (packet << 4) | csum;
872 
873  return packet;
874 }

877  fill in a DMA buffer for dshot
878  */
879 void RCOutput::fill_DMA_buffer_dshot(uint32_t *buffer, uint8_t stride, uint16_t packet, uint16_t clockmul)
880 {
881  const uint32_t DSHOT_MOTOR_BIT_0 = 7 * clockmul;
882  const uint32_t DSHOT_MOTOR_BIT_1 = 14 * clockmul;
883  uint16_t i = 0;

885  buffer[i * stride] = 0;
886  }

889  packet <<= 1;
890  }
891  for (; i<dshot_bit_length; i++) {
892  buffer[i * stride] = 0;

896 /*
897  send a set of DShot packets for a channel group
898  This call be called in blocking mode from the timer, in which case it waits for the DMA lock.
899  In normal operation it doesn't wait for the DMA lock.
900  */
901 void RCOutput::dshot_send(pwm_group &group, bool blocking)
902 {
903 #ifndef DISABLE_DSHOT
904  if (irq.waiter) {
905  // doing serial output, don't send DShot pulses
906  return;
907  }
908 
909  if (blocking) {
910  group.dma_handle->lock();
911  } else {
912  if (!group.dma_handle->lock_nonblock()) {
913  return;
914  }
915  }
916 
917  bool safety_on = hal.util->safety_switch_state() == AP_HAL::Util::SAFETY_DISARMED;
918 
919  memset((uint8_t *)group.dma_buffer, 0, dshot_buffer_length);
920 
921  for (uint8_t i=0; i<4; i++) {
922  uint8_t chan = group.chan[i];
923  if (chan != CHAN_DISABLED) {
924  uint16_t pwm = period[chan];
925 
926  if (safety_on && !(safety_mask & (1U<<(chan+chan_offset)))) {
927  // safety is on, overwride pwm
928  pwm = safe_pwm[chan+chan_offset];
929  }
930 
931  const uint16_t chan_mask = (1U<<chan);
932  if (pwm == 0) {
933  // no output
934  continue;
935  }
936 
937  pwm = constrain_int16(pwm, 1000, 2000);
938  uint16_t value = 2 * (pwm - 1000);
939 
940  if (chan_mask & (reversible_mask>>chan_offset)) {
941  // this is a DShot-3D output, map so that 1500 PWM is zero throttle reversed
942  if (value < 1000) {
943  value = 2000 - value;
944  } else if (value > 1000) {
945  value = value - 1000;
946  } else {
947  // mid-throttle is off
948  value = 0;
949  }
950  }
951  if (value != 0) {
952  // dshot values are from 48 to 2047. Zero means off.
953  value += 47;
954  }
955 
956  bool request_telemetry = (telem_request_mask & chan_mask)?true:false;
957  uint16_t packet = create_dshot_packet(value, request_telemetry);
958  if (request_telemetry) {
959  telem_request_mask &= ~chan_mask;
960  }
961  fill_DMA_buffer_dshot(group.dma_buffer + i, 4, packet, group.bit_width_mul);
962  }
963  }
964 
965  // start sending the pulses out
966  send_pulses_DMAR(group, dshot_buffer_length);
967 
968  group.last_dshot_send_us = AP_HAL::micros64();
969 #endif //#ifndef DISABLE_DSHOT
970 }

972 /*
973  send a series of pulses for a group using DMAR. Pulses must have
974  been encoded into the group dma_buffer with interleaving for the 4
975  channels in the group
976  */
977 void RCOutput::send_pulses_DMAR(pwm_group &group, uint32_t buffer_length)
978 {
979 #ifndef DISABLE_DSHOT
980  /*
981  The DMA approach we are using is based on the DMAR method from
982  betaflight. We use the TIMn_UP DMA channel for the timer, and
983  setup an interleaved set of pulse durations, with a stride of 4
984  (for the 4 channels). We use the DMAR register to point the DMA
985  engine at the 4 CCR registers of the timer, so it fills in the
986  pulse widths for each timer in turn. This means we only use a
987  single DMA channel for groups of 4 timer channels. See the "DMA
988  address for full transfer TIMx_DMAR" section of the
989  datasheet. Many thanks to the betaflight developers for coming
990  up with this great method.
991  */
992  dmaStreamSetPeripheral(group.dma, &(group.pwm_drv->tim->DMAR));
993  cacheBufferFlush(group.dma_buffer, buffer_length);
994  dmaStreamSetMemory0(group.dma, group.dma_buffer);
995  dmaStreamSetTransactionSize(group.dma, buffer_length/sizeof(uint32_t));
996  dmaStreamSetFIFO(group.dma, STM32_DMA_FCR_DMDIS | STM32_DMA_FCR_FTH_FULL);
997  dmaStreamSetMode(group.dma,
998  STM32_DMA_CR_CHSEL(group.dma_up_channel) |
999  STM32_DMA_CR_DIR_M2P | STM32_DMA_CR_PSIZE_WORD | STM32_DMA_CR_MSIZE_WORD |
1000  STM32_DMA_CR_MINC | STM32_DMA_CR_PL(3) |
1001  STM32_DMA_CR_TEIE | STM32_DMA_CR_TCIE);
1002 
1003  // setup for 4 burst strided transfers. 0x0D is the register
1004  // address offset of the CCR registers in the timer peripheral
1005  group.pwm_drv->tim->DCR = 0x0D | STM32_TIM_DCR_DBL(3);
1006 
1007  dmaStreamEnable(group.dma);
1008 #endif //#ifndef DISABLE_DSHOT
1009 }

1046  firmware flashing

1052 {

1054  chan -= chan_offset;
1055  chanmask >>= chan_offset;

1057 
1058  // find the channel group
1059  for (uint8_t i = 0; i < NUM_GROUPS; i++ ) {
1060  pwm_group &group = pwm_group_list[i];

1062  // can't do serial output with brushed motors

1067  for (uint8_t j=0; j<4; j++) {
1068  if (group.chan[j] == chan) {
1069  group.serial.chan = j;
1070  }

1072  break;

1077  if (serial_group) {
1078  // shutdown old group
1079  serial_end();
