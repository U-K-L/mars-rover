26 // to use the minimum number of floating point operations

385 float Vector3<T>::angle(const Vector3<T> &v2) const
386 {
387  const float len = this->length() * v2.length();
388  if (len <= 0) {
389  return 0.0f;
390  }
391  const float cosv = ((*this)*v2) / len;
392  if (fabsf(cosv) >= 1) {
393  return 0.0f;
394  }
395  return acosf(cosv);

417 // distance from the tip of this vector to a line segment specified by two vectors
418 template <typename T>
419 float Vector3<T>::distance_to_segment(const Vector3<T> &seg_start, const Vector3<T> &seg_end) const
420 {
421  // triangle side lengths
422  const float a = (*this-seg_start).length();
423  const float b = (seg_start-seg_end).length();
424  const float c = (seg_end-*this).length();
425 
426  // protect against divide by zero later
427  if (::is_zero(b)) {
428  return 0.0f;
429  }
430 
431  // semiperimeter of triangle
432  const float s = (a+b+c) * 0.5f;
433 
434  float area_squared = s*(s-a)*(s-b)*(s-c);
435  // area must be constrained above 0 because a triangle could have 3 points could be on a line and float rounding could push this under 0
436  if (area_squared < 0.0f) {
437  area_squared = 0.0f;
438  }
439  const float area = safe_sqrt(area_squared);
440  return 2.0f*area/b;
441 }
