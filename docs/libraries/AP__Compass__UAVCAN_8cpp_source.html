76  AP_Compass_UAVCAN* driver = nullptr;
77  for (uint8_t i = 0; i < COMPASS_MAX_BACKEND; i++) {
78  if (!_detected_modules[i].driver && _detected_modules[i].ap_uavcan) {
79  // Register new Compass mode to a backend

124 AP_Compass_UAVCAN* AP_Compass_UAVCAN::get_uavcan_backend(AP_UAVCAN* ap_uavcan, uint8_t node_id, uint8_t sensor_id)
125 {
126  if (ap_uavcan == nullptr) {
127  return nullptr;
128  }
129  for (uint8_t i=0; i<COMPASS_MAX_BACKEND; i++) {
130  if (_detected_modules[i].driver &&
131  _detected_modules[i].ap_uavcan == ap_uavcan &&
132  _detected_modules[i].node_id == node_id &&
133  _detected_modules[i].sensor_id == sensor_id) {
134  return _detected_modules[i].driver;
135  }
136  }
137 
138  bool already_detected = false;
139  // Check if there's an empty spot for possible registeration
140  for (uint8_t i = 0; i < COMPASS_MAX_BACKEND; i++) {
141  if (_detected_modules[i].ap_uavcan == ap_uavcan &&
142  _detected_modules[i].node_id == node_id &&
143  _detected_modules[i].sensor_id == sensor_id) {
144  // Already Detected
145  already_detected = true;
146  break;
147  }
148  }
149  if (!already_detected) {
150  for (uint8_t i = 0; i < COMPASS_MAX_BACKEND; i++) {
151  if (nullptr == _detected_modules[i].ap_uavcan) {
152  _detected_modules[i].ap_uavcan = ap_uavcan;
153  _detected_modules[i].node_id = node_id;
154  _detected_modules[i].sensor_id = sensor_id;
155  break;
156  }
157  }
158  }
159  return nullptr;
160 }
