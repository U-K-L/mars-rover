28 void SRV_Channel::output_ch(void)
29 {
30  int8_t passthrough_from = -1;
31 
32  // take care of special function cases
33  switch(function)
34  {
35  case k_manual: // manual
36  passthrough_from = ch_num;
37  break;
38  case k_rcin1 ... k_rcin16: // rc pass-thru
39  passthrough_from = int8_t(function - k_rcin1);
40  break;
41  }
42  if (passthrough_from != -1) {
43  // we are doing passthrough from input to output for this channel
44  RC_Channel *c = rc().channel(passthrough_from);
45  if (c) {
46  if (SRV_Channels::passthrough_disabled()) {
47  output_pwm = c->get_radio_trim();
48  } else {
49  const int16_t radio_in = c->get_radio_in();
50  if (!ign_small_rcin_changes) {
51  output_pwm = radio_in;
52  previous_radio_in = radio_in;
53  } else {
54  // check if rc input value has changed by more than the deadzone
55  if (abs(radio_in - previous_radio_in) > c->get_dead_zone()) {
56  output_pwm = radio_in;
57  ign_small_rcin_changes = false;
58  }
59  }
60  }
61  }
62  }
63  if (!(SRV_Channels::disabled_mask & (1U<<ch_num))) {
64  hal.rcout->write(ch_num, output_pwm);
65  }
66 }

142 {

148  for (uint8_t i = 0; i < SRV_Channel::k_nr_aux_servo_functions; i++) {

164 void SRV_Channels::enable_aux_servos()
165 {
166  hal.rcout->set_default_rate(uint16_t(_singleton->default_rate.get()));
167 
168  update_aux_servo_function();
169 
170  // enable all channels that are set to a valid function. This
171  // includes k_none servos, which allows those to get their initial
172  // trim value on startup
173  for (uint8_t i = 0; i < NUM_SERVO_CHANNELS; i++) {
174  SRV_Channel &c = channels[i];
175  // see if it is a valid function
176  if ((uint8_t)c.function.get() < SRV_Channel::k_nr_aux_servo_functions) {
177  hal.rcout->enable_ch(c.ch_num);
178  }
179 
180  /*
181  for channels which have been marked as digital output then the
182  MIN/MAX/TRIM values have no meaning for controlling output, as
183  the HAL handles the scaling. We still need to cope with places
184  in the code that may try to set a PWM value however, so to
185  ensure consistency we force the MIN/MAX/TRIM to be consistent
186  across all digital channels. We use a MIN/MAX of 1000/2000, and
187  set TRIM to either 1000 or 1500 depending on whether the channel
188  is reversible
189  */
190  if (digital_mask & (1U<<i)) {
191  c.servo_min.set(1000);
192  c.servo_max.set(2000);
193  if (reversible_mask & (1U<<i)) {
194  c.servo_trim.set(1500);
195  } else {
196  c.servo_trim.set(1000);
197  }
198  }
199  }
200 
201 #if HAL_SUPPORT_RCOUT_SERIAL
202  blheli_ptr->update();
203 #endif
204 }

403 /*
404  set servo_out and angle_min/max, then calc_pwm and output a
405  value. This is used to move a AP_Mount servo
406  */
407 void
408 SRV_Channels::move_servo(SRV_Channel::Aux_servo_function_t function,
409  int16_t value, int16_t angle_min, int16_t angle_max)
410 {
411  if (!function_assigned(function)) {
412  return;
413  }
414  if (angle_max <= angle_min) {
415  return;
416  }
417  float v = float(value - angle_min) / float(angle_max - angle_min);
418  v = constrain_float(v, 0.0f, 1.0f);
419  for (uint8_t i = 0; i < NUM_SERVO_CHANNELS; i++) {
420  SRV_Channel &c = channels[i];
421  if (c.function.get() == function) {
422  float v2 = c.get_reversed()? (1-v) : v;
423  uint16_t pwm = c.servo_min + v2 * (c.servo_max - c.servo_min);
424  c.set_output_pwm(pwm);
425  }
426  }
427 }

564 /*
565  auto-adjust channel trim from an integrator value. Positive v means
566  adjust trim up. Negative means decrease
567  */
568 void SRV_Channels::adjust_trim(SRV_Channel::Aux_servo_function_t function, float v)
569 {
570  if (is_zero(v)) {
571  return;
572  }
573  for (uint8_t i=0; i<NUM_SERVO_CHANNELS; i++) {
574  SRV_Channel &c = channels[i];
575  if (function != (SRV_Channel::Aux_servo_function_t)(c.function.get())) {
576  continue;
577  }
578  float change = c.reversed?-v:v;
579  uint16_t new_trim = c.servo_trim;
580  float trim_scaled = float(c.servo_trim - c.servo_min) / (c.servo_max - c.servo_min);
581  if (change > 0 && trim_scaled < 0.6f) {
582  new_trim++;
583  } else if (change < 0 && trim_scaled > 0.4f) {
584  new_trim--;
585  } else {
586  return;
587  }
588  c.servo_trim.set(new_trim);
589 
590  trimmed_mask |= 1U<<i;
591  }
592 }

641 /*
642  limit slew rate for an output function to given rate in percent per
643  second. This assumes output has not yet done to the hal
644  */
645 void SRV_Channels::limit_slew_rate(SRV_Channel::Aux_servo_function_t function, float slew_rate, float dt)
646 {
647  if (slew_rate <= 0) {
648  // nothing to do
649  return;
650  }
651  for (uint8_t i=0; i<NUM_SERVO_CHANNELS; i++) {
652  SRV_Channel &c = channels[i];
653  if (c.function == function) {
654  c.calc_pwm(functions[function].output_scaled);
655  uint16_t last_pwm = hal.rcout->read_last_sent(c.ch_num);
656  if (last_pwm == c.output_pwm) {
657  continue;
658  }
659  uint16_t max_change = (c.get_output_max() - c.get_output_min()) * slew_rate * dt * 0.01f;
660  if (max_change == 0 || dt > 1) {
661  // always allow some change. If dt > 1 then assume we
662  // are just starting out, and only allow a small
663  // change for this loop

713 /*
714  upgrade RC* parameters into SERVO* parameters. This does the following:
715 
716  - copies MIN/MAX/TRIM values from old RC parameters into new RC* parameters and SERVO* parameters.
717  - copies RCn_FUNCTION to SERVOn_FUNCTION
718  - maps old RCn_REV to SERVOn_REVERSE and RCn_REVERSE
719 
720  aux_channel_mask is a bitmask of which channels were RC_Channel_aux channels
721 
722  Note that this code is highly dependent on the parameter indexing of
723  the old RC_Channel and RC_Channel_aux objects.
724 
725  If rcmap is passed in then the vehicle code also wants functions for
726  the first 4 output channels to be remapped
727 
728  We return true if an upgrade has been done. This allows the caller
729  to make any vehicle specific upgrades that may be needed
730 */
731 bool SRV_Channels::upgrade_parameters(const uint8_t rc_keys[14], uint16_t aux_channel_mask, RCMapper *rcmap)
732 {
733  // use SERVO16_FUNCTION as a marker to say that we have run the upgrade already
734  if (channels[15].function.configured_in_storage()) {
735  // upgrade already done
736  return false;
737  }
738 
739  // old system had 14 RC channels
740  for (uint8_t i=0; i<14; i++) {
741  uint8_t k = rc_keys[i];
742  if (k == 0) {
743  // missing parameter. Some vehicle types didn't have all parameters
744  continue;
745  }
746  SRV_Channel &srv_chan = channels[i];
747  RC_Channel *rc_chan = rc().channel(i);
748  enum {
749  FLAG_NONE=0,
750  FLAG_IS_REVERSE=1,
751  FLAG_AUX_ONLY=2
752  };
753  const struct mapping {
754  uint8_t old_index;
755  AP_Param *new_srv_param;
756  AP_Param *new_rc_param;
757  enum ap_var_type type;
758  uint8_t flags;
759  } mapping[] = {
760  { 0, &srv_chan.servo_min, &rc_chan->radio_min, AP_PARAM_INT16, FLAG_NONE },
761  { 1, &srv_chan.servo_trim, &rc_chan->radio_trim, AP_PARAM_INT16, FLAG_NONE },
762  { 2, &srv_chan.servo_max, &rc_chan->radio_max, AP_PARAM_INT16, FLAG_NONE },
763  { 3, &srv_chan.reversed, &rc_chan->reversed, AP_PARAM_INT8, FLAG_IS_REVERSE },
764  { 1, &srv_chan.function, nullptr, AP_PARAM_INT8, FLAG_AUX_ONLY },
765  };
766  bool is_aux = aux_channel_mask & (1U<<i);
767 
768  for (uint8_t j=0; j<ARRAY_SIZE(mapping); j++) {
769  const struct mapping &m = mapping[j];
770  AP_Param::ConversionInfo info;
771  AP_Int8 v8;
772  AP_Int16 v16;
773  AP_Param *v = m.type == AP_PARAM_INT16?(AP_Param*)&v16:(AP_Param*)&v8;
774  bool aux_only = (m.flags & FLAG_AUX_ONLY)!=0;
775  if (!is_aux && aux_only) {
776  continue;
777  }
778  info.old_key = k;
779  info.type = m.type;
780  info.new_name = nullptr;
781 
782  // if this was an aux channel we need to shift by 6 bits, but not for RCn_FUNCTION
783  info.old_group_element = (is_aux && !aux_only)?(m.old_index<<6):m.old_index;
784 
785  if (!AP_Param::find_old_parameter(&info, v)) {
786  // the parameter wasn't set in the old eeprom
787  continue;
788  }
789 
790  if (m.flags & FLAG_IS_REVERSE) {
791  // special mapping from RCn_REV to RCn_REVERSED
792  v8.set(v8.get() == -1?1:0);
793  }
794 
795  if (!m.new_srv_param->configured_in_storage()) {
796  // not configured yet in new eeprom
797  if (m.type == AP_PARAM_INT16) {
798  ((AP_Int16 *)m.new_srv_param)->set_and_save_ifchanged(v16.get());
799  } else {
800  ((AP_Int8 *)m.new_srv_param)->set_and_save_ifchanged(v8.get());
801  }
802  }
803  if (m.new_rc_param && !m.new_rc_param->configured_in_storage()) {
804  // not configured yet in new eeprom
805  if (m.type == AP_PARAM_INT16) {
806  ((AP_Int16 *)m.new_rc_param)->set_and_save_ifchanged(v16.get());
807  } else {
808  ((AP_Int8 *)m.new_rc_param)->set_and_save_ifchanged(v8.get());
809  }
810  }
811  }
812  }
813 
814  if (rcmap != nullptr) {
815  // we need to make the output functions from the rcmapped inputs
816  const int8_t func_map[4] = { channels[0].function.get(),
817  channels[1].function.get(),
818  channels[2].function.get(),
819  channels[3].function.get() };
820  const uint8_t map[4] = { rcmap->roll(), rcmap->pitch(), rcmap->throttle(), rcmap->yaw() };
821  for (uint8_t i=0; i<4; i++) {
822  uint8_t m = uint8_t(map[i]-1);
823  if (m != i && m < 4) {
824  channels[m].function.set_and_save_ifchanged(func_map[i]);
825  }
826  }
827  }
828 
829  // mark the upgrade as having been done
830  channels[15].function.set_and_save(channels[15].function.get());
831 
832  return true;
833 }
