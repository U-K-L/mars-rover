50 bool AP_Landing::type_slope_verify_land(const Location &prev_WP_loc, Location &next_WP_loc, const Location &current_loc,
51  const float height, const float sink_rate, const float wp_proportion, const uint32_t last_flying_ms, const bool is_armed, const bool is_flying, const bool rangefinder_state_in_range)
52 {
53  // we don't 'verify' landing in the sense that it never completes,
54  // so we don't verify command completion. Instead we use this to
55  // adjust final landing parameters
56 
57  // determine stage
58  if (type_slope_stage == SLOPE_STAGE_NORMAL) {
59  const bool heading_lined_up = abs(nav_controller->bearing_error_cd()) < 1000 && !nav_controller->data_is_stale();
60  const bool on_flight_line = fabsf(nav_controller->crosstrack_error()) < 5.0f && !nav_controller->data_is_stale();
61  const bool below_prev_WP = current_loc.alt < prev_WP_loc.alt;
62  if ((mission.get_prev_nav_cmd_id() == MAV_CMD_NAV_LOITER_TO_ALT) ||
63  (wp_proportion >= 0 && heading_lined_up && on_flight_line) ||
64  (wp_proportion > 0.15f && heading_lined_up && below_prev_WP) ||
65  (wp_proportion > 0.5f)) {
66  type_slope_stage = SLOPE_STAGE_APPROACH;
67  }
68  }
69 
70  /* Set land_complete (which starts the flare) under 3 conditions:
71  1) we are within LAND_FLARE_ALT meters of the landing altitude
72  2) we are within LAND_FLARE_SEC of the landing point vertically
73  by the calculated sink rate (if LAND_FLARE_SEC != 0)
74  3) we have gone past the landing point and don't have
75  rangefinder data (to prevent us keeping throttle on
76  after landing if we've had positive baro drift)
77  */
78 
79  // flare check:
80  // 1) below flare alt/sec requires approach stage check because if sec/alt are set too
81  // large, and we're on a hard turn to line up for approach, we'll prematurely flare by
82  // skipping approach phase and the extreme roll limits will make it hard to line up with runway
83  // 2) passed land point and don't have an accurate AGL
84  // 3) probably crashed (ensures motor gets turned off)
85 
86  const bool on_approach_stage = type_slope_is_on_approach();
87  const bool below_flare_alt = (height <= flare_alt);
88  const bool below_flare_sec = (flare_sec > 0 && height <= sink_rate * flare_sec);
89  const bool probably_crashed = (aparm.crash_detection_enable && fabsf(sink_rate) < 0.2f && !is_flying);
90 
91  if ((on_approach_stage && below_flare_alt) ||
92  (on_approach_stage && below_flare_sec && (wp_proportion > 0.5)) ||
93  (!rangefinder_state_in_range && wp_proportion >= 1) ||
94  probably_crashed) {
95 
96  const AP_GPS &gps = AP::gps();
97  if (type_slope_stage != SLOPE_STAGE_FINAL) {
98  type_slope_flags.post_stats = true;
99  if (is_flying && (AP_HAL::millis()-last_flying_ms) > 3000) {
100  gcs().send_text(MAV_SEVERITY_CRITICAL, "Flare crash detected: speed=%.1f", (double)gps.ground_speed());
101  } else {
102  gcs().send_text(MAV_SEVERITY_INFO, "Flare %.1fm sink=%.2f speed=%.1f dist=%.1f",
103  (double)height, (double)sink_rate,
104  (double)gps.ground_speed(),
105  (double)current_loc.get_distance(next_WP_loc));
106  }
107 
108  type_slope_stage = SLOPE_STAGE_FINAL;
109 
110  // Check if the landing gear was deployed before landing
111  // If not - go around
112  AP_LandingGear *LG_inst = AP_LandingGear::get_singleton();
113  if (LG_inst != nullptr && !LG_inst->check_before_land()) {
114  type_slope_request_go_around();
115  gcs().send_text(MAV_SEVERITY_CRITICAL, "Landing gear was not deployed");
116  }
117  }
118 
119  if (gps.ground_speed() < 3) {
120  // reload any airspeed or groundspeed parameters that may have
121  // been set for landing. We don't do this till ground
122  // speed drops below 3.0 m/s as otherwise we will change
123  // target speeds too early.
124  aparm.airspeed_cruise_cm.load();
125  aparm.min_gndspeed_cm.load();
126  aparm.throttle_cruise.load();
127  }
128  } else if (type_slope_stage == SLOPE_STAGE_APPROACH && pre_flare_airspeed > 0) {
129  bool reached_pre_flare_alt = pre_flare_alt > 0 && (height <= pre_flare_alt);
130  bool reached_pre_flare_sec = pre_flare_sec > 0 && (height <= sink_rate * pre_flare_sec);
131  if (reached_pre_flare_alt || reached_pre_flare_sec) {
132  type_slope_stage = SLOPE_STAGE_PREFLARE;
133  }
134  }
135 
136  /*
137  when landing we keep the L1 navigation waypoint 200m ahead. This
138  prevents sudden turns if we overshoot the landing point
139  */
140  struct Location land_WP_loc = next_WP_loc;
141 
142  int32_t land_bearing_cd = prev_WP_loc.get_bearing_to(next_WP_loc);
143  land_WP_loc.offset_bearing(land_bearing_cd * 0.01f, prev_WP_loc.get_distance(current_loc) + 200);
144  nav_controller->update_waypoint(prev_WP_loc, land_WP_loc);
145 
146  // once landed and stationary, post some statistics
147  // this is done before disarm_if_autoland_complete() so that it happens on the next loop after the disarm
148  if (type_slope_flags.post_stats && !is_armed) {
149  type_slope_flags.post_stats = false;
150  gcs().send_text(MAV_SEVERITY_INFO, "Distance from LAND point=%.2fm", (double)current_loc.get_distance(next_WP_loc));
151  }
152 
153  // check if we should auto-disarm after a confirmed landing
154  if (type_slope_stage == SLOPE_STAGE_FINAL) {
155  disarm_if_autoland_complete_fn();
156  }
157 
158  /*
159  we return false as a landing mission item never completes
160 
161  we stay on this waypoint unless the GCS commands us to change
162  mission item, reset the mission, command a go-around or finish
163  a land_abort procedure.
164  */
165  return false;
166 }

168 void AP_Landing::type_slope_adjust_landing_slope_for_rangefinder_bump(AP_Vehicle::FixedWing::Rangefinder_State &rangefinder_state, Location &prev_WP_loc, Location &next_WP_loc, const Location &current_loc, const float wp_distance, int32_t &target_altitude_offset_cm)
169 {
170  // check the rangefinder correction for a large change. When found, recalculate the glide slope. This is done by
171  // determining the slope from your current location to the land point then following that back up to the approach
172  // altitude and moving the prev_wp to that location. From there
173  float correction_delta = fabsf(rangefinder_state.last_stable_correction) - fabsf(rangefinder_state.correction);
174 
175  if (slope_recalc_shallow_threshold <= 0 ||
176  fabsf(correction_delta) < slope_recalc_shallow_threshold) {
177  return;
178  }
179 
180  rangefinder_state.last_stable_correction = rangefinder_state.correction;
181 
182  float corrected_alt_m = (adjusted_altitude_cm_fn() - next_WP_loc.alt)*0.01f - rangefinder_state.correction;
183  float total_distance_m = prev_WP_loc.get_distance(next_WP_loc);
184  float top_of_glide_slope_alt_m = total_distance_m * corrected_alt_m / wp_distance;
185  prev_WP_loc.alt = top_of_glide_slope_alt_m*100 + next_WP_loc.alt;
186 
187  // re-calculate auto_state.land_slope with updated prev_WP_loc
188  setup_landing_glide_slope(prev_WP_loc, next_WP_loc, current_loc, target_altitude_offset_cm);
189 
190  if (rangefinder_state.correction >= 0) { // we're too low or object is below us
191  // correction positive means we're too low so we should continue on with
192  // the newly computed shallower slope instead of pitching/throttling up
193 
194  } else if (slope_recalc_steep_threshold_to_abort > 0 && !type_slope_flags.has_aborted_due_to_slope_recalc) {
195  // correction negative means we're too high and need to point down (and speed up) to re-align
196  // to land on target. A large negative correction means we would have to dive down a lot and will
197  // generating way too much speed that we can not bleed off in time. It is better to remember
198  // the large baro altitude offset and abort the landing to come around again with the correct altitude
199  // offset and "perfect" slope.
200 
201  // calculate projected slope with projected alt
202  float new_slope_deg = degrees(atanf(slope));
203  float initial_slope_deg = degrees(atanf(initial_slope));
204 
205  // is projected slope too steep?
206  if (new_slope_deg - initial_slope_deg > slope_recalc_steep_threshold_to_abort) {
207  gcs().send_text(MAV_SEVERITY_INFO, "Landing slope too steep, aborting (%.0fm %.1fdeg)",
208  (double)rangefinder_state.correction, (double)(new_slope_deg - initial_slope_deg));
209  alt_offset = rangefinder_state.correction;
210  flags.commanded_go_around = true;
211  type_slope_flags.has_aborted_due_to_slope_recalc = true; // only allow this once.
212  Log();
213  }
214  }
215 }

223 /*
224  a special glide slope calculation for the landing approach
225 
226  During the land approach use a linear glide slope to a point
227  projected through the landing point. We don't use the landing point
228  itself as that leads to discontinuities close to the landing point,
229  which can lead to erratic pitch control
230  */
231 void AP_Landing::type_slope_setup_landing_glide_slope(const Location &prev_WP_loc, const Location &next_WP_loc, const Location &current_loc, int32_t &target_altitude_offset_cm)
232 {
233  float total_distance = prev_WP_loc.get_distance(next_WP_loc);
234 
235  // If someone mistakenly puts all 0's in their LAND command then total_distance
236  // will be calculated as 0 and cause a divide by 0 error below. Lets avoid that.
237  if (total_distance < 1) {
238  total_distance = 1;
239  }
240 
241  // height we need to sink for this WP
242  float sink_height = (prev_WP_loc.alt - next_WP_loc.alt)*0.01f;
243 
244  // current ground speed
245  float groundspeed = ahrs.groundspeed();
246  if (groundspeed < 0.5f) {
247  groundspeed = 0.5f;
248  }
249 
250  // calculate time to lose the needed altitude
251  float sink_time = total_distance / groundspeed;
252  if (sink_time < 0.5f) {
253  sink_time = 0.5f;
254  }
255 
256  // find the sink rate needed for the target location
257  float sink_rate = sink_height / sink_time;
258 
259  // the height we aim for is the one to give us the right flare point
260  float aim_height = flare_sec * sink_rate;
261  if (aim_height <= 0) {
262  aim_height = flare_alt;
263  }
264 
265  // don't allow the aim height to be too far above LAND_FLARE_ALT
266  if (flare_alt > 0 && aim_height > flare_alt*2) {
267  aim_height = flare_alt*2;
268  }
269 
270  // calculate slope to landing point
271  bool is_first_calc = is_zero(slope);
272  slope = (sink_height - aim_height) / total_distance;
273  if (is_first_calc) {
274  gcs().send_text(MAV_SEVERITY_INFO, "Landing glide slope %.1f degrees", (double)degrees(atanf(slope)));
275  }
276 
277  // time before landing that we will flare
278  float flare_time = aim_height / SpdHgt_Controller->get_land_sinkrate();
279 
280  // distance to flare is based on ground speed, adjusted as we
281  // get closer. This takes into account the wind
282  float flare_distance = groundspeed * flare_time;
283 
284  // don't allow the flare before half way along the final leg
285  if (flare_distance > total_distance/2) {
286  flare_distance = total_distance/2;
287  }
288 
289  // project a point 500 meters past the landing point, passing
290  // through the landing point
291  const float land_projection = 500;
292  int32_t land_bearing_cd = prev_WP_loc.get_bearing_to(next_WP_loc);
293 
294  // now calculate our aim point, which is before the landing
295  // point and above it
296  Location loc = next_WP_loc;
297  loc.offset_bearing(land_bearing_cd * 0.01f, -flare_distance);
298  loc.alt += aim_height*100;
299 
300  // calculate point along that slope 500m ahead
301  loc.offset_bearing(land_bearing_cd * 0.01f, land_projection);
302  loc.alt -= slope * land_projection * 100;
303 
304  // setup the offset_cm for set_target_altitude_proportion()
305  target_altitude_offset_cm = loc.alt - prev_WP_loc.alt;
306 
307  // calculate the proportion we are to the target
308  float land_proportion = current_loc.line_path_proportion(prev_WP_loc, loc);
309 
310  // now setup the glide slope for landing
311  set_target_altitude_proportion_fn(loc, 1.0f - land_proportion);
312 
313  // stay within the range of the start and end locations in altitude
314  constrain_target_altitude_location_fn(loc, prev_WP_loc);
315 }

317 int32_t AP_Landing::type_slope_get_target_airspeed_cm(void)
318 {
319  // we're landing, check for custom approach and
320  // pre-flare airspeeds. Also increase for head-winds
321 
322  const float land_airspeed = SpdHgt_Controller->get_land_airspeed();
323  int32_t target_airspeed_cm = aparm.airspeed_cruise_cm;
324 
325  switch (type_slope_stage) {
326  case SLOPE_STAGE_APPROACH:
327  if (land_airspeed >= 0) {
328  target_airspeed_cm = land_airspeed * 100;
329  }
330  break;
331 
332  case SLOPE_STAGE_PREFLARE:
333  case SLOPE_STAGE_FINAL:
334  if (pre_flare_airspeed > 0) {
335  // if we just preflared then continue using the pre-flare airspeed during final flare
336  target_airspeed_cm = pre_flare_airspeed * 100;
337  } else if (land_airspeed >= 0) {
338  target_airspeed_cm = land_airspeed * 100;
339  }
340  break;
341 
342  default:
343  break;
344  }
345 
346  // when landing, add half of head-wind.
347  const int32_t head_wind_compensation_cm = head_wind() * 0.5f * 100;
348 
349  // Do not lower it or exceed cruise speed
350  return constrain_int32(target_airspeed_cm + head_wind_compensation_cm, target_airspeed_cm, aparm.airspeed_cruise_cm);
351 }
