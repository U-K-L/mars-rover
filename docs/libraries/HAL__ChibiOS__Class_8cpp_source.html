136 /*
137  set the priority of the main APM task

139 void hal_chibios_set_priority(uint8_t priority)
140 {
141  chSysLock();
142 #if CH_CFG_USE_MUTEXES == TRUE
143  if ((daemon_task->prio == daemon_task->realprio) || (priority > daemon_task->prio)) {
144  daemon_task->prio = priority;
145  }
146  daemon_task->realprio = priority;
147 #endif
148  chSchRescheduleS();
149  chSysUnlock();
150 }

159 static void main_loop()
160 {
161  daemon_task = chThdGetSelfX();
162 
163  /*
164  switch to high priority for main loop
165  */
166  chThdSetPriority(APM_MAIN_PRIORITY);
167 
168 #ifdef HAL_I2C_CLEAR_BUS
169  // Clear all I2C Buses. This can be needed on some boards which
170  // can get a stuck I2C peripheral on boot
171  ChibiOS::I2CBus::clear_all();
172 #endif
173 
174 #if STM32_DMA_ADVANCED
175  ChibiOS::Shared_DMA::init();
176 #endif
177  peripheral_power_enable();
178 
179  hal.uartA->begin(115200);
180 
181 #ifdef HAL_SPI_CHECK_CLOCK_FREQ
182  // optional test of SPI clock frequencies
183  ChibiOS::SPIDevice::test_clock_freq();
184 #endif
185 
186  hal.uartB->begin(38400);
187  hal.uartC->begin(57600);
188  hal.analogin->init();
189  hal.scheduler->init();
190 
191  /*
192  run setup() at low priority to ensure CLI doesn't hang the
193  system, and to allow initial sensor read loops to run
194  */
195  hal_chibios_set_priority(APM_STARTUP_PRIORITY);
196 
197  if (stm32_was_watchdog_reset()) {
198  // load saved watchdog data
199  stm32_watchdog_load((uint32_t *)&utilInstance.persistent_data, (sizeof(utilInstance.persistent_data)+3)/4);
200  }
201 
202  schedulerInstance.hal_initialized();
203 
204  g_callbacks->setup();
205 
206 #ifdef IOMCU_FW
207  stm32_watchdog_init();
208 #elif !defined(HAL_BOOTLOADER_BUILD)
209  // setup watchdog to reset if main loop stops
210  if (AP_BoardConfig::watchdog_enabled()) {
211  stm32_watchdog_init();
212  }
213 
214  if (hal.util->was_watchdog_reset()) {
215  AP::internalerror().error(AP_InternalError::error_t::watchdog_reset);
216  const AP_HAL::Util::PersistentData &pd = hal.util->persistent_data;
217  AP::logger().WriteCritical("WDOG", "TimeUS,Task,IErr,IErrCnt,MavMsg,MavCmd,SemLine", "QbIIHHH",
218  AP_HAL::micros64(),
219  pd.scheduler_task,
220  pd.internal_errors,
221  pd.internal_error_count,
222  pd.last_mavlink_msgid,
223  pd.last_mavlink_cmd,
224  pd.semaphore_line);
225  }
226 #endif
227 
228  schedulerInstance.watchdog_pat();
229 
230  hal.scheduler->system_initialized();
231 
232  thread_running = true;
233  chRegSetThreadName(SKETCHNAME);
234 
235  /*
236  switch to high priority for main loop
237  */
238  chThdSetPriority(APM_MAIN_PRIORITY);
239 
240  while (true) {
241  g_callbacks->loop();
242 
243  /*
244  give up 50 microseconds of time if the INS loop hasn't
245  called delay_microseconds_boost(), to ensure low priority
246  drivers get a chance to run. Calling
247  delay_microseconds_boost() means we have already given up
248  time from the main loop, so we don't need to do it again
249  here
250  */
251 #ifndef HAL_DISABLE_LOOP_DELAY
252  if (!schedulerInstance.check_called_boost()) {
253  hal.scheduler->delay_microseconds(50);
254  }
255 #endif
256  schedulerInstance.watchdog_pat();
257  }
258  thread_running = false;
259 }
