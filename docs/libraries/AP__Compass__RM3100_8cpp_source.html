94 bool AP_Compass_RM3100::init()
95 {
96  if (!dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
97  return false;
98  }
99 
100  // high retries for init
101  dev->set_retries(10);
102 
103  // use default cycle count values as a whoami test
104  uint8_t ccx0;
105  uint8_t ccx1;
106  uint8_t ccy0;
107  uint8_t ccy1;
108  uint8_t ccz0;
109  uint8_t ccz1;
110  if (!dev->read_registers(RM3100_CCX1_REG, &ccx1, 1) ||
111  !dev->read_registers(RM3100_CCX0_REG, &ccx0, 1) ||
112  !dev->read_registers(RM3100_CCY1_REG, &ccy1, 1) ||
113  !dev->read_registers(RM3100_CCY0_REG, &ccy0, 1) ||
114  !dev->read_registers(RM3100_CCZ1_REG, &ccz1, 1) ||
115  !dev->read_registers(RM3100_CCZ0_REG, &ccz0, 1) ||
116  ccx1 != CCP1_DEFAULT || ccx0 != CCP0_DEFAULT ||
117  ccy1 != CCP1_DEFAULT || ccy0 != CCP0_DEFAULT ||
118  ccz1 != CCP1_DEFAULT || ccz0 != CCP0_DEFAULT) {
119  // couldn't read one of the cycle count registers or didn't recognize the default cycle count values
120  goto fail;
121  }
122 
123  dev->setup_checked_registers(8);
124 
125  dev->write_register(RM3100_TMRC_REG, TMRC, true); // cycle count z
126  dev->write_register(RM3100_CMM_REG, CMM, false); // CMM configuration
127  dev->write_register(RM3100_CCX1_REG, CCP1, true); // cycle count x
128  dev->write_register(RM3100_CCX0_REG, CCP0, true); // cycle count x
129  dev->write_register(RM3100_CCY1_REG, CCP1, true); // cycle count y
130  dev->write_register(RM3100_CCY0_REG, CCP0, true); // cycle count y
131  dev->write_register(RM3100_CCZ1_REG, CCP1, true); // cycle count z
132  dev->write_register(RM3100_CCZ0_REG, CCP0, true); // cycle count z
133 
134  _scaler = (1 / GAIN_CC200) * UTESLA_TO_MGAUSS; // has to be changed if using a different cycle count
135 
136  // lower retries for run
137  dev->set_retries(3);
138 
139  dev->get_semaphore()->give();
140 
141  /* register the compass instance in the frontend */
142  compass_instance = register_compass();
143 
144  printf("Found a RM3100 at address 0x%x as compass %u\n", dev->get_bus_address(), compass_instance);
145 
146  set_rotation(compass_instance, rotation);
147 
148  if (force_external) {
149  set_external(compass_instance, true);
150  }
151 
152  dev->set_device_type(DEVTYPE_RM3100);
153  set_dev_id(compass_instance, dev->get_bus_id());
154 
155  // call timer() at 80Hz
156  dev->register_periodic_callback(1000000U/80U,
157  FUNCTOR_BIND_MEMBER(&AP_Compass_RM3100::timer, void));
158 
159  return true;
160 
161 fail:
162  dev->get_semaphore()->give();
163  return false;
164 }

166 void AP_Compass_RM3100::timer()
167 {
168  struct PACKED {
169  uint8_t magx_2;
170  uint8_t magx_1;
171  uint8_t magx_0;
172  uint8_t magy_2;
173  uint8_t magy_1;
174  uint8_t magy_0;
175  uint8_t magz_2;
176  uint8_t magz_1;
177  uint8_t magz_0;
178  } data;
179  Vector3f field;
180 
181  int32_t magx = 0;
182  int32_t magy = 0;
183  int32_t magz = 0;
184 
185  // check data ready on 3 axis
186  uint8_t status;
187  if (!dev->read_registers(RM3100_STATUS_REG, (uint8_t *)&status, 1)) {
188  goto check_registers;
189  }
190 
191  if (!(status & 0x80)) {
192  // data not available yet
193  goto check_registers;
194  }
195 
196  if (!dev->read_registers(RM3100_MX2_REG, (uint8_t *)&data, sizeof(data))) {
197  goto check_registers;
198  }
199 
200  // the 24 bits of data for each axis are in 2s complement representation
201  // each byte is shifted to its position in a 24-bit unsigned integer and from 8 more bits to be left-aligned in a 32-bit integer
202  magx = ((uint32_t)data.magx_2 << 24) | ((uint32_t)data.magx_1 << 16) | ((uint32_t)data.magx_0 << 8);
203  magy = ((uint32_t)data.magy_2 << 24) | ((uint32_t)data.magy_1 << 16) | ((uint32_t)data.magy_0 << 8);
204  magz = ((uint32_t)data.magz_2 << 24) | ((uint32_t)data.magz_1 << 16) | ((uint32_t)data.magz_0 << 8);
205 
206  // right-shift signed integer back to get correct measurement value
207  magx >>= 8;
208  magy >>= 8;
209  magz >>= 8;
210 
211  // apply scaler and store in field vector
212  field(magx * _scaler, magy * _scaler, magz * _scaler);
213 
214  accumulate_sample(field, compass_instance);
215 
216 check_registers:
217  dev->check_next_register();
218 }
