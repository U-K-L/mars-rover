97 bool AP_ToneAlarm::init()
98 {
99  if (pNotify->buzzer_enabled() == false) {
100  return false;
101  }
102  if (!hal.util->toneAlarm_init()) {
103  return false;
104  }
105 
106  // set initial boot states. This prevents us issuing a arming
107  // warning in plane and rover on every boot
108  flags.armed = AP_Notify::flags.armed;
109  flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
110  flags.pre_arm_check = 1;
111  _cont_tone_playing = -1;
112  hal.scheduler->register_timer_process(FUNCTOR_BIND(this, &AP_ToneAlarm::_timer_task, void));
113  play_tone(AP_NOTIFY_TONE_STARTUP);
114  return true;
115 }

118 void AP_ToneAlarm::play_tone(const uint8_t tone_index)
119 {
120  uint32_t tnow_ms = AP_HAL::millis();
121  const Tone &tone_requested = _tones[tone_index];
122 
123  if (tone_requested.continuous) {
124  _cont_tone_playing = tone_index;
125  }
126 
127  _tone_playing = tone_index;
128  _tone_beginning_ms = tnow_ms;
129 
130  play_tune(tone_requested.str);
131 }

158 void AP_ToneAlarm::check_cont_tone()
159 {
160  uint32_t tnow_ms = AP_HAL::millis();
161  // if we are supposed to be playing a continuous tone,
162  // and it was interrupted, and the interrupting tone has timed out,
163  // resume the continuous tone
164 
165  if (_cont_tone_playing != -1 && _tone_playing != _cont_tone_playing && tnow_ms-_tone_beginning_ms > AP_NOTIFY_TONEALARM_MAX_TONE_LENGTH_MS) {
166  play_tone(_cont_tone_playing);
167  }
168 }

170 // update - updates led according to timed_updated. Should be called at 50Hz
171 void AP_ToneAlarm::update()
172 {
173  // exit if buzzer is not enabled
174  if (pNotify->buzzer_enabled() == false) {
175  return;
176  }
177 
178  check_cont_tone();
179 
180  if (AP_Notify::flags.powering_off) {
181  if (!flags.powering_off) {
182  play_tone(AP_NOTIFY_TONE_QUIET_SHUTDOWN);
183  }
184  flags.powering_off = AP_Notify::flags.powering_off;
185  return;
186  }
187 
188  if (AP_Notify::flags.compass_cal_running != flags.compass_cal_running) {
189  if (AP_Notify::flags.compass_cal_running) {
190  play_tone(AP_NOTIFY_TONE_QUIET_COMPASS_CALIBRATING_CTS);
191  play_tone(AP_NOTIFY_TONE_QUIET_POS_FEEDBACK);
192  } else {
193  if (_cont_tone_playing == AP_NOTIFY_TONE_QUIET_COMPASS_CALIBRATING_CTS) {
194  stop_cont_tone();
195  }
196  }
197  }
198  flags.compass_cal_running = AP_Notify::flags.compass_cal_running;
199 
200  if (AP_Notify::events.compass_cal_canceled) {
201  play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
202  return;
203  }
204 
205  if (AP_Notify::events.initiated_compass_cal) {
206  play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
207  return;
208  }
209 
210  if (AP_Notify::events.compass_cal_saved) {
211  play_tone(AP_NOTIFY_TONE_QUIET_READY_OR_FINISHED);
212  return;
213  }
214 
215  if (AP_Notify::events.compass_cal_failed) {
216  play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
217  return;
218  }
219 
220  // don't play other tones if compass cal is running
221  if (AP_Notify::flags.compass_cal_running) {
222  return;
223  }
224 
225  // notify the user when autotune or mission completes
226  if (AP_Notify::flags.armed && (AP_Notify::events.autotune_complete || AP_Notify::events.mission_complete)) {
227  play_tone(AP_NOTIFY_TONE_LOUD_READY_OR_FINISHED);
228  }
229 
230  //notify the user when autotune fails
231  if (AP_Notify::flags.armed && (AP_Notify::events.autotune_failed)) {
232  play_tone(AP_NOTIFY_TONE_LOUD_NEG_FEEDBACK);
233  }
234 
235  // notify the user when a waypoint completes
236  if (AP_Notify::events.waypoint_complete) {
237  play_tone(AP_NOTIFY_TONE_LOUD_WP_COMPLETE);
238  }
239 
240  // notify the user when their mode change was successful
241  if (AP_Notify::events.user_mode_change) {
242  if (AP_Notify::flags.armed) {
243  play_tone(AP_NOTIFY_TONE_LOUD_NEU_FEEDBACK);
244  } else {
245  play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
246  }
247  }
248 
249  // notify the user when their mode change failed
250  if (AP_Notify::events.user_mode_change_failed) {
251  if (AP_Notify::flags.armed) {
252  play_tone(AP_NOTIFY_TONE_LOUD_NEG_FEEDBACK);
253  } else {
254  play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
255  }
256  }
257 
258  // failsafe initiated mode change
259  if (AP_Notify::events.failsafe_mode_change) {
260  play_tone(AP_NOTIFY_TONE_LOUD_ATTENTION_NEEDED);
261  }
262 
263  // notify the user when arming fails
264  if (AP_Notify::events.arming_failed) {
265  play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
266  }
267 
268  // notify the user when RC contact is lost
269  if (flags.failsafe_radio != AP_Notify::flags.failsafe_radio) {
270  flags.failsafe_radio = AP_Notify::flags.failsafe_radio;
271  if (flags.failsafe_radio) {
272  // armed case handled by events.failsafe_mode_change
273  if (!AP_Notify::flags.armed) {
274  play_tone(AP_NOTIFY_TONE_QUIET_NEG_FEEDBACK);
275  }
276  } else {
277  if (AP_Notify::flags.armed) {
278  play_tone(AP_NOTIFY_TONE_LOUD_POS_FEEDBACK);
279  } else {
280  play_tone(AP_NOTIFY_TONE_QUIET_POS_FEEDBACK);
281  }
282  }
283  }
284 
285  // notify the user when pre_arm checks are passing
286  if (flags.pre_arm_check != AP_Notify::flags.pre_arm_check) {
287  flags.pre_arm_check = AP_Notify::flags.pre_arm_check;
288  if (flags.pre_arm_check) {
289  play_tone(AP_NOTIFY_TONE_QUIET_READY_OR_FINISHED);
290  _have_played_ready_tone = true;
291  } else {
292  // only play sad tone if we've ever played happy tone:
293  if (_have_played_ready_tone) {
294  play_tone(AP_NOTIFY_TONE_QUIET_NOT_READY_OR_NOT_FINISHED);
295  }
296  }
297  }
298 
299  // check if arming status has changed
300  if (flags.armed != AP_Notify::flags.armed) {
301  flags.armed = AP_Notify::flags.armed;
302  if (flags.armed) {
303  // arming tune
304  play_tone(AP_NOTIFY_TONE_QUIET_ARMING_WARNING);
305  } else {
306  // disarming tune
307  play_tone(AP_NOTIFY_TONE_QUIET_NEU_FEEDBACK);
308  if (!flags.leak_detected) {
309  stop_cont_tone();
310  }
311  }
312  }
313 
314  // check if battery status has changed
315  if (flags.failsafe_battery != AP_Notify::flags.failsafe_battery) {
316  flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
317  if (flags.failsafe_battery) {
318  // battery warning tune
319  play_tone(AP_NOTIFY_TONE_LOUD_BATTERY_ALERT_CTS);
320  }
321  }
322 
323  // check parachute release
324  if (flags.parachute_release != AP_Notify::flags.parachute_release) {
325  flags.parachute_release = AP_Notify::flags.parachute_release;
326  if (flags.parachute_release) {
327  // parachute release warning tune
328  play_tone(AP_NOTIFY_TONE_LOUD_ATTENTION_NEEDED);
329  }
330  }
331 
332  // lost vehicle tone
333  if (flags.vehicle_lost != AP_Notify::flags.vehicle_lost) {
334  flags.vehicle_lost = AP_Notify::flags.vehicle_lost;
335  if (flags.vehicle_lost) {
336  play_tone(AP_NOTIFY_TONE_LOUD_VEHICLE_LOST_CTS);
337  } else {
338  stop_cont_tone();
339  }
340  }
341 
342  // waiting to be thrown vehicle tone
343  if (flags.waiting_for_throw != AP_Notify::flags.waiting_for_throw) {
344  flags.waiting_for_throw = AP_Notify::flags.waiting_for_throw;
345  if (flags.waiting_for_throw) {
346  play_tone(AP_NOTIFY_TONE_WAITING_FOR_THROW);
347  } else {
348  stop_cont_tone();
349  }
350  }
351 
352  if (flags.leak_detected != AP_Notify::flags.leak_detected) {
353  flags.leak_detected = AP_Notify::flags.leak_detected;
354  if (flags.leak_detected) {
355  play_tone(AP_NOTIFY_TONE_LEAK_DETECTED);
356  } else {
357  stop_cont_tone();
358  }
359  }
360 
361  if (AP_Notify::events.tune_started) {
362  play_tone(AP_NOTIFY_TONE_TUNING_START);
363  AP_Notify::events.tune_started = 0;
364  }
365  if (AP_Notify::events.tune_next) {
366  // signify which parameter in the set is starting
367  play_tone(AP_NOTIFY_TONE_LOUD_1 + (AP_Notify::events.tune_next-1));
368  AP_Notify::events.tune_next = 0;
369  }
370  if (AP_Notify::events.tune_save) {
371  play_tone(AP_NOTIFY_TONE_TUNING_SAVE);
372  AP_Notify::events.tune_save = 0;
373  }
374  if (AP_Notify::events.tune_error) {
375  play_tone(AP_NOTIFY_TONE_TUNING_ERROR);
376  AP_Notify::events.tune_error = 0;
377  }
378 }

401  _tone_buf[sizeof(_tone_buf)-1] = 0;
402  _mml_player.play(_tone_buf);
