126 int sumd_decode(uint8_t byte, uint8_t *rssi, uint8_t *rx_count, uint16_t *channel_count, uint16_t *channels,
127  uint16_t max_chan_count)
128 {
129 
130  int ret = 1;
131 
132  switch (_decode_state) {
133  case SUMD_DECODE_STATE_UNSYNCED:
134  if (_debug) {
135  printf(" SUMD_DECODE_STATE_UNSYNCED \n") ;
136  }
137 
138  if (byte == SUMD_HEADER_ID) {
139  _rxpacket.header = byte;
140  _sumd = true;
141  _rxlen = 0;
142  _crc16 = 0x0000;
143  _crc8 = 0x00;
144  _crcOK = false;
145  _crc16 = crc_xmodem_update(_crc16, byte);
146  _crc8 = sumd_crc8(_crc8, byte);
147  _decode_state = SUMD_DECODE_STATE_GOT_HEADER;
148 
149  if (_debug) {
150  printf(" SUMD_DECODE_STATE_GOT_HEADER: %x \n", byte) ;
151  }
152 
153  } else {
154  ret = 3;
155  }
156 
157  break;
158 
159  case SUMD_DECODE_STATE_GOT_HEADER:
160  if (byte == SUMD_ID_SUMD || byte == SUMD_ID_SUMH) {
161  _rxpacket.status = byte;
162 
163  if (byte == SUMD_ID_SUMH) {
164  _sumd = false;
165  }
166 
167  if (_sumd) {
168  _crc16 = crc_xmodem_update(_crc16, byte);
169 
170  } else {
171  _crc8 = sumd_crc8(_crc8, byte);
172  }
173 
174  _decode_state = SUMD_DECODE_STATE_GOT_STATE;
175 
176  if (_debug) {
177  printf(" SUMD_DECODE_STATE_GOT_STATE: %x \n", byte) ;
178  }
179 
180  } else {
181  _decode_state = SUMD_DECODE_STATE_UNSYNCED;
182  }
183 
184  break;
185 
186  case SUMD_DECODE_STATE_GOT_STATE:
187  if (byte >= 2 && byte <= SUMD_MAX_CHANNELS) {
188  _rxpacket.length = byte;
189 
190  if (_sumd) {
191  _crc16 = crc_xmodem_update(_crc16, byte);
192 
193  } else {
194  _crc8 = sumd_crc8(_crc8, byte);
195  }
196 
197  _rxlen++;
198  _decode_state = SUMD_DECODE_STATE_GOT_LEN;
199 
200  if (_debug) {
201  printf(" SUMD_DECODE_STATE_GOT_LEN: %x (%d) \n", byte, byte) ;
202  }
203 
204  } else {
205  _decode_state = SUMD_DECODE_STATE_UNSYNCED;
206  }
207 
208  break;
209 
210  case SUMD_DECODE_STATE_GOT_LEN:
211  _rxpacket.sumd_data[_rxlen] = byte;
212 
213  if (_sumd) {
214  _crc16 = crc_xmodem_update(_crc16, byte);
215 
216  } else {
217  _crc8 = sumd_crc8(_crc8, byte);
218  }
219 
220  _rxlen++;
221 
222  if (_rxlen <= ((_rxpacket.length * 2))) {
223  if (_debug) {
224  printf(" SUMD_DECODE_STATE_GOT_DATA[%d]: %x\n", _rxlen - 2, byte) ;
225  }
226 
227  } else {
228  _decode_state = SUMD_DECODE_STATE_GOT_DATA;
229 
230  if (_debug) {
231  printf(" SUMD_DECODE_STATE_GOT_DATA -- finish --\n") ;
232  }
233 
234  }
235 
236  break;
237 
238  case SUMD_DECODE_STATE_GOT_DATA:
239  _rxpacket.crc16_high = byte;
240 
241  if (_debug) {
242  printf(" SUMD_DECODE_STATE_GOT_CRC16[1]: %x [%x]\n", byte, ((_crc16 >> 8) & 0xff)) ;
243  }
244 
245  if (_sumd) {
246  _decode_state = SUMD_DECODE_STATE_GOT_CRC;
247 
248  } else {
249  _decode_state = SUMD_DECODE_STATE_GOT_CRC16_BYTE_1;
250  }
251 
252  break;
253 
254  case SUMD_DECODE_STATE_GOT_CRC16_BYTE_1:
255  _rxpacket.crc16_low = byte;
256 
257  if (_debug) {
258  printf(" SUMD_DECODE_STATE_GOT_CRC16[2]: %x [%x]\n", byte, (_crc16 & 0xff)) ;
259  }
260 
261  _decode_state = SUMD_DECODE_STATE_GOT_CRC16_BYTE_2;
262 
263  break;
264 
265  case SUMD_DECODE_STATE_GOT_CRC16_BYTE_2:
266  _rxpacket.telemetry = byte;
267 
268  if (_debug) {
269  printf(" SUMD_DECODE_STATE_GOT_SUMH_TELEMETRY: %x\n", byte) ;
270  }
271 
272  _decode_state = SUMD_DECODE_STATE_GOT_CRC;
273 
274  break;
275 
276  case SUMD_DECODE_STATE_GOT_CRC:
277  if (_sumd) {
278  _rxpacket.crc16_low = byte;
279 
280  if (_debug) {
281  printf(" SUMD_DECODE_STATE_GOT_CRC[2]: %x [%x]\n\n", byte, (_crc16 & 0xff)) ;
282  }
283 
284  if (_crc16 == (uint16_t)(_rxpacket.crc16_high << 8) + _rxpacket.crc16_low) {
285  _crcOK = true;
286  }
287 
288  } else {
289  _rxpacket.crc8 = byte;
290 
291  if (_debug) {
292  printf(" SUMD_DECODE_STATE_GOT_CRC8_SUMH: %x [%x]\n\n", byte, _crc8) ;
293  }
294 
295  if (_crc8 == _rxpacket.crc8) {
296  _crcOK = true;
297  }
298  }
299 
300  if (_crcOK) {
301  if (_debug) {
302  printf(" CRC - OK \n") ;
303  }
304 
305  if (_sumd) {
306  if (_debug) {
307  printf(" Got valid SUMD Packet\n") ;
308  }
309 
310  } else {
311  if (_debug) {
312  printf(" Got valid SUMH Packet\n") ;
313  }
314 
315  }
316 
317  if (_debug) {
318  printf(" RXLEN: %d [Chans: %d] \n\n", _rxlen - 1, (_rxlen - 1) / 2) ;
319  }
320 
321  ret = 0;
322  unsigned i;
323  uint8_t _cnt = *rx_count + 1;
324  *rx_count = _cnt;
325 
326  *rssi = 100;
327 
328  /* received Channels */
329  if ((uint16_t)_rxpacket.length > max_chan_count) {
330  _rxpacket.length = (uint8_t) max_chan_count;
331  }
332 
333  *channel_count = (uint16_t)_rxpacket.length;
334 
335  /* decode the actual packet */
336  /* reorder first 4 channels */
337 
338  /* ch1 = roll -> sumd = ch2 */
339  channels[0] = (uint16_t)((_rxpacket.sumd_data[1 * 2 + 1] << 8) | _rxpacket.sumd_data[1 * 2 + 2]) >> 3;
340  /* ch2 = pitch -> sumd = ch2 */
341  channels[1] = (uint16_t)((_rxpacket.sumd_data[2 * 2 + 1] << 8) | _rxpacket.sumd_data[2 * 2 + 2]) >> 3;
342  /* ch3 = throttle -> sumd = ch2 */
343  channels[2] = (uint16_t)((_rxpacket.sumd_data[0 * 2 + 1] << 8) | _rxpacket.sumd_data[0 * 2 + 2]) >> 3;
344  /* ch4 = yaw -> sumd = ch2 */
345  channels[3] = (uint16_t)((_rxpacket.sumd_data[3 * 2 + 1] << 8) | _rxpacket.sumd_data[3 * 2 + 2]) >> 3;
346 
347  /* we start at channel 5(index 4) */
348  unsigned chan_index = 4;
349 
350  for (i = 4; i < _rxpacket.length; i++) {
351  if (_debug) {
352  printf("ch[%d] : %x %x [ %x %d ]\n", i + 1, _rxpacket.sumd_data[i * 2 + 1], _rxpacket.sumd_data[i * 2 + 2],
353  ((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3,
354  ((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3);
355  }
356 
357  channels[chan_index] = (uint16_t)((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3;
358  /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
359  //channels[chan_index] = (uint16_t)(channels[chan_index] * SUMD_SCALE_FACTOR + .5f) + SUMD_SCALE_OFFSET;
360 
361  chan_index++;
362  }
363 
364  } else {
365  /* decoding failed */
366  ret = 4;
367 
368  if (_debug) {
369  printf(" CRC - fail \n") ;
370  }
371 
372  }
373 
374  _decode_state = SUMD_DECODE_STATE_UNSYNCED;
375  break;
376  }
377 
378  return ret;
