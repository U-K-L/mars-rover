23 // return the rotation matrix equivalent for this quaternion
24 void Quaternion::rotation_matrix(Matrix3f &m) const
25 {
26  const float q3q3 = q3 * q3;
27  const float q3q4 = q3 * q4;
28  const float q2q2 = q2 * q2;
29  const float q2q3 = q2 * q3;
30  const float q2q4 = q2 * q4;
31  const float q1q2 = q1 * q2;
32  const float q1q3 = q1 * q3;
33  const float q1q4 = q1 * q4;
34  const float q4q4 = q4 * q4;
35 
36  m.a.x = 1.0f-2.0f*(q3q3 + q4q4);
37  m.a.y = 2.0f*(q2q3 - q1q4);
38  m.a.z = 2.0f*(q2q4 + q1q3);
39  m.b.x = 2.0f*(q2q3 + q1q4);
40  m.b.y = 1.0f-2.0f*(q2q2 + q4q4);
41  m.b.z = 2.0f*(q3q4 - q1q2);
42  m.c.x = 2.0f*(q2q4 - q1q3);
43  m.c.y = 2.0f*(q3q4 + q1q2);
44  m.c.z = 1.0f-2.0f*(q2q2 + q3q3);
45 }

47 // return the rotation matrix equivalent for this quaternion after normalization
48 void Quaternion::rotation_matrix_norm(Matrix3f &m) const
49 {
50  const float q1q1 = q1 * q1;
51  const float q1q2 = q1 * q2;
52  const float q1q3 = q1 * q3;
53  const float q1q4 = q1 * q4;
54  const float q2q2 = q2 * q2;
55  const float q2q3 = q2 * q3;
56  const float q2q4 = q2 * q4;
57  const float q3q3 = q3 * q3;
58  const float q3q4 = q3 * q4;
59  const float q4q4 = q4 * q4;
60  const float invs = 1.0f / (q1q1 + q2q2 + q3q3 + q4q4);
61 
62  m.a.x = ( q2q2 - q3q3 - q4q4 + q1q1)*invs;
63  m.a.y = 2.0f*(q2q3 - q1q4)*invs;
64  m.a.z = 2.0f*(q2q4 + q1q3)*invs;
65  m.b.x = 2.0f*(q2q3 + q1q4)*invs;
66  m.b.y = (-q2q2 + q3q3 - q4q4 + q1q1)*invs;
67  m.b.z = 2.0f*(q3q4 - q1q2)*invs;
68  m.c.x = 2.0f*(q2q4 - q1q3)*invs;
69  m.c.y = 2.0f*(q3q4 + q1q2)*invs;
70  m.c.z = (-q2q2 - q3q3 + q4q4 + q1q1)*invs;
71 }

75 // http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/index.htm
76 void Quaternion::from_rotation_matrix(const Matrix3f &m)
77 {
78  const float &m00 = m.a.x;
79  const float &m11 = m.b.y;
80  const float &m22 = m.c.z;
81  const float &m10 = m.b.x;
82  const float &m01 = m.a.y;
83  const float &m20 = m.c.x;
84  const float &m02 = m.a.z;
85  const float &m21 = m.c.y;
86  const float &m12 = m.b.z;
87  float &qw = q1;
88  float &qx = q2;
89  float &qy = q3;
90  float &qz = q4;
91 
92  const float tr = m00 + m11 + m22;
93 
94  if (tr > 0) {
95  const float S = sqrtf(tr+1) * 2;
96  qw = 0.25f * S;
97  qx = (m21 - m12) / S;
98  qy = (m02 - m20) / S;
99  qz = (m10 - m01) / S;
100  } else if ((m00 > m11) && (m00 > m22)) {
101  const float S = sqrtf(1.0f + m00 - m11 - m22) * 2.0f;
102  qw = (m21 - m12) / S;
103  qx = 0.25f * S;
104  qy = (m01 + m10) / S;
105  qz = (m02 + m20) / S;
106  } else if (m11 > m22) {
107  const float S = sqrtf(1.0f + m11 - m00 - m22) * 2.0f;
108  qw = (m02 - m20) / S;
109  qx = (m01 + m10) / S;
110  qy = 0.25f * S;
111  qz = (m12 + m21) / S;
112  } else {
113  const float S = sqrtf(1.0f + m22 - m00 - m11) * 2.0f;
114  qw = (m10 - m01) / S;
115  qx = (m02 + m20) / S;
116  qy = (m12 + m21) / S;
117  qz = 0.25f * S;
118  }
119 }

130 void Quaternion::from_euler(float roll, float pitch, float yaw)
131 {
132  const float cr2 = cosf(roll*0.5f);
133  const float cp2 = cosf(pitch*0.5f);
134  const float cy2 = cosf(yaw*0.5f);
135  const float sr2 = sinf(roll*0.5f);
136  const float sp2 = sinf(pitch*0.5f);
137  const float sy2 = sinf(yaw*0.5f);
138 
139  q1 = cr2*cp2*cy2 + sr2*sp2*sy2;
140  q2 = sr2*cp2*cy2 - cr2*sp2*sy2;
141  q3 = cr2*sp2*cy2 + sr2*cp2*sy2;
142  q4 = cr2*cp2*sy2 - sr2*sp2*cy2;
143 }

223 void Quaternion::rotate_fast(const Vector3f &v)
224 {
225  const float theta = v.length();
226  if (is_zero(theta)) {
227  return;
228  }
229  const float t2 = theta/2.0f;
230  const float sqt2 = sq(t2);
231  float st2 = t2-sqt2*t2/6.0f;
232  st2 /= theta;
233 
234  //"rotation quaternion"
235  const float w2 = 1.0f-(sqt2/2.0f)+sq(sqt2)/24.0f;
236  const float x2 = v.x * st2;
237  const float y2 = v.y * st2;
238  const float z2 = v.z * st2;
239 
240  //copy our quaternion
241  const float w1 = q1;
242  const float x1 = q2;
243  const float y1 = q3;
244  const float z1 = q4;
245 
246  //do the multiply into our quaternion
247  q1 = w1*w2 - x1*x2 - y1*y2 - z1*z2;
248  q2 = w1*x2 + x1*w2 + y1*z2 - z1*y2;
249  q3 = w1*y2 - x1*z2 + y1*w2 + z1*x2;
250  q4 = w1*z2 + x1*y2 - y1*x2 + z1*w2;
251 }
