68 void print_latlon(AP_HAL::BetterStream *s, int32_t lat_or_lon);
69 void print_latlon(AP_HAL::BetterStream *s, int32_t lat_or_lon)
70 {
71  int32_t dec_portion, frac_portion;
72  int32_t abs_lat_or_lon = labs(lat_or_lon);
73 
74  // extract decimal portion (special handling of negative numbers to ensure we round towards zero)
75  dec_portion = abs_lat_or_lon / 10000000UL;
76 
77  // extract fractional portion
78  frac_portion = abs_lat_or_lon - dec_portion*10000000UL;
79 
80  // print output including the minus sign
81  if( lat_or_lon < 0 ) {
82  s->printf("-");
83  }
84  s->printf("%ld.%07ld",(long)dec_portion,(long)frac_portion);
85 }

87 // loop
88 void loop()
89 {
90  static uint32_t last_msg_ms;
91 
92  // Update GPS state based on possible bytes received from the module.
93  gps.update();
94 
95  // If new GPS data is received, output it's contents to the console
96  // Here we rely on the time of the message in GPS class and the time of last message
97  // saved in static variable last_msg_ms. When new message is received, the time
98  // in GPS class will be updated.
99  if (last_msg_ms != gps.last_message_time_ms()) {
100  // Reset the time of message
101  last_msg_ms = gps.last_message_time_ms();
102 
103  // Acquire location
104  const Location &loc = gps.location();
105 
106  // Print the contents of message
107  hal.console->printf("Lat: ");
108  print_latlon(hal.console, loc.lat);
109  hal.console->printf(" Lon: ");
110  print_latlon(hal.console, loc.lng);
111  hal.console->printf(" Alt: %.2fm GSP: %.2fm/s CoG: %d SAT: %d TIM: %u/%lu STATUS: %u\n",
112  (double)(loc.alt * 0.01f),
113  (double)gps.ground_speed(),
114  (int)gps.ground_course_cd() / 100,
115  gps.num_sats(),
116  gps.time_week(),
117  (long unsigned int)gps.time_week_ms(),
118  gps.status());
119  }
120 
121  // Delay for 10 mS will give us 100 Hz invocation rate
122  hal.scheduler->delay(10);
123 }
