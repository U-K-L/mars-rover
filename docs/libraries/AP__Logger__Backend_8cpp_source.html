115  // permit any message to go in as its timestamp will be before
116  // any we wrote in. Time going backwards annoys log readers.
117 
118  // sorry! currently busy writing out startup messages...
119  return false;

176 bool AP_Logger_Backend::Write(const uint8_t msg_type, va_list arg_list, bool is_critical)
177 {
178  // stack-allocate a buffer so we can WriteBlock(); this could be
179  // 255 bytes! If we were willing to lose the WriteBlock
180  // abstraction we could do WriteBytes() here instead?
181  const char *fmt = nullptr;
182  uint8_t msg_len;
183  AP_Logger::log_write_fmt *f;
184  for (f = _front.log_write_fmts; f; f=f->next) {
185  if (f->msg_type == msg_type) {
186  fmt = f->fmt;
187  msg_len = f->msg_len;
188  break;
189  }
190  }
191  if (fmt == nullptr) {
192  AP::internalerror().error(AP_InternalError::error_t::logger_logwrite_missingfmt);
193  return false;
194  }
195  if (bufferspace_available() < msg_len) {
196  return false;
197  }
198  uint8_t buffer[msg_len];
199  uint8_t offset = 0;
200  buffer[offset++] = HEAD_BYTE1;
201  buffer[offset++] = HEAD_BYTE2;
202  buffer[offset++] = msg_type;
203  for (uint8_t i=0; i<strlen(fmt); i++) {
204  uint8_t charlen = 0;
205  switch(fmt[i]) {
206  case 'b': {
207  int8_t tmp = va_arg(arg_list, int);
208  memcpy(&buffer[offset], &tmp, sizeof(int8_t));
209  offset += sizeof(int8_t);
210  break;
211  }
212  case 'h':
213  case 'c': {
214  int16_t tmp = va_arg(arg_list, int);
215  memcpy(&buffer[offset], &tmp, sizeof(int16_t));
216  offset += sizeof(int16_t);
217  break;
218  }
219  case 'd': {
220  double tmp = va_arg(arg_list, double);
221  memcpy(&buffer[offset], &tmp, sizeof(double));
222  offset += sizeof(double);
223  break;
224  }
225  case 'i':
226  case 'L':
227  case 'e': {
228  int32_t tmp = va_arg(arg_list, int);
229  memcpy(&buffer[offset], &tmp, sizeof(int32_t));
230  offset += sizeof(int32_t);
231  break;
232  }
233  case 'f': {
234  float tmp = va_arg(arg_list, double);
235  memcpy(&buffer[offset], &tmp, sizeof(float));
236  offset += sizeof(float);
237  break;
238  }
239  case 'n':
240  charlen = 4;
241  break;
242  case 'M':
243  case 'B': {
244  uint8_t tmp = va_arg(arg_list, int);
245  memcpy(&buffer[offset], &tmp, sizeof(uint8_t));
246  offset += sizeof(uint8_t);
247  break;
248  }
249  case 'H':
250  case 'C': {
251  uint16_t tmp = va_arg(arg_list, int);
252  memcpy(&buffer[offset], &tmp, sizeof(uint16_t));
253  offset += sizeof(uint16_t);
254  break;
255  }
256  case 'I':
257  case 'E': {
258  uint32_t tmp = va_arg(arg_list, uint32_t);
259  memcpy(&buffer[offset], &tmp, sizeof(uint32_t));
260  offset += sizeof(uint32_t);
261  break;
262  }
263  case 'N':
264  charlen = 16;
265  break;
266  case 'Z':
267  charlen = 64;
268  break;
269  case 'q': {
270  int64_t tmp = va_arg(arg_list, int64_t);
271  memcpy(&buffer[offset], &tmp, sizeof(int64_t));
272  offset += sizeof(int64_t);
273  break;
274  }
275  case 'Q': {
276  uint64_t tmp = va_arg(arg_list, uint64_t);
277  memcpy(&buffer[offset], &tmp, sizeof(uint64_t));
278  offset += sizeof(uint64_t);
279  break;
280  }
281  }
282  if (charlen != 0) {
283  char *tmp = va_arg(arg_list, char*);
284  memcpy(&buffer[offset], tmp, charlen);
285  offset += charlen;
286  }
287  }
288 
289  return WritePrioritisedBlock(buffer, msg_len, is_critical);
290 }

310 void AP_Logger_Backend::validate_WritePrioritisedBlock(const void *pBuffer,
311  uint16_t size)
312 {
313  // just check the first few packets to avoid too much overhead
314  // (finding the structures is expensive)
315  static uint16_t count = 0;
316  if (count > 65534) {
317  return;
318  }
319  count++;
320 
321  // we assume here that we ever WritePrioritisedBlock for a single
322  // message. If this assumption becomes false we can't do these
323  // checks.
324  if (size < 3) {
325  AP_HAL::panic("Short prioritised block");
326  }
327  if (((uint8_t*)pBuffer)[0] != HEAD_BYTE1 ||
328  ((uint8_t*)pBuffer)[1] != HEAD_BYTE2) {
329  AP_HAL::panic("Not passed a message");
330  }
331  const uint8_t type = ((uint8_t*)pBuffer)[2];
332  uint8_t type_len;
333  const struct LogStructure *s = _front.structure_for_msg_type(type);
334  if (s == nullptr) {
335  const struct AP_Logger::log_write_fmt *t = _front.log_write_fmt_for_msg_type(type);
336  if (t == nullptr) {
337  AP_HAL::panic("No structure for msg_type=%u", type);
338  }
339  type_len = t->msg_len;
340  } else {
341  type_len = s->msg_len;
342  }
343  if (type_len != size) {
344  char name[5] = {}; // get a null-terminated string
345  memcpy(name, s->name, 4);
346  AP_HAL::panic("Size mismatch for %u (%s) (expected=%u got=%u)\n",
347  type, name, type_len, size);
348  }
349 }

369 bool AP_Logger_Backend::ShouldLog(bool is_critical)
370 {
371  if (!_front.WritesEnabled()) {

377

380  // only the main thread may write startup messages out
381  return false;
382  }
383

392  return false;
