277 int16_t AP_Logger_Block::get_log_data(uint16_t log_num, uint16_t page, uint32_t offset, uint16_t len, uint8_t *data)
278 {
279  WITH_SEMAPHORE(sem);
280  if (offset == 0) {
281  uint8_t header[3];
282  if (get_log_data_raw(log_num, page, 0, 3, header) == -1) {
283  return -1;
284  }
285  adding_fmt_headers = (header[0] != HEAD_BYTE1 || header[1] != HEAD_BYTE2 || header[2] != LOG_FORMAT_MSG);
286  }
287  uint16_t ret = 0;
288 
289  if (adding_fmt_headers) {
290  // the log doesn't start with a FMT message, we need to add
291  // them
292  const uint16_t fmt_header_size = num_types() * sizeof(struct log_Format);
293  while (offset < fmt_header_size && len > 0) {
294  struct log_Format pkt;
295  uint8_t t = offset / sizeof(pkt);
296  uint8_t ofs = offset % sizeof(pkt);
297  Fill_Format(structure(t), pkt);
298  uint8_t n = sizeof(pkt) - ofs;
299  if (n > len) {
300  n = len;
301  }
302  memcpy(data, ofs + (uint8_t *)&pkt, n);
303  data += n;
304  offset += n;
305  len -= n;
306  ret += n;
307  }
308  offset -= fmt_header_size;
309  }
310 
311  if (len > 0) {
312  const int16_t bytes = get_log_data_raw(log_num, page, offset, len, data);
313  if (bytes == -1) {
314  return ret == 0 ? -1 : ret;
315  }
316  ret += bytes;
317  }
318 
319  return ret;
320 }

461 // This function finds the last page of the last file
462 uint32_t AP_Logger_Block::find_last_page(void)
463 {
464  uint32_t look;
465  uint32_t bottom = 1;
466  uint32_t top = df_NumPages;
467  uint32_t look_hash;
468  uint32_t bottom_hash;
469  uint32_t top_hash;
470 
471  WITH_SEMAPHORE(sem);
472 
473  StartRead(bottom);
474  bottom_hash = ((int32_t)GetFileNumber()<<16) | df_FilePage;
475 
476  while (top-bottom > 1) {
477  look = (top+bottom)/2;
478  StartRead(look);
479  look_hash = (int32_t)GetFileNumber()<<16 | df_FilePage;
480  if (look_hash >= 0xFFFF0000) {
481  look_hash = 0;
482  }
483 
484  if (look_hash < bottom_hash) {
485  // move down
486  top = look;
487  } else {
488  // move up
489  bottom = look;
490  bottom_hash = look_hash;
491  }
492  }
493 
494  StartRead(top);
495  top_hash = ((int32_t)GetFileNumber()<<16) | df_FilePage;
496  if (top_hash >= 0xFFFF0000) {
497  top_hash = 0;
498  }
499  if (top_hash > bottom_hash) {
500  return top;
501  }
502 
503  return bottom;
504 }
