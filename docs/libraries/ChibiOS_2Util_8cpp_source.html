89 void *Util::heap_realloc(void *heap, void *ptr, size_t new_size)
90 {
91  if (heap == nullptr) {
92  return nullptr;
93  }
94  if (new_size == 0) {

96  chHeapFree(ptr);

102  }
103  void *new_mem = chHeapAlloc((memory_heap_t *)heap, new_size);
104  if (new_mem != nullptr) {
105  memcpy(new_mem, ptr, chHeapGetSize(ptr) > new_size ? new_size : chHeapGetSize(ptr));
106  chHeapFree(ptr);

126 void Util::set_imu_temp(float current)
127 {
128 #if HAL_HAVE_IMU_HEATER
129  if (!heater.target || *heater.target == -1) {
130  return;
131  }
132 
133  // average over temperatures to remove noise
134  heater.count++;
135  heater.sum += current;
136 
137  // update once a second
138  uint32_t now = AP_HAL::millis();
139  if (now - heater.last_update_ms < 1000) {
140 #if defined(HAL_HEATER_GPIO_PIN)
141  // output as duty cycle to local pin
142  hal.gpio->write(HAL_HEATER_GPIO_PIN, heater.duty_counter < heater.output);
143  heater.duty_counter = (heater.duty_counter+1) % 100;
144 #endif
145  return;
146  }
147  heater.last_update_ms = now;
148 
149  current = heater.sum / heater.count;
150  heater.sum = 0;
151  heater.count = 0;
152 
153  // experimentally tweaked for Pixhawk2
154  const float kI = 0.3f;
155  const float kP = 200.0f;
156  float target = (float)(*heater.target);
157 
158  // limit to 65 degrees to prevent damage
159  target = constrain_float(target, 0, 65);
160 
161  float err = target - current;
162 
163  heater.integrator += kI * err;
164  heater.integrator = constrain_float(heater.integrator, 0, 70);
165 
166  heater.output = constrain_float(kP * err + heater.integrator, 0, 100);
167 
168  //hal.console->printf("integrator %.1f out=%.1f temp=%.2f err=%.2f\n", heater.integrator, heater.output, current, err);
169 
170 #if HAL_WITH_IO_MCU
171  if (AP_BoardConfig::io_enabled()) {
172  // tell IOMCU to setup heater
173  iomcu.set_heater_duty_cycle(heater.output);
174  }
175 #endif
176 #endif // HAL_HAVE_IMU_HEATER
177 }

228 bool Util::flash_bootloader()
229 {
230  uint32_t fw_size;
231  const char *fw_name = "bootloader.bin";
232 
233  EXPECT_DELAY_MS(11000);
234 
235  uint8_t *fw = AP_ROMFS::find_decompress(fw_name, fw_size);
236  if (!fw) {
237  hal.console->printf("failed to find %s\n", fw_name);
238  return false;
239  }
240 
241  const uint32_t addr = hal.flash->getpageaddr(0);
242  if (!memcmp(fw, (const void*)addr, fw_size)) {
243  hal.console->printf("Bootloader up-to-date\n");
244  free(fw);
245  return true;
246  }
247 
248  hal.console->printf("Erasing\n");
249  if (!hal.flash->erasepage(0)) {
250  hal.console->printf("Erase failed\n");
251  free(fw);
252  return false;
253  }
254  hal.console->printf("Flashing %s @%08x\n", fw_name, (unsigned int)addr);
255  const uint8_t max_attempts = 10;
256  for (uint8_t i=0; i<max_attempts; i++) {
257  bool ok = hal.flash->write(addr, fw, fw_size);
258  if (!ok) {
259  hal.console->printf("Flash failed! (attempt=%u/%u)\n",
260  i+1,
261  max_attempts);
262  hal.scheduler->delay(1000);
263  continue;
264  }
265  hal.console->printf("Flash OK\n");
266  free(fw);
267  return true;
268  }
269 
270  hal.console->printf("Flash failed after %u attempts\n", max_attempts);
271  free(fw);
272  return false;
273 }
274 #endif //#ifndef HAL_NO_FLASH_SUPPORT
