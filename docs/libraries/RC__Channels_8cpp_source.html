132 // support for auxillary switches:
133 // read_aux_switches - checks aux switch positions and invokes configured actions
134 void RC_Channels::read_aux_all()
135 {
136  if (!has_valid_input()) {
137  // exit immediately when no RC input
138  return;
139  }
140  bool need_log = false;
141 
142  for (uint8_t i=0; i<NUM_RC_CHANNELS; i++) {
143  RC_Channel *c = channel(i);
144  if (c == nullptr) {
145  continue;
146  }
147  need_log |= c->read_aux();
148  }
149  if (need_log) {
150  // guarantee that we log when a switch changes
151  AP::logger().Write_RCIN();
152  }
153 }
