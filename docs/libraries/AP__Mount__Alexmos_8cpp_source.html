34  control_axis(_state._neutral_angles.get(), true);
35  break;

38  case MAV_MOUNT_MODE_MAVLINK_TARGETING:

40  control_axis(_angle_ef_target_rad, false);
41  break;
42 
43  // RC radio manual angle control, but with stabilization from the AHRS
44  case MAV_MOUNT_MODE_RC_TARGETING:

168 {

174  _port->write( cmd ); // write command id

177

181  }

222 void AP_Mount_Alexmos::read_incoming()
223 {
224  uint8_t data;
225  int16_t numc;
226 
227  numc = _port->available();
228 
229  if (numc < 0 ){
230  return;
231  }
232 
233  for (int16_t i = 0; i < numc; i++) { // Process bytes received
234  data = _port->read();
235  switch (_step) {
236  case 0:
237  if ( '>' == data) {
238  _step = 1;
239  _checksum = 0; //reset checksum accumulator
240  _last_command_confirmed = false;
241  }
242  break;
243 
244  case 1: // command ID
245  _checksum = data;
246  _command_id = data;
247  _step++;
248  break;
249 
250  case 2: // Size of the body of the message
251  _checksum += data;
252  _payload_length = data;
253  _step++;
254  break;
255 
256  case 3: // checksum of the header
257  if (_checksum != data ) {
258  _step = 0;
259  _checksum = 0;
260  // checksum error
261  break;
262  }
263  _step++;
264  _checksum = 0;
265  _payload_counter = 0; // prepare to receive payload
266  break;
267 
268  case 4: // parsing body
269  _checksum += data;
270  if (_payload_counter < sizeof(_buffer)) {
271  _buffer[_payload_counter] = data;
272  }
273  if (++_payload_counter == _payload_length)
274  _step++;
275  break;
276 
277  case 5:// body checksum
278  _step = 0;
279  if (_checksum != data) {
280  break;
281  }
282  parse_body();
283  }
284  }
