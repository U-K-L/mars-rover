42 /*
43 * This library is used for the Safe Return-to-Launch feature. The vehicle's
44 * position (aka "bread crumbs") are stored into an array in memory at
45 * regular intervals. After a certain number of bread crumbs have been
46 * stored and space within the array is low, clean-up algorithms are run to
47 * reduce the total number of points. When Safe-RTL is initiated by the
48 * vehicle code, a more thorough cleanup runs and the resulting path is fed
49 * into navigation controller to return the vehicle to home.
50 *
51 * The cleanup consists of two parts, pruning and simplification:
52 *
53 * 1. Pruning calculates the closest distance between two line segments formed
54 * by two pairs of sequential points, and then cuts out anything between two
55 * points when their line segments get close. This algorithm will never
56 * compare two consecutive line segments. Obviously the segments (p1,p2) and
57 * (p2,p3) will get very close (they touch), but there would be nothing to
58 * trim between them.
59 *
60 * 2. Simplification uses the Ramer-Douglas-Peucker algorithm. See Wikipedia
61 * for a more complete description.
62 *
63 * The simplification and pruning algorithms run in the background and do not
64 * alter the path in memory. Two definitions, SMARTRTL_SIMPLIFY_TIME_US and
65 * SMARTRTL_PRUNING_LOOP_TIME_US are used to limit how long each algorithm will
66 * be run before they save their state and return.
67 *
68 * Both algorithms are "anytime algorithms" meaning they can be interrupted
69 * before they complete which is helpful when memory is filling up and we just
70 * need to quickly identify a handful of points which can be deleted.
71 *
72 * Once the algorithms have completed the simplify.complete and
73 * prune.complete flags are set to true. The "thorough cleanup" procedure,
74 * which is run as the vehicle initiates the SmartRTL flight mode, waits for
75 * these flags to become true. This can force the vehicle to pause for a few
76 * seconds before initiating the return journey.
77 */

86 // initialise safe rtl including setting up background processes
87 void AP_SmartRTL::init()
88 {
89  // protect against repeated call to init
90  if (_path != nullptr) {
91  return;
92  }
93 
94  // constrain the path length, in case the user decided to make the path unreasonably long.
95  _points_max = constrain_int16(_points_max, 0, SMARTRTL_POINTS_MAX);
96 
97  // check if user has disabled SmartRTL
98  if (_points_max == 0 || !is_positive(_accuracy)) {
99  return;
100  }
101 
102  // allocate arrays
103  _path = (Vector3f*)calloc(_points_max, sizeof(Vector3f));
104 
105  _prune.loops_max = _points_max * SMARTRTL_PRUNING_LOOP_BUFFER_LEN_MULT;
106  _prune.loops = (prune_loop_t*)calloc(_prune.loops_max, sizeof(prune_loop_t));
107 
108  _simplify.stack_max = _points_max * SMARTRTL_SIMPLIFY_STACK_LEN_MULT;
109  _simplify.stack = (simplify_start_finish_t*)calloc(_simplify.stack_max, sizeof(simplify_start_finish_t));
110 
111  // check if memory allocation failed
112  if (_path == nullptr || _prune.loops == nullptr || _simplify.stack == nullptr) {
113  log_action(SRTL_DEACTIVATED_INIT_FAILED);
114  gcs().send_text(MAV_SEVERITY_WARNING, "SmartRTL deactivated: init failed");
115  free(_path);
116  free(_prune.loops);
117  free(_simplify.stack);
118  return;
119  }
120 
121  _path_points_max = _points_max;
122 
123  // when running the example sketch, we want the cleanup tasks to run when we tell them to, not in the background (so that they can be timed.)
124  if (!_example_mode){
125  // register background cleanup to run in IO thread
126  hal.scheduler->register_io_process(FUNCTOR_BIND_MEMBER(&AP_SmartRTL::run_background_cleanup, void));
127  }
128 }

222 {

284 // add point to end of path (if necessary), returns true on success
285 bool AP_SmartRTL::add_point(const Vector3f& point)
286 {
287  // get semaphore
288  if (!_path_sem.take_nonblocking()) {
289  log_action(SRTL_ADD_FAILED_NO_SEMAPHORE, point);
290  return false;
291  }
292 
293  // check if we have traveled far enough
294  if (_path_points_count > 0) {
295  const Vector3f& last_pos = _path[_path_points_count-1];
296  if (last_pos.distance_squared(point) < sq(_accuracy.get())) {
297  _path_sem.give();
298  return true;
299  }
300  }
301 
302  // check we have space in the path
303  if (_path_points_count >= _path_points_max) {
304  _path_sem.give();
305  log_action(SRTL_ADD_FAILED_PATH_FULL, point);
306  return false;
307  }
308 
309  // add point to path
310  _path[_path_points_count++] = point;
311  log_action(SRTL_POINT_ADD, point);
312 
313  _path_sem.give();
314  return true;
315 }

317 // run background cleanup - should be run regularly from the IO thread
318 void AP_SmartRTL::run_background_cleanup()
319 {
320  if (!_active) {
321  return;
322  }
323 
324  // get semaphore
325  if (!_path_sem.take_nonblocking()) {
326  return;
327  }
328  // local copy of _path_points_count and _path_points_completed_limit
329  const uint16_t path_points_count = _path_points_count;
330  const uint16_t path_points_completed_limit = _path_points_completed_limit;
331  _path_points_completed_limit = SMARTRTL_POINTS_MAX;
332  _path_sem.give();
333 
334  // check if thorough cleanup is required
335  if (_thorough_clean_request_ms > 0) {
336  // check if we have already completed the request
337  if (_thorough_clean_complete_ms != _thorough_clean_request_ms) {
338  if (thorough_cleanup(path_points_count, _thorough_clean_type)) {
339  // record completion
340  _thorough_clean_complete_ms = _thorough_clean_request_ms;
341  }
342  }
343  // we do not perform any further detection or cleanup until the requester acknowledges
344  // they have what they need by setting _thorough_clean_request_ms back to zero
345  return;
346  }
347 
348  // ensure clean complete time is zero
349  _thorough_clean_complete_ms = 0;
350 
351  // perform routine cleanup which removes 10 to 50 points if possible
352  routine_cleanup(path_points_count, path_points_completed_limit);
353 }
354 
355 // routine cleanup is called regularly from run_background_cleanup
356 // simplifies the path after SMARTRTL_CLEANUP_POINT_TRIGGER points (50 points) have been added OR
357 // SMARTRTL_CLEANUP_POINT_MIN (10 points) have been added and the path has less than SMARTRTL_CLEANUP_START_MARGIN spaces (10 spaces) remaining
358 // prunes the path if the path has less than SMARTRTL_CLEANUP_START_MARGIN spaces (10 spaces) remaining
359 void AP_SmartRTL::routine_cleanup(uint16_t path_points_count, uint16_t path_points_completed_limit)
360 {
361  // if simplify is running, let it run to completion
362  if (!_simplify.complete) {
363  detect_simplifications();
364  return;
365  }
366 
367  // remove simplified from path if required
368  if (_simplify.removal_required) {
369  remove_points_by_simplify_bitmask();
370  return;
371  }
372 
373  // if necessary restart detect_pruning up to last point simplified
374  if (_prune.complete) {
375  restart_pruning_if_new_points();
376  }
377  // if pruning is running, let it run to completion
378  if (!_prune.complete) {
379  detect_loops();
380  return;
381  }
382 
383  // detect path shrinkage and reduce simplify and prune path_points_completed count
384  if (_simplify.path_points_completed > path_points_completed_limit) {
385  _simplify.path_points_completed = path_points_completed_limit;
386  }
387  if (_prune.path_points_completed > path_points_completed_limit) {
388  _prune.path_points_completed = path_points_completed_limit;
389  }
390 
391  // calculate the number of points we could simplify
392  const uint16_t points_to_simplify = (path_points_count > _simplify.path_points_completed) ? (path_points_count - _simplify.path_points_completed) : 0 ;
393  const bool low_on_space = (_path_points_max - path_points_count) <= SMARTRTL_CLEANUP_START_MARGIN;
394 
395  // if 50 points can be simplified or we are low on space and at least 10 points can be simplified
396  if ((points_to_simplify >= SMARTRTL_CLEANUP_POINT_TRIGGER) || (low_on_space && (points_to_simplify >= SMARTRTL_CLEANUP_POINT_MIN))) {
397  restart_simplification(path_points_count);
398  return;
399  }
400 
401  // we are low on space, prune
402  if (low_on_space) {
403  // remove at least 10 points
404  remove_points_by_loops(SMARTRTL_CLEANUP_POINT_MIN);
405  }
406 }

408 // thorough cleanup simplifies and prunes all loops. returns true if the cleanup was completed.
409 // path_points_count is _path_points_count but passed in to avoid having to take the semaphore
410 bool AP_SmartRTL::thorough_cleanup(uint16_t path_points_count, ThoroughCleanupType clean_type)
411 {
412  if (clean_type != THOROUGH_CLEAN_PRUNE_ONLY) {
413  // restart simplify if new points have appeared on path
414  if (_simplify.complete) {
415  restart_simplify_if_new_points(path_points_count);
416  }
417  // if simplification is not complete, run it
418  if (!_simplify.complete) {
419  detect_simplifications();
420  return false;
421  }
422  // remove simplified points from path if required
423  if (_simplify.removal_required) {
424  remove_points_by_simplify_bitmask();
425  return false;
426  }
427  }
428 
429  if (clean_type != THOROUGH_CLEAN_SIMPLIFY_ONLY) {
430  // if necessary restart detect_pruning up to last point simplified
431  if (_prune.complete) {
432  restart_pruning_if_new_points();
433  }
434  // if pruning is not complete, run it
435  if (!_prune.complete) {
436  detect_loops();
437  return false;
438  }
439  // remove pruning points
440  if (!remove_points_by_loops(SMARTRTL_POINTS_MAX)) {
441  return false;
442  }
443  }
444 
445  return true;
446 }

448 // Simplifies a 3D path, according to the Ramer-Douglas-Peucker algorithm.
449 // _simplify.complete is set to true when all simplifications on the path have been identified
450 void AP_SmartRTL::detect_simplifications()
451 {
452  // complete immediately if only one segment
453  if (_simplify.path_points_count < 3) {
454  _simplify.complete = true;
455  return;
456  }
457 
458  // if not complete but also nothing to do, we must be restarting
459  if (_simplify.stack_count == 0) {
460  // reset to beginning state. add a single element in the array with:
461  // start = first path point OR the index of the last already-simplified point
462  // finish = final path point
463  _simplify.stack[0].start = (_simplify.path_points_completed > 0) ? _simplify.path_points_completed - 1 : 0;
464  _simplify.stack[0].finish = _simplify.path_points_count-1;
465  _simplify.stack_count++;
466  }
467 
468  const uint32_t start_time_us = AP_HAL::micros();
469  while (_simplify.stack_count > 0) { // while there is something to do
470 
471  // if this method has run for long enough, exit
472  if (AP_HAL::micros() - start_time_us > SMARTRTL_SIMPLIFY_TIME_US) {
473  return;
474  }
475 
476  // pop last item off the simplify stack
477  const simplify_start_finish_t tmp = _simplify.stack[--_simplify.stack_count];
478  const uint16_t start_index = tmp.start;
479  const uint16_t end_index = tmp.finish;
480 
481  // find the point between start and end points that is farthest from the start-end line segment
482  float max_dist = 0.0f;
483  uint16_t farthest_point_index = start_index;
484  for (uint16_t i = start_index + 1; i < end_index; i++) {
485  // only check points that have not already been flagged for simplification
486  if (_simplify.bitmask.get(i)) {
487  const float dist = _path[i].distance_to_segment(_path[start_index], _path[end_index]);
488  if (dist > max_dist) {
489  farthest_point_index = i;
490  max_dist = dist;
491  }
492  }
493  }
494 
495  // if the farthest point is more than ACCURACY * 0.5 add two new elements to the _simplification_stack
496  // so that on the next iteration we will check between start-to-farthestpoint and farthestpoint-to-end
497  if (max_dist > SMARTRTL_SIMPLIFY_EPSILON) {
498  // if the to-do list is full, give up on simplifying. This should never happen.
499  if (_simplify.stack_count >= _simplify.stack_max) {
500  _simplify.complete = true;
501  return;
502  }
503  _simplify.stack[_simplify.stack_count++] = simplify_start_finish_t {start_index, farthest_point_index};
504  _simplify.stack[_simplify.stack_count++] = simplify_start_finish_t {farthest_point_index, end_index};
505  } else {
506  // if the farthest point was closer than ACCURACY * 0.5 we can simplify all points between start and end
507  for (uint16_t i = start_index + 1; i < end_index; i++) {
508  _simplify.bitmask.clear(i);
509  _simplify.removal_required = true;
510  }
511  }
512  }
513  _simplify.path_points_completed = _simplify.path_points_count;
514  _simplify.complete = true;
515 }

524 void AP_SmartRTL::detect_loops()
525 {
526  // if there are less than 4 points (3 segments), mark complete
527  if (_prune.path_points_count < 4) {
528  _prune.complete = true;
529  return;
530  }
531 
532  // capture start time
533  const uint32_t start_time_us = AP_HAL::micros();
534 
535  // run for defined amount of time
536  while (AP_HAL::micros() - start_time_us < SMARTRTL_PRUNING_LOOP_TIME_US) {
537 
538  // advance inner loop
539  _prune.j++;
540  if (_prune.j > _prune.i - 2) {
541  // set inner loop back to first point
542  _prune.j = 1;
543  // reduce outer loop
544  _prune.i--;
545  // complete when outer loop has run out of new points to check
546  if (_prune.i < 4 || _prune.i < _prune.path_points_completed) {
547  _prune.complete = true;
548  _prune.path_points_completed = _prune.path_points_count;
549  return;
550  }
551  }
552 
553  // find the closest distance between two line segments and the mid-point
554  dist_point dp = segment_segment_dist(_path[_prune.i], _path[_prune.i-1], _path[_prune.j-1], _path[_prune.j]);
555  if (dp.distance < SMARTRTL_PRUNING_DELTA) {
556  // if there is a loop here, add to loop array
557  if (!add_loop(_prune.j, _prune.i-1, dp.midpoint)) {
558  // if the buffer is full, stop trying to prune
559  _prune.complete = true;
560  }
561  // set inner loop forward to trigger outer loop move to next segment
562  _prune.j = _prune.i;
563  }
564  }
565 }

620 // remove all simplify-able points from the path
621 void AP_SmartRTL::remove_points_by_simplify_bitmask()
622 {
623  // get semaphore before modifying path
624  if (!_path_sem.take_nonblocking()) {
625  return;
626  }
627  uint16_t dest = 1;
628  uint16_t removed = 0;
629  for (uint16_t src = 1; src < _path_points_count; src++) {
630  if (!_simplify.bitmask.get(src)) {
631  log_action(SRTL_POINT_SIMPLIFY, _path[src]);
632  removed++;
633  } else {
634  _path[dest] = _path[src];
635  dest++;
636  }
637  }
638 
639  // reduce count of the number of points simplified
640  if (_path_points_count > removed && _simplify.path_points_count > removed) {
641  _path_points_count -= removed;
642  _simplify.path_points_count -= removed;
643  _simplify.path_points_completed = _simplify.path_points_count;
644  } else {
645  // this is an error that should never happen so deactivate
646  deactivate(SRTL_DEACTIVATED_PROGRAM_ERROR, "program error");
647  }
648 
649  _path_sem.give();
650 
651  // flag point removal is complete
652  _simplify.bitmask.setall();
653  _simplify.removal_required = false;
654 }

656 // remove loops until at least num_point_to_delete have been removed from path
657 // does not necessarily prune all loops
658 // returns false if it failed to remove points (because it could not take semaphore)
659 bool AP_SmartRTL::remove_points_by_loops(uint16_t num_points_to_remove)
660 {
661  // exit immediately if no loops to prune
662  if (_prune.loops_count == 0) {
663  return true;
664  }
665 
666  // get semaphore before modifying path
667  if (!_path_sem.take_nonblocking()) {
668  return false;
669  }
670 
671  uint16_t removed_points = 0;
672  uint16_t i = _prune.loops_count;
673  while ((i > 0) && (removed_points < num_points_to_remove)) {
674  i--;
675  prune_loop_t loop = _prune.loops[i];
676 
677  // midpoint goes into start_index (this is the end point of the first segment)
678  _path[loop.start_index] = loop.midpoint;
679 
680  // shift points after the end of the loop down by the number of points in the loop
681  uint16_t loop_num_points_to_remove = loop.end_index - loop.start_index;
682  for (uint16_t dest = loop.start_index + 1; dest < _path_points_count - loop_num_points_to_remove; dest++) {
683  log_action(SRTL_POINT_PRUNE, _path[dest]);
684  _path[dest] = _path[dest + loop_num_points_to_remove];
685  }
686 
687  if (_path_points_count > loop_num_points_to_remove) {
688  _path_points_count -= loop_num_points_to_remove;
689  removed_points += loop_num_points_to_remove;
690  } else {
691  // this is an error that should never happen so deactivate
692  deactivate(SRTL_DEACTIVATED_PROGRAM_ERROR, "program error");
693  _path_sem.give();
694  // we return true so thorough_cleanup does not get stuck
695  return true;
696  }
697 
698  // fix the indices of any existing prune loops
699  // we do not check for overlapping loops because add_loops should have caught them
700  for (uint16_t loop_cnt = 0; loop_cnt < i; loop_cnt++) {
701  if (_prune.loops[loop_cnt].start_index >= loop.end_index) {
702  _prune.loops[loop_cnt].start_index -= loop_num_points_to_remove;
703  }
704  if (_prune.loops[loop_cnt].end_index >= loop.end_index) {
705  _prune.loops[loop_cnt].end_index -= loop_num_points_to_remove;
706  }
707  }
708 
709  // remove last prune loop from array
710  _prune.loops_count--;
711  }
712 
713  _path_sem.give();
714  return true;
715 }

717 // add loop to loops array
718 // returns true if loop added successfully, false if loop array is full
719 // checks if loop overlaps with an existing loop, keeps only the longer loop
720 bool AP_SmartRTL::add_loop(uint16_t start_index, uint16_t end_index, const Vector3f& midpoint)
721 {
722  // if the buffer is full, return failure
723  if (_prune.loops_count >= _prune.loops_max) {
724  return false;
725  }
726 
727  // sanity check indices
728  if (end_index <= start_index) {
729  return false;
730  }
731 
732  // create new loop structure and calculate length squared of loop
733  prune_loop_t new_loop = {start_index, end_index, midpoint, 0.0f};
734  new_loop.length_squared = midpoint.distance_squared(_path[start_index]) + midpoint.distance_squared(_path[end_index]);
735  for (uint16_t i = start_index; i < end_index; i++) {
736  new_loop.length_squared += _path[i].distance_squared(_path[i+1]);
737  }
738 
739  // look for overlapping loops and find their combined length
740  bool overlapping_loops = false;
741  float overlapping_loop_length = 0.0f;
742  for (uint16_t loop_idx = 0; loop_idx < _prune.loops_count; loop_idx++) {
743  if (loops_overlap(_prune.loops[loop_idx], new_loop)) {
744  overlapping_loops = true;
745  overlapping_loop_length += _prune.loops[loop_idx].length_squared;
746  }
747  }
748 
749  // handle overlapping loops
750  if (overlapping_loops) {
751  // if adding this loop would lengthen the path, discard the new loop but return success
752  if (overlapping_loop_length > new_loop.length_squared) {
753  return true;
754  }
755  // remove overlapping loops
756  uint16_t dest_idx = 0;
757  uint16_t removed = 0;
758  for (uint16_t src_idx = 0; src_idx < _prune.loops_count; src_idx++) {
759  if (loops_overlap(_prune.loops[src_idx], new_loop)) {
760  removed++;
761  } else {
762  _prune.loops[dest_idx] = _prune.loops[src_idx];
763  dest_idx++;
764  }
765  }
766  _prune.loops_count -= removed;
767  }
768 
769  // add new loop to _prune.loops array
770  _prune.loops[_prune.loops_count] = new_loop;
771  _prune.loops_count++;
772  return true;
773 }

783 AP_SmartRTL::dist_point AP_SmartRTL::segment_segment_dist(const Vector3f &p1, const Vector3f &p2, const Vector3f &p3, const Vector3f &p4)
784 {
785  const Vector3f line1 = p2-p1;
786  const Vector3f line2 = p4-p3;
787  const Vector3f line_start_diff = p1-p3; // from the beginning of the second line to the beginning of the first line
788 
789  // these don't really have a physical representation. They're only here to break up the longer formulas below.
790  const float a = line1*line1;
791  const float b = line1*line2;
792  const float c = line2*line2;
793  const float d = line1*line_start_diff;
794  const float e = line2*line_start_diff;
795 
796  // the parameter for the position on line1 and line2 which define the closest points.
797  float t1 = 0.0f;
798  float t2 = 0.0f;
799 
800  // if lines are almost parallel, return a garbage answer. This is irrelevant, since the loop
801  // could always be pruned start/end of the previous/subsequent line segment
802  if (is_zero((a*c)-(b*b))) {
803  return {FLT_MAX, Vector3f(0.0f, 0.0f, 0.0f)};
804  }
805 
806  t1 = (b*e-c*d)/(a*c-b*b);
807  t2 = (a*e-b*d)/(a*c-b*b);
808 
809  // restrict both parameters between 0 and 1.
810  t1 = constrain_float(t1, 0.0f, 1.0f);
811  t2 = constrain_float(t2, 0.0f, 1.0f);
812 
813  // difference between two closest points
814  const Vector3f dP = line_start_diff+line1*t1-line2*t2;
815 
816  const Vector3f midpoint = (p1+line1*t1 + p3+line2*t2)/2.0f;
817  return {dP.length(), midpoint};
818 }

836 // returns true if the two loops overlap (used within add_loop to determine which loops to keep or throw away)
837 bool AP_SmartRTL::loops_overlap(const prune_loop_t &loop1, const prune_loop_t &loop2) const
838 {
839  // check if loop1 within loop2
840  if (loop1.start_index >= loop2.start_index && loop1.end_index <= loop2.end_index) {
841  return true;
842  }
843  // check if loop2 within loop1
844  if (loop2.start_index >= loop1.start_index && loop2.end_index <= loop1.end_index) {
845  return true;
846  }
847  // check for partial overlap (loop1's start OR end point is within loop2)

849  const bool loop1_end_in_loop2 = (loop1.end_index >= loop2.start_index) && (loop1.end_index <= loop2.end_index);
850  if (loop1_start_in_loop2 != loop1_end_in_loop2) {
851  return true;
852  }
853  // if we got here, no overlap
854  return false;
