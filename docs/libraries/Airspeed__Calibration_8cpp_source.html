39  update the state of the airspeed calibration - needs to be called

41  */
42 float Airspeed_Calibration::update(float airspeed, const Vector3f &vg, int16_t max_airspeed_allowed_during_cal)
43 {
44  // Perform the covariance prediction
45  // Q is a diagonal matrix so only need to add three terms in
46  // C code implementation
47  // P = P + Q;
48  P.a.x += Q0;
49  P.b.y += Q0;
50  P.c.z += Q1;
51 
52  // Perform the predicted measurement using the current state estimates
53  // No state prediction required because states are assumed to be time
54  // invariant plus process noise
55  // Ignore vertical wind component
56  float TAS_pred = state.z * norm(vg.x - state.x, vg.y - state.y, vg.z);
57  float TAS_mea = airspeed;
58 
59  // Calculate the observation Jacobian H_TAS
60  float SH1 = sq(vg.y - state.y) + sq(vg.x - state.x);
61  if (SH1 < 0.000001f) {
62  // avoid division by a small number
63  return state.z;
64  }
65  float SH2 = 1/sqrtf(SH1);
66 
67  // observation Jacobian
68  Vector3f H_TAS(
69  -(state.z*SH2*(2*vg.x - 2*state.x))/2,
70  -(state.z*SH2*(2*vg.y - 2*state.y))/2,
71  1/SH2);
72 
73  // Calculate the fusion innovation covariance assuming a TAS measurement
74  // noise of 1.0 m/s
75  // S = H_TAS*P*H_TAS' + 1.0; % [1 x 3] * [3 x 3] * [3 x 1] + [1 x 1]
76  Vector3f PH = P * H_TAS;
77  float S = H_TAS * PH + 1.0f;
78 
79  // Calculate the Kalman gain
80  // [3 x 3] * [3 x 1] / [1 x 1]
81  Vector3f KG = PH / S;
82 
83  // Update the states
84  state += KG*(TAS_mea - TAS_pred); // [3 x 1] + [3 x 1] * [1 x 1]
85 
86  // Update the covariance matrix
87  Vector3f HP2 = H_TAS * P;
88  P -= KG.mul_rowcol(HP2);
89 
90  // force symmetry on the covariance matrix - necessary due to rounding
91  // errors
92  float P12 = 0.5f * (P.a.y + P.b.x);
93  float P13 = 0.5f * (P.a.z + P.c.x);
94  float P23 = 0.5f * (P.b.z + P.c.y);
95  P.a.y = P.b.x = P12;
96  P.a.z = P.c.x = P13;
97  P.b.z = P.c.y = P23;
98 
99  // Constrain diagonals to be non-negative - protects against rounding errors
100  P.a.x = MAX(P.a.x, 0.0f);
101  P.b.y = MAX(P.b.y, 0.0f);
102  P.c.z = MAX(P.c.z, 0.0f);
103 
104  state.x = constrain_float(state.x, -max_airspeed_allowed_during_cal, max_airspeed_allowed_during_cal);
105  state.y = constrain_float(state.y, -max_airspeed_allowed_during_cal, max_airspeed_allowed_during_cal);
106  state.z = constrain_float(state.z, 0.5f, 1.0f);
107 
108  return state.z;
109 }

115 void AP_Airspeed::update_calibration(uint8_t i, const Vector3f &vground, int16_t max_airspeed_allowed_during_cal)
116 {
117  if (!param[i].autocal) {
118  // auto-calibration not enabled
119  return;
120  }
121 
122  // set state.z based on current ratio, this allows the operator to
123  // override the current ratio in flight with autocal, which is
124  // very useful both for testing and to force a reasonable value.
125  float ratio = constrain_float(param[i].ratio, 1.0f, 4.0f);
126 
127  state[i].calibration.state.z = 1.0f / sqrtf(ratio);
128 
129  // calculate true airspeed, assuming a airspeed ratio of 1.0
130  float dpress = MAX(get_differential_pressure(), 0);
131  float true_airspeed = sqrtf(dpress) * AP::baro().get_EAS2TAS();
132 
133  float zratio = state[i].calibration.update(true_airspeed, vground, max_airspeed_allowed_during_cal);
134 
135  if (isnan(zratio) || isinf(zratio)) {
136  return;
137  }
138 
139  // this constrains the resulting ratio to between 1.0 and 4.0
140  zratio = constrain_float(zratio, 0.5f, 1.0f);
141  param[i].ratio.set(1/sq(zratio));
142  if (state[i].counter > 60) {
143  if (state[i].last_saved_ratio > 1.05f*param[i].ratio ||
144  state[i].last_saved_ratio < 0.95f*param[i].ratio) {
145  param[i].ratio.save();
146  state[i].last_saved_ratio = param[i].ratio;
147  state[i].counter = 0;
148  }
149  } else {
150  state[i].counter++;
151  }
152 }
