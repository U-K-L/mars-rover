51 Flow_PX4::Flow_PX4(uint32_t width, uint32_t bytesperline,
52  uint32_t max_flow_pixel,
53  float bottom_flow_feature_threshold,
54  float bottom_flow_value_threshold) :
55  _width(width),
56  _bytesperline(bytesperline),
57  _search_size(max_flow_pixel),
58  _bottom_flow_feature_threshold(bottom_flow_feature_threshold),
59  _bottom_flow_value_threshold(bottom_flow_value_threshold)
60 {
61  /* _pixlo is _search_size + 1 because if we need to evaluate
62  * the subpixels up/left of the first pixel, the index
63  * will be equal to _pixlo - _search_size -1
64  * idem if we need to evaluate the subpixels down/right
65  * the index will be equal to _pixhi + _search_size + 1
66  * which needs to remain inferior to _width - 1
67  */
68  _pixlo = _search_size + 1;
69  _pixhi = _width - 1 - (_search_size + 1);
70  /* 1 block is of size 2*_search_size + 1 + 1 pixel on each
71  * side for subpixel calculation.
72  * So _num_blocks = _width / (2 * _search_size + 3)
73  */
74  _num_blocks = _width / (2 * _search_size + 3);
75  _pixstep = ceilf(((float)(_pixhi - _pixlo)) / _num_blocks);
76 }

86 static inline uint32_t compute_diff(uint8_t *image, uint16_t offx, uint16_t offy,
87  uint16_t row_size, uint8_t window_size)
88 {
89  /* calculate position in image buffer */
90  /* we calc only the 4x4 pattern */
91  uint16_t off = (offy + 2) * row_size + (offx + 2);
92  uint32_t acc = 0;
93  unsigned int i;
94 
95  for (i = 0; i < window_size; i++) {
96  /* accumulate differences between line1/2, 2/3, 3/4 for 4 pixels
97  * starting at offset off
98  */
99  acc += abs(image[off + i] - image[off + i + row_size]);
100  acc += abs(image[off + i + row_size] - image[off + i + 2 * row_size]);
101  acc += abs(image[off + i + 2 * row_size] -
102  image[off + i + 3 * row_size]);
103 
104  /* accumulate differences between col1/2, 2/3, 3/4 for 4 pixels starting
105  * at off
106  */
107  acc += abs(image[off + row_size * i] - image[off + row_size * i + 1]);
108  acc += abs(image[off + row_size * i + 1] -
109  image[off + row_size * i + 2]);
110  acc += abs(image[off + row_size * i + 2] -
111  image[off + row_size * i + 3]);
112  }
113 
114  return acc;
115 }

127 static inline uint32_t compute_sad(uint8_t *image1, uint8_t *image2,
128  uint16_t off1x, uint16_t off1y,
129  uint16_t off2x, uint16_t off2y,
130  uint16_t row_size, uint16_t window_size)
131 {
132  /* calculate position in image buffer
133  * off1 for image1 and off2 for image2
134  */
135  uint16_t off1 = off1y * row_size + off1x;
136  uint16_t off2 = off2y * row_size + off2x;
137  unsigned int i,j;
138  uint32_t acc = 0;
139 
140  for (i = 0; i < window_size; i++) {
141  for (j = 0; j < window_size; j++) {
142  acc += abs(image1[off1 + i + j*row_size] -
143  image2[off2 + i + j*row_size]);
144  }
145  }
146  return acc;
147 }

160 static inline uint32_t compute_subpixel(uint8_t *image1, uint8_t *image2,
161  uint16_t off1x, uint16_t off1y,
162  uint16_t off2x, uint16_t off2y,
163  uint32_t *acc, uint16_t row_size,
164  uint16_t window_size)
165 {
166  /* calculate position in image buffer */
167  uint16_t off1 = off1y * row_size + off1x; // image1
168  uint16_t off2 = off2y * row_size + off2x; // image2
169  uint8_t sub[8];
170  uint16_t i, j, k;
171 
172  memset(acc, 0, window_size * sizeof(uint32_t));
173 
174  for (i = 0; i < window_size; i++) {
175  for (j = 0; j < window_size; j++) {
176  /* the 8 s values are from following positions for each pixel (X):
177  * + - + - + - +
178  * + 5 7 +
179  * + - + 6 + - +
180  * + 4 X 0 +
181  * + - + 2 + - +
182  * + 3 1 +
183  * + - + - + - +
184  */
185 
186  /* subpixel 0 is the mean value of base pixel and
187  * the pixel on the right, subpixel 1 is the mean
188  * value of base pixel, the pixel on the right,
189  * the pixel down from it, and the pixel down on
190  * the right. etc...
191  */
192  sub[0] = (image2[off2 + i + j*row_size] +
193  image2[off2 + i + 1 + j*row_size])/2;
194 
195  sub[1] = (image2[off2 + i + j*row_size] +
196  image2[off2 + i + 1 + j*row_size] +
197  image2[off2 + i + (j+1)*row_size] +
198  image2[off2 + i + 1 + (j+1)*row_size])/4;
199 
200  sub[2] = (image2[off2 + i + j*row_size] +
201  image2[off2 + i + 1 + (j+1)*row_size])/2;
202 
203  sub[3] = (image2[off2 + i + j*row_size] +
204  image2[off2 + i - 1 + j*row_size] +
205  image2[off2 + i - 1 + (j+1)*row_size] +
206  image2[off2 + i + (j+1)*row_size])/4;
207 
208  sub[4] = (image2[off2 + i + j*row_size] +
209  image2[off2 + i - 1 + (j+1)*row_size])/2;
210 
211  sub[5] = (image2[off2 + i + j*row_size] +
212  image2[off2 + i - 1 + j*row_size] +
213  image2[off2 + i - 1 + (j-1)*row_size] +
214  image2[off2 + i + (j-1)*row_size])/4;
215 
216  sub[6] = (image2[off2 + i + j*row_size] +
217  image2[off2 + i + (j-1)*row_size])/2;
218 
219  sub[7] = (image2[off2 + i + j*row_size] +
220  image2[off2 + i + 1 + j*row_size] +
221  image2[off2 + i + (j-1)*row_size] +
222  image2[off2 + i + 1 + (j-1)*row_size])/4;
223 
224  for (k = 0; k < 8; k++) {
225  acc[k] += abs(image1[off1 + i + j*row_size] - sub[k]);
226  }
227  }
228  }
229 
230  return 0;
231 }

233 uint8_t Flow_PX4::compute_flow(uint8_t *image1, uint8_t *image2,
234  uint32_t delta_time, float *pixel_flow_x,
235  float *pixel_flow_y)
236 {
237  /* constants */
238  const int16_t winmin = -_search_size;
239  const int16_t winmax = _search_size;
240  uint16_t i, j;
241  uint32_t acc[2*_search_size];
242  int8_t dirsx[_num_blocks*_num_blocks];
243  int8_t dirsy[_num_blocks*_num_blocks];
244  uint8_t subdirs[_num_blocks*_num_blocks];
245  float meanflowx = 0.0f;
246  float meanflowy = 0.0f;
247  uint16_t meancount = 0;
248  float histflowx = 0.0f;
249  float histflowy = 0.0f;
250 
251  /* iterate over all patterns
252  */
253  for (j = _pixlo; j < _pixhi; j += _pixstep) {
254  for (i = _pixlo; i < _pixhi; i += _pixstep) {
255  /* test pixel if it is suitable for flow tracking */
256  uint32_t diff = compute_diff(image1, i, j, (uint16_t) _bytesperline,
257  _search_size);
258  if (diff < _bottom_flow_feature_threshold) {
259  continue;
260  }
261 
262  uint32_t dist = 0xFFFFFFFF; // set initial distance to "infinity"
263  int8_t sumx = 0;
264  int8_t sumy = 0;
265  int8_t ii, jj;
266 
267  for (jj = winmin; jj <= winmax; jj++) {
268  for (ii = winmin; ii <= winmax; ii++) {
269  uint32_t temp_dist = compute_sad(image1, image2, i, j,
270  i + ii, j + jj,
271  (uint16_t)_bytesperline,
272  2 * _search_size);
273  if (temp_dist < dist) {
274  sumx = ii;
275  sumy = jj;
276  dist = temp_dist;
277  }
278  }
279  }
280 
281  /* acceptance SAD distance threshold */
282  if (dist < _bottom_flow_value_threshold) {
283  meanflowx += (float)sumx;
284  meanflowy += (float) sumy;
285 
286  compute_subpixel(image1, image2, i, j, i + sumx, j + sumy,
287  acc, (uint16_t) _bytesperline,
288  2 * _search_size);
289  uint32_t mindist = dist; // best SAD until now
290  uint8_t mindir = 8; // direction 8 for no direction
291  for (uint8_t k = 0; k < 2 * _search_size; k++) {
292  if (acc[k] < mindist) {
293  // SAD becomes better in direction k
294  mindist = acc[k];
295  mindir = k;
296  }
297  }
298  dirsx[meancount] = sumx;
299  dirsy[meancount] = sumy;
300  subdirs[meancount] = mindir;
301  meancount++;
302  }
303  }
304  }
305 
306  /* evaluate flow calculation */
307  if (meancount > _num_blocks * _num_blocks / 2) {
308  meanflowx /= meancount;
309  meanflowy /= meancount;
310 
311  /* use average of accepted flow values */
312  uint32_t meancount_x = 0;
313  uint32_t meancount_y = 0;
314 
315  for (uint16_t h = 0; h < meancount; h++) {
316  float subdirx = 0.0f;
317  if (subdirs[h] == 0 || subdirs[h] == 1 || subdirs[h] == 7) {
318  subdirx = 0.5f;
319  }
320  if (subdirs[h] == 3 || subdirs[h] == 4 || subdirs[h] == 5) {
321  subdirx = -0.5f;
322  }
323  histflowx += (float)dirsx[h] + subdirx;
324  meancount_x++;
325 
326  float subdiry = 0.0f;
327  if (subdirs[h] == 5 || subdirs[h] == 6 || subdirs[h] == 7) {
328  subdiry = -0.5f;
329  }
330  if (subdirs[h] == 1 || subdirs[h] == 2 || subdirs[h] == 3) {
331  subdiry = 0.5f;
332  }
333  histflowy += (float)dirsy[h] + subdiry;
334  meancount_y++;
335  }
336 
337  histflowx /= meancount_x;
338  histflowy /= meancount_y;
339 
340  *pixel_flow_x = histflowx;
341  *pixel_flow_y = histflowy;
342  } else {
343  *pixel_flow_x = 0.0f;
344  *pixel_flow_y = 0.0f;
345  return 0;
346  }
347 
348  /* calc quality */
349  uint8_t qual = (uint8_t)(meancount * 255 / (_num_blocks*_num_blocks));
350 
351  return qual;
352 }
