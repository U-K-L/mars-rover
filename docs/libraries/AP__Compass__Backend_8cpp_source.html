46 void AP_Compass_Backend::correct_field(Vector3f &mag, uint8_t i)
47 {
48  Compass::mag_state &state = _compass._state[i];
49 
50  if (state.diagonals.get().is_zero()) {
51  state.diagonals.set(Vector3f(1.0f,1.0f,1.0f));
52  }
53 
54  const Vector3f &offsets = state.offset.get();
55  const Vector3f &diagonals = state.diagonals.get();
56  const Vector3f &offdiagonals = state.offdiagonals.get();
57  const Vector3f &mot = state.motor_compensation.get();
58 
59  // add in the basic offsets
60  mag += offsets;
61 
62  // apply eliptical correction
63  Matrix3f mat(
64  diagonals.x, offdiagonals.x, offdiagonals.y,
65  offdiagonals.x, diagonals.y, offdiagonals.z,
66  offdiagonals.y, offdiagonals.z, diagonals.z
67  );
68 
69  mag = mat * mag;
70 
71  /*
72  calculate motor-power based compensation
73  note that _motor_offset[] is kept even if compensation is not
74  being applied so it can be logged correctly
75  */
76  state.motor_offset.zero();
77  if (_compass._per_motor.enabled() && i == 0) {
78  // per-motor correction is only valid for first compass
79  _compass._per_motor.compensate(state.motor_offset);
80  } else if (_compass._motor_comp_type == AP_COMPASS_MOT_COMP_THROTTLE) {
81  state.motor_offset = mot * _compass._thr;
82  } else if (_compass._motor_comp_type == AP_COMPASS_MOT_COMP_CURRENT) {
83  AP_BattMonitor &battery = AP::battery();
84  if (battery.has_current()) {
85  state.motor_offset = mot * battery.current_amps();
86  }
87  }
88 
89  /*
90  we apply the motor offsets after we apply the eliptical
91  correction. This is needed to match the way that the motor
92  compensation values are calculated, as they are calculated based
93  on final field outputs, not on the raw outputs
94  */
95  mag += state.motor_offset;
96 }

216 /* Check that the compass value is valid by using a mean filter. If
217  * the value is further than filtrer_range from mean value, it is
218  * rejected.
219 */
220 bool AP_Compass_Backend::field_ok(const Vector3f &field)
221 {
222 
223 
224  if (field.is_inf() || field.is_nan()) {
225  return false;
226  }
227 
228  const float range = (float)_compass.get_filter_range();
229  if (range <= 0) {
230  return true;
231  }
232 
233  const float length = field.length();
234 
235  if (is_zero(_mean_field_length)) {
236  _mean_field_length = length;
237  return true;
238  }
239 
240  bool ret = true;
241  const float d = fabsf(_mean_field_length - length) / (_mean_field_length + length); // diff divide by mean value in percent ( with the *200.0f on later line)
242  float koeff = FILTER_KOEF;
243 
244  if (d * 200.0f > range) { // check the difference from mean value outside allowed range
245  // printf("\nCompass field length error: mean %f got %f\n", (double)_mean_field_length, (double)length );
246  ret = false;
247  koeff /= (d * 10.0f); // 2.5 and more, so one bad sample never change mean more than 4%
248  _error_count++;
249  }
250  _mean_field_length = _mean_field_length * (1 - koeff) + length * koeff; // complimentary filter 1/k
251 
252  return ret;
253 }
