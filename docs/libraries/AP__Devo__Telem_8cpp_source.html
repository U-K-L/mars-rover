51 uint32_t AP_DEVO_Telem::gpsDdToDmsFormat(float ddm)
52 {
53  int32_t deg = (int32_t)ddm;
54  float mm = (ddm - deg) * 60.0f;
55 
56  mm = ((float)deg * 100.0f + mm) /100.0f;
57 
58  if ((mm < -180.0f) || (mm > 180.0f)) {
59  mm = 0.0f;
60  }
61 
62  return mm * 1.0e7f;
63 }

73 void AP_DEVO_Telem::send_frames()
74 {
75  // return immediately if not initialised
76  if (_port == nullptr) {
77  return;
78  }
79 
80  struct PACKED {
81  uint8_t header;
82  int32_t lon;
83  int32_t lat;
84  int32_t alt;
85  int16_t speed;
86  int16_t temp;
87  int16_t volt;
88  uint8_t checksum8;
89  } devoPacket{};
90 
91  devoPacket.header = DEVOM_SYNC_BYTE;
92 
93  const AP_AHRS &_ahrs = AP::ahrs();
94  const AP_GPS &gps = AP::gps();
95  Location loc;
96 
97  if (_ahrs.get_position(loc)) {
98  devoPacket.lat = gpsDdToDmsFormat(loc.lat);
99  devoPacket.lon = gpsDdToDmsFormat(loc.lng);
100  devoPacket.speed = (int16_t)(gps.ground_speed() * DEVO_SPEED_FACTOR * 100.0f); // * 100 for cm
101 
102  /*
103  Note that this isn't actually barometric altitude, it is the
104  inertial nav estimate of altitude above home.
105  */
106  float alt;
107  _ahrs.get_relative_position_D_home(alt);
108  devoPacket.alt = alt * -100.0f; // coordinates was in NED, so it needs to change sign. Protocol requires in cm!
109  }
110 
111 
112 
113  devoPacket.volt = roundf(AP::battery().voltage() * 10.0f);
114  devoPacket.temp = gcs().custom_mode(); // Send mode as temperature
115 
116  // emit the packet to the port byte-by-byte, calculating checksum
117  // as we go. Note we are stepping backwards through the structure
118  // - presumably to get endianness correct on the entries!
119  uint8_t *b = (uint8_t *)&devoPacket;
120  for (uint8_t i = sizeof(devoPacket)-1; i !=0; i--) {
121  _port->write(b, 1);
122  devoPacket.checksum8 += *b++; // Add Checksum
123  }
124  _port->write(&devoPacket.checksum8, 1);
125 }
