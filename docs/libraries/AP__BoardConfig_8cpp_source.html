251 void AP_BoardConfig::init()
252 {
253  board_setup();
254 
255 #if HAL_HAVE_IMU_HEATER
256  // let the HAL know the target temperature. We pass a pointer as
257  // we want the user to be able to change the parameter without
258  // rebooting
259  hal.util->set_imu_target_temp((int8_t *)&_imu_target_temperature);
260 #endif
261 
262  AP::rtc().set_utc_usec(hal.util->get_hw_rtc(), AP_RTC::SOURCE_HW);
263 
264 #if CONFIG_HAL_BOARD == HAL_BOARD_CHIBIOS && defined(USE_POSIX)
265  uint8_t slowdown = constrain_int16(_sdcard_slowdown.get(), 0, 32);
266  const uint8_t max_slowdown = 8;
267  do {
268  if (hal.util->fs_init()) {
269  break;
270  }
271  slowdown++;
272  hal.scheduler->delay(5);
273  } while (slowdown < max_slowdown);
274  if (slowdown < max_slowdown) {
275  _sdcard_slowdown.set(slowdown);
276  } else {
277  printf("SDCard failed to start\n");
278  }
279 #endif
280 }

300 void AP_BoardConfig::sensor_config_error(const char *reason)
301 {
302  _in_sensor_config_error = true;
303  /*
304  to give the user the opportunity to connect to USB we keep
305  repeating the error. The mavlink delay callback is initialised
306  before this, so the user can change parameters (and in
307  particular BRD_TYPE if needed)
308  */
309  uint32_t last_print_ms = 0;
310  while (true) {
311  uint32_t now = AP_HAL::millis();
312  if (now - last_print_ms >= 3000) {
313  last_print_ms = now;
314  printf("Sensor failure: %s\n", reason);
315 #if !APM_BUILD_TYPE(APM_BUILD_UNKNOWN)
316  gcs().send_text(MAV_SEVERITY_ERROR, "Check BRD_TYPE: %s", reason);
317 #endif
318  }
319 #if !APM_BUILD_TYPE(APM_BUILD_UNKNOWN)
320  gcs().update_receive();
321  gcs().update_send();
322 #endif
323  hal.scheduler->delay(5);
324  }
325 }
