252 // init - // init - initialise serial ports
253 void AP_SerialManager::init()
254 {
255  // always reset passthru port2 on boot
256  passthru_port2.set_and_save_ifchanged(-1);
257 
258  // initialise pointers to serial ports
259  state[1].uart = hal.uartC; // serial1, uartC, normally telem1
260  state[2].uart = hal.uartD; // serial2, uartD, normally telem2
261  state[3].uart = hal.uartB; // serial3, uartB, normally 1st GPS
262  state[4].uart = hal.uartE; // serial4, uartE, normally 2nd GPS
263  state[5].uart = hal.uartF; // serial5
264  state[6].uart = hal.uartG; // serial6
265 
266  if (state[0].uart == nullptr) {
267  init_console();
268  }
269 
270  // initialise serial ports
271  for (uint8_t i=1; i<SERIALMANAGER_NUM_PORTS; i++) {
272 
273  if (state[i].uart != nullptr) {
274 
275  // see if special options have been requested
276  if (state[i].protocol != SerialProtocol_None && state[i].options) {
277  set_options(i);
278  }
279 
280  switch (state[i].protocol) {
281  case SerialProtocol_None:
282  break;
283  case SerialProtocol_Console:
284  case SerialProtocol_MAVLink:
285  case SerialProtocol_MAVLink2:
286  state[i].uart->begin(map_baudrate(state[i].baud),
287  AP_SERIALMANAGER_MAVLINK_BUFSIZE_RX,
288  AP_SERIALMANAGER_MAVLINK_BUFSIZE_TX);
289  break;
290  case SerialProtocol_FrSky_D:
291  // Note baudrate is hardcoded to 9600
292  state[i].baud = AP_SERIALMANAGER_FRSKY_D_BAUD/1000; // update baud param in case user looks at it
293  // begin is handled by AP_Frsky_telem library
294  break;
295  case SerialProtocol_FrSky_SPort:
296  case SerialProtocol_FrSky_SPort_Passthrough:
297  // Note baudrate is hardcoded to 57600
298  state[i].baud = AP_SERIALMANAGER_FRSKY_SPORT_BAUD/1000; // update baud param in case user looks at it
299  // begin is handled by AP_Frsky_telem library
300  break;
301  case SerialProtocol_GPS:
302  case SerialProtocol_GPS2:
303  state[i].uart->begin(map_baudrate(state[i].baud),
304  AP_SERIALMANAGER_GPS_BUFSIZE_RX,
305  AP_SERIALMANAGER_GPS_BUFSIZE_TX);
306  break;
307  case SerialProtocol_AlexMos:
308  // Note baudrate is hardcoded to 115200
309  state[i].baud = AP_SERIALMANAGER_ALEXMOS_BAUD / 1000; // update baud param in case user looks at it
310  state[i].uart->begin(AP_SERIALMANAGER_ALEXMOS_BAUD,
311  AP_SERIALMANAGER_ALEXMOS_BUFSIZE_RX,
312  AP_SERIALMANAGER_ALEXMOS_BUFSIZE_TX);
313  break;
314  case SerialProtocol_SToRM32:
315  // Note baudrate is hardcoded to 115200
316  state[i].baud = AP_SERIALMANAGER_SToRM32_BAUD / 1000; // update baud param in case user looks at it
317  state[i].uart->begin(map_baudrate(state[i].baud),
318  AP_SERIALMANAGER_SToRM32_BUFSIZE_RX,
319  AP_SERIALMANAGER_SToRM32_BUFSIZE_TX);
320  break;
321  case SerialProtocol_Aerotenna_uLanding:
322  state[i].protocol.set_and_save(SerialProtocol_Rangefinder);
323  break;
324  case SerialProtocol_Volz:
325  // Note baudrate is hardcoded to 115200
326  state[i].baud = AP_SERIALMANAGER_VOLZ_BAUD; // update baud param in case user looks at it
327  state[i].uart->begin(map_baudrate(state[i].baud),
328  AP_SERIALMANAGER_VOLZ_BUFSIZE_RX,
329  AP_SERIALMANAGER_VOLZ_BUFSIZE_TX);
330  state[i].uart->set_unbuffered_writes(true);
331  state[i].uart->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
332  break;
333  case SerialProtocol_Sbus1:
334  state[i].baud = AP_SERIALMANAGER_SBUS1_BAUD / 1000; // update baud param in case user looks at it
335  state[i].uart->begin(map_baudrate(state[i].baud),
336  AP_SERIALMANAGER_SBUS1_BUFSIZE_RX,
337  AP_SERIALMANAGER_SBUS1_BUFSIZE_TX);
338  state[i].uart->configure_parity(2); // enable even parity
339  state[i].uart->set_stop_bits(2);
340  state[i].uart->set_unbuffered_writes(true);
341  state[i].uart->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
342  break;
343 
344  case SerialProtocol_ESCTelemetry:
345  // ESC telemetry protocol from BLHeli32 ESCs. Note that baudrate is hardcoded to 115200
346  state[i].baud = 115200;
347  state[i].uart->begin(map_baudrate(state[i].baud), 30, 30);
348  state[i].uart->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
349  break;
350 
351  case SerialProtocol_Robotis:
352  state[i].uart->begin(map_baudrate(state[i].baud),
353  AP_SERIALMANAGER_ROBOTIS_BUFSIZE_RX,
354  AP_SERIALMANAGER_ROBOTIS_BUFSIZE_TX);
355  state[i].uart->set_unbuffered_writes(true);
356  state[i].uart->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
357  break;
358 
359  default:
360  state[i].uart->begin(map_baudrate(state[i].baud));
361  break;
362  }
363  }
364  }

368 const AP_SerialManager::UARTState *AP_SerialManager::find_protocol_instance(enum SerialProtocol protocol, uint8_t instance) const
369 {
370  uint8_t found_instance = 0;
371 
372  // search for matching protocol
373  for(uint8_t i=0; i<SERIALMANAGER_NUM_PORTS; i++) {
374  if (protocol_match(protocol, (enum SerialProtocol)state[i].protocol.get())) {
375  if (found_instance == instance) {
376  return &state[i];
377  }
378  found_instance++;
379  }
380  }
381 
382  // if we got this far we did not find the uart
383  return nullptr;
384 }

386 // find_serial - searches available serial ports for the first instance that allows the given protocol
387 // instance should be zero if searching for the first instance, 1 for the second, etc
388 // returns uart on success, nullptr if a serial port cannot be found
389 AP_HAL::UARTDriver *AP_SerialManager::find_serial(enum SerialProtocol protocol, uint8_t instance) const
390 {
391  const struct UARTState *_state = find_protocol_instance(protocol, instance);
392  if (_state == nullptr) {
393  return nullptr;
394  }
395  return _state->uart;
396 }

465 /*
466  * map from a 16 bit EEPROM baud rate to a real baud rate. For
467  * stm32-based boards we can do 1.5MBit, although 921600 is more
468  * reliable.
469  */
470 uint32_t AP_SerialManager::map_baudrate(int32_t rate) const
471 {
472  if (rate <= 0) {
473  rate = 57;
474  }
475  switch (rate) {
476  case 1: return 1200;
477  case 2: return 2400;
478  case 4: return 4800;
479  case 9: return 9600;
480  case 19: return 19200;
481  case 38: return 38400;
482  case 57: return 57600;
483  case 100: return 100000;
484  case 111: return 111100;
485  case 115: return 115200;
486  case 230: return 230400;
487  case 256: return 256000;
488  case 460: return 460800;
489  case 500: return 500000;
490  case 921: return 921600;
491  case 1500: return 1500000;
492  }
493 
494  if (rate > 2000) {
495  // assume it is a direct baudrate. This allows for users to
496  // set an exact baudrate as long as it is over 2000 baud
497  return (uint32_t)rate;
498  }
499 
500  // otherwise allow any other kbaud rate
501  return rate*1000;
502 }

505 bool AP_SerialManager::protocol_match(enum SerialProtocol protocol1, enum SerialProtocol protocol2) const
506 {
507  // check for obvious match
508  if (protocol1 == protocol2) {
509  return true;
510  }
511 
512  // mavlink match
513  if (((protocol1 == SerialProtocol_MAVLink) || (protocol1 == SerialProtocol_MAVLink2)) &&
514  ((protocol2 == SerialProtocol_MAVLink) || (protocol2 == SerialProtocol_MAVLink2))) {
515  return true;
516  }
517 
518  // gps match
519  if (((protocol1 == SerialProtocol_GPS) || (protocol1 == SerialProtocol_GPS2)) &&
520  ((protocol2 == SerialProtocol_GPS) || (protocol2 == SerialProtocol_GPS2))) {
521  return true;
522  }
523 
524  return false;
