280 // return a steering servo output from -1.0 to +1.0 given a desired lateral acceleration rate in m/s/s.
281 // positive lateral acceleration is to the right.
282 float AR_AttitudeControl::get_steering_out_lat_accel(float desired_accel, bool motor_limit_left, bool motor_limit_right, float dt)
283 {
284  // record desired accel for reporting purposes
285  _steer_lat_accel_last_ms = AP_HAL::millis();
286  _desired_lat_accel = desired_accel;
287 
288  // get speed forward
289  float speed;
290  if (!get_forward_speed(speed)) {
291  // we expect caller will not try to control heading using rate control without a valid speed estimate
292  // on failure to get speed we do not attempt to steer
293  return 0.0f;
294  }
295 
296  const float desired_rate = get_turn_rate_from_lat_accel(desired_accel, speed);
297 
298  return get_steering_out_rate(desired_rate, motor_limit_left, motor_limit_right, dt);
299 }

312 // return a desired turn-rate given a desired heading in radians
313 float AR_AttitudeControl::get_turn_rate_from_heading(float heading_rad, float rate_max_rads) const
314 {
315  const float yaw_error = wrap_PI(heading_rad - _ahrs.yaw);
316 
317  // Calculate the desired turn rate (in radians) from the angle error (also in radians)
318  float desired_rate = _steer_angle_p.get_p(yaw_error);
319 
320  // limit desired_rate if a custom pivot turn rate is selected, otherwise use ATC_STR_RAT_MAX
321  if (is_positive(rate_max_rads)) {
322  desired_rate = constrain_float(desired_rate, -rate_max_rads, rate_max_rads);
323  }
324 
325  return desired_rate;
326 }

328 // return a steering servo output from -1 to +1 given a
329 // desired yaw rate in radians/sec. Positive yaw is to the right.
330 float AR_AttitudeControl::get_steering_out_rate(float desired_rate, bool motor_limit_left, bool motor_limit_right, float dt)
331 {
332  // sanity check dt
333  dt = constrain_float(dt, 0.0f, 1.0f);
334 
335  // if not called recently, reset input filter and desired turn rate to actual turn rate (used for accel limiting)
336  const uint32_t now = AP_HAL::millis();
337  if ((_steer_turn_last_ms == 0) || ((now - _steer_turn_last_ms) > AR_ATTCONTROL_TIMEOUT_MS)) {
338  _steer_rate_pid.reset_filter();
339  _steer_rate_pid.reset_I();
340  _desired_turn_rate = _ahrs.get_yaw_rate_earth();
341  }
342  _steer_turn_last_ms = now;
343 
344  // acceleration limit desired turn rate
345  if (is_positive(_steer_accel_max)) {
346  const float change_max = radians(_steer_accel_max) * dt;
347  desired_rate = constrain_float(desired_rate, _desired_turn_rate - change_max, _desired_turn_rate + change_max);
348  }
349  _desired_turn_rate = desired_rate;
350 
351  // rate limit desired turn rate
352  if (is_positive(_steer_rate_max)) {
353  const float steer_rate_max_rad = radians(_steer_rate_max);
354  _desired_turn_rate = constrain_float(_desired_turn_rate, -steer_rate_max_rad, steer_rate_max_rad);
355  }
356 
357  // Calculate the steering rate error (rad/sec)
358  // We do this in earth frame to allow for rover leaning over in hard corners
359  const float rate_error = (_desired_turn_rate - _ahrs.get_yaw_rate_earth());
360 
361  // set PID's dt
362  _steer_rate_pid.set_dt(dt);
363 
364  // record desired rate for logging purposes only
365  _steer_rate_pid.set_desired_rate(_desired_turn_rate);
366 
367  // pass error to PID controller
368  _steer_rate_pid.set_input_filter_all(rate_error);
369 
370  // get feed-forward
371  const float ff = _steer_rate_pid.get_ff(_desired_turn_rate);
372 
373  // get p
374  const float p = _steer_rate_pid.get_p();
375 
376  // get i unless non-skid-steering rover at low speed or steering output has hit a limit
377  float i = _steer_rate_pid.get_integrator();
378  if ((is_negative(rate_error) && !motor_limit_left) || (is_positive(rate_error) && !motor_limit_right)) {
379  i = _steer_rate_pid.get_i();
380  }
381 
382  // get d
383  const float d = _steer_rate_pid.get_d();
384 
385  // constrain and return final output
386  return (ff + p + i + d);
387 }

420 // calculate the turn rate in rad/sec given a lateral acceleration (in m/s/s) and speed (in m/s)
421 float AR_AttitudeControl::get_turn_rate_from_lat_accel(float lat_accel, float speed) const
422 {
423  // enforce minimum speed to stop oscillations when first starting to move
424  if (fabsf(speed) < AR_ATTCONTROL_STEER_SPEED_MIN) {
425  if (is_negative(speed)) {
426  speed = -AR_ATTCONTROL_STEER_SPEED_MIN;
427  } else {
428  speed = AR_ATTCONTROL_STEER_SPEED_MIN;
429  }
430  }
431 
432  return lat_accel / speed;
433 }

435 // return a throttle output from -1 to +1 given a desired speed in m/s (use negative speeds to travel backwards)
436 // motor_limit should be true if motors have hit their upper or lower limits
437 // cruise speed should be in m/s, cruise throttle should be a number from -1 to +1
438 float AR_AttitudeControl::get_throttle_out_speed(float desired_speed, bool motor_limit_low, bool motor_limit_high, float cruise_speed, float cruise_throttle, float dt)
439 {
440  // sanity check dt
441  dt = constrain_float(dt, 0.0f, 1.0f);
442 
443  // get speed forward
444  float speed;
445  if (!get_forward_speed(speed)) {
446  // we expect caller will not try to control heading using rate control without a valid speed estimate
447  // on failure to get speed we do not attempt to steer
448  return 0.0f;
449  }
450 
451  // if not called recently, reset input filter and desired speed to actual speed (used for accel limiting)
452  if (!speed_control_active()) {
453  _throttle_speed_pid.reset_filter();
454  _throttle_speed_pid.reset_I();
455  _desired_speed = speed;
456  }
457  _speed_last_ms = AP_HAL::millis();
458 
459  // acceleration limit desired speed
460  _desired_speed = get_desired_speed_accel_limited(desired_speed, dt);
461 
462  // set PID's dt
463  _throttle_speed_pid.set_dt(dt);
464 
465  // calculate speed error and pass to PID controller
466  const float speed_error = desired_speed - speed;
467  _throttle_speed_pid.set_input_filter_all(speed_error);
468 
469  // record desired speed for logging purposes only
470  _throttle_speed_pid.set_desired_rate(desired_speed);
471 
472  // get feed-forward
473  const float ff = _throttle_speed_pid.get_ff(desired_speed);
474 
475  // get p
476  const float p = _throttle_speed_pid.get_p();
477 
478  // get i unless moving at low speed or motors have hit a limit
479  float i = _throttle_speed_pid.get_integrator();
480  if ((is_negative(speed_error) && !motor_limit_low && !_throttle_limit_low) || (is_positive(speed_error) && !motor_limit_high && !_throttle_limit_high)) {
481  i = _throttle_speed_pid.get_i();
482  }
483 
484  // get d
485  const float d = _throttle_speed_pid.get_d();
486 
487  // calculate base throttle (protect against divide by zero)
488  float throttle_base = 0.0f;
489  if (is_positive(cruise_speed) && is_positive(cruise_throttle)) {
490  throttle_base = desired_speed * (cruise_throttle / cruise_speed);
491  }
492 
493  // calculate final output
494  float throttle_out = (ff+p+i+d+throttle_base);
495 
496  // clear local limit flags used to stop i-term build-up as we stop reversed outputs going to motors
497  _throttle_limit_low = false;
498  _throttle_limit_high = false;
499 
500  // protect against reverse output being sent to the motors unless braking has been enabled
501  if (!_brake_enable) {
502  // if both desired speed and actual speed are positive, do not allow negative values
503  if ((desired_speed >= 0.0f) && (throttle_out <= 0.0f)) {
504  throttle_out = 0.0f;
505  _throttle_limit_low = true;
506  } else if ((desired_speed <= 0.0f) && (throttle_out >= 0.0f)) {
507  throttle_out = 0.0f;
508  _throttle_limit_high = true;
509  }
510  }
511 
512  // final output throttle in range -1 to 1
513  return throttle_out;
514 }

516 // return a throttle output from -1 to +1 to perform a controlled stop. returns true once the vehicle has stopped
517 float AR_AttitudeControl::get_throttle_out_stop(bool motor_limit_low, bool motor_limit_high, float cruise_speed, float cruise_throttle, float dt, bool &stopped)
518 {
519  // get current system time
520  const uint32_t now = AP_HAL::millis();
521 
522  // if we were stopped in the last 300ms, assume we are still stopped
523  bool _stopped = (_stop_last_ms != 0) && (now - _stop_last_ms) < 300;
524 
525  // get deceleration limited speed
526  float desired_speed_limited = get_desired_speed_accel_limited(0.0f, dt);
527 
528  // get speed forward
529  float speed;
530  if (!get_forward_speed(speed)) {
531  // could not get speed so assume stopped
532  _stopped = true;
533  } else {
534  // if desired speed is zero and vehicle drops below _stop_speed consider it stopped
535  if (is_zero(desired_speed_limited) && fabsf(speed) <= fabsf(_stop_speed)) {
536  _stopped = true;
537  }
538  }
539 
540  // set stopped status for caller
541  stopped = _stopped;
542 
543  // if stopped return zero
544  if (stopped) {
545  // update last time we thought we were stopped
546  _stop_last_ms = now;
547  // set last time speed controller was run so accelerations are limited
548  _speed_last_ms = now;
549  return 0.0f;
550  }
551 
552  // clear stopped system time
553  _stop_last_ms = 0;
554  // run speed controller to bring vehicle to stop
555  return get_throttle_out_speed(desired_speed_limited, motor_limit_low, motor_limit_high, cruise_speed, cruise_throttle, dt);
556 }

558 // balancebot pitch to throttle controller
559 // returns a throttle output from -100 to +100 given a desired pitch angle and vehicle's current speed (from wheel encoders)
560 // desired_pitch is in radians, veh_speed_pct is supplied as a percentage (-100 to +100) of vehicle's top speed
561 float AR_AttitudeControl::get_throttle_out_from_pitch(float desired_pitch, float vehicle_speed_pct, bool motor_limit_low, bool motor_limit_high, float dt)
562 {
563  // sanity check dt
564  dt = constrain_float(dt, 0.0f, 1.0f);
565 
566  // if not called recently, reset input filter
567  const uint32_t now = AP_HAL::millis();
568  if ((_balance_last_ms == 0) || ((now - _balance_last_ms) > AR_ATTCONTROL_TIMEOUT_MS)) {
569  _pitch_to_throttle_pid.reset_filter();
570  _pitch_to_throttle_pid.reset_I();
571  }
572  _balance_last_ms = now;
573 
574  // calculate pitch error
575  const float pitch_error = desired_pitch - _ahrs.pitch;
576 
577  // set PID's dt
578  _pitch_to_throttle_pid.set_dt(dt);
579 
580  // record desired speed for logging purposes only
581  _pitch_to_throttle_pid.set_desired_rate(desired_pitch);
582 
583  // pitch error is given as input to PID contoller
584  _pitch_to_throttle_pid.set_input_filter_all(pitch_error);
585 
586  // get feed-forward
587  const float ff = _pitch_to_throttle_pid.get_ff(desired_pitch);
588 
589  // get p
590  const float p = _pitch_to_throttle_pid.get_p();
591 
592  // get i unless non-skid-steering rover at low speed or steering output has hit a limit
593  float i = _pitch_to_throttle_pid.get_integrator();
594  if ((is_negative(pitch_error) && !motor_limit_low) || (is_positive(pitch_error) && !motor_limit_high)) {
595  i = _pitch_to_throttle_pid.get_i();
596  }
597 
598  // get d
599  const float d = _pitch_to_throttle_pid.get_d();
600 
601  // add feed forward from speed
602  const float spd_ff = vehicle_speed_pct * 0.01f * _pitch_to_throttle_speed_ff;
603 
604  // constrain and return final output
605  return (ff + p + i + d + spd_ff);
606 }

671 // get forward speed in m/s (earth-frame horizontal velocity but only along vehicle x-axis). returns true on success
672 bool AR_AttitudeControl::get_forward_speed(float &speed) const
673 {
674  Vector3f velocity;
675  if (!_ahrs.get_velocity_NED(velocity)) {
676  // use less accurate GPS, assuming entire length is along forward/back axis of vehicle
677  if (AP::gps().status() >= AP_GPS::GPS_OK_FIX_3D) {
678  if (fabsf(wrap_180_cd(_ahrs.yaw_sensor - AP::gps().ground_course_cd())) <= 9000) {
679  speed = AP::gps().ground_speed();
680  } else {
681  speed = -AP::gps().ground_speed();
682  }
683  return true;
684  } else {
685  return false;
686  }
687  }
688  // calculate forward speed velocity into body frame
689  speed = velocity.x*_ahrs.cos_yaw() + velocity.y*_ahrs.sin_yaw();
690  return true;

722 // get acceleration limited desired speed
723 float AR_AttitudeControl::get_desired_speed_accel_limited(float desired_speed, float dt) const
724 {
725  // return input value if no recent calls to speed controller
726  // apply no limiting when ATC_ACCEL_MAX is set to zero
727  if (!speed_control_active() || !is_positive(_throttle_accel_max)) {
728  return desired_speed;
729  }
730 
731  // sanity check dt
732  dt = constrain_float(dt, 0.0f, 1.0f);
733 
734  // use previous desired speed as basis for accel limiting
735  float speed_prev = _desired_speed;
736 
737  // if no recent calls to speed controller limit based on current speed
738  if (!speed_control_active()) {
739  get_forward_speed(speed_prev);
740  }
741 
742  // acceleration limit desired speed
743  float speed_change_max;
744  if (fabsf(desired_speed) < fabsf(_desired_speed) && is_positive(_throttle_decel_max)) {
745  speed_change_max = _throttle_decel_max * dt;
746  } else {
747  speed_change_max = _throttle_accel_max * dt;
748  }
749  return constrain_float(desired_speed, speed_prev - speed_change_max, speed_prev + speed_change_max);
750 }

752 // get minimum stopping distance (in meters) given a speed (in m/s)
753 float AR_AttitudeControl::get_stopping_distance(float speed) const
754 {

758  // avoid divide by zero
