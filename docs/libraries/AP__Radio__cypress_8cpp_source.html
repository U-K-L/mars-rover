744 /*
745  parse channels from a packet
746  */
747 bool AP_Radio_cypress::parse_dsm_channels(const uint8_t *data)
748 {
749  uint16_t num_values = 0;
750  uint16_t pwm_channels[max_channels] {};
751 
752  // default value for channels above 4 is previous value
753  memcpy(&pwm_channels[4], &dsm.pwm_channels[4], (max_channels-4)*sizeof(uint16_t));
754 
755  if (!dsm_decode(AP_HAL::micros64(),
756  data,
757  pwm_channels,
758  &num_values,
759  ARRAY_SIZE(pwm_channels))) {
760  // invalid packet
761  Debug(2, "DSM: bad decode\n");
762  return false;
763  }
764  if (num_values < 5) {
765  Debug(2, "DSM: num_values=%u\n", num_values);
766  return false;
767  }
768 
769  // cope with mode1/mode2
770  map_stick_mode(pwm_channels);
771 
772  memcpy(dsm.pwm_channels, pwm_channels, num_values*sizeof(uint16_t));
773 
774  dsm.last_parse_us = AP_HAL::micros();
775 
776  // suppress channel 8 ack values
777  dsm.num_channels = num_values==8?7:num_values;
778 
779  if (num_values == 8) {
780  // decode telemetry ack value and version
781  uint16_t d=0;
782  if (is_DSM2()) {
783  d = data[14] << 8 | data[15];
784  } else {
785  // see chan_order[] for DSMX
786  d = data[10] << 8 | data[11];
787  }
788  // extra data is sent on channel 8, with 3 bit key and 8 bit data
789  uint8_t chan = d>>11;
790  uint8_t key = (d >> 8) & 0x7;
791  uint8_t v = d & 0xFF;
792  if (chan == 7 && key == 0) {
793  // got an ack from key 0
794  Debug(4, "ack %u seq=%u acked=%u length=%u len=%u\n",
795  v, fwupload.sequence, unsigned(fwupload.acked), unsigned(fwupload.length), fwupload.len);
796  if (fwupload.sequence == v && sem.take_nonblocking()) {
797  fwupload.sequence++;
798  fwupload.acked += fwupload.len;
799  if (fwupload.acked == fwupload.length) {
800  // trigger send of DATA16 ack to client
801  fwupload.need_ack = true;
802  }
803  sem.give();
804  }
805  }
806  if (chan == 7) {
807  // extract telemetry extra data
808  switch (key) {
809  case 1:
810  dsm.tx_firmware_year = v;
811  break;
812  case 2:
813  dsm.tx_firmware_month = v;
814  break;
815  case 3:
816  dsm.tx_firmware_day = v;
817  break;
818  case 4:
819  dsm.tx_rssi = v;
820  break;
821  case 5:
822  dsm.tx_pps = v;
823  dsm.have_tx_pps = true;
824  check_double_bind();
825  break;
826  case 6:
827  if (v != dsm.tx_bl_version) {
828  if (v == 2) {
829  // TX with new filter gets a default power of 6
830  set_tx_max_power_default(6);
831  }
832  }
833  dsm.tx_bl_version = v;
834  break;
835  }
836  }
837  }
838  return true;
839 }

841 /*
842  process an incoming bind packet
843  */
844 void AP_Radio_cypress::process_bind(const uint8_t *pkt, uint8_t len)
845 {
846  if (len != 16) {
847  return;
848  }
849  bool ok = (len==16 && pkt[0] == pkt[4] && pkt[1] == pkt[5] && pkt[2] == pkt[6] && pkt[3] == pkt[7]);
850 
851  // Calculate the first sum
852  uint16_t bind_sum = 384 - 0x10;
853  for (uint8_t i = 0; i < 8; i++) {
854  bind_sum += pkt[i];
855  }
856 
857  // Check the first sum
858  if (pkt[8] != bind_sum >> 8 || pkt[9] != (bind_sum & 0xFF)) {
859  ok = false;
860  }
861 
862  // Calculate second sum
863  for (uint8_t i = 8; i < 14; i++) {
864  bind_sum += pkt[i];
865  }
866 
867  // Check the second sum
868  if (pkt[14] != bind_sum >> 8 || pkt[15] != (bind_sum & 0xFF)) {
869  ok = false;
870  }
871 
872  if (state == STATE_AUTOBIND) {
873  uint8_t rssi = read_register(CYRF_RSSI) & 0x1F;
874  Debug(3,"bind RSSI %u\n", rssi);
875  if (rssi < get_autobind_rssi()) {
876  ok = false;
877  }
878  }
879 
880  if (ok) {
881  uint8_t mfg_id[4] = {uint8_t(~pkt[0]), uint8_t(~pkt[1]), uint8_t(~pkt[2]), uint8_t(~pkt[3])};
882  uint8_t num_chan = pkt[11];
883  uint8_t protocol = pkt[12];
884  (void)num_chan;
885  // change to normal receive
886  memcpy(dsm.mfg_id, mfg_id, 4);
887  state = STATE_RECV;
888 
889  radio_set_config(cyrf_transfer_config, ARRAY_SIZE(cyrf_transfer_config));
890 
891  if (get_disable_crc()) {
892  write_register(CYRF_RX_OVERRIDE, CYRF_DIS_RXCRC);
893  }
894 
895  dsm.protocol = (enum dsm_protocol)protocol;
896  dsm_setup_transfer_dsmx();
897 
898  Debug(1, "BIND OK: mfg_id={0x%02x, 0x%02x, 0x%02x, 0x%02x} N=%u P=0x%02x DSM2=%u\n",
899  mfg_id[0], mfg_id[1], mfg_id[2], mfg_id[3],
900  num_chan,
901  protocol,
902  is_DSM2());
903 
904  dsm.last_recv_us = AP_HAL::micros();
905 
906  if (is_DSM2()) {
907  dsm2_start_sync();
908  }
909 
910  dsm.need_bind_save = true;
911  }
912 }

940 /*
941  process an incoming packet
942  */
943 void AP_Radio_cypress::process_packet(const uint8_t *pkt, uint8_t len)
944 {
945  if (len == 16) {
946  bool ok;
947  const uint8_t *id = dsm.mfg_id;
948  uint32_t now = AP_HAL::micros();
949 
950  if (is_DSM2()) {
951  ok = (pkt[0] == ((~id[2])&0xFF) && pkt[1] == (~id[3]&0xFF));
952  } else {
953  ok = (pkt[0] == id[2] && pkt[1] == id[3]);
954  }
955  if (ok && is_DSM2() && dsm.sync < DSM2_OK) {
956  if (dsm.sync == DSM2_SYNC_A) {
957  dsm.channels[0] = dsm.current_rf_channel;
958  dsm.sync = DSM2_SYNC_B;
959  Debug(2, "DSM2 SYNCA chan=%u\n", dsm.channels[0]);
960  dsm.last_recv_us = now;
961  } else {
962  if (dsm.current_rf_channel != dsm.channels[0]) {
963  dsm.channels[1] = dsm.current_rf_channel;
964  dsm.sync = DSM2_OK;
965  Debug(2, "DSM2 SYNCB chan=%u\n", dsm.channels[1]);
966  dsm.last_recv_us = now;
967  }
968  }
969  return;
970  }
971  if (ok && (!is_DSM2() || dsm.sync >= DSM2_SYNC_B)) {
972  ok = parse_dsm_channels(pkt);
973  }
974  if (ok) {
975  uint32_t packet_dt_us = now - dsm.last_recv_us;
976 
977  dsm.last_recv_chan = dsm.current_channel;
978  dsm.last_recv_us = now;
979  if (dsm.crc_errors > 2) {
980  dsm.crc_errors -= 2;
981  }
982 
983  stats.recv_packets++;
984 
985  // sample the RSSI
986  uint8_t rssi = read_register(CYRF_RSSI) & 0x1F;
987  dsm.rssi = 0.95 * dsm.rssi + 0.05 * rssi;
988 
989  if (packet_dt_us < 5000) {
990  dsm.pkt_time1 = packet_dt_us;
991  } else if (packet_dt_us < 8000) {
992  dsm.pkt_time2 = packet_dt_us;
993  }
994 
995  if (get_telem_enable()) {
996  if (packet_dt_us < 5000 &&
997  (get_autobind_time() == 0 || dsm.have_tx_pps)) {
998  /*
999  we have just received two packets rapidly, which
1000  means we have about 7ms before the next
1001  one. That gives time for a telemetry packet. We
1002  send it 1ms after we receive the incoming packet
1003 
1004  If auto-bind is enabled we don't send telemetry
1005  till we've received a tx_pps value from the
1006  TX. This allows us to detect double binding (two
1007  RX bound to the same TX)
1008  */
1009  state = STATE_SEND_TELEM;
1010  setup_timeout(1);
1011  }
1012  }
1013  } else {
1014  stats.bad_packets++;
1015  }
1016  } else {
1017  stats.bad_packets++;
1018  }
1019 }

1273  */
1274 void AP_Radio_cypress::dsm_generate_channels_dsmx(uint8_t mfg_id[4], uint8_t channels[23])
1275 {
1276  // Calculate the DSMX channels
1277  int idx = 0;
1278  uint32_t id = ~((mfg_id[0] << 24) | (mfg_id[1] << 16) |
1279  (mfg_id[2] << 8) | (mfg_id[3] << 0));
1280  uint32_t id_tmp = id;
1281 
1282  // While not all channels are set
1283  while(idx < 23) {
1284  int i;
1285  int count_3_27 = 0, count_28_51 = 0, count_52_76 = 0;
1286 
1287  id_tmp = id_tmp * 0x0019660D + 0x3C6EF35F; // Randomization
1288  uint8_t next_ch = ((id_tmp >> 8) % 0x49) + 3; // Use least-significant byte and must be larger than 3
1289  if (((next_ch ^ id) & 0x01 ) == 0) {
1290  continue;
1291  }
1292 
1293  // Go trough all already set channels
1294  for (i = 0; i < idx; i++) {
1295  // Channel is already used
1296  if (channels[i] == next_ch) {
1297  break;
1298  }
1299 
1300  // Count the channel groups
1301  if(channels[i] <= 27) {
1302  count_3_27++;
1303  } else if (channels[i] <= 51) {
1304  count_28_51++;
1305  } else {
1306  count_52_76++;
1307  }
1308  }
1309 
1310  // When channel is already used continue
1311  if (i != idx) {
1312  continue;
1313  }
1314 
1315  // Set the channel when channel groups aren't full
1316  if ((next_ch < 28 && count_3_27 < 8) // Channels 3-27: max 8
1317  || (next_ch >= 28 && next_ch < 52 && count_28_51 < 7) // Channels 28-52: max 7
1318  || (next_ch >= 52 && count_52_76 < 8)) { // Channels 52-76: max 8
1319  channels[idx++] = next_ch;
1320  }
1321  }
1322 
1323  Debug(2, "Generated DSMX channels\n");
1324 }

1379  if (now - dsm.last_chan_change_us > 15000) {

1382  dsm.current_rf_channel = (dsm.current_rf_channel+2) % DSM_MAX_CHANNEL;
1383  dsm.last_chan_change_us = now;

1387  dsm.sop_col, dsm.data_col,
1388  dsm.sync==DSM2_SYNC_B?~dsm.crc_seed:dsm.crc_seed);
1389  return;
1390  }
1391 
1392  if (dt < 1000) {
1393  // normal channel advance
1394  next_channel = dsm.last_recv_chan + 1;
1395  } else if (dt > 20*cycle_time) {
1396  // change channel slowly
1397  next_channel = dsm.last_recv_chan + (dt / (cycle_time*2));
1398  } else {
1399  // predict next channel
1400  next_channel = dsm.last_recv_chan + 1;
1401  next_channel += (dt / cycle_time) * 2;
1402  if (dt % cycle_time > (unsigned)(dsm.pkt_time1 + 1000U)) {
1403  next_channel++;
1404  }
1405  }

1412  dsm.crc_seed = ~dsm.crc_seed;
1413  }
1414  }
1415 
1416  if (is_DSM2() && dsm.sync == DSM2_SYNC_B && dsm.current_channel == 1) {
1417  // scan to next channelb
1418  do {
1419  dsm.channels[1] &= ~1;
1420  dsm.channels[1] = (dsm.channels[1]+2) % DSM_MAX_CHANNEL;
1421  } while (dsm.channels[1] == dsm.channels[0]);
1422  }
1423

1425

1427  if (dsm.current_channel & 1) {
1428  seed = ~seed;
1429  }
1430 
1431  if (is_DSM2()) {
1432  if (now - dsm.last_recv_us > 5000000) {
1433  dsm2_start_sync();
1434  }
1435  }
1436 
1437  dsm_set_channel(dsm.current_rf_channel, is_DSM2(),
1438  dsm.sop_col, dsm.data_col, seed);
