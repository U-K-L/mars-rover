596 // Fuse the SoloGimbalEKF velocity estimates - this enables alevel reference to be maintained during constant turns
597 void SoloGimbalEKF::fuseVelocity()
598 {
599  const AP_AHRS_NavEKF &_ahrs = AP::ahrs_navekf();
600 
601  if (!_ahrs.have_inertial_nav()) {
602  return;
603  }
604 
605  float R_OBS = 0.25f;
606  float innovation[3];
607  float varInnov[3];
608  Vector3f angErrVec;
609  uint8_t stateIndex;
610  float K[9];
611  // Fuse measurements sequentially
612  for (uint8_t obsIndex=0;obsIndex<=2;obsIndex++) {
613  stateIndex = 3 + obsIndex;
614 
615  // Calculate the velocity measurement innovation using the SoloGimbalEKF estimate as the observation
616  // if heading isn't aligned, use zero velocity (static assumption)
617  if (YawAligned) {
618  Vector3f measVelNED = Vector3f(0,0,0);
619  nav_filter_status main_ekf_status;
620 
621  if (_ahrs.get_filter_status(main_ekf_status)) {
622  if (main_ekf_status.flags.horiz_vel) {
623  _ahrs.get_velocity_NED(measVelNED);
624  }
625  }
626 
627  innovation[obsIndex] = state.velocity[obsIndex] - measVelNED[obsIndex];
628  } else {
629  innovation[obsIndex] = state.velocity[obsIndex];
630  }
631 
632  // Zero the attitude error states - they represent the incremental error so must be zero before corrections are applied
633  state.angErr.zero();
634  // Calculate the innovation variance
635  varInnov[obsIndex] = Cov[stateIndex][stateIndex] + R_OBS;
636  // Calculate the Kalman gain and correct states, taking advantage of direct state observation
637  for (uint8_t rowIndex=0;rowIndex<=8;rowIndex++) {
638  K[rowIndex] = Cov[rowIndex][stateIndex]/varInnov[obsIndex];
639  states[rowIndex] -= K[rowIndex] * innovation[obsIndex];
640  }
641 
642  // Store tilt error estimate for external monitoring
643  angErrVec = angErrVec + state.angErr;
644 
645  // the first 3 states represent the angular error vector where truth = estimate + error. This is is used to correct the estimated quaternion
646  // Bring the quaternion state estimate back to 'truth' by adding the error
647  state.quat.rotate(state.angErr);
648 
649  // re-normalise the quaternion
650  state.quat.normalize();
651 
652  // Update the covariance
653  for (uint8_t rowIndex=0;rowIndex<=8;rowIndex++) {
654  for (uint8_t colIndex=0;colIndex<=8;colIndex++) {
655  Cov[rowIndex][colIndex] = Cov[rowIndex][colIndex] - K[rowIndex]*Cov[stateIndex][colIndex];
656  }
657  }
658 
659  // force symmetry and constrain diagonals to be non-negative
660  fixCovariance();
661  }
662 
663  // calculate tilt component of angle correction
664  TiltCorrectionSquared = sq(angErrVec.x) + sq(angErrVec.y);
665 }

775  // Calculate innovation variance and Kalman gains, taking advantage of the fact that only the first 3 elements in H are non zero
776  float PH[3];
777  float varInnov = R_MAG;
778  for (uint8_t rowIndex=0;rowIndex<=2;rowIndex++) {
779  PH[rowIndex] = 0.0f;
780  for (uint8_t colIndex=0;colIndex<=2;colIndex++) {
781  PH[rowIndex] += Cov[rowIndex][colIndex]*H_MAG[colIndex];
782  }
783  varInnov += H_MAG[rowIndex]*PH[rowIndex];

787  for (uint8_t rowIndex=0;rowIndex<=8;rowIndex++) {
788  K_MAG[rowIndex] = 0.0f;
789  for (uint8_t colIndex=0;colIndex<=2;colIndex++) {

791  }
792  K_MAG[rowIndex] *= varInnovInv;
793  }

799  if (innovation > 0.5f) {
800  innovation = 0.5f;
801  } else if (innovation < -0.5f) {
802  innovation = -0.5f;
803  }
804 
805  // correct the state vector
806  state.angErr.zero();
807  for (uint8_t i=0;i<=8;i++) {
808  states[i] -= K_MAG[i] * innovation;
809  }
810

812  // Bring the quaternion state estimate back to 'truth' by adding the error
813  state.quat.rotate(state.angErr);
814 
815  // re-normalise the quaternion
816  state.quat.normalize();
817 
818  // correct the covariance using P = P - K*H*P taking advantage of the fact that only the first 3 elements in H are non zero
819  float HP[9];
820  for (uint8_t colIndex=0;colIndex<=8;colIndex++) {

825  }

827  for (uint8_t colIndex=0;colIndex<=8;colIndex++) {
828  Cov[rowIndex][colIndex] -= K_MAG[rowIndex] * HP[colIndex];
829  }
830  }

832  // force symmetry and constrain diagonals to be non-negative
833  fixCovariance();
834 }

854 // Calculate magnetic heading innovation
855 float SoloGimbalEKF::calcMagHeadingInnov()
856 {
857  // Define rotation from magnetometer to sensor using a 312 rotation sequence
858  Matrix3f Tms;
859  Tms[0][0] = cosTheta*cosPsi-sinPsi*sinPhi*sinTheta;
860  Tms[1][0] = -sinPsi*cosPhi;
861  Tms[2][0] = cosPsi*sinTheta+cosTheta*sinPsi*sinPhi;
862  Tms[0][1] = cosTheta*sinPsi+cosPsi*sinPhi*sinTheta;
863  Tms[1][1] = cosPsi*cosPhi;
864  Tms[2][1] = sinPsi*sinTheta-cosTheta*cosPsi*sinPhi;
865  Tms[0][2] = -sinTheta*cosPhi;
866  Tms[1][2] = sinPhi;
867  Tms[2][2] = cosTheta*cosPhi;
868 
869  const AP_AHRS_NavEKF &_ahrs = AP::ahrs_navekf();
870 
871  // get earth magnetic field estimate from main ekf if available to take advantage of main ekf magnetic field learning
872  Vector3f earth_magfield = Vector3f(0,0,0);
873  _ahrs.get_mag_field_NED(earth_magfield);
874 
875  float declination;
876  if (!earth_magfield.is_zero()) {
877  declination = atan2f(earth_magfield.y,earth_magfield.x);
878  } else {
879  declination = _ahrs.get_compass()->get_declination();
880  }
881 
882  Vector3f body_magfield = Vector3f(0,0,0);
883  _ahrs.get_mag_field_correction(body_magfield);
884 
885  // Define rotation from magnetometer to NED axes
886  Matrix3f Tmn = Tsn*Tms;
887 
888  // rotate magentic field measured at top plate into NED axes afer applying bias values learnt by main EKF
889  Vector3f magMeasNED = Tmn*(magData - body_magfield);
890 
891  // calculate the innovation where the predicted measurement is the angle wrt magnetic north of the horizontal component of the measured field
892  float innovation = atan2f(magMeasNED.y,magMeasNED.x) - declination;
893 
894  // wrap the innovation so it sits on the range from +-pi
895  if (innovation > M_PI) {
896  innovation = innovation - 2*M_PI;
897  } else if (innovation < -M_PI) {
898  innovation = innovation + 2*M_PI;
899  }
900 
901  // Unwrap so that a large yaw gyro bias offset that causes the heading to wrap does not lead to continual uncontrolled heading drift
902  if (innovation - lastInnovation > M_PI) {
903  // Angle has wrapped in the positive direction to subtract an additional 2*Pi
904  innovationIncrement -= 2*M_PI;
905  } else if (innovation -innovationIncrement < -M_PI) {
906  // Angle has wrapped in the negative direction so add an additional 2*Pi
907  innovationIncrement += 2*M_PI;
908  }
909  lastInnovation = innovation;
910 
911  return innovation + innovationIncrement;
912 }
