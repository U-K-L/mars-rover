115 /*
116  probe and initialise accelerometer
117  */
118 bool AP_InertialSensor_BMI055::accel_init()
119 {
120  dev_accel->get_semaphore()->take_blocking();
121 
122  uint8_t v;
123  if (!dev_accel->read_registers(REGA_BGW_CHIPID, &v, 1) || v != 0xFA) {
124  goto failed;
125  }
126 
127  if (!dev_accel->write_register(REGA_BGW_SOFTRESET, 0xB6)) {
128  goto failed;
129  }
130  hal.scheduler->delay(10);
131 
132  dev_accel->setup_checked_registers(5, 20);
133 
134  // setup 16g range
135  if (!dev_accel->write_register(REGA_PMU_RANGE, 0x0C, true)) {
136  goto failed;
137  }
138 
139  // setup filter bandwidth 1kHz
140  if (!dev_accel->write_register(REGA_PMU_BW, 0x0F, true)) {
141  goto failed;
142  }
143 
144  // disable low-power mode
145  if (!dev_accel->write_register(REGA_PMU_LPW, 0, true)) {
146  goto failed;
147  }
148 
149  // setup for unfiltered data
150  if (!dev_accel->write_register(REGA_ACCD_HBW, 0x80, true)) {
151  goto failed;
152  }
153 
154  // setup FIFO for streaming X,Y,Z
155  if (!dev_accel->write_register(REGA_FIFO_CONFIG_1, 0x80, true)) {
156  goto failed;
157  }
158 
159 
160  hal.console->printf("BMI055: found accel\n");
161 
162  dev_accel->get_semaphore()->give();
163  return true;
164 
165 failed:
166  dev_accel->get_semaphore()->give();
167  return false;
168 }

232 /*
233  read accel fifo
234  */
235 void AP_InertialSensor_BMI055::read_fifo_accel(void)
236 {
237  uint8_t num_frames;
238  if (!dev_accel->read_registers(REGA_FIFO_STATUS, &num_frames, 1)) {
239  _inc_accel_error_count(accel_instance);
240  return;
241  }
242  num_frames &= 0x7F;
243 
244  // don't read more than 8 frames at a time
245  if (num_frames > 8) {
246  num_frames = 8;
247  }
248 
249  if (num_frames == 0) {
250  return;
251  }
252 
253  uint8_t data[6*num_frames];
254  if (!dev_accel->read_registers(REGA_FIFO_DATA, data, num_frames*6)) {
255  _inc_accel_error_count(accel_instance);
256  return;
257  }
258  // data is 12 bits with 16g range, 7.81mg/LSB
259  const float scale = 7.81 * 0.001 * GRAVITY_MSS / 16.0f;
260  for (uint8_t i = 0; i < num_frames; i++) {
261  const uint8_t *d = &data[i*6];
262  int16_t xyz[3] {
263  int16_t(uint16_t((d[0]&0xF0) | (d[1]<<8))),
264  int16_t(uint16_t((d[2]&0xF0) | (d[3]<<8))),
265  int16_t(uint16_t((d[4]&0xF0) | (d[5]<<8))) };
266  Vector3f accel(xyz[0], xyz[1], xyz[2]);
267 
268  accel *= scale;
269 
270  _rotate_and_correct_accel(accel_instance, accel);
271  _notify_new_accel_raw_sample(accel_instance, accel);
272  }
273 
274  if (temperature_counter++ == 100) {
275  temperature_counter = 0;
276  int8_t t;
277  if (!dev_accel->read_registers(REGA_ACCD_TEMP, (uint8_t *)&t, 1)) {
278  _inc_accel_error_count(accel_instance);
279  } else {
280  float temp_degc = (0.5f * t) + 23.0f;
281  _publish_temperature(accel_instance, temp_degc);
282  }
283  }
284 
285  if (!dev_accel->check_next_register()) {
286  _inc_accel_error_count(accel_instance);
287  }
288 }
