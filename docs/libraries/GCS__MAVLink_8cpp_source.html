54 /*
55  lock a channel, preventing use by MAVLink
56  */
57 void GCS_MAVLINK::lock_channel(mavlink_channel_t _chan, bool lock)
58 {
59  if (!valid_channel(chan)) {
60  return;
61  }
62  if (lock) {
63  mavlink_locked_mask |= (1U<<(unsigned)_chan);
64  } else {
65  mavlink_locked_mask &= ~(1U<<(unsigned)_chan);
66  }
67 }

99 uint16_t comm_get_txspace(mavlink_channel_t chan)
100 {
101  if (!valid_channel(chan)) {
102  return 0;
103  }
104  if ((1U<<chan) & mavlink_locked_mask) {
105  return 0;
106  }
107  int16_t ret = mavlink_comm_port[chan]->txspace();
108  if (ret < 0) {
109  ret = 0;
110  }
111  return (uint16_t)ret;
112 }

118 uint16_t comm_get_available(mavlink_channel_t chan)
119 {
120  if (!valid_channel(chan)) {
121  return 0;
122  }
123  if ((1U<<chan) & mavlink_locked_mask) {
124  return 0;
125  }
126  int16_t bytes = mavlink_comm_port[chan]->available();
127  if (bytes == -1) {
128  return 0;
129  }
130  return (uint16_t)bytes;

133 /*
134  send a buffer out a MAVLink channel
135  */
136 void comm_send_buffer(mavlink_channel_t chan, const uint8_t *buf, uint8_t len)
137 {
138  if (!valid_channel(chan)) {
139  return;
140  }
141  if (gcs_alternative_active[chan]) {
142  // an alternative protocol is active
143  return;
144  }
145  const size_t written = mavlink_comm_port[chan]->write(buf, len);
146 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
147  if (written < len) {
148  AP_HAL::panic("Short write on UART: %lu < %u", written, len);
149  }
150 #else
151  (void)written;
152 #endif
153 }
