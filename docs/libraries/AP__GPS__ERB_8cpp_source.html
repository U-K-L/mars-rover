35 // Process bytes available from the stream
36 //
37 // The stream is assumed to contain only messages we recognise. If it
38 // contains other messages, and those messages contain the preamble
39 // bytes, it is possible for this code to fail to synchronise to the
40 // stream immediately. Without buffering the entire message and
41 // re-processing it from the top, this is unavoidable. The parser
42 // attempts to avoid this when possible.
43 //
44 bool
45 AP_GPS_ERB::read(void)
46 {
47  uint8_t data;
48  int16_t numc;
49  bool parsed = false;
50 
51  numc = port->available();
52  for (int16_t i = 0; i < numc; i++) { // Process bytes received
53 
54  // read the next byte
55  data = port->read();
56 
57  reset:
58  switch(_step) {
59 
60  // Message preamble detection
61  //
62  case 1:
63  if (PREAMBLE2 == data) {
64  _step++;
65  break;
66  }
67  _step = 0;
68  Debug("reset %u", __LINE__);
69  FALLTHROUGH;
70  case 0:
71  if(PREAMBLE1 == data)
72  _step++;
73  break;
74 
75  // Message header processing
76  //
77  case 2:
78  _step++;
79  _msg_id = data;
80  _ck_b = _ck_a = data; // reset the checksum accumulators
81  break;
82  case 3:
83  _step++;
84  _ck_b += (_ck_a += data); // checksum byte
85  _payload_length = data; // payload length low byte
86  break;
87  case 4:
88  _step++;
89  _ck_b += (_ck_a += data); // checksum byte
90  _payload_length += (uint16_t)(data<<8);
91  _payload_counter = 0; // prepare to receive payload
92  break;
93 
94  // Receive message data
95  //
96  case 5:
97  _ck_b += (_ck_a += data); // checksum byte
98  if (_payload_counter < sizeof(_buffer)) {
99  _buffer[_payload_counter] = data;
100  }
101  if (++_payload_counter == _payload_length)
102  _step++;
103  break;
104 
105  // Checksum and message processing
106  //
107  case 6:
108  _step++;
109  if (_ck_a != data) {
110  Debug("bad cka %x should be %x", data, _ck_a);
111  _step = 0;
112  goto reset;
113  }
114  break;
115  case 7:
116  _step = 0;
117  if (_ck_b != data) {
118  Debug("bad ckb %x should be %x", data, _ck_b);
119  break; // bad checksum
120  }
121 
122  if (_parse_gps()) {
123  parsed = true;
124  }
125  break;
126  }
127  }
128  return parsed;
129 }

131 bool
132 AP_GPS_ERB::_parse_gps(void)
133 {
134  switch (_msg_id) {
135  case MSG_VER:
136  Debug("Version of ERB protocol %u.%u.%u",
137  _buffer.ver.ver_high,
138  _buffer.ver.ver_medium,
139  _buffer.ver.ver_low);
140  break;
141  case MSG_POS:
142  Debug("Message POS");
143  _last_pos_time = _buffer.pos.time;
144  state.location.lng = (int32_t)(_buffer.pos.longitude * (double)1e7);
145  state.location.lat = (int32_t)(_buffer.pos.latitude * (double)1e7);
146  state.location.alt = (int32_t)(_buffer.pos.altitude_msl * 100);
147  state.status = next_fix;
148  _new_position = true;
149  state.horizontal_accuracy = _buffer.pos.horizontal_accuracy * 1.0e-3f;
150  state.vertical_accuracy = _buffer.pos.vertical_accuracy * 1.0e-3f;
151  state.have_horizontal_accuracy = true;
152  state.have_vertical_accuracy = true;
153  break;
154  case MSG_STAT:
155  Debug("Message STAT fix_status=%u fix_type=%u",
156  _buffer.stat.fix_status,
157  _buffer.stat.fix_type);
158  if (_buffer.stat.fix_status & STAT_FIX_VALID) {
159  if (_buffer.stat.fix_type == AP_GPS_ERB::FIX_FIX) {
160  next_fix = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
161  } else if (_buffer.stat.fix_type == AP_GPS_ERB::FIX_FLOAT) {
162  next_fix = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
163  } else if (_buffer.stat.fix_type == AP_GPS_ERB::FIX_SINGLE) {
164  next_fix = AP_GPS::GPS_OK_FIX_3D;
165  } else {
166  next_fix = AP_GPS::NO_FIX;
167  state.status = AP_GPS::NO_FIX;
168  }
169  } else {
170  next_fix = AP_GPS::NO_FIX;
171  state.status = AP_GPS::NO_FIX;
172  }
173  state.num_sats = _buffer.stat.satellites;
174  if (next_fix >= AP_GPS::GPS_OK_FIX_3D) {
175  // use the uart receive time to make packet timestamps more accurate
176  set_uart_timestamp(_payload_length + sizeof(erb_header) + 2);
177  state.last_gps_time_ms = AP_HAL::millis();
178  state.time_week_ms = _buffer.stat.time;
179  state.time_week = _buffer.stat.week;
180  }
181  break;
182  case MSG_DOPS:
183  Debug("Message DOPS");
184  state.hdop = _buffer.dops.hDOP;
185  state.vdop = _buffer.dops.vDOP;
186  break;
187  case MSG_VEL:
188  Debug("Message VEL");
189  _last_vel_time = _buffer.vel.time;
190  state.ground_speed = _buffer.vel.speed_2d * 0.01f; // m/s
191  // Heading 2D deg * 100000 rescaled to deg * 100
192  state.ground_course = wrap_360(_buffer.vel.heading_2d * 1.0e-5f);
193  state.have_vertical_velocity = true;
194  state.velocity.x = _buffer.vel.vel_north * 0.01f;
195  state.velocity.y = _buffer.vel.vel_east * 0.01f;
196  state.velocity.z = _buffer.vel.vel_down * 0.01f;
197  state.have_speed_accuracy = true;
198  state.speed_accuracy = _buffer.vel.speed_accuracy * 0.01f;
199  _new_speed = true;
200  break;
201  case MSG_RTK:
202  Debug("Message RTK");
203  state.rtk_baseline_coords_type = RTK_BASELINE_COORDINATE_SYSTEM_NED;
204  state.rtk_num_sats = _buffer.rtk.base_num_sats;
205  if (_buffer.rtk.age_cs == 0xFFFF) {
206  state.rtk_age_ms = 0xFFFFFFFF;
207  } else {
208  state.rtk_age_ms = _buffer.rtk.age_cs * 10;
209  }
210  state.rtk_baseline_x_mm = _buffer.rtk.baseline_N_mm;
211  state.rtk_baseline_y_mm = _buffer.rtk.baseline_E_mm;
212  state.rtk_baseline_z_mm = _buffer.rtk.baseline_D_mm;
213  state.rtk_accuracy = _buffer.rtk.ar_ratio;
214 
215  state.rtk_week_number = _buffer.rtk.base_week_number;
216  state.rtk_time_week_ms = _buffer.rtk.base_time_week_ms;
217  break;
218  default:
219  Debug("Unexpected message 0x%02x", (unsigned)_msg_id);
220  return false;
221  }
222  // we only return true when we get new position and speed data
223  // this ensures we don't use stale data
224  if (_new_position && _new_speed && _last_vel_time == _last_pos_time) {
225  _new_speed = _new_position = false;
226  _fix_count++;
227  return true;
228  }
229  return false;
230 }

232 /*
233  detect a ERB GPS. Adds one byte, and returns true if the stream
234  matches a ERB
235  */
236 bool
237 AP_GPS_ERB::_detect(struct ERB_detect_state &state, uint8_t data)
238 {
239 reset:
240  switch (state.step) {
241  case 1:
242  if (PREAMBLE2 == data) {
243  state.step++;
244  break;
245  }
246  state.step = 0;
247  FALLTHROUGH;
248  case 0:
249  if (PREAMBLE1 == data)
250  state.step++;
251  break;
252  case 2:
253  state.step++;
254  state.ck_b = state.ck_a = data;
255  break;
256  case 3:
257  state.step++;
258  state.ck_b += (state.ck_a += data);
259  state.payload_length = data;
260  break;
261  case 4:
262  state.step++;
263  state.ck_b += (state.ck_a += data);
264  state.payload_counter = 0;
265  break;
266  case 5:
267  state.ck_b += (state.ck_a += data);
268  if (++state.payload_counter == state.payload_length)
269  state.step++;
270  break;
271  case 6:
272  state.step++;
273  if (state.ck_a != data) {
274  state.step = 0;
275  goto reset;
276  }
277  break;
278  case 7:
279  state.step = 0;
280  if (state.ck_b == data) {
281  return true;
282  } else {
283  goto reset;
284  }
285  }
286  return false;
287 }
