80 void AC_Avoid::adjust_velocity(float kP, float accel_cmss, Vector2f &desired_vel_cms, float dt)
81 {
82  // exit immediately if disabled
83  if (_enabled == AC_AVOID_DISABLED) {
84  return;
85  }
86 
87  // limit acceleration
88  const float accel_cmss_limited = MIN(accel_cmss, AC_AVOID_ACCEL_CMSS_MAX);
89 
90  if ((_enabled & AC_AVOID_STOP_AT_FENCE) > 0) {
91  adjust_velocity_circle_fence(kP, accel_cmss_limited, desired_vel_cms, dt);
92  adjust_velocity_polygon_fence(kP, accel_cmss_limited, desired_vel_cms, dt);
93  }
94 
95  if ((_enabled & AC_AVOID_STOP_AT_BEACON_FENCE) > 0) {
96  adjust_velocity_beacon_fence(kP, accel_cmss_limited, desired_vel_cms, dt);
97  }
98 
99  if ((_enabled & AC_AVOID_USE_PROXIMITY_SENSOR) > 0 && _proximity_enabled) {
100  adjust_velocity_proximity(kP, accel_cmss_limited, desired_vel_cms, dt);
101  }
102 }

113 // adjust desired horizontal speed so that the vehicle stops before the fence or object
114 // accel (maximum acceleration/deceleration) is in m/s/s
115 // heading is in radians
116 // speed is in m/s
117 // kP should be zero for linear response, non-zero for non-linear response
118 void AC_Avoid::adjust_speed(float kP, float accel, float heading, float &speed, float dt)
119 {
120  // convert heading and speed into velocity vector
121  Vector2f vel_xy;
122  vel_xy.x = cosf(heading) * speed * 100.0f;
123  vel_xy.y = sinf(heading) * speed * 100.0f;
124  adjust_velocity(kP, accel * 100.0f, vel_xy, dt);
125 
126  // adjust speed towards zero
127  if (is_negative(speed)) {
128  speed = -vel_xy.length() * 0.01f;
129  } else {
130  speed = vel_xy.length() * 0.01f;
131  }
132 }

134 // adjust vertical climb rate so vehicle does not break the vertical fence
135 void AC_Avoid::adjust_velocity_z(float kP, float accel_cmss, float& climb_rate_cms, float dt)
136 {
137  // exit immediately if disabled
138  if (_enabled == AC_AVOID_DISABLED) {
139  return;
140  }
141 
142  // do not adjust climb_rate if level or descending
143  if (climb_rate_cms <= 0.0f) {
144  return;
145  }
146 
147  // limit acceleration
148  const float accel_cmss_limited = MIN(accel_cmss, AC_AVOID_ACCEL_CMSS_MAX);
149 
150  bool limit_alt = false;
151  float alt_diff = 0.0f; // distance from altitude limit to vehicle in metres (positive means vehicle is below limit)
152 
153  const AP_AHRS &_ahrs = AP::ahrs();
154 
155  // calculate distance below fence
156  AC_Fence *fence = AP::fence();
157  if ((_enabled & AC_AVOID_STOP_AT_FENCE) > 0 && fence && (fence->get_enabled_fences() & AC_FENCE_TYPE_ALT_MAX) > 0) {
158  // calculate distance from vehicle to safe altitude
159  float veh_alt;
160  _ahrs.get_relative_position_D_home(veh_alt);
161  // _fence.get_safe_alt_max() is UP, veh_alt is DOWN:
162  alt_diff = fence->get_safe_alt_max() + veh_alt;
163  limit_alt = true;
164  }
165 
166  // calculate distance to (e.g.) optical flow altitude limit
167  // AHRS values are always in metres
168  float alt_limit;
169  float curr_alt;
170  if (_ahrs.get_hgt_ctrl_limit(alt_limit) &&
171  _ahrs.get_relative_position_D_origin(curr_alt)) {
172  // alt_limit is UP, curr_alt is DOWN:
173  const float ctrl_alt_diff = alt_limit + curr_alt;
174  if (!limit_alt || ctrl_alt_diff < alt_diff) {
175  alt_diff = ctrl_alt_diff;
176  limit_alt = true;
177  }
178  }
179 
180  // get distance from proximity sensor
181  float proximity_alt_diff;
182  AP_Proximity *proximity = AP::proximity();
183  if (proximity && proximity->get_upward_distance(proximity_alt_diff)) {
184  proximity_alt_diff -= _margin;
185  if (!limit_alt || proximity_alt_diff < alt_diff) {
186  alt_diff = proximity_alt_diff;
187  limit_alt = true;
188  }
189  }
190 
191  // limit climb rate
192  if (limit_alt) {
193  // do not allow climbing if we've breached the safe altitude
194  if (alt_diff <= 0.0f) {
195  climb_rate_cms = MIN(climb_rate_cms, 0.0f);
196  return;
197  }
198 
199  // limit climb rate
200  const float max_speed = get_max_speed(kP, accel_cmss_limited, alt_diff*100.0f, dt);
201  climb_rate_cms = MIN(max_speed, climb_rate_cms);
202  }
203 }

205 // adjust roll-pitch to push vehicle away from objects
206 // roll and pitch value are in centi-degrees
207 void AC_Avoid::adjust_roll_pitch(float &roll, float &pitch, float veh_angle_max)
208 {
209  // exit immediately if proximity based avoidance is disabled
210  if ((_enabled & AC_AVOID_USE_PROXIMITY_SENSOR) == 0 || !_proximity_enabled) {
211  return;
212  }
213 
214  // exit immediately if angle max is zero
215  if (_angle_max <= 0.0f || veh_angle_max <= 0.0f) {
216  return;
217  }
218 
219  float roll_positive = 0.0f; // maximum positive roll value
220  float roll_negative = 0.0f; // minimum negative roll value
221  float pitch_positive = 0.0f; // maximum positive pitch value
222  float pitch_negative = 0.0f; // minimum negative pitch value
223 
224  // get maximum positive and negative roll and pitch percentages from proximity sensor
225  get_proximity_roll_pitch_pct(roll_positive, roll_negative, pitch_positive, pitch_negative);
226 
227  // add maximum positive and negative percentages together for roll and pitch, convert to centi-degrees
228  Vector2f rp_out((roll_positive + roll_negative) * 4500.0f, (pitch_positive + pitch_negative) * 4500.0f);
229 
230  // apply avoidance angular limits
231  // the object avoidance lean angle is never more than 75% of the total angle-limit to allow the pilot to override
232  const float angle_limit = constrain_float(_angle_max, 0.0f, veh_angle_max * AC_AVOID_ANGLE_MAX_PERCENT);
233  float vec_len = rp_out.length();
234  if (vec_len > angle_limit) {
235  rp_out *= (angle_limit / vec_len);
236  }
237 
238  // add passed in roll, pitch angles
239  rp_out.x += roll;
240  rp_out.y += pitch;
241 
242  // apply total angular limits
243  vec_len = rp_out.length();
244  if (vec_len > veh_angle_max) {
245  rp_out *= (veh_angle_max / vec_len);
246  }
247 
248  // return adjusted roll, pitch
249  roll = rp_out.x;
250  pitch = rp_out.y;
251 }

253 /*
254  * Limits the component of desired_vel_cms in the direction of the unit vector
255  * limit_direction to be at most the maximum speed permitted by the limit_distance_cm.
256  *
257  * Uses velocity adjustment idea from Randy's second email on this thread:
258  * https://groups.google.com/forum/#!searchin/drones-discuss/obstacle/drones-discuss/QwUXz__WuqY/qo3G8iTLSJAJ
259  */
260 void AC_Avoid::limit_velocity(float kP, float accel_cmss, Vector2f &desired_vel_cms, const Vector2f& limit_direction, float limit_distance_cm, float dt) const
261 {
262  const float max_speed = get_max_speed(kP, accel_cmss, limit_distance_cm, dt);
263  // project onto limit direction
264  const float speed = desired_vel_cms * limit_direction;
265  if (speed > max_speed) {
266  // subtract difference between desired speed and maximum acceptable speed
267  desired_vel_cms += limit_direction*(max_speed - speed);
268  }
269 }

272  * Computes the speed such that the stopping distance
273  * of the vehicle will be exactly the input distance.
274  */
275 float AC_Avoid::get_max_speed(float kP, float accel_cmss, float distance_cm, float dt) const
276 {
277  if (is_zero(kP)) {
278  return safe_sqrt(2.0f * distance_cm * accel_cmss);
279  } else {
280  return AC_AttitudeControl::sqrt_controller(distance_cm, kP, accel_cmss, dt);
281  }

284 /*
285  * Adjusts the desired velocity for the circular fence.
286  */
287 void AC_Avoid::adjust_velocity_circle_fence(float kP, float accel_cmss, Vector2f &desired_vel_cms, float dt)
288 {
289  AC_Fence *fence = AP::fence();
290  if (fence == nullptr) {
291  return;
292  }
293 
294  AC_Fence &_fence = *fence;
295 
296  // exit if circular fence is not enabled
297  if ((_fence.get_enabled_fences() & AC_FENCE_TYPE_CIRCLE) == 0) {
298  return;
299  }
300 
301  // exit if the circular fence has already been breached
302  if ((_fence.get_breaches() & AC_FENCE_TYPE_CIRCLE) != 0) {
303  return;
304  }
305 
306  // get desired speed
307  const float desired_speed = desired_vel_cms.length();
308  if (is_zero(desired_speed)) {
309  // no avoidance necessary when desired speed is zero
310  return;
311  }
312 
313  const AP_AHRS &_ahrs = AP::ahrs();
314 
315  // get position as a 2D offset from ahrs home
316  Vector2f position_xy;
317  if (!_ahrs.get_relative_position_NE_home(position_xy)) {
318  // we have no idea where we are....
319  return;
320  }
321  position_xy *= 100.0f; // m -> cm
322 
323  // get the fence radius in cm
324  const float fence_radius = _fence.get_radius() * 100.0f;
325  // get the margin to the fence in cm
326  const float margin_cm = _fence.get_margin() * 100.0f;
327 
328  // get vehicle distance from home
329  const float dist_from_home = position_xy.length();
330  if (dist_from_home > fence_radius) {
331  // outside of circular fence, no velocity adjustments
332  return;
333  }
334 
335  // vehicle is inside the circular fence
336  if ((AC_Avoid::BehaviourType)_behavior.get() == BEHAVIOR_SLIDE) {
337  // implement sliding behaviour
338  const Vector2f stopping_point = position_xy + desired_vel_cms*(get_stopping_distance(kP, accel_cmss, desired_speed)/desired_speed);
339  const float stopping_point_dist_from_home = stopping_point.length();
340  if (stopping_point_dist_from_home <= fence_radius - margin_cm) {
341  // stopping before before fence so no need to adjust
342  return;
343  }
344  // unsafe desired velocity - will not be able to stop before reaching margin from fence
345  // Project stopping point radially onto fence boundary
346  // Adjusted velocity will point towards this projected point at a safe speed
347  const Vector2f target_offset = stopping_point * ((fence_radius - margin_cm) / stopping_point_dist_from_home);
348  const Vector2f target_direction = target_offset - position_xy;
349  const float distance_to_target = target_direction.length();
350  const float max_speed = get_max_speed(kP, accel_cmss, distance_to_target, dt);
351  desired_vel_cms = target_direction * (MIN(desired_speed,max_speed) / distance_to_target);
352  } else {
353  // implement stopping behaviour
354  // calculate stopping point plus a margin so we look forward far enough to intersect with circular fence
355  const Vector2f stopping_point_plus_margin = position_xy + desired_vel_cms*((2.0f + margin_cm + get_stopping_distance(kP, accel_cmss, desired_speed))/desired_speed);
356  const float stopping_point_plus_margin_dist_from_home = stopping_point_plus_margin.length();
357  if (dist_from_home >= fence_radius - margin_cm) {
358  // if vehicle has already breached margin around fence
359  // if stopping point is even further from home (i.e. in wrong direction) then adjust speed to zero
360  // otherwise user is backing away from fence so do not apply limits
361  if (stopping_point_plus_margin_dist_from_home >= dist_from_home) {
362  desired_vel_cms.zero();
363  }
364  } else {
365  // shorten vector without adjusting its direction
366  Vector2f intersection;
367  if (Vector2f::circle_segment_intersection(position_xy, stopping_point_plus_margin, Vector2f(0.0f,0.0f), fence_radius - margin_cm, intersection)) {
368  const float distance_to_target = MAX((intersection - position_xy).length() - margin_cm, 0.0f);
369  const float max_speed = get_max_speed(kP, accel_cmss, distance_to_target, dt);
370  if (max_speed < desired_speed) {
371  desired_vel_cms *= MAX(max_speed, 0.0f) / desired_speed;
372  }
373  }
374  }
375  }
376 }

458 /*
459  * Adjusts the desired velocity for the polygon fence.
460  */
461 void AC_Avoid::adjust_velocity_polygon(float kP, float accel_cmss, Vector2f &desired_vel_cms, const Vector2f* boundary, uint16_t num_points, bool earth_frame, float margin, float dt)
462 {
463  // exit if there are no points
464  if (boundary == nullptr || num_points == 0) {
465  return;
466  }
467 
468  // exit immediately if no desired velocity
469  if (desired_vel_cms.is_zero()) {
470  return;
471  }
472 
473  const AP_AHRS &_ahrs = AP::ahrs();
474 
475  // do not adjust velocity if vehicle is outside the polygon fence
476  Vector2f position_xy;
477  if (earth_frame) {
478  if (!_ahrs.get_relative_position_NE_origin(position_xy)) {
479  // boundary is in earth frame but we have no idea
480  // where we are
481  return;
482  }
483  position_xy = position_xy * 100.0f; // m to cm
484  }
485 
486  if (Polygon_outside(position_xy, boundary, num_points)) {
487  return;
488  }
489 
490  // Safe_vel will be adjusted to remain within fence.
491  // We need a separate vector in case adjustment fails,
492  // e.g. if we are exactly on the boundary.
493  Vector2f safe_vel(desired_vel_cms);
494 
495  // if boundary points are in body-frame, rotate velocity vector from earth frame to body-frame
496  if (!earth_frame) {
497  safe_vel.x = desired_vel_cms.y * _ahrs.sin_yaw() + desired_vel_cms.x * _ahrs.cos_yaw(); // right
498  safe_vel.y = desired_vel_cms.y * _ahrs.cos_yaw() - desired_vel_cms.x * _ahrs.sin_yaw(); // forward
499  }
500 
501  // calc margin in cm
502  const float margin_cm = MAX(margin * 100.0f, 0.0f);
503 
504  // for stopping
505  const float speed = safe_vel.length();
506  const Vector2f stopping_point_plus_margin = position_xy + safe_vel*((2.0f + margin_cm + get_stopping_distance(kP, accel_cmss, speed))/speed);
507 
508  for (uint16_t i=0; i<num_points; i++) {
509  uint16_t j = i+1;
510  if (j >= num_points) {
511  j = 0;
512  }
513  // end points of current edge
514  Vector2f start = boundary[j];
515  Vector2f end = boundary[i];
516  if ((AC_Avoid::BehaviourType)_behavior.get() == BEHAVIOR_SLIDE) {
517  // vector from current position to closest point on current edge
518  Vector2f limit_direction = Vector2f::closest_point(position_xy, start, end) - position_xy;
519  // distance to closest point
520  const float limit_distance_cm = limit_direction.length();
521  if (!is_zero(limit_distance_cm)) {
522  // We are strictly inside the given edge.
523  // Adjust velocity to not violate this edge.
524  limit_direction /= limit_distance_cm;
525  limit_velocity(kP, accel_cmss, safe_vel, limit_direction, MAX(limit_distance_cm - margin_cm, 0.0f), dt);
526  } else {
527  // We are exactly on the edge - treat this as a fence breach.
528  // i.e. do not adjust velocity.
529  return;
530  }
531  } else {
532  // find intersection with line segment
533  Vector2f intersection;
534  if (Vector2f::segment_intersection(position_xy, stopping_point_plus_margin, start, end, intersection)) {
535  // vector from current position to point on current edge
536  Vector2f limit_direction = intersection - position_xy;
537  const float limit_distance_cm = limit_direction.length();
538  if (!is_zero(limit_distance_cm)) {
539  if (limit_distance_cm <= margin_cm) {
540  // we are within the margin so stop vehicle
541  safe_vel.zero();
542  } else {
543  // vehicle inside the given edge, adjust velocity to not violate this edge
544  limit_direction /= limit_distance_cm;
545  limit_velocity(kP, accel_cmss, safe_vel, limit_direction, MAX(limit_distance_cm - margin_cm, 0.0f), dt);
546  }
547  } else {
548  // We are exactly on the edge - treat this as a fence breach.
549  // i.e. do not adjust velocity.
550  return;
551  }
552  }
553  }
554  }
555 
556  // set modified desired velocity vector
557  if (earth_frame) {
558  desired_vel_cms = safe_vel;
559  } else {
560  // if points were in body-frame, rotate resulting vector back to earth-frame
561  desired_vel_cms.x = safe_vel.x * _ahrs.cos_yaw() - safe_vel.y * _ahrs.sin_yaw();
562  desired_vel_cms.y = safe_vel.x * _ahrs.sin_yaw() + safe_vel.y * _ahrs.cos_yaw();
563  }
564 }

566 /*
567  * Computes distance required to stop, given current speed.
568  *
569  * Implementation copied from AC_PosControl.
570  */
571 float AC_Avoid::get_stopping_distance(float kP, float accel_cmss, float speed_cms) const
572 {
573  // avoid divide by zero by using current position if the velocity is below 10cm/s, kP is very low or acceleration is zero
574  if (accel_cmss <= 0.0f || is_zero(speed_cms)) {
575  return 0.0f;
576  }
577 
578  // handle linear deceleration
579  if (kP <= 0.0f) {
580  return 0.5f * sq(speed_cms) / accel_cmss;
581  }
582 
583  // calculate distance within which we can stop
584  // accel_cmss/kP is the point at which velocity switches from linear to sqrt
585  if (speed_cms < accel_cmss/kP) {
586  return speed_cms/kP;
587  } else {
588  // accel_cmss/(2.0f*kP*kP) is the distance at which we switch from linear to sqrt response
589  return accel_cmss/(2.0f*kP*kP) + (speed_cms*speed_cms)/(2.0f*accel_cmss);

594 float AC_Avoid::distance_to_lean_pct(float dist_m)

598  return 0.0f;
599  }
600  // inverted but linear response
601  return 1.0f - (dist_m / _dist_max);

604 // returns the maximum positive and negative roll and pitch percentages (in -1 ~ +1 range) based on the proximity sensor
605 void AC_Avoid::get_proximity_roll_pitch_pct(float &roll_positive, float &roll_negative, float &pitch_positive, float &pitch_negative)
606 {
607  AP_Proximity *proximity = AP::proximity();
608  if (proximity == nullptr) {
609  return;
610  }
611  AP_Proximity &_proximity = *proximity;
612 
613  // exit immediately if proximity sensor is not present
614  if (_proximity.get_status() != AP_Proximity::Proximity_Good) {
615  return;
616  }
617 
618  const uint8_t obj_count = _proximity.get_object_count();
619 
620  // if no objects return
621  if (obj_count == 0) {
622  return;
623  }
624 
625  // calculate maximum roll, pitch values from objects
626  for (uint8_t i=0; i<obj_count; i++) {
627  float ang_deg, dist_m;
628  if (_proximity.get_object_angle_and_distance(i, ang_deg, dist_m)) {
629  if (dist_m < _dist_max) {
630  // convert distance to lean angle (in 0 to 1 range)
631  const float lean_pct = distance_to_lean_pct(dist_m);
632  // convert angle to roll and pitch lean percentages
633  const float angle_rad = radians(ang_deg);
634  const float roll_pct = -sinf(angle_rad) * lean_pct;
635  const float pitch_pct = cosf(angle_rad) * lean_pct;
636  // update roll, pitch maximums
637  if (roll_pct > 0.0f) {
638  roll_positive = MAX(roll_positive, roll_pct);
639  } else if (roll_pct < 0.0f) {
640  roll_negative = MIN(roll_negative, roll_pct);
641  }
642  if (pitch_pct > 0.0f) {
643  pitch_positive = MAX(pitch_positive, pitch_pct);
644  } else if (pitch_pct < 0.0f) {
645  pitch_negative = MIN(pitch_negative, pitch_pct);
646  }
647  }
648  }
649  }
