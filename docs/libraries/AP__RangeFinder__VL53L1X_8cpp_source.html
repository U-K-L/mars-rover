40 /*
41  detect if a VL53L1X rangefinder is connected. We'll detect by
42  trying to take a reading on I2C. If we get a result the sensor is
43  there.

45 AP_RangeFinder_Backend *AP_RangeFinder_VL53L1X::detect(RangeFinder::RangeFinder_State &_state, AP_RangeFinder_Params &_params, AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev)
46 {
47  if (!dev) {
48  return nullptr;
49  }
50 
51  AP_RangeFinder_VL53L1X *sensor
52  = new AP_RangeFinder_VL53L1X(_state, _params, std::move(dev));
53 
54  if (!sensor) {
55  delete sensor;
56  return nullptr;
57  }
58 
59  sensor->dev->get_semaphore()->take_blocking();
60 
61  if (!sensor->check_id() || !sensor->init()) {
62  sensor->dev->get_semaphore()->give();
63  delete sensor;
64  return nullptr;
65  }
66 
67  sensor->dev->get_semaphore()->give();
68 
69  return sensor;
70 }

92 bool AP_RangeFinder_VL53L1X::init()
93 {
94  uint8_t pad_i2c_hv_extsup_config = 0;
95  uint16_t mm_config_outer_offset_mm = 0;

97  read_register(PAD_I2C_HV__EXTSUP_CONFIG, pad_i2c_hv_extsup_config) &&
98  write_register(PAD_I2C_HV__EXTSUP_CONFIG,
99  pad_i2c_hv_extsup_config | 0x01) &&
100 
101  // store oscillator info for later use
102  read_register16(OSC_MEASURED__FAST_OSC__FREQUENCY, fast_osc_frequency) &&
103  read_register16(RESULT__OSC_CALIBRATE_VAL, osc_calibrate_val) &&
104 
105  // static config
106  write_register16(DSS_CONFIG__TARGET_TOTAL_RATE_MCPS, TargetRate) && // should already be this value after reset
107  write_register(GPIO__TIO_HV_STATUS, 0x02) &&
108  write_register(SIGMA_ESTIMATOR__EFFECTIVE_PULSE_WIDTH_NS, 8) && // tuning parm default
109  write_register(SIGMA_ESTIMATOR__EFFECTIVE_AMBIENT_WIDTH_NS, 16) && // tuning parm default
110  write_register(ALGO__CROSSTALK_COMPENSATION_VALID_HEIGHT_MM, 0x01) &&
111  write_register(ALGO__RANGE_IGNORE_VALID_HEIGHT_MM, 0xFF) &&
112  write_register(ALGO__RANGE_MIN_CLIP, 0) && // tuning parm default
113  write_register(ALGO__CONSISTENCY_CHECK__TOLERANCE, 2) && // tuning parm default
114 
115  // general config
116  write_register16(SYSTEM__THRESH_RATE_HIGH, 0x0000) &&
117  write_register16(SYSTEM__THRESH_RATE_LOW, 0x0000) &&
118  write_register(DSS_CONFIG__APERTURE_ATTENUATION, 0x38) &&
119 
120  // timing config
121  write_register16(RANGE_CONFIG__SIGMA_THRESH, 360) && // tuning parm default
122  write_register16(RANGE_CONFIG__MIN_COUNT_RATE_RTN_LIMIT_MCPS, 192) && // tuning parm default
123 
124  // dynamic config
125  write_register(SYSTEM__GROUPED_PARAMETER_HOLD_0, 0x01) &&
126  write_register(SYSTEM__GROUPED_PARAMETER_HOLD_1, 0x01) &&
127  write_register(SD_CONFIG__QUANTIFIER, 2) && // tuning parm default

136  // from VL53L1_config_low_power_auto_mode
137  write_register(SYSTEM__SEQUENCE_CONFIG, 0x8B) && // VHV, PHASECAL, DSS1, RANGE

141  setDistanceMode(Long) &&
142  setMeasurementTimingBudget(40000) &&
143  // the API triggers this change in VL53L1_init_and_start_range() once a
144  // measurement is started; assumes MM1 and MM2 are disabled
145  write_register16(ALGO__PART_TO_PART_RANGE_OFFSET_MM, mm_config_outer_offset_mm * 4) &&
146  // set continuous mode
147  startContinuous(50)
148  )) {
149  return false;
150  }
151 
152  // call timer() every 50ms. We expect new data to be available every 50ms
153  dev->register_periodic_callback(50000,
154  FUNCTOR_BIND_MEMBER(&AP_RangeFinder_VL53L1X::timer, void));
155 
156  return true;

159 // set distance mode to Short, Medium, or Long
160 // based on VL53L1_SetDistanceMode()
161 bool AP_RangeFinder_VL53L1X::setDistanceMode(DistanceMode distance_mode)
162 {
163  // save existing timing budget
164  uint32_t budget_us = 0;
165  if (!getMeasurementTimingBudget(budget_us)) {
166  return false;
167  }
168 
169  switch (distance_mode) {
170  case Short:
171  // from VL53L1_preset_mode_standard_ranging_short_range()
172 
173  if (!(// timing config
174  write_register(RANGE_CONFIG__VCSEL_PERIOD_A, 0x07) &&
175  write_register(RANGE_CONFIG__VCSEL_PERIOD_B, 0x05) &&
176  write_register(RANGE_CONFIG__VALID_PHASE_HIGH, 0x38) &&
177 
178  // dynamic config
179  write_register(SD_CONFIG__WOI_SD0, 0x07) &&
180  write_register(SD_CONFIG__WOI_SD1, 0x05) &&
181  write_register(SD_CONFIG__INITIAL_PHASE_SD0, 6) && // tuning parm default
182  write_register(SD_CONFIG__INITIAL_PHASE_SD1, 6))) { // tuning parm default
183  return false;
184  }
185 
186  break;
187 
188  case Medium:
189  // from VL53L1_preset_mode_standard_ranging()
190 
191  if (!(// timing config
192  write_register(RANGE_CONFIG__VCSEL_PERIOD_A, 0x0B) &&
193  write_register(RANGE_CONFIG__VCSEL_PERIOD_B, 0x09) &&
194  write_register(RANGE_CONFIG__VALID_PHASE_HIGH, 0x78) &&
195 
196  // dynamic config
197  write_register(SD_CONFIG__WOI_SD0, 0x0B) &&
198  write_register(SD_CONFIG__WOI_SD1, 0x09) &&
199  write_register(SD_CONFIG__INITIAL_PHASE_SD0, 10) && // tuning parm default
200  write_register(SD_CONFIG__INITIAL_PHASE_SD1, 10))) { // tuning parm default
201  return false;
202  }
203 
204  break;
205 
206  case Long:
207  // from VL53L1_preset_mode_standard_ranging_long_range()
208 
209  if (!(// timing config
210  write_register(RANGE_CONFIG__VCSEL_PERIOD_A, 0x0F) &&
211  write_register(RANGE_CONFIG__VCSEL_PERIOD_B, 0x0D) &&
212  write_register(RANGE_CONFIG__VALID_PHASE_HIGH, 0xB8) &&
213 
214  // dynamic config
215  write_register(SD_CONFIG__WOI_SD0, 0x0F) &&
216  write_register(SD_CONFIG__WOI_SD1, 0x0D) &&
217  write_register(SD_CONFIG__INITIAL_PHASE_SD0, 14) && // tuning parm default
218  write_register(SD_CONFIG__INITIAL_PHASE_SD1, 14))) { // tuning parm default
219  return false;
220  }
221 
222  break;
223 
224  default:
225  // unrecognized mode - do nothing
226  return false;
227  }
228 
229  // reapply timing budget
230  return setMeasurementTimingBudget(budget_us);
231 }

233 // Set the measurement timing budget in microseconds, which is the time allowed
234 // for one measurement. A longer timing budget allows for more accurate measurements.
235 // based on VL53L1_SetMeasurementTimingBudgetMicroSeconds()
236 bool AP_RangeFinder_VL53L1X::setMeasurementTimingBudget(uint32_t budget_us)
237 {
238  // assumes PresetMode is LOWPOWER_AUTONOMOUS
239  if (budget_us <= TimingGuard) {
240  return false;
241  }
242 
243  uint32_t range_config_timeout_us = budget_us - TimingGuard;
244  if (range_config_timeout_us > 1100000) {
245  return false; // FDA_MAX_TIMING_BUDGET_US * 2
246  }
247 
248  range_config_timeout_us /= 2;
249 
250  // VL53L1_calc_timeout_register_values() begin
251 
252  uint8_t range_config_vcsel_period = 0;
253  if(!read_register(RANGE_CONFIG__VCSEL_PERIOD_A, range_config_vcsel_period)) {
254  return false;
255  }
256 
257  // "Update Macro Period for Range A VCSEL Period"
258  uint32_t macro_period_us = calcMacroPeriod(range_config_vcsel_period);
259 
260  // "Update Phase timeout - uses Timing A"
261  // Timeout of 1000 is tuning parm default (TIMED_PHASECAL_CONFIG_TIMEOUT_US_DEFAULT)
262  // via VL53L1_get_preset_mode_timing_cfg().
263  uint32_t phasecal_timeout_mclks = timeoutMicrosecondsToMclks(1000, macro_period_us);
264  if (phasecal_timeout_mclks > 0xFF) {
265  phasecal_timeout_mclks = 0xFF;
266  }
267 
268  if (!( write_register(PHASECAL_CONFIG__TIMEOUT_MACROP, phasecal_timeout_mclks) &&
269 
270  // "Update MM Timing A timeout"
271  // Timeout of 1 is tuning parm default (LOWPOWERAUTO_MM_CONFIG_TIMEOUT_US_DEFAULT)
272  // via VL53L1_get_preset_mode_timing_cfg(). With the API, the register
273  // actually ends up with a slightly different value because it gets assigned,
274  // retrieved, recalculated with a different macro period, and reassigned,
275  // but it probably doesn't matter because it seems like the MM ("mode
276  // mitigation"?) sequence steps are disabled in low power auto mode anyway.
277  write_register16(MM_CONFIG__TIMEOUT_MACROP_A, encodeTimeout(
278  timeoutMicrosecondsToMclks(1, macro_period_us))) &&
279 
280  // "Update Range Timing A timeout"
281  write_register16(RANGE_CONFIG__TIMEOUT_MACROP_A, encodeTimeout(
282  timeoutMicrosecondsToMclks(range_config_timeout_us, macro_period_us))) &&
283 
284  // "Update Macro Period for Range B VCSEL Period"
285  read_register(RANGE_CONFIG__VCSEL_PERIOD_B, range_config_vcsel_period)
286  )) {
287  return false;
288  }
289 
290  // "Update Macro Period for Range B VCSEL Period"
291  macro_period_us = calcMacroPeriod(range_config_vcsel_period);
292 
293  // "Update MM Timing B timeout"
294  // (See earlier comment about MM Timing A timeout.)
295  return write_register16(MM_CONFIG__TIMEOUT_MACROP_B, encodeTimeout(
296  timeoutMicrosecondsToMclks(1, macro_period_us))) &&
297 
298  // "Update Range Timing B timeout"
299  write_register16(RANGE_CONFIG__TIMEOUT_MACROP_B, encodeTimeout(
300  timeoutMicrosecondsToMclks(range_config_timeout_us, macro_period_us)));
301 }

303 // Get the measurement timing budget in microseconds
304 // based on VL53L1_SetMeasurementTimingBudgetMicroSeconds()
305 bool AP_RangeFinder_VL53L1X::getMeasurementTimingBudget(uint32_t &budget)
306 {
307  // assumes PresetMode is LOWPOWER_AUTONOMOUS and these sequence steps are
308  // enabled: VHV, PHASECAL, DSS1, RANGE
309 
310  // "Update Macro Period for Range A VCSEL Period"
311  uint8_t range_config_vcsel_period_a = 0;
312  if (!read_register(RANGE_CONFIG__VCSEL_PERIOD_A, range_config_vcsel_period_a)) {
313  return false;
314  }
315 
316  uint32_t macro_period_us = calcMacroPeriod(range_config_vcsel_period_a);
317 
318  uint16_t timeout_macrop_a = 0;
319  if (!read_register16(RANGE_CONFIG__TIMEOUT_MACROP_A, timeout_macrop_a)) {
320  return false;
321  }
322 
323  // "Get Range Timing A timeout"
324  uint32_t range_config_timeout_us = timeoutMclksToMicroseconds(decodeTimeout(timeout_macrop_a), macro_period_us);
325 
326  budget = 2 * range_config_timeout_us + TimingGuard;
327  return true;
328 }

350 // Encode sequence step timeout register value from timeout in MCLKs
351 // based on VL53L1_encode_timeout()
352 uint16_t AP_RangeFinder_VL53L1X::encodeTimeout(uint32_t timeout_mclks)
353 {
354  // encoded format: "(LSByte * 2^MSByte) + 1"
355  uint32_t ls_byte = 0;
356  uint16_t ms_byte = 0;
357 
358  if (timeout_mclks > 0) {
359  ls_byte = timeout_mclks - 1;
360  while ((ls_byte & 0xFFFFFF00) > 0) {
361  ls_byte >>= 1;
362  ms_byte++;
363  }
364  return (ms_byte << 8) | (ls_byte & 0xFF);
365  }
366  else {
367  return 0;
368  }
369 }

380 // macro period in microseconds (12.12 format)
381 // based on VL53L1_calc_timeout_mclks()
382 uint32_t AP_RangeFinder_VL53L1X::timeoutMicrosecondsToMclks(uint32_t timeout_us, uint32_t macro_period_us)
383 {

387 // Calculate macro period in microseconds (12.12 format) with given VCSEL period
388 // assumes fast_osc_frequency has been read and stored
389 // based on VL53L1_calc_macro_period_us()
390 uint32_t AP_RangeFinder_VL53L1X::calcMacroPeriod(uint8_t vcsel_period)
391 {
392  // from VL53L1_calc_pll_period_us()
393  // fast osc frequency in 4.12 format; PLL period in 0.24 format
394  uint32_t pll_period_us = ((uint32_t)0x01 << 30) / fast_osc_frequency;
395 
396  // from VL53L1_decode_vcsel_period()
397  uint8_t vcsel_period_pclks = (vcsel_period + 1) << 1;
398 
399  // VL53L1_MACRO_PERIOD_VCSEL_PERIODS = 2304
400  uint32_t macro_period_us = (uint32_t)2304 * pll_period_us;
401  macro_period_us >>= 6;
402  macro_period_us *= vcsel_period_pclks;
403  macro_period_us >>= 6;
404 
405  return macro_period_us;
406 }

444 // read - return last value measured by sensor
445 bool AP_RangeFinder_VL53L1X::get_reading(uint16_t &reading_mm)
446 {
447  uint8_t tries = 10;
448  while (!dataReady()) {
449  tries--;
450  hal.scheduler->delay(1);
451  if (tries == 0) {
452  return false;
453  }
454  }
455 
456  uint8_t range_status = 0;
457 
458  if (!(read_register(RESULT__RANGE_STATUS, range_status) &&
459  read_register16(RESULT__FINAL_CROSSTALK_CORRECTED_RANGE_MM_SD0, reading_mm))) {
460  return false;
461  }
462 
463  // "apply correction gain"
464  // gain factor of 2011 is tuning parm default (VL53L1_TUNINGPARM_LITE_RANGING_GAIN_FACTOR_DEFAULT)
465  // Basically, this appears to scale the result by 2011/2048, or about 98%
466  // (with the 1024 added for proper rounding).
467  reading_mm = ((uint32_t)reading_mm * 2011 + 0x0400) / 0x0800;
468 
469  if (!write_register(SYSTEM__INTERRUPT_CLEAR, 0x01)) { // sys_interrupt_clear_range
470  return false;
471  }
472 
473  switch ((DeviceError)range_status) {
474  case RANGECOMPLETE:
475  break;
476 
477  default:
478 #ifdef VL53L1X_DEBUG
479  hal.console->printf("VL53L1X: %d ms status %d\n", AP_HAL::millis(), (int)range_status);
480 #endif // VL53L1X_DEBUG
481  return false;
482  }
483 
484  if (!calibrated) {
485  calibrated = setupManualCalibration();
486  }
487 
488  return calibrated;
489 }

544 /*
545  update the state of the sensor
546 */
547 void AP_RangeFinder_VL53L1X::update(void)
548 {
549  WITH_SEMAPHORE(_sem);
550  if (counter > 0) {
551  state.distance_cm = sum_mm / (10*counter);
552  state.last_reading_ms = AP_HAL::millis();
553  update_status();
554  sum_mm = 0;
555  counter = 0;
