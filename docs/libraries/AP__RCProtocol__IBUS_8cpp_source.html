24 // decode a full IBUS frame
25 bool AP_RCProtocol_IBUS::ibus_decode(const uint8_t frame[IBUS_FRAME_SIZE], uint16_t *values, bool *ibus_failsafe)
26 {
27  uint32_t chksum = 96;
28 
29  /* check frame boundary markers to avoid out-of-sync cases */
30  if ((frame[0] != 0x20) || (frame[1] != 0x40)) {
31  return false;
32  }
33 
34  /* use the decoder matrix to extract channel data */
35  for (uint8_t channel = 0, pick=2; channel < IBUS_INPUT_CHANNELS; channel++, pick+=2) {
36  values[channel]=frame[pick]|(frame[pick+1] & 0x0F)<<8;
37  chksum+=frame[pick]+frame[pick+1];
38  }
39 
40  chksum += frame[IBUS_FRAME_SIZE-2]|frame[IBUS_FRAME_SIZE-1]<<8;
41 
42  if (chksum!=0xFFFF) {
43  return false;
44  }
45 
46  if ((frame[3]&0xF0) || (frame[9]&0xF0)) {
47  *ibus_failsafe = true;
48  } else {
49  *ibus_failsafe = false;
50  }
51 
52  return true;
53 }

67 // support byte input
68 void AP_RCProtocol_IBUS::_process_byte(uint32_t timestamp_us, uint8_t b)
69 {
70  const bool have_frame_gap = (timestamp_us - byte_input.last_byte_us >= 2000U);
71  byte_input.last_byte_us = timestamp_us;
72 
73  if (have_frame_gap) {
74  // if we have a frame gap then this must be the start of a new
75  // frame
76  byte_input.ofs = 0;
77  }
78  if (b != 0x20 && byte_input.ofs == 0) {
79  // definately not IBUS, missing header byte
80  return;
81  }
82  if (byte_input.ofs == 0 && !have_frame_gap) {
83  // must have a frame gap before the start of a new IBUS frame
84  return;
85  }
86 
87  byte_input.buf[byte_input.ofs++] = b;
88 
89  if (byte_input.ofs == sizeof(byte_input.buf)) {
90  uint16_t values[IBUS_INPUT_CHANNELS];
91  bool ibus_failsafe = false;
92  if (ibus_decode(byte_input.buf, values, &ibus_failsafe)) {
93  add_input(IBUS_INPUT_CHANNELS, values, ibus_failsafe);
94  }
95  byte_input.ofs = 0;
96  }
97 }
