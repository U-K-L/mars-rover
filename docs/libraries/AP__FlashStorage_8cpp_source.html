45 // initialise storage
46 bool AP_FlashStorage::init(void)
47 {
48  debug("running init()\n");
49 
50  // start with empty memory buffer
51  memset(mem_buffer, 0, storage_size);
52 
53  // find state of sectors
54  struct sector_header header[2];
55 
56  // read headers and possibly initialise if bad signature
57  for (uint8_t i=0; i<2; i++) {
58  if (!flash_read(i, 0, (uint8_t *)&header[i], sizeof(header[i]))) {
59  return false;
60  }
61  bool bad_header = (header[i].signature != signature);
62  enum SectorState state = (enum SectorState)header[i].state;
63  if (state != SECTOR_STATE_AVAILABLE &&
64  state != SECTOR_STATE_IN_USE &&
65  state != SECTOR_STATE_FULL) {
66  bad_header = true;
67  }
68 
69  // initialise if bad header
70  if (bad_header) {
71  return erase_all();
72  }
73  }
74 
75  // work out the first sector to read from using sector states
76  enum SectorState states[2] {(enum SectorState)header[0].state, (enum SectorState)header[1].state};
77  uint8_t first_sector;
78 
79  if (states[0] == states[1]) {
80  if (states[0] != SECTOR_STATE_AVAILABLE) {
81  return erase_all();
82  }
83  first_sector = 0;
84  } else if (states[0] == SECTOR_STATE_FULL) {
85  first_sector = 0;
86  } else if (states[1] == SECTOR_STATE_FULL) {
87  first_sector = 1;
88  } else if (states[0] == SECTOR_STATE_IN_USE) {
89  first_sector = 0;
90  } else if (states[1] == SECTOR_STATE_IN_USE) {
91  first_sector = 1;
92  } else {
93  // doesn't matter which is first
94  first_sector = 0;
95  }
96 
97  // load data from any current sectors
98  for (uint8_t i=0; i<2; i++) {
99  uint8_t sector = (first_sector + i) & 1;
100  if (states[sector] == SECTOR_STATE_IN_USE ||
101  states[sector] == SECTOR_STATE_FULL) {
102  if (!load_sector(sector)) {
103  return erase_all();
104  }
105  }
106  }
107 
108  // clear any write error
109  write_error = false;
110  reserved_space = 0;
111 
112  // if the first sector is full then write out all data so we can erase it
113  if (states[first_sector] == SECTOR_STATE_FULL) {
114  current_sector = first_sector ^ 1;
115  if (!write_all()) {
116  return erase_all();
117  }
118  }
119 
120  // erase any sectors marked full
121  for (uint8_t i=0; i<2; i++) {
122  if (states[i] == SECTOR_STATE_FULL) {
123  if (!erase_sector(i)) {
124  return false;
125  }
126  }
127  }
128 
129  reserved_space = 0;
130 
131  // ready to use
132  return true;
133 }

158 // write some data to virtual EEPROM
159 bool AP_FlashStorage::write(uint16_t offset, uint16_t length)
160 {
161  if (write_error) {
162  return false;
163  }
164  //debug("write at %u for %u write_offset=%u\n", offset, length, write_offset);
165 
166  while (length > 0) {
167  uint8_t n = max_write;
168  if (length < n) {
169  n = length;
170  }
171 
172  if (write_offset > flash_sector_size - (sizeof(struct block_header) + max_write + reserved_space)) {
173  if (!switch_sectors()) {
174  if (!flash_erase_ok()) {
175  return false;
176  }
177  if (!switch_full_sector()) {
178  return false;
179  }
180  }
181  }
182 
183  struct block_header header;
184  header.state = BLOCK_STATE_WRITING;
185  header.block_num = offset / block_size;
186  header.num_blocks_minus_one = ((n + (block_size - 1)) / block_size)-1;
187  uint16_t block_ofs = header.block_num*block_size;
188  uint16_t block_nbytes = (header.num_blocks_minus_one+1)*block_size;
189 
190  if (!flash_write(current_sector, write_offset, (uint8_t*)&header, sizeof(header))) {
191  return false;
192  }
193  if (!flash_write(current_sector, write_offset+sizeof(header), &mem_buffer[block_ofs], block_nbytes)) {
194  return false;
195  }
196  header.state = BLOCK_STATE_VALID;
197  if (!flash_write(current_sector, write_offset, (uint8_t*)&header, sizeof(header))) {
198  return false;
199  }
200  write_offset += sizeof(header) + block_nbytes;
201 
202  uint8_t n2 = block_nbytes - (offset % block_size);
203  //debug("write_block at %u for %u n2=%u\n", block_ofs, block_nbytes, n2);
204  if (n2 > length) {
205  break;
206  }
207  offset += n2;
208  length -= n2;
209  }
210 
211  // handle wrap to next sector
212  // write data
213  // write header word
214  return true;
215 }

217 /*
218  load all data from a flash sector into mem_buffer
219  */
220 bool AP_FlashStorage::load_sector(uint8_t sector)
221 {
222  uint32_t ofs = sizeof(sector_header);
223  while (ofs < flash_sector_size - sizeof(struct block_header)) {
224  struct block_header header;
225  if (!flash_read(sector, ofs, (uint8_t *)&header, sizeof(header))) {
226  return false;
227  }
228  enum BlockState state = (enum BlockState)header.state;
229 
230  switch (state) {
231  case BLOCK_STATE_AVAILABLE:
232  // we've reached the end
233  write_offset = ofs;
234  return true;
235 
236  case BLOCK_STATE_WRITING: {
237  /*
238  we were interrupted while writing a block. We can't
239  re-use the data in this block as it may have some bits
240  that are not set to 1, so by flash rules can't be set to
241  an arbitrary value. So we skip over this block, leaving
242  a gap. The gap size is limited to (7+1)*8=64 bytes. That
243  gap won't be recovered until we next do an erase of this
244  sector
245  */
246  uint16_t block_nbytes = (header.num_blocks_minus_one+1)*block_size;
247  ofs += block_nbytes + sizeof(header);
248  break;
249  }
250 
251  case BLOCK_STATE_VALID: {
252  uint16_t block_nbytes = (header.num_blocks_minus_one+1)*block_size;
253  uint16_t block_ofs = header.block_num*block_size;
254  if (block_ofs + block_nbytes > storage_size) {
255  // the data is invalid (out of range)
256  return false;
257  }
258  if (!flash_read(sector, ofs+sizeof(header), &mem_buffer[block_ofs], block_nbytes)) {
259  return false;
260  }
261  //debug("read at %u for %u\n", block_ofs, block_nbytes);
262  ofs += block_nbytes + sizeof(header);
263  break;
264  }
265  default:
266  // invalid state
267  return false;
268  }
269  }
270  write_offset = ofs;
271  return true;
272 }

311  write all of mem_buffer to current sector

313 bool AP_FlashStorage::write_all()
314 {
315  debug("write_all to sector %u at %u with reserved_space=%u\n",
316  current_sector, write_offset, reserved_space);
317  for (uint16_t ofs=0; ofs<storage_size; ofs += max_write) {
318  // local variable needed to overcome problem with MIN() macro and -O0
319  const uint8_t max_write_local = max_write;
320  uint8_t n = MIN(max_write_local, storage_size-ofs);
321  if (!all_zero(ofs, n)) {
322  if (!write(ofs, n)) {
323  return false;
324  }
325  }
326  }
327  return true;
328 }

341 // switch to next sector for writing
342 bool AP_FlashStorage::switch_sectors(void)
343 {
344  if (reserved_space != 0) {
345  // other sector is already full
346  debug("both sectors are full\n");
347  return false;
348  }
349 
350  struct sector_header header;
351  header.signature = signature;
352 
353  uint8_t new_sector = current_sector ^ 1;
354  debug("switching to sector %u\n", new_sector);
355 
356  // check sector is available
357  if (!flash_read(new_sector, 0, (uint8_t *)&header, sizeof(header))) {
358  return false;
359  }
360  if (header.signature != signature) {
361  write_error = true;
362  return false;
363  }
364  if (SECTOR_STATE_AVAILABLE != (enum SectorState)header.state) {
365  write_error = true;
366  debug("both sectors full\n");
367  return false;
368  }
369 
370  // mark current sector as full. This needs to be done before we
371  // mark the new sector as in-use so that a power failure between
372  // the two steps doesn't leave us with an erase on the
373  // reboot. Thanks to night-ghost for spotting this.
374  header.state = SECTOR_STATE_FULL;
375  if (!flash_write(current_sector, 0, (const uint8_t *)&header, sizeof(header))) {
376  return false;
377  }
378 
379  // mark new sector as in-use
380  header.state = SECTOR_STATE_IN_USE;
381  if (!flash_write(new_sector, 0, (const uint8_t *)&header, sizeof(header))) {
382  return false;
383  }
384 
385  // switch sectors
386  current_sector = new_sector;
387 
388  // we need to reserve some space in next sector to ensure we can successfully do a
389  // full write out on init()
390  reserved_space = reserve_size;
391 
392  write_offset = sizeof(header);
393  return true;
394 }
