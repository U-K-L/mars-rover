61 // loop
62 static void loop()
63 {
64  static const uint8_t compass_count = compass.get_count();
65  static float min[COMPASS_MAX_INSTANCES][3];
66  static float max[COMPASS_MAX_INSTANCES][3];
67  static float offset[COMPASS_MAX_INSTANCES][3];
68 
69  // run read() at 10Hz
70  if ((AP_HAL::micros() - timer) > 100000L) {
71  timer = AP_HAL::micros();
72  compass.read();
73  const uint32_t read_time = AP_HAL::micros() - timer;
74 
75  for (uint8_t i = 0; i < compass_count; i++) {
76  float heading;
77 
78  hal.console->printf("Compass #%u: ", i);
79 
80  if (!compass.healthy()) {
81  hal.console->printf("not healthy\n");
82  continue;
83  }
84 
85  Matrix3f dcm_matrix;
86  // use roll = 0, pitch = 0 for this example
87  dcm_matrix.from_euler(0, 0, 0);
88  heading = compass.calculate_heading(dcm_matrix, i);
89 
90  const Vector3f &mag = compass.get_field(i);
91 
92  // capture min
93  min[i][0] = MIN(mag.x, min[i][0]);
94  min[i][1] = MIN(mag.y, min[i][1]);
95  min[i][2] = MIN(mag.z, min[i][2]);
96 
97  // capture max
98  max[i][0] = MAX(mag.x, max[i][0]);
99  max[i][1] = MAX(mag.y, max[i][1]);
100  max[i][2] = MAX(mag.z, max[i][2]);
101 
102  // calculate offsets
103  offset[i][0] = -(max[i][0] + min[i][0]) / 2;
104  offset[i][1] = -(max[i][1] + min[i][1]) / 2;
105  offset[i][2] = -(max[i][2] + min[i][2]) / 2;
106 
107  // display all to user
108  hal.console->printf("Heading: %.2f (%3d, %3d, %3d)",
109  (double)ToDeg(heading),
110  (int)mag.x,
111  (int)mag.y,
112  (int)mag.z);
113 
114  // display offsets
115  hal.console->printf(" offsets(%.2f, %.2f, %.2f)",
116  (double)offset[i][0],
117  (double)offset[i][1],
118  (double)offset[i][2]);
119 
120  hal.console->printf(" t=%u", (unsigned)read_time);
121 
122  hal.console->printf("\n");
123  }
124  } else {
125 
126  // if stipulated time has not passed between two distinct readings, delay the program for a millisecond
127  hal.scheduler->delay(1);
128  }
129 }
