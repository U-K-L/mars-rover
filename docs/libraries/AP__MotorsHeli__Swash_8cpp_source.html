103 // CCPM Mixers - calculate mixing scale factors by swashplate type
104 void AP_MotorsHeli_Swash::calculate_roll_pitch_collective_factors()
105 {
106  if (_swash_type == SWASHPLATE_TYPE_H1) {
107  // CCPM mixing not used
108  _collectiveFactor[CH_1] = 0;
109  _collectiveFactor[CH_2] = 0;
110  _collectiveFactor[CH_3] = 1;
111  } else if ((_swash_type == SWASHPLATE_TYPE_H4_90) || (_swash_type == SWASHPLATE_TYPE_H4_45)) {
112  // collective mixer for four-servo CCPM
113  _collectiveFactor[CH_1] = 1;
114  _collectiveFactor[CH_2] = 1;
115  _collectiveFactor[CH_3] = 1;
116  _collectiveFactor[CH_4] = 1;
117  } else {
118  // collective mixer for three-servo CCPM
119  _collectiveFactor[CH_1] = 1;
120  _collectiveFactor[CH_2] = 1;
121  _collectiveFactor[CH_3] = 1;
122  }
123 
124  if (_swash_type == SWASHPLATE_TYPE_H3) {
125  // Three-servo roll/pitch mixer for adjustable servo position
126  // can be any style swashplate, phase angle is adjustable
127  _rollFactor[CH_1] = cosf(radians(_servo1_pos + 90 - _phase_angle));
128  _rollFactor[CH_2] = cosf(radians(_servo2_pos + 90 - _phase_angle));
129  _rollFactor[CH_3] = cosf(radians(_servo3_pos + 90 - _phase_angle));
130  _pitchFactor[CH_1] = cosf(radians(_servo1_pos - _phase_angle));
131  _pitchFactor[CH_2] = cosf(radians(_servo2_pos - _phase_angle));
132  _pitchFactor[CH_3] = cosf(radians(_servo3_pos - _phase_angle));
133 
134  // defined swashplates, servo1 is always left, servo2 is right,
135  // servo3 is elevator
136  } else if (_swash_type == SWASHPLATE_TYPE_H3_140) { //
137  // Three-servo roll/pitch mixer for H3-140
138  // HR3-140 uses reversed servo and collective direction in heli setup
139  // 1:1 pure input style, phase angle not adjustable
140  _rollFactor[CH_1] = 1;
141  _rollFactor[CH_2] = -1;
142  _rollFactor[CH_3] = 0;
143  _pitchFactor[CH_1] = 1;
144  _pitchFactor[CH_2] = 1;
145  _pitchFactor[CH_3] = -1;
146  } else if (_swash_type == SWASHPLATE_TYPE_H3_120) {
147  // three-servo roll/pitch mixer for H3-120
148  // HR3-120 uses reversed servo and collective direction in heli setup
149  // not a pure mixing swashplate, phase angle is adjustable
150  _rollFactor[CH_1] = 0.866025f;
151  _rollFactor[CH_2] = -0.866025f;
152  _rollFactor[CH_3] = 0;
153  _pitchFactor[CH_1] = 0.5f;
154  _pitchFactor[CH_2] = 0.5f;
155  _pitchFactor[CH_3] = -1;
156  } else if (_swash_type == SWASHPLATE_TYPE_H4_90) {
157  // four-servo roll/pitch mixer for H4-90
158  // 1:1 pure input style, phase angle not adjustable
159  // servos 3 & 7 are elevator
160  // can also be used for all versions of 90 deg three-servo swashplates
161  _rollFactor[CH_1] = 1;
162  _rollFactor[CH_2] = -1;
163  _rollFactor[CH_3] = 0;
164  _rollFactor[CH_4] = 0;
165  _pitchFactor[CH_1] = 0;
166  _pitchFactor[CH_2] = 0;
167  _pitchFactor[CH_3] = -1;
168  _pitchFactor[CH_4] = 1;
169  } else if (_swash_type == SWASHPLATE_TYPE_H4_45) {
170  // four-servo roll/pitch mixer for H4-45
171  // 1:1 pure input style, phase angle not adjustable
172  // for 45 deg plates servos 1&2 are LF&RF, 3&7 are LR&RR.
173  _rollFactor[CH_1] = 0.707107f;
174  _rollFactor[CH_2] = -0.707107f;
175  _rollFactor[CH_3] = 0.707107f;
176  _rollFactor[CH_4] = -0.707107f;
177  _pitchFactor[CH_1] = 0.707107f;
178  _pitchFactor[CH_2] = 0.707107f;
179  _pitchFactor[CH_3] = -0.707f;
180  _pitchFactor[CH_4] = -0.707f;
181  } else {
182  // CCPM mixing not being used, so H1 straight outputs
183  _rollFactor[CH_1] = 1;
184  _rollFactor[CH_2] = 0;
185  _rollFactor[CH_3] = 0;
186  _pitchFactor[CH_1] = 0;
187  _pitchFactor[CH_2] = 1;
188  _pitchFactor[CH_3] = 0;
189  }

192 // get_servo_out - calculates servo output
193 float AP_MotorsHeli_Swash::get_servo_out(int8_t ch_num, float pitch, float roll, float collective) const
194 {
195  // Collective control direction. Swash moves up for negative collective pitch, down for positive collective pitch
196  if (_collective_direction == COLLECTIVE_DIRECTION_REVERSED){
197  collective = 1 - collective;
198  }
199 
200  float servo = ((_rollFactor[ch_num] * roll) + (_pitchFactor[ch_num] * pitch))*0.45f + _collectiveFactor[ch_num] * collective;
201  if (_swash_type == SWASHPLATE_TYPE_H1 && (ch_num == CH_1 || ch_num == CH_2)) {
202  servo += 0.5f;
203  }
204 
205  // rescale from -1..1, so we can use the pwm calc that includes trim
206  servo = 2.0f * servo - 1.0f;
207 
208  if (_make_servo_linear == 1) {
209  servo = get_linear_servo_output(servo);
210  }
211 
212  return servo;
213 }

215 // set_linear_servo_out - sets swashplate servo output to be linear
216 float AP_MotorsHeli_Swash::get_linear_servo_output(float input) const
217 {
218 
219  input = constrain_float(input, -1.0f, 1.0f);
220 
221  //servo output is calculated by normalizing input to 50 deg arm rotation as full input for a linear throw
222  return safe_asin(0.766044f * input) * 1.145916;
223 
224 }
