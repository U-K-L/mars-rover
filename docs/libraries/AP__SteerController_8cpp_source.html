119 /*
120  steering rate controller. Returns servo out -4500 to 4500 given
121  desired yaw rate in degrees/sec. Positive yaw rate means clockwise yaw.
122 */
123 int32_t AP_SteerController::get_steering_out_rate(float desired_rate)
124 {
125  uint32_t tnow = AP_HAL::millis();
126  uint32_t dt = tnow - _last_t;
127  if (_last_t == 0 || dt > 1000) {
128  dt = 0;
129  }
130  _last_t = tnow;
131 
132  float speed = _ahrs.groundspeed();
133  if (speed < _minspeed) {
134  // assume a minimum speed. This stops oscillations when first starting to move
135  speed = _minspeed;
136  }
137 
138  // this is a linear approximation of the inverse steering
139  // equation for a ground vehicle. It returns steering as an angle from -45 to 45
140  float scaler = 1.0f / speed;
141 
142  _pid_info.desired = desired_rate;
143 
144  // Calculate the steering rate error (deg/sec) and apply gain scaler
145  // We do this in earth frame to allow for rover leaning over in hard corners
146  float yaw_rate_earth = ToDeg(_ahrs.get_yaw_rate_earth());
147  if (_reverse) {
148  yaw_rate_earth *= -1.0f;
149  }
150  _pid_info.actual = yaw_rate_earth;
151 
152  float rate_error = (desired_rate - yaw_rate_earth) * scaler;
153 
154  // Calculate equivalent gains so that values for K_P and K_I can be taken across from the old PID law
155  // No conversion is required for K_D
156  float ki_rate = _K_I * _tau * 45.0f;
157  float kp_ff = MAX((_K_P - _K_I * _tau) * _tau - _K_D , 0) * 45.0f;
158  float k_ff = _K_FF * 45.0f;
159  float delta_time = (float)dt * 0.001f;
160 
161  // Multiply yaw rate error by _ki_rate and integrate
162  // Don't integrate if in stabilize mode as the integrator will wind up against the pilots inputs
163  if (ki_rate > 0 && speed >= _minspeed) {
164  // only integrate if gain and time step are positive.
165  if (dt > 0) {
166  float integrator_delta = rate_error * ki_rate * delta_time * scaler;
167  // prevent the integrator from increasing if steering defln demand is above the upper limit
168  if (_last_out < -45) {
169  integrator_delta = MAX(integrator_delta , 0);
170  } else if (_last_out > 45) {
171  // prevent the integrator from decreasing if steering defln demand is below the lower limit
172  integrator_delta = MIN(integrator_delta, 0);
173  }
174  _pid_info.I += integrator_delta;
175  }
176  } else {
177  _pid_info.I = 0;
178  }
179 
180  // Scale the integration limit
181  float intLimScaled = _imax * 0.01f;
182 
183  // Constrain the integrator state
184  _pid_info.I = constrain_float(_pid_info.I, -intLimScaled, intLimScaled);
185 
186  _pid_info.D = rate_error * _K_D * 4.0f;
187  _pid_info.P = (ToRad(desired_rate) * kp_ff) * scaler;
188  _pid_info.FF = (ToRad(desired_rate) * k_ff) * scaler;
189 
190  // Calculate the demanded control surface deflection
191  _last_out = _pid_info.D + _pid_info.FF + _pid_info.P + _pid_info.I;
192 
193  float derate_constraint = 4500;
194 
195  // Calculate required constrain based on speed
196  if (!is_zero(_deratespeed) && speed > _deratespeed) {
197  derate_constraint = 4500 - (speed - _deratespeed) * _deratefactor * 100;
198  if (derate_constraint < _mindegree) {
199  derate_constraint = _mindegree;
200  }
201  }
202 
203  // Convert to centi-degrees and constrain
204  return constrain_float(_last_out * 100, -derate_constraint, derate_constraint);
205 }

208 /*
209  lateral acceleration controller. Returns servo value -4500 to 4500
210  given a desired lateral acceleration
211 */
212 int32_t AP_SteerController::get_steering_out_lat_accel(float desired_accel)
213 {
214  float speed = _ahrs.groundspeed();
215  if (speed < _minspeed) {
216  // assume a minimum speed. This reduces osciallations when first starting to move
217  speed = _minspeed;
218  }
219 
220  // Calculate the desired steering rate given desired_accel and speed
221  float desired_rate = ToDeg(desired_accel / speed);
222  if (_reverse) {
223  desired_rate *= -1;
224  }
225  return get_steering_out_rate(desired_rate);
226 }

228 /*
229  return a steering servo value from -4500 to 4500 given an angular
230  steering error in centidegrees.
231 */
232 int32_t AP_SteerController::get_steering_out_angle_error(int32_t angle_err)
233 {
234  if (_tau < 0.1f) {
235  _tau = 0.1f;
236  }
237 
238  // Calculate the desired steering rate (deg/sec) from the angle error
239  float desired_rate = angle_err * 0.01f / _tau;
240 
241  return get_steering_out_rate(desired_rate);
242 }
