133 // get target's estimated location
134 bool AP_Follow::get_target_location_and_velocity(Location &loc, Vector3f &vel_ned) const
135 {
136  // exit immediately if not enabled
137  if (!_enabled) {
138  return false;
139  }
140 
141  // check for timeout
142  if ((_last_location_update_ms == 0) || (AP_HAL::millis() - _last_location_update_ms > AP_FOLLOW_TIMEOUT_MS)) {
143  return false;
144  }
145 
146  // calculate time since last actual position update
147  const float dt = (AP_HAL::millis() - _last_location_update_ms) * 0.001f;
148 
149  // get velocity estimate
150  if (!get_velocity_ned(vel_ned, dt)) {
151  return false;
152  }
153 
154  // project the vehicle position
155  Location last_loc = _target_location;
156  last_loc.offset(vel_ned.x * dt, vel_ned.y * dt);
157  last_loc.alt -= vel_ned.z * 100.0f * dt; // convert m/s to cm/s, multiply by dt. minus because NED
158 
159  // return latest position estimate
160  loc = last_loc;
161  return true;
162 }

164 // get distance vector to target (in meters) and target's velocity all in NED frame
165 bool AP_Follow::get_target_dist_and_vel_ned(Vector3f &dist_ned, Vector3f &dist_with_offs, Vector3f &vel_ned)
166 {
167  // get our location
168  Location current_loc;
169  if (!AP::ahrs().get_position(current_loc)) {
170  clear_dist_and_bearing_to_target();
171  return false;
172  }
173 
174  // get target location and velocity
175  Location target_loc;
176  Vector3f veh_vel;
177  if (!get_target_location_and_velocity(target_loc, veh_vel)) {
178  clear_dist_and_bearing_to_target();
179  return false;
180  }
181 
182  // change to altitude above home if relative altitude is being used
183  if (target_loc.relative_alt == 1) {
184  current_loc.alt -= AP::ahrs().get_home().alt;
185  }
186 
187  // calculate difference
188  const Vector3f dist_vec = current_loc.get_distance_NED(target_loc);
189 
190  // fail if too far
191  if (is_positive(_dist_max.get()) && (dist_vec.length() > _dist_max)) {
192  clear_dist_and_bearing_to_target();
193  return false;
194  }
195 
196  // initialise offsets from distance vector if required
197  init_offsets_if_required(dist_vec);
198 
199  // get offsets
200  Vector3f offsets;
201  if (!get_offsets_ned(offsets)) {
202  clear_dist_and_bearing_to_target();
203  return false;
204  }
205 
206  // calculate results
207  dist_ned = dist_vec;
208  dist_with_offs = dist_vec + offsets;
209  vel_ned = veh_vel;
210 
211  // record distance and heading for reporting purposes
212  if (is_zero(dist_with_offs.x) && is_zero(dist_with_offs.y)) {
213  clear_dist_and_bearing_to_target();
214  } else {
215  _dist_to_target = safe_sqrt(sq(dist_with_offs.x) + sq(dist_with_offs.y));
216  _bearing_to_target = degrees(atan2f(dist_with_offs.y, dist_with_offs.x));
217  }
218 
219  return true;
220 }

240 // handle mavlink DISTANCE_SENSOR messages
241 void AP_Follow::handle_msg(const mavlink_message_t &msg)
242 {
243  // exit immediately if not enabled
244  if (!_enabled) {
245  return;
246  }
247 
248  // skip our own messages
249  if (msg.sysid == mavlink_system.sysid) {
250  return;
251  }
252 
253  // skip message if not from our target
254  if (_sysid != 0 && msg.sysid != _sysid) {
255  if (_automatic_sysid) {
256  // maybe timeout who we were following...
257  if ((_last_location_update_ms == 0) || (AP_HAL::millis() - _last_location_update_ms > AP_FOLLOW_SYSID_TIMEOUT_MS)) {
258  _sysid.set(0);
259  }
260  }
261  return;
262  }
263 
264  // decode global-position-int message
265  if (msg.msgid == MAVLINK_MSG_ID_GLOBAL_POSITION_INT) {
266 
267  // get estimated location and velocity (for logging)
268  Location loc_estimate{};
269  Vector3f vel_estimate;
270  UNUSED_RESULT(get_target_location_and_velocity(loc_estimate, vel_estimate));
271 
272  // decode message
273  mavlink_global_position_int_t packet;
274  mavlink_msg_global_position_int_decode(&msg, &packet);
275 
276  // ignore message if lat and lon are (exactly) zero
277  if ((packet.lat == 0 && packet.lon == 0)) {
278  return;
279  }
280 
281  _target_location.lat = packet.lat;
282  _target_location.lng = packet.lon;
283 
284  // select altitude source based on FOLL_ALT_TYPE param
285  if (_alt_type == AP_FOLLOW_ALTITUDE_TYPE_RELATIVE) {
286  // relative altitude
287  _target_location.alt = packet.relative_alt / 10; // convert millimeters to cm
288  _target_location.relative_alt = 1; // set relative_alt flag
289  } else {
290  // absolute altitude
291  _target_location.alt = packet.alt / 10; // convert millimeters to cm
292  _target_location.relative_alt = 0; // reset relative_alt flag
293  }
294 
295  _target_velocity_ned.x = packet.vx * 0.01f; // velocity north
296  _target_velocity_ned.y = packet.vy * 0.01f; // velocity east
297  _target_velocity_ned.z = packet.vz * 0.01f; // velocity down
298 
299  // get a local timestamp with correction for transport jitter
300  _last_location_update_ms = _jitter.correct_offboard_timestamp_msec(packet.time_boot_ms, AP_HAL::millis());
301  if (packet.hdg <= 36000) { // heading (UINT16_MAX if unknown)
302  _target_heading = packet.hdg * 0.01f; // convert centi-degrees to degrees
303  _last_heading_update_ms = _last_location_update_ms;
304  }
305  // initialise _sysid if zero to sender's id
306  if (_sysid == 0) {
307  _sysid.set(msg.sysid);
308  _automatic_sysid = true;
309  }
310 
311  // log lead's estimated vs reported position
312  AP::logger().Write("FOLL",
313  "TimeUS,Lat,Lon,Alt,VelN,VelE,VelD,LatE,LonE,AltE", // labels
314  "sDUmnnnDUm", // units
315  "F--B000--B", // mults
316  "QLLifffLLi", // fmt
317  AP_HAL::micros64(),
318  _target_location.lat,
319  _target_location.lng,
320  _target_location.alt,
321  (double)_target_velocity_ned.x,
322  (double)_target_velocity_ned.y,
323  (double)_target_velocity_ned.z,
324  loc_estimate.lat,
325  loc_estimate.lng,
326  loc_estimate.alt
327  );
328  }
329 }

338 // initialise offsets to provided distance vector to other vehicle (in meters in NED frame) if required
339 void AP_Follow::init_offsets_if_required(const Vector3f &dist_vec_ned)
340 {
341  // return immediately if offsets have already been set
342  if (!_offset.get().is_zero()) {
343  return;
344  }
345 
346  float target_heading_deg;
347  if ((_offset_type == AP_FOLLOW_OFFSET_TYPE_RELATIVE) && get_target_heading_deg(target_heading_deg)) {
348  // rotate offsets from north facing to vehicle's perspective
349  _offset = rotate_vector(-dist_vec_ned, -target_heading_deg);
350  } else {
351  // initialise offset in NED frame
352  _offset = -dist_vec_ned;
353  // ensure offset_type used matches frame of offsets saved
354  _offset_type = AP_FOLLOW_OFFSET_TYPE_NED;
355  }
356 }

358 // get offsets in meters in NED frame
359 bool AP_Follow::get_offsets_ned(Vector3f &offset) const
360 {
361  const Vector3f &off = _offset.get();
362 
363  // if offsets are zero or type is NED, simply return offset vector
364  if (off.is_zero() || (_offset_type == AP_FOLLOW_OFFSET_TYPE_NED)) {
365  offset = off;
366  return true;
367  }
368 
369  // offset type is relative, exit if we cannot get vehicle's heading
370  float target_heading_deg;
371  if (!get_target_heading_deg(target_heading_deg)) {
372  return false;
373  }
374 
375  // rotate offsets from vehicle's perspective to NED
376  offset = rotate_vector(off, target_heading_deg);
377  return true;
378 }

380 // rotate 3D vector clockwise by specified angle (in degrees)
381 Vector3f AP_Follow::rotate_vector(const Vector3f &vec, float angle_deg) const
382 {
383  // rotate roll, pitch input from north facing to vehicle's perspective
384  const float cos_yaw = cosf(radians(angle_deg));
385  const float sin_yaw = sinf(radians(angle_deg));
386  return Vector3f((vec.x * cos_yaw) - (vec.y * sin_yaw), (vec.y * cos_yaw) + (vec.x * sin_yaw), vec.z);
387 }
