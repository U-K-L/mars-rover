130 // Read the receiver RSSI value as a float 0.0f - 1.0f.
131 // 0.0 represents weakest signal, 1.0 represents maximum signal.
132 float AP_RSSI::read_receiver_rssi()
133 {
134  switch (RssiType(rssi_type.get())) {
135  case RssiType::TYPE_DISABLED:
136  return 0.0f;
137  case RssiType::ANALOG_PIN:
138  return read_pin_rssi();
139  case RssiType::RC_CHANNEL_VALUE:
140  return read_channel_rssi();
141  case RssiType::RECEIVER: {
142  int16_t rssi = RC_Channels::get_receiver_rssi();
143  if (rssi != -1) {
144  return rssi / 255.0;
145  }
146  return 0.0f;
147  }
148  case RssiType::PWM_PIN:
149  return read_pwm_pin_rssi();
150  }
151  // should never get to here
152  return 0.0f;
153 }

186 void AP_RSSI::check_pwm_pin_rssi()
187 {
188  if (rssi_analog_pin == pwm_state.last_rssi_analog_pin) {
189  return;
190  }
191 
192  // detach last one
193  if (pwm_state.last_rssi_analog_pin) {
194  if (!hal.gpio->detach_interrupt(pwm_state.last_rssi_analog_pin)) {
195  gcs().send_text(MAV_SEVERITY_WARNING,
196  "RSSI: Failed to detach from pin %u",
197  pwm_state.last_rssi_analog_pin);
198  // ignore this failure or the user may be stuck
199  }
200  }
201 
202  pwm_state.last_rssi_analog_pin = rssi_analog_pin;
203 
204  if (!rssi_analog_pin) {
205  // don't need to install handler
206  return;
207  }
208 
209  // install interrupt handler on rising and falling edge
210  hal.gpio->pinMode(rssi_analog_pin, HAL_GPIO_INPUT);
211  if (!hal.gpio->attach_interrupt(
212  rssi_analog_pin,
213  FUNCTOR_BIND_MEMBER(&AP_RSSI::irq_handler,
214  void,
215  uint8_t,
216  bool,
217  uint32_t),
218  AP_HAL::GPIO::INTERRUPT_BOTH)) {
219  // failed to attach interrupt
220  gcs().send_text(MAV_SEVERITY_WARNING,
221  "RSSI: Failed to attach to pin %u",
222  rssi_analog_pin);
223  return;
224  }
225 }

227 // read the PWM value from a pin
228 float AP_RSSI::read_pwm_pin_rssi()
229 {
230  // check if pin has changed and configure interrupt handlers if required:
231  check_pwm_pin_rssi();
232 
233  if (!pwm_state.last_rssi_analog_pin) {
234  // disabled (either by configuration or failure to attach interrupt)
235  return 0.0f;
236  }
237 
238  // disable interrupts and grab state
239  void *irqstate = hal.scheduler->disable_interrupts_save();
240  const uint32_t irq_value_us = pwm_state.irq_value_us;
241  pwm_state.irq_value_us = 0;
242  hal.scheduler->restore_interrupts(irqstate);
243 
244  const uint32_t now = AP_HAL::millis();
245  if (irq_value_us == 0) {
246  // no reading; check for timeout:
247  if (now - pwm_state.last_reading_ms > 1000) {
248  // no reading for a second - something is broken
249  pwm_state.rssi_value = 0.0f;
250  }
251  } else {
252  // a new reading - convert pwm value to rssi value
253  pwm_state.rssi_value = scale_and_constrain_float_rssi(irq_value_us, rssi_channel_low_pwm_value, rssi_channel_high_pwm_value);
254  pwm_state.last_reading_ms = now;
255  }
256 
257  return pwm_state.rssi_value;
258 }

260 // Scale and constrain a float rssi value to 0.0 to 1.0 range
261 float AP_RSSI::scale_and_constrain_float_rssi(float current_rssi_value, float low_rssi_range, float high_rssi_range)
262 {
263  float rssi_value_range = fabsf(high_rssi_range - low_rssi_range);
264  if (is_zero(rssi_value_range)) {
265  // User range isn't meaningful, return 0 for RSSI (and avoid divide by zero)
266  return 0.0f;
267  }
268  // Note that user-supplied ranges may be inverted and we accommodate that here.
269  // (Some radio receivers put out inverted ranges for RSSI-type values).
270  bool range_is_inverted = (high_rssi_range < low_rssi_range);
271  // Constrain to the possible range - values outside are clipped to ends
272  current_rssi_value = constrain_float(current_rssi_value,
273  range_is_inverted ? high_rssi_range : low_rssi_range,
274  range_is_inverted ? low_rssi_range : high_rssi_range);
275 
276  if (range_is_inverted)
277  {
278  // Swap values so we can treat them as low->high uniformly in the code that follows
279  current_rssi_value = high_rssi_range + fabsf(current_rssi_value - low_rssi_range);
280  std::swap(low_rssi_range, high_rssi_range);
281  }
282 
283  // Scale the value down to a 0.0 - 1.0 range
284  float rssi_value_scaled = (current_rssi_value - low_rssi_range) / rssi_value_range;
285  // Make absolutely sure the value is clipped to the 0.0 - 1.0 range. This should handle things if the
286  // value retrieved falls outside the user-supplied range.
287  return constrain_float(rssi_value_scaled, 0.0f, 1.0f);
288 }

291 void AP_RSSI::irq_handler(uint8_t pin, bool pin_high, uint32_t timestamp_us)
292 {

299  }
