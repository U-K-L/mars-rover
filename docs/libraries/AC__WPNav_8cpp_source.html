315 bool AC_WPNav::advance_wp_target_along_track(float dt)
316 {
317  float track_covered; // distance (in cm) along the track that the vehicle has traveled. Measured by drawing a perpendicular line from the track to the vehicle.
318  Vector3f track_error; // distance error (in cm) from the track_covered position (i.e. closest point on the line to the vehicle) and the vehicle
319  float track_desired_max; // the farthest distance (in cm) along the track that the leash will allow
320  float track_leash_slack; // additional distance (in cm) along the track from our track_covered position that our leash will allow
321  bool reached_leash_limit = false; // true when track has reached leash limit and we need to slow down the target point
322 
323  // get current location
324  Vector3f curr_pos = _inav.get_position();
325 
326  // calculate terrain adjustments
327  float terr_offset = 0.0f;
328  if (_terrain_alt && !get_terrain_offset(terr_offset)) {
329  return false;
330  }
331 
332  // calculate 3d vector from segment's origin
333  Vector3f curr_delta = (curr_pos - Vector3f(0,0,terr_offset)) - _origin;
334 
335  // calculate how far along the track we are
336  track_covered = curr_delta.x * _pos_delta_unit.x + curr_delta.y * _pos_delta_unit.y + curr_delta.z * _pos_delta_unit.z;
337 
338  // calculate the point closest to the vehicle on the segment from origin to destination
339  Vector3f track_covered_pos = _pos_delta_unit * track_covered;
340 
341  // calculate the distance vector from the vehicle to the closest point on the segment from origin to destination
342  track_error = curr_delta - track_covered_pos;
343 
344  // calculate the horizontal error
345  _track_error_xy = norm(track_error.x, track_error.y);
346 
347  // calculate the vertical error
348  float track_error_z = fabsf(track_error.z);
349 
350  // get up leash if we are moving up, down leash if we are moving down
351  float leash_z = track_error.z >= 0 ? _pos_control.get_leash_up_z() : _pos_control.get_leash_down_z();
352 
353  // use pythagoras's theorem calculate how far along the track we could move the intermediate target before reaching the end of the leash
354  // track_desired_max is the distance from the vehicle to our target point along the track. It is the "hypotenuse" which we want to be no longer than our leash (aka _track_leash_length)
355  // track_error is the line from the vehicle to the closest point on the track. It is the "opposite" side
356  // track_leash_slack is the line from the closest point on the track to the target point. It is the "adjacent" side. We adjust this so the track_desired_max is no longer than the leash
357  float track_leash_length_abs = fabsf(_track_leash_length);
358  float track_error_max_abs = MAX(_track_leash_length*track_error_z/leash_z, _track_leash_length*_track_error_xy/_pos_control.get_leash_xy());
359  track_leash_slack = (track_leash_length_abs > track_error_max_abs) ? safe_sqrt(sq(_track_leash_length) - sq(track_error_max_abs)) : 0;
360  track_desired_max = track_covered + track_leash_slack;
361 
362  // check if target is already beyond the leash
363  if (_track_desired > track_desired_max) {
364  reached_leash_limit = true;
365  }
366 
367  // get current velocity
368  const Vector3f &curr_vel = _inav.get_velocity();
369  // get speed along track
370  float speed_along_track = curr_vel.x * _pos_delta_unit.x + curr_vel.y * _pos_delta_unit.y + curr_vel.z * _pos_delta_unit.z;
371 
372  // calculate point at which velocity switches from linear to sqrt
373  float linear_velocity = _pos_control.get_max_speed_xy();
374  float kP = _pos_control.get_pos_xy_p().kP();
375  if (is_positive(kP)) { // avoid divide by zero
376  linear_velocity = _track_accel/kP;
377  }
378 
379  // let the limited_speed_xy_cms be some range above or below current velocity along track
380  if (speed_along_track < -linear_velocity) {
381  // we are traveling fast in the opposite direction of travel to the waypoint so do not move the intermediate point
382  _limited_speed_xy_cms = 0;
383  }else{
384  // increase intermediate target point's velocity if not yet at the leash limit
385  if(dt > 0 && !reached_leash_limit) {
386  _limited_speed_xy_cms += 2.0f * _track_accel * dt;
387  }
388  // do not allow speed to be below zero or over top speed
389  _limited_speed_xy_cms = constrain_float(_limited_speed_xy_cms, 0.0f, _track_speed);
390 
391  // check if we should begin slowing down
392  if (!_flags.fast_waypoint) {
393  float dist_to_dest = _track_length - _track_desired;
394  if (!_flags.slowing_down && dist_to_dest <= _slow_down_dist) {
395  _flags.slowing_down = true;
396  }
397  // if target is slowing down, limit the speed
398  if (_flags.slowing_down) {
399  _limited_speed_xy_cms = MIN(_limited_speed_xy_cms, get_slow_down_speed(dist_to_dest, _track_accel));
400  }
401  }
402 
403  // if our current velocity is within the linear velocity range limit the intermediate point's velocity to be no more than the linear_velocity above or below our current velocity
404  if (fabsf(speed_along_track) < linear_velocity) {
405  _limited_speed_xy_cms = constrain_float(_limited_speed_xy_cms,speed_along_track-linear_velocity,speed_along_track+linear_velocity);
406  }
407  }
408  // advance the current target
409  if (!reached_leash_limit) {
410  _track_desired += _limited_speed_xy_cms * dt;
411 
412  // reduce speed if we reach end of leash
413  if (_track_desired > track_desired_max) {
414  _track_desired = track_desired_max;
415  _limited_speed_xy_cms -= 2.0f * _track_accel * dt;
416  if (_limited_speed_xy_cms < 0.0f) {
417  _limited_speed_xy_cms = 0.0f;
418  }
419  }
420  }
421 
422  // do not let desired point go past the end of the track unless it's a fast waypoint
423  if (!_flags.fast_waypoint) {
424  _track_desired = constrain_float(_track_desired, 0, _track_length);
425  } else {
426  _track_desired = constrain_float(_track_desired, 0, _track_length + WPNAV_WP_FAST_OVERSHOOT_MAX);
427  }
428 
429  // recalculate the desired position
430  Vector3f final_target = _origin + _pos_delta_unit * _track_desired;
431  // convert final_target.z to altitude above the ekf origin
432  final_target.z += terr_offset;
433  _pos_control.set_pos_target(final_target);
434 
435  // check if we've reached the waypoint
436  if( !_flags.reached_destination ) {
437  if( _track_desired >= _track_length ) {
438  // "fast" waypoints are complete once the intermediate point reaches the destination
439  if (_flags.fast_waypoint) {
440  _flags.reached_destination = true;
441  }else{
442  // regular waypoints also require the copter to be within the waypoint radius
443  Vector3f dist_to_dest = (curr_pos - Vector3f(0,0,terr_offset)) - _destination;
444  if( dist_to_dest.length() <= _wp_radius_cm ) {
445  _flags.reached_destination = true;
446  }
447  }
448  }
449  }
450 
451  // update the target yaw if origin and destination are at least 2m apart horizontally
452  if (_track_length_xy >= WPNAV_YAW_DIST_MIN) {
453  if (_pos_control.get_leash_xy() < WPNAV_YAW_DIST_MIN) {
454  // if the leash is short (i.e. moving slowly) and destination is at least 2m horizontally, point along the segment from origin to destination
455  set_yaw_cd(get_bearing_cd(_origin, _destination));
456  } else {
457  Vector3f horiz_leash_xy = final_target - curr_pos;
458  horiz_leash_xy.z = 0;
459  if (horiz_leash_xy.length() > MIN(WPNAV_YAW_DIST_MIN, _pos_control.get_leash_xy()*WPNAV_YAW_LEASH_PCT_MIN)) {
460  set_yaw_cd(RadiansToCentiDegrees(atan2f(horiz_leash_xy.y,horiz_leash_xy.x)));
461  }
462  }
463  }
464 
465  // successfully advanced along track
466  return true;
467 }

527 void AC_WPNav::calculate_wp_leash_length()
528 {
529  // length of the unit direction vector in the horizontal
530  float pos_delta_unit_xy = norm(_pos_delta_unit.x, _pos_delta_unit.y);
531  float pos_delta_unit_z = fabsf(_pos_delta_unit.z);
532 
533  float speed_z;
534  float leash_z;
535  if (_pos_delta_unit.z >= 0.0f) {
536  speed_z = _pos_control.get_max_speed_up();
537  leash_z = _pos_control.get_leash_up_z();
538  }else{
539  speed_z = fabsf(_pos_control.get_max_speed_down());
540  leash_z = _pos_control.get_leash_down_z();
541  }
542 
543  // calculate the maximum acceleration, maximum velocity, and leash length in the direction of travel
544  if(is_zero(pos_delta_unit_z) && is_zero(pos_delta_unit_xy)){
545  _track_accel = 0;
546  _track_speed = 0;
547  _track_leash_length = WPNAV_LEASH_LENGTH_MIN;
548  }else if(is_zero(_pos_delta_unit.z)){
549  _track_accel = _wp_accel_cmss/pos_delta_unit_xy;
550  _track_speed = _pos_control.get_max_speed_xy() / pos_delta_unit_xy;
551  _track_leash_length = _pos_control.get_leash_xy()/pos_delta_unit_xy;
552  }else if(is_zero(pos_delta_unit_xy)){
553  _track_accel = _wp_accel_z_cmss/pos_delta_unit_z;
554  _track_speed = speed_z/pos_delta_unit_z;
555  _track_leash_length = leash_z/pos_delta_unit_z;
556  }else{
557  _track_accel = MIN(_wp_accel_z_cmss/pos_delta_unit_z, _wp_accel_cmss/pos_delta_unit_xy);
558  _track_speed = MIN(speed_z/pos_delta_unit_z, _pos_control.get_max_speed_xy() / pos_delta_unit_xy);
559  _track_leash_length = MIN(leash_z/pos_delta_unit_z, _pos_control.get_leash_xy()/pos_delta_unit_xy);
560  }
561 
562  // calculate slow down distance (the distance from the destination when the target point should begin to slow down)
563  calc_slow_down_distance(_track_speed, _track_accel);
564 
565  // set recalc leash flag to false
566  _flags.recalc_wp_leash = false;

656 bool AC_WPNav::set_spline_origin_and_destination(const Vector3f& origin, const Vector3f& destination, bool terrain_alt, bool stopped_at_start, spline_segment_end_type seg_end_type, const Vector3f& next_destination)
657 {
658  // mission is "active" if wpnav has been called recently and vehicle reached the previous waypoint
659  bool prev_segment_exists = (_flags.reached_destination && ((AP_HAL::millis() - _wp_last_update) < 1000));
660 
661  // get dt from pos controller
662  float dt = _pos_control.get_dt();
663 
664  // check _wp_accel_cmss is reasonable to avoid divide by zero
665  if (_wp_accel_cmss <= 0) {
666  _wp_accel_cmss.set_and_save(WPNAV_ACCELERATION);
667  }
668 
669  // segment start types
670  // stop - vehicle is not moving at origin
671  // straight-fast - vehicle is moving, previous segment is straight. vehicle will fly straight through the waypoint before beginning it's spline path to the next wp
672  // _flag.segment_type holds whether prev segment is straight vs spline but we don't know if it has a delay
673  // spline-fast - vehicle is moving, previous segment is splined, vehicle will fly through waypoint but previous segment should have it flying in the correct direction (i.e. exactly parallel to position difference vector from previous segment's origin to this segment's destination)
674 
675  // calculate spline velocity at origin
676  if (stopped_at_start || !prev_segment_exists) {
677  // if vehicle is stopped at the origin, set origin velocity to 0.02 * distance vector from origin to destination
678  _spline_origin_vel = (destination - origin) * dt;
679  _spline_time = 0.0f;
680  _spline_vel_scaler = 0.0f;
681  }else{
682  // look at previous segment to determine velocity at origin
683  if (_flags.segment_type == SEGMENT_STRAIGHT) {
684  // previous segment is straight, vehicle is moving so vehicle should fly straight through the origin
685  // before beginning it's spline path to the next waypoint. Note: we are using the previous segment's origin and destination
686  _spline_origin_vel = (_destination - _origin);
687  _spline_time = 0.0f; // To-Do: this should be set based on how much overrun there was from straight segment?
688  _spline_vel_scaler = _pos_control.get_vel_target().length(); // start velocity target from current target velocity
689  }else{
690  // previous segment is splined, vehicle will fly through origin
691  // we can use the previous segment's destination velocity as this segment's origin velocity
692  // Note: previous segment will leave destination velocity parallel to position difference vector
693  // from previous segment's origin to this segment's destination)
694  _spline_origin_vel = _spline_destination_vel;
695  if (_spline_time > 1.0f && _spline_time < 1.1f) { // To-Do: remove hard coded 1.1f
696  _spline_time -= 1.0f;
697  }else{
698  _spline_time = 0.0f;
699  }
700  // Note: we leave _spline_vel_scaler as it was from end of previous segment
701  }
702  }
703 
704  // calculate spline velocity at destination
705  switch (seg_end_type) {
706 
707  case SEGMENT_END_STOP:
708  // if vehicle stops at the destination set destination velocity to 0.02 * distance vector from origin to destination
709  _spline_destination_vel = (destination - origin) * dt;
710  _flags.fast_waypoint = false;
711  break;
712 
713  case SEGMENT_END_STRAIGHT:
714  // if next segment is straight, vehicle's final velocity should face along the next segment's position
715  _spline_destination_vel = (next_destination - destination);
716  _flags.fast_waypoint = true;
717  break;
718 
719  case SEGMENT_END_SPLINE:
720  // if next segment is splined, vehicle's final velocity should face parallel to the line from the origin to the next destination
721  _spline_destination_vel = (next_destination - origin);
722  _flags.fast_waypoint = true;
723  break;
724  }
725 
726  // code below ensures we don't get too much overshoot when the next segment is short
727  float vel_len = _spline_origin_vel.length() + _spline_destination_vel.length();
728  float pos_len = (destination - origin).length() * 4.0f;
729  if (vel_len > pos_len) {
730  // if total start+stop velocity is more than twice position difference
731  // use a scaled down start and stop velocityscale the start and stop velocities down
732  float vel_scaling = pos_len / vel_len;
733  // update spline calculator
734  update_spline_solution(origin, destination, _spline_origin_vel * vel_scaling, _spline_destination_vel * vel_scaling);
735  }else{
736  // update spline calculator
737  update_spline_solution(origin, destination, _spline_origin_vel, _spline_destination_vel);
738  }
739 
740  // store origin and destination locations
741  _origin = origin;
742  _destination = destination;
743  _terrain_alt = terrain_alt;
744 
745  // calculate slow down distance
746  calc_slow_down_distance(_pos_control.get_max_speed_xy(), _wp_accel_cmss);
747 
748  // get alt-above-terrain
749  float terr_offset = 0.0f;
750  if (terrain_alt) {
751  if (!get_terrain_offset(terr_offset)) {
752  return false;
753  }
754  }
755 
756  // initialise intermediate point to the origin
757  _pos_control.set_pos_target(origin + Vector3f(0,0,terr_offset));
758  _flags.reached_destination = false;
759  _flags.segment_type = SEGMENT_SPLINE;
760  _flags.new_wp_destination = true; // flag new waypoint so we can freeze the pos controller's feed forward and smooth the transition
761  _flags.wp_yaw_set = false;
762 
763  // initialise yaw related variables
764  _track_length_xy = safe_sqrt(sq(_destination.x - _origin.x)+sq(_destination.y - _origin.y)); // horizontal track length (used to decide if we should update yaw)
765 
766  return true;
767 }

804 void AC_WPNav::update_spline_solution(const Vector3f& origin, const Vector3f& dest, const Vector3f& origin_vel, const Vector3f& dest_vel)
805 {

813 bool AC_WPNav::advance_spline_target_along_track(float dt)
814 {
815  if (!_flags.reached_destination) {
816  Vector3f target_pos, target_vel;
817 
818  // update target position and velocity from spline calculator
819  calc_spline_pos_vel(_spline_time, target_pos, target_vel);
820 
821  // if target velocity is zero the origin and destination must be the same
822  // so flag reached destination (and protect against divide by zero)
823  float target_vel_length = target_vel.length();
824  if (is_zero(target_vel_length)) {
825  _flags.reached_destination = true;
826  return true;
827  }
828 
829  _pos_delta_unit = target_vel / target_vel_length;
830  calculate_wp_leash_length();
831 
832  // get current location
833  Vector3f curr_pos = _inav.get_position();
834 
835  // get terrain altitude offset for origin and current position (i.e. change in terrain altitude from a position vs ekf origin)
836  float terr_offset = 0.0f;
837  if (_terrain_alt && !get_terrain_offset(terr_offset)) {
838  return false;
839  }
840 
841  // calculate position error
842  Vector3f track_error = curr_pos - target_pos;
843  track_error.z -= terr_offset;
844 
845  // calculate the horizontal error
846  _track_error_xy = norm(track_error.x, track_error.y);
847 
848  // calculate the vertical error
849  float track_error_z = fabsf(track_error.z);
850 
851  // get position control leash lengths
852  float leash_xy = _pos_control.get_leash_xy();
853  float leash_z;
854  if (track_error.z >= 0) {
855  leash_z = _pos_control.get_leash_up_z();
856  }else{
857  leash_z = _pos_control.get_leash_down_z();
858  }
859 
860  // calculate how far along the track we could move the intermediate target before reaching the end of the leash
861  float track_leash_slack = MIN(_track_leash_length*(leash_z-track_error_z)/leash_z, _track_leash_length*(leash_xy-_track_error_xy)/leash_xy);
862  if (track_leash_slack < 0.0f) {
863  track_leash_slack = 0.0f;
864  }
865 
866  // update velocity
867  float spline_dist_to_wp = (_destination - target_pos).length();
868  float vel_limit = _pos_control.get_max_speed_xy();
869  if (!is_zero(dt)) {
870  vel_limit = MIN(vel_limit, track_leash_slack/dt);
871  }
872 
873  // if within the stopping distance from destination, set target velocity to sqrt of distance * 2 * acceleration
874  if (!_flags.fast_waypoint && spline_dist_to_wp < _slow_down_dist) {
875  _spline_vel_scaler = safe_sqrt(spline_dist_to_wp * 2.0f * _wp_accel_cmss);
876  }else if(_spline_vel_scaler < vel_limit) {
877  // increase velocity using acceleration
878  _spline_vel_scaler += _wp_accel_cmss * dt;
879  }
880 
881  // constrain target velocity
882  _spline_vel_scaler = constrain_float(_spline_vel_scaler, 0.0f, vel_limit);
883 
884  // scale the spline_time by the velocity we've calculated vs the velocity that came out of the spline calculator
885  _spline_time_scale = _spline_vel_scaler / target_vel_length;
886 
887  // update target position
888  target_pos.z += terr_offset;
889  _pos_control.set_pos_target(target_pos);
890 
891  // update the target yaw if origin and destination are at least 2m apart horizontally
892  if (_track_length_xy >= WPNAV_YAW_DIST_MIN) {
893  if (_pos_control.get_leash_xy() < WPNAV_YAW_DIST_MIN) {
894  // if the leash is very short (i.e. flying at low speed) use the target point's velocity along the track
895  if (!is_zero(target_vel.x) && !is_zero(target_vel.y)) {
896  set_yaw_cd(RadiansToCentiDegrees(atan2f(target_vel.y,target_vel.x)));
897  }
898  } else {
899  // point vehicle along the leash (i.e. point vehicle towards target point on the segment from origin to destination)
900  float track_error_xy_length = safe_sqrt(sq(track_error.x)+sq(track_error.y));
901  if (track_error_xy_length > MIN(WPNAV_YAW_DIST_MIN, _pos_control.get_leash_xy()*WPNAV_YAW_LEASH_PCT_MIN)) {
902  // To-Do: why is track_error sign reversed?
903  set_yaw_cd(RadiansToCentiDegrees(atan2f(-track_error.y,-track_error.x)));
904  }
905  }
906  }
907 
908  // advance spline time to next step
909  _spline_time += _spline_time_scale*dt;
910 
911  // we will reach the next waypoint in the next step so set reached_destination flag
912  // To-Do: is this one step too early?
913  if (_spline_time >= 1.0f) {
914  _flags.reached_destination = true;
915  }
916  }
917  return true;
918 }

920 // calc_spline_pos_vel_accel - calculates target position, velocity and acceleration for the given "spline_time"
922 void AC_WPNav::calc_spline_pos_vel(float spline_time, Vector3f& position, Vector3f& velocity)
923 {
924  float spline_time_sqrd = spline_time * spline_time;
925  float spline_time_cubed = spline_time_sqrd * spline_time;
926 
927  position = _hermite_spline_solution[0] + \
928  _hermite_spline_solution[1] * spline_time + \
929  _hermite_spline_solution[2] * spline_time_sqrd + \
930  _hermite_spline_solution[3] * spline_time_cubed;
931 
932  velocity = _hermite_spline_solution[1] + \
933  _hermite_spline_solution[2] * 2.0f * spline_time + \
934  _hermite_spline_solution[3] * 3.0f * spline_time_sqrd;
