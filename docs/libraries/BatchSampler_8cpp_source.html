171 void AP_InertialSensor::BatchSampler::push_data_to_log()
172 {
173  if (!initialised) {
174  return;
175  }
176  if (_sensor_mask == 0) {
177  return;
178  }
179  if (data_write_offset - data_read_offset < samples_per_msg) {
180  // insuffucient data to pack a packet
181  return;
182  }
183  if (AP_HAL::millis() - last_sent_ms < (uint16_t)push_interval_ms) {
184  // avoid flooding AP_Logger's buffer
185  return;
186  }
187  AP_Logger *logger = AP_Logger::get_singleton();
188  if (logger == nullptr) {
189  // should not have been called
190  return;
191  }
192 
193  // possibly send isb header:
194  if (!isbh_sent && data_read_offset == 0) {
195  float sample_rate = 0; // avoid warning about uninitialised values
196  switch(type) {
197  case IMU_SENSOR_TYPE_GYRO:
198  sample_rate = _imu._gyro_raw_sample_rates[instance];
199  if (_doing_sensor_rate_logging) {
200  sample_rate *= _imu._gyro_over_sampling[instance];
201  }
202  break;
203  case IMU_SENSOR_TYPE_ACCEL:
204  sample_rate = _imu._accel_raw_sample_rates[instance];
205  if (_doing_sensor_rate_logging) {
206  sample_rate *= _imu._accel_over_sampling[instance];
207  }
208  break;
209  }
210  if (!logger->Write_ISBH(isb_seqnum,
211  type,
212  instance,
213  multiplier,
214  _required_count,
215  measurement_started_us,
216  sample_rate)) {
217  // buffer full?
218  return;
219  }
220  isbh_sent = true;
221  }
222  // pack and send a data packet:
223  if (!logger->Write_ISBD(isb_seqnum,
224  data_read_offset/samples_per_msg,

226  &data_y[data_read_offset],
227  &data_z[data_read_offset])) {
228  // maybe later?!
229  return;
230  }
231  data_read_offset += samples_per_msg;

233  if (data_read_offset >= _required_count) {
234  // that was the last one. Clean up:
235  data_read_offset = 0;
236  isb_seqnum++;
