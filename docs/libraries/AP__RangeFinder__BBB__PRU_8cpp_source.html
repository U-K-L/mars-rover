47 /*
48  Stop PRU, load firmware (check if firmware is present), start PRU.
49  If we get a result the sensor seems to be there.
50 */
51 bool AP_RangeFinder_BBB_PRU::detect()
52 {
53  bool result = true;
54  uint32_t mem_fd;
55  uint32_t *ctrl;
56  void *ram;
57 
58  mem_fd = open("/dev/mem", O_RDWR | O_SYNC | O_CLOEXEC);
59  ctrl = (uint32_t*)mmap(0, 0x1000, PROT_READ | PROT_WRITE, MAP_SHARED, mem_fd, PRU0_CTRL_BASE);
60  ram = mmap(0, PRU0_IRAM_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED, mem_fd, PRU0_IRAM_BASE);
61 
62  // Reset PRU 0
63  *ctrl = 0;
64  hal.scheduler->delay(1);
65 
66  // Load firmware (.text)
67  FILE *file = fopen("/lib/firmware/rangefinderprutext.bin", "rb");
68  if(file == nullptr)
69  {
70  result = false;
71  }
72 
73  if(fread(ram, PRU0_IRAM_SIZE, 1, file) != 1)
74  {
75  result = false;
76  }
77 
78  fclose(file);
79 
80  munmap(ram, PRU0_IRAM_SIZE);
81 
82  ram = mmap(0, PRU0_DRAM_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED, mem_fd, PRU0_DRAM_BASE);
83 
84  // Load firmware (.data)
85  file = fopen("/lib/firmware/rangefinderprudata.bin", "rb");
86  if(file == nullptr)
87  {
88  result = false;
89  }
90 
91  if(fread(ram, PRU0_DRAM_SIZE, 1, file) != 1)
92  {
93  result = false;
94  }
95 
96  fclose(file);
97 
98  munmap(ram, PRU0_DRAM_SIZE);
99 
100  // Map PRU RAM
101  ram = mmap(0, 0x1000, PROT_READ | PROT_WRITE, MAP_SHARED, mem_fd, PRU0_DRAM_BASE);
102  close(mem_fd);
103 
104  // Start PRU 0
105  *ctrl = 2;
106 
107  rangerpru = (volatile struct range*)ram;
108 
109  return result;
110 }
