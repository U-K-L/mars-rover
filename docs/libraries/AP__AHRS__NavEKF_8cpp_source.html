89 void AP_AHRS_NavEKF::update(bool skip_ins_update)
90 {
91  // support locked access functions to AHRS data
92  WITH_SEMAPHORE(_rsem);
93 
94  // drop back to normal priority if we were boosted by the INS
95  // calling delay_microseconds_boost()
96  hal.scheduler->boost_end();
97 
98  // EKF1 is no longer supported - handle case where it is selected
99  if (_ekf_type == 1) {
100  _ekf_type.set(2);
101  }
102 
103  update_DCM(skip_ins_update);
104 
105 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
106  update_SITL();
107 #endif
108 
109  if (_ekf_type == 2) {
110  // if EK2 is primary then run EKF2 first to give it CPU
111  // priority
112  update_EKF2();
113  update_EKF3();
114  } else {
115  // otherwise run EKF3 first
116  update_EKF3();
117  update_EKF2();
118  }
119 
120 #if AP_MODULE_SUPPORTED
121  // call AHRS_update hook if any
122  AP_Module::call_hook_AHRS_update(*this);
123 #endif
124 
125  // push gyros if optical flow present
126  if (hal.opticalflow) {
127  const Vector3f &exported_gyro_bias = get_gyro_drift();
128  hal.opticalflow->push_gyro_bias(exported_gyro_bias.x, exported_gyro_bias.y);
129  }
130 
131  if (_view != nullptr) {
132  // update optional alternative attitude view
133  _view->update(skip_ins_update);
134  }
135 
136 #if !HAL_MINIMIZE_FEATURES && AP_AHRS_NAVEKF_AVAILABLE
137  // update NMEA output
138  update_nmea_out();
139 #endif
140 }

158 void AP_AHRS_NavEKF::update_EKF2(void)
159 {
160  if (!_ekf2_started) {
161  // wait 1 second for DCM to output a valid tilt error estimate
162  if (start_time_ms == 0) {
163  start_time_ms = AP_HAL::millis();
164  }
165  if (AP_HAL::millis() - start_time_ms > startup_delay_ms || _force_ekf) {
166  _ekf2_started = EKF2.InitialiseFilter();
167  if (_force_ekf) {
168  return;
169  }
170  }
171  }
172  if (_ekf2_started) {
173  EKF2.UpdateFilter();
174  if (active_EKF_type() == EKF_TYPE2) {
175  Vector3f eulers;
176  EKF2.getRotationBodyToNED(_dcm_matrix);
177  EKF2.getEulerAngles(-1,eulers);
178  roll = eulers.x;
179  pitch = eulers.y;
180  yaw = eulers.z;
181 
182  update_cd_values();
183  update_trig();
184 
185  // Use the primary EKF to select the primary gyro
186  const int8_t primary_imu = EKF2.getPrimaryCoreIMUIndex();
187 
188  const AP_InertialSensor &_ins = AP::ins();
189 
190  // get gyro bias for primary EKF and change sign to give gyro drift
191  // Note sign convention used by EKF is bias = measurement - truth
192  _gyro_drift.zero();
193  EKF2.getGyroBias(-1,_gyro_drift);
194  _gyro_drift = -_gyro_drift;
195 
196  // calculate corrected gyro estimate for get_gyro()
197  _gyro_estimate.zero();
198  if (primary_imu == -1 || !_ins.get_gyro_health(primary_imu)) {
199  // the primary IMU is undefined so use an uncorrected default value from the INS library
200  _gyro_estimate = _ins.get_gyro();
201  } else {
202  // use the same IMU as the primary EKF and correct for gyro drift
203  _gyro_estimate = _ins.get_gyro(primary_imu) + _gyro_drift;
204  }
205 
206  // get z accel bias estimate from active EKF (this is usually for the primary IMU)
207  float abias = 0;
208  EKF2.getAccelZBias(-1,abias);
209 
210  // This EKF is currently using primary_imu, and abias applies to only that IMU
211  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
212  Vector3f accel = _ins.get_accel(i);
213  if (i == primary_imu) {
214  accel.z -= abias;
215  }
216  if (_ins.get_accel_health(i)) {
217  _accel_ef_ekf[i] = _dcm_matrix * get_rotation_autopilot_body_to_vehicle_body() * accel;
218  }
219  }
220  _accel_ef_ekf_blended = _accel_ef_ekf[primary_imu>=0?primary_imu:_ins.get_primary_accel()];
221  nav_filter_status filt_state;
222  EKF2.getFilterStatus(-1,filt_state);
223  AP_Notify::flags.gps_fusion = filt_state.flags.using_gps; // Drives AP_Notify flag for usable GPS.
224  AP_Notify::flags.gps_glitching = filt_state.flags.gps_glitching;
225  AP_Notify::flags.have_pos_abs = filt_state.flags.horiz_pos_abs;
226  }
227  }

231 void AP_AHRS_NavEKF::update_EKF3(void)
232 {
233  if (!_ekf3_started) {
234  // wait 1 second for DCM to output a valid tilt error estimate
235  if (start_time_ms == 0) {
236  start_time_ms = AP_HAL::millis();
237  }
238  if (AP_HAL::millis() - start_time_ms > startup_delay_ms || _force_ekf) {
239  _ekf3_started = EKF3.InitialiseFilter();
240  if (_force_ekf) {
241  return;
242  }
243  }
244  }
245  if (_ekf3_started) {
246  EKF3.UpdateFilter();
247  if (active_EKF_type() == EKF_TYPE3) {
248  Vector3f eulers;
249  EKF3.getRotationBodyToNED(_dcm_matrix);
250  EKF3.getEulerAngles(-1,eulers);
251  roll = eulers.x;
252  pitch = eulers.y;
253  yaw = eulers.z;
254 
255  update_cd_values();
256  update_trig();
257 
258  const AP_InertialSensor &_ins = AP::ins();
259 
260  // Use the primary EKF to select the primary gyro
261  const int8_t primary_imu = EKF3.getPrimaryCoreIMUIndex();
262 
263  // get gyro bias for primary EKF and change sign to give gyro drift
264  // Note sign convention used by EKF is bias = measurement - truth
265  _gyro_drift.zero();
266  EKF3.getGyroBias(-1,_gyro_drift);
267  _gyro_drift = -_gyro_drift;
268 
269  // calculate corrected gyro estimate for get_gyro()
270  _gyro_estimate.zero();
271  if (primary_imu == -1 || !_ins.get_gyro_health(primary_imu)) {
272  // the primary IMU is undefined so use an uncorrected default value from the INS library
273  _gyro_estimate = _ins.get_gyro();
274  } else {
275  // use the same IMU as the primary EKF and correct for gyro drift
276  _gyro_estimate = _ins.get_gyro(primary_imu) + _gyro_drift;
277  }
278 
279  // get 3-axis accel bias festimates for active EKF (this is usually for the primary IMU)
280  Vector3f abias;
281  EKF3.getAccelBias(-1,abias);
282 
283  // This EKF uses the primary IMU
284  // Eventually we will run a separate instance of the EKF for each IMU and do the selection and blending of EKF outputs upstream
285  // update _accel_ef_ekf
286  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
287  Vector3f accel = _ins.get_accel(i);
288  if (i==_ins.get_primary_accel()) {
289  accel -= abias;
290  }
291  if (_ins.get_accel_health(i)) {
292  _accel_ef_ekf[i] = _dcm_matrix * accel;
293  }
294  }
295  _accel_ef_ekf_blended = _accel_ef_ekf[_ins.get_primary_accel()];
296  nav_filter_status filt_state;
297  EKF3.getFilterStatus(-1,filt_state);
298  AP_Notify::flags.gps_fusion = filt_state.flags.using_gps; // Drives AP_Notify flag for usable GPS.
299  AP_Notify::flags.gps_glitching = filt_state.flags.gps_glitching;
300  AP_Notify::flags.have_pos_abs = filt_state.flags.horiz_pos_abs;
301  }
302  }

306 void AP_AHRS_NavEKF::update_SITL(void)
307 {
308  if (_sitl == nullptr) {
309  _sitl = AP::sitl();
310  if (_sitl == nullptr) {
311  return;
312  }
313  }
314 
315  const struct SITL::sitl_fdm &fdm = _sitl->state;
316  const AP_InertialSensor &_ins = AP::ins();
317 
318  if (active_EKF_type() == EKF_TYPE_SITL) {
319 
320  fdm.quaternion.rotation_matrix(_dcm_matrix);
321  _dcm_matrix = _dcm_matrix * get_rotation_vehicle_body_to_autopilot_body();
322  _dcm_matrix.to_euler(&roll, &pitch, &yaw);
323 
324  update_cd_values();
325  update_trig();
326 
327  _gyro_drift.zero();
328 
329  _gyro_estimate = _ins.get_gyro();
330 
331  for (uint8_t i=0; i<INS_MAX_INSTANCES; i++) {
332  const Vector3f &accel = _ins.get_accel(i);
333  _accel_ef_ekf[i] = _dcm_matrix * get_rotation_autopilot_body_to_vehicle_body() * accel;
334  }
335  _accel_ef_ekf_blended = _accel_ef_ekf[0];
336 
337  }
338 
339  if (_sitl->odom_enable) {
340  // use SITL states to write body frame odometry data at 20Hz
341  uint32_t timeStamp_ms = AP_HAL::millis();
342  if (timeStamp_ms - _last_body_odm_update_ms > 50) {
343  const float quality = 100.0f;
344  const Vector3f posOffset(0.0f, 0.0f, 0.0f);
345  const float delTime = 0.001f * (timeStamp_ms - _last_body_odm_update_ms);
346  _last_body_odm_update_ms = timeStamp_ms;
347  timeStamp_ms -= (timeStamp_ms - _last_body_odm_update_ms)/2; // correct for first order hold average delay
348  Vector3f delAng = _ins.get_gyro();
349 
350  delAng *= delTime;
351  // rotate earth velocity into body frame and calculate delta position
352  Matrix3f Tbn;
353  Tbn.from_euler(radians(fdm.rollDeg),radians(fdm.pitchDeg),radians(fdm.yawDeg));
354  const Vector3f earth_vel(fdm.speedN,fdm.speedE,fdm.speedD);
355  const Vector3f delPos = Tbn.transposed() * (earth_vel * delTime);
356  // write to EKF
357  EKF3.writeBodyFrameOdom(quality, delPos, delAng, delTime, timeStamp_ms, posOffset);
358  }
359  }
360 }

966 AP_AHRS_NavEKF::EKF_TYPE AP_AHRS_NavEKF::active_EKF_type(void) const
967 {
968  EKF_TYPE ret = EKF_TYPE_NONE;
969 
970  switch (ekf_type()) {
971  case 0:
972  return EKF_TYPE_NONE;
973 
974  case 2: {
975  // do we have an EKF2 yet?
976  if (!_ekf2_started) {
977  return EKF_TYPE_NONE;
978  }
979  if (always_use_EKF()) {
980  uint16_t ekf2_faults;
981  EKF2.getFilterFaults(-1,ekf2_faults);
982  if (ekf2_faults == 0) {
983  ret = EKF_TYPE2;
984  }
985  } else if (EKF2.healthy()) {
986  ret = EKF_TYPE2;
987  }
988  break;
989  }
990 
991  case 3: {
992  // do we have an EKF3 yet?
993  if (!_ekf3_started) {
994  return EKF_TYPE_NONE;
995  }
996  if (always_use_EKF()) {
997  uint16_t ekf3_faults;
998  EKF3.getFilterFaults(-1,ekf3_faults);
999  if (ekf3_faults == 0) {
1000  ret = EKF_TYPE3;
1001  }
1002  } else if (EKF3.healthy()) {
1003  ret = EKF_TYPE3;
1004  }
1005  break;
1006  }
1007 
1008 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1009  case EKF_TYPE_SITL:
1010  ret = EKF_TYPE_SITL;
1011  break;
1012 #endif
1013  }
1014 
1015  /*
1016  fixed wing and rover when in fly_forward mode will fall back to
1017  DCM if the EKF doesn't have GPS. This is the safest option as
1018  DCM is very robust. Note that we also check the filter status
1019  when fly_forward is false and we are disarmed. This is to ensure
1020  that the arming checks do wait for good GPS position on fixed
1021  wing and rover
1022  */
1023  if (ret != EKF_TYPE_NONE &&
1024  (_vehicle_class == AHRS_VEHICLE_FIXED_WING ||
1025  _vehicle_class == AHRS_VEHICLE_GROUND) &&
1026  (_flags.fly_forward || !hal.util->get_soft_armed())) {
1027  nav_filter_status filt_state;
1028  if (ret == EKF_TYPE2) {
1029  EKF2.getFilterStatus(-1,filt_state);
1030  } else if (ret == EKF_TYPE3) {
1031  EKF3.getFilterStatus(-1,filt_state);
1032 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
1033  } else if (ret == EKF_TYPE_SITL) {
1034  get_filter_status(filt_state);
1035 #endif
1036  }
1037  if (hal.util->get_soft_armed() && !filt_state.flags.using_gps && AP::gps().status() >= AP_GPS::GPS_OK_FIX_3D) {
1038  // if the EKF is not fusing GPS and we have a 3D lock, then
1039  // plane and rover would prefer to use the GPS position from
1040  // DCM. This is a safety net while some issues with the EKF
1041  // get sorted out
1042  return EKF_TYPE_NONE;
1043  }
1044  if (hal.util->get_soft_armed() && filt_state.flags.const_pos_mode) {
1045  return EKF_TYPE_NONE;
1046  }
1047  if (!filt_state.flags.attitude ||
1048  !filt_state.flags.vert_vel ||
1049  !filt_state.flags.vert_pos) {
1050  return EKF_TYPE_NONE;
1051  }
1052  if (!filt_state.flags.horiz_vel ||
1053  (!filt_state.flags.horiz_pos_abs && !filt_state.flags.horiz_pos_rel)) {
1054  if ((!_compass || !_compass->use_for_yaw()) &&
1055  AP::gps().status() >= AP_GPS::GPS_OK_FIX_3D &&
1056  AP::gps().ground_speed() < 2) {
1057  /*
1058  special handling for non-compass mode when sitting
1059  still. The EKF may not yet have aligned its yaw. We
1060  accept EKF as healthy to allow arming. Once we reach
1061  speed the EKF should get yaw alignment
1062  */
1063  if (filt_state.flags.gps_quality_good) {
1064  return ret;
1065  }
1066  }
1067  return EKF_TYPE_NONE;
1068  }
1069  }
1070  return ret;
1071 }
