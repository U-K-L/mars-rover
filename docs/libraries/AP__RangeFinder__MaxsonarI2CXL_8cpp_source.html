46 /*
47  detect if a Maxbotix rangefinder is connected. We'll detect by
48  trying to take a reading on I2C. If we get a result the sensor is

51 AP_RangeFinder_Backend *AP_RangeFinder_MaxsonarI2CXL::detect(RangeFinder::RangeFinder_State &_state,
52  AP_RangeFinder_Params &_params,
53  AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev)
54 {
55  if (!dev) {
56  return nullptr;

73 /*
74  initialise sensor
75  */
76 bool AP_RangeFinder_MaxsonarI2CXL::_init(void)
77 {
78  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
79  return false;
80  }
81 
82  if (!start_reading()) {
83  _dev->get_semaphore()->give();
84  return false;
85  }
86 
87  // give time for the sensor to process the request
88  hal.scheduler->delay(100);
89 
90  uint16_t reading_cm;
91  if (!get_reading(reading_cm)) {
92  _dev->get_semaphore()->give();
93  return false;
94  }
95 
96  _dev->get_semaphore()->give();
97 
98  _dev->register_periodic_callback(100000,
99  FUNCTOR_BIND_MEMBER(&AP_RangeFinder_MaxsonarI2CXL::_timer, void));
100 
101  return true;
102 }

104 // start_reading() - ask sensor to make a range reading
105 bool AP_RangeFinder_MaxsonarI2CXL::start_reading()
106 {
107  uint8_t cmd = AP_RANGE_FINDER_MAXSONARI2CXL_COMMAND_TAKE_RANGE_READING;
108 
109  // send command to take reading
110  return _dev->transfer(&cmd, sizeof(cmd), nullptr, 0);
111 }

113 // read - return last value measured by sensor
114 bool AP_RangeFinder_MaxsonarI2CXL::get_reading(uint16_t &reading_cm)
115 {
116  be16_t val;
117 
118  // take range reading and read back results
119  bool ret = _dev->transfer(nullptr, 0, (uint8_t *) &val, sizeof(val));
120 
121  if (ret) {
122  // combine results into distance
123  reading_cm = be16toh(val);
124 
125  // trigger a new reading
126  start_reading();
127  }
128 
129  return ret;
130 }

132 /*

134 */
135 void AP_RangeFinder_MaxsonarI2CXL::_timer(void)
136 {
137  uint16_t d;
