92 // read - return last value measured by sensor
93 bool AP_RangeFinder_Wasp::get_reading(uint16_t &reading_cm) {
94  if (uart == nullptr) {
95  return false;
96  }
97 
98  // read any available lines from the lidar
99  float sum = 0;
100  uint16_t count = 0;
101  int16_t nbytes = uart->available();
102  while (nbytes-- > 0) {
103  char c = uart->read();
104  if (c == '\n') {
105  linebuf[linebuf_len] = 0;
106  linebuf_len = 0;
107  state.last_reading_ms = AP_HAL::millis();
108  if (isalpha(linebuf[0])) {
109  parse_response();
110  } else {
111  float read_value = (float)atof(linebuf);
112  if (read_value > 0) {
113  sum += read_value;
114  count++;
115  }
116  }
117  } else if (isalnum(c) || c == '.' || c == '-') {
118  linebuf[linebuf_len++] = c;
119  }
120 
121  // discard excessively long buffers
122  if (linebuf_len == sizeof(linebuf)) {
123  linebuf_len = 0;
124  }
125  }
126 
127  if (configuration_state == WASP_CFG_RATE && uart->tx_pending() == 0) {
128  configuration_state = WASP_CFG_ENCODING;
129  }
130 
131  if (count == 0) {
132  return false;
133  }
134 
135  reading_cm = 100 * sum / count;
136  set_status(RangeFinder::RangeFinder_Good);
137 
138  return true;
139 }

143 void AP_RangeFinder_Wasp::update(void) {
144  if (!get_reading(state.distance_cm)) {
145  set_status(RangeFinder::RangeFinder_NoData);
146  }
147 
148  if (AP_HAL::millis() - state.last_reading_ms > 500) {
149  // attempt to reconfigure on the assumption this was a bad baud setting
150  configuration_state = WASP_CFG_RATE;
151  }
152 
153  char command[COMMAND_BUFFER_LEN] = {};
154 
155  switch (configuration_state) {
156  case WASP_CFG_RATE:
157  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">BAUD %s\n", baud > 0 ? "HIGH" : "LOW");
158  break;
159  case WASP_CFG_ENCODING:
160  uart->end();
161  uart->begin(baud > 0 ? 921600 : 115200);
162  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">LBE LITTLE\n");
163  break;
164  case WASP_CFG_PROTOCOL:
165  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">FMT ASCII\n");
166  break;
167  case WASP_CFG_FRQ:
168  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">FRQ %d\n", constrain_int16(frq, 20, baud > 0 ? 10000 : 1440));
169  break;
170  case WASP_CFG_GO:
171  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">GO\n");
172  break;
173  case WASP_CFG_AUT:
174  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">AUT %d\n", thr >= 0 ? 0 : 1);
175  break;
176  case WASP_CFG_THR:
177  if (thr >= 0) {
178  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">THR %d\n", constrain_int16(thr, 0,255));
179  } else {
180  configuration_state = WASP_CFG_MAVG;
181  }
182  break;
183  case WASP_CFG_MAVG:
184  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">MAVG %d\n", constrain_int16(mavg, 0, 255));
185  break;
186  case WASP_CFG_MEDF:
187  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">MEDF %d\n", constrain_int16(medf, 0, 255));
188  break;
189  case WASP_CFG_AVG:
190  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">AVG %d\n", constrain_int16(avg, 0, 255));
191  break;
192  case WASP_CFG_AUV:
193  hal.util->snprintf(command, COMMAND_BUFFER_LEN, ">AUV 1\n");
194  break;
195  case WASP_CFG_DONE:
196  return;
197  }
198 
199  if (command[0] != 0) {
200  uart->write((uint8_t *)command, strlen(command));
201  }
202 }

204 void AP_RangeFinder_Wasp::parse_response(void) {
205  switch (configuration_state) {
206  case WASP_CFG_RATE:
207  configuration_state = WASP_CFG_ENCODING;
208  break;
209  case WASP_CFG_ENCODING:
210  if (strncmp(linebuf, "LBE", 3) == 0) {
211  configuration_state = WASP_CFG_PROTOCOL;
212  }
213  break;
214  case WASP_CFG_PROTOCOL:
215  if (strncmp(linebuf, "FMT", 3) == 0) {
216  configuration_state = WASP_CFG_FRQ;
217  }
218  break;
219  case WASP_CFG_FRQ:
220  if (strncmp(linebuf, "FRQ", 3) == 0) {
221  configuration_state = WASP_CFG_GO;
222  }
223  break;
224  case WASP_CFG_GO:
225  if (strncmp(linebuf, "GO", 2) == 0) {
226  configuration_state = WASP_CFG_AUT;

228  break;
