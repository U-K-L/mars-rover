186 void SRV_Channels::save_trim(void)
187 {
188  for (uint8_t i=0; i<NUM_SERVO_CHANNELS; i++) {
189  if (trimmed_mask & (1U<<i)) {
190  channels[i].servo_trim.set_and_save(channels[i].servo_trim.get());
191  }
192  }
193  trimmed_mask = 0;
194 }

196 void SRV_Channels::setup_failsafe_trim_all_non_motors(void)
197 {
198  for (uint8_t i = 0; i < NUM_SERVO_CHANNELS; i++) {
199  if (!SRV_Channel::is_motor(channels[i].get_function())) {
200  hal.rcout->set_failsafe_pwm(1U<<channels[i].ch_num, channels[i].servo_trim);
201  }
202  }

234 void SRV_Channels::push()
235 {
236  hal.rcout->push();
237 
238  // give volz library a chance to update
239  volz_ptr->update();
240 
241  // give sbus library a chance to update
242  sbus_ptr->update();
243 
244  // give robotis library a chance to update
245  robotis_ptr->update();
246 
247 #if HAL_SUPPORT_RCOUT_SERIAL
248  // give blheli telemetry a chance to update
249  blheli_ptr->update_telemetry();
250 #endif
251 
252 #if HAL_WITH_UAVCAN
253  // push outputs to CAN
254  uint8_t can_num_drivers = AP::can().get_num_drivers();
255  for (uint8_t i = 0; i < can_num_drivers; i++) {
256  switch (AP::can().get_protocol_type(i)) {
257  case AP_BoardConfig_CAN::Protocol_Type_UAVCAN: {
258  AP_UAVCAN *ap_uavcan = AP_UAVCAN::get_uavcan(i);
259  if (ap_uavcan == nullptr) {
260  continue;
261  }
262  ap_uavcan->SRV_push_servos();
263  break;
264  }
265  case AP_BoardConfig_CAN::Protocol_Type_KDECAN: {
266 // To be replaced with macro saying if KDECAN library is included
267 #if APM_BUILD_TYPE(APM_BUILD_ArduCopter) || APM_BUILD_TYPE(APM_BUILD_ArduPlane) || APM_BUILD_TYPE(APM_BUILD_ArduSub)
268  AP_KDECAN *ap_kdecan = AP_KDECAN::get_kdecan(i);
269  if (ap_kdecan == nullptr) {
270  continue;
271  }
272  ap_kdecan->update();
273  break;
274 #endif
275  }
276  case AP_BoardConfig_CAN::Protocol_Type_ToshibaCAN: {
277  AP_ToshibaCAN *ap_tcan = AP_ToshibaCAN::get_tcan(i);
278  if (ap_tcan == nullptr) {
279  continue;
280  }
281  ap_tcan->update();
282  break;
283  }
284  case AP_BoardConfig_CAN::Protocol_Type_None:
285  default:
286  break;
287  }
288  }
289 #endif // HAL_WITH_UAVCAN
