37 // initialise the driver
38 bool AP_RAMTRON::init(void)
39 {
40  dev = hal.spi->get_device("ramtron");
41  if (!dev) {
42  return false;
43  }
44  WITH_SEMAPHORE(dev->get_semaphore());
45 
46  struct rdid {
47  uint8_t manufacturer[6];
48  uint8_t memory;
49  uint8_t id1;
50  uint8_t id2;
51  } rdid;
52  if (!dev->read_registers(RAMTRON_RDID, (uint8_t *)&rdid, sizeof(rdid))) {
53  return false;
54  }
55 
56  for (uint8_t i=0; i<ARRAY_SIZE(ramtron_ids); i++) {
57  if (ramtron_ids[i].id1 == rdid.id1 &&
58  ramtron_ids[i].id2 == rdid.id2) {
59  id = i;
60  return true;
61  }
62  }
63  hal.console->printf("Unknown RAMTRON manufacturer=%02x memory=%02x id1=%02x id2=%02x\n",
64  rdid.manufacturer[0], rdid.memory, rdid.id1, rdid.id2);
65  return false;
66 }
