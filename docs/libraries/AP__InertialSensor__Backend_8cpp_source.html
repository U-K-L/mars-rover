49 /*
50  update the sensor rate for FIFO sensors
51 
52  FIFO sensors produce samples at a fixed rate, but the clock in the
53  sensor may vary slightly from the system clock. This slowly adjusts
54  the rate to the observed rate
55 */
56 void AP_InertialSensor_Backend::_update_sensor_rate(uint16_t &count, uint32_t &start_us, float &rate_hz) const
57 {
58  uint32_t now = AP_HAL::micros();
59  if (start_us == 0) {
60  count = 0;
61  start_us = now;
62  } else {
63  count++;
64  if (now - start_us > 1000000UL) {
65  float observed_rate_hz = count * 1.0e6f / (now - start_us);
66 #if SENSOR_RATE_DEBUG
67  printf("RATE: %.1f should be %.1f\n", observed_rate_hz, rate_hz);
68 #endif
69  float filter_constant = 0.98f;
70  float upper_limit = 1.05f;
71  float lower_limit = 0.95f;
72  if (AP_HAL::millis() < 30000) {
73  // converge quickly for first 30s, then more slowly
74  filter_constant = 0.8f;
75  upper_limit = 2.0f;
76  lower_limit = 0.5f;
77  }
78  observed_rate_hz = constrain_float(observed_rate_hz, rate_hz*lower_limit, rate_hz*upper_limit);
79  rate_hz = filter_constant * rate_hz + (1-filter_constant) * observed_rate_hz;
80  count = 0;
81  start_us = now;
82  }
83  }

146 void AP_InertialSensor_Backend::_notify_new_gyro_raw_sample(uint8_t instance,
147  const Vector3f &gyro,
148  uint64_t sample_us)
149 {
150  if ((1U<<instance) & _imu.imu_kill_mask) {
151  return;
152  }
153  float dt;
154 
155  _update_sensor_rate(_imu._sample_gyro_count[instance], _imu._sample_gyro_start_us[instance],
156  _imu._gyro_raw_sample_rates[instance]);
157 
158  uint64_t last_sample_us = _imu._gyro_last_sample_us[instance];
159 
160  /*
161  we have two classes of sensors. FIFO based sensors produce data
162  at a very predictable overall rate, but the data comes in
163  bunches, so we use the provided sample rate for deltaT. Non-FIFO
164  sensors don't bunch up samples, but also tend to vary in actual
165  rate, so we use the provided sample_us to get the deltaT. The
166  difference between the two is whether sample_us is provided.
167  */
168  if (sample_us != 0 && _imu._gyro_last_sample_us[instance] != 0) {
169  dt = (sample_us - _imu._gyro_last_sample_us[instance]) * 1.0e-6f;
170  _imu._gyro_last_sample_us[instance] = sample_us;
171  } else {
172  // don't accept below 100Hz
173  if (_imu._gyro_raw_sample_rates[instance] < 100) {
174  return;
175  }
176 
177  dt = 1.0f / _imu._gyro_raw_sample_rates[instance];
178  _imu._gyro_last_sample_us[instance] = AP_HAL::micros64();
179  }
180 
181 #if AP_MODULE_SUPPORTED
182  // call gyro_sample hook if any
183  AP_Module::call_hook_gyro_sample(instance, dt, gyro);
184 #endif
185 
186  // push gyros if optical flow present
187  if (hal.opticalflow) {
188  hal.opticalflow->push_gyro(gyro.x, gyro.y, dt);
189  }
190 
191  // compute delta angle
192  Vector3f delta_angle = (gyro + _imu._last_raw_gyro[instance]) * 0.5f * dt;
193 
194  // compute coning correction
195  // see page 26 of:
196  // Tian et al (2010) Three-loop Integration of GPS and Strapdown INS with Coning and Sculling Compensation
197  // Available: http://www.sage.unsw.edu.au/snap/publications/tian_etal2010b.pdf
198  // see also examples/coning.py
199  Vector3f delta_coning = (_imu._delta_angle_acc[instance] +
200  _imu._last_delta_angle[instance] * (1.0f / 6.0f));
201  delta_coning = delta_coning % delta_angle;
202  delta_coning *= 0.5f;
203 
204  {
205  WITH_SEMAPHORE(_sem);
206  uint64_t now = AP_HAL::micros64();
207 
208  if (now - last_sample_us > 100000U) {
209  // zero accumulator if sensor was unhealthy for 0.1s
210  _imu._delta_angle_acc[instance].zero();
211  _imu._delta_angle_acc_dt[instance] = 0;
212  dt = 0;
213  delta_angle.zero();
214  }
215 
216  // integrate delta angle accumulator
217  // the angles and coning corrections are accumulated separately in the
218  // referenced paper, but in simulation little difference was found between
219  // integrating together and integrating separately (see examples/coning.py)
220  _imu._delta_angle_acc[instance] += delta_angle + delta_coning;
221  _imu._delta_angle_acc_dt[instance] += dt;
222 
223  // save previous delta angle for coning correction
224  _imu._last_delta_angle[instance] = delta_angle;
225  _imu._last_raw_gyro[instance] = gyro;
226 
227  // apply the low pass filter
228  _imu._gyro_filtered[instance] = _imu._gyro_filter[instance].apply(gyro);
229  // apply the notch filter
230  if (_gyro_notch_enabled()) {
231  _imu._gyro_filtered[instance] = _imu._gyro_notch_filter[instance].apply(_imu._gyro_filtered[instance]);
232  }
233  if (_imu._gyro_filtered[instance].is_nan() || _imu._gyro_filtered[instance].is_inf()) {
234  _imu._gyro_filter[instance].reset();
235  _imu._gyro_notch_filter[instance].reset();
236  }
237  _imu._new_gyro_data[instance] = true;
238  }
239 
240  if (!_imu.batchsampler.doing_post_filter_logging()) {
241  log_gyro_raw(instance, sample_us, gyro);
242  }
243  else {
244  log_gyro_raw(instance, sample_us, _imu._gyro_filtered[instance]);
245  }
