180 void AP_Notify::add_backend_helper(NotifyDevice *backend)
181 {
182  _devices[_num_devices] = backend;
183  _devices[_num_devices]->pNotify = this;
184  if(!_devices[_num_devices]->init()) {
185  delete _devices[_num_devices];
186  _devices[_num_devices] = nullptr;
187  } else {
188  _num_devices++;
189  }
190 }

194 // add notify backends to _devices array
195 void AP_Notify::add_backends(void)
196 {
197  if (_num_devices != 0) {
198  return;
199  }
200 
201  for (uint32_t i = 1; i < Notify_LED_MAX; i = i << 1) {
202  switch(_led_type & i) {
203  case Notify_LED_None:
204  break;
205  case Notify_LED_Board:
206  // select the most appropriate built in LED driver type
207 #if CONFIG_HAL_BOARD == HAL_BOARD_LINUX
208  #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_NAVIO2
209  ADD_BACKEND(new Led_Sysfs("rgb_led0", "rgb_led2", "rgb_led1"));
210  #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_EDGE
211  ADD_BACKEND(new RCOutputRGBLedInverted(12, 13, 14));
212  #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BH
213  ADD_BACKEND(new RCOutputRGBLed(HAL_RCOUT_RGBLED_RED, HAL_RCOUT_RGBLED_GREEN, HAL_RCOUT_RGBLED_BLUE));
214  #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_DISCO
215  ADD_BACKEND(new DiscoLED());
216  #endif
217 #endif // CONFIG_HAL_BOARD == HAL_BOARD_LINUX
218 
219 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V51 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V52 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRUBRAIN_V51 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRCORE_V10 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V54
220  ADD_BACKEND(new ExternalLED()); // despite the name this is a built in set of onboard LED's
221 #endif // CONFIG_HAL_BOARD_SUBTYPE == various CHIBIOS-VRBRAINs
222 
223 #if defined(HAL_HAVE_PIXRACER_LED)
224  ADD_BACKEND(new PixRacerLED());
225 #elif (defined(HAL_GPIO_A_LED_PIN) && defined(HAL_GPIO_B_LED_PIN) && defined(HAL_GPIO_C_LED_PIN))
226  #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V51 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V52 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRUBRAIN_V51 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRCORE_V10 || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_CHIBIOS_VRBRAIN_V54
227  ADD_BACKEND(new VRBoard_LED());
228  #else
229  ADD_BACKEND(new AP_BoardLED());
230  #endif
231 #elif (defined(HAL_GPIO_A_LED_PIN) && defined(HAL_GPIO_B_LED_PIN))
232  ADD_BACKEND(new AP_BoardLED2());
233 #endif
234  break;
235  case Notify_LED_ToshibaLED_I2C_Internal:
236  ADD_BACKEND(new ToshibaLED_I2C(TOSHIBA_LED_I2C_BUS_INTERNAL));
237  break;
238  case Notify_LED_ToshibaLED_I2C_External:
239  ADD_BACKEND(new ToshibaLED_I2C(TOSHIBA_LED_I2C_BUS_EXTERNAL));
240  break;
241 #if !HAL_MINIMIZE_FEATURES
242  case Notify_LED_NCP5623_I2C_External:
243  FOREACH_I2C_EXTERNAL(b) {
244  ADD_BACKEND(new NCP5623(b));
245  }
246  break;
247  case Notify_LED_NCP5623_I2C_Internal:
248  ADD_BACKEND(new NCP5623(TOSHIBA_LED_I2C_BUS_INTERNAL));
249  break;
250 #endif
251  case Notify_LED_PCA9685LED_I2C_External:
252  ADD_BACKEND(new PCA9685LED_I2C());
253  break;
254  case Notify_LED_OreoLED:
255 #if !HAL_MINIMIZE_FEATURES
256  if (_oreo_theme) {
257  ADD_BACKEND(new OreoLED_I2C(0, _oreo_theme));
258  }
259 #endif
260  break;
261  case Notify_LED_UAVCAN:
262 #if HAL_WITH_UAVCAN
263  ADD_BACKEND(new UAVCAN_RGB_LED(0));
264 #endif // HAL_WITH_UAVCAN
265  break;
266 
267  }
268  }
