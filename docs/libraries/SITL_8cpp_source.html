229 /*
230  convert a set of roll rates from earth frame to body frame
231  output values are in radians/second
232 */
233 void SITL::convert_body_frame(double rollDeg, double pitchDeg,
234  double rollRate, double pitchRate, double yawRate,
235  double *p, double *q, double *r)
236 {
237  double phi, theta, phiDot, thetaDot, psiDot;
238 
239  phi = ToRad(rollDeg);
240  theta = ToRad(pitchDeg);
241  phiDot = ToRad(rollRate);
242  thetaDot = ToRad(pitchRate);
243  psiDot = ToRad(yawRate);
244 
245  *p = phiDot - psiDot*sin(theta);
246  *q = cos(phi)*thetaDot + sin(phi)*psiDot*cos(theta);
247  *r = cos(phi)*psiDot*cos(theta) - sin(phi)*thetaDot;
248 }

251 /*
252  convert angular velocities from body frame to
253  earth frame.
254 
255  all inputs and outputs are in radians/s
256 */
257 Vector3f SITL::convert_earth_frame(const Matrix3f &dcm, const Vector3f &gyro)
258 {
259  float p = gyro.x;
260  float q = gyro.y;
261  float r = gyro.z;
262 
263  float phi, theta, psi;
264  dcm.to_euler(&phi, &theta, &psi);
265 
266  float phiDot = p + tanf(theta)*(q*sinf(phi) + r*cosf(phi));
267  float thetaDot = q*cosf(phi) - r*sinf(phi);
268  if (fabsf(cosf(theta)) < 1.0e-20f) {
269  theta += 1.0e-10f;
270  }
271  float psiDot = (q*sinf(phi) + r*cosf(phi))/cosf(theta);
272  return Vector3f(phiDot, thetaDot, psiDot);
273 }
