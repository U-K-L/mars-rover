77 /*
78  update and report, called from main loop
79  */
80 void AP_Button::update(void)
81 {
82  if (!enable) {
83  return;
84  }
85 
86  // call setup pins at update rate (5Hz) to allow for runtime parameter change of pins
87  setup_pins();
88 
89  if (!initialised) {
90  initialised = true;
91 
92  // get initial mask
93  last_mask = get_mask();
94 
95  // register 1kHz timer callback
96  hal.scheduler->register_timer_process(FUNCTOR_BIND_MEMBER(&AP_Button::timer_update, void));
97  }
98 
99  if (last_change_time_ms != 0 &&
100  (AP_HAL::millis() - last_report_ms) > AP_BUTTON_REPORT_PERIOD_MS &&
101  (AP_HAL::millis64() - last_change_time_ms) < report_send_time*1000ULL) {
102  // send a change report
103  last_report_ms = AP_HAL::millis();
104 
105  // send a report to GCS
106  send_report();
107  }
108 }

110 /*
111  get current mask
112  */
113 uint8_t AP_Button::get_mask(void)
114 {
115  uint8_t mask = 0;
116  for (uint8_t i=0; i<AP_BUTTON_NUM_PINS; i++) {
117  if (pin[i] == -1) {
118  continue;
119  }
120  mask |= hal.gpio->read(pin[i]) << i;
121  }
122  return mask;
123 }

125 /*
126  called at 1kHz to check for button state change
127  */
128 void AP_Button::timer_update(void)
129 {
130  if (!enable) {
131  return;
132  }
133  uint8_t mask = get_mask();
134  if (mask != last_mask) {
135  last_mask = mask;
136  last_change_time_ms = AP_HAL::millis64();
137  }

154 /*
155  setup the pins as input with pullup. We need pullup to give reliable
156  input with a pulldown button
157  */
158 void AP_Button::setup_pins(void)
159 {
160  for (uint8_t i=0; i<AP_BUTTON_NUM_PINS; i++) {
161  if (pin[i] == -1) {
162  continue;
163  }
164  hal.gpio->pinMode(pin[i], HAL_GPIO_INPUT);
165  // setup pullup
