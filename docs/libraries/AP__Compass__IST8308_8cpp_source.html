108 bool AP_Compass_IST8308::init()
109 {
110  uint8_t reset_count = 0;
111 
112  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
113  return false;
114  }
115 
116  // high retries for init
117  _dev->set_retries(10);
118 
119  uint8_t whoami;
120  if (!_dev->read_registers(WAI_REG, &whoami, 1) ||
121  whoami != DEVICE_ID) {
122  // not an IST8308
123  goto fail;
124  }
125 
126  for (; reset_count < 5; reset_count++) {
127  if (!_dev->write_register(CNTL3_REG, CNTL3_VAL_SRST)) {
128  hal.scheduler->delay(10);
129  continue;
130  }
131 
132  hal.scheduler->delay(20);
133 
134  uint8_t cntl3 = 0xFF;
135  if (_dev->read_registers(CNTL3_REG, &cntl3, 1) &&
136  (cntl3 & 0x01) == 0) {
137  break;
138  }
139  }
140 
141  if (reset_count == 5) {
142  printf("IST8308: failed to reset device\n");
143  goto fail;
144  }
145 
146  // DRDY enabled
147  // Dynamic Range=±500 uT, Sensitivity=6.6 LSB/uT
148  // OSR 16 (max 100Hz)
149  // Start continuous mode at 100Hz
150  if (!_dev->write_register(CNTL3_REG, CNTL3_VAL_DRDY_EN) ||
151  !_dev->write_register(CNTL4_REG, CNTL4_VAL_DYNAMIC_RANGE_500) ||
152  !_dev->write_register(OSRCNTL_REG, OSRCNTL_VAL_Y_16 | OSRCNTL_VAL_XZ_16) ||
153  !_dev->write_register(CNTL2_REG, CNTL2_VAL_CONT_ODR100_MODE)) {
154  printf("IST8308: found device but could not set it up\n");
155  goto fail;
156  }
157 
158  // lower retries for run
159  _dev->set_retries(3);
160 
161  _dev->get_semaphore()->give();
162 
163  _instance = register_compass();
164 
165  printf("%s found on bus %u id %u address 0x%02x\n", name,
166  _dev->bus_num(), _dev->get_bus_id(), _dev->get_bus_address());
167 
168  set_rotation(_instance, _rotation);
169 
170  _dev->set_device_type(DEVTYPE_IST8308);
171  set_dev_id(_instance, _dev->get_bus_id());
172 
173  if (_force_external) {
174  set_external(_instance, true);
175  }
176 
177  _dev->register_periodic_callback(SAMPLING_PERIOD_USEC,
178  FUNCTOR_BIND_MEMBER(&AP_Compass_IST8308::timer, void));
179 
180  _perf_xfer_err = hal.util->perf_alloc(AP_HAL::Util::PC_COUNT, "IST8308_xfer_err");
181 
182  return true;
183 
184 fail:
185  _dev->get_semaphore()->give();
186  return false;
187 }

189 void AP_Compass_IST8308::timer()
190 {
191  struct PACKED {
192  le16_t rx;
193  le16_t ry;
194  le16_t rz;
195  } buffer;
196  uint8_t stat;
197 
198  if (!_dev->read_registers(STAT1_REG, &stat, 1) ||
199  !(stat & STAT1_VAL_DRDY)) {
200  hal.util->perf_count(_perf_xfer_err);
201  return;
202  }
203 
204  if (stat & STAT1_VAL_DOR) {
205  printf("IST8308: data overrun\n");
206  }
207 
208  if (!_dev->read_registers(DATAX_L_REG, (uint8_t *) &buffer,
209  sizeof(buffer))) {
210  hal.util->perf_count(_perf_xfer_err);
211  return;
212  }
213 
214  auto x = static_cast<int16_t>(le16toh(buffer.rx));
215  auto y = static_cast<int16_t>(le16toh(buffer.ry));
216  auto z = static_cast<int16_t>(le16toh(buffer.rz));
217 
218  // flip Z to conform to right-hand rule convention
219  z = -z;
220 
221  /* Resolution: 0.1515 µT/LSB - already convert to milligauss */
222  Vector3f field = Vector3f{x * 1.515f, y * 1.515f, z * 1.515f};
223 
224  accumulate_sample(field, _instance);
225 }
