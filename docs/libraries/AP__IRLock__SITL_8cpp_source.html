34 // retrieve latest sensor data - returns true if new data is available
35 bool AP_IRLock_SITL::update()
36 {
37  // return immediately if not healthy
38  _flags.healthy = _sitl->precland_sim.healthy();
39  if (!_flags.healthy) {
40  return false;
41  }
42 
43  if (_sitl->precland_sim.last_update_ms() != _last_timestamp) {
44  const Vector3f position = _sitl->precland_sim.get_target_position();
45  const Matrix3f &body_to_ned = AP::ahrs().get_rotation_body_to_ned();
46  const Vector3f real_position = body_to_ned.mul_transpose(-position);
47  _last_timestamp = _sitl->precland_sim.last_update_ms();
48  _last_update_ms = _last_timestamp;
49  _target_info.timestamp = _last_timestamp;
50  _target_info.pos_x = real_position.y;
51  _target_info.pos_y = -real_position.x;
52  _target_info.pos_z = real_position.z;
53  return true;
54  }
55  return false;
56 }
