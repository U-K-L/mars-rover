54 bool BusEvent::wait(uavcan::MonotonicDuration duration)
55 {
56  // set maximum time to allow for 16 bit timers running at 1MHz
57  static const uavcan::int64_t MaxDelayUSec = 0x000FFFF;
58 
59  const uavcan::int64_t usec = duration.toUSec();
60  msg_t ret = msg_t();
61 
62  if (usec <= 0) {
63 # if (CH_KERNEL_MAJOR == 2)
64  ret = sem_.waitTimeout(TIME_IMMEDIATE);
65 # else // ChibiOS 3+
66  ret = sem_.wait(TIME_IMMEDIATE);
67 # endif
68  }
69  else {
70 # if (CH_KERNEL_MAJOR == 2)
71  ret = sem_.waitTimeout((usec > MaxDelayUSec) ? US2ST(MaxDelayUSec) : US2ST(usec));
72 # elif (CH_KERNEL_MAJOR >= 5)
73  ret = sem_.wait((usec > MaxDelayUSec) ? chTimeUS2I(MaxDelayUSec) : chTimeUS2I(usec));
74 # else // ChibiOS 3+
75  ret = sem_.wait((usec > MaxDelayUSec) ? US2ST(MaxDelayUSec) : US2ST(usec));
76 # endif
77  }
78 # if (CH_KERNEL_MAJOR == 2)
79  return ret == RDY_OK;
80 # else // ChibiOS 3+
81  return ret == MSG_OK;
82 # endif
83 }
