56 static float get_declination(float lat, float lon)
57 {
58  // Validate input values
59  lat = constrain_float(lat, -90, 90);
60  lon = constrain_float(lon, -180, 180);
61 
62  const int16_t latmin = floorf(lat / 5) * 5;
63  const int16_t lonmin = floorf(lon / 5) * 5;
64 
65  const uint8_t latmin_index = (90 + latmin) / 5;
66  const uint8_t lonmin_index = (180 + lonmin) / 5;
67 
68  const int16_t decSW = dec_tbl[latmin_index][lonmin_index];
69  const int16_t decSE = dec_tbl[latmin_index][lonmin_index + 1];
70  const int16_t decNE = dec_tbl[latmin_index+1][lonmin_index + 1];
71  const int16_t decNW = dec_tbl[latmin_index+1][lonmin_index];
72 
73  const float decmin = (lon - lonmin) / 5 * (decSE - decSW) + decSW;
74  const float decmax = (lon - lonmin) / 5 * (decNE - decNW) + decNW;
75  return (lat - latmin) / 5 * (decmax - decmin) + decmin;
76 }
