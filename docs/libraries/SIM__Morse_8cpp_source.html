84  Aircraft(home_str, frame_str)

90  char *args2 = strtok_r(nullptr, ":", &saveptr);
91  char *args3 = strtok_r(nullptr, ":", &saveptr);
92  /*
93  allow setting of IP, sensors port and control port
94  format morse:IPADDRESS:SENSORS_PORT:CONTROL_PORT
95  */
96  if (args1) {
97  morse_ip = args1;
98  }
99  if (args2) {
100  morse_sensors_port = atoi(args2);
101  morse_control_port = morse_sensors_port+1;
102  }
103  if (args3) {
104  morse_control_port = atoi(args3);
105  }
106 
107  if (strstr(frame_option, "-rover")) {
108  output_type = OUTPUT_ROVER;
109  } else if (strstr(frame_option, "-quad")) {
110  output_type = OUTPUT_QUAD;
111  } else if (strstr(frame_option, "-pwm")) {
112  output_type = OUTPUT_PWM;
113  } else {
114  // default to rover
115  output_type = OUTPUT_ROVER;
116  }
117 
118  for (uint8_t i=0; i<ARRAY_SIZE(sim_defaults); i++) {
119  AP_Param::set_default_by_name(sim_defaults[i].name, sim_defaults[i].value);
120  if (sim_defaults[i].save) {
121  enum ap_var_type ptype;
122  AP_Param *p = AP_Param::find(sim_defaults[i].name, &ptype);
123  if (!p->configured()) {

129  morse_ip, morse_sensors_port, morse_control_port,

131 }

133 /*
134  very simple JSON parser for sensor data
135  called with pointer to one row of sensor data, nul terminated
136 
137  This parser does not do any syntax checking, and is not at all
138  general purpose
139 */
140 bool Morse::parse_sensors(const char *json)
141 {
142  //printf("%s\n", json);
143  for (uint16_t i=0; i<ARRAY_SIZE(keytable); i++) {
144  struct keytable &key = keytable[i];
145 
146  /* look for section header */
147  const char *p = strstr(json, key.section);
148  if (!p) {
149  // we don't have this sensor
150  continue;
151  }
152  p += strlen(key.section)+1;
153 
154  // find key inside section
155  p = strstr(p, key.key);
156  if (!p) {
157  printf("Failed to find key %s/%s\n", key.section, key.key);
158  return false;
159  }
160 
161  p += strlen(key.key)+3;
162  switch (key.type) {
163  case DATA_FLOAT:
164  *((float *)key.ptr) = atof(p);
165  break;
166 
167  case DATA_DOUBLE:
168  *((double *)key.ptr) = atof(p);
169  break;
170 
171  case DATA_VECTOR3F: {
172  Vector3f *v = (Vector3f *)key.ptr;
173  if (sscanf(p, "[%f, %f, %f]", &v->x, &v->y, &v->z) != 3) {
174  printf("Failed to parse Vector3f for %s/%s\n", key.section, key.key);
175  return false;
176  }
177  break;
178  }
179 
180  case DATA_VECTOR3F_ARRAY: {
181  // example: [[0.0, 0.0, 0.0], [-8.97607135772705, -8.976069450378418, -8.642673492431641e-07]]
182  if (*p++ != '[') {
183  return false;
184  }
185  uint16_t n = 0;
186  struct vector3f_array *v = (struct vector3f_array *)key.ptr;
187  while (true) {
188  if (n >= v->length) {
189  Vector3f *d = (Vector3f *)realloc(v->data, sizeof(Vector3f)*(n+1));
190  if (d == nullptr) {
191  return false;
192  }
193  v->data = d;
194  v->length = n+1;
195  }
196  if (sscanf(p, "[%f, %f, %f]", &v->data[n].x, &v->data[n].y, &v->data[n].z) != 3) {
197  printf("Failed to parse Vector3f for %s/%s[%u]\n", key.section, key.key, n);
198  return false;
199  }
200  n++;
201  p = strchr(p,']');
202  if (!p) {
203  return false;
204  }
205  p++;
206  if (p[0] != ',') {
207  break;
208  }
209  if (p[1] != ' ') {
210  return false;
211  }
212  p += 2;
213  }
214  if (p[0] != ']') {
215  return false;
216  }
217  v->length = n;
218  break;
219  }
220 
221  case DATA_FLOAT_ARRAY: {
222  // example: [18.0, 12.694079399108887]
223  if (*p++ != '[') {
224  return false;
225  }
226  uint16_t n = 0;
227  struct float_array *v = (struct float_array *)key.ptr;
228  while (true) {
229  if (n >= v->length) {
230  float *d = (float *)realloc(v->data, sizeof(float)*(n+1));
231  if (d == nullptr) {
232  return false;
233  }
234  v->data = d;
235  v->length = n+1;
236  }
237  v->data[n] = atof(p);
238  n++;
239  p = strchr(p,',');
240  if (!p) {
241  break;
242  }
243  p++;
244  }
245  v->length = n;
246  break;
247  }
248  }
249  }
250  socket_frame_counter++;
251  return true;
252 }

257 bool Morse::connect_sockets(void)
258 {
259  if (!sensors_sock) {
260  sensors_sock = new SocketAPM(false);
261  if (!sensors_sock) {
262  AP_HAL::panic("Out of memory for sensors socket");
263  }
264  if (!sensors_sock->connect(morse_ip, morse_sensors_port)) {
265  usleep(100000);
266  if (connect_counter++ == 20) {
267  printf("Waiting to connect to sensors control on %s:%u\n",
268  morse_ip, morse_sensors_port);
269  connect_counter = 0;
270  }
271  delete sensors_sock;
272  sensors_sock = nullptr;
273  return false;
274  }
275  sensors_sock->reuseaddress();
276  printf("Sensors connected\n");
277  }
278  if (!control_sock) {
279  control_sock = new SocketAPM(false);
280  if (!control_sock) {
281  AP_HAL::panic("Out of memory for control socket");
282  }
283  if (!control_sock->connect(morse_ip, morse_control_port)) {
284  usleep(100000);
285  if (connect_counter++ == 20) {
286  printf("Waiting to connect to control control on %s:%u\n",
287  morse_ip, morse_control_port);
288  connect_counter = 0;
289  }
290  delete control_sock;
291  control_sock = nullptr;
292  return false;
293  }
294  control_sock->reuseaddress();
295  printf("Control connected\n");
296  }
297  return true;

300 /*
301  get any new data from the sensors socket
302 */
303 bool Morse::sensors_receive(void)
304 {
305  ssize_t ret = sensors_sock->recv(&sensor_buffer[sensor_buffer_len], sizeof(sensor_buffer)-sensor_buffer_len, 0);
306  if (ret <= 0) {
307  no_data_counter++;
308  if (no_data_counter == 1000) {
309  no_data_counter = 0;
310  delete sensors_sock;
311  delete control_sock;
312  sensors_sock = nullptr;
313  control_sock = nullptr;
314  }
315  return false;
316  }
317  no_data_counter = 0;
318 
319  // convert '\n' into nul
320  while (uint8_t *p = (uint8_t *)memchr(&sensor_buffer[sensor_buffer_len], '\n', ret)) {
321  *p = 0;
322  }
323  sensor_buffer_len += ret;
324 
325  const uint8_t *p2 = (const uint8_t *)memrchr(sensor_buffer, 0, sensor_buffer_len);
326  if (p2 == nullptr || p2 == sensor_buffer) {
327  return false;
328  }
329  const uint8_t *p1 = (const uint8_t *)memrchr(sensor_buffer, 0, p2 - sensor_buffer);
330  if (p1 == nullptr) {
331  return false;
332  }
333 
334  bool parse_ok = parse_sensors((const char *)(p1+1));
335 
336  memmove(sensor_buffer, p2, sensor_buffer_len - (p2 - sensor_buffer));
337  sensor_buffer_len = sensor_buffer_len - (p2 - sensor_buffer);
338 
339  return parse_ok;
340 }

342 /*
343  output control command assuming skid-steering rover
344  */
345 void Morse::output_rover(const struct sitl_input &input)
346 {
347  float motor1 = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
348  float motor2 = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
349  const float steer_rate_max_dps = 60;
350  const float max_speed = 2;
351 
352  const float steering_rps = (motor1 - motor2) * radians(steer_rate_max_dps);
353  const float speed_ms = 0.5*(motor1 + motor2) * max_speed;
354 
355  // construct a JSON packet for v and w
356  char buf[60];
357  snprintf(buf, sizeof(buf)-1, "{\"v\": %.3f, \"w\": %.2f}\n",
358  speed_ms, -steering_rps);
359  buf[sizeof(buf)-1] = 0;
360 
361  control_sock->send(buf, strlen(buf));
362 }

364 /*
365  output control command assuming a 4 channel quad
366  */
367 void Morse::output_quad(const struct sitl_input &input)
368 {
369  const float max_thrust = 1500;
370  float motors[4];
371  for (uint8_t i=0; i<4; i++) {
372  motors[i] = constrain_float(((input.servos[i]-1000)/1000.0f) * max_thrust, 0, max_thrust);
373  }
374  const float &m_right = motors[0];
375  const float &m_left = motors[1];
376  const float &m_front = motors[2];
377  const float &m_back = motors[3];
378 
379  // quad format in Morse is:
380  // m1: back
381  // m2: right
382  // m3: front
383  // m4: left
384 
385  // construct a JSON packet for motors
386  char buf[60];
387  snprintf(buf, sizeof(buf)-1, "{\"engines\": [%.3f, %.3f, %.3f, %.3f]}\n",
388  m_back, m_right, m_front, m_left);
389  buf[sizeof(buf)-1] = 0;
390 
391  control_sock->send(buf, strlen(buf));
392 }

411 /*
412  update the Morse simulation by one time step
413  */
414 void Morse::update(const struct sitl_input &input)
415 {
416  if (!connect_sockets()) {
417  return;
418  }
419 
420  last_state = state;
421 
422  if (sensors_receive()) {
423  // update average frame time used for extrapolation
424  double dt = constrain_float(state.timestamp - last_state.timestamp, 0.001, 1.0/50);
425  if (average_frame_time_s < 1.0e-6) {
426  average_frame_time_s = dt;
427  }
428  average_frame_time_s = average_frame_time_s * 0.98 + dt * 0.02;
429  }
430 
431  double dt_s = state.timestamp - last_state.timestamp;
432  if (dt_s < 0 || dt_s > 1) {
433  // cope with restarting while connected
434  initial_time_s = time_now_us * 1.0e-6f;
435  last_time_s = state.timestamp;
436  return;
437  }
438 
439  if (dt_s < 0.00001f) {
440  float delta_time = 0.001;
441  // don't go past the next expected frame
442  if (delta_time + extrapolated_s > average_frame_time_s) {
443  delta_time = average_frame_time_s - extrapolated_s;
444  }
445  if (delta_time <= 0) {
446  usleep(1000);
447  return;
448  }
449  time_now_us += delta_time * 1.0e6;
450  extrapolate_sensors(delta_time);
451  update_position();
452  update_mag_field_bf();
453  usleep(delta_time*1.0e6);
454  extrapolated_s += delta_time;
455  report_FPS();
456  return;
457  }
458 
459  extrapolated_s = 0;
460 
461  if (initial_time_s <= 0) {
462  dt_s = 0.001f;
463  initial_time_s = state.timestamp - dt_s;
464  }
465 
466  // convert from state variables to ardupilot conventions
467  dcm.from_euler(state.pose.roll, -state.pose.pitch, -state.pose.yaw);
468 
469  gyro = Vector3f(state.imu.angular_velocity[0],
470  -state.imu.angular_velocity[1],
471  -state.imu.angular_velocity[2]);
472 
473  velocity_ef = Vector3f(state.velocity.world_linear_velocity[0],
474  -state.velocity.world_linear_velocity[1],
475  -state.velocity.world_linear_velocity[2]);
476 
477  position = Vector3f(state.gps.x, -state.gps.y, -state.gps.z);
478 
479  // Morse IMU accel is NEU, convert to NED
480  accel_body = Vector3f(state.imu.linear_acceleration[0],
481  -state.imu.linear_acceleration[1],
482  -state.imu.linear_acceleration[2]);
483 
484  // limit to 16G to match pixhawk1
485  float a_limit = GRAVITY_MSS*16;
486  accel_body.x = constrain_float(accel_body.x, -a_limit, a_limit);
487  accel_body.y = constrain_float(accel_body.y, -a_limit, a_limit);
488  accel_body.z = constrain_float(accel_body.z, -a_limit, a_limit);
489 
490  // fill in laser scanner results, if available
491  scanner.points = state.scanner.points;
492  scanner.ranges = state.scanner.ranges;
493 
494  update_position();
495  time_advance();
496  uint64_t new_time_us = (state.timestamp - initial_time_s)*1.0e6;
497  if (new_time_us < time_now_us) {
498  uint64_t dt_us = time_now_us - new_time_us;
499  if (dt_us > 500000) {
500  // time going backwards
501  time_now_us = new_time_us;
502  }
503  } else {
504  time_now_us = new_time_us;
505  }
506 
507  last_time_s = state.timestamp;
508 
509  // update magnetic field
510  update_mag_field_bf();
511 
512  switch (output_type) {
513  case OUTPUT_ROVER:
514  output_rover(input);
515  break;
516  case OUTPUT_QUAD:
517  output_quad(input);
518  break;
519  case OUTPUT_PWM:
520  output_pwm(input);
521  break;
522  }
523 
524  report_FPS();
525 }
