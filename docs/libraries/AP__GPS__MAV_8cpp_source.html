40 // handles an incoming mavlink message (HIL_GPS) and sets
41 // corresponding gps data appropriately;
42 void AP_GPS_MAV::handle_msg(const mavlink_message_t *msg)
43 {
44  switch (msg->msgid) {
45 
46  case MAVLINK_MSG_ID_GPS_INPUT: {
47  mavlink_gps_input_t packet;
48  mavlink_msg_gps_input_decode(msg, &packet);
49 
50  bool have_alt = ((packet.ignore_flags & GPS_INPUT_IGNORE_FLAG_ALT) == 0);
51  bool have_hdop = ((packet.ignore_flags & GPS_INPUT_IGNORE_FLAG_HDOP) == 0);
52  bool have_vdop = ((packet.ignore_flags & GPS_INPUT_IGNORE_FLAG_VDOP) == 0);
53  bool have_vel_h = ((packet.ignore_flags & GPS_INPUT_IGNORE_FLAG_VEL_HORIZ) == 0);
54  bool have_vel_v = ((packet.ignore_flags & GPS_INPUT_IGNORE_FLAG_VEL_VERT) == 0);
55  bool have_sa = ((packet.ignore_flags & GPS_INPUT_IGNORE_FLAG_SPEED_ACCURACY) == 0);
56  bool have_ha = ((packet.ignore_flags & GPS_INPUT_IGNORE_FLAG_HORIZONTAL_ACCURACY) == 0);
57  bool have_va = ((packet.ignore_flags & GPS_INPUT_IGNORE_FLAG_VERTICAL_ACCURACY) == 0);
58 
59  state.time_week = packet.time_week;
60  state.time_week_ms = packet.time_week_ms;
61  state.status = (AP_GPS::GPS_Status)packet.fix_type;
62 
63  Location loc = {};
64  loc.lat = packet.lat;
65  loc.lng = packet.lon;
66  if (have_alt) {
67  loc.alt = packet.alt * 100; // convert to centimeters
68  }
69  state.location = loc;
70 
71  if (have_hdop) {
72  state.hdop = packet.hdop * 100; // convert to centimeters
73  }
74 
75  if (have_vdop) {
76  state.vdop = packet.vdop * 100; // convert to centimeters
77  }
78 
79  if (have_vel_h) {
80  Vector3f vel(packet.vn, packet.ve, 0);
81  if (have_vel_v) {
82  vel.z = packet.vd;
83  state.have_vertical_velocity = true;
84  }
85 
86  state.velocity = vel;
87  state.ground_course = wrap_360(degrees(atan2f(vel.y, vel.x)));
88  state.ground_speed = norm(vel.x, vel.y);
89  }
90 
91  if (have_sa) {
92  state.speed_accuracy = packet.speed_accuracy;
93  state.have_speed_accuracy = true;
94  }
95 
96  if (have_ha) {
97  state.horizontal_accuracy = packet.horiz_accuracy;
98  state.have_horizontal_accuracy = true;
99  }
100 
101  if (have_va) {
102  state.vertical_accuracy = packet.vert_accuracy;
103  state.have_vertical_accuracy = true;
104  }
105 
106  state.num_sats = packet.satellites_visible;
107  state.last_gps_time_ms = AP_HAL::millis();
108  _new_data = true;
109  break;
110  }
111 
112  case MAVLINK_MSG_ID_HIL_GPS: {
113  mavlink_hil_gps_t packet;
114  mavlink_msg_hil_gps_decode(msg, &packet);
115 
116  state.time_week = 0;
117  state.time_week_ms = packet.time_usec/1000;
118  state.status = (AP_GPS::GPS_Status)packet.fix_type;
119 
120  Location loc = {};
121  loc.lat = packet.lat;
122  loc.lng = packet.lon;
123  loc.alt = packet.alt * 0.1f;
124  state.location = loc;
125  state.hdop = MIN(packet.eph, GPS_UNKNOWN_DOP);
126  state.vdop = MIN(packet.epv, GPS_UNKNOWN_DOP);
127  if (packet.vel < 65535) {
128  state.ground_speed = packet.vel / 100.0f;
129  }
130  Vector3f vel(packet.vn/100.0f, packet.ve/100.0f, packet.vd/100.0f);
131  state.velocity = vel;
132  if (packet.vd != 0) {
133  state.have_vertical_velocity = true;
134  }
135  if (packet.cog < 36000) {
136  state.ground_course = packet.cog / 100.0f;
137  }
138  state.have_speed_accuracy = false;
139  state.have_horizontal_accuracy = false;
140  state.have_vertical_accuracy = false;
141  if (packet.satellites_visible < 255) {
142  state.num_sats = packet.satellites_visible;
143  }
144  state.last_gps_time_ms = AP_HAL::millis();
145  _new_data = true;
146  break;
147  }
148  default:
149  // ignore all other messages
150  break;
151  }
152 }
