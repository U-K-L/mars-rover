45 // Process bytes available from the stream
46 //
47 // The stream is assumed to contain only our custom message. If it
48 // contains other messages, and those messages contain the preamble bytes,
49 // it is possible for this code to become de-synchronised. Without
50 // buffering the entire message and re-processing it from the top,
51 // this is unavoidable.
52 //
53 // The lack of a standard header length field makes it impossible to skip
54 // unrecognised messages.
55 //
56 bool
57 AP_GPS_MTK::read(void)
58 {
59  uint8_t data;
60  int16_t numc;
61  bool parsed = false;
62 
63  numc = port->available();
64  for (int16_t i = 0; i < numc; i++) { // Process bytes received
65 
66  // read the next byte
67  data = port->read();
68 
69 restart:
70  switch(_step) {
71 
72  // Message preamble, class, ID detection
73  //
74  // If we fail to match any of the expected bytes, we
75  // reset the state machine and re-consider the failed
76  // byte as the first byte of the preamble. This
77  // improves our chances of recovering from a mismatch
78  // and makes it less likely that we will be fooled by
79  // the preamble appearing as data in some other message.
80  //
81  case 0:
82  if(PREAMBLE1 == data)
83  _step++;
84  break;
85  case 1:
86  if (PREAMBLE2 == data) {
87  _step++;
88  break;
89  }
90  _step = 0;
91  goto restart;
92  case 2:
93  if (MESSAGE_CLASS == data) {
94  _step++;
95  _ck_b = _ck_a = data; // reset the checksum accumulators
96  } else {
97  _step = 0; // reset and wait for a message of the right class
98  goto restart;
99  }
100  break;
101  case 3:
102  if (MESSAGE_ID == data) {
103  _step++;
104  _ck_b += (_ck_a += data);
105  _payload_counter = 0;
106  } else {
107  _step = 0;
108  goto restart;
109  }
110  break;
111 
112  // Receive message data
113  //
114  case 4:
115  _buffer[_payload_counter++] = data;
116  _ck_b += (_ck_a += data);
117  if (_payload_counter == sizeof(_buffer))
118  _step++;
119  break;
120 
121  // Checksum and message processing
122  //
123  case 5:
124  _step++;
125  if (_ck_a != data) {
126  _step = 0;
127  }
128  break;
129  case 6:
130  _step = 0;
131  if (_ck_b != data) {
132  break;
133  }
134 
135  // set fix type
136  if (_buffer.msg.fix_type == FIX_3D) {
137  state.status = AP_GPS::GPS_OK_FIX_3D;
138  }else if (_buffer.msg.fix_type == FIX_2D) {
139  state.status = AP_GPS::GPS_OK_FIX_2D;
140  }else{
141  state.status = AP_GPS::NO_FIX;
142  }
143  state.location.lat = swap_int32(_buffer.msg.latitude) * 10;
144  state.location.lng = swap_int32(_buffer.msg.longitude) * 10;
145  state.location.alt = swap_int32(_buffer.msg.altitude);
146  state.ground_speed = swap_int32(_buffer.msg.ground_speed) * 0.01f;
147  state.ground_course = wrap_360(swap_int32(_buffer.msg.ground_course) * 1.0e-6f);
148  state.num_sats = _buffer.msg.satellites;
149 
150  if (state.status >= AP_GPS::GPS_OK_FIX_2D) {
151  make_gps_time(0, swap_int32(_buffer.msg.utc_time)*10);
152  }
153  // we don't change _last_gps_time as we don't know the
154  // full date
155 
156  fill_3d_velocity();
157 
158  parsed = true;
159  }
160  }
161  return parsed;
162 }

164 /*
165  detect a MTK GPS
166  */
167 bool
168 AP_GPS_MTK::_detect(struct MTK_detect_state &state, uint8_t data)
169 {
170  switch (state.step) {
171  case 1:
172  if (PREAMBLE2 == data) {
173  state.step++;
174  break;
175  }
176  state.step = 0;
177  FALLTHROUGH;
178  case 0:
179  state.ck_b = state.ck_a = state.payload_counter = 0;
180  if(PREAMBLE1 == data)
181  state.step++;
182  break;
183  case 2:
184  if (MESSAGE_CLASS == data) {
185  state.step++;
186  state.ck_b = state.ck_a = data;
187  } else {
188  state.step = 0;
189  }
190  break;
191  case 3:
192  if (MESSAGE_ID == data) {
193  state.step++;
194  state.ck_b += (state.ck_a += data);
195  state.payload_counter = 0;
196  } else {
197  state.step = 0;
198  }
199  break;
200  case 4:
201  state.ck_b += (state.ck_a += data);
202  if (++state.payload_counter == sizeof(struct diyd_mtk_msg))
203  state.step++;
204  break;
205  case 5:
206  state.step++;
207  if (state.ck_a != data) {
208  state.step = 0;
209  }
210  break;
211  case 6:
212  state.step = 0;
213  if (state.ck_b == data) {
214  return true;
215  }
216  }
217  return false;
218 }
