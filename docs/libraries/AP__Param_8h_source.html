467 /* This header is prepended to a variable stored in EEPROM.
468  * The meaning is as follows:
469  *
470  * - key: the k_param enum value from Parameter.h in the sketch
471  *
472  * - group_element: This is zero for top level parameters. For
473  * parameters stored within an object this is divided
474  * into 3 lots of 6 bits, allowing for three levels
475  * of object to be stored in the eeprom
476  *
477  * - type: the ap_var_type value for the variable
478  */
479  struct Param_header {
480  // to get 9 bits for key we needed to split it into two parts to keep binary compatibility
481  uint32_t key_low : 8;
482  uint32_t type : 5;
483  uint32_t key_high : 1;
484  uint32_t group_element : 18;
485  };

679  void set_and_notify(const T &v) {
680 // We do want to compare each value, even floats, since it being the same here
681 // is the result of previously setting it.
682 #pragma GCC diagnostic push
683 #pragma GCC diagnostic ignored "-Wfloat-equal"
684  if (v != _value) {
685 #pragma GCC diagnostic pop
686  set(v);
687  notify();
688  }
689  }

694  bool force = fabsf((float)(_value - v)) < FLT_EPSILON;
695  set(v);
696  save(force);
697  }
698 
704  void set_and_save_ifchanged(const T &v) {
705  if (v == _value) {
706  return;
707  }
708  set(v);
709  save(true);
710  }

717  return _value;
718  }
