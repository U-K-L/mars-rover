182 void AP_LandingGear::update(float height_above_ground_m)
183 {
184  if (_pin_weight_on_wheels == -1) {
185  last_wow_event_ms = 0;
186  wow_state_current = LG_WOW_UNKNOWN;
187  } else {
188  LG_WOW_State wow_state_new = hal.gpio->read(_pin_weight_on_wheels) == _pin_weight_on_wheels_polarity ? LG_WOW : LG_NO_WOW;
189 
190  if (wow_state_new != wow_state_current) {
191  // we changed states, lets note the time.
192  last_wow_event_ms = AP_HAL::millis();
193  log_wow_state(wow_state_new);
194  }
195 
196  wow_state_current = wow_state_new;
197  }
198 
199  if (_pin_deployed == -1) {
200  last_gear_event_ms = 0;
201 
202  // If there was no pilot input and state is still unknown - leave it as it is
203  if (gear_state_current != LG_UNKNOWN) {
204  gear_state_current = (_deployed == true ? LG_DEPLOYED : LG_RETRACTED);
205  }
206  } else {
207  LG_LandingGear_State gear_state_new;
208 
209  if (_deployed) {
210  gear_state_new = (deployed() == true ? LG_DEPLOYED : LG_DEPLOYING);
211  } else {
212  gear_state_new = (deployed() == false ? LG_RETRACTED : LG_RETRACTING);
213  }
214 
215  if (gear_state_new != gear_state_current) {
216  // we changed states, lets note the time.
217  last_gear_event_ms = AP_HAL::millis();
218 
219  log_wow_state(wow_state_current);
220  }
221 
222  gear_state_current = gear_state_new;
223  }
224 
225  /*
226  check for height based triggering
227  */
228  int16_t alt_m = constrain_int16(height_above_ground_m, 0, INT16_MAX);
229 
230  if (hal.util->get_soft_armed()) {
231  // only do height based triggering when armed
232  if ((!_deployed || !_have_changed) &&
233  _deploy_alt > 0 &&
234  alt_m <= _deploy_alt &&
235  _last_height_above_ground > _deploy_alt) {
236  deploy();
237  }
238  if ((_deployed || !_have_changed) &&
239  _retract_alt > 0 &&
240  _retract_alt >= _deploy_alt &&
241  alt_m >= _retract_alt &&
242  _last_height_above_ground < _retract_alt) {
243  retract();
244  }
245  }
246 
247  _last_height_above_ground = alt_m;
248 }
