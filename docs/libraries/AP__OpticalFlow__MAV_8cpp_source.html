30 // read latest values from sensor and fill in x,y and totals.
31 void AP_OpticalFlow_MAV::update(void)
32 {
33  // record gyro values as long as they are being used
34  // the sanity check of dt below ensures old gyro values are not used
35  if (gyro_sum_count < 1000) {
36  const Vector3f& gyro = AP::ahrs().get_gyro();
37  gyro_sum.x += gyro.x;
38  gyro_sum.y += gyro.y;
39  gyro_sum_count++;
40  }
41 
42  // return without updating state if no readings
43  if (count == 0) {
44  return;
45  }
46 
47  struct OpticalFlow::OpticalFlow_state state {};
48 
49  state.surface_quality = quality_sum / count;
50 
51  // calculate dt
52  const float dt = (latest_frame_us - prev_frame_us) * 1.0e-6;
53  prev_frame_us = latest_frame_us;
54 
55  // sanity check dt
56  if (is_positive(dt) && (dt < OPTFLOW_MAV_TIMEOUT_SEC)) {
57  // calculate flow values
58  const float flow_scale_factor_x = 1.0f + 0.001f * _flowScaler().x;
59  const float flow_scale_factor_y = 1.0f + 0.001f * _flowScaler().y;
60 
61  // copy flow rates to state structure
62  state.flowRate = { ((float)flow_sum.x / count) * flow_scale_factor_x * dt,
63  ((float)flow_sum.y / count) * flow_scale_factor_y * dt };
64 
65  // copy average body rate to state structure
66  state.bodyRate = { gyro_sum.x / gyro_sum_count, gyro_sum.y / gyro_sum_count };
67 
68  _applyYaw(state.flowRate);
69  _applyYaw(state.bodyRate);
70  } else {
71  // first frame received in some time so cannot calculate flow values
72  state.flowRate.zero();
73  state.bodyRate.zero();
74  }
75 
76  _update_frontend(state);
77 
78  // reset local buffers
79  flow_sum.zero();
80  quality_sum = 0;
81  count = 0;
82 
83  // reset gyro sum
84  gyro_sum.zero();
85  gyro_sum_count = 0;
86 }
