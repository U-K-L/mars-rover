97 /*
98  erase all storage
99  */
100 void StorageManager::erase(void)
101 {
102  uint8_t blk[16];
103  memset(blk, 0, sizeof(blk));
104  for (uint8_t i=0; i<STORAGE_NUM_AREAS; i++) {
105  const StorageManager::StorageArea &area = StorageManager::layout[i];
106  uint16_t length = area.length;
107  uint16_t offset = area.offset;
108  for (uint16_t ofs=0; ofs<length; ofs += sizeof(blk)) {
109  uint8_t n = 16;
110  if (ofs + n > length) {
111  n = length - ofs;
112  }
113  hal.storage->write_block(offset + ofs, blk, n);
114  }
115  }
116 
117 }

135 /*
136  base read function. The src offset is within the bytes allocated
137  for the storage type of this StorageAccess object
138 */
139 bool StorageAccess::read_block(void *data, uint16_t addr, size_t n) const
140 {
141  uint8_t *b = (uint8_t *)data;
142  for (uint8_t i=0; i<STORAGE_NUM_AREAS; i++) {
143  const StorageManager::StorageArea &area = StorageManager::layout[i];
144  uint16_t length = area.length;
145  uint16_t offset = area.offset;
146  if (area.type != type) {
147  continue;
148  }
149  if (addr >= length) {
150  // the data isn't in this area
151  addr -= length;
152  continue;
153  }
154  uint8_t count = n;
155  if (count+addr > length) {
156  // the data crosses a boundary between two areas
157  count = length - addr;
158  }
159  hal.storage->read_block(b, addr+offset, count);
160  n -= count;
161 
162  if (n == 0) {
163  break;
164  }
165 
166  // move pointer after written bytes
167  b += count;
168  // continue writing at the beginning of next valid area
169  addr = 0;
170  }
171  return (n == 0);
172 }
