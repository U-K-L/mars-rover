310 /* update; triggers by distance moved
311 */
312 void AP_Camera::update()
313 {
314  if (AP::gps().status() < AP_GPS::GPS_OK_FIX_3D) {
315  return;
316  }
317 
318  if (is_zero(_trigg_dist)) {
319  return;
320  }
321  if (_last_location.lat == 0 && _last_location.lng == 0) {
322  _last_location = current_loc;
323  return;
324  }
325  if (_last_location.lat == current_loc.lat && _last_location.lng == current_loc.lng) {
326  // we haven't moved - this can happen as update() may
327  // be called without a new GPS fix
328  return;
329  }
330 
331  if (current_loc.get_distance(_last_location) < _trigg_dist) {
332  return;
333  }
334 
335  if (_max_roll > 0 && fabsf(AP::ahrs().roll_sensor*1e-2f) > _max_roll) {
336  return;
337  }
338 
339  if (_is_in_auto_mode != true && _auto_mode_only != 0) {
340  return;
341  }
342 
343  uint32_t tnow = AP_HAL::millis();
344  if (tnow - _last_photo_time < (unsigned) _min_interval) {
345  return;
346  }
347 
348  take_picture();
349 
350  _last_location = current_loc;
351  _last_photo_time = tnow;
352 }

379 /*
380  setup a callback for a feedback pin. When on PX4 with the right FMU
381  mode we can use the microsecond timer.
382  */
383 void AP_Camera::setup_feedback_callback(void)
384 {
385  if (_feedback_pin <= 0 || _timer_installed || _isr_installed) {
386  // invalid or already installed
387  return;
388  }
389 
390  // ensure we are in input mode
391  hal.gpio->pinMode(_feedback_pin, HAL_GPIO_INPUT);
392 
393  // enable pullup/pulldown
394  uint8_t trigger_polarity = _feedback_polarity==0?0:1;
395  hal.gpio->write(_feedback_pin, !trigger_polarity);
396 
397  if (hal.gpio->attach_interrupt(_feedback_pin, FUNCTOR_BIND_MEMBER(&AP_Camera::feedback_pin_isr, void, uint8_t, bool, uint32_t),
398  trigger_polarity?AP_HAL::GPIO::INTERRUPT_RISING:AP_HAL::GPIO::INTERRUPT_FALLING)) {
399  _isr_installed = true;
400  } else {
401  // install a 1kHz timer to check feedback pin
402  hal.scheduler->register_timer_process(FUNCTOR_BIND_MEMBER(&AP_Camera::feedback_pin_timer, void));
403 
404  _timer_installed = true;
