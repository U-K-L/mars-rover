286 // return current_limit as a number from 0 ~ 1 in the range throttle_min to throttle_max
287 float AP_MotorsMulticopter::get_current_limit_max_throttle()
288 {
289  AP_BattMonitor &battery = AP::battery();
290 
291  if (_batt_current_max <= 0 || // return maximum if current limiting is disabled
292  !_flags.armed || // remove throttle limit if disarmed
293  !battery.has_current(_batt_idx)) { // no current monitoring is available
294  _throttle_limit = 1.0f;
295  return 1.0f;
296  }
297 
298  float _batt_resistance = battery.get_resistance(_batt_idx);
299 
300  if (is_zero(_batt_resistance)) {
301  _throttle_limit = 1.0f;
302  return 1.0f;
303  }
304 
305  float _batt_current = battery.current_amps(_batt_idx);
306 
307  // calculate the maximum current to prevent voltage sag below _batt_voltage_min
308  float batt_current_max = MIN(_batt_current_max, _batt_current + (battery.voltage(_batt_idx)-_batt_voltage_min)/_batt_resistance);
309 
310  float batt_current_ratio = _batt_current/batt_current_max;
311 
312  float loop_interval = 1.0f/_loop_rate;
313  _throttle_limit += (loop_interval/(loop_interval+_batt_current_time_constant))*(1.0f - batt_current_ratio);
314 
315  // throttle limit drops to 20% between hover and full throttle
316  _throttle_limit = constrain_float(_throttle_limit, 0.2f, 1.0f);
317 
318  // limit max throttle
319  return get_throttle_hover() + ((1.0-get_throttle_hover())*_throttle_limit);
320 }

322 // apply_thrust_curve_and_volt_scaling - returns throttle in the range 0 ~ 1
323 float AP_MotorsMulticopter::apply_thrust_curve_and_volt_scaling(float thrust) const
324 {
325  float throttle_ratio = thrust;
326  // apply thrust curve - domain 0.0 to 1.0, range 0.0 to 1.0
327  float thrust_curve_expo = constrain_float(_thrust_curve_expo, -1.0f, 1.0f);
328  if (fabsf(thrust_curve_expo) < 0.001) {
329  // zero expo means linear, avoid floating point exception for small values
330  return thrust;
331  }
332  if(!is_zero(_batt_voltage_filt.get())) {
333  throttle_ratio = ((thrust_curve_expo-1.0f) + safe_sqrt((1.0f-thrust_curve_expo)*(1.0f-thrust_curve_expo) + 4.0f*thrust_curve_expo*_lift_max*thrust))/(2.0f*thrust_curve_expo*_batt_voltage_filt.get());
334  } else {
335  throttle_ratio = ((thrust_curve_expo-1.0f) + safe_sqrt((1.0f-thrust_curve_expo)*(1.0f-thrust_curve_expo) + 4.0f*thrust_curve_expo*_lift_max*thrust))/(2.0f*thrust_curve_expo);
336  }
337 
338  return constrain_float(throttle_ratio, 0.0f, 1.0f);
339 }

341 // update_lift_max from battery voltage - used for voltage compensation
342 void AP_MotorsMulticopter::update_lift_max_from_batt_voltage()
343 {
344  // sanity check battery_voltage_min is not too small
345  // if disabled or misconfigured exit immediately
346  float _batt_voltage_resting_estimate = AP::battery().voltage_resting_estimate(_batt_idx);
347  if((_batt_voltage_max <= 0) || (_batt_voltage_min >= _batt_voltage_max) || (_batt_voltage_resting_estimate < 0.25f*_batt_voltage_min)) {
348  _batt_voltage_filt.reset(1.0f);
349  _lift_max = 1.0f;
350  return;
351  }
352 
353  _batt_voltage_min = MAX(_batt_voltage_min, _batt_voltage_max * 0.6f);
354 
355  // contrain resting voltage estimate (resting voltage is actual voltage with sag removed based on current draw and resistance)
356  _batt_voltage_resting_estimate = constrain_float(_batt_voltage_resting_estimate, _batt_voltage_min, _batt_voltage_max);
357 
358  // filter at 0.5 Hz
359  float batt_voltage_filt = _batt_voltage_filt.apply(_batt_voltage_resting_estimate/_batt_voltage_max, 1.0f/_loop_rate);
360 
361  // calculate lift max
362  float thrust_curve_expo = constrain_float(_thrust_curve_expo, -1.0f, 1.0f);
363  _lift_max = batt_voltage_filt*(1-thrust_curve_expo) + thrust_curve_expo*batt_voltage_filt*batt_voltage_filt;
364 }

366 float AP_MotorsMulticopter::get_compensation_gain() const
367 {
368  // avoid divide by zero
369  if (_lift_max <= 0.0f) {
370  return 1.0f;
371  }
372 
373  float ret = 1.0f / _lift_max;
374 
375 #if AP_MOTORS_DENSITY_COMP == 1
376  // air density ratio is increasing in density / decreasing in altitude
377  if (_air_density_ratio > 0.3f && _air_density_ratio < 1.5f) {
378  ret *= 1.0f / constrain_float(_air_density_ratio,0.5f,1.25f);
379  }
380 #endif
381  return ret;
382 }

384 // convert actuator output (0~1) range to pwm range
385 int16_t AP_MotorsMulticopter::output_to_pwm(float actuator)
386 {
387  float pwm_output;
388  if (_spool_state == SpoolState::SHUT_DOWN) {
389  // in shutdown mode, use PWM 0 or minimum PWM
390  if (_disarm_disable_pwm && _disarm_safety_timer == 0 && !armed()) {
391  pwm_output = 0;
392  } else {
393  pwm_output = get_pwm_output_min();
394  }
395  } else {
396  // in all other spool modes, covert to desired PWM
397  pwm_output = get_pwm_output_min() + (get_pwm_output_max()-get_pwm_output_min()) * actuator;
398  }
399 
400  return pwm_output;
401 }

410 // adds slew rate limiting to actuator output
411 void AP_MotorsMulticopter::set_actuator_with_slew(float& actuator_output, float input)
412 {
413  /*
414  If MOT_SLEW_UP_TIME is 0 (default), no slew limit is applied to increasing output.
415  If MOT_SLEW_DN_TIME is 0 (default), no slew limit is applied to decreasing output.
416  MOT_SLEW_UP_TIME and MOT_SLEW_DN_TIME are constrained to 0.0~0.5 for sanity.
417  If spool mode is shutdown, no slew limit is applied to allow immediate disarming of motors.
418  */
419 
420  // Output limits with no slew time applied
421  float output_slew_limit_up = 1.0f;
422  float output_slew_limit_dn = 0.0f;
423 
424  // If MOT_SLEW_UP_TIME is set, calculate the highest allowed new output value, constrained 0.0~1.0
425  if (is_positive(_slew_up_time)) {
426  float output_delta_up_max = 1.0f/(constrain_float(_slew_up_time,0.0f,0.5f) * _loop_rate);
427  output_slew_limit_up = constrain_float(actuator_output + output_delta_up_max, 0.0f, 1.0f);
428  }
429 
430  // If MOT_SLEW_DN_TIME is set, calculate the lowest allowed new output value, constrained 0.0~1.0
431  if (is_positive(_slew_dn_time)) {
432  float output_delta_dn_max = 1.0f/(constrain_float(_slew_dn_time,0.0f,0.5f) * _loop_rate);
433  output_slew_limit_dn = constrain_float(actuator_output - output_delta_dn_max, 0.0f, 1.0f);
434  }
435 
436  // Constrain change in output to within the above limits
437  actuator_output = constrain_float(input, output_slew_limit_dn, output_slew_limit_up);
438 }

496 // run spool logic
497 void AP_MotorsMulticopter::output_logic()
498 {
499  if (_flags.armed) {
500  _disarm_safety_timer = 100;
501  } else if (_disarm_safety_timer != 0) {
502  _disarm_safety_timer--;
503  }
504 
505  // force desired and current spool mode if disarmed or not interlocked
506  if (!_flags.armed || !_flags.interlock) {
507  _spool_desired = DesiredSpoolState::SHUT_DOWN;
508  _spool_state = SpoolState::SHUT_DOWN;
509  }
510 
511  if (_spool_up_time < 0.05) {
512  // prevent float exception
513  _spool_up_time.set(0.05);
514  }
515 
516  switch (_spool_state) {
517  case SpoolState::SHUT_DOWN:
518  // Motors should be stationary.
519  // Servos set to their trim values or in a test condition.
520 
521  // set limits flags
522  limit.roll_pitch = true;
523  limit.yaw = true;
524  limit.throttle_lower = true;
525  limit.throttle_upper = true;
526 
527  // make sure the motors are spooling in the correct direction
528  if (_spool_desired != DesiredSpoolState::SHUT_DOWN) {
529  _spool_state = SpoolState::GROUND_IDLE;
530  break;
531  }
532 
533  // set and increment ramp variables
534  _spin_up_ratio = 0.0f;
535  _throttle_thrust_max = 0.0f;
536 
537  // initialise motor failure variables
538  _thrust_boost = false;
539  _thrust_boost_ratio = 0.0f;
540  break;
541 
542  case SpoolState::GROUND_IDLE: {
543  // Motors should be stationary or at ground idle.
544  // Servos should be moving to correct the current attitude.
545 
546  // set limits flags
547  limit.roll_pitch = true;
548  limit.yaw = true;
549  limit.throttle_lower = true;
550  limit.throttle_upper = true;
551 
552  // set and increment ramp variables
553  float spool_step = 1.0f/(_spool_up_time*_loop_rate);
554  switch (_spool_desired) {
555  case DesiredSpoolState::SHUT_DOWN:
556  _spin_up_ratio -= spool_step;
557  // constrain ramp value and update mode
558  if (_spin_up_ratio <= 0.0f) {
559  _spin_up_ratio = 0.0f;
560  _spool_state = SpoolState::SHUT_DOWN;
561  }
562  break;
563  case DesiredSpoolState::THROTTLE_UNLIMITED:
564  _spin_up_ratio += spool_step;
565  // constrain ramp value and update mode
566  if (_spin_up_ratio >= 1.0f) {
567  _spin_up_ratio = 1.0f;
568  _spool_state = SpoolState::SPOOLING_UP;
569  }
570  break;
571  case DesiredSpoolState::GROUND_IDLE:
572  float spin_up_armed_ratio = 0.0f;
573  if (_spin_min > 0.0f) {
574  spin_up_armed_ratio = _spin_arm / _spin_min;
575  }
576  _spin_up_ratio += constrain_float(spin_up_armed_ratio-_spin_up_ratio, -spool_step, spool_step);
577  break;
578  }
579  _throttle_thrust_max = 0.0f;
580 
581  // initialise motor failure variables
582  _thrust_boost = false;
583  _thrust_boost_ratio = 0.0f;
584  break;
585  }
586  case SpoolState::SPOOLING_UP:
587  // Maximum throttle should move from minimum to maximum.
588  // Servos should exhibit normal flight behavior.
589 
590  // initialize limits flags
591  limit.roll_pitch = false;
592  limit.yaw = false;
593  limit.throttle_lower = false;
594  limit.throttle_upper = false;
595 
596  // make sure the motors are spooling in the correct direction
597  if (_spool_desired != DesiredSpoolState::THROTTLE_UNLIMITED ){
598  _spool_state = SpoolState::SPOOLING_DOWN;
599  break;
600  }
601 
602  // set and increment ramp variables
603  _spin_up_ratio = 1.0f;
604  _throttle_thrust_max += 1.0f/(_spool_up_time*_loop_rate);
605 
606  // constrain ramp value and update mode
607  if (_throttle_thrust_max >= MIN(get_throttle(), get_current_limit_max_throttle())) {
608  _throttle_thrust_max = get_current_limit_max_throttle();
609  _spool_state = SpoolState::THROTTLE_UNLIMITED;
610  } else if (_throttle_thrust_max < 0.0f) {
611  _throttle_thrust_max = 0.0f;
612  }
613 
614  // initialise motor failure variables
615  _thrust_boost = false;
616  _thrust_boost_ratio = MAX(0.0, _thrust_boost_ratio - 1.0 / (_spool_up_time * _loop_rate));
617  break;
618 
619  case SpoolState::THROTTLE_UNLIMITED:
620  // Throttle should exhibit normal flight behavior.
621  // Servos should exhibit normal flight behavior.
622 
623  // initialize limits flags
624  limit.roll_pitch = false;
625  limit.yaw = false;
626  limit.throttle_lower = false;
627  limit.throttle_upper = false;
628 
629  // make sure the motors are spooling in the correct direction
630  if (_spool_desired != DesiredSpoolState::THROTTLE_UNLIMITED) {
631  _spool_state = SpoolState::SPOOLING_DOWN;
632  break;
633  }
634 
635  // set and increment ramp variables
636  _spin_up_ratio = 1.0f;
637  _throttle_thrust_max = get_current_limit_max_throttle();
638 
639  if (_thrust_boost && !_thrust_balanced) {
640  _thrust_boost_ratio = MIN(1.0, _thrust_boost_ratio+1.0f/(_spool_up_time*_loop_rate));
641  } else {
642  _thrust_boost_ratio = MAX(0.0, _thrust_boost_ratio-1.0f/(_spool_up_time*_loop_rate));
643  }
644  break;
645 
646  case SpoolState::SPOOLING_DOWN:
647  // Maximum throttle should move from maximum to minimum.
648  // Servos should exhibit normal flight behavior.
649 
650  // initialize limits flags
651  limit.roll_pitch = false;
652  limit.yaw = false;
653  limit.throttle_lower = false;
654  limit.throttle_upper = false;
655 
656  // make sure the motors are spooling in the correct direction
657  if (_spool_desired == DesiredSpoolState::THROTTLE_UNLIMITED) {
658  _spool_state = SpoolState::SPOOLING_UP;
659  break;
660  }
661 
662  // set and increment ramp variables
663  _spin_up_ratio = 1.0f;
664  _throttle_thrust_max -= 1.0f/(_spool_up_time*_loop_rate);
665 
666  // constrain ramp value and update mode
667  if (_throttle_thrust_max <= 0.0f){
668  _throttle_thrust_max = 0.0f;
669  }
670  if (_throttle_thrust_max >= get_current_limit_max_throttle()) {
671  _throttle_thrust_max = get_current_limit_max_throttle();
672  } else if (is_zero(_throttle_thrust_max)) {
673  _spool_state = SpoolState::GROUND_IDLE;
674  }
675 
676  _thrust_boost_ratio = MAX(0.0, _thrust_boost_ratio-1.0f/(_spool_up_time*_loop_rate));
677  break;
678  }
679 }

699 // output a thrust to all motors that match a given motor mask. This
700 // is used to control tiltrotor motors in forward flight. Thrust is in
701 // the range 0 to 1
702 void AP_MotorsMulticopter::output_motor_mask(float thrust, uint8_t mask, float rudder_dt)
703 {
704  for (uint8_t i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
705  if (motor_enabled[i]) {
706  if (mask & (1U<<i)) {
707  /*
708  apply rudder mixing differential thrust
709  copter frame roll is plane frame yaw as this only
710  apples to either tilted motors or tailsitters
711  */
712  float diff_thrust = get_roll_factor(i) * rudder_dt * 0.5f;
713  set_actuator_with_slew(_actuator[i], thrust_to_actuator(thrust + diff_thrust));
714  int16_t pwm_output = get_pwm_output_min() +
715  (get_pwm_output_max()-get_pwm_output_min()) * _actuator[i];
716  rc_write(i, pwm_output);
717  } else {
718  rc_write(i, get_pwm_output_min());
719  }
720  }
721  }
722 }
