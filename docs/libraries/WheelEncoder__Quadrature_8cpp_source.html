30 // check if pin has changed and initialise gpio event callback
31 void AP_WheelEncoder_Quadrature::update_pin(uint8_t &pin,
32  uint8_t new_pin,
33  uint8_t &pin_value)
34 {
35  if (new_pin == pin) {
36  // no change
37  return;
38  }
39 
40  // remove old gpio event callback if present
41  if (pin != (uint8_t)-1 &&
42  !hal.gpio->detach_interrupt(pin)) {
43  gcs().send_text(MAV_SEVERITY_WARNING, "WEnc: Failed to detach from pin %u", pin);
44  // ignore this failure or the user may be stuck
45  }
46 
47  pin = new_pin;
48 
49  // install interrupt handler on rising or falling edge of gpio for pin a
50  if (new_pin != (uint8_t)-1) {
51  hal.gpio->pinMode(pin, HAL_GPIO_INPUT);
52  if (!hal.gpio->attach_interrupt(
53  pin,
54  FUNCTOR_BIND_MEMBER(&AP_WheelEncoder_Quadrature::irq_handler,
55  void,
56  uint8_t,
57  bool,
58  uint32_t),
59  AP_HAL::GPIO::INTERRUPT_BOTH)) {
60  gcs().send_text(MAV_SEVERITY_WARNING, "WEnc: Failed to attach to pin %u", pin);
61  }
62  pin_value = hal.gpio->read(pin);
63  }
64 }

84 // convert pin a and pin b state to a wheel encoder phase
85 uint8_t AP_WheelEncoder_Quadrature::pin_ab_to_phase(bool pin_a, bool pin_b)
86 {
87  if (!pin_a) {
88  if (!pin_b) {
89  // A = 0, B = 0
90  return 0;
91  } else {
92  // A = 0, B = 1
93  return 1;
94  }
95  } else {
96  if (!pin_b) {
97  // A = 1, B = 0
98  return 3;
99  } else {
100  // A = 1, B = 1
101  return 2;
102  }
103  }
104  return (uint8_t)pin_a << 1 | (uint8_t)pin_b;
105 }

107 void AP_WheelEncoder_Quadrature::update_phase_and_error_count()
108 {
109  // convert pin state before and after to phases
110  uint8_t phase_after = pin_ab_to_phase(last_pin_a_value, last_pin_b_value);
111 
112  // look for invalid changes
113  uint8_t step_forward = irq_state.phase < 3 ? irq_state.phase+1 : 0;
114  uint8_t step_back = irq_state.phase > 0 ? irq_state.phase-1 : 3;
115  if (phase_after == step_forward) {
116  irq_state.phase = phase_after;
117  irq_state.distance_count++;
118  } else if (phase_after == step_back) {
119  irq_state.phase = phase_after;
120  irq_state.distance_count--;
121  } else {
122  irq_state.error_count++;
123  }
124  irq_state.total_count++;
125 }

128  bool pin_value,

130 {

133  return;
