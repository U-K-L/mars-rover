93 void SoftSigReaderInt::_irq_handler(EICUDriver *eicup, eicuchannel_t aux_channel)
94 {
95  eicuchannel_t channel = get_pair_channel(aux_channel);
96  pulse_t pulse;
97  pulse.w0 = eicup->tim->CCR[channel];
98  pulse.w1 = eicup->tim->CCR[aux_channel];
99 
100  _singleton->sigbuf.push(pulse);
101 
102  //check for missed interrupt
103  uint32_t mask = (STM32_TIM_SR_CC1OF << channel) | (STM32_TIM_SR_CC1OF << aux_channel);
104  if ((eicup->tim->SR & mask) != 0) {
105  //we have missed some pulses
106  //try to reset RCProtocol parser by returning invalid value (i.e. 0 width pulse)
107  pulse.w0 = 0;
108  pulse.w1 = 0;
109  _singleton->sigbuf.push(pulse);
110  //reset overcapture mask
111  eicup->tim->SR &= ~mask;
112  }
113 }

115 bool SoftSigReaderInt::read(uint32_t &widths0, uint32_t &widths1)
116 {
117  if (sigbuf.available() >= 2) {
118  pulse_t pulse;
119  if (sigbuf.pop(pulse)) {
120  widths0 = uint16_t(pulse.w0 - last_value);
121  widths1 = uint16_t(pulse.w1 - pulse.w0);
122  last_value = pulse.w1;
123  return true;
124  }
125  }
126  return false;
127 }
