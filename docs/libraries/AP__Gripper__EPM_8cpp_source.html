97 // update - moves the pwm back to neutral after the timeout has passed
98 // should be called at at least 10hz
99 void AP_Gripper_EPM::update_gripper()
100 {
101  // move EPM PWM output back to neutral after the last grab or release
102  if (AP_HAL::millis() - _last_grab_or_release > EPM_RETURN_TO_NEUTRAL_MS) {
103  if (config.state == AP_Gripper::STATE_GRABBING) {
104  neutral();
105  config.state = AP_Gripper::STATE_GRABBED;
106  } else if (config.state == AP_Gripper::STATE_RELEASING) {
107  neutral();
108  config.state = AP_Gripper::STATE_RELEASED;
109  }
110  }
111 
112  // re-grab the cargo intermittently
113  if (config.state == AP_Gripper::STATE_GRABBED &&
114  (config.regrab_interval > 0) &&
115  (AP_HAL::millis() - _last_grab_or_release > ((uint32_t)config.regrab_interval * 1000))) {
116  grab();
117  }
118 }
