44 // main loop of graphics thread
45 void Display_SITL::update_thread(void)
46 {
47  {
48  WITH_SEMAPHORE(AP::notify().sf_window_mutex);
49  w = new sf::RenderWindow(sf::VideoMode(COLUMNS*SCALE, ROWS*SCALE), "Display");
50  }
51  if (!w) {
52  AP_HAL::panic("Unable to create Display_SITL window");
53  }
54 
55  const sf::Color color_black = sf::Color(0,0,0);
56  const sf::Color color_white = sf::Color(255,255,255);
57 
58  const sf::Uint8 pixels[ROWS*COLUMNS*4]{};
59  sf::Image image;
60  image.create(COLUMNS, ROWS, pixels);
61 
62  while (true) {
63  {
64  WITH_SEMAPHORE(AP::notify().sf_window_mutex);
65  sf::Event event;
66  while (w->pollEvent(event)) {
67  if (event.type == sf::Event::Closed) {
68  w->close();
69  }
70  }
71  if (!w->isOpen()) {
72  break;
73  }
74  if (_need_hw_update) {
75  _need_hw_update = false;
76 
77  uint8_t buffer2[ROWS*COLUMNS];
78  {
79  WITH_SEMAPHORE(mutex);
80  memcpy(buffer2, _displaybuffer, sizeof(buffer2));
81  }
82  w->clear();
83 
84  for (uint16_t y=0; y<ROWS; y++) {
85  for (uint16_t x=0; x<COLUMNS; x++) {
86  if (buffer2[x+y/8*COLUMNS] & 1<<y%8) {
87  image.setPixel(x, y, color_white);
88  } else {
89  image.setPixel(x, y, color_black);
90  }
91  }
92  }
93 
94  sf::Texture texture;
95  texture.loadFromImage(image);
96  sf::Sprite sprite;
97  sprite.setTexture(texture, true);
98  sprite.setScale(SCALE, SCALE);
99  w->draw(sprite);
100 
101  w->display();
102  }
103  }
104  usleep(10000);
105  }
106 }

128 void Display_SITL::set_pixel(uint16_t x, uint16_t y)
129 {
130  // check x, y range
131  if ((x >= COLUMNS) || (y >= ROWS)) {
132  return;
133  }
134  // set pixel in buffer
135  WITH_SEMAPHORE(mutex);
136  _displaybuffer[x + (y / 8 * COLUMNS)] |= 1 << (y % 8);
137  _need_hw_update = true;
138 }
