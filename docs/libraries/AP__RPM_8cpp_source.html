102 /*
103  initialise the AP_RPM class.
104  */
105 void AP_RPM::init(void)
106 {
107  if (num_instances != 0) {
108  // init called a 2nd time?
109  return;
110  }
111  for (uint8_t i=0; i<RPM_MAX_INSTANCES; i++) {
112  uint8_t type = _type[i];
113 
114  if (type == RPM_TYPE_PX4_PWM) {
115  // on non-PX4 treat PX4-pin as AUXPIN option, for upgrade
116  type = RPM_TYPE_PIN;
117  }
118  if (type == RPM_TYPE_PIN) {
119  drivers[i] = new AP_RPM_Pin(*this, i, state[i]);
120  }
121 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
122  drivers[i] = new AP_RPM_SITL(*this, i, state[i]);
123 #endif
124  if (drivers[i] != nullptr) {
125  // we loaded a driver for this instance, so it must be
126  // present (although it may not be healthy)
127  num_instances = i+1; // num_instances is a high-water-mark
128  }
129  }
130 }

132 /*
133  update RPM state for all instances. This should be called by main loop
134  */
135 void AP_RPM::update(void)
136 {
137  for (uint8_t i=0; i<num_instances; i++) {
138  if (drivers[i] != nullptr) {
139  if (_type[i] == RPM_TYPE_NONE) {
140  // allow user to disable an RPM sensor at runtime and force it to re-learn the quality if re-enabled.
141  state[i].signal_quality = 0;
142  continue;
143  }
144 
145  drivers[i]->update();
146  }
147  }
148 }

150 /*
151  check if an instance is healthy
152  */
153 bool AP_RPM::healthy(uint8_t instance) const
154 {
155  if (instance >= num_instances || _type[instance] == RPM_TYPE_NONE) {
156  return false;
157  }
158 
159  // check that data quality is above minimum required
160  if (state[instance].signal_quality < _quality_min[0]) {
161  return false;
162  }
163 
164  return true;
165 }

167 /*

169  */
170 bool AP_RPM::enabled(uint8_t instance) const
171 {
172  if (instance >= num_instances) {
173  return false;
174  }
175  // if no sensor type is selected, the sensor is not activated.
176  if (_type[instance] == RPM_TYPE_NONE) {
177  return false;
178  }
179  return true;
