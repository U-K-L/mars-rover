140 /*
141  parse a seconds since 1970 into a utc_tm structure
142  code based on _der_gmtime from samba
143  */
144 static void parse_utc_seconds(uint64_t utc_sec, struct utc_tm *tm)
145 {
146  uint32_t secday = utc_sec % (3600U * 24U);
147  uint32_t days = utc_sec / (3600U * 24U);
148 
149  memset(tm, 0, sizeof(*tm));
150 
151  tm->tm_sec = secday % 60U;
152  tm->tm_min = (secday % 3600U) / 60U;
153  tm->tm_hour = secday / 3600U;
154  tm->tm_year = 70;
155 
156  if (days > (2000 * 365)) {
157  // don't look for dates too far into the future
158  return;
159  }
160 
161  while (true) {
162  unsigned dayinyear = (is_leap(tm->tm_year) ? 366 : 365);
163  if (days < dayinyear) {
164  break;
165  }
166  tm->tm_year += 1;
167  days -= dayinyear;
168  }
169  tm->tm_mon = 0;
170 
171  while (true) {
172  unsigned daysinmonth = ndays[is_leap(tm->tm_year)?1:0][tm->tm_mon];
173  if (days < daysinmonth) {
174  break;
175  }
176  days -= daysinmonth;
177  tm->tm_mon++;
178  }
179  tm->tm_mday = days + 1;
180 }

183 /*
184  get time for fat filesystem. This is based on
185  rtcConvertDateTimeToFAT from the ChibiOS RTC driver. We don't use
186  the hw RTC clock as it is very inaccurate
187  */
188 uint32_t get_fattime()
189 {
190  if (utc_time_offset == 0) {
191  // return a fixed time
192  return ((uint32_t)0 | (1 << 16)) | (1 << 21);
193  }
194  uint64_t utc_usec = stm32_get_utc_usec();
195  uint64_t utc_sec = utc_usec / 1000000UL;
196  struct utc_tm tm;
197 
198  parse_utc_seconds(utc_sec, &tm);
199 
200  uint32_t fattime;
201 
202  fattime = tm.tm_sec >> 1U;
203  fattime |= tm.tm_min << 5U;
204  fattime |= tm.tm_hour << 11U;
205  fattime |= tm.tm_mday << 16U;
206  fattime |= (tm.tm_mon+1) << 21U;
207  fattime |= (uint32_t)((tm.tm_year-80) << 25U);
208 
209  return fattime;
210 }

222 // set n RTC backup registers starting at given idx
223 void set_rtc_backup(uint8_t idx, const uint32_t *v, uint8_t n)
224 {
225  if ((RCC->BDCR & RCC_BDCR_RTCEN) == 0) {
226  RCC->BDCR |= STM32_RTCSEL;
227  RCC->BDCR |= RCC_BDCR_RTCEN;
228  }
229 #ifdef PWR_CR_DBP
230  PWR->CR |= PWR_CR_DBP;
231 #else
232  PWR->CR1 |= PWR_CR1_DBP;
233 #endif
234  while (n--) {
235  ((__IO uint32_t *)&RTC->BKP0R)[idx++] = *v++;
236  }
237 }
