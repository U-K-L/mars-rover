54 /*
55  look for the sensor on different buses
56  */
57 bool AP_OpticalFlow_PX4Flow::scan_buses(void)
58 {
59  bool success = false;
60  uint8_t retry_attempt = 0;
61 
62  while (!success && retry_attempt < PX4FLOW_INIT_RETRIES) {
63  FOREACH_I2C_EXTERNAL(bus) {
64  #ifdef HAL_OPTFLOW_PX4FLOW_I2C_BUS
65  // only one bus from HAL
66  if (bus != HAL_OPTFLOW_PX4FLOW_I2C_BUS) {
67  continue;
68  }
69  #endif
70  AP_HAL::OwnPtr<AP_HAL::Device> tdev = hal.i2c_mgr->get_device(bus, PX4FLOW_BASE_I2C_ADDR + get_address());
71  if (!tdev) {
72  continue;
73  }
74  WITH_SEMAPHORE(tdev->get_semaphore());
75 
76  struct i2c_integral_frame frame;
77  success = tdev->read_registers(REG_INTEGRAL_FRAME, (uint8_t *)&frame, sizeof(frame));
78  if (success) {
79  printf("Found PX4Flow on bus %u\n", bus);
80  dev = std::move(tdev);
81  break;
82  }
83  }
84  retry_attempt++;
85  if (!success) {
86  hal.scheduler->delay(10);
87  }
88  }
89  return success;
90 }

109 // timer to read sensor
110 void AP_OpticalFlow_PX4Flow::timer(void)
111 {
112  struct i2c_integral_frame frame;
113  if (!dev->read_registers(REG_INTEGRAL_FRAME, (uint8_t *)&frame, sizeof(frame))) {
114  return;
115  }
116  struct OpticalFlow::OpticalFlow_state state {};
117 
118  if (frame.integration_timespan > 0) {
119  const Vector2f flowScaler = _flowScaler();
120  float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
121  float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
122  float integralToRate = 1.0e6 / frame.integration_timespan;
123 
124  state.surface_quality = frame.qual;
125  state.flowRate = Vector2f(frame.pixel_flow_x_integral * flowScaleFactorX,
126  frame.pixel_flow_y_integral * flowScaleFactorY) * 1.0e-4 * integralToRate;
127  state.bodyRate = Vector2f(frame.gyro_x_rate_integral, frame.gyro_y_rate_integral) * 1.0e-4 * integralToRate;
128 
129  _applyYaw(state.flowRate);
130  _applyYaw(state.bodyRate);
131  }
132 
133  _update_frontend(state);
134 }
