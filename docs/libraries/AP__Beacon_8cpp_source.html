239 // create fence boundary points
240 void AP_Beacon::update_boundary_points()
241 {
242  // we need three beacons at least to create boundary fence.
243  // update boundary fence if number of beacons changes
244  if (!device_ready() || num_beacons < AP_BEACON_MINIMUM_FENCE_BEACONS || boundary_num_beacons == num_beacons) {
245  return;
246  }
247 
248  // record number of beacons so we do not repeat calculations
249  boundary_num_beacons = num_beacons;
250 
251  // accumulate beacon points
252  Vector2f beacon_points[AP_BEACON_MAX_BEACONS];
253  for (uint8_t index = 0; index < num_beacons; index++) {
254  const Vector3f& point_3d = beacon_position(index);
255  beacon_points[index].x = point_3d.x;
256  beacon_points[index].y = point_3d.y;
257  }
258 
259  // create polygon around boundary points using the following algorithm
260  // set the "current point" as the first boundary point
261  // loop through all the boundary points looking for the point which creates a vector (from the current point to this new point) with the lowest angle
262  // check if point is already in boundary
263  // - no: add to boundary, move current point to this new point and repeat the above
264  // - yes: we've completed the bounding box, delete any boundary points found earlier than the duplicate
265 
266  Vector2f boundary_points[AP_BEACON_MAX_BEACONS+1]; // array of boundary points
267  uint8_t curr_boundary_idx = 0; // index into boundary_sorted index. always points to the highest filled in element of the array
268  uint8_t curr_beacon_idx = 0; // index into beacon_point array. point indexed is same point as curr_boundary_idx's
269 
270  // initialise first point of boundary_sorted with first beacon's position (this point may be removed later if it is found to not be on the outer boundary)
271  boundary_points[curr_boundary_idx] = beacon_points[curr_beacon_idx];
272 
273  bool boundary_success = false; // true once the boundary has been successfully found
274  bool boundary_failure = false; // true if we fail to build the boundary
275  float start_angle = 0.0f; // starting angle used when searching for next boundary point, on each iteration this climbs but never climbs past PI * 2
276  while (!boundary_success && !boundary_failure) {
277  // look for next outer point
278  uint8_t next_idx;
279  float next_angle;
280  if (get_next_boundary_point(beacon_points, num_beacons, curr_beacon_idx, start_angle, next_idx, next_angle)) {
281  // add boundary point to boundary_sorted array
282  curr_boundary_idx++;
283  boundary_points[curr_boundary_idx] = beacon_points[next_idx];
284  curr_beacon_idx = next_idx;
285  start_angle = next_angle;
286 
287  // check if we have a complete boundary by looking for duplicate points within the boundary_sorted
288  uint8_t dup_idx = 0;
289  bool dup_found = false;
290  while (dup_idx < curr_boundary_idx && !dup_found) {
291  dup_found = (boundary_points[dup_idx] == boundary_points[curr_boundary_idx]);
292  if (!dup_found) {
293  dup_idx++;
294  }
295  }
296  // if duplicate is found, remove all boundary points before the duplicate because they are inner points
297  if (dup_found) {
298  // note that the closing/duplicate point is not
299  // included in the boundary points.
300  const uint8_t num_pts = curr_boundary_idx - dup_idx;
301  if (num_pts >= AP_BEACON_MINIMUM_FENCE_BEACONS) { // we consider three points to be a polygon
302  // success, copy boundary points to boundary array and convert meters to cm
303  for (uint8_t j = 0; j < num_pts; j++) {
304  boundary[j] = boundary_points[j+dup_idx] * 100.0f;
305  }
306  boundary_num_points = num_pts;
307  boundary_success = true;
308  } else {
309  // boundary has too few points
310  boundary_failure = true;
311  }
312  }
313  } else {
314  // failed to create boundary - give up!
315  boundary_failure = true;
316  }
317  }
318 
319  // clear boundary on failure
320  if (boundary_failure) {
321  boundary_num_points = 0;
322  }
323 }
324 
325 // find next boundary point from an array of boundary points given the current index into that array
326 // returns true if a next point can be found
327 // current_index should be an index into the boundary_pts array
328 // start_angle is an angle (in radians), the search will sweep clockwise from this angle
329 // the index of the next point is returned in the next_index argument
330 // the angle to the next point is returned in the next_angle argument
331 bool AP_Beacon::get_next_boundary_point(const Vector2f* boundary_pts, uint8_t num_points, uint8_t current_index, float start_angle, uint8_t& next_index, float& next_angle)
332 {
333  // sanity check
334  if (boundary_pts == nullptr || current_index >= num_points) {
335  return false;
336  }
337 
338  // get current point
339  Vector2f curr_point = boundary_pts[current_index];
340 
341  // search through all points for next boundary point in a clockwise direction
342  float lowest_angle = M_PI_2;
343  float lowest_angle_relative = M_PI_2;
344  bool lowest_found = false;
345  uint8_t lowest_index = 0;
346  for (uint8_t i=0; i < num_points; i++) {
347  if (i != current_index) {
348  Vector2f vec = boundary_pts[i] - curr_point;
349  if (!vec.is_zero()) {
350  float angle = wrap_2PI(atan2f(vec.y, vec.x));
351  float angle_relative = wrap_2PI(angle - start_angle);
352  if ((angle_relative < lowest_angle_relative) || !lowest_found) {
353  lowest_angle = angle;
354  lowest_angle_relative = angle_relative;
355  lowest_index = i;
356  lowest_found = true;
357  }
358  }
359  }
360  }
361 
362  // return results
363  if (lowest_found) {
364  next_index = lowest_index;
365  next_angle = lowest_angle;
366  }
367  return lowest_found;
368 }
