61  struct termios options;
62 
63  tcgetattr(fd, &options);
64  cfsetspeed(&options, baudrate);
65  tcgetattr(fd, &options);
66 
67  if (baudrate == 100000) {
68  // SBUS: 100000bps, even parity, two stop bits
69  options.c_cflag |= (CSTOPB | PARENB);
70  } else {
71  // DSM: 115200, one stop, no parity
72  options.c_cflag &= ~(PARENB|CSTOPB|CSIZE);
73  options.c_cflag |= CS8;
74 
75  }
76  options.c_lflag &= ~(ICANON|ECHO|ECHOE|ISIG);
77  options.c_iflag &= ~(IXON|IXOFF|IXANY);
78  options.c_oflag &= ~OPOST;
79  if (tcsetattr(fd, TCSANOW, &options) != 0) {
80  perror("tcsetattr");
81  exit(1);
82  }
83  tcflush(fd, TCIOFLUSH);
84 
85  rcprot = new AP_RCProtocol();

87 }

89 //Main loop where the action takes place
90 void loop()
91 {
92  uint8_t buf[32];
93  ssize_t ret = read(fd, buf, sizeof(buf));
94  if (ret <= 0) {
95  return;
96  }
97  for (uint8_t i=0; i<ret; i++) {
98  rcprot->process_byte(buf[i], 115200);
99  if (rcprot->new_input()) {
100  uint8_t nchan = rcprot->num_channels();
101  printf("%u: ", nchan);
102  for (uint8_t j=0; j<nchan; j++) {
103  uint16_t v = rcprot->read(j);
104  printf("%04u ", v);
105  }
106  printf("\n");
107  }
108  }
109 }
