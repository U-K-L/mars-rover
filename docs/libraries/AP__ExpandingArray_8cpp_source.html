28 // expand the array by specified number of chunks, returns true on success
29 bool AP_ExpandingArrayGeneric::expand(uint16_t num_chunks)
30 {
31  // expand chunk_ptrs array if necessary
32  if (chunk_count + num_chunks >= chunk_count_max) {
33  uint16_t chunk_ptr_size = chunk_count + num_chunks + chunk_ptr_increment;
34  chunk_ptr_t *chunk_ptrs_new = (chunk_ptr_t*)realloc(chunk_ptrs, chunk_ptr_size * sizeof(chunk_ptr_t));
35  if (chunk_ptrs_new == nullptr) {
36  return false;
37  }
38 
39  // use new pointers array
40  chunk_ptrs = chunk_ptrs_new;
41  chunk_count_max = chunk_ptr_size;
42  }
43 
44  // allocate new chunks
45  for (uint16_t i = 0; i < num_chunks; i++) {
46  uint8_t *new_chunk = (uint8_t *)calloc(chunk_size, elem_size);
47  if (new_chunk == nullptr) {
48  // failed to allocate new chunk
49  return false;
50  }
51  chunk_ptrs[chunk_count] = new_chunk;
52  chunk_count++;
53  }
54  return true;
55 }
