77 bool AP_GPS_NMEA::_decode(char c)
78 {
79  bool valid_sentence = false;
80 
81  _sentence_length++;
82 
83  switch (c) {
84  case ',': // term terminators
85  _parity ^= c;
86  FALLTHROUGH;
87  case '\r':
88  case '\n':
89  case '*':
90  if (_term_offset < sizeof(_term)) {
91  _term[_term_offset] = 0;
92  valid_sentence = _term_complete();
93  }
94  ++_term_number;
95  _term_offset = 0;
96  _is_checksum_term = c == '*';
97  return valid_sentence;
98 
99  case '$': // sentence begin
100  _term_number = _term_offset = 0;
101  _parity = 0;
102  _sentence_type = _GPS_SENTENCE_OTHER;
103  _is_checksum_term = false;
104  _gps_data_good = false;
105  _sentence_length = 1;
106  return valid_sentence;
107  }
108 
109  // ordinary characters
110  if (_term_offset < sizeof(_term) - 1)
111  _term[_term_offset++] = c;
112  if (!_is_checksum_term)
113  _parity ^= c;
114 
115  return valid_sentence;
116 }

131 int32_t AP_GPS_NMEA::_parse_decimal_100(const char *p)
132 {
133  char *endptr = nullptr;
134  long ret = 100 * strtol(p, &endptr, 10);
135  int sign = ret < 0 ? -1 : 1;
136 
137  if (ret >= (long)INT32_MAX) {
138  return INT32_MAX;
139  }
140  if (ret <= (long)INT32_MIN) {
141  return INT32_MIN;
142  }
143  if (endptr == nullptr || *endptr != '.') {
144  return ret;
145  }
146 
147  if (isdigit(endptr[1])) {
148  ret += sign * 10 * DIGIT_TO_VAL(endptr[1]);
149  if (isdigit(endptr[2])) {
150  ret += sign * DIGIT_TO_VAL(endptr[2]);
151  if (isdigit(endptr[3])) {
152  ret += sign * (DIGIT_TO_VAL(endptr[3]) >= 5);
153  }
154  }
155  }
156  return ret;
157 }

159 /*
160  parse a NMEA latitude/longitude degree value. The result is in degrees*1e7
161  */
162 uint32_t AP_GPS_NMEA::_parse_degrees()
163 {
164  char *p, *q;
165  uint8_t deg = 0, min = 0;
166  float frac_min = 0;
167  int32_t ret = 0;
168 
169  // scan for decimal point or end of field
170  for (p = _term; *p && isdigit(*p); p++)
171  ;
172  q = _term;
173 
174  // convert degrees
175  while ((p - q) > 2 && *q) {
176  if (deg)
177  deg *= 10;
178  deg += DIGIT_TO_VAL(*q++);
179  }
180 
181  // convert minutes
182  while (p > q && *q) {
183  if (min)
184  min *= 10;
185  min += DIGIT_TO_VAL(*q++);
186  }
187 
188  // convert fractional minutes
189  if (*p == '.') {
190  q = p + 1;
191  float frac_scale = 0.1f;
192  while (*q && isdigit(*q)) {
193  frac_min += DIGIT_TO_VAL(*q) * frac_scale;
194  q++;
195  frac_scale *= 0.1f;
196  }
197  }
198  ret = (deg * (int32_t)10000000UL);
199  ret += (min * (int32_t)10000000UL / 60);
200  ret += (int32_t) (frac_min * (1.0e7f / 60.0f));
201  return ret;
202 }

204 /*
205  see if we have a new set of NMEA messages
206  */
207 bool AP_GPS_NMEA::_have_new_message()
208 {
209  if (_last_RMC_ms == 0 ||
210  _last_GGA_ms == 0) {
211  return false;
212  }
213  uint32_t now = AP_HAL::millis();
214  if (now - _last_RMC_ms > 150 ||
215  now - _last_GGA_ms > 150) {
216  return false;
217  }
218  if (_last_VTG_ms != 0 &&
219  now - _last_VTG_ms > 150) {
220  return false;
221  }
222  // prevent these messages being used again
223  if (_last_VTG_ms != 0) {
224  _last_VTG_ms = 1;
225  }
226  _last_GGA_ms = 1;
227  _last_RMC_ms = 1;
228  return true;
229 }

231 // Processes a just-completed term
232 // Returns true if new sentence has just passed checksum test and is validated
233 bool AP_GPS_NMEA::_term_complete()
234 {
235  // handle the last term in a message
236  if (_is_checksum_term) {
237  uint8_t checksum = 16 * _from_hex(_term[0]) + _from_hex(_term[1]);
238  if (checksum == _parity) {
239  if (_gps_data_good) {
240  uint32_t now = AP_HAL::millis();
241  switch (_sentence_type) {
242  case _GPS_SENTENCE_RMC:
243  _last_RMC_ms = now;
244  //time = _new_time;
245  //date = _new_date;
246  state.location.lat = _new_latitude;
247  state.location.lng = _new_longitude;
248  state.ground_speed = _new_speed*0.01f;
249  state.ground_course = wrap_360(_new_course*0.01f);
250  make_gps_time(_new_date, _new_time * 10);
251  set_uart_timestamp(_sentence_length);
252  state.last_gps_time_ms = now;
253  fill_3d_velocity();
254  break;
255  case _GPS_SENTENCE_GGA:
256  _last_GGA_ms = now;
257  state.location.alt = _new_altitude;
258  state.location.lat = _new_latitude;
259  state.location.lng = _new_longitude;
260  state.num_sats = _new_satellite_count;
261  state.hdop = _new_hdop;
262  switch(_new_quality_indicator) {
263  case 0: // Fix not available or invalid
264  state.status = AP_GPS::NO_FIX;
265  break;
266  case 1: // GPS SPS Mode, fix valid
267  state.status = AP_GPS::GPS_OK_FIX_3D;
268  break;
269  case 2: // Differential GPS, SPS Mode, fix valid
270  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
271  break;
272  case 3: // GPS PPS Mode, fix valid
273  state.status = AP_GPS::GPS_OK_FIX_3D;
274  break;
275  case 4: // Real Time Kinematic. System used in RTK mode with fixed integers
276  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
277  break;
278  case 5: // Float RTK. Satellite system used in RTK mode, floating integers
279  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
280  break;
281  case 6: // Estimated (dead reckoning) Mode
282  state.status = AP_GPS::NO_FIX;
283  break;
284  default://to maintain compatibility with MAV_GPS_INPUT and others
285  state.status = AP_GPS::GPS_OK_FIX_3D;
286  break;
287  }
288  break;
289  case _GPS_SENTENCE_VTG:
290  _last_VTG_ms = now;
291  state.ground_speed = _new_speed*0.01f;
292  state.ground_course = wrap_360(_new_course*0.01f);
293  fill_3d_velocity();
294  // VTG has no fix indicator, can't change fix status
295  break;
296  }
297  } else {
298  switch (_sentence_type) {
299  case _GPS_SENTENCE_RMC:
300  case _GPS_SENTENCE_GGA:
301  // Only these sentences give us information about
302  // fix status.
303  state.status = AP_GPS::NO_FIX;
304  }
305  }
306  // see if we got a good message
307  return _have_new_message();
308  }
309  // we got a bad message, ignore it
310  return false;
311  }
312 
313  // the first term determines the sentence type
314  if (_term_number == 0) {
315  /*
316  The first two letters of the NMEA term are the talker
317  ID. The most common is 'GP' but there are a bunch of others
318  that are valid. We accept any two characters here.
319  */
320  if (_term[0] < 'A' || _term[0] > 'Z' ||
321  _term[1] < 'A' || _term[1] > 'Z') {
322  _sentence_type = _GPS_SENTENCE_OTHER;
323  return false;
324  }
325  const char *term_type = &_term[2];
326  if (strcmp(term_type, "RMC") == 0) {
327  _sentence_type = _GPS_SENTENCE_RMC;
328  } else if (strcmp(term_type, "GGA") == 0) {
329  _sentence_type = _GPS_SENTENCE_GGA;
330  } else if (strcmp(term_type, "VTG") == 0) {
331  _sentence_type = _GPS_SENTENCE_VTG;
332  // VTG may not contain a data qualifier, presume the solution is good
333  // unless it tells us otherwise.
334  _gps_data_good = true;
335  } else {
336  _sentence_type = _GPS_SENTENCE_OTHER;
337  }
338  return false;
339  }
340 
341  // 32 = RMC, 64 = GGA, 96 = VTG
342  if (_sentence_type != _GPS_SENTENCE_OTHER && _term[0]) {
343  switch (_sentence_type + _term_number) {
344  // operational status
345  //
346  case _GPS_SENTENCE_RMC + 2: // validity (RMC)
347  _gps_data_good = _term[0] == 'A';
348  break;
349  case _GPS_SENTENCE_GGA + 6: // Fix data (GGA)
350  _gps_data_good = _term[0] > '0';
351  _new_quality_indicator = _term[0] - '0';
352  break;
353  case _GPS_SENTENCE_VTG + 9: // validity (VTG) (we may not see this field)
354  _gps_data_good = _term[0] != 'N';
355  break;
356  case _GPS_SENTENCE_GGA + 7: // satellite count (GGA)
357  _new_satellite_count = atol(_term);
358  break;
359  case _GPS_SENTENCE_GGA + 8: // HDOP (GGA)
360  _new_hdop = (uint16_t)_parse_decimal_100(_term);
361  break;
362 
363  // time and date
364  //
365  case _GPS_SENTENCE_RMC + 1: // Time (RMC)
366  case _GPS_SENTENCE_GGA + 1: // Time (GGA)
367  _new_time = _parse_decimal_100(_term);
368  break;
369  case _GPS_SENTENCE_RMC + 9: // Date (GPRMC)
370  _new_date = atol(_term);
371  break;
372 
373  // location
374  //
375  case _GPS_SENTENCE_RMC + 3: // Latitude
376  case _GPS_SENTENCE_GGA + 2:
377  _new_latitude = _parse_degrees();
378  break;
379  case _GPS_SENTENCE_RMC + 4: // N/S
380  case _GPS_SENTENCE_GGA + 3:
381  if (_term[0] == 'S')
382  _new_latitude = -_new_latitude;
383  break;
384  case _GPS_SENTENCE_RMC + 5: // Longitude
385  case _GPS_SENTENCE_GGA + 4:
386  _new_longitude = _parse_degrees();
387  break;
388  case _GPS_SENTENCE_RMC + 6: // E/W
389  case _GPS_SENTENCE_GGA + 5:
390  if (_term[0] == 'W')
391  _new_longitude = -_new_longitude;
392  break;
393  case _GPS_SENTENCE_GGA + 9: // Altitude (GPGGA)
394  _new_altitude = _parse_decimal_100(_term);
395  break;
396 
397  // course and speed
398  //
399  case _GPS_SENTENCE_RMC + 7: // Speed (GPRMC)
400  case _GPS_SENTENCE_VTG + 5: // Speed (VTG)
401  _new_speed = (_parse_decimal_100(_term) * 514) / 1000; // knots-> m/sec, approximiates * 0.514
402  break;
403  case _GPS_SENTENCE_RMC + 8: // Course (GPRMC)
404  case _GPS_SENTENCE_VTG + 1: // Course (VTG)
405  _new_course = _parse_decimal_100(_term);
406  break;
407  }
408  }
409 
410  return false;
411 }

413 /*
414  detect a NMEA GPS. Adds one byte, and returns true if the stream
415  matches a NMEA string
416  */
417 bool
418 AP_GPS_NMEA::_detect(struct NMEA_detect_state &state, uint8_t data)
419 {
420  switch (state.step) {
421  case 0:
422  state.ck = 0;
423  if ('$' == data) {
424  state.step++;
425  }
426  break;
427  case 1:
428  if ('*' == data) {
429  state.step++;
430  } else {
431  state.ck ^= data;
432  }
433  break;
434  case 2:
435  if (hexdigit(state.ck>>4) == data) {
436  state.step++;
437  } else {
438  state.step = 0;
439  }
440  break;
441  case 3:
442  if (hexdigit(state.ck&0xF) == data) {
443  state.step = 0;
444  return true;
445  }
446  state.step = 0;
447  break;
448  }
449  return false;
450 }
