182 // The attitude controller works around the concept of the desired attitude, target attitude
183 // and measured attitude. The desired attitude is the attitude input into the attitude controller
184 // that expresses where the higher level code would like the aircraft to move to. The target attitude is moved
185 // to the desired attitude with jerk, acceleration, and velocity limits. The target angular velocities are fed
186 // directly into the rate controllers. The angular error between the measured attitude and the target attitude is
187 // fed into the angle controller and the output of the angle controller summed at the input of the rate controllers.
188 // By feeding the target angular velocity directly into the rate controllers the measured and target attitudes
189 // remain very close together.
190 //
191 // All input functions below follow the same procedure
192 // 1. define the desired attitude the aircraft should attempt to achieve using the input variables
193 // 2. using the desired attitude and input variables, define the target angular velocity so that it should
194 // move the target attitude towards the desired attitude
195 // 3. if _rate_bf_ff_enabled is not being used then make the target attitude
196 // and target angular velocities equal to the desired attitude and desired angular velocities.
197 // 4. ensure _attitude_target_quat, _attitude_target_euler_angle, _attitude_target_euler_rate and
198 // _attitude_target_ang_vel have been defined. This ensures input modes can be changed without discontinuity.
199 // 5. attitude_controller_run_quat is then run to pass the target angular velocities to the rate controllers and
200 // integrate them into the target attitude. Any errors between the target attitude and the measured attitude are
201 // corrected by first correcting the thrust vector until the angle between the target thrust vector measured
202 // trust vector drops below 2*AC_ATTITUDE_THRUST_ERROR_ANGLE. At this point the heading is also corrected.

436  // rotate body_roll axis by |sin(pitch angle)|
437  Quaternion bf_roll_Q;
438  bf_roll_Q.from_axis_angle(Vector3f(0, 0, spitch * body_roll));

440  // rotate body_yaw axis by cos(pitch angle)
441  Quaternion bf_yaw_Q;
442  bf_yaw_Q.from_axis_angle(Vector3f(cpitch, 0, 0), euler_yaw_rate);

446  // These should be used only for logging target eulers, with the caveat noted above
447  // Also note that _attitude_target_quat.from_euler() should only be used in special circumstances
448  // such as when attitude is specified directly in terms of Euler angles.
449  // _attitude_target_euler_angle.x = _attitude_target_quat.get_euler_roll();
450  // _attitude_target_euler_angle.y = euler_pitch;

453  _attitude_target_euler_rate = Vector3f(0.0f, 0.0f, 0.0f);
454  _attitude_target_ang_vel = Vector3f(0.0f, 0.0f, 0.0f);

645 // Calculates the body frame angular velocities to follow the target attitude
646 void AC_AttitudeControl::attitude_controller_run_quat()
647 {
648  // Retrieve quaternion vehicle attitude
649  Quaternion attitude_vehicle_quat;
650  _ahrs.get_quat_body_to_ned(attitude_vehicle_quat);
651 
652  // Compute attitude error
653  Vector3f attitude_error_vector;
654  thrust_heading_rotation_angles(_attitude_target_quat, attitude_vehicle_quat, attitude_error_vector, _thrust_error_angle);
655 
656  // Compute the angular velocity target from the attitude error
657  _rate_target_ang_vel = update_ang_vel_target_from_att_error(attitude_error_vector);
658 
659  // Add feedforward term that attempts to ensure that roll and pitch errors rotate with the body frame rather than the reference frame.
660  // todo: this should probably be a matrix that couples yaw as well.
661  _rate_target_ang_vel.x += constrain_float(attitude_error_vector.y, -M_PI / 4, M_PI / 4) * _ahrs.get_gyro().z;
662  _rate_target_ang_vel.y += -constrain_float(attitude_error_vector.x, -M_PI / 4, M_PI / 4) * _ahrs.get_gyro().z;
663 
664  ang_vel_limit(_rate_target_ang_vel, radians(_ang_vel_roll_max), radians(_ang_vel_pitch_max), radians(_ang_vel_yaw_max));
665 
666  // Add the angular velocity feedforward, rotated into vehicle frame
667  Quaternion attitude_target_ang_vel_quat = Quaternion(0.0f, _attitude_target_ang_vel.x, _attitude_target_ang_vel.y, _attitude_target_ang_vel.z);
668  Quaternion to_to_from_quat = attitude_vehicle_quat.inverse() * _attitude_target_quat;
669  Quaternion desired_ang_vel_quat = to_to_from_quat.inverse() * attitude_target_ang_vel_quat * to_to_from_quat;
670 
671  // Correct the thrust vector and smoothly add feedforward and yaw input
672  if (_thrust_error_angle > AC_ATTITUDE_THRUST_ERROR_ANGLE * 2.0f) {
673  _rate_target_ang_vel.z = _ahrs.get_gyro().z;
674  } else if (_thrust_error_angle > AC_ATTITUDE_THRUST_ERROR_ANGLE) {
675  float feedforward_scalar = (1.0f - (_thrust_error_angle - AC_ATTITUDE_THRUST_ERROR_ANGLE) / AC_ATTITUDE_THRUST_ERROR_ANGLE);
676  _rate_target_ang_vel.x += desired_ang_vel_quat.q2 * feedforward_scalar;
677  _rate_target_ang_vel.y += desired_ang_vel_quat.q3 * feedforward_scalar;
678  _rate_target_ang_vel.z += desired_ang_vel_quat.q4;
679  _rate_target_ang_vel.z = _ahrs.get_gyro().z * (1.0 - feedforward_scalar) + _rate_target_ang_vel.z * feedforward_scalar;
680  } else {
681  _rate_target_ang_vel.x += desired_ang_vel_quat.q2;
682  _rate_target_ang_vel.y += desired_ang_vel_quat.q3;
683  _rate_target_ang_vel.z += desired_ang_vel_quat.q4;
684  }
685 
686  if (_rate_bf_ff_enabled) {
687  // rotate target and normalize
688  Quaternion attitude_target_update_quat;
689  attitude_target_update_quat.from_axis_angle(Vector3f(_attitude_target_ang_vel.x * _dt, _attitude_target_ang_vel.y * _dt, _attitude_target_ang_vel.z * _dt));
690  _attitude_target_quat = _attitude_target_quat * attitude_target_update_quat;
691  _attitude_target_quat.normalize();
692  }
693 
694  // ensure Quaternions stay normalized
695  _attitude_target_quat.normalize();
696 
697  // Record error to handle EKF resets
698  _attitude_ang_error = attitude_vehicle_quat.inverse() * _attitude_target_quat;
699 }

701 // thrust_heading_rotation_angles - calculates two ordered rotations to move the att_from_quat quaternion to the att_to_quat quaternion.
702 // The first rotation corrects the thrust vector and the second rotation corrects the heading vector.
703 void AC_AttitudeControl::thrust_heading_rotation_angles(Quaternion& att_to_quat, const Quaternion& att_from_quat, Vector3f& att_diff_angle, float& thrust_vec_dot)
704 {
705  Matrix3f att_to_rot_matrix; // rotation from the target body frame to the inertial frame.
706  att_to_quat.rotation_matrix(att_to_rot_matrix);
707  Vector3f att_to_thrust_vec = att_to_rot_matrix * Vector3f(0.0f, 0.0f, 1.0f);
708 
709  Matrix3f att_from_rot_matrix; // rotation from the current body frame to the inertial frame.
710  att_from_quat.rotation_matrix(att_from_rot_matrix);
711  Vector3f att_from_thrust_vec = att_from_rot_matrix * Vector3f(0.0f, 0.0f, 1.0f);
712 
713  // the cross product of the desired and target thrust vector defines the rotation vector
714  Vector3f thrust_vec_cross = att_from_thrust_vec % att_to_thrust_vec;
715 
716  // the dot product is used to calculate the angle between the target and desired thrust vectors
717  thrust_vec_dot = acosf(constrain_float(att_from_thrust_vec * att_to_thrust_vec, -1.0f, 1.0f));
718 
719  // Normalize the thrust rotation vector
720  float thrust_vector_length = thrust_vec_cross.length();
721  if (is_zero(thrust_vector_length) || is_zero(thrust_vec_dot)) {
722  thrust_vec_cross = Vector3f(0, 0, 1);
723  thrust_vec_dot = 0.0f;
724  } else {
725  thrust_vec_cross /= thrust_vector_length;
726  }
727  Quaternion thrust_vec_correction_quat;
728  thrust_vec_correction_quat.from_axis_angle(thrust_vec_cross, thrust_vec_dot);
729 
730  // Rotate thrust_vec_correction_quat to the att_from frame
731  thrust_vec_correction_quat = att_from_quat.inverse() * thrust_vec_correction_quat * att_from_quat;
732 
733  // calculate the remaining rotation required after thrust vector is rotated transformed to the att_from frame
734  Quaternion yaw_vec_correction_quat = thrust_vec_correction_quat.inverse() * att_from_quat.inverse() * att_to_quat;
735 
736  // calculate the angle error in x and y.
737  Vector3f rotation;
738  thrust_vec_correction_quat.to_axis_angle(rotation);
739  att_diff_angle.x = rotation.x;
740  att_diff_angle.y = rotation.y;
741 
742  // calculate the angle error in z (x and y should be zero here).
743  yaw_vec_correction_quat.to_axis_angle(rotation);
744  att_diff_angle.z = rotation.z;
745 
746  // Todo: Limit roll an pitch error based on output saturation and maximum error.
747 
748  // Limit Yaw Error based on maximum acceleration - Update to include output saturation and maximum error.
749  // Currently the limit is based on the maximum acceleration using the linear part of the SQRT controller.
750  // This should be updated to be based on an angle limit, saturation, or unlimited based on user defined parameters.
751  if (!is_zero(_p_angle_yaw.kP()) && fabsf(att_diff_angle.z) > AC_ATTITUDE_ACCEL_Y_CONTROLLER_MAX_RADSS / _p_angle_yaw.kP()) {
752  att_diff_angle.z = constrain_float(wrap_PI(att_diff_angle.z), -AC_ATTITUDE_ACCEL_Y_CONTROLLER_MAX_RADSS / _p_angle_yaw.kP(), AC_ATTITUDE_ACCEL_Y_CONTROLLER_MAX_RADSS / _p_angle_yaw.kP());
753  yaw_vec_correction_quat.from_axis_angle(Vector3f(0.0f, 0.0f, att_diff_angle.z));
754  att_to_quat = att_from_quat * thrust_vec_correction_quat * yaw_vec_correction_quat;
755  }
756 }

1027 // Proportional controller with piecewise sqrt sections to constrain second derivative
1028 float AC_AttitudeControl::sqrt_controller(float error, float p, float second_ord_lim, float dt)
1029 {
1030  float correction_rate;
1031  if (is_negative(second_ord_lim) || is_zero(second_ord_lim)) {
1032  // second order limit is zero or negative.
1033  correction_rate = error * p;
1034  } else if (is_zero(p)) {
1035  // P term is zero but we have a second order limit.
1036  if (is_positive(error)) {
1037  correction_rate = safe_sqrt(2.0f * second_ord_lim * (error));
1038  } else if (is_negative(error)) {
1039  correction_rate = -safe_sqrt(2.0f * second_ord_lim * (-error));
1040  } else {
1041  correction_rate = 0.0f;
1042  }
1043  } else {
1044  // Both the P and second order limit have been defined.
1045  float linear_dist = second_ord_lim / sq(p);
1046  if (error > linear_dist) {
1047  correction_rate = safe_sqrt(2.0f * second_ord_lim * (error - (linear_dist / 2.0f)));
1048  } else if (error < -linear_dist) {
1049  correction_rate = -safe_sqrt(2.0f * second_ord_lim * (-error - (linear_dist / 2.0f)));
1050  } else {
1051  correction_rate = error * p;
1052  }
1053  }
1054  if (!is_zero(dt)) {
1055  // this ensures we do not get small oscillations by over shooting the error correction in the last time step.
1056  return constrain_float(correction_rate, -fabsf(error) / dt, fabsf(error) / dt);
1057  } else {
1058  return correction_rate;
1059  }
1060 }

1062 // Inverse proportional controller with piecewise sqrt sections to constrain second derivative
1063 float AC_AttitudeControl::stopping_point(float first_ord_mag, float p, float second_ord_lim)
1064 {
1065  if (is_positive(second_ord_lim) && !is_zero(second_ord_lim) && is_zero(p)) {
1066  return (first_ord_mag * first_ord_mag) / (2.0f * second_ord_lim);
1067  } else if ((is_negative(second_ord_lim) || is_zero(second_ord_lim)) && !is_zero(p)) {
1068  return first_ord_mag / p;
1069  } else if ((is_negative(second_ord_lim) || is_zero(second_ord_lim)) && is_zero(p)) {
1070  return 0.0f;
1071  }
1072 
1073  // calculate the velocity at which we switch from calculating the stopping point using a linear function to a sqrt function
1074  float linear_velocity = second_ord_lim / p;
1075

1077  // if our current velocity is below the cross-over point we use a linear function

1079  } else {
