154 // find the intersection between two line segments
155 // returns true if they intersect, false if they do not
156 // the point of intersection is returned in the intersection argument
157 template <typename T>
158 bool Vector2<T>::segment_intersection(const Vector2<T>& seg1_start, const Vector2<T>& seg1_end, const Vector2<T>& seg2_start, const Vector2<T>& seg2_end, Vector2<T>& intersection)
159 {
160  // implementation borrowed from http://stackoverflow.com/questions/563198/how-do-you-detect-where-two-line-segments-intersect
161  const Vector2<T> r1 = seg1_end - seg1_start;
162  const Vector2<T> r2 = seg2_end - seg2_start;
163  const Vector2<T> ss2_ss1 = seg2_start - seg1_start;
164  const float r1xr2 = r1 % r2;
165  const float q_pxr = ss2_ss1 % r1;
166  if (fabsf(r1xr2) < FLT_EPSILON) {
167  // either collinear or parallel and non-intersecting
168  return false;
169  } else {
170  // t = (q - p) * s / (r * s)
171  // u = (q - p) * r / (r * s)
172  const float t = (ss2_ss1 % r2) / r1xr2;
173  const float u = q_pxr / r1xr2;
174  if ((u >= 0) && (u <= 1) && (t >= 0) && (t <= 1)) {
175  // lines intersect
176  // t can be any non-negative value because (p, p + r) is a ray
177  // u must be between 0 and 1 because (q, q + s) is a line segment
178  intersection = seg1_start + (r1*t);
179  return true;
180  } else {
181  // non-parallel and non-intersecting
182  return false;
183  }
184  }
185 }

187 // find the intersection between a line segment and a circle
188 // returns true if they intersect and intersection argument is updated with intersection closest to seg_start
189 // solution adapted from http://stackoverflow.com/questions/1073336/circle-line-segment-collision-detection-algorithm
190 template <typename T>
191 bool Vector2<T>::circle_segment_intersection(const Vector2<T>& seg_start, const Vector2<T>& seg_end, const Vector2<T>& circle_center, float radius, Vector2<T>& intersection)
192 {
193  // calculate segment start and end as offsets from circle's center
194  const Vector2f seg_start_local = seg_start - circle_center;
195 
196  // calculate vector from start to end
197  const Vector2f seg_end_minus_start = seg_end - seg_start;
198 
199  const float a = sq(seg_end_minus_start.x) + sq(seg_end_minus_start.y);
200  const float b = 2 * ((seg_end_minus_start.x * seg_start_local.x) + (seg_end_minus_start.y * seg_start_local.y));
201  const float c = sq(seg_start_local.x) + sq(seg_start_local.y) - sq(radius);
202  const float delta = sq(b) - (4.0f * a * c);
203 
204  // check for invalid data
205  if (::is_zero(a)) {
206  return false;
207  }
208  if (isnan(a) || isnan(b) || isnan(c) || isnan(delta)) {
209  return false;
210  }
211 
212  // check for invalid delta (i.e. discriminant)
213  if (delta < 0.0f) {
214  return false;
215  }
216 
217  const float delta_sqrt = sqrtf(delta);
218  const float t1 = (-b + delta_sqrt) / (2.0f * a);
219  const float t2 = (-b - delta_sqrt) / (2.0f * a);
220 
221  // Three hit cases:
222  // -o-> --|--> | | --|->
223  // Impale(t1 hit,t2 hit), Poke(t1 hit,t2>1), ExitWound(t1<0, t2 hit),
224 
225  // Three miss cases:
226  // -> o o -> | -> |
227  // FallShort (t1>1,t2>1), Past (t1<0,t2<0), CompletelyInside(t1<0, t2>1)
228 
229  // intersection = new Vector3(E.x + t1 * d.x, secondPoint.y, E.y + t1 * d.y);
230  // intersection.x = seg_start.x + t1 * seg_end_minus_start.x;
231  // intersection.y = seg_start.y + t1 * seg_end_minus_start.y;
232 
233  if ((t1 >= 0.0f) && (t1 <= 1.0f)) {
234  // t1 is the intersection, and it is closer than t2 (since t1 uses -b - discriminant)
235  // Impale, Poke
236  intersection = seg_start + (seg_end_minus_start * t1);
237  return true;
238  }
239 
240  // here t1 did not intersect so we are either started inside the sphere or completely past it
241  if ((t2 >= 0.0f) && (t2 <= 1.0f)) {
242  // ExitWound
243  intersection = seg_start + (seg_end_minus_start * t2);
244  return true;
245  }
246 
247  // no intersection: FallShort, Past or CompletelyInside
248  return false;
249 }

303 /*
304  * Returns the point closest to p on the line segment (v,w).
305  *
306  * Comments and implementation taken from
307  * http://stackoverflow.com/questions/849211/shortest-distance-between-a-point-and-a-line-segment
308  */
309 template <typename T>
310 Vector2<T> Vector2<T>::closest_point(const Vector2<T> &p, const Vector2<T> &v, const Vector2<T> &w)
311 {
312  // length squared of line segment
313  const float l2 = (v - w).length_squared();
314  if (l2 < FLT_EPSILON) {
315  // v == w case
316  return v;
317  }
318  // Consider the line extending the segment, parameterized as v + t (w - v).
319  // We find projection of point p onto the line.
320  // It falls where t = [(p-v) . (w-v)] / |w-v|^2
321  // We clamp t from [0,1] to handle points outside the segment vw.
322  const float t = ((p - v) * (w - v)) / l2;
323  if (t <= 0) {
324  return v;
325  } else if (t >= 1) {
326  return w;
327  } else {
328  return v + (w - v)*t;
329  }
330 }

332 /*
333  * Returns the point closest to p on the line segment (0,w).
334  *
335  * this is a simplification of closest point with a general segment, with v=(0,0)
336  */
337 template <typename T>
338 Vector2<T> Vector2<T>::closest_point(const Vector2<T> &p, const Vector2<T> &w)
339 {
340  // length squared of line segment
341  const float l2 = w.length_squared();
342  if (l2 < FLT_EPSILON) {
343  // v == w case
344  return w;
345  }
346  const float t = (p * w) / l2;
347  if (t <= 0) {
348  return Vector2<T>(0,0);
349  } else if (t >= 1) {
350  return w;
351  } else {
352  return w*t;
353  }
