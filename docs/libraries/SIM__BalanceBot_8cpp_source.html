43 /*
44  update the Balance Bot simulation by one time step
45  */
46 /*
47  * The balance bot is physically modeled as an inverted pendulum(cuboid) on wheels
48  * Further details on the equations used can be found here:
49  * 1) http://robotics.ee.uwa.edu.au/theses/2003-Balance-Ooi.pdf page 33 onwards
50  * 2) http://journals.sagepub.com/doi/pdf/10.5772/63933
51  */
52 void BalanceBot::update(const struct sitl_input &input)
53 {
54  // pendulum/chassis constants
55  const float m_p = 3.060f; //pendulum mass(kg)
56  const float width = 0.0650f; //width(m)
57  const float height = 0.240f; //height(m)
58  const float l = 0.120f; //height of center of mass from base(m)
59  const float i_p = (1/12.0f)*m_p*(width*width + height*height); //Moment of inertia about pitch axis(SI units)
60 
61  // wheel constants
62  const float r_w = 0.10f; //wheel radius(m)
63  const float m_w = 0.120f; //wheel mass(kg)
64  const float i_w = 0.5f*m_w*r_w*r_w; // moment of inertia of wheel(SI units)
65 
66  // motor constants
67  const float R = 1.0f; //Winding resistance(ohm)
68  const float k_e = 0.13f; //back-emf constant(SI units)
69  const float k_t = 0.242f; //torque constant(SI units)
70  const float v_max = 12.0f; //max input voltage(V)
71 
72  // balance bot uses skid steering
73  const float motor1 = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
74  const float motor2 = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
75  const float steering = motor1 - motor2;
76  const float throttle = 0.5 * (motor1 + motor2);
77 
78  // motor input voltage: (throttle/max_throttle)*v_max
79  const float v = throttle*v_max;
80 
81  // how much time has passed?
82  const float delta_time = frame_time_us * 1.0e-6f;
83 
84  // yaw rate in degrees/s
85  const float yaw_rate = calc_yaw_rate(steering);
86 
87  // obtain roll, pitch, yaw from dcm
88  float r, p, y;
89  dcm.to_euler(&r, &p, &y);
90  float theta = p; //radians
91 
92  float ang_vel = gyro.y; //radians/s
93 
94  // t1,t2,t3 are terms in the equation to find vehicle frame x acceleration
95  const float t1 = ((2.0f*k_t*v/(R*r_w)) - (2.0f*k_t*k_e*velocity_vf_x/(R*r_w*r_w)) - (m_p*l*ang_vel*ang_vel*sin(theta))) * (i_p + m_p*l*l);
96  const float t2 = -m_p*l*cos(theta)*((2.0f*k_t*k_e*velocity_vf_x/(R*r_w)) - (2.0f*k_t*v/(R)) + (m_p*GRAVITY_MSS*l*sin(theta)));
97  const float t3 = ( ((2.0f*m_w + 2.0f*i_w/(r_w*r_w) + m_p) * (i_p + m_p*l*l)) - (m_p*m_p*l*l*cos(theta)*cos(theta)) );
98 
99  //vehicle frame x acceleration
100  const float accel_vf_x = (t1-t2)/t3;
101 
102  const float angular_accel_bf_y = ((2.0f*k_t*k_e*velocity_vf_x/(R*r_w)) - (2.0f*k_t*v/(R)) + m_p*l*accel_vf_x*cos(theta) + m_p*GRAVITY_MSS*l*sin(theta))
103  / (i_p + m_p*l*l);
104 
105  // update theta and angular velocity
106  ang_vel += angular_accel_bf_y * delta_time;
107  theta += ang_vel * delta_time;
108  theta = fmod(theta, radians(360));
109 
110  // update x velocity in vehicle frame
111  velocity_vf_x += accel_vf_x * delta_time;
112 
113  gyro = Vector3f(0, ang_vel, radians(yaw_rate));
114 
115  // update attitude
116  dcm.rotate(gyro * delta_time);
117  dcm.normalize();
118 
119  // accel in body frame due to motor
120  accel_body = Vector3f(accel_vf_x*cos(theta), 0, -accel_vf_x*sin(theta));
121 
122  // add in accel due to direction change
123  accel_body.y += radians(yaw_rate) * velocity_vf_x;
124 
125  // now in earth frame
126  Vector3f accel_earth = dcm * accel_body;
127  accel_earth += Vector3f(0, 0, GRAVITY_MSS);
128 
129  // we are on the ground, so our vertical accel is zero
130  accel_earth.z = 0;
131 
132  if (!hal.util->get_soft_armed()) {
133  // reset to vertical when not armed for faster testing
134  accel_earth.zero();
135  velocity_ef.zero();
136  dcm.identity();
137  gyro.zero();
138  velocity_vf_x =0;
139  theta = radians(0);
140  ang_vel = 0;
141  }
142 
143  // work out acceleration as seen by the accelerometers. It sees the kinematic
144  // acceleration (ie. real movement), plus gravity
145  accel_body += dcm.transposed() * (Vector3f(0, 0, -GRAVITY_MSS));
146 
147  // new velocity vector
148  velocity_ef += accel_earth * delta_time;
149 
150  // new position vector
151  position += (velocity_ef * delta_time);
152 
153  // neglect roll
154  dcm.to_euler(&r, &p, &y);
155  dcm.from_euler(0.0f, p, y);
156  use_smoothing = true;
157 
158  // update lat/lon/altitude
159  update_position();
160  time_advance();
161 
162  // update magnetic field
163  update_mag_field_bf();
164 }
