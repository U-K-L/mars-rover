35 /*
36  request any missing 4x4 grids from a block, given a grid_cache
37  */
38 bool AP_Terrain::request_missing(mavlink_channel_t chan, struct grid_cache &gcache)
39 {
40  struct grid_block &grid = gcache.grid;
41 
42  if (grid.spacing != grid_spacing) {
43  // an invalid grid
44  return false;
45  }
46 
47  // see if we are waiting for disk read
48  if (gcache.state == GRID_CACHE_DISKWAIT) {
49  // don't request data from the GCS till we know it's not on disk
50  return false;
51  }
52 
53  // see if it is fully populated
54  if ((grid.bitmap & bitmap_mask) == bitmap_mask) {
55  // it is fully populated, nothing to do
56  return false;
57  }
58 
59  if (!HAVE_PAYLOAD_SPACE(chan, TERRAIN_REQUEST)) {
60  // not enough buffer space
61  return false;
62  }
63 
64  /*
65  ask the GCS to send a set of 4x4 grids
66  */
67  mavlink_msg_terrain_request_send(chan, grid.lat, grid.lon, grid_spacing, bitmap_mask & ~grid.bitmap);
68  last_request_time_ms[chan] = AP_HAL::millis();
69 
70  return true;
71 }

83 /*
84  send any pending terrain request to the GCS
85  */
86 void AP_Terrain::send_request(mavlink_channel_t chan)
87 {
88  if (!allocate()) {
89  // not enabled
90  return;
91  }
92 
93  // see if we need to schedule some disk IO
94  schedule_disk_io();
95 
96  Location loc;
97  if (!AP::ahrs().get_position(loc)) {
98  // we don't know where we are
99  return;
100  }
101 
102  // always send a terrain report
103  send_terrain_report(chan, loc, true);
104 
105  // did we request recently?
106  if (AP_HAL::millis() - last_request_time_ms[chan] < 2000) {
107  // too soon to request again
108  return;
109  }
110 
111  // request any missing 4x4 blocks in the current grid
112  struct grid_info info;
113  calculate_grid_info(loc, info);
114 
115  if (request_missing(chan, info)) {
116  return;
117  }
118 
119  // also request a larger set of up to 9 grids
120  for (int8_t x=-1; x<=1; x++) {
121  for (int8_t y=-1; y<=1; y++) {
122  Location loc2 = loc;
123  loc2.offset(x*TERRAIN_GRID_BLOCK_SIZE_X*0.7f*grid_spacing,
124  y*TERRAIN_GRID_BLOCK_SIZE_Y*0.7f*grid_spacing);
125  struct grid_info info2;
126  calculate_grid_info(loc2, info2);
127  if (request_missing(chan, info2)) {
128  return;
129  }
130  }
131  }
132 
133  // check cache blocks that may have been setup by a TERRAIN_CHECK
134  for (uint16_t i=0; i<cache_size; i++) {
135  if (cache[i].state >= GRID_CACHE_VALID) {
136  if (request_missing(chan, cache[i])) {
137  return;
138  }
139  }
140  }
141 
142  // request the current loc last to ensure it has highest last
143  // access time
144  if (request_missing(chan, info)) {
145  return;
146  }
147 }

157 /*
158  get some statistics for TERRAIN_REPORT
159 */
160 void AP_Terrain::get_statistics(uint16_t &pending, uint16_t &loaded)
161 {
162  pending = 0;
163  loaded = 0;
164  for (uint16_t i=0; i<cache_size; i++) {
165  if (cache[i].grid.spacing != grid_spacing) {
166  continue;
167  }
168  if (cache[i].state == GRID_CACHE_INVALID) {
169  continue;
170  }
171  uint8_t maskbits = TERRAIN_GRID_BLOCK_MUL_X*TERRAIN_GRID_BLOCK_MUL_Y;
172  if (cache[i].state == GRID_CACHE_DISKWAIT) {
173  pending += maskbits;
174  continue;
175  }
176  if (cache[i].state == GRID_CACHE_DIRTY) {
177  // count dirty grids as a pending, so we know where there
178  // are disk writes pending
179  pending++;
180  }
181  uint8_t bitcount = bitcount64(cache[i].grid.bitmap);
182  pending += maskbits - bitcount;
183  loaded += bitcount;
184  }
185 }

201 /*
202  send a TERRAIN_REPORT for a location
203  */
204 void AP_Terrain::send_terrain_report(mavlink_channel_t chan, const Location &loc, bool extrapolate)
205 {
206  float terrain_height = 0;
207  float home_terrain_height = 0;
208  uint16_t spacing = 0;
209  Location current_loc;
210  const AP_AHRS &ahrs = AP::ahrs();
211  if (ahrs.get_position(current_loc) &&
212  height_amsl(ahrs.get_home(), home_terrain_height, false) &&
213  height_amsl(loc, terrain_height, false)) {
214  // non-zero spacing indicates we have data
215  spacing = grid_spacing;
216  } else if (extrapolate && have_current_loc_height) {
217  // show the extrapolated height, so logs show what height is
218  // being used for navigation
219  terrain_height = last_current_loc_height;
220  }
221  uint16_t pending, loaded;
222  get_statistics(pending, loaded);
223 
224  float current_height;
225  if (spacing == 0 && !(extrapolate && have_current_loc_height)) {
226  current_height = 0;
227  } else {
228  if (current_loc.relative_alt) {
229  current_height = current_loc.alt*0.01f;
230  } else {
231  current_height = (current_loc.alt - ahrs.get_home().alt)*0.01f;
232  }
233  }
234  current_height += home_terrain_height - terrain_height;
235 
236  if (HAVE_PAYLOAD_SPACE(chan, TERRAIN_REPORT)) {
237  mavlink_msg_terrain_report_send(chan, loc.lat, loc.lng, spacing,
238  terrain_height, current_height,
239  pending, loaded);
240  }
241 }

256 /*
257  handle TERRAIN_DATA messages from GCS
258  */
259 void AP_Terrain::handle_terrain_data(mavlink_message_t *msg)
260 {
261  mavlink_terrain_data_t packet;
262  mavlink_msg_terrain_data_decode(msg, &packet);
263 
264  uint16_t i;
265  for (i=0; i<cache_size; i++) {
266  if (cache[i].grid.lat == packet.lat &&
267  cache[i].grid.lon == packet.lon &&
268  cache[i].grid.spacing == packet.grid_spacing &&
269  grid_spacing == packet.grid_spacing &&
270  packet.gridbit < 56) {
271  break;
272  }
273  }
274  if (i == cache_size) {
275  // we don't have that grid, ignore data
276  return;
277  }
278  struct grid_cache &gcache = cache[i];
279  struct grid_block &grid = gcache.grid;
280  uint8_t idx_x = (packet.gridbit / TERRAIN_GRID_BLOCK_MUL_Y) * TERRAIN_GRID_MAVLINK_SIZE;
281  uint8_t idx_y = (packet.gridbit % TERRAIN_GRID_BLOCK_MUL_Y) * TERRAIN_GRID_MAVLINK_SIZE;
282  ASSERT_RANGE(idx_x,0,(TERRAIN_GRID_BLOCK_MUL_X-1)*TERRAIN_GRID_MAVLINK_SIZE);
283  ASSERT_RANGE(idx_y,0,(TERRAIN_GRID_BLOCK_MUL_Y-1)*TERRAIN_GRID_MAVLINK_SIZE);
284  for (uint8_t x=0; x<TERRAIN_GRID_MAVLINK_SIZE; x++) {
285  for (uint8_t y=0; y<TERRAIN_GRID_MAVLINK_SIZE; y++) {
286  grid.height[idx_x+x][idx_y+y] = packet.data[x*TERRAIN_GRID_MAVLINK_SIZE+y];
287  }
288  }
289  gcache.grid.bitmap |= ((uint64_t)1) << packet.gridbit;
290 
291  // mark dirty for disk IO
292  gcache.state = GRID_CACHE_DIRTY;
293 
294 #if TERRAIN_DEBUG
295  hal.console->printf("Filled bit %u idx_x=%u idx_y=%u\n",
296  (unsigned)packet.gridbit, (unsigned)idx_x, (unsigned)idx_y);
297  if (gcache.grid.bitmap == bitmap_mask) {
298  hal.console->printf("--lat=%12.7f --lon=%12.7f %u\n",
299  grid.lat*1.0e-7f,
300  grid.lon*1.0e-7f,
301  grid.height[0][0]);
302  Location loc2;
303  loc2.lat = grid.lat;
304  loc2.lng = grid.lon;
305  loc2.offset(28*grid_spacing, 32*grid_spacing);
306  hal.console->printf("--lat=%12.7f --lon=%12.7f %u\n",
307  loc2.lat*1.0e-7f,
308  loc2.lng*1.0e-7f,
309  grid.height[27][31]);
310  }
311 #endif
312 
313  // see if we need to schedule some disk IO
314  update();
315 }
