34 void SLCANRouter::run()
35 {
36  _port = AP_SerialManager::get_singleton()->get_serial_by_id(AP::can().get_slcan_serial());
37  if (_slcan_if.init(921600, SLCAN::CAN::OperatingMode::NormalMode, _port) < 0) {
38  return;
39  }
40  _slcan_rt_timeout = AP::can().get_slcan_timeout();
41  if (!_thread_started) {
42  hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&SLCANRouter::can2slcan_router_trampoline, void), "C2SRouter", 512, AP_HAL::Scheduler::PRIORITY_CAN, 0);
43  hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&SLCANRouter::slcan2can_router_trampoline, void), "S2CRouter", 512, AP_HAL::Scheduler::PRIORITY_CAN, 0);
44  _thread_started = true;
45  _thread_suspended = false;
46  }
47  else if (_thread_suspended) { //wake up threads
48  chSysLock();
49  if (_c2s_thd_ref && _s2c_thd_ref) {
50  chThdResumeS(&_c2s_thd_ref, MSG_OK);
51  chThdResumeS(&_s2c_thd_ref, MSG_OK);
52  _thread_suspended = false;
53  }
54  chSysUnlock();
55  }
56 }

58 void SLCANRouter::timer()
59 {
60  if ((!_thread_started || _thread_suspended) && (AP::can().get_slcan_serial() != -1)) {
61  run();
62  AP::can().reset_slcan_serial();
63  _last_active_time = AP_HAL::millis();
64  }
65  if (!_slcan_if.closed()) {
66  _last_active_time = AP_HAL::millis();
67  }
68  if (_thread_suspended) {
69  return;
70  }
71  if (AP_HAL::millis() - _last_active_time > (_slcan_rt_timeout * 1000) && _slcan_rt_timeout != 0) {
72  chSysLock();
73  _port->lock_port(0, 0);
74  _port->flush();
75  _thread_suspended = true;
76  chSysUnlock();
77  }
78 }

105 void SLCANRouter::slcan2can_router_trampoline(void)
106 {
107  CanRouteItem it;
108  while (true) {
109  chSysLock();
110  _s2c_thd_ref = nullptr;
111  if (_thread_suspended) {
112  chThdSuspendS(&_s2c_thd_ref);
113  _port->flush();
114  }
115  chSysUnlock();
116  _slcan_if.reader();
117  if (_can_tx_queue.available() && _can_if) {
118  _can_tx_queue.peek(it);
119  if (_can_if->send(it.frame, uavcan::MonotonicTime::fromUSec(AP_HAL::micros64() + 1000), 0)) {
120  _can_tx_queue.pop();
121  }
122  }
123  }
124 }
