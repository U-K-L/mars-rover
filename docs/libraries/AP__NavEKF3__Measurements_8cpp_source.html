687 // correct the height of the EKF origin to be consistent with GPS Data using a Bayes filter.
688 void NavEKF3_core::correctEkfOriginHeight()
689 {
690  // Estimate the WGS-84 height of the EKF's origin using a Bayes filter
691 
692  // calculate the variance of our a-priori estimate of the ekf origin height
693  float deltaTime = constrain_float(0.001f * (imuDataDelayed.time_ms - lastOriginHgtTime_ms), 0.0f, 1.0f);
694  if (activeHgtSource == HGT_SOURCE_BARO) {
695  // Use the baro drift rate
696  const float baroDriftRate = 0.05f;
697  ekfOriginHgtVar += sq(baroDriftRate * deltaTime);
698  } else if (activeHgtSource == HGT_SOURCE_RNG) {
699  // use the worse case expected terrain gradient and vehicle horizontal speed
700  const float maxTerrGrad = 0.25f;
701  ekfOriginHgtVar += sq(maxTerrGrad * norm(stateStruct.velocity.x , stateStruct.velocity.y) * deltaTime);
702  } else {
703  // by definition our height source is absolute so cannot run this filter
704  return;
705  }
706  lastOriginHgtTime_ms = imuDataDelayed.time_ms;
707 
708  // calculate the observation variance assuming EKF error relative to datum is independent of GPS observation error
709  // when not using GPS as height source
710  float originHgtObsVar = sq(gpsHgtAccuracy) + P[9][9];
711 
712  // calculate the correction gain
713  float gain = ekfOriginHgtVar / (ekfOriginHgtVar + originHgtObsVar);
714 
715  // calculate the innovation
716  float innovation = - stateStruct.position.z - gpsDataDelayed.hgt;
717 
718  // check the innovation variance ratio
719  float ratio = sq(innovation) / (ekfOriginHgtVar + originHgtObsVar);
720 
721  // correct the EKF origin and variance estimate if the innovation is less than 5-sigma
722  if (ratio < 25.0f && gpsAccuracyGood) {
723  ekfGpsRefHgt -= (double)(gain * innovation);
724  ekfOriginHgtVar -= MAX(gain * ekfOriginHgtVar , 0.0f);
725  }
