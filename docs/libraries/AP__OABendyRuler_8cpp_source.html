21 const int16_t OA_BENDYRULER_BEARING_INC = 5; // check every 5 degrees around vehicle
22 const float OA_BENDYRULER_LOOKAHEAD_STEP2_RATIO = 1.0f; // step2's lookahead length as a ratio of step1's lookahead length
23 const float OA_BENDYRULER_LOOKAHEAD_STEP2_MIN = 2.0f; // step2 checks at least this many meters past step1's location
24 const float OA_BENDYRULER_LOOKAHEAD_PAST_DEST = 2.0f; // lookahead length will be at least this many meters past the destination
25 const float OA_BENDYRULER_LOW_SPEED_SQUARED = (0.2f * 0.2f); // when ground course is below this speed squared, vehicle's heading will be used

27 // run background task to find best path and update avoidance_results
28 // returns true and updates origin_new and destination_new if a best path has been found
29 bool AP_OABendyRuler::update(const Location& current_loc, const Location& destination, const Vector2f &ground_speed_vec, Location &origin_new, Location &destination_new)
30 {
31  // bendy ruler always sets origin to current_loc
32  origin_new = current_loc;
33 
34  // calculate bearing and distance to final destination
35  const float bearing_to_dest = current_loc.get_bearing_to(destination) * 0.01f;
36  const float distance_to_dest = current_loc.get_distance(destination);
37 
38  // lookahead distance is adjusted dynamically based on avoidance results
39  _current_lookahead = constrain_float(_current_lookahead, _lookahead * 0.5f, _lookahead);
40 
41  // calculate lookahead dist and time for step1. distance can be slightly longer than
42  // the distance to the destination to allow room to dodge after reaching the destination
43  const float lookahead_step1_dist = MIN(_current_lookahead, distance_to_dest + OA_BENDYRULER_LOOKAHEAD_PAST_DEST);
44 
45  // calculate lookahead dist for step2
46  const float lookahead_step2_dist = _current_lookahead * OA_BENDYRULER_LOOKAHEAD_STEP2_RATIO;
47 
48  // get ground course
49  float ground_course_deg;
50  if (ground_speed_vec.length_squared() < OA_BENDYRULER_LOW_SPEED_SQUARED) {
51  // with zero ground speed use vehicle's heading
52  ground_course_deg = AP::ahrs().yaw_sensor * 0.01f;
53  } else {
54  ground_course_deg = degrees(ground_speed_vec.angle());
55  }
56 
57  // check OA_BEARING_INC definition allows checking in all directions
58  static_assert(360 % OA_BENDYRULER_BEARING_INC == 0, "check 360 is a multiple of OA_BEARING_INC");
59 
60  // search in OA_BENDYRULER_BEARING_INC degree increments around the vehicle alternating left
61  // and right. For each direction check if vehicle would avoid all obstacles
62  float best_bearing = bearing_to_dest;
63  bool have_best_bearing = false;
64  float best_margin = -FLT_MAX;
65  float best_margin_bearing = best_bearing;
66 
67  for (uint8_t i = 0; i <= (170 / OA_BENDYRULER_BEARING_INC); i++) {
68  for (uint8_t bdir = 0; bdir <= 1; bdir++) {
69  // skip duplicate check of bearing straight towards destination
70  if ((i==0) && (bdir > 0)) {
71  continue;
72  }
73  // bearing that we are probing
74  const float bearing_delta = i * OA_BENDYRULER_BEARING_INC * (bdir == 0 ? -1.0f : 1.0f);
75  const float bearing_test = wrap_180(bearing_to_dest + bearing_delta);
76 
77  // ToDo: add effective groundspeed calculations using airspeed
78  // ToDo: add prediction of vehicle's position change as part of turn to desired heading
79 
80  // test location is projected from current location at test bearing
81  Location test_loc = current_loc;
82  test_loc.offset_bearing(bearing_test, lookahead_step1_dist);
83 
84  // calculate margin from fence for this scenario
85  float margin = calc_avoidance_margin(current_loc, test_loc);
86  if (margin > best_margin) {
87  best_margin_bearing = bearing_test;
88  best_margin = margin;
89  }
90  if (margin > _margin_max) {
91  // this bearing avoids obstacles out to the lookahead_step1_dist
92  // now check in there is a clear path in three directions towards the destination
93  if (!have_best_bearing) {
94  best_bearing = bearing_test;
95  have_best_bearing = true;
96  } else if (fabsf(wrap_180(ground_course_deg - bearing_test)) <
97  fabsf(wrap_180(ground_course_deg - best_bearing))) {
98  // replace bearing with one that is closer to our current ground course
99  best_bearing = bearing_test;
100  }
101 
102  // perform second stage test in three directions looking for obstacles
103  const float test_bearings[] { 0.0f, 45.0f, -45.0f };
104  const float bearing_to_dest2 = test_loc.get_bearing_to(destination) * 0.01f;
105  float distance2 = constrain_float(lookahead_step2_dist, OA_BENDYRULER_LOOKAHEAD_STEP2_MIN, test_loc.get_distance(destination));
106  for (uint8_t j = 0; j < ARRAY_SIZE(test_bearings); j++) {
107  float bearing_test2 = wrap_180(bearing_to_dest2 + test_bearings[j]);
108  Location test_loc2 = test_loc;
109  test_loc2.offset_bearing(bearing_test2, distance2);
110 
111  // calculate minimum margin to fence and obstacles for this scenario
112  float margin2 = calc_avoidance_margin(test_loc, test_loc2);
113  if (margin2 > _margin_max) {
114  // all good, now project in the chosen direction by the full distance
115  destination_new = current_loc;
116  destination_new.offset_bearing(bearing_test, distance_to_dest);
117  _current_lookahead = MIN(_lookahead, _current_lookahead * 1.1f);
118  // if the chosen direction is directly towards the destination turn off avoidance
119  return (i != 0 || j != 0);
120  }
121  }
122  }
123  }
124  }
125 
126  float chosen_bearing;
127  if (have_best_bearing) {
128  // none of the directions tested were OK for 2-step checks. Choose the direction
129  // that was best for the first step
130  chosen_bearing = best_bearing;
131  _current_lookahead = MIN(_lookahead, _current_lookahead * 1.05f);
132  } else {
133  // none of the possible paths had a positive margin. Choose
134  // the one with the highest margin
135  chosen_bearing = best_margin_bearing;
136  _current_lookahead = MAX(_lookahead * 0.5f, _current_lookahead * 0.9f);
137  }
138 
139  // calculate new target based on best effort
140  destination_new = current_loc;
141  destination_new.offset_bearing(chosen_bearing, distance_to_dest);
142 
143  return true;
144 }

146 // calculate minimum distance between a segment and any obstacle
147 float AP_OABendyRuler::calc_avoidance_margin(const Location &start, const Location &end)
148 {
149  float circular_fence_margin;
150  if (!calc_margin_from_circular_fence(start, end, circular_fence_margin)) {
151  circular_fence_margin = FLT_MAX;
152  }
153 
154  float polygon_fence_margin;
155  if (!calc_margin_from_polygon_fence(start, end, polygon_fence_margin)) {
156  polygon_fence_margin = FLT_MAX;
157  }
158 
159  float proximity_margin;
160  if (!calc_margin_from_proximity_sensors(start, end, proximity_margin)) {
161  proximity_margin = FLT_MAX;
162  }
163 
164  // return smallest margin from any obstacle
165  return MIN(MIN(circular_fence_margin, polygon_fence_margin), proximity_margin);
166 }

168 // calculate minimum distance between a path and the circular fence (centered on home)
169 // on success returns true and updates margin
170 bool AP_OABendyRuler::calc_margin_from_circular_fence(const Location &start, const Location &end, float &margin)
171 {
172  // exit immediately if polygon fence is not enabled
173  const AC_Fence *fence = AC_Fence::get_singleton();
174  if (fence == nullptr) {
175  return false;
176  }
177  if ((fence->get_enabled_fences() & AC_FENCE_TYPE_CIRCLE) == 0) {
178  return false;
179  }
180 
181  // calculate start and end point's distance from home
182  const Location &ahrs_home = AP::ahrs().get_home();
183  const float start_dist_sq = ahrs_home.get_distance_NE(start).length_squared();
184  const float end_dist_sq = ahrs_home.get_distance_NE(end).length_squared();
185 
186  // get circular fence radius
187  const float fence_radius = fence->get_radius();
188 
189  // margin is fence radius minus the longer of start or end distance
190  margin = fence_radius - sqrtf(MAX(start_dist_sq, end_dist_sq));
191  return true;
192 }

194 // calculate minimum distance between a path and the polygon fence
195 // on success returns true and updates margin
196 bool AP_OABendyRuler::calc_margin_from_polygon_fence(const Location &start, const Location &end, float &margin)
197 {
198  // exit immediately if polygon fence is not enabled
199  const AC_Fence *fence = AC_Fence::get_singleton();
200  if (fence == nullptr) {
201  return false;
202  }
203  if (((fence->get_enabled_fences() & AC_FENCE_TYPE_POLYGON) == 0) || !fence->is_polygon_valid()) {
204  return false;
205  }
206 
207  // get polygon boundary
208  uint16_t num_points;
209  const Vector2f* boundary = fence->get_boundary_points(num_points);
210  if (num_points < 3) {
211  // this should have already been checked by is_polygon_valid() but just in case
212  return false;
213  }
214 
215  // convert start and end to offsets from EKF origin
216  Vector2f start_NE, end_NE;
217  if (!start.get_vector_xy_from_origin_NE(start_NE) || !end.get_vector_xy_from_origin_NE(end_NE)) {
218  return false;
219  }
220 
221  // if outside the fence margin is the closest distance but with negative sign
222  const float sign = Polygon_outside(start_NE, boundary, num_points) ? -1.0f : 1.0f;
223 
224  // calculate min distance (in meters) from line to polygon
225  margin = sign * Polygon_closest_distance_line(boundary, num_points, start_NE, end_NE) * 0.01f;
226  return true;
227 }

229 // calculate minimum distance between a path and proximity sensor obstacles
230 // on success returns true and updates margin
231 bool AP_OABendyRuler::calc_margin_from_proximity_sensors(const Location &start, const Location &end, float &margin)
232 {
233  AP_Proximity* prx = AP_Proximity::get_singleton();
234  if (prx == nullptr) {
235  return false;
236  }
237 
238  // retrieve obstacles from proximity library
239  if (!prx->copy_locations(_prx_locs, ARRAY_SIZE(_prx_locs), _prx_loc_count)) {
240  return false;
241  }
242 
243  // exit if no obstacles
244  if (_prx_loc_count == 0) {
245  return false;
246  }
247 
248  // convert start and end to offsets (in cm) from EKF origin
249  Vector2f start_NE, end_NE;
250  if (!start.get_vector_xy_from_origin_NE(start_NE) || !end.get_vector_xy_from_origin_NE(end_NE)) {
251  return false;
252  }
253 
254  // check each obstacle's distance from segment
255  bool valid_loc_found = false;
256  float smallest_margin = FLT_MAX;
257  uint32_t now = AP_HAL::millis();
258  for (uint16_t i=0; i<_prx_loc_count; i++) {
259 
260  // check time of objects is still valid
261  if ((now - _prx_locs[i].last_update_ms) > PROXIMITY_LOCATION_TIMEOUT_MS) {
262  continue;
263  }
264  valid_loc_found = true;
265 
266  // convert obstacle's location to offset (in cm) from EKF origin
267  Vector2f point;
268  if (!_prx_locs[i].loc.get_vector_xy_from_origin_NE(point)) {
269  continue;
270  }
271 
272  // margin is distance between line segment and obstacle minus obstacle's radius
273  const float m = Vector2f::closest_distance_between_line_and_point(start_NE, end_NE, point) * 0.01f - _prx_locs[i].radius_m;
274  if (m < smallest_margin) {
275  smallest_margin = m;
276  }
277  }
278 
279  // if no valid obstacles found then they've all expired so clear buffer
280  if (!valid_loc_found) {
281  _prx_loc_count = 0;
282  }
283 
284  // return smallest margin
285  if (smallest_margin < FLT_MAX) {
286  margin = smallest_margin;
287  return true;
288  }
289 
290  return false;
291 }
