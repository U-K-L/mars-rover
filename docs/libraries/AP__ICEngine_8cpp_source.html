128 /*
129  update engine state
130  */
131 void AP_ICEngine::update(void)
132 {
133  if (!enable) {
134  return;
135  }
136 
137  uint16_t cvalue = 1500;
138  RC_Channel *c = rc().channel(start_chan-1);
139  if (c != nullptr) {
140  // get starter control channel
141  cvalue = c->get_radio_in();
142  }
143 
144  bool should_run = false;
145  uint32_t now = AP_HAL::millis();
146 
147  if (state == ICE_OFF && cvalue >= 1700) {
148  should_run = true;
149  } else if (cvalue <= 1300) {
150  should_run = false;
151  } else if (state != ICE_OFF) {
152  should_run = true;
153  }
154 
155  // switch on current state to work out new state
156  switch (state) {
157  case ICE_OFF:
158  if (should_run) {
159  state = ICE_START_DELAY;
160  }
161  break;
162 
163  case ICE_START_HEIGHT_DELAY: {
164  Vector3f pos;
165  if (!should_run) {
166  state = ICE_OFF;
167  } else if (AP::ahrs().get_relative_position_NED_origin(pos)) {
168  if (height_pending) {
169  height_pending = false;
170  initial_height = -pos.z;
171  } else if ((-pos.z) >= initial_height + height_required) {
172  gcs().send_text(MAV_SEVERITY_INFO, "Starting height reached %.1f",
173  (double)(-pos.z - initial_height));
174  state = ICE_STARTING;
175  }
176  }
177  break;
178  }
179 
180  case ICE_START_DELAY:
181  if (!should_run) {
182  state = ICE_OFF;
183  } else if (now - starter_last_run_ms >= starter_delay*1000) {
184  gcs().send_text(MAV_SEVERITY_INFO, "Starting engine");
185  state = ICE_STARTING;
186  }
187  break;
188 
189  case ICE_STARTING:
190  if (!should_run) {
191  state = ICE_OFF;
192  } else if (now - starter_start_time_ms >= starter_time*1000) {
193  state = ICE_RUNNING;
194  }
195  break;
196 
197  case ICE_RUNNING:
198  if (!should_run) {
199  state = ICE_OFF;
200  gcs().send_text(MAV_SEVERITY_INFO, "Stopped engine");
201  } else if (rpm_instance > 0) {
202  // check RPM to see if still running
203  if (!rpm.healthy(rpm_instance-1) ||
204  rpm.get_rpm(rpm_instance-1) < rpm_threshold) {
205  // engine has stopped when it should be running
206  state = ICE_START_DELAY;
207  }
208  }
209  break;
210  }
211 
212  if (!hal.util->get_soft_armed()) {
213  if (state == ICE_START_HEIGHT_DELAY) {
214  // when disarmed we can be waiting for takeoff
215  Vector3f pos;
216  if (AP::ahrs().get_relative_position_NED_origin(pos)) {
217  // reset initial height while disarmed
218  initial_height = -pos.z;
219  }
220  } else if (idle_percent <= 0) { // check if we should idle
221  // force ignition off when disarmed
222  state = ICE_OFF;
223  }
224  }
225 
226  /* now set output channels */
227  switch (state) {
228  case ICE_OFF:
229  SRV_Channels::set_output_pwm(SRV_Channel::k_ignition, pwm_ignition_off);
230  SRV_Channels::set_output_pwm(SRV_Channel::k_starter, pwm_starter_off);
231  starter_start_time_ms = 0;
232  break;
233 
234  case ICE_START_HEIGHT_DELAY:
235  case ICE_START_DELAY:
236  SRV_Channels::set_output_pwm(SRV_Channel::k_ignition, pwm_ignition_on);
237  SRV_Channels::set_output_pwm(SRV_Channel::k_starter, pwm_starter_off);
238  break;
239 
240  case ICE_STARTING:
241  SRV_Channels::set_output_pwm(SRV_Channel::k_ignition, pwm_ignition_on);
242  SRV_Channels::set_output_pwm(SRV_Channel::k_starter, pwm_starter_on);
243  if (starter_start_time_ms == 0) {
244  starter_start_time_ms = now;
245  }
246  starter_last_run_ms = now;
247  break;
248 
249  case ICE_RUNNING:
250  SRV_Channels::set_output_pwm(SRV_Channel::k_ignition, pwm_ignition_on);
251  SRV_Channels::set_output_pwm(SRV_Channel::k_starter, pwm_starter_off);
252  starter_start_time_ms = 0;
253  break;
254  }
255 }

258 /*
259  check for throttle override. This allows the ICE controller to force
260  the correct starting throttle when starting the engine and maintain idle when disarmed
261  */
262 bool AP_ICEngine::throttle_override(uint8_t &percentage)
263 {
264  if (!enable) {
265  return false;
266  }
267 
268  if (state == ICE_RUNNING &&
269  idle_percent > 0 &&
270  idle_percent < 100 &&
271  (int16_t)idle_percent > SRV_Channels::get_output_scaled(SRV_Channel::k_throttle))
272  {
273  percentage = (uint8_t)idle_percent;
274  return true;
275  }
276 
277  if (state == ICE_STARTING || state == ICE_START_DELAY) {
278  percentage = (uint8_t)start_percent.get();
279  return true;
280  }
281  return false;
282 }

285 /*
286  handle DO_ENGINE_CONTROL messages via MAVLink or mission
287 */
288 bool AP_ICEngine::engine_control(float start_control, float cold_start, float height_delay)
289 {
290  if (start_control <= 0) {
291  state = ICE_OFF;
292  return true;
293  }
294  RC_Channel *c = rc().channel(start_chan-1);
295  if (c != nullptr) {
296  // get starter control channel
297  if (c->get_radio_in() <= 1300) {
298  gcs().send_text(MAV_SEVERITY_INFO, "Engine: start control disabled");
299  return false;
300  }
301  }
302  if (height_delay > 0) {
303  height_pending = true;
304  initial_height = 0;
305  height_required = height_delay;
306  state = ICE_START_HEIGHT_DELAY;
307  gcs().send_text(MAV_SEVERITY_INFO, "Takeoff height set to %.1fm", (double)height_delay);
308  return true;
309  }
310  state = ICE_STARTING;
311  return true;
312 }
