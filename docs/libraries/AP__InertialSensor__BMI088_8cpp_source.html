111 /*
112  read from accelerometer registers, special SPI handling needed
113 */
114 bool AP_InertialSensor_BMI088::read_accel_registers(uint8_t reg, uint8_t *data, uint8_t len)
115 {
116  // when on I2C we just read normally
117  if (dev_accel->bus_type() != AP_HAL::Device::BUS_TYPE_SPI) {
118  return dev_accel->read_registers(reg, data, len);
119  }
120  // for SPI we need to discard the first returned byte. See
121  // datasheet for explanation
122  uint8_t b[len+2];
123  b[0] = reg | 0x80;
124  memset(&b[1], 0, len+1);
125  if (!dev_accel->transfer(b, len+2, b, len+2)) {
126  return false;
127  }
128  memcpy(data, &b[2], len);
129  return true;
130 }

132 /*
133  write to accel registers with retries. The SPI sensor may take
134  several tries to correctly write a register
135 */
136 bool AP_InertialSensor_BMI088::write_accel_register(uint8_t reg, uint8_t v)
137 {
138  for (uint8_t i=0; i<8; i++) {
139  dev_accel->write_register(reg, v);
140  uint8_t v2 = 0;
141  if (read_accel_registers(reg, &v2, 1) && v2 == v) {
142  return true;
143  }
144  }
145  return false;
146 }

163 bool AP_InertialSensor_BMI088::setup_accel_config(void)
164 {
165  if (done_accel_config) {
166  return true;
167  }
168  accel_config_count++;
169  for (uint8_t i=0; i<ARRAY_SIZE(accel_config); i++) {
170  uint8_t v;
171  if (!read_accel_registers(accel_config[i].reg, &v, 1)) {
172  return false;
173  }
174  if (v == accel_config[i].value) {
175  continue;
176  }
177  if (!write_accel_register(accel_config[i].reg, accel_config[i].value)) {
178  return false;
179  }
180  }
181  done_accel_config = true;
182  hal.console->printf("BMI088: accel config OK (%u tries)\n", (unsigned)accel_config_count);
183  return true;
184 }

211 /*
212  probe and initialise gyro
213  */
214 bool AP_InertialSensor_BMI088::gyro_init()
215 {
216  WITH_SEMAPHORE(dev_gyro->get_semaphore());
217 
218  uint8_t v;
219  if (!dev_gyro->read_registers(REGG_CHIPID, &v, 1) || v != 0x0F) {
220  return false;
221  }
222 
223  if (!dev_gyro->write_register(REGG_BGW_SOFTRESET, 0xB6)) {
224  return false;
225  }
226  hal.scheduler->delay(10);
227 
228  dev_gyro->setup_checked_registers(5, 20);
229 
230  // setup 2000dps range
231  if (!dev_gyro->write_register(REGG_RANGE, 0x00, true)) {
232  return false;
233  }
234 
235  // setup filter bandwidth 230Hz, no decimation
236  if (!dev_gyro->write_register(REGG_BW, 0x81, true)) {
237  return false;
238  }
239 
240  // disable low-power mode
241  if (!dev_gyro->write_register(REGG_LPM1, 0, true)) {
242  return false;
243  }
244 
245  // setup for filtered data
246  if (!dev_gyro->write_register(REGG_RATE_HBW, 0x00, true)) {
247  return false;
248  }
249 
250  // setup FIFO for streaming X,Y,Z
251  if (!dev_gyro->write_register(REGG_FIFO_CONFIG_1, 0x80, true)) {
252  return false;
253  }
254 
255  hal.console->printf("BMI088: found gyro\n");
256 
257  return true;
258 }

268 /*
269  read accel fifo
270  */
271 void AP_InertialSensor_BMI088::read_fifo_accel(void)
272 {
273  if (!setup_accel_config()) {
274  return;
275  }
276  uint8_t len[2];
277  if (!read_accel_registers(REGA_FIFO_LEN0, len, 2)) {
278  _inc_accel_error_count(accel_instance);
279  return;
280  }
281  uint16_t fifo_length = len[0] + (len[1]<<8);
282  if (fifo_length & 0x8000) {
283  // empty
284  return;
285  }
286 
287  // don't read more than 8 frames at a time
288  if (fifo_length > 8*7) {
289  fifo_length = 8*7;
290  }
291  if (fifo_length == 0) {
292  return;
293  }
294 
295  uint8_t data[fifo_length];
296  if (!read_accel_registers(REGA_FIFO_DATA, data, fifo_length)) {
297  _inc_accel_error_count(accel_instance);
298  return;
299  }
300  // assume configured for 24g range
301  const float scale = (1.0/32768.0) * GRAVITY_MSS * 24.0;
302  const uint8_t *p = &data[0];
303  while (fifo_length >= 7) {
304  /*
305  the fifo frames are variable length, with the frame type in the first byte
306  */
307  uint8_t frame_len = 2;
308  switch (p[0] & 0xFC) {
309  case 0x84: {
310  // accel frame
311  frame_len = 7;
312  const uint8_t *d = p+1;
313  int16_t xyz[3] {
314  int16_t(uint16_t(d[0] | (d[1]<<8))),
315  int16_t(uint16_t(d[2] | (d[3]<<8))),
316  int16_t(uint16_t(d[4] | (d[5]<<8)))};
317  Vector3f accel(xyz[0], xyz[1], xyz[2]);
318 
319  accel *= scale;
320 
321  _rotate_and_correct_accel(accel_instance, accel);
322  _notify_new_accel_raw_sample(accel_instance, accel);
323  break;
324  }
325  case 0x40:
326  // skip frame
327  frame_len = 2;
328  break;
329  case 0x44:
330  // sensortime frame
331  frame_len = 4;
332  break;
333  case 0x48:
334  // fifo config frame
335  frame_len = 2;
336  break;
337  case 0x50:
338  // sample drop frame
339  frame_len = 2;
340  break;
341  }
342  p += frame_len;
343  fifo_length -= frame_len;
344  }
345 
346  if (temperature_counter++ == 100) {
347  temperature_counter = 0;
348  uint8_t tbuf[2];
349  if (!read_accel_registers(REGA_TEMP_LSB, tbuf, 2)) {
350  _inc_accel_error_count(accel_instance);
351  } else {
352  uint16_t temp_uint11 = (tbuf[0]<<3) | (tbuf[1]>>5);
353  int16_t temp_int11 = temp_uint11>1023?temp_uint11-2048:temp_uint11;
354  float temp_degc = temp_int11 * 0.125f + 23;
355  _publish_temperature(accel_instance, temp_degc);
356  }
357  }
358 }

360 /*
361  read gyro fifo
362  */
363 void AP_InertialSensor_BMI088::read_fifo_gyro(void)
364 {
365  uint8_t num_frames;
366  if (!dev_gyro->read_registers(REGG_FIFO_STATUS, &num_frames, 1)) {
367  _inc_gyro_error_count(gyro_instance);
368  return;
369  }
370  num_frames &= 0x7F;
371 
372  // don't read more than 8 frames at a time
373  if (num_frames > 8) {
374  num_frames = 8;
375  }
376  if (num_frames == 0) {
377  return;
378  }
379  uint8_t data[6*num_frames];
380  if (!dev_gyro->read_registers(REGG_FIFO_DATA, data, num_frames*6)) {
381  _inc_gyro_error_count(gyro_instance);
382  return;
383  }
384 
385  // data is 16 bits with 2000dps range
386  const float scale = radians(2000.0f) / 32767.0f;
387  for (uint8_t i = 0; i < num_frames; i++) {
388  const uint8_t *d = &data[i*6];
389  int16_t xyz[3] {
390  int16_t(uint16_t(d[0] | d[1]<<8)),
391  int16_t(uint16_t(d[2] | d[3]<<8)),
392  int16_t(uint16_t(d[4] | d[5]<<8)) };
393  Vector3f gyro(xyz[0], xyz[1], xyz[2]);
394  gyro *= scale;
395 
396  _rotate_and_correct_gyro(gyro_instance, gyro);
397  _notify_new_gyro_raw_sample(gyro_instance, gyro);
398  }
399 
400  if (!dev_gyro->check_next_register()) {
401  _inc_gyro_error_count(gyro_instance);
402  }
403 }
