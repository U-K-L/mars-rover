77 /*
78  format a SBUS output frame into a 25 byte buffer
79  */
80 void AP_SBusOut::sbus_format_frame(uint16_t *channels, uint8_t num_channels, uint8_t buffer[SBUS_BSIZE])
81 {
82  uint8_t index = 1;
83  uint8_t offset = 0;
84 
85  memset(buffer, 0, SBUS_BSIZE);
86  buffer[0] = 0x0f;
87 
88  /* construct sbus frame representing channels 1 through 16 (max) */
89  uint8_t nchan = MIN(num_channels, SBUS_CHANNELS);
90  for (unsigned i = 0; i < nchan; ++i) {
91  /*protect from out of bounds values and limit to 11 bits*/
92  uint16_t pwmval = MAX(channels[i], SBUS_MIN);
93  uint16_t value = (uint16_t)((pwmval - SBUS_MIN) * SBUS_SCALE);
94  if (value > 0x07ff) {
95  value = 0x07ff;
96  }
97 
98 #if SBUS_DEBUG
99  static uint16_t lastch9 = 0;
100  if ((i==8) && (pwmval != lastch9)) {
101  lastch9 = pwmval;
102  hal.console->printf("channel 9 pwm: %04d\n", pwmval);
103  }
104 #endif
105 
106  while (offset >= 8) {
107  ++index;
108  offset -= 8;
109  }
110 
111  buffer[index] |= (value << (offset)) & 0xff;
112  buffer[index + 1] |= (value >> (8 - offset)) & 0xff;
113  buffer[index + 2] |= (value >> (16 - offset)) & 0xff;
114  offset += 11;
115  }
116 }

122 void
123 AP_SBusOut::update()
124 {
125  if (!initialised) {
126  initialised = true;
127  init();
128  }
129 
130  if (sbus1_uart == nullptr) {
131  return;
132  }
133 
134  // constrain output rate using sbus_frame_interval
135  static uint32_t last_micros = 0;
136  uint32_t now = AP_HAL::micros();
137  if ((now - last_micros) <= sbus_frame_interval) {
138  return;
139  }
140 
141  last_micros = now;
142 
143  /* construct sbus frame representing channels 1 through 16 (max) */
144  uint8_t nchan = MIN(NUM_SERVO_CHANNELS, SBUS_CHANNELS);
145  uint16_t channels[SBUS_CHANNELS] {};
146 
147  for (unsigned i = 0; i < nchan; ++i) {
148  SRV_Channel *c = SRV_Channels::srv_channel(i);
149  if (c == nullptr) {
150  continue;
151  }
152  channels[i] = c->get_output_pwm();
153  }
154  uint8_t buffer[SBUS_BSIZE];
155  sbus_format_frame(channels, nchan, buffer);
156 
157 #if SBUS_DEBUG
158  hal.gpio->pinMode(55, HAL_GPIO_OUTPUT);
159  hal.gpio->write(55, 1);
160 #endif
161 
162  sbus1_uart->write(buffer, sizeof(buffer));
163 
164 #if SBUS_DEBUG
165  hal.gpio->pinMode(55, HAL_GPIO_OUTPUT);
166  hal.gpio->write(55, 0);
167 #endif
168 }

170 void AP_SBusOut::init() {
171  uint16_t rate = sbus_rate.get();
172 
173 #if SBUS_DEBUG
174  hal.console->printf("AP_SBusOut: init %d Hz\n", rate);
175 #endif
176 
177  // subtract 500usec from requested frame interval to allow for latency
178  sbus_frame_interval = (1000UL * 1000UL) / rate - 500;
179  // at 100,000 bps, a 300 bit sbus frame takes 3msec to transfer
180  // require a minimum 700usec interframe gap
181  if (sbus_frame_interval < 3700) {
182  sbus_frame_interval = 3700;
183  }
184 
185  AP_SerialManager *serial_manager = AP_SerialManager::get_singleton();
186  if (!serial_manager) {
187  return;
188  }
189  sbus1_uart = serial_manager->find_serial(AP_SerialManager::SerialProtocol_Sbus1,0);
190 }
