86 uint16_t RCInput::read(uint8_t channel)
87 {
88  if (!_init || (channel >= MIN(RC_INPUT_MAX_CHANNELS, _num_channels))) {
89  return 0;
90  }
91  rcin_mutex.take(HAL_SEMAPHORE_BLOCK_FOREVER);
92  uint16_t v = _rc_values[channel];
93  rcin_mutex.give();
94 #if HAL_RCINPUT_WITH_AP_RADIO
95  if (radio && channel == 0) {
96  // hook to allow for update of radio on main thread, for mavlink sends
97  radio->update();
98  }
99 #endif
100  return v;
101 }

118 void RCInput::_timer_tick(void)
119 {
120  if (!_init) {
121  return;
122  }
123 
124 #if HAL_USE_ICU == TRUE
125  const uint32_t *p;
126  uint32_t n;
127  while ((p = (const uint32_t *)sig_reader.sigbuf.readptr(n)) != nullptr) {
128  rcin_prot.process_pulse_list(p, n*2, sig_reader.need_swap);
129  sig_reader.sigbuf.advance(n);
130  }
131 #endif
132 
133 #if HAL_USE_EICU == TRUE
134  uint32_t width_s0, width_s1;
135  while(sig_reader.read(width_s0, width_s1)) {
136  rcin_prot.process_pulse(width_s0, width_s1);
137  }
138 #endif
139 
140 #ifndef HAL_NO_UARTDRIVER
141  const char *rc_protocol = nullptr;
142 #endif
143 
144 #if HAL_USE_ICU == TRUE || HAL_USE_EICU == TRUE
145  if (rcin_prot.new_input()) {
146  rcin_mutex.take(HAL_SEMAPHORE_BLOCK_FOREVER);
147  _rcin_timestamp_last_signal = AP_HAL::micros();
148  _num_channels = rcin_prot.num_channels();
149  _num_channels = MIN(_num_channels, RC_INPUT_MAX_CHANNELS);
150  for (uint8_t i=0; i<_num_channels; i++) {
151  _rc_values[i] = rcin_prot.read(i);
152  }
153  rcin_mutex.give();
154 #ifndef HAL_NO_UARTDRIVER
155  rc_protocol = rcin_prot.protocol_name();
156 #endif
157  }
158 #endif
159 
160 #if HAL_RCINPUT_WITH_AP_RADIO
161  if (radio && radio->last_recv_us() != last_radio_us) {
162  last_radio_us = radio->last_recv_us();
163  rcin_mutex.take(HAL_SEMAPHORE_BLOCK_FOREVER);
164  _rcin_timestamp_last_signal = last_radio_us;
165  _num_channels = radio->num_channels();
166  _num_channels = MIN(_num_channels, RC_INPUT_MAX_CHANNELS);
167  for (uint8_t i=0; i<_num_channels; i++) {
168  _rc_values[i] = radio->read(i);
169  }
170  rcin_mutex.give();
171  }
172 #endif
173 
174 #if HAL_WITH_IO_MCU
175  rcin_mutex.take(HAL_SEMAPHORE_BLOCK_FOREVER);
176  if (AP_BoardConfig::io_enabled() &&
177  iomcu.check_rcinput(last_iomcu_us, _num_channels, _rc_values, RC_INPUT_MAX_CHANNELS)) {
178  _rcin_timestamp_last_signal = last_iomcu_us;
179 #ifndef HAL_NO_UARTDRIVER
180  rc_protocol = iomcu.get_rc_protocol();
181 #endif
182  }
183  rcin_mutex.give();
184 #endif
185 
186 #ifndef HAL_NO_UARTDRIVER
187  if (rc_protocol && rc_protocol != last_protocol) {
188  last_protocol = rc_protocol;
189  gcs().send_text(MAV_SEVERITY_DEBUG, "RCInput: decoding %s", last_protocol);
190  }
191 #endif
192 
193  // note, we rely on the vehicle code checking new_input()
194  // and a timeout for the last valid input to handle failsafe
195 }
