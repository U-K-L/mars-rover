28 uint16_t crc_crc4(uint16_t *data)
29 {
30  uint16_t n_rem = 0;
31  uint8_t n_bit;
32 
33  for (uint8_t cnt = 0; cnt < 16; cnt++) {
34  /* uneven bytes */
35  if (cnt & 1) {
36  n_rem ^= (uint8_t)((data[cnt >> 1]) & 0x00FF);
37  } else {
38  n_rem ^= (uint8_t)(data[cnt >> 1] >> 8);
39  }
40 
41  for (n_bit = 8; n_bit > 0; n_bit--) {
42  if (n_rem & 0x8000) {
43  n_rem = (n_rem << 1) ^ 0x3000;
44  } else {
45  n_rem = (n_rem << 1);
46  }
47  }
48  }
49 
50  return (n_rem >> 12) & 0xF;
51 }

81 uint8_t crc_crc8(const uint8_t *p, uint8_t len)
82 {
83  uint16_t crc = 0x0;
84 
85  while (len--) {
86  const uint16_t i = (crc ^ *p++) & 0xFF;
87  crc = (crc8_table[i] ^ (crc << 8)) & 0xFF;
88  }
89 
90  return crc & 0xFF;
91 }

97 uint16_t crc_xmodem_update(uint16_t crc, uint8_t data)
98 {
99  crc = crc ^ ((uint16_t)data << 8);
100  for (uint16_t i=0; i<8; i++)
101  {
102  if(crc & 0x8000) {
103  crc = (crc << 1) ^ 0x1021;
104  } else {
105  crc <<= 1;
106  }
107  }
108 
109  return crc;
110 }
