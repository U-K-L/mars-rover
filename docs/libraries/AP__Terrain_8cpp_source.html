76 /*
77  return terrain height in meters above average sea level (WGS84) for
78  a given position
79 
80  This is the base function that other height calculations are derived
81  from. The functions below are more convenient for most uses
82 
83  This function costs about 20 microseconds on Pixhawk
84  */
85 bool AP_Terrain::height_amsl(const Location &loc, float &height, bool corrected)
86 {
87  if (!allocate()) {
88  return false;
89  }
90 
91  const AP_AHRS &ahrs = AP::ahrs();
92 
93  // quick access for home altitude
94  if (loc.lat == home_loc.lat &&
95  loc.lng == home_loc.lng) {
96  height = home_height;
97  // apply correction which assumes home altitude is at terrain altitude
98  if (corrected) {
99  height += (ahrs.get_home().alt * 0.01f) - home_height;
100  }
101  return true;
102  }
103 
104  struct grid_info info;
105 
106  calculate_grid_info(loc, info);
107 
108  // find the grid
109  const struct grid_block &grid = find_grid_cache(info).grid;
110 
111  /*
112  note that we rely on the one square overlap to ensure these
113  calculations don't go past the end of the arrays
114  */
115  ASSERT_RANGE(info.idx_x, 0, TERRAIN_GRID_BLOCK_SIZE_X-2);
116  ASSERT_RANGE(info.idx_y, 0, TERRAIN_GRID_BLOCK_SIZE_Y-2);
117 
118 
119  // check we have all 4 required heights
120  if (!check_bitmap(grid, info.idx_x, info.idx_y) ||
121  !check_bitmap(grid, info.idx_x, info.idx_y+1) ||
122  !check_bitmap(grid, info.idx_x+1, info.idx_y) ||
123  !check_bitmap(grid, info.idx_x+1, info.idx_y+1)) {
124  return false;
125  }
126 
127  // hXY are the heights of the 4 surrounding grid points
128  int16_t h00, h01, h10, h11;
129 
130  h00 = grid.height[info.idx_x+0][info.idx_y+0];
131  h01 = grid.height[info.idx_x+0][info.idx_y+1];
132  h10 = grid.height[info.idx_x+1][info.idx_y+0];
133  h11 = grid.height[info.idx_x+1][info.idx_y+1];
134 
135  // do a simple dual linear interpolation. We could do something
136  // fancier, but it probably isn't worth it as long as the
137  // grid_spacing is kept small enough
138  float avg1 = (1.0f-info.frac_x) * h00 + info.frac_x * h10;
139  float avg2 = (1.0f-info.frac_x) * h01 + info.frac_x * h11;
140  float avg = (1.0f-info.frac_y) * avg1 + info.frac_y * avg2;
141 
142  height = avg;
143 
144  if (loc.lat == ahrs.get_home().lat &&
145  loc.lng == ahrs.get_home().lng) {
146  // remember home altitude as a special case
147  home_height = height;
148  home_loc = loc;
149  }
150 
151  // apply correction which assumes home altitude is at terrain altitude
152  if (corrected) {
153  height += (ahrs.get_home().alt * 0.01f) - home_height;
154  }
155 
156  return true;
157 }

160 /*
161  find difference between home terrain height and the terrain
162  height at the current location in meters. A positive result
163  means the terrain is higher than home.
164 
165  return false is terrain at the current location or at home
166  location is not available
167 
168  If extrapolate is true then allow return of an extrapolated
169  terrain altitude based on the last available data
170 */
171 bool AP_Terrain::height_terrain_difference_home(float &terrain_difference, bool extrapolate)
172 {
173  const AP_AHRS &ahrs = AP::ahrs();
174 
175  float height_home, height_loc;
176  if (!height_amsl(ahrs.get_home(), height_home, false)) {
177  // we don't know the height of home
178  return false;
179  }
180 
181  Location loc;
182  if (!ahrs.get_position(loc)) {
183  // we don't know where we are
184  return false;
185  }
186 
187  if (!height_amsl(loc, height_loc, false)) {
188  if (!extrapolate || !have_current_loc_height) {
189  // we don't know the height of the given location
190  return false;
191  }
192  // we don't have data at the current location, but the caller
193  // has asked for extrapolation, so use the last available
194  // terrain height. This can be used to fill in while new data
195  // is fetched. It should be very rarely used
196  height_loc = last_current_loc_height;
197  }
198 
199  terrain_difference = height_loc - height_home;
200 
201  return true;
202 }
203 
204 /*
205  return current height above terrain at current AHRS
206  position.
207 
208  If extrapolate is true then extrapolate from most recently
209  available terrain data is terrain data is not available for the
210  current location.
211 
212  Return true if height is available, otherwise false.
213 */
214 bool AP_Terrain::height_above_terrain(float &terrain_altitude, bool extrapolate)
215 {
216  float terrain_difference;
217  if (!height_terrain_difference_home(terrain_difference, extrapolate)) {
218  return false;
219  }
220 
221  float relative_home_altitude;
222  AP::ahrs().get_relative_position_D_home(relative_home_altitude);
223  relative_home_altitude = -relative_home_altitude;
224 
225  terrain_altitude = relative_home_altitude - terrain_difference;
226  return true;
227 }
228 
229 /*
230  return estimated equivalent relative-to-home altitude in meters
231  of a given height above the terrain at the current location
232 
233  This function allows existing height controllers which work on
234  barometric altitude (relative to home) to be used with terrain
235  based target altitude, by translating the "above terrain" altitude
236  into an equivalent barometric relative height.
237 
238  return false if terrain data is not available either at the given
239  location or at the home location.
240 
241  If extrapolate is true then allow return of an extrapolated
242  terrain altitude based on the last available data
243 */
244 bool AP_Terrain::height_relative_home_equivalent(float terrain_altitude,
245  float &relative_home_altitude,
246  bool extrapolate)
247 {
248  float terrain_difference;
249  if (!height_terrain_difference_home(terrain_difference, extrapolate)) {
250  return false;
251  }
252  relative_home_altitude = terrain_altitude + terrain_difference;
253  return true;
254 }

257 /*
258  calculate lookahead rise in terrain. This returns extra altitude
259  needed to clear upcoming terrain in meters
260 */
261 float AP_Terrain::lookahead(float bearing, float distance, float climb_ratio)
262 {
263  if (!allocate() || grid_spacing <= 0) {
264  return 0;
265  }
266 
267  Location loc;
268  if (!AP::ahrs().get_position(loc)) {
269  // we don't know where we are
270  return 0;
271  }
272  float base_height;
273  if (!height_amsl(loc, base_height, false)) {
274  // we don't know our current terrain height
275  return 0;
276  }
277 
278  float climb = 0;
279  float lookahead_estimate = 0;
280 
281  // check for terrain at grid spacing intervals
282  while (distance > 0) {
283  loc.offset_bearing(bearing, grid_spacing);
284  climb += climb_ratio * grid_spacing;
285  distance -= grid_spacing;
286  float height;
287  if (height_amsl(loc, height, false)) {
288  float rise = (height - base_height) - climb;
289  if (rise > lookahead_estimate) {
290  lookahead_estimate = rise;
291  }
292  }
293  }
294 
295  return lookahead_estimate;
296 }

299 /*
300  1hz update function. This is here to ensure progress is made on disk
301  IO even if no MAVLink send_request() operations are called for a
302  while.
303  */
304 void AP_Terrain::update(void)
305 {
306  // just schedule any needed disk IO
307  schedule_disk_io();
308 
309  const AP_AHRS &ahrs = AP::ahrs();
310 
311  // try to ensure the home location is populated
312  float height;
313  height_amsl(ahrs.get_home(), height, false);
314 
315  // update the cached current location height
316  Location loc;
317  bool pos_valid = ahrs.get_position(loc);
318  bool terrain_valid = pos_valid && height_amsl(loc, height, false);
319  if (pos_valid && terrain_valid) {
320  last_current_loc_height = height;
321  have_current_loc_height = true;
322  }
323 
324  // check for pending mission data
325  update_mission_data();
326 
327  // check for pending rally data
328  update_rally_data();
329 
330  // update capabilities and status
331  if (allocate()) {
332  if (!pos_valid) {
333  // we don't know where we are
334  system_status = TerrainStatusUnhealthy;
335  } else if (!terrain_valid) {
336  // we don't have terrain data at current location
337  system_status = TerrainStatusUnhealthy;
338  } else {
339  system_status = TerrainStatusOK;
340  }
341  } else {
342  system_status = TerrainStatusDisabled;
343  }
344 
345 }

380 /*
381  allocate terrain cache. Making this dynamically allocated allows
382  memory to be saved when terrain functionality is disabled
383  */
384 bool AP_Terrain::allocate(void)
385 {
386  if (enable == 0) {
387  return false;
388  }
389  if (cache != nullptr) {
390  return true;
391  }

393  if (cache == nullptr) {
394  enable.set(0);
395  gcs().send_text(MAV_SEVERITY_CRITICAL, "Terrain: Allocation failed");
396  return false;
