175 /*
176  start JSBSim child
177  */
178 bool JSBSim::start_JSBSim(void)
179 {
180  if (started_jsbsim) {
181  return true;
182  }
183  if (!open_fdm_socket()) {
184  return false;
185  }
186 
187  int p[2];
188  int devnull = open("/dev/null", O_RDWR|O_CLOEXEC);
189  if (pipe(p) != 0) {
190  AP_HAL::panic("Unable to create pipe");
191  }
192  pid_t child_pid = fork();
193  if (child_pid == 0) {
194  // in child
195  setsid();
196  dup2(devnull, 0);
197  dup2(p[1], 1);
198  close(p[0]);
199  for (uint8_t i=3; i<100; i++) {
200  close(i);
201  }
202  char *logdirective;
203  char *script;
204  char *nice;
205  char *rate;
206 
207  asprintf(&logdirective, "--logdirectivefile=%s", jsbsim_fgout);
208  asprintf(&script, "--script=%s", jsbsim_script);
209  asprintf(&nice, "--nice=%.8f", 10*1e-9);
210  asprintf(&rate, "--simulation-rate=%f", rate_hz);
211 
212  if (chdir(autotest_dir) != 0) {
213  perror(autotest_dir);
214  exit(1);
215  }
216 
217  int ret = execlp("JSBSim",
218  "JSBSim",
219  "--suspend",
220  rate,
221  nice,
222  logdirective,
223  script,
224  nullptr);
225  if (ret != 0) {
226  perror("JSBSim");
227  }
228  exit(1);
229  }
230  close(p[1]);
231  jsbsim_stdout = p[0];
232 
233  // read startup to be sure it is running
234  char c;
235  if (read(jsbsim_stdout, &c, 1) != 1) {
236  AP_HAL::panic("Unable to start JSBSim");
237  }
238 
239  if (!expect("JSBSim Execution beginning")) {
240  AP_HAL::panic("Failed to start JSBSim");
241  }
242  if (!open_control_socket()) {
243  AP_HAL::panic("Failed to open JSBSim control socket");
244  }
245 
246  fcntl(jsbsim_stdout, F_SETFL, fcntl(jsbsim_stdout, F_GETFL, 0) | O_NONBLOCK);
247 
248  started_jsbsim = true;
249  check_stdout();
250  close(devnull);
251  return true;
252 }

334 /*
335  decode and send servos
336 */
337 void JSBSim::send_servos(const struct sitl_input &input)
338 {
339  char *buf = nullptr;
340  float aileron = filtered_servo_angle(input, 0);
341  float elevator = filtered_servo_angle(input, 1);
342  float throttle = filtered_servo_range(input, 2);
343  float rudder = filtered_servo_angle(input, 3);
344  if (frame == FRAME_ELEVON) {
345  // fake an elevon plane
346  float ch1 = aileron;
347  float ch2 = elevator;
348  aileron = (ch2-ch1)/2.0f;
349  // the minus does away with the need for RC2_REV=-1
350  elevator = -(ch2+ch1)/2.0f;
351  } else if (frame == FRAME_VTAIL) {
352  // fake a vtail plane
353  float ch1 = elevator;
354  float ch2 = rudder;
355  // this matches VTAIL_OUTPUT==2
356  elevator = (ch2-ch1)/2.0f;
357  rudder = (ch2+ch1)/2.0f;
358  }
359  float wind_speed_fps = input.wind.speed / FEET_TO_METERS;
360  asprintf(&buf,
361  "set fcs/aileron-cmd-norm %f\n"
362  "set fcs/elevator-cmd-norm %f\n"
363  "set fcs/rudder-cmd-norm %f\n"
364  "set fcs/throttle-cmd-norm %f\n"
365  "set atmosphere/psiw-rad %f\n"
366  "set atmosphere/wind-mag-fps %f\n"
367  "set atmosphere/turbulence/milspec/windspeed_at_20ft_AGL-fps %f\n"
368  "set atmosphere/turbulence/milspec/severity %f\n"
369  "iterate 1\n",
370  aileron, elevator, rudder, throttle,
371  radians(input.wind.direction),
372  wind_speed_fps,
373  wind_speed_fps/3,
374  input.wind.turbulence);
375  ssize_t buflen = strlen(buf);
376  ssize_t sent = sock_control.send(buf, buflen);
377  free(buf);
378  if (sent < 0) {
379  if (errno != EAGAIN) {
380  fprintf(stderr, "Fatal: Failed to send on control socket: %s\n",
381  strerror(errno));
382  exit(1);
383  }
384  }
385  if (sent < buflen) {
386  fprintf(stderr, "Failed to send all bytes on control socket\n");
387  }
388 }

393 void FGNetFDM::ByteSwap(void)
394 {
395  uint32_t *buf = (uint32_t *)this;
396  for (uint16_t i=0; i<sizeof(*this)/4; i++) {

410 /*
411  receive an update from the FDM
412  This is a blocking function
413  */
414 void JSBSim::recv_fdm(const struct sitl_input &input)
415 {
416  FGNetFDM fdm;
417  check_stdout();
418 
419  do {
420  while (sock_fgfdm.recv(&fdm, sizeof(fdm), 100) != sizeof(fdm)) {
421  send_servos(input);
422  check_stdout();
423  }
424  fdm.ByteSwap();
425  } while (fdm.cur_time == time_now_us);
426 
427  accel_body = Vector3f(fdm.A_X_pilot, fdm.A_Y_pilot, fdm.A_Z_pilot) * FEET_TO_METERS;
428 
429  double p, q, r;
430  SITL::convert_body_frame(degrees(fdm.phi), degrees(fdm.theta),
431  degrees(fdm.phidot), degrees(fdm.thetadot), degrees(fdm.psidot),
432  &p, &q, &r);
433  gyro = Vector3f(p, q, r);
434 
435  velocity_ef = Vector3f(fdm.v_north, fdm.v_east, fdm.v_down) * FEET_TO_METERS;
436  location.lat = degrees(fdm.latitude) * 1.0e7;
437  location.lng = degrees(fdm.longitude) * 1.0e7;
438  location.alt = fdm.agl*100 + home.alt;
439  dcm.from_euler(fdm.phi, fdm.theta, fdm.psi);
440  airspeed = fdm.vcas * KNOTS_TO_METERS_PER_SECOND;
441  airspeed_pitot = airspeed;
442 
443  // update magnetic field
444  update_mag_field_bf();
445 
446  rpm1 = fdm.rpm[0];
447  rpm2 = fdm.rpm[1];
448 
449  time_now_us = fdm.cur_time;
450 }

461 /*
462  update the JSBSim simulation by one time step
463  */
464 void JSBSim::update(const struct sitl_input &input)
465 {
466  while (!initialised) {
467  if (!create_templates() ||
468  !start_JSBSim() ||
469  !open_control_socket() ||
470  !open_fdm_socket()) {
471  time_now_us = 1;
472  return;
473  }
474  initialised = true;
475  }
476  send_servos(input);
477  recv_fdm(input);
478  adjust_frame_time(rate_hz);
479  sync_frame_time();
480  drain_control_socket();
481 }
