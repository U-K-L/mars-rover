198 /*
199  addStuffing() from Robotis SDK. This pads the packet as required by the protocol
200 */
201 void AP_RobotisServo::add_stuffing(uint8_t *packet)
202 {
203  int packet_length_in = DXL_MAKEWORD(packet[PKT_LENGTH_L], packet[PKT_LENGTH_H]);
204  int packet_length_out = packet_length_in;
205 
206  if (packet_length_in < 8) {
207  // INSTRUCTION, ADDR_L, ADDR_H, CRC16_L, CRC16_H + FF FF FD
208  return;
209  }
210 
211  uint8_t *packet_ptr;
212  uint16_t packet_length_before_crc = packet_length_in - 2;
213  for (uint16_t i = 3; i < packet_length_before_crc; i++) {
214  packet_ptr = &packet[i+PKT_INSTRUCTION-2];
215  if (packet_ptr[0] == 0xFF && packet_ptr[1] == 0xFF && packet_ptr[2] == 0xFD) {
216  packet_length_out++;
217  }
218  }
219 
220  if (packet_length_in == packet_length_out) {
221  // no stuffing required
222  return;
223  }
224 
225  uint16_t out_index = packet_length_out + 6 - 2; // last index before crc
226  uint16_t in_index = packet_length_in + 6 - 2; // last index before crc
227 
228  while (out_index != in_index) {
229  if (packet[in_index] == 0xFD && packet[in_index-1] == 0xFF && packet[in_index-2] == 0xFF) {
230  packet[out_index--] = 0xFD; // byte stuffing
231  if (out_index != in_index) {
232  packet[out_index--] = packet[in_index--]; // FD
233  packet[out_index--] = packet[in_index--]; // FF
234  packet[out_index--] = packet[in_index--]; // FF
235  }
236  } else {
237  packet[out_index--] = packet[in_index--];
238  }
239  }
240 
241  packet[PKT_LENGTH_L] = DXL_LOBYTE(packet_length_out);
242  packet[PKT_LENGTH_H] = DXL_HIBYTE(packet_length_out);
243 }

245 /*
246  send a protocol 2.0 packet
247  */
248 void AP_RobotisServo::send_packet(uint8_t *txpacket)
249 {
250  add_stuffing(txpacket);
251 
252  // check max packet length
253  uint16_t total_packet_length = DXL_MAKEWORD(txpacket[PKT_LENGTH_L], txpacket[PKT_LENGTH_H]) + 7;
254 
255  // make packet header
256  txpacket[PKT_HEADER0] = 0xFF;
257  txpacket[PKT_HEADER1] = 0xFF;
258  txpacket[PKT_HEADER2] = 0xFD;
259  txpacket[PKT_RESERVED] = 0x00;
260 
261  // add CRC16
262  uint16_t crc = update_crc(0, txpacket, total_packet_length - 2); // 2: CRC16
263  txpacket[total_packet_length - 2] = DXL_LOBYTE(crc);
264  txpacket[total_packet_length - 1] = DXL_HIBYTE(crc);
265 
266  port->write(txpacket, total_packet_length);
267 
268  delay_time_us += total_packet_length * us_per_byte + us_gap;
269 }

312 void AP_RobotisServo::send_command(uint8_t id, uint16_t reg, uint32_t value, uint8_t len)
313 {
314  uint8_t txpacket[16] {};
315 
316  txpacket[PKT_ID] = id;
317  txpacket[PKT_LENGTH_L] = 5 + len;
318  txpacket[PKT_LENGTH_H] = 0;
319  txpacket[PKT_INSTRUCTION] = INST_WRITE;
320  txpacket[PKT_INSTRUCTION+1] = DXL_LOBYTE(reg);
321  txpacket[PKT_INSTRUCTION+2] = DXL_HIBYTE(reg);
322  memcpy(&txpacket[PKT_INSTRUCTION+3], &value, MIN(len,4));
323 
324  send_packet(txpacket);
325 }

327 /*
328  read response bytes
329  */
330 void AP_RobotisServo::read_bytes(void)
331 {
332  uint32_t n = port->available();
333  if (n == 0 && pktbuf_ofs < PKT_INSTRUCTION) {
334  return;
335  }
336 
337  if (n > sizeof(pktbuf) - pktbuf_ofs) {
338  n = sizeof(pktbuf) - pktbuf_ofs;
339  }
340  for (uint8_t i=0; i<n; i++) {
341  pktbuf[pktbuf_ofs++] = port->read();
342  }
343 
344  // discard bad leading data. This should be rare
345  while (pktbuf_ofs >= 4 &&
346  (pktbuf[0] != 0xFF || pktbuf[1] != 0xFF || pktbuf[2] != 0xFD || pktbuf[3] != 0x00)) {
347  memmove(pktbuf, &pktbuf[1], pktbuf_ofs-1);
348  pktbuf_ofs--;
349  }
350 
351  if (pktbuf_ofs < 10) {
352  // not enough data yet
353  return;
354  }
355 
356  const uint16_t total_packet_length = DXL_MAKEWORD(pktbuf[PKT_LENGTH_L], pktbuf[PKT_LENGTH_H]) + PKT_INSTRUCTION;
357  if (total_packet_length > sizeof(pktbuf)) {
358  pktbuf_ofs = 0;
359  return;
360  }
361  if (pktbuf_ofs < total_packet_length) {
362  // more data needed
363  return;
364  }
365 
366  // check CRC
367  const uint16_t crc = DXL_MAKEWORD(pktbuf[total_packet_length-2], pktbuf[total_packet_length-1]);
368  const uint16_t calc_crc = update_crc(0, pktbuf, total_packet_length - 2);
369  if (calc_crc != crc) {
370  memmove(pktbuf, &pktbuf[total_packet_length], pktbuf_ofs - total_packet_length);
371  pktbuf_ofs -= total_packet_length;
372  return;
373  }
374 
375  // process full packet
376  process_packet(pktbuf, total_packet_length);
377 
378  memmove(pktbuf, &pktbuf[total_packet_length], pktbuf_ofs - total_packet_length);
379  pktbuf_ofs -= total_packet_length;
380 }

382 /*
383  process a packet from a servo
384  */
385 void AP_RobotisServo::process_packet(const uint8_t *pkt, uint8_t length)
386 {
387  uint8_t id = pkt[PKT_ID];
388  if (id > 16 || id < 1) {
389  // discard packets from servos beyond max or min. Note that we
390  // don't allow servo 0, to make mapping to SERVOn_* parameters
391  // easier
392  return;
393  }
394  uint16_t id_mask = (1U<<(id-1));
395  if (!(id_mask & servo_mask)) {
396  // mark the servo as present
397  servo_mask |= id_mask;
398  hal.console->printf("Robotis: new servo %u\n", id);
399  }
400 }

403 void AP_RobotisServo::update()
404 {
405  if (!initialised) {
406  initialised = true;
407  init();
408  last_send_us = AP_HAL::micros();
409  return;
410  }
411 
412  if (port == nullptr) {
413  return;
414  }
415 
416  read_bytes();
417 
418  uint32_t now = AP_HAL::micros();
419  if (last_send_us != 0 && now - last_send_us < delay_time_us) {
420  // waiting for last send to complete
421  return;
422  }
423 
424  if (detection_count < DETECT_SERVO_COUNT) {
425  detection_count++;
426  detect_servos();
427  }
428 
429  if (servo_mask == 0) {
430  return;
431  }
432 
433  if (configured_servos < CONFIGURE_SERVO_COUNT) {
434  configured_servos++;
435  last_send_us = now;
436  configure_servos();
437  return;
438  }
439 
440  last_send_us = now;
441  delay_time_us = 0;
442 
443  // loop for all 16 channels
444  for (uint8_t i=0; i<NUM_SERVO_CHANNELS; i++) {
445  if (((1U<<i) & servo_mask) == 0) {
446  continue;
447  }
448  SRV_Channel *c = SRV_Channels::srv_channel(i);
449  if (c == nullptr) {
450  continue;
451  }
452  const uint16_t pwm = c->get_output_pwm();
453  const uint16_t min = c->get_output_min();
454  const uint16_t max = c->get_output_max();
455  float v = float(pwm - min) / (max - min);
456  uint32_t value = pos_min + v * (pos_max - pos_min);
457  send_command(i+1, REG_GOAL_POSITION, value, 4);
458  }
