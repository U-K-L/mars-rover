165 /*
166  return the center stick position expressed as a control_in value
167  used for thr_mid in copter
168  */
169 int16_t RC_Channel::get_control_mid() const
170 {
171  if (type_in == RC_CHANNEL_TYPE_RANGE) {
172  int16_t r_in = (radio_min.get() + radio_max.get())/2;
173 
174  if (reversed) {
175  r_in = radio_max.get() - (r_in - radio_min.get());
176  }
177 
178  int16_t radio_trim_low = radio_min + dead_zone;
179 
180  return (((int32_t)(high_in) * (int32_t)(r_in - radio_trim_low)) / (int32_t)(radio_max - radio_trim_low));
181  } else {
182  return 0;
183  }
184 }

186 /*
187  return an "angle in centidegrees" (normally -4500 to 4500) from
188  the current radio_in value using the specified dead_zone
189  */
190 int16_t
191 RC_Channel::pwm_to_angle_dz_trim(uint16_t _dead_zone, uint16_t _trim) const
192 {
193  int16_t radio_trim_high = _trim + _dead_zone;
194  int16_t radio_trim_low = _trim - _dead_zone;
195 
196  int16_t reverse_mul = (reversed?-1:1);
197  if (radio_in > radio_trim_high && radio_max != radio_trim_high) {
198  return reverse_mul * ((int32_t)high_in * (int32_t)(radio_in - radio_trim_high)) / (int32_t)(radio_max - radio_trim_high);
199  } else if (radio_in < radio_trim_low && radio_trim_low != radio_min) {
200  return reverse_mul * ((int32_t)high_in * (int32_t)(radio_in - radio_trim_low)) / (int32_t)(radio_trim_low - radio_min);
201  } else {
202  return 0;
203  }
204 }

269 float
270 RC_Channel::norm_input() const
271 {
272  float ret;
273  int16_t reverse_mul = (reversed?-1:1);
274  if (radio_in < radio_trim) {
275  if (radio_min >= radio_trim) {
276  return 0.0f;
277  }
278  ret = reverse_mul * (float)(radio_in - radio_trim) / (float)(radio_trim - radio_min);
279  } else {
280  if (radio_max <= radio_trim) {
281  return 0.0f;
282  }
283  ret = reverse_mul * (float)(radio_in - radio_trim) / (float)(radio_max - radio_trim);
284  }
285  return constrain_float(ret, -1.0f, 1.0f);
286 }

288 float
289 RC_Channel::norm_input_dz() const
290 {
291  int16_t dz_min = radio_trim - dead_zone;
292  int16_t dz_max = radio_trim + dead_zone;
293  float ret;
294  int16_t reverse_mul = (reversed?-1:1);
295  if (radio_in < dz_min && dz_min > radio_min) {
296  ret = reverse_mul * (float)(radio_in - dz_min) / (float)(dz_min - radio_min);
297  } else if (radio_in > dz_max && radio_max > dz_max) {
298  ret = reverse_mul * (float)(radio_in - dz_max) / (float)(radio_max - dz_max);
299  } else {
300  ret = 0;
301  }
302  return constrain_float(ret, -1.0f, 1.0f);
303 }

305 /*
306  get percentage input from 0 to 100. This ignores the trim value.
307  */
308 uint8_t
309 RC_Channel::percent_input() const
310 {
311  if (radio_in <= radio_min) {
312  return reversed?100:0;
313  }
314  if (radio_in >= radio_max) {
315  return reversed?0:100;
316  }
317  uint8_t ret = 100.0f * (radio_in - radio_min) / (float)(radio_max - radio_min);
318  if (reversed) {
319  ret = 100 - ret;
320  }
321  return ret;

359 /*
360  perform stick mixing on one channel
361  This type of stick mixing reduces the influence of the auto
362  controller as it increases the influence of the users stick input,
363  allowing the user full deflection if needed
364  */
365 int16_t RC_Channel::stick_mixing(const int16_t servo_in)
366 {
367  float ch_inf = (float)(radio_in - radio_trim);
368  ch_inf = fabsf(ch_inf);
369  ch_inf = MIN(ch_inf, 400.0f);
370  ch_inf = ((400.0f - ch_inf) / 400.0f);
371 
372  int16_t servo_out = servo_in;
373  servo_out *= ch_inf;
374  servo_out += control_in;
375 
376  return servo_out;
377 }

394 void RC_Channel::read_mode_switch()
395 {
396  // calculate position of flight mode switch
397  const uint16_t pulsewidth = get_radio_in();
398  if (pulsewidth <= 900 || pulsewidth >= 2200) {
399  return; // This is an error condition
400  }
401 
402  modeswitch_pos_t position;
403  if (pulsewidth < 1231) position = 0;
404  else if (pulsewidth < 1361) position = 1;
405  else if (pulsewidth < 1491) position = 2;
406  else if (pulsewidth < 1621) position = 3;
407  else if (pulsewidth < 1750) position = 4;
408  else position = 5;
409 
410  if (mode_switch_state.last_position == position) {
411  // nothing to do
412  return;
413  }
414 
415  const uint32_t tnow_ms = AP_HAL::millis();
416  if (position != mode_switch_state.debounced_position) {
417  mode_switch_state.debounced_position = position;
418  // store time that switch last moved
419  mode_switch_state.last_edge_time_ms = tnow_ms;
420  return;
421  }
422 
423  if (tnow_ms - mode_switch_state.last_edge_time_ms < MODE_SWITCH_DEBOUNCE_TIME_MS) {
424  // still in debounce
425  return;
426  }
427 
428  // set flight mode and simple mode setting
429  mode_switch_changed(position);
430 
431  // set the last switch position. This marks the
432  // transition as complete, even if the mode switch actually
433  // failed. This prevents the vehicle changing modes
434  // unexpectedly some time later.
435  mode_switch_state.last_position = position;
436 }

482 /*
483  read an aux channel. Return true if a switch has changed
484  */
485 bool RC_Channel::read_aux()
486 {
487  const aux_func_t _option = (aux_func_t)option.get();
488  if (_option == AUX_FUNC::DO_NOTHING) {
489  // may wish to add special cases for other "AUXSW" things
490  // here e.g. RCMAP_ROLL etc once they become options
491  return false;
492  }
493  aux_switch_pos_t new_position;
494  if (!read_3pos_switch(new_position)) {
495  return false;
496  }
497  const aux_switch_pos_t old_position = old_switch_position();
498  if (new_position == old_position) {
499  debounce.count = 0;
500  return false;
501  }
502  if (debounce.new_position != new_position) {
503  debounce.new_position = new_position;
504  debounce.count = 0;
505  }
506  // a value of 2 means we need 3 values in a row with the same
507  // value to activate
508  if (debounce.count++ < 2) {
509  return false;
510  }
511 
512  // debounced; undertake the action:
513  do_aux_function(_option, new_position);
514  set_old_switch_position(new_position);
515  return true;
516 }
