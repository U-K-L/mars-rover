37 // update the state of the sensor
38 void AP_Proximity_MAV::update(void)
39 {
40  // check for timeout and set health status
41  if ((_last_update_ms == 0 || (AP_HAL::millis() - _last_update_ms > PROXIMITY_MAV_TIMEOUT_MS)) &&
42  (_last_upward_update_ms == 0 || (AP_HAL::millis() - _last_upward_update_ms > PROXIMITY_MAV_TIMEOUT_MS))) {
43  set_status(AP_Proximity::Proximity_NoData);
44  } else {
45  set_status(AP_Proximity::Proximity_Good);
46  }
47 }

49 // get distance upwards in meters. returns true on success
50 bool AP_Proximity_MAV::get_upward_distance(float &distance) const

59 // handle mavlink DISTANCE_SENSOR messages
60 void AP_Proximity_MAV::handle_msg(mavlink_message_t *msg)
61 {
62  if (msg->msgid == MAVLINK_MSG_ID_DISTANCE_SENSOR) {
63  mavlink_distance_sensor_t packet;
64  mavlink_msg_distance_sensor_decode(msg, &packet);
65 
66  // store distance to appropriate sector based on orientation field
67  if (packet.orientation <= MAV_SENSOR_ROTATION_YAW_315) {
68  uint8_t sector = packet.orientation;
69  _angle[sector] = sector * 45;
70  _distance[sector] = packet.current_distance / 100.0f;
71  _distance_min = packet.min_distance / 100.0f;
72  _distance_max = packet.max_distance / 100.0f;
73  _distance_valid[sector] = (_distance[sector] >= _distance_min) && (_distance[sector] <= _distance_max);
74  _last_update_ms = AP_HAL::millis();
75  update_boundary_for_sector(sector);
76  }
77 
78  // store upward distance
79  if (packet.orientation == MAV_SENSOR_ROTATION_PITCH_90) {
80  _distance_upward = packet.current_distance / 100.0f;
81  _last_upward_update_ms = AP_HAL::millis();
82  }
83  return;
84  }
85 
86  if (msg->msgid == MAVLINK_MSG_ID_OBSTACLE_DISTANCE) {
87  mavlink_obstacle_distance_t packet;
88  mavlink_msg_obstacle_distance_decode(msg, &packet);
89 
90  // check increment (message's sector width)
91  float increment;
92  if (packet.increment_f > 0) {
93  // use increment float
94  increment = packet.increment_f;
95  } else if (packet.increment > 0) {
96  // use increment uint8_t
97  increment = packet.increment;
98  } else {
99  // invalid increment
100  return;
101  }
102 
103  const float MAX_DISTANCE = 9999.0f;
104  const float total_distances = MIN(360.0f / increment, MAVLINK_MSG_OBSTACLE_DISTANCE_FIELD_DISTANCES_LEN); // usually 72
105 
106  // set distance min and max
107  _distance_min = packet.min_distance / 100.0f;
108  _distance_max = packet.max_distance / 100.0f;
109  _last_update_ms = AP_HAL::millis();
110 
111  // get user configured yaw correction from front end
112  const float param_yaw_offset = constrain_float(frontend.get_yaw_correction(state.instance), -360.0f, +360.0f);
113  const float yaw_correction = wrap_360(param_yaw_offset + packet.angle_offset);
114  if (frontend.get_orientation(state.instance) != 0) {
115  increment *= -1;
116  }
117 
118  // initialise updated array and proximity sector angles (to closest object) and distances
119  bool sector_updated[_num_sectors];
120  float sector_width_half[_num_sectors];
121  for (uint8_t i = 0; i < _num_sectors; i++) {
122  sector_updated[i] = false;
123  sector_width_half[i] = _sector_width_deg[i] * 0.5f;
124  _angle[i] = _sector_middle_deg[i];
125  _distance[i] = MAX_DISTANCE;
126  }
127 
128  // iterate over message's sectors
129  for (uint8_t j = 0; j < total_distances; j++) {
130  const float packet_distance_m = packet.distances[j] * 0.01f;
131  const float mid_angle = wrap_360(j * increment + yaw_correction);
132 
133  // iterate over proximity sectors
134  for (uint8_t i = 0; i < _num_sectors; i++) {
135  float angle_diff = fabsf(wrap_180(_sector_middle_deg[i] - mid_angle));
136  // update distance array sector with shortest distance from message
137  if ((angle_diff <= sector_width_half[i]) && (packet_distance_m < _distance[i])) {
138  _distance[i] = packet_distance_m;
139  _angle[i] = mid_angle;
140  sector_updated[i] = true;
141  }
142  }
143  }
144 
145  // update proximity sectors validity and boundary point
146  for (uint8_t i = 0; i < _num_sectors; i++) {
147  _distance_valid[i] = (_distance[i] >= _distance_min) && (_distance[i] <= _distance_max);
148  if (sector_updated[i]) {
149  update_boundary_for_sector(i);
150  }
151  }
152  }
153 }
