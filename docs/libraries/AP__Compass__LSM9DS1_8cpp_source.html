137 void AP_Compass_LSM9DS1::_update(void)
138 {
139  struct sample_regs regs;
140  Vector3f raw_field;
141 
142  if (!_block_read(LSM9DS1M_STATUS_REG_M, (uint8_t *) &regs, sizeof(regs))) {
143  return;
144  }
145 
146  if (regs.status & 0x80) {
147  return;

149 
150  raw_field = Vector3f(regs.val[0], regs.val[1], regs.val[2]);

153  return;
154  }
155 
156  raw_field *= _scaling;
