90 void AP_RCProtocol_DSM::dsm_guess_format(bool reset, const uint8_t dsm_frame[16])
91 {
92  /* reset the 10/11 bit sniffed channel masks */
93  if (reset) {
94  cs10 = 0;
95  cs11 = 0;
96  samples = 0;
97  channel_shift = 0;
98  return;
99  }
100 
101  /* scan the channels in the current dsm_frame in both 10- and 11-bit mode */
102  for (unsigned i = 0; i < DSM_FRAME_CHANNELS; i++) {
103 
104  const uint8_t *dp = &dsm_frame[2 + (2 * i)];
105  uint16_t raw = (dp[0] << 8) | dp[1];
106  unsigned channel, value;
107 
108  /* if the channel decodes, remember the assigned number */
109  if (dsm_decode_channel(raw, 10, &channel, &value) && (channel < 31)) {
110  cs10 |= (1 << channel);
111  }
112 
113  if (dsm_decode_channel(raw, 11, &channel, &value) && (channel < 31)) {
114  cs11 |= (1 << channel);
115  }
116 
117  /* XXX if we cared, we could look for the phase bit here to decide 1 vs. 2-dsm_frame format */
118  }
119 
120  /* wait until we have seen plenty of frames - 5 should normally be enough */
121  if (samples++ < 5) {
122  return;
123  }
124 
125  /*
126  * Iterate the set of sensible sniffed channel sets and see whether
127  * decoding in 10 or 11-bit mode has yielded anything we recognize.
128  *
129  * XXX Note that due to what seem to be bugs in the DSM2 high-resolution
130  * stream, we may want to sniff for longer in some cases when we think we
131  * are talking to a DSM2 receiver in high-resolution mode (so that we can
132  * reject it, ideally).
133  * See e.g. http://git.openpilot.org/cru/OPReview-116 for a discussion
134  * of this issue.
135  */
136  static const uint32_t masks[] = {
137  0x3f, /* 6 channels (DX6) */
138  0x7f, /* 7 channels (DX7) */
139  0xff, /* 8 channels (DX8) */
140  0x1ff, /* 9 channels (DX9, etc.) */
141  0x3ff, /* 10 channels (DX10) */
142  0x1fff, /* 13 channels (DX10t) */
143  0x3fff /* 18 channels (DX10) */
144  };
145  unsigned votes10 = 0;
146  unsigned votes11 = 0;
147 
148  for (unsigned i = 0; i < sizeof(masks)/sizeof(masks[0]); i++) {
149 
150  if (cs10 == masks[i]) {
151  votes10++;
152  }
153 
154  if (cs11 == masks[i]) {
155  votes11++;
156  }
157  }
158 
159  if ((votes11 == 1) && (votes10 == 0)) {
160  channel_shift = 11;
161  debug("DSM: 11-bit format");
162  return;
163  }
164 
165  if ((votes10 == 1) && (votes11 == 0)) {
166  channel_shift = 10;
167  debug("DSM: 10-bit format");
168  return;
169  }
170 
171  /* call ourselves to reset our state ... we have to try again */
172  debug("DSM: format detect fail, 10: 0x%08x %d 11: 0x%08x %d", cs10, votes10, cs11, votes11);
173  dsm_guess_format(true, dsm_frame);
174 }

180 bool AP_RCProtocol_DSM::dsm_decode(uint32_t frame_time_ms, const uint8_t dsm_frame[16],
181  uint16_t *values, uint16_t *num_values, uint16_t max_values)
182 {
183  /*
184  * If we have lost signal for at least 200ms, reset the
185  * format guessing heuristic.
186  */
187  if (((frame_time_ms - last_frame_time_ms) > 200U) && (channel_shift != 0)) {
188  dsm_guess_format(true, dsm_frame);
189  }
190 
191  /* we have received something we think is a dsm_frame */
192  last_frame_time_ms = frame_time_ms;
193 
194  /* if we don't know the dsm_frame format, update the guessing state machine */
195  if (channel_shift == 0) {
196  dsm_guess_format(false, dsm_frame);
197  return false;
198  }
199 
200  /*
201  * The encoding of the first two bytes is uncertain, so we're
202  * going to ignore them for now.
203  *
204  * Each channel is a 16-bit unsigned value containing either a 10-
205  * or 11-bit channel value and a 4-bit channel number, shifted
206  * either 10 or 11 bits. The MSB may also be set to indicate the
207  * second dsm_frame in variants of the protocol where more than
208  * seven channels are being transmitted.
209  */
210 
211  for (unsigned i = 0; i < DSM_FRAME_CHANNELS; i++) {
212 
213  const uint8_t *dp = &dsm_frame[2 + (2 * i)];
214  uint16_t raw = (dp[0] << 8) | dp[1];
215  unsigned channel, value;
216 
217  if (!dsm_decode_channel(raw, channel_shift, &channel, &value)) {
218  continue;
219  }
220 
221  /* ignore channels out of range */
222  if (channel >= max_values) {
223  continue;
224  }
225 
226  /* update the decoded channel count */
227  if (channel >= *num_values) {
228  *num_values = channel + 1;
229  }
230 
231  /* convert 0-1024 / 0-2048 values to 1000-2000 ppm encoding. */
232  if (channel_shift == 10) {
233  value *= 2;
234  }
235 
236  /*
237  * Spektrum scaling is special. There are these basic considerations
238  *
239  * * Midpoint is 1520 us
240  * * 100% travel channels are +- 400 us
241  *
242  * We obey the original Spektrum scaling (so a default setup will scale from
243  * 1100 - 1900 us), but we do not obey the weird 1520 us center point
244  * and instead (correctly) center the center around 1500 us. This is in order
245  * to get something useful without requiring the user to calibrate on a digital
246  * link for no reason.
247  */
248 
249  /* scaled integer for decent accuracy while staying efficient */
250  value = ((((int)value - 1024) * 1000) / 1700) + 1500;
251 
252  /*
253  * Store the decoded channel into the R/C input buffer, taking into
254  * account the different ideas about channel assignement that we have.
255  *
256  * Specifically, the first four channels in rc_channel_data are roll, pitch, thrust, yaw,
257  * but the first four channels from the DSM receiver are thrust, roll, pitch, yaw.
258  */
259  switch (channel) {
260  case 0:
261  channel = 2;
262  break;
263 
264  case 1:
265  channel = 0;
266  break;
267 
268  case 2:
269  channel = 1;
270 
271  default:
272  break;
273  }
274 
275  values[channel] = value;
276  }
277 
278  /*
279  * Spektrum likes to send junk in higher channel numbers to fill
280  * their packets. We don't know about a 13 channel model in their TX
281  * lines, so if we get a channel count of 13, we'll return 12 (the last
282  * data index that is stable).
283  */
284  if (*num_values == 13) {
285  *num_values = 12;
286  }
287 
288 #if 0
289  if (channel_shift == 11) {
290  /* Set the 11-bit data indicator */
291  *num_values |= 0x8000;
292  }
293 #endif
294 
295  /*
296  * XXX Note that we may be in failsafe here; we need to work out how to detect that.
297  */
298  return true;
299 }

314 void AP_RCProtocol_DSM::update(void)
315 {
316 #if defined(HAL_GPIO_SPEKTRUM_PWR) && defined(HAL_GPIO_SPEKTRUM_RC)
317  switch (bind_state) {
318  case BIND_STATE_NONE:
319  break;
320 
321  case BIND_STATE1:
322  hal.gpio->write(HAL_GPIO_SPEKTRUM_PWR, !HAL_SPEKTRUM_PWR_ENABLED);
323  hal.gpio->pinMode(HAL_GPIO_SPEKTRUM_RC, 1);
324  hal.gpio->write(HAL_GPIO_SPEKTRUM_RC, 1);
325  bind_last_ms = AP_HAL::millis();
326  bind_state = BIND_STATE2;
327  break;
328 
329  case BIND_STATE2: {
330  uint32_t now = AP_HAL::millis();
331  if (now - bind_last_ms > 500) {
332  hal.gpio->write(HAL_GPIO_SPEKTRUM_PWR, HAL_SPEKTRUM_PWR_ENABLED);
333  bind_last_ms = now;
334  bind_state = BIND_STATE3;
335  }
336  break;
337  }
338 
339  case BIND_STATE3: {
340  uint32_t now = AP_HAL::millis();
341  if (now - bind_last_ms > 72) {
342  // 9 pulses works with all satellite receivers, and supports the highest
343  // available protocol
344  const uint8_t num_pulses = 9;
345  for (uint8_t i=0; i<num_pulses; i++) {
346  hal.scheduler->delay_microseconds(120);
347  hal.gpio->write(HAL_GPIO_SPEKTRUM_RC, 0);
348  hal.scheduler->delay_microseconds(120);
349  hal.gpio->write(HAL_GPIO_SPEKTRUM_RC, 1);
350  }
351  bind_last_ms = now;
352  bind_state = BIND_STATE4;
353  }
354  break;
355  }
356 
357  case BIND_STATE4: {
358  uint32_t now = AP_HAL::millis();
359  if (now - bind_last_ms > 50) {
360  hal.gpio->pinMode(HAL_GPIO_SPEKTRUM_RC, 0);
361  bind_state = BIND_STATE_NONE;
362  }
363  break;
364  }
365  }
366 #endif
367 }

372 bool AP_RCProtocol_DSM::dsm_parse_byte(uint32_t frame_time_ms, uint8_t b, uint16_t *values,
373  uint16_t *num_values, uint16_t max_channels)
374 {
375  /* this is set by the decoding state machine and will default to false
376  * once everything that was decodable has been decoded.
377  */
378  bool decode_ret = false;
379 
380  /* overflow check */
381  if (byte_input.ofs == sizeof(byte_input.buf) / sizeof(byte_input.buf[0])) {
382  byte_input.ofs = 0;
383  dsm_decode_state = DSM_DECODE_STATE_DESYNC;
384  debug("DSM: RESET (BUF LIM)\n");
385  }
386 
387  if (byte_input.ofs == DSM_FRAME_SIZE) {
388  byte_input.ofs = 0;
389  dsm_decode_state = DSM_DECODE_STATE_DESYNC;
390  debug("DSM: RESET (PACKET LIM)\n");
391  }
392 
393 #ifdef DSM_DEBUG
394  debug("dsm state: %s%s, count: %d, val: %02x\n",
395  (dsm_decode_state == DSM_DECODE_STATE_DESYNC) ? "DSM_DECODE_STATE_DESYNC" : "",
396  (dsm_decode_state == DSM_DECODE_STATE_SYNC) ? "DSM_DECODE_STATE_SYNC" : "",
397  byte_input.ofs,
398  (unsigned)b);
399 #endif
400 
401  switch (dsm_decode_state) {
402  case DSM_DECODE_STATE_DESYNC:
403 
404  /* we are de-synced and only interested in the frame marker */
405  if ((frame_time_ms - last_rx_time_ms) >= 5) {
406  dsm_decode_state = DSM_DECODE_STATE_SYNC;
407  byte_input.ofs = 0;
408  byte_input.buf[byte_input.ofs++] = b;
409  }
410  break;
411 
412  case DSM_DECODE_STATE_SYNC: {
413  if ((frame_time_ms - last_rx_time_ms) >= 5 && byte_input.ofs > 0) {
414  byte_input.ofs = 0;
415  dsm_decode_state = DSM_DECODE_STATE_DESYNC;
416  break;
417  }
418  byte_input.buf[byte_input.ofs++] = b;
419 
420  /* decode whatever we got and expect */
421  if (byte_input.ofs < DSM_FRAME_SIZE) {
422  break;
423  }
424 
425  /*
426  * Great, it looks like we might have a frame. Go ahead and
427  * decode it.
428  */
429  decode_ret = dsm_decode(frame_time_ms, byte_input.buf, values, &chan_count, max_channels);
430 
431  /* we consumed the partial frame, reset */
432  byte_input.ofs = 0;
433 
434  /* if decoding failed, set proto to desync */
435  if (decode_ret == false) {
436  dsm_decode_state = DSM_DECODE_STATE_DESYNC;
437  }
438  break;
439  }
440 
441  default:
442  debug("UNKNOWN PROTO STATE");
443  decode_ret = false;
444  }
445 
446 
447  if (decode_ret) {
448  *num_values = chan_count;
449  }
450 
451  last_rx_time_ms = frame_time_ms;
452 
453  /* return false as default */
454  return decode_ret;
455 }
