62 /* Round up to the specified alignment.
63  *
64  * Let u be the address p rounded up to the alignment a. Then:
65  * u = p + a - 1 - r, where r = (p + a - 1) % a
66  *
67  * If p % a = 0, i.e. if p is already aligned, then:
68  * r = a - 1 ==> u = p
69  *
70  * Otherwise:
71  * r = p % a -1 ==> u = p + a - p % a
72  *
73  * p can be written p = q + p % a, where q is rounded down to the
74  * alignment a. Then u = q + a.
75  */
76 static inline void *align_to(void *p, size_t align)
77 {
78  return (void *)(((uintptr_t)p + align - 1) & ~(align - 1));
79 }

81 void Thread::_poison_stack()
82 {
83  pthread_attr_t attr;
84  size_t stack_size, guard_size;
85  void *stackp;
86  uint32_t *p, *curr, *begin, *end;
87 
88  if (pthread_getattr_np(_ctx, &attr) != 0 ||
89  pthread_attr_getstack(&attr, &stackp, &stack_size) != 0 ||
90  pthread_attr_getguardsize(&attr, &guard_size) != 0) {
91  return;
92  }
93 
94  stack_size /= sizeof(uint32_t);
95  guard_size /= sizeof(uint32_t);
96 
97  /* The stack either grows upward or downard. The guard part always
98  * protects the end */
99  end = (uint32_t *)stackp;
100  begin = end + stack_size;
101  curr = (uint32_t *)align_to(alloca(sizeof(uint32_t)), alignof(uint32_t));
102 
103  /* if curr is closer to @end, the stack actually grows from low to high
104  * virtual address: this is because this function should be executing very
105  * early in the thread's life and close to the thread creation, assuming
106  * the actual stack size is greater than the guard size and the stack
107  * until now is resonably small */
108  if (abs(curr - begin) > abs(curr - end)) {
109  std::swap(end, begin);
110  end -= guard_size;
111 
112  for (p = end; p > curr; p--) {
113  *p = STACK_POISON;
114  }
115  } else {
116  end += guard_size;
117 
118  /* we aligned curr to the up boundary, make sure this didn't cause us
119  * to lose some bytes */
120  curr--;
121 
122  for (p = end; p < curr; p++) {
123  *p = STACK_POISON;
124  }
125  }
126 
127  _stack_debug.start = begin;
128  _stack_debug.end = end;
129 }

131 size_t Thread::get_stack_usage()
132 {
133  uint32_t *p;
134  size_t result = 0;
135 
136  /* Make sure we are tracking usage for this thread */
137  if (_stack_debug.start == 0 || _stack_debug.end == 0) {
138  return 0;
139  }
140 
141  if (_stack_debug.start < _stack_debug.end) {
142  for (p = _stack_debug.end; p > _stack_debug.start; p--) {
143  if (*p != STACK_POISON) {
144  break;
145  }
146  }
147  result = p - _stack_debug.start;
148  } else {
149  for (p = _stack_debug.end; p < _stack_debug.start; p++) {
150  if (*p != STACK_POISON) {
151  break;
152  }
153  }
154  result = _stack_debug.start - p;
155  }
156 
157  return result;
158 }

160 bool Thread::start(const char *name, int policy, int prio)
161 {
162  if (_started) {
163  return false;
164  }
165 
166  struct sched_param param = { .sched_priority = prio };
167  pthread_attr_t attr;
168  int r;
169 
170  pthread_attr_init(&attr);
171 
172  /*
173  we need to run as root to get realtime scheduling. Allow it to
174  run as non-root for debugging purposes, plus to allow the Replay
175  tool to run
176  */
177  if (geteuid() == 0) {
178  if ((r = pthread_attr_setinheritsched(&attr, PTHREAD_EXPLICIT_SCHED)) != 0 ||
179  (r = pthread_attr_setschedpolicy(&attr, policy)) != 0 ||
180  (r = pthread_attr_setschedparam(&attr, &param)) != 0) {
181  AP_HAL::panic("Failed to set attributes for thread '%s': %s",
182  name, strerror(r));
183  }
184  }
185 
186  if (_stack_size) {
187  if (pthread_attr_setstacksize(&attr, _stack_size) != 0) {
188  return false;
189  }
190  }
191 
192  r = pthread_create(&_ctx, &attr, &Thread::_run_trampoline, this);
193  if (r != 0) {
194  AP_HAL::panic("Failed to create thread '%s': %s",
195  name, strerror(r));
196  }
197  pthread_attr_destroy(&attr);
198 
199  if (name) {
200  pthread_setname_np(_ctx, name);
201  }
202 
203  _started = true;
204 
205  return true;
206 }
