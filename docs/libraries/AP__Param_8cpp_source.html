131 /* the 'group_id' of a element of a group is the 18 bit identifier
132  used to distinguish between this element of the group and other
133  elements of the same group. It is calculated using a bit shift per
134  level of nesting, so the first level of nesting gets 6 bits the 2nd
135  level gets the next 6 bits, and the 3rd level gets the last 6
136  bits. This limits groups to having at most 64 elements.
137 */
138 uint32_t AP_Param::group_id(const struct GroupInfo *grpinfo, uint32_t base, uint8_t i, uint8_t shift)
139 {
140  if (grpinfo[i].idx == 0 && shift != 0 && !(grpinfo[i].flags & AP_PARAM_NO_SHIFT)) {
141  /*
142  this is a special case for a bug in the original design. An
143  idx of 0 shifted by n bits is still zero, which makes it
144  indistinguishable from a different parameter. This can lead
145  to parameter loops. We use index 63 for that case.
146  */
147  return base + (63U<<shift);
148  }
149  return base + (grpinfo[i].idx<<shift);
150 }

167 // validate a group info table
168 bool AP_Param::check_group_info(const struct AP_Param::GroupInfo * group_info,
169  uint16_t * total_size,
170  uint8_t group_shift,
171  uint8_t prefix_length)
172 {
173  uint8_t type;
174  uint64_t used_mask = 0;
175  for (uint8_t i=0;
176  (type=group_info[i].type) != AP_PARAM_NONE;
177  i++) {
178  uint8_t idx = group_info[i].idx;
179  if (idx >= (1<<_group_level_shift)) {
180  Debug("idx too large (%u) in %s", idx, group_info[i].name);
181  return false;
182  }
183  if (group_shift != 0 && idx == 0) {
184  // great idx 0 as 63 for duplicates. See group_id()
185  idx = 63;
186  }
187  if (used_mask & (1ULL<<idx)) {
188  Debug("Duplicate group idx %u for %s", idx, group_info[i].name);
189  return false;
190  }
191  used_mask |= (1ULL<<idx);
192  if (type == AP_PARAM_GROUP) {
193  // a nested group
194  if (group_shift + _group_level_shift >= _group_bits) {
195  Debug("double group nesting in %s", group_info[i].name);
196  return false;
197  }
198  const struct GroupInfo *ginfo = get_group_info(group_info[i]);
199  if (ginfo == nullptr) {
200  continue;
201  }
202  if (!check_group_info(ginfo, total_size, group_shift + _group_level_shift, prefix_length + strlen(group_info[i].name))) {
203  return false;
204  }
205  continue;
206  }
207  uint8_t size = type_size((enum ap_var_type)type);
208  if (size == 0) {
209  Debug("invalid type in %s", group_info[i].name);
210  return false;
211  }
212  if (prefix_length + strlen(group_info[i].name) > 16) {
213  Debug("suffix is too long in %s", group_info[i].name);
214  return false;
215  }
216  (*total_size) += size + sizeof(struct Param_header);
217  }
218  return true;
219 }

256 // validate the _var_info[] table
257 bool AP_Param::check_var_info(void)
258 {
259  uint16_t total_size = sizeof(struct EEPROM_header);
260 
261  for (uint16_t i=0; i<_num_vars; i++) {
262  uint8_t type = _var_info[i].type;
263  uint16_t key = _var_info[i].key;
264  if (type == AP_PARAM_GROUP) {
265  if (i == 0) {
266  // first element can't be a group, for first() call
267  return false;
268  }
269  const struct GroupInfo *group_info = get_group_info(_var_info[i]);
270  if (group_info == nullptr) {
271  continue;
272  }
273  if (!check_group_info(group_info, &total_size, 0, strlen(_var_info[i].name))) {
274  return false;
275  }
276  } else {
277  uint8_t size = type_size((enum ap_var_type)type);
278  if (size == 0) {
279  // not a valid type - the top level list can't contain
280  // AP_PARAM_NONE
281  return false;
282  }
283  total_size += size + sizeof(struct Param_header);
284  }
285  if (duplicate_key(i, key)) {
286  return false;
287  }
288  if (type != AP_PARAM_GROUP && (_var_info[i].flags & AP_PARAM_FLAG_POINTER)) {
289  // only groups can be pointers
290  return false;
291  }
292  }
293 
294  // we no longer check if total_size is larger than _eeprom_size,
295  // as we allow for more variables than could fit, relying on not
296  // saving default values
297 
298  return true;
299 }

328  adjust offset of a group element for nested groups and group pointers
329 
330  The new_offset variable is relative to the vindex base. This makes
331  dealing with pointer groups tricky
332  */
333 bool AP_Param::adjust_group_offset(uint16_t vindex, const struct GroupInfo &group_info, ptrdiff_t &new_offset)
334 {
335  if (group_info.flags & AP_PARAM_FLAG_NESTED_OFFSET) {
336  new_offset += group_info.offset;
337  return true;
338  }
339  if (group_info.flags & AP_PARAM_FLAG_POINTER) {
340  // group_info.offset refers to a pointer
341  ptrdiff_t base;
342  if (!get_base(_var_info[vindex], base)) {
343  // the object is not allocated yet
344  return false;
345  }
346  void **p = (void **)(base + new_offset + group_info.offset);
347  if (*p == nullptr) {
348  // the object is not allocated yet
349  return false;
350  }
351  // calculate offset that is needed to take base object and adjust for this object
352  new_offset = ((ptrdiff_t)*p) - base;
353  }
354  return true;
355 }

371 // find the info structure given a header and a group_info table
372 // return the Info structure and a pointer to the variables storage
373 const struct AP_Param::Info *AP_Param::find_by_header_group(struct Param_header phdr, void **ptr,
374  uint16_t vindex,
375  const struct GroupInfo *group_info,
376  uint32_t group_base,
377  uint8_t group_shift,
378  ptrdiff_t group_offset)
379 {
380  uint8_t type;
381  for (uint8_t i=0;
382  (type=group_info[i].type) != AP_PARAM_NONE;
383  i++) {
384  if (type == AP_PARAM_GROUP) {
385  // a nested group
386  if (group_shift + _group_level_shift >= _group_bits) {
387  // too deeply nested - this should have been caught by
388  // setup() !
389  return nullptr;
390  }
391  const struct GroupInfo *ginfo = get_group_info(group_info[i]);
392  if (ginfo == nullptr) {
393  continue;
394  }
395  ptrdiff_t new_offset = group_offset;
396 
397  if (!adjust_group_offset(vindex, group_info[i], new_offset)) {
398  continue;
399  }
400 
401  const struct AP_Param::Info *ret = find_by_header_group(phdr, ptr, vindex, ginfo,
402  group_id(group_info, group_base, i, group_shift),
403  group_shift + _group_level_shift, new_offset);
404  if (ret != nullptr) {
405  return ret;
406  }
407  continue;
408  }
409  if (group_id(group_info, group_base, i, group_shift) == phdr.group_element && type == phdr.type) {
410  // found a group element
411  ptrdiff_t base;
412  if (!get_base(_var_info[vindex], base)) {
413  continue;
414  }
415  *ptr = (void*)(base + group_info[i].offset + group_offset);
416  return &_var_info[vindex];
417  }
418  }
419  return nullptr;
420 }

454 // find the info structure for a variable in a group
455 const struct AP_Param::Info *AP_Param::find_var_info_group(const struct GroupInfo * group_info,
456  uint16_t vindex,
457  uint32_t group_base,
458  uint8_t group_shift,
459  ptrdiff_t group_offset,
460  uint32_t * group_element,
461  const struct GroupInfo * &group_ret,
462  struct GroupNesting &group_nesting,
463  uint8_t * idx) const
464 {
465  ptrdiff_t base;
466  if (!get_base(_var_info[vindex], base)) {
467  return nullptr;
468  }
469  uint8_t type;
470  for (uint8_t i=0;
471  (type=group_info[i].type) != AP_PARAM_NONE;
472  i++) {
473  ptrdiff_t ofs = group_info[i].offset + group_offset;
474  if (type == AP_PARAM_GROUP) {
475  const struct GroupInfo *ginfo = get_group_info(group_info[i]);
476  if (ginfo == nullptr) {
477  continue;
478  }
479  // a nested group
480  if (group_shift + _group_level_shift >= _group_bits) {
481  // too deeply nested - this should have been caught by
482  // setup() !
483  return nullptr;
484  }
485  const struct AP_Param::Info *info;
486  ptrdiff_t new_offset = group_offset;
487  if (!adjust_group_offset(vindex, group_info[i], new_offset)) {
488  continue;
489  }
490  if (group_nesting.level >= group_nesting.numlevels) {
491  return nullptr;
492  }
493  group_nesting.group_ret[group_nesting.level++] = &group_info[i];
494  info = find_var_info_group(ginfo, vindex,
495  group_id(group_info, group_base, i, group_shift),
496  group_shift + _group_level_shift,
497  new_offset,
498  group_element,
499  group_ret,
500  group_nesting,
501  idx);
502  if (info != nullptr) {
503  return info;
504  }
505  group_nesting.level--;
506  } else if ((ptrdiff_t) this == base + ofs) {
507  *group_element = group_id(group_info, group_base, i, group_shift);
508  group_ret = &group_info[i];
509  *idx = 0;
510  return &_var_info[vindex];
511  } else if (type == AP_PARAM_VECTOR3F &&
512  (base+ofs+(ptrdiff_t)sizeof(float) == (ptrdiff_t) this ||
513  base+ofs+2*(ptrdiff_t)sizeof(float) == (ptrdiff_t) this)) {
514  // we are inside a Vector3f. We need to work out which
515  // element of the vector the current object refers to.
516  *idx = (((ptrdiff_t) this) - (base+ofs))/sizeof(float);
517  *group_element = group_id(group_info, group_base, i, group_shift);
518  group_ret = &group_info[i];
519  return &_var_info[vindex];
520  }
521  }
522  return nullptr;
523 }

647 /*
648  return true if a header is the end of eeprom sentinal
649  */
650 bool AP_Param::is_sentinal(const Param_header &phdr)
651 {
652  // note that this is an ||, not an &&, as this makes us more
653  // robust to power off while adding a variable to EEPROM
654  if (phdr.type == _sentinal_type ||
655  get_key(phdr) == _sentinal_key ||
656  phdr.group_element == _sentinal_group) {
657  return true;
658  }
659  // also check for 0xFFFFFFFF and 0x00000000, which are the fill
660  // values for storage. These can appear if power off occurs while
661  // writing data
662  uint32_t v = *(uint32_t *)&phdr;

670 // return true if found, along with the offset in the EEPROM where
671 // the variable is stored
672 // if not found return the offset of the sentinal
673 // if the sentinal isn't found either, the offset is set to 0xFFFF
674 bool AP_Param::scan(const AP_Param::Param_header *target, uint16_t *pofs)
675 {
676  struct Param_header phdr;

678  while (ofs < _storage.size()) {

681  get_key(phdr) == get_key(*target) &&
682  phdr.group_element == target->group_element) {
683  // found it
684  *pofs = ofs;
685  return true;
686  }

689  *pofs = ofs;

691  }

693  }

696  return false;

987 /*
988  Save the variable to HAL storage, synchronous version
989 */
990 void AP_Param::save_sync(bool force_save)
991 {
992  uint32_t group_element = 0;
993  const struct GroupInfo *ginfo;
994  struct GroupNesting group_nesting {};
995  uint8_t idx;
996  const struct AP_Param::Info *info = find_var_info(&group_element, ginfo, group_nesting, &idx);
997  const AP_Param *ap;
998 
999  if (info == nullptr) {
1000  // we don't have any info on how to store it
1001  return;
1002  }
1003 
1004  struct Param_header phdr;
1005 
1006  // create the header we will use to store the variable
1007  if (ginfo != nullptr) {
1008  phdr.type = ginfo->type;
1009  } else {
1010  phdr.type = info->type;
1011  }
1012  set_key(phdr, info->key);
1013  phdr.group_element = group_element;
1014 
1015  ap = this;
1016  if (phdr.type != AP_PARAM_VECTOR3F && idx != 0) {
1017  // only vector3f can have non-zero idx for now
1018  return;
1019  }
1020  if (idx != 0) {
1021  ap = (const AP_Param *)((ptrdiff_t)ap) - (idx*sizeof(float));
1022  }
1023 
1024  if (phdr.type == AP_PARAM_INT8 && ginfo != nullptr && (ginfo->flags & AP_PARAM_FLAG_ENABLE)) {
1025  // clear cached parameter count
1026  _parameter_count = 0;
1027  }
1028 
1029  char name[AP_MAX_NAME_SIZE+1];
1030  copy_name_info(info, ginfo, group_nesting, idx, name, sizeof(name), true);
1031 
1032  // scan EEPROM to find the right location
1033  uint16_t ofs;
1034  if (scan(&phdr, &ofs)) {
1035  // found an existing copy of the variable
1036  eeprom_write_check(ap, ofs+sizeof(phdr), type_size((enum ap_var_type)phdr.type));
1037  send_parameter(name, (enum ap_var_type)phdr.type, idx);
1038  return;
1039  }
1040  if (ofs == (uint16_t) ~0) {
1041  return;
1042  }
1043 
1044  // if the value is the default value then don't save
1045  if (phdr.type <= AP_PARAM_FLOAT) {
1046  float v1 = cast_to_float((enum ap_var_type)phdr.type);
1047  float v2;
1048  if (ginfo != nullptr) {
1049  v2 = get_default_value(this, &ginfo->def_value);
1050  } else {
1051  v2 = get_default_value(this, &info->def_value);
1052  }
1053  if (is_equal(v1,v2) && !force_save) {
1054  gcs().send_parameter_value(name, (enum ap_var_type)info->type, v2);
1055  return;
1056  }
1057  if (!force_save &&
1058  (phdr.type != AP_PARAM_INT32 &&
1059  (fabsf(v1-v2) < 0.0001f*fabsf(v1)))) {
1060  // for other than 32 bit integers, we accept values within
1061  // 0.01 percent of the current value as being the same
1062  gcs().send_parameter_value(name, (enum ap_var_type)info->type, v2);
1063  return;
1064  }
1065  }
1066 
1067  if (ofs+type_size((enum ap_var_type)phdr.type)+2*sizeof(phdr) >= _storage.size()) {
1068  // we are out of room for saving variables
1069  hal.console->printf("EEPROM full\n");
1070  return;
1071  }
1072 
1073  // write a new sentinal, then the data, then the header
1074  write_sentinal(ofs + sizeof(phdr) + type_size((enum ap_var_type)phdr.type));
1075  eeprom_write_check(ap, ofs+sizeof(phdr), type_size((enum ap_var_type)phdr.type));
1076  eeprom_write_check(&phdr, ofs, sizeof(phdr));
1077 
1078  send_parameter(name, (enum ap_var_type)phdr.type, idx);
1079 }

1684 // convert one old vehicle parameter to new object parameter
1685 void AP_Param::convert_old_parameter(const struct ConversionInfo *info, float scaler, uint8_t flags)
1686 {
1687  uint8_t old_value[type_size(info->type)];
1688  AP_Param *ap = (AP_Param *)&old_value[0];
1689 
1690  if (!find_old_parameter(info, ap)) {
1691  // the old parameter isn't saved in the EEPROM. It was
1692  // probably still set to the default value, which isn't stored
1693  // no need to convert
1694  return;
1695  }
1696

1698  enum ap_var_type ptype;

1700  ap2 = find(&info->new_name[0], &ptype);
1701  if (ap2 == nullptr) {
1702  hal.console->printf("Unknown conversion '%s'\n", info->new_name);
1703  return;
1704  }
1705

1707  if (!(flags & CONVERT_FLAG_FORCE) && ap2->configured_in_storage()) {
1708  // the new parameter already has a value set by the user, or
1709  // has already been converted
1710  return;
1711  }
1712 
1713  // see if they are the same type and no scaling applied
1714  if (ptype == info->type && is_equal(scaler, 1.0f) && flags == 0) {
1715  // copy the value over only if the new parameter does not already
1716  // have the old value (via a default).
1717  if (memcmp(ap2, ap, sizeof(old_value)) != 0) {
1718  memcpy(ap2, ap, sizeof(old_value));
1719  // and save
1720  ap2->save();
1721  }
1722  } else if (ptype <= AP_PARAM_FLOAT && info->type <= AP_PARAM_FLOAT) {
1723  // perform scalar->scalar conversion
1724  float v = ap->cast_to_float(info->type);
1725  if (flags & CONVERT_FLAG_REVERSE) {
1726  // convert a _REV parameter to a _REVERSED parameter
1727  v = is_equal(v, -1.0f)?1:0;
1728  }
1729  if (!is_equal(v,ap2->cast_to_float(ptype))) {
1730  // the value needs to change
1731  set_value(ptype, ap2, v * scaler);
1732  ap2->save();
1733  }
1734  } else {
1735  // can't do vector<->scalar conversion, or different vector types
1736  hal.console->printf("Bad conversion type '%s'\n", info->new_name);
1737  }
