227 bool AP_Arming::ins_accels_consistent(const AP_InertialSensor &ins)
228 {
229  const uint8_t accel_count = ins.get_accel_count();
230  if (accel_count <= 1) {
231  return true;
232  }
233 
234  const Vector3f &prime_accel_vec = ins.get_accel();
235  const uint32_t now = AP_HAL::millis();
236  for(uint8_t i=0; i<accel_count; i++) {
237  if (!ins.use_accel(i)) {
238  continue;
239  }
240  // get next accel vector
241  const Vector3f &accel_vec = ins.get_accel(i);
242  Vector3f vec_diff = accel_vec - prime_accel_vec;
243  // allow for user-defined difference, typically 0.75 m/s/s. Has to pass in last 10 seconds
244  float threshold = accel_error_threshold;
245  if (i >= 2) {
246  /*
247  we allow for a higher threshold for IMU3 as it
248  runs at a different temperature to IMU1/IMU2,
249  and is not used for accel data in the EKF
250  */
251  threshold *= 3;
252  }
253 
254  // EKF is less sensitive to Z-axis error
255  vec_diff.z *= 0.5f;
256 
257  if (vec_diff.length() <= threshold) {
258  last_accel_pass_ms[i] = now;
259  }
260  if (now - last_accel_pass_ms[i] > 10000) {
261  return false;
262  }
263  }
264 
265  return true;
266 }

268 bool AP_Arming::ins_gyros_consistent(const AP_InertialSensor &ins)
269 {
270  const uint8_t gyro_count = ins.get_gyro_count();
271  if (gyro_count <= 1) {
272  return true;
273  }
274 
275  const Vector3f &prime_gyro_vec = ins.get_gyro();
276  const uint32_t now = AP_HAL::millis();
277  for(uint8_t i=0; i<gyro_count; i++) {
278  if (!ins.use_gyro(i)) {
279  continue;
280  }
281  // get next gyro vector
282  const Vector3f &gyro_vec = ins.get_gyro(i);
283  const Vector3f vec_diff = gyro_vec - prime_gyro_vec;
284  // allow for up to 5 degrees/s difference. Pass if it has
285  // been OK in last 10 seconds
286  if (vec_diff.length() <= radians(5)) {
287  last_gyro_pass_ms[i] = now;
288  }
289  if (now - last_gyro_pass_ms[i] > 10000) {
290  return false;
291  }
292  }
293 
294  return true;
295 }

297 bool AP_Arming::ins_checks(bool report)
298 {
299  if ((checks_to_perform & ARMING_CHECK_ALL) ||
300  (checks_to_perform & ARMING_CHECK_INS)) {
301  const AP_InertialSensor &ins = AP::ins();
302  if (!ins.get_gyro_health_all()) {
303  check_failed(ARMING_CHECK_INS, report, "Gyros not healthy");
304  return false;
305  }
306  if (!ins.gyro_calibrated_ok_all()) {
307  check_failed(ARMING_CHECK_INS, report, "Gyros not calibrated");
308  return false;
309  }
310  if (!ins.get_accel_health_all()) {
311  check_failed(ARMING_CHECK_INS, report, "Accels not healthy");
312  return false;
313  }
314  if (!ins.accel_calibrated_ok_all()) {
315  check_failed(ARMING_CHECK_INS, report, "3D Accel calibration needed");
316  return false;
317  }
318 
319  //check if accelerometers have calibrated and require reboot
320  if (ins.accel_cal_requires_reboot()) {
321  check_failed(ARMING_CHECK_INS, report, "Accels calibrated requires reboot");
322  return false;
323  }
324 
325  // check all accelerometers point in roughly same direction
326  if (!ins_accels_consistent(ins)) {
327  check_failed(ARMING_CHECK_INS, report, "Accels inconsistent");
328  return false;
329  }
330 
331  // check all gyros are giving consistent readings
332  if (!ins_gyros_consistent(ins)) {
333  check_failed(ARMING_CHECK_INS, report, "Gyros inconsistent");
334  return false;
335  }
336 
337  // check AHRS attitudes are consistent
338  char failure_msg[50] = {};
339  if (!AP::ahrs().attitudes_consistent(failure_msg, ARRAY_SIZE(failure_msg))) {
340  check_failed(ARMING_CHECK_INS, report, "%s", failure_msg);
341  return false;
342  }
343  }
344 
345  return true;
346 }

348 bool AP_Arming::compass_checks(bool report)
349 {
350  Compass &_compass = AP::compass();
351 
352  // check if compass is calibrating
353  if (_compass.is_calibrating()) {
354  check_failed(ARMING_CHECK_NONE, report, "Compass calibration running");
355  return false;
356  }
357 
358  // check if compass has calibrated and requires reboot
359  if (_compass.compass_cal_requires_reboot()) {
360  check_failed(ARMING_CHECK_NONE, report, "Compass calibrated requires reboot");
361  return false;
362  }
363 
364  if ((checks_to_perform) & ARMING_CHECK_ALL ||
365  (checks_to_perform) & ARMING_CHECK_COMPASS) {
366 
367  // avoid Compass::use_for_yaw(void) as it implicitly calls healthy() which can
368  // incorrectly skip the remaining checks, pass the primary instance directly
369  if (!_compass.use_for_yaw(_compass.get_primary())) {
370  // compass use is disabled
371  return true;
372  }
373 
374  if (!_compass.healthy()) {
375  check_failed(ARMING_CHECK_COMPASS, report, "Compass not healthy");
376  return false;
377  }
378  // check compass learning is on or offsets have been set
379  if (!_compass.learn_offsets_enabled() && !_compass.configured()) {
380  check_failed(ARMING_CHECK_COMPASS, report, "Compass not calibrated");
381  return false;
382  }
383 
384  // check for unreasonable compass offsets
385  const Vector3f offsets = _compass.get_offsets();
386  if (offsets.length() > _compass.get_offsets_max()) {
387  check_failed(ARMING_CHECK_COMPASS, report, "Compass offsets too high");
388  return false;
389  }
390 
391  // check for unreasonable mag field length
392  const float mag_field = _compass.get_field().length();
393  if (mag_field > AP_ARMING_COMPASS_MAGFIELD_MAX || mag_field < AP_ARMING_COMPASS_MAGFIELD_MIN) {
394  check_failed(ARMING_CHECK_COMPASS, report, "Check mag field");
395  return false;
396  }
397 
398  // check all compasses point in roughly same direction
399  if (!_compass.consistent()) {
400  check_failed(ARMING_CHECK_COMPASS, report, "Compasses inconsistent");
401  return false;
402  }
403  }
404 
405  return true;
406 }

408 bool AP_Arming::gps_checks(bool report)
409 {
410  const AP_GPS &gps = AP::gps();
411  if ((checks_to_perform & ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_GPS)) {
412 
413  //GPS OK?
414  if (!AP::ahrs().home_is_set() ||
415  gps.status() < AP_GPS::GPS_OK_FIX_3D) {
416  check_failed(ARMING_CHECK_GPS, report, "Bad GPS Position");
417  return false;
418  }
419 
420  //GPS update rate acceptable
421  if (!gps.is_healthy()) {
422  check_failed(ARMING_CHECK_GPS, report, "GPS is not healthy");
423  return false;
424  }
425 
426  // check GPSs are within 50m of each other and that blending is healthy
427  float distance_m;
428  if (!gps.all_consistent(distance_m)) {
429  check_failed(ARMING_CHECK_GPS, report, "GPS positions differ by %4.1fm",
430  (double)distance_m);
431  return false;
432  }
433  if (!gps.blend_health_check()) {
434  check_failed(ARMING_CHECK_GPS, report, "GPS blending unhealthy");
435  return false;
436  }
437 
438  // check AHRS and GPS are within 10m of each other
439  const Location gps_loc = gps.location();
440  Location ahrs_loc;
441  if (AP::ahrs().get_position(ahrs_loc)) {
442  const float distance = gps_loc.get_distance(ahrs_loc);
443  if (distance > AP_ARMING_AHRS_GPS_ERROR_MAX) {
444  check_failed(ARMING_CHECK_GPS, report, "GPS and AHRS differ by %4.1fm", (double)distance);
445  return false;
446  }
447  }
448  }
449 
450  if ((checks_to_perform & ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_GPS_CONFIG)) {
451  const uint8_t first_unconfigured = gps.first_unconfigured_gps();
452  if (first_unconfigured != AP_GPS::GPS_ALL_CONFIGURED) {
453  check_failed(ARMING_CHECK_GPS_CONFIG,
454  report,
455  "GPS %d failing configuration checks",
456  first_unconfigured + 1);
457  if (report) {
458  gps.broadcast_first_configuration_failure_reason();
459  }
460  return false;
461  }
462  }
463 
464  return true;
465 }

496 bool AP_Arming::rc_calibration_checks(bool report)
497 {
498  bool check_passed = true;
499  const uint8_t num_channels = RC_Channels::get_valid_channel_count();
500  for (uint8_t i = 0; i < NUM_RC_CHANNELS; i++) {
501  const RC_Channel *c = rc().channel(i);
502  if (c == nullptr) {
503  continue;
504  }
505  if (i >= num_channels && !(c->has_override())) {
506  continue;
507  }
508  const uint16_t trim = c->get_radio_trim();
509  if (c->get_radio_min() > trim) {
510  check_failed(ARMING_CHECK_RC, report, "RC%d minimum is greater than trim", i + 1);
511  check_passed = false;
512  }
513  if (c->get_radio_max() < trim) {
514  check_failed(ARMING_CHECK_RC, report, "RC%d maximum is less than trim", i + 1);
515  check_passed = false;
516  }
517  }
518 
519  return check_passed;
520 }
