532  // reset the position to the current GPS position
533  ResetPosition();
534  // reset the velocity to the GPS velocity
535  ResetVelocity();
536  // don't fuse GPS data on this time step
537  fusePosData = false;
538  fuseVelData = false;
539  // Reset the position variances and corresponding covariances to a value that will pass the checks
540  zeroRows(P,6,7);
541  zeroCols(P,6,7);
542  P[6][6] = sq(float(0.5f*frontend->_gpsGlitchRadiusMax));
543  P[7][7] = P[6][6];
544  // Reset the normalised innovation to avoid failing the bad fusion tests
545  posTestRatio = 0.0f;
546  velTestRatio = 0.0f;
547  }
548  } else {
549  posHealth = false;
550  }
551  }
552 
553  // test velocity measurements
554  if (fuseVelData) {
555  // test velocity measurements
556  uint8_t imax = 2;
557  // Don't fuse vertical velocity observations if inhibited by the user or if we are using synthetic data
558  if (frontend->_fusionModeGPS > 0 || PV_AidingMode != AID_ABSOLUTE || frontend->inhibitGpsVertVelUse) {
559  imax = 1;
560  }
561  float innovVelSumSq = 0; // sum of squares of velocity innovations
562  float varVelSum = 0; // sum of velocity innovation variances
563  for (uint8_t i = 0; i<=imax; i++) {
564  // velocity states start at index 3
565  stateIndex = i + 3;
566  // calculate innovations using blended and single IMU predicted states
567  velInnov[i] = stateStruct.velocity[i] - velPosObs[i]; // blended
568  // calculate innovation variance
569  varInnovVelPos[i] = P[stateIndex][stateIndex] + R_OBS_DATA_CHECKS[i];
570  // sum the innovation and innovation variances
571  innovVelSumSq += sq(velInnov[i]);
572  varVelSum += varInnovVelPos[i];
573  }
574  // apply an innovation consistency threshold test, but don't fail if bad IMU data
575  // calculate the test ratio
576  velTestRatio = innovVelSumSq / (varVelSum * sq(MAX(0.01f * (float)frontend->_gpsVelInnovGate, 1.0f)));
577  // fail if the ratio is greater than 1
578  velHealth = ((velTestRatio < 1.0f) || badIMUdata);
579  // use velocity data if healthy, timed out, or in constant position mode
580  if (velHealth || velTimeout) {
581  velHealth = true;
582  // restart the timeout count
583  lastVelPassTime_ms = imuSampleTime_ms;
584  // If we are doing full aiding and velocity fusion times out, reset to the GPS velocity
585  if (PV_AidingMode == AID_ABSOLUTE && velTimeout) {
586  // reset the velocity to the GPS velocity
587  ResetVelocity();
588  // don't fuse GPS velocity data on this time step
589  fuseVelData = false;
590  // Reset the normalised innovation to avoid failing the bad fusion tests
591  velTestRatio = 0.0f;
592  }
593  } else {
594  velHealth = false;
595  }
596  }
597 
598  // test height measurements
599  if (fuseHgtData) {
600  // calculate height innovations
601  innovVelPos[5] = stateStruct.position.z - velPosObs[5];
602  varInnovVelPos[5] = P[8][8] + R_OBS_DATA_CHECKS[5];
603  // calculate the innovation consistency test ratio
604  hgtTestRatio = sq(innovVelPos[5]) / (sq(MAX(0.01f * (float)frontend->_hgtInnovGate, 1.0f)) * varInnovVelPos[5]);
605  // fail if the ratio is > 1, but don't fail if bad IMU data
606  hgtHealth = ((hgtTestRatio < 1.0f) || badIMUdata);
607  // Fuse height data if healthy or timed out or in constant position mode
608  if (hgtHealth || hgtTimeout || (PV_AidingMode == AID_NONE && onGround)) {
609  // Calculate a filtered value to be used by pre-flight health checks
610  // We need to filter because wind gusts can generate significant baro noise and we want to be able to detect bias errors in the inertial solution
611  if (onGround) {
612  float dtBaro = (imuSampleTime_ms - lastHgtPassTime_ms)*1.0e-3f;
613  const float hgtInnovFiltTC = 2.0f;
614  float alpha = constrain_float(dtBaro/(dtBaro+hgtInnovFiltTC),0.0f,1.0f);
615  hgtInnovFiltState += (innovVelPos[5]-hgtInnovFiltState)*alpha;
616  } else {
617  hgtInnovFiltState = 0.0f;
618  }
619 
620  // if timed out, reset the height
621  if (hgtTimeout) {
622  ResetHeight();
623  }
624 
625  // If we have got this far then declare the height data as healthy and reset the timeout counter
626  hgtHealth = true;
627  lastHgtPassTime_ms = imuSampleTime_ms;
628  }
629  }
630 
631  // set range for sequential fusion of velocity and position measurements depending on which data is available and its health
632  if (fuseVelData && velHealth) {
633  fuseData[0] = true;
634  fuseData[1] = true;
635  if (useGpsVertVel) {
636  fuseData[2] = true;
637  }
638  tiltErrVec.zero();
639  }
640  if (fusePosData && posHealth) {
641  fuseData[3] = true;
642  fuseData[4] = true;
643  tiltErrVec.zero();
644  }
645  if (fuseHgtData && hgtHealth) {
646  fuseData[5] = true;
647  }
648 
649  // fuse measurements sequentially
650  for (obsIndex=0; obsIndex<=5; obsIndex++) {
651  if (fuseData[obsIndex]) {
652  stateIndex = 3 + obsIndex;
653  // calculate the measurement innovation, using states from a different time coordinate if fusing height data
654  // adjust scaling on GPS measurement noise variances if not enough satellites
655  if (obsIndex <= 2)
656  {
657  innovVelPos[obsIndex] = stateStruct.velocity[obsIndex] - velPosObs[obsIndex];
658  R_OBS[obsIndex] *= sq(gpsNoiseScaler);
659  }
660  else if (obsIndex == 3 || obsIndex == 4) {
661  innovVelPos[obsIndex] = stateStruct.position[obsIndex-3] - velPosObs[obsIndex];
662  R_OBS[obsIndex] *= sq(gpsNoiseScaler);
663  } else if (obsIndex == 5) {
664  innovVelPos[obsIndex] = stateStruct.position[obsIndex-3] - velPosObs[obsIndex];
665  const float gndMaxBaroErr = 4.0f;
666  const float gndBaroInnovFloor = -0.5f;
667 
668  if(getTouchdownExpected() && activeHgtSource == HGT_SOURCE_BARO) {
669  // when a touchdown is expected, floor the barometer innovation at gndBaroInnovFloor
670  // constrain the correction between 0 and gndBaroInnovFloor+gndMaxBaroErr
671  // this function looks like this:
672  // |/
673  //---------|---------
674  // ____/|
675  // / |
676  // / |
677  innovVelPos[5] += constrain_float(-innovVelPos[5]+gndBaroInnovFloor, 0.0f, gndBaroInnovFloor+gndMaxBaroErr);
678  }
679  }
680 
681  // calculate the Kalman gain and calculate innovation variances
682  varInnovVelPos[obsIndex] = P[stateIndex][stateIndex] + R_OBS[obsIndex];
683  SK = 1.0f/varInnovVelPos[obsIndex];
684  for (uint8_t i= 0; i<=15; i++) {
685  Kfusion[i] = P[i][stateIndex]*SK;
686  }
687 
688  // inhibit magnetic field state estimation by setting Kalman gains to zero
689  if (!inhibitMagStates) {
690  for (uint8_t i = 16; i<=21; i++) {
691  Kfusion[i] = P[i][stateIndex]*SK;
692  }
693  } else {
694  for (uint8_t i = 16; i<=21; i++) {
695  Kfusion[i] = 0.0f;
696  }
697  }
698 
699  // inhibit wind state estimation by setting Kalman gains to zero
700  if (!inhibitWindStates) {
701  Kfusion[22] = P[22][stateIndex]*SK;
702  Kfusion[23] = P[23][stateIndex]*SK;
703  } else {
704  Kfusion[22] = 0.0f;
705  Kfusion[23] = 0.0f;
706  }
707 
708  // update the covariance - take advantage of direct observation of a single state at index = stateIndex to reduce computations
709  // this is a numerically optimised implementation of standard equation P = (I - K*H)*P;
710  for (uint8_t i= 0; i<=stateIndexLim; i++) {
711  for (uint8_t j= 0; j<=stateIndexLim; j++)
712  {
713  KHP[i][j] = Kfusion[i] * P[stateIndex][j];
714  }
715  }
716  // Check that we are not going to drive any variances negative and skip the update if so
717  bool healthyFusion = true;
718  for (uint8_t i= 0; i<=stateIndexLim; i++) {
719  if (KHP[i][i] > P[i][i]) {
720  healthyFusion = false;
721  }
722  }
723  if (healthyFusion) {
724  // update the covariance matrix
725  for (uint8_t i= 0; i<=stateIndexLim; i++) {
726  for (uint8_t j= 0; j<=stateIndexLim; j++) {
727  P[i][j] = P[i][j] - KHP[i][j];
728  }
729  }
730 
731  // force the covariance matrix to be symmetrical and limit the variances to prevent ill-conditioning.
732  ForceSymmetry();
733  ConstrainVariances();
734 
735  // update the states
736  // zero the attitude error state - by definition it is assumed to be zero before each observation fusion
737  stateStruct.angErr.zero();
738 
739  // calculate state corrections and re-normalise the quaternions for states predicted using the blended IMU data
740  for (uint8_t i = 0; i<=stateIndexLim; i++) {
741  statesArray[i] = statesArray[i] - Kfusion[i] * innovVelPos[obsIndex];
742  }
743 
744  // the first 3 states represent the angular misalignment vector. This is
745  // is used to correct the estimated quaternion
746  stateStruct.quat.rotate(stateStruct.angErr);
747 
748  // sum the attitude error from velocity and position fusion only
749  // used as a metric for convergence monitoring
750  if (obsIndex != 5) {
751  tiltErrVec += stateStruct.angErr;
752  }
753  // record good fusion status
754  if (obsIndex == 0) {
755  faultStatus.bad_nvel = false;
756  } else if (obsIndex == 1) {
757  faultStatus.bad_evel = false;
758  } else if (obsIndex == 2) {
759  faultStatus.bad_dvel = false;
760  } else if (obsIndex == 3) {
761  faultStatus.bad_npos = false;
762  } else if (obsIndex == 4) {
763  faultStatus.bad_epos = false;
764  } else if (obsIndex == 5) {
765  faultStatus.bad_dpos = false;
766  }
767  } else {
768  // record bad fusion status
769  if (obsIndex == 0) {
770  faultStatus.bad_nvel = true;
771  } else if (obsIndex == 1) {
772  faultStatus.bad_evel = true;
773  } else if (obsIndex == 2) {
774  faultStatus.bad_dvel = true;
775  } else if (obsIndex == 3) {
776  faultStatus.bad_npos = true;
777  } else if (obsIndex == 4) {
778  faultStatus.bad_epos = true;
779  } else if (obsIndex == 5) {
780  faultStatus.bad_dpos = true;
781  }
782  }
783  }
784  }
785  }
786 
787  // stop performance timer
788  hal.util->perf_end(_perf_FuseVelPosNED);
789 }

795 // select the height measurement to be fused from the available baro, range finder and GPS sources
796 void NavEKF2_core::selectHeightForFusion()
797 {
798  // Read range finder data and check for new data in the buffer
799  // This data is used by both height and optical flow fusion processing
800  readRangeFinder();
801  rangeDataToFuse = storedRange.recall(rangeDataDelayed,imuDataDelayed.time_ms);
802 
803  // correct range data for the body frame position offset relative to the IMU
804  // the corrected reading is the reading that would have been taken if the sensor was
805  // co-located with the IMU
806  if (rangeDataToFuse) {
807  AP_RangeFinder_Backend *sensor = frontend->_rng.get_backend(rangeDataDelayed.sensor_idx);
808  if (sensor != nullptr) {
809  Vector3f posOffsetBody = sensor->get_pos_offset() - accelPosOffset;
810  if (!posOffsetBody.is_zero()) {
811  Vector3f posOffsetEarth = prevTnb.mul_transpose(posOffsetBody);
812  rangeDataDelayed.rng += posOffsetEarth.z / prevTnb.c.z;
813  }
814  }
815  }
816 
817  // read baro height data from the sensor and check for new data in the buffer
818  readBaroData();
819  baroDataToFuse = storedBaro.recall(baroDataDelayed, imuDataDelayed.time_ms);
820 
821  // select height source
822  if (extNavUsedForPos) {
823  // always use external vision as the height source if using for position.
824  activeHgtSource = HGT_SOURCE_EV;
825  } else if (((frontend->_useRngSwHgt > 0) || (frontend->_altSource == 1)) && (imuSampleTime_ms - rngValidMeaTime_ms < 500)) {
826  if (frontend->_altSource == 1) {
827  // always use range finder
828  activeHgtSource = HGT_SOURCE_RNG;
829  } else {
830  // determine if we are above or below the height switch region
831  float rangeMaxUse = 1e-4f * (float)frontend->_rng.max_distance_cm_orient(ROTATION_PITCH_270) * (float)frontend->_useRngSwHgt;
832  bool aboveUpperSwHgt = (terrainState - stateStruct.position.z) > rangeMaxUse;
833  bool belowLowerSwHgt = (terrainState - stateStruct.position.z) < 0.7f * rangeMaxUse;
834 
835  // If the terrain height is consistent and we are moving slowly, then it can be
836  // used as a height reference in combination with a range finder
837  // apply a hysteresis to the speed check to prevent rapid switching
838  float horizSpeed = norm(stateStruct.velocity.x, stateStruct.velocity.y);
839  bool dontTrustTerrain = ((horizSpeed > frontend->_useRngSwSpd) && filterStatus.flags.horiz_vel) || !terrainHgtStable;
840  float trust_spd_trigger = MAX((frontend->_useRngSwSpd - 1.0f),(frontend->_useRngSwSpd * 0.5f));
841  bool trustTerrain = (horizSpeed < trust_spd_trigger) && terrainHgtStable;
842 
843  /*
844  * Switch between range finder and primary height source using height above ground and speed thresholds with
845  * hysteresis to avoid rapid switching. Using range finder for height requires a consistent terrain height
846  * which cannot be assumed if the vehicle is moving horizontally.
847  */
848  if ((aboveUpperSwHgt || dontTrustTerrain) && (activeHgtSource == HGT_SOURCE_RNG)) {
849  // cannot trust terrain or range finder so stop using range finder height
850  if (frontend->_altSource == 0) {
851  activeHgtSource = HGT_SOURCE_BARO;
852  } else if (frontend->_altSource == 2) {
853  activeHgtSource = HGT_SOURCE_GPS;
854  }
855  } else if (belowLowerSwHgt && trustTerrain && (activeHgtSource != HGT_SOURCE_RNG)) {
856  // reliable terrain and range finder so start using range finder height
857  activeHgtSource = HGT_SOURCE_RNG;
858  }
859  }
860  } else if ((frontend->_altSource == 2) && ((imuSampleTime_ms - lastTimeGpsReceived_ms) < 500) && validOrigin && gpsAccuracyGood) {
861  activeHgtSource = HGT_SOURCE_GPS;
862  } else if ((frontend->_altSource == 3) && validOrigin && rngBcnGoodToAlign) {
863  activeHgtSource = HGT_SOURCE_BCN;
864  } else {
865  activeHgtSource = HGT_SOURCE_BARO;
866  }
867 
868  // Use Baro alt as a fallback if we lose range finder, GPS or external nav
869  bool lostRngHgt = ((activeHgtSource == HGT_SOURCE_RNG) && ((imuSampleTime_ms - rngValidMeaTime_ms) > 500));
870  bool lostGpsHgt = ((activeHgtSource == HGT_SOURCE_GPS) && ((imuSampleTime_ms - lastTimeGpsReceived_ms) > 2000));
871  bool lostExtNavHgt = ((activeHgtSource == HGT_SOURCE_EV) && ((imuSampleTime_ms - extNavMeasTime_ms) > 2000));
872  if (lostRngHgt || lostGpsHgt || lostExtNavHgt) {
873  activeHgtSource = HGT_SOURCE_BARO;
874  }
875 
876  // if there is new baro data to fuse, calculate filtered baro data required by other processes
877  if (baroDataToFuse) {
878  // calculate offset to baro data that enables us to switch to Baro height use during operation
879  if (activeHgtSource != HGT_SOURCE_BARO) {
880  calcFiltBaroOffset();
881  }
882  // filtered baro data used to provide a reference for takeoff
883  // it is is reset to last height measurement on disarming in performArmingChecks()
884  if (!getTakeoffExpected()) {
885  const float gndHgtFiltTC = 0.5f;
886  const float dtBaro = frontend->hgtAvg_ms*1.0e-3f;
887  float alpha = constrain_float(dtBaro / (dtBaro+gndHgtFiltTC),0.0f,1.0f);
888  meaHgtAtTakeOff += (baroDataDelayed.hgt-meaHgtAtTakeOff)*alpha;
889  }
890  }
891 
892  // If we are not using GPS as the primary height sensor, correct EKF origin height so that
893  // combined local NED position height and origin height remains consistent with the GPS altitude
894  // This also enables the GPS height to be used as a backup height source
895  if (gpsDataToFuse &&
896  (((frontend->_originHgtMode & (1 << 0)) && (activeHgtSource == HGT_SOURCE_BARO)) ||
897  ((frontend->_originHgtMode & (1 << 1)) && (activeHgtSource == HGT_SOURCE_RNG)))
898  ) {
899  correctEkfOriginHeight();
900  }
901 
902  // Select the height measurement source
903  if (extNavDataToFuse && (activeHgtSource == HGT_SOURCE_EV)) {
904  hgtMea = -extNavDataDelayed.pos.z;
905  posDownObsNoise = sq(constrain_float(extNavDataDelayed.posErr, 0.01f, 10.0f));
906  } else if (rangeDataToFuse && (activeHgtSource == HGT_SOURCE_RNG)) {
907  // using range finder data
908  // correct for tilt using a flat earth model
909  if (prevTnb.c.z >= 0.7) {
910  // calculate height above ground
911  hgtMea = MAX(rangeDataDelayed.rng * prevTnb.c.z, rngOnGnd);
912  // correct for terrain position relative to datum
913  hgtMea -= terrainState;
914  // enable fusion
915  fuseHgtData = true;
916  velPosObs[5] = -hgtMea;
917  // set the observation noise
918  posDownObsNoise = sq(constrain_float(frontend->_rngNoise, 0.1f, 10.0f));
919  // add uncertainty created by terrain gradient and vehicle tilt
920  posDownObsNoise += sq(rangeDataDelayed.rng * frontend->_terrGradMax) * MAX(0.0f , (1.0f - sq(prevTnb.c.z)));
921  } else {
922  // disable fusion if tilted too far
923  fuseHgtData = false;
924  }
925  } else if (gpsDataToFuse && (activeHgtSource == HGT_SOURCE_GPS)) {
926  // using GPS data
927  hgtMea = gpsDataDelayed.hgt;
928  // enable fusion
929  velPosObs[5] = -hgtMea;
930  fuseHgtData = true;
931  // set the observation noise using receiver reported accuracy or the horizontal noise scaled for typical VDOP/HDOP ratio
932  if (gpsHgtAccuracy > 0.0f) {
933  posDownObsNoise = sq(constrain_float(gpsHgtAccuracy, 1.5f * frontend->_gpsHorizPosNoise, 100.0f));
934  } else {
935  posDownObsNoise = sq(constrain_float(1.5f * frontend->_gpsHorizPosNoise, 0.1f, 10.0f));
936  }
937  } else if (baroDataToFuse && (activeHgtSource == HGT_SOURCE_BARO)) {
938  // using Baro data
939  hgtMea = baroDataDelayed.hgt - baroHgtOffset;
940  // enable fusion
941  velPosObs[5] = -hgtMea;
942  fuseHgtData = true;
943  // set the observation noise
944  posDownObsNoise = sq(constrain_float(frontend->_baroAltNoise, 0.1f, 10.0f));
945  // reduce weighting (increase observation noise) on baro if we are likely to be in ground effect
946  if (getTakeoffExpected() || getTouchdownExpected()) {
947  posDownObsNoise *= frontend->gndEffectBaroScaler;
948  }
949  // If we are in takeoff mode, the height measurement is limited to be no less than the measurement at start of takeoff
950  // This prevents negative baro disturbances due to copter downwash corrupting the EKF altitude during initial ascent
951  if (motorsArmed && getTakeoffExpected()) {
952  hgtMea = MAX(hgtMea, meaHgtAtTakeOff);
953  }
954  } else {
955  fuseHgtData = false;
956  }
957 
958  // If we haven't fused height data for a while, then declare the height data as being timed out
959  // set timeout period based on whether we have vertical GPS velocity available to constrain drift
960  hgtRetryTime_ms = (useGpsVertVel && !velTimeout) ? frontend->hgtRetryTimeMode0_ms : frontend->hgtRetryTimeMode12_ms;
961  if (imuSampleTime_ms - lastHgtPassTime_ms > hgtRetryTime_ms) {
962  hgtTimeout = true;
963  } else {
964  hgtTimeout = false;
965  }
966 }
