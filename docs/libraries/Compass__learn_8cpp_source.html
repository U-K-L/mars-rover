175 /*
176  process a new compass sample
177  */
178 void CompassLearn::process_sample(const struct sample &s)
179 {
180  uint16_t besti = 0;
181  float bestv = 0, worstv=0;
182 
183  /*
184  we run through the 72 possible yaw error values, and for each
185  one we calculate a value for the compass offsets if that yaw
186  error is correct.
187  */
188  for (uint16_t i=0; i<num_sectors; i++) {
189  float yaw_err_deg = i*(360/num_sectors);
190 
191  // form rotation matrix for the euler attitude
192  Matrix3f dcm;
193  dcm.from_euler(s.attitude.x, s.attitude.y, wrap_2PI(s.attitude.z + radians(yaw_err_deg)));
194 
195  // calculate the field we would expect to get if this yaw error is correct
196  Vector3f expected_field = dcm.transposed() * mag_ef;
197 
198  // calculate a value for the compass offsets for this yaw error
199  Vector3f v1 = mat * s.field;
200  Vector3f v2 = mat * expected_field;
201  Vector3f offsets = (v2 - v1) + s.offsets;
202  float delta = (offsets - predicted_offsets[i]).length();
203 
204  if (num_samples == 1) {
205  predicted_offsets[i] = offsets;
206  } else {
207  // lowpass the predicted offsets and the error
208  const float learn_rate = 0.92f;
209  predicted_offsets[i] = predicted_offsets[i] * learn_rate + offsets * (1-learn_rate);
210  errors[i] = errors[i] * learn_rate + delta * (1-learn_rate);
211  }
212 
213  // keep track of the current best prediction
214  if (i == 0 || errors[i] < bestv) {
215  besti = i;
216  bestv = errors[i];
217  }
218  // also keep the worst error. This is used as part of the convergence test
219  if (i == 0 || errors[i] > worstv) {
220  worstv = errors[i];
221  }
222  }
223 
224  WITH_SEMAPHORE(sem);
225 
226  // pass the current estimate to the front-end
227  best_offsets = predicted_offsets[besti];
228  best_error = bestv;
229  worst_error = worstv;
230  best_yaw_deg = wrap_360(degrees(s.attitude.z) + besti * (360/num_sectors));
231 }
