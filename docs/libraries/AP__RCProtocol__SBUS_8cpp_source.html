63 #define SBUS_RANGE_MIN 200.0f

69 /* pre-calculate the floating point stuff as far as possible at compile time */
70 #define SBUS_SCALE_FACTOR ((SBUS_TARGET_MAX - SBUS_TARGET_MIN) / (SBUS_RANGE_MAX - SBUS_RANGE_MIN))

73 /*
74  * S.bus decoder matrix.
75  *
76  * Each channel value can come from up to 3 input bytes. Each row in the
77  * matrix describes up to three bytes, and each entry gives:
78  *
79  * - byte offset in the data portion of the frame
80  * - right shift applied to the data byte
81  * - mask for the data byte
82  * - left shift applied to the result into the channel value
83  */
84 struct sbus_bit_pick {
85  uint8_t byte;
86  uint8_t rshift;
87  uint8_t mask;
88  uint8_t lshift;
89 };
90 static const struct sbus_bit_pick sbus_decoder[SBUS_INPUT_CHANNELS][3] = {
91  /* 0 */ { { 0, 0, 0xff, 0}, { 1, 0, 0x07, 8}, { 0, 0, 0x00, 0} },
92  /* 1 */ { { 1, 3, 0x1f, 0}, { 2, 0, 0x3f, 5}, { 0, 0, 0x00, 0} },
93  /* 2 */ { { 2, 6, 0x03, 0}, { 3, 0, 0xff, 2}, { 4, 0, 0x01, 10} },
94  /* 3 */ { { 4, 1, 0x7f, 0}, { 5, 0, 0x0f, 7}, { 0, 0, 0x00, 0} },
95  /* 4 */ { { 5, 4, 0x0f, 0}, { 6, 0, 0x7f, 4}, { 0, 0, 0x00, 0} },
96  /* 5 */ { { 6, 7, 0x01, 0}, { 7, 0, 0xff, 1}, { 8, 0, 0x03, 9} },
97  /* 6 */ { { 8, 2, 0x3f, 0}, { 9, 0, 0x1f, 6}, { 0, 0, 0x00, 0} },
98  /* 7 */ { { 9, 5, 0x07, 0}, {10, 0, 0xff, 3}, { 0, 0, 0x00, 0} },
99  /* 8 */ { {11, 0, 0xff, 0}, {12, 0, 0x07, 8}, { 0, 0, 0x00, 0} },
100  /* 9 */ { {12, 3, 0x1f, 0}, {13, 0, 0x3f, 5}, { 0, 0, 0x00, 0} },
101  /* 10 */ { {13, 6, 0x03, 0}, {14, 0, 0xff, 2}, {15, 0, 0x01, 10} },
102  /* 11 */ { {15, 1, 0x7f, 0}, {16, 0, 0x0f, 7}, { 0, 0, 0x00, 0} },
103  /* 12 */ { {16, 4, 0x0f, 0}, {17, 0, 0x7f, 4}, { 0, 0, 0x00, 0} },
104  /* 13 */ { {17, 7, 0x01, 0}, {18, 0, 0xff, 1}, {19, 0, 0x03, 9} },

116 // decode a full SBUS frame
117 bool AP_RCProtocol_SBUS::sbus_decode(const uint8_t frame[25], uint16_t *values, uint16_t *num_values,
118  bool *sbus_failsafe, bool *sbus_frame_drop, uint16_t max_values)
119 {
120  /* check frame boundary markers to avoid out-of-sync cases */
121  if ((frame[0] != 0x0f)) {
122  return false;
123  }
124 
125  switch (frame[24]) {
126  case 0x00:
127  /* this is S.BUS 1 */
128  break;
129  case 0x03:
130  /* S.BUS 2 SLOT0: RX battery and external voltage */
131  break;
132  case 0x83:
133  /* S.BUS 2 SLOT1 */
134  break;
135  case 0x43:
136  case 0xC3:
137  case 0x23:
138  case 0xA3:
139  case 0x63:
140  case 0xE3:
141  break;
142  default:
143  /* we expect one of the bits above, but there are some we don't know yet */
144  break;
145  }
146 
147  unsigned chancount = (max_values > SBUS_INPUT_CHANNELS) ?
148  SBUS_INPUT_CHANNELS : max_values;
149 
150  /* use the decoder matrix to extract channel data */
151  for (unsigned channel = 0; channel < chancount; channel++) {
152  unsigned value = 0;
153 
154  for (unsigned pick = 0; pick < 3; pick++) {
155  const struct sbus_bit_pick *decode = &sbus_decoder[channel][pick];
156 
157  if (decode->mask != 0) {
158  unsigned piece = frame[1 + decode->byte];
159  piece >>= decode->rshift;
160  piece &= decode->mask;
161  piece <<= decode->lshift;
162 
163  value |= piece;
164  }
165  }
166 
167 
168  /* convert 0-2048 values to 1000-2000 ppm encoding in a not too sloppy fashion */
169  values[channel] = (uint16_t)(value * SBUS_SCALE_FACTOR +.5f) + SBUS_SCALE_OFFSET;
170  }
171 
172  /* decode switch channels if data fields are wide enough */
173  if (max_values > 17 && chancount > 15) {
174  chancount = 18;
175 
176  /* channel 17 (index 16) */
177  values[16] = (frame[SBUS_FLAGS_BYTE] & (1 << 0))?1998:998;
178  /* channel 18 (index 17) */
179  values[17] = (frame[SBUS_FLAGS_BYTE] & (1 << 1))?1998:998;
180  }
181 
182  /* note the number of channels decoded */
183  *num_values = chancount;
184 
185  /* decode and handle failsafe and frame-lost flags */
186  if (frame[SBUS_FLAGS_BYTE] & (1 << SBUS_FAILSAFE_BIT)) { /* failsafe */
187  /* report that we failed to read anything valid off the receiver */
188  *sbus_failsafe = true;
189  *sbus_frame_drop = true;
190  } else if (frame[SBUS_FLAGS_BYTE] & (1 << SBUS_FRAMELOST_BIT)) { /* a frame was lost */
191  /* set a special warning flag
192  *
193  * Attention! This flag indicates a skipped frame only, not a total link loss! Handling this
194  * condition as fail-safe greatly reduces the reliability and range of the radio link,
195  * e.g. by prematurely issuing return-to-launch!!! */
196 
197  *sbus_failsafe = false;
198  *sbus_frame_drop = true;
199  } else {
200  *sbus_failsafe = false;
201  *sbus_frame_drop = false;
202  }
203 
204  return true;
205 }

226 // support byte input
227 void AP_RCProtocol_SBUS::_process_byte(uint32_t timestamp_us, uint8_t b)
228 {
229  const bool have_frame_gap = (timestamp_us - byte_input.last_byte_us >= 2000U);
230  byte_input.last_byte_us = timestamp_us;
231 
232  if (have_frame_gap) {
233  // if we have a frame gap then this must be the start of a new
234  // frame
235  byte_input.ofs = 0;
236  }
237  if (b != 0x0F && byte_input.ofs == 0) {
238  // definately not SBUS, missing header byte
239  return;
240  }
241  if (byte_input.ofs == 0 && !have_frame_gap) {
242  // must have a frame gap before the start of a new SBUS frame
243  return;
244  }
245 
246  byte_input.buf[byte_input.ofs++] = b;
247 
248  if (byte_input.ofs == sizeof(byte_input.buf)) {
249  uint16_t values[SBUS_INPUT_CHANNELS];
250  uint16_t num_values=0;
251  bool sbus_failsafe = false;
252  bool sbus_frame_drop = false;
253  if (sbus_decode(byte_input.buf, values, &num_values,
254  &sbus_failsafe, &sbus_frame_drop, SBUS_INPUT_CHANNELS) &&
255  num_values >= MIN_RCIN_CHANNELS) {
256  add_input(num_values, values, sbus_failsafe);
257  }
258  byte_input.ofs = 0;
259  }
260 }
