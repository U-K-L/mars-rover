620  * because the EKF is running on a delayed time horizon. Note that the quaternion is
621  * not used by the EKF equations, which instead estimate the error in the attitude of
622  * the vehicle when each observation is fused. This attitude error is then used to correct
623  * the quaternion.
624 */
625 void NavEKF2_core::UpdateStrapdownEquationsNED()
626 {
627  // update the quaternion states by rotating from the previous attitude through
628  // the delta angle rotation quaternion and normalise
629  // apply correction for earth's rotation rate
630  // % * - and + operators have been overloaded
631  stateStruct.quat.rotate(delAngCorrected - prevTnb * earthRateNED*imuDataDelayed.delAngDT);
632  stateStruct.quat.normalize();
633 
634  // transform body delta velocities to delta velocities in the nav frame
635  // use the nav frame from previous time step as the delta velocities
636  // have been rotated into that frame
637  // * and + operators have been overloaded
638  Vector3f delVelNav; // delta velocity vector in earth axes
639  delVelNav = prevTnb.mul_transpose(delVelCorrected);
640  delVelNav.z += GRAVITY_MSS*imuDataDelayed.delVelDT;
641 
642  // calculate the body to nav cosine matrix
643  stateStruct.quat.inverse().rotation_matrix(prevTnb);
644 
645  // calculate the rate of change of velocity (used for launch detect and other functions)
646  velDotNED = delVelNav / imuDataDelayed.delVelDT;
647 
648  // apply a first order lowpass filter
649  velDotNEDfilt = velDotNED * 0.05f + velDotNEDfilt * 0.95f;
650 
651  // calculate a magnitude of the filtered nav acceleration (required for GPS
652  // variance estimation)
653  accNavMag = velDotNEDfilt.length();
654  accNavMagHoriz = norm(velDotNEDfilt.x , velDotNEDfilt.y);
655 
656  // if we are not aiding, then limit the horizontal magnitude of acceleration
657  // to prevent large manoeuvre transients disturbing the attitude
658  if ((PV_AidingMode == AID_NONE) && (accNavMagHoriz > 5.0f)) {
659  float gain = 5.0f/accNavMagHoriz;
660  delVelNav.x *= gain;
661  delVelNav.y *= gain;
662  }
663 
664  // save velocity for use in trapezoidal integration for position calcuation
665  Vector3f lastVelocity = stateStruct.velocity;
666 
667  // sum delta velocities to get velocity
668  stateStruct.velocity += delVelNav;
669 
670  // apply a trapezoidal integration to velocities to calculate position
671  stateStruct.position += (stateStruct.velocity + lastVelocity) * (imuDataDelayed.delVelDT*0.5f);
672 
673  // accumulate the bias delta angle and time since last reset by an OF measurement arrival
674  delAngBodyOF += delAngCorrected;
675  delTimeOF += imuDataDelayed.delAngDT;
676 
677  // limit states to protect against divergence
678  ConstrainStates();

681 /*
682  * Propagate PVA solution forward from the fusion time horizon to the current time horizon
683  * using simple observer which performs two functions:
684  * 1) Corrects for the delayed time horizon used by the EKF.
685  * 2) Applies a LPF to state corrections to prevent 'stepping' in states due to measurement
686  * fusion introducing unwanted noise into the control loops.
687  * The inspiration for using a complementary filter to correct for time delays in the EKF
688  * is based on the work by A Khosravian.
689  *
690  * "Recursive Attitude Estimation in the Presence of Multi-rate and Multi-delay Vector Measurements"
691  * A Khosravian, J Trumpf, R Mahony, T Hamel, Australian National University
692 */
693 void NavEKF2_core::calcOutputStates()
694 {
695  // apply corrections to the IMU data
696  Vector3f delAngNewCorrected = imuDataNew.delAng;
697  Vector3f delVelNewCorrected = imuDataNew.delVel;
698  correctDeltaAngle(delAngNewCorrected, imuDataNew.delAngDT, imuDataNew.gyro_index);
699  correctDeltaVelocity(delVelNewCorrected, imuDataNew.delVelDT, imuDataNew.accel_index);
700 
701  // apply corections to track EKF solution
702  Vector3f delAng = delAngNewCorrected + delAngCorrection;
703 
704  // convert the rotation vector to its equivalent quaternion
705  Quaternion deltaQuat;
706  deltaQuat.from_axis_angle(delAng);
707 
708  // update the quaternion states by rotating from the previous attitude through
709  // the delta angle rotation quaternion and normalise
710  outputDataNew.quat *= deltaQuat;
711  outputDataNew.quat.normalize();
712 
713  // calculate the body to nav cosine matrix
714  Matrix3f Tbn_temp;
715  outputDataNew.quat.rotation_matrix(Tbn_temp);
716 
717  // transform body delta velocities to delta velocities in the nav frame
718  Vector3f delVelNav = Tbn_temp*delVelNewCorrected;
719  delVelNav.z += GRAVITY_MSS*imuDataNew.delVelDT;
720 
721  // save velocity for use in trapezoidal integration for position calcuation
722  Vector3f lastVelocity = outputDataNew.velocity;
723 
724  // sum delta velocities to get velocity
725  outputDataNew.velocity += delVelNav;
726 
727  // apply a trapezoidal integration to velocities to calculate position
728  outputDataNew.position += (outputDataNew.velocity + lastVelocity) * (imuDataNew.delVelDT*0.5f);
729 
730  // If the IMU accelerometer is offset from the body frame origin, then calculate corrections
731  // that can be added to the EKF velocity and position outputs so that they represent the velocity
732  // and position of the body frame origin.
733  // Note the * operator has been overloaded to operate as a dot product
734  if (!accelPosOffset.is_zero()) {
735  // calculate the average angular rate across the last IMU update
736  // note delAngDT is prevented from being zero in readIMUData()
737  Vector3f angRate = imuDataNew.delAng * (1.0f/imuDataNew.delAngDT);
738 
739  // Calculate the velocity of the body frame origin relative to the IMU in body frame
740  // and rotate into earth frame. Note % operator has been overloaded to perform a cross product
741  Vector3f velBodyRelIMU = angRate % (- accelPosOffset);
742  velOffsetNED = Tbn_temp * velBodyRelIMU;
743 
744  // calculate the earth frame position of the body frame origin relative to the IMU
745  posOffsetNED = Tbn_temp * (- accelPosOffset);
746  } else {
747  velOffsetNED.zero();
748  posOffsetNED.zero();
749  }
750 
751  // store INS states in a ring buffer that with the same length and time coordinates as the IMU data buffer
752  if (runUpdates) {
753  // store the states at the output time horizon
754  storedOutput[storedIMU.get_youngest_index()] = outputDataNew;
755 
756  // recall the states from the fusion time horizon
757  outputDataDelayed = storedOutput[storedIMU.get_oldest_index()];
758 
759  // compare quaternion data with EKF quaternion at the fusion time horizon and calculate correction
760 
761  // divide the demanded quaternion by the estimated to get the error
762  Quaternion quatErr = stateStruct.quat / outputDataDelayed.quat;
763 
764  // Convert to a delta rotation using a small angle approximation
765  quatErr.normalize();
766  Vector3f deltaAngErr;
767  float scaler;
768  if (quatErr[0] >= 0.0f) {
769  scaler = 2.0f;
770  } else {
771  scaler = -2.0f;
772  }
773  deltaAngErr.x = scaler * quatErr[1];
774  deltaAngErr.y = scaler * quatErr[2];
775  deltaAngErr.z = scaler * quatErr[3];
776 
777  // calculate a gain that provides tight tracking of the estimator states and
778  // adjust for changes in time delay to maintain consistent damping ratio of ~0.7
779  float timeDelay = 1e-3f * (float)(imuDataNew.time_ms - imuDataDelayed.time_ms);
780  timeDelay = fmaxf(timeDelay, dtIMUavg);
781  float errorGain = 0.5f / timeDelay;
782 
783  // calculate a corrrection to the delta angle
784  // that will cause the INS to track the EKF quaternions
785  delAngCorrection = deltaAngErr * errorGain * dtIMUavg;
786 
787  // calculate velocity and position tracking errors
788  Vector3f velErr = (stateStruct.velocity - outputDataDelayed.velocity);
789  Vector3f posErr = (stateStruct.position - outputDataDelayed.position);
790 
791  // collect magnitude tracking error for diagnostics
792  outputTrackError.x = deltaAngErr.length();
793  outputTrackError.y = velErr.length();
794  outputTrackError.z = posErr.length();
795 
796  // convert user specified time constant from centi-seconds to seconds
797  float tauPosVel = constrain_float(0.01f*(float)frontend->_tauVelPosOutput, 0.1f, 0.5f);
798 
799  // calculate a gain to track the EKF position states with the specified time constant
800  float velPosGain = dtEkfAvg / constrain_float(tauPosVel, dtEkfAvg, 10.0f);
801 
802  // use a PI feedback to calculate a correction that will be applied to the output state history
803  posErrintegral += posErr;
804  velErrintegral += velErr;
805  Vector3f velCorrection = velErr * velPosGain + velErrintegral * sq(velPosGain) * 0.1f;
806  Vector3f posCorrection = posErr * velPosGain + posErrintegral * sq(velPosGain) * 0.1f;
807 
808  // loop through the output filter state history and apply the corrections to the velocity and position states
809  // this method is too expensive to use for the attitude states due to the quaternion operations required
810  // but does not introduce a time delay in the 'correction loop' and allows smaller tracking time constants
811  // to be used
812  output_elements outputStates;
813  for (unsigned index=0; index < imu_buffer_length; index++) {
814  outputStates = storedOutput[index];
815 
816  // a constant velocity correction is applied
817  outputStates.velocity += velCorrection;
818 
819  // a constant position correction is applied
820  outputStates.position += posCorrection;
821 
822  // push the updated data to the buffer
823  storedOutput[index] = outputStates;
824  }
825 
826  // update output state to corrected values
827  outputDataNew = storedOutput[storedIMU.get_youngest_index()];
828 
829  }
