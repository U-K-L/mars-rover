159 /*
160  apply a reading in ADC counts
161  */
162 void AnalogSource::_add_value(float v, float vcc5V)
163 {
164  WITH_SEMAPHORE(_semaphore);
165 
166  _latest_value = v;
167  _sum_value += v;
168  if (vcc5V < 3.0f) {
169  _sum_ratiometric += v;
170  } else {
171  // this compensates for changes in the 5V rail relative to the
172  // 3.3V reference used by the ADC.
173  _sum_ratiometric += v * 5.0f / vcc5V;
174  }
175  _sum_count++;
176  if (_sum_count == 254) {
177  _sum_value /= 2;
178  _sum_ratiometric /= 2;
179  _sum_count /= 2;
180  }
181 }

184 /*
185  callback from ADC driver when sample buffer is filled
186  */
187 void AnalogIn::adccallback(ADCDriver *adcp)
188 {
189  const adcsample_t *buffer = samples;
190 
191  cacheBufferInvalidate(buffer, sizeof(adcsample_t)*ADC_DMA_BUF_DEPTH*ADC_GRP1_NUM_CHANNELS);
192  for (uint8_t i = 0; i < ADC_DMA_BUF_DEPTH; i++) {
193  for (uint8_t j = 0; j < ADC_GRP1_NUM_CHANNELS; j++) {
194  sample_sum[j] += *buffer++;
195  }
196  }
197  sample_count += ADC_DMA_BUF_DEPTH;
198 }

200 /*
201  setup adc peripheral to capture samples with DMA into a buffer
202  */
203 void AnalogIn::init()
204 {
205  if (ADC_GRP1_NUM_CHANNELS == 0) {
206  return;
207  }
208 
209  samples = (adcsample_t *)hal.util->malloc_type(sizeof(adcsample_t)*ADC_DMA_BUF_DEPTH*ADC_GRP1_NUM_CHANNELS, AP_HAL::Util::MEM_DMA_SAFE);
210 
211  adcStart(&ADCD1, NULL);
212  memset(&adcgrpcfg, 0, sizeof(adcgrpcfg));
213  adcgrpcfg.circular = true;
214  adcgrpcfg.num_channels = ADC_GRP1_NUM_CHANNELS;
215  adcgrpcfg.end_cb = adccallback;
216 #if defined(STM32H7)
217  // use 12 bits resolution to keep scaling factors the same as other boards.
218  // todo: enable oversampling in cfgr2 ?
219  adcgrpcfg.cfgr = ADC_CFGR_CONT | ADC_CFGR_RES_12BITS;
220 #else
221  adcgrpcfg.sqr1 = ADC_SQR1_NUM_CH(ADC_GRP1_NUM_CHANNELS);
222  adcgrpcfg.cr2 = ADC_CR2_SWSTART;
223 #endif
224 
225  for (uint8_t i=0; i<ADC_GRP1_NUM_CHANNELS; i++) {
226  uint8_t chan = pin_config[i].channel;
227  // setup cycles per sample for the channel
228 #if defined(STM32H7)
229  adcgrpcfg.pcsel |= (1<<chan);
230  adcgrpcfg.smpr[chan/10] |= ADC_SMPR_SMP_384P5 << (3*(chan%10));
231  if (i < 4) {
232  adcgrpcfg.sqr[0] |= chan << (6*(i+1));
233  } else if (i < 9) {
234  adcgrpcfg.sqr[1] |= chan << (6*(i-4));
235  } else {
236  adcgrpcfg.sqr[2] |= chan << (6*(i-9));
237  }
238 #else
239  if (chan < 10) {
240  adcgrpcfg.smpr2 |= ADC_SAMPLE_480 << (3*chan);
241  } else {
242  adcgrpcfg.smpr1 |= ADC_SAMPLE_480 << (3*(chan-10));
243  }
244  // setup channel sequence
245  if (i < 6) {
246  adcgrpcfg.sqr3 |= chan << (5*i);
247  } else if (i < 12) {
248  adcgrpcfg.sqr2 |= chan << (5*(i-6));
249  } else {
250  adcgrpcfg.sqr1 |= chan << (5*(i-12));
251  }
252 #endif
253  }
254  adcStartConversion(&ADCD1, &adcgrpcfg, samples, ADC_DMA_BUF_DEPTH);
255 }

257 /*
258  calculate average sample since last read for all channels
259  */
260 void AnalogIn::read_adc(uint32_t *val)
261 {
262  chSysLock();
263  for (uint8_t i = 0; i < ADC_GRP1_NUM_CHANNELS; i++) {
264  val[i] = sample_sum[i] / sample_count;
265  }
266  memset(sample_sum, 0, sizeof(sample_sum));
267  sample_count = 0;
268  chSysUnlock();

271 /*
272  called at 1kHz
273  */
274 void AnalogIn::_timer_tick(void)
275 {
276  // read adc at 100Hz
277  uint32_t now = AP_HAL::micros();
278  uint32_t delta_t = now - _last_run;
279  if (delta_t < 10000) {
280  return;
281  }
282  _last_run = now;
283 
284  uint32_t buf_adc[ADC_GRP1_NUM_CHANNELS];
285 
286  /* read all channels available */
287  read_adc(buf_adc);
288 
289  // update power status flags
290  update_power_flags();
291 
292  // match the incoming channels to the currently active pins
293  for (uint8_t i=0; i < ADC_GRP1_NUM_CHANNELS; i++) {
294 #ifdef ANALOG_VCC_5V_PIN
295  if (pin_config[i].channel == ANALOG_VCC_5V_PIN) {
296  // record the Vcc value for later use in
297  // voltage_average_ratiometric()
298  _board_voltage = buf_adc[i] * pin_config[i].scaling;
299  }
300 #endif
301 #ifdef FMU_SERVORAIL_ADC_CHAN
302  if (pin_config[i].channel == FMU_SERVORAIL_ADC_CHAN) {
303  _servorail_voltage = buf_adc[i] * pin_config[i].scaling;
304  }
305 #endif
306  }
307 
308 #if HAL_WITH_IO_MCU
309  // now handle special inputs from IOMCU
310  _servorail_voltage = iomcu.get_vservo();
311  _rssi_voltage = iomcu.get_vrssi();
312 #endif
313 
314  for (uint8_t i=0; i<ADC_GRP1_NUM_CHANNELS; i++) {
315  Debug("chan %u value=%u\n",
316  (unsigned)pin_config[i].channel,
317  (unsigned)buf_adc[i]);
318  for (uint8_t j=0; j < ANALOG_MAX_CHANNELS; j++) {
319  ChibiOS::AnalogSource *c = _channels[j];
320  if (c != nullptr) {
321  if (pin_config[i].channel == c->_pin) {
322  // add a value
323  c->_add_value(buf_adc[i], _board_voltage);
324  } else if (c->_pin == ANALOG_SERVO_VRSSI_PIN) {
325  c->_add_value(_rssi_voltage / VOLTAGE_SCALING, 0);
326  }
327  }
328  }
329  }
330 
331 #if CHIBIOS_ADC_MAVLINK_DEBUG
332  static uint8_t count;
333  if (AP_HAL::millis() > 5000 && count++ == 10) {
334  count = 0;
335  uint16_t adc[6] {};
336  uint8_t n = ADC_GRP1_NUM_CHANNELS;
337  if (n > 6) {
338  n = 6;
339  }
340  for (uint8_t i=0; i < n; i++) {
341  adc[i] = buf_adc[i];
342  }
343  mavlink_msg_ap_adc_send(MAVLINK_COMM_0, adc[0], adc[1], adc[2], adc[3], adc[4], adc[5]);
344  }
345 #endif
346 }
