154 void AP_Landing_Deepstall::do_land(const AP_Mission::Mission_Command& cmd, const float relative_altitude)
155 {
156  stage = DEEPSTALL_STAGE_FLY_TO_LANDING;
157  ds_PID.reset();
158  L1_xtrack_i = 0.0f;
159  hold_level = false; // come out of yaw lock
160 
161  // load the landing point in, the rest of path building is deferred for a better wind estimate
162  memcpy(&landing_point, &cmd.content.location, sizeof(Location));
163 
164  if (!landing_point.relative_alt && !landing_point.terrain_alt) {
165  approach_alt_offset = cmd.p1;
166  landing_point.alt += approach_alt_offset * 100;
167  } else {
168  approach_alt_offset = 0.0f;
169  }
170 }

185 bool AP_Landing_Deepstall::verify_land(const Location &prev_WP_loc, Location &next_WP_loc, const Location &current_loc,
186  const float height, const float sink_rate, const float wp_proportion, const uint32_t last_flying_ms,
187  const bool is_armed, const bool is_flying, const bool rangefinder_state_in_range)
188 {
189  switch (stage) {
190  case DEEPSTALL_STAGE_FLY_TO_LANDING:
191  if (current_loc.get_distance(landing_point) > abs(2 * landing.aparm.loiter_radius)) {
192  landing.nav_controller->update_waypoint(current_loc, landing_point);
193  return false;
194  }
195  stage = DEEPSTALL_STAGE_ESTIMATE_WIND;
196  loiter_sum_cd = 0; // reset the loiter counter
197  FALLTHROUGH;
198  case DEEPSTALL_STAGE_ESTIMATE_WIND:
199  {
200  landing.nav_controller->update_loiter(landing_point, landing.aparm.loiter_radius, landing_point.loiter_ccw ? -1 : 1);
201  if (!landing.nav_controller->reached_loiter_target() || (fabsf(height - approach_alt_offset) > DEEPSTALL_LOITER_ALT_TOLERANCE)) {
202  // wait until the altitude is correct before considering a breakout
203  return false;
204  }
205  // only count loiter progress when within the target altitude
206  int32_t target_bearing = landing.nav_controller->target_bearing_cd();
207  int32_t delta = wrap_180_cd(target_bearing - last_target_bearing);
208  delta *= (landing_point.loiter_ccw ? -1 : 1);
209  if (delta > 0) { // only accumulate turns in the correct direction
210  loiter_sum_cd += delta;
211  }
212  last_target_bearing = target_bearing;
213  if (loiter_sum_cd < 36000) {
214  // wait until we've done at least one complete loiter at the correct altitude
215  return false;
216  }
217  stage = DEEPSTALL_STAGE_WAIT_FOR_BREAKOUT;
218  loiter_sum_cd = 0; // reset the loiter counter
219  FALLTHROUGH;
220  }
221  case DEEPSTALL_STAGE_WAIT_FOR_BREAKOUT:
222  // rebuild the approach path if we have done less then a full circle to allow it to be
223  // more into the wind as the estimator continues to refine itself, and allow better
224  // compensation on windy days. This is limited to a single full circle though, as on
225  // a no wind day you could be in this loop forever otherwise.
226  if (loiter_sum_cd < 36000) {
227  build_approach_path(false);
228  }
229  if (!verify_breakout(current_loc, arc_entry, height - approach_alt_offset)) {
230  int32_t target_bearing = landing.nav_controller->target_bearing_cd();
231  int32_t delta = wrap_180_cd(target_bearing - last_target_bearing);
232  if (delta > 0) { // only accumulate turns in the correct direction
233  loiter_sum_cd += delta;
234  }
235  last_target_bearing = target_bearing;
236  landing.nav_controller->update_loiter(landing_point, landing.aparm.loiter_radius, landing_point.loiter_ccw ? -1 : 1);
237  return false;
238  }
239  stage = DEEPSTALL_STAGE_FLY_TO_ARC;
240  memcpy(&breakout_location, &current_loc, sizeof(Location));
241  FALLTHROUGH;
242  case DEEPSTALL_STAGE_FLY_TO_ARC:
243  if (current_loc.get_distance(arc_entry) > 2 * landing.aparm.loiter_radius) {
244  landing.nav_controller->update_waypoint(breakout_location, arc_entry);
245  return false;
246  }
247  stage = DEEPSTALL_STAGE_ARC;
248  FALLTHROUGH;
249  case DEEPSTALL_STAGE_ARC:
250  {
251  Vector2f groundspeed = landing.ahrs.groundspeed_vector();
252  if (!landing.nav_controller->reached_loiter_target() ||
253  (fabsf(wrap_180(target_heading_deg -
254  degrees(atan2f(-groundspeed.y, -groundspeed.x) + M_PI))) >= 10.0f)) {
255  landing.nav_controller->update_loiter(arc, landing.aparm.loiter_radius, landing_point.loiter_ccw ? -1 : 1);
256  return false;
257  }
258  stage = DEEPSTALL_STAGE_APPROACH;
259  }
260  FALLTHROUGH;
261  case DEEPSTALL_STAGE_APPROACH:
262  {
263  Location entry_point;
264  landing.nav_controller->update_waypoint(arc_exit, extended_approach);
265 
266  float height_above_target;
267  if (is_zero(approach_alt_offset)) {
268  landing.ahrs.get_relative_position_D_home(height_above_target);
269  height_above_target = -height_above_target;
270  } else {
271  Location position;
272  if (landing.ahrs.get_position(position)) {
273  height_above_target = (position.alt - landing_point.alt + approach_alt_offset * 100) * 1e-2f;
274  } else {
275  height_above_target = approach_alt_offset;
276  }
277  }
278 
279  const float travel_distance = predict_travel_distance(landing.ahrs.wind_estimate(), height_above_target, false);
280 
281  memcpy(&entry_point, &landing_point, sizeof(Location));
282  entry_point.offset_bearing(target_heading_deg + 180.0, travel_distance);
283 
284  if (!current_loc.past_interval_finish_line(arc_exit, entry_point)) {
285  if (current_loc.past_interval_finish_line(arc_exit, extended_approach)) {
286  // this should never happen, but prevent against an indefinite fly away
287  stage = DEEPSTALL_STAGE_FLY_TO_LANDING;
288  }
289  return false;
290  }
291  predict_travel_distance(landing.ahrs.wind_estimate(), height_above_target, true);
292  stage = DEEPSTALL_STAGE_LAND;
293  stall_entry_time = AP_HAL::millis();
294 
295  const SRV_Channel* elevator = SRV_Channels::get_channel_for(SRV_Channel::k_elevator);
296  if (elevator != nullptr) {
297  // take the last used elevator angle as the starting deflection
298  // don't worry about bailing here if the elevator channel can't be found
299  // that will be handled within override_servos
300  initial_elevator_pwm = elevator->get_output_pwm();
301  }
302  }
303  FALLTHROUGH;
304  case DEEPSTALL_STAGE_LAND:
305  // while in deepstall the only thing verify needs to keep the extended approach point sufficently far away
306  landing.nav_controller->update_waypoint(current_loc, extended_approach);
307  landing.disarm_if_autoland_complete_fn();
308  return false;
309  default:
310  return true;
311  }
312 }

314 bool AP_Landing_Deepstall::override_servos(void)
315 {
316  if (stage != DEEPSTALL_STAGE_LAND) {
317  return false;
318  }
319 
320  SRV_Channel* elevator = SRV_Channels::get_channel_for(SRV_Channel::k_elevator);
321 
322  if (elevator == nullptr) {
323  // deepstalls are impossible without these channels, abort the process
324  gcs().send_text(MAV_SEVERITY_CRITICAL, "Deepstall: Unable to find the elevator channels");
325  request_go_around();
326  return false;
327  }
328 
329  // calculate the progress on slewing the elevator
330  float slew_progress = 1.0f;
331  if (slew_speed > 0) {
332  slew_progress = (AP_HAL::millis() - stall_entry_time) / (100.0f * slew_speed);
333  }
334 
335  // mix the elevator to the correct value
336  elevator->set_output_pwm(linear_interpolate(initial_elevator_pwm, elevator_pwm,
337  slew_progress, 0.0f, 1.0f));
338 
339  // use the current airspeed to dictate the travel limits
340  float airspeed;
341  if (!landing.ahrs.airspeed_estimate(&airspeed)) {
342  airspeed = 0; // safely forces control to the deepstall steering since we don't have an estimate
343  }
344 
345 
346  // only allow the deepstall steering controller to run below the handoff airspeed
347  if (slew_progress >= 1.0f || airspeed <= handoff_airspeed) {
348  // run the steering conntroller
349  float pid = update_steering();
350 
351 
352  float travel_limit = constrain_float((handoff_airspeed - airspeed) /
353  (handoff_airspeed - handoff_lower_limit_airspeed) *
354  0.5f + 0.5f,
355  0.5f, 1.0f);
356 
357  float output = constrain_float(pid, -travel_limit, travel_limit);
358  SRV_Channels::set_output_scaled(SRV_Channel::k_aileron, output*4500*aileron_scalar);
359  SRV_Channels::set_output_scaled(SRV_Channel::k_rudder, output*4500);
360  SRV_Channels::set_output_scaled(SRV_Channel::k_throttle, 0); // this will normally be managed as part of landing,
361  // but termination needs to set throttle control here
362  }
363 
364  // hand off rudder control to deepstall controlled
365  return true;
366 }

480 void AP_Landing_Deepstall::build_approach_path(bool use_current_heading)
481 {
482  float loiter_radius = landing.nav_controller->loiter_radius(landing.aparm.loiter_radius);
483 
484  Vector3f wind = landing.ahrs.wind_estimate();
485  // TODO: Support a user defined approach heading
486  target_heading_deg = use_current_heading ? landing.ahrs.yaw_sensor * 1e-2 : (degrees(atan2f(-wind.y, -wind.x)));
487 
488  memcpy(&extended_approach, &landing_point, sizeof(Location));
489  memcpy(&arc_exit, &landing_point, sizeof(Location));
490 
491  //extend the approach point to 1km away so that there is always a navigational target
492  extended_approach.offset_bearing(target_heading_deg, 1000.0);
493 
494  float expected_travel_distance = predict_travel_distance(wind, is_zero(approach_alt_offset) ? landing_point.alt * 0.01f : approach_alt_offset,
495  false);
496  float approach_extension_m = expected_travel_distance + approach_extension;
497  float loiter_radius_m_abs = fabsf(loiter_radius);
498  // an approach extensions must be at least half the loiter radius, or the aircraft has a
499  // decent chance to be misaligned on final approach
500  approach_extension_m = MAX(approach_extension_m, loiter_radius_m_abs * 0.5f);
501 
502  arc_exit.offset_bearing(target_heading_deg + 180, approach_extension_m);
503  memcpy(&arc, &arc_exit, sizeof(Location));
504  memcpy(&arc_entry, &arc_exit, sizeof(Location));
505 
506  float arc_heading_deg = target_heading_deg + (landing_point.loiter_ccw ? -90.0f : 90.0f);
507  arc.offset_bearing(arc_heading_deg, loiter_radius_m_abs);
508  arc_entry.offset_bearing(arc_heading_deg, loiter_radius_m_abs * 2);
509 
510 #ifdef DEBUG_PRINTS
511  // TODO: Send this information via a MAVLink packet
512  gcs().send_text(MAV_SEVERITY_INFO, "Arc: %3.8f %3.8f",
513  (double)(arc.lat / 1e7),(double)( arc.lng / 1e7));
514  gcs().send_text(MAV_SEVERITY_INFO, "Loiter en: %3.8f %3.8f",
515  (double)(arc_entry.lat / 1e7), (double)(arc_entry.lng / 1e7));
516  gcs().send_text(MAV_SEVERITY_INFO, "Loiter ex: %3.8f %3.8f",
517  (double)(arc_exit.lat / 1e7), (double)(arc_exit.lng / 1e7));
518  gcs().send_text(MAV_SEVERITY_INFO, "Extended: %3.8f %3.8f",
519  (double)(extended_approach.lat / 1e7), (double)(extended_approach.lng / 1e7));
520  gcs().send_text(MAV_SEVERITY_INFO, "Extended by: %f (%f)", (double)approach_extension_m,
521  (double)expected_travel_distance);
522  gcs().send_text(MAV_SEVERITY_INFO, "Target Heading: %3.1f", (double)target_heading_deg);
523 #endif // DEBUG_PRINTS
524 
525 }

527 float AP_Landing_Deepstall::predict_travel_distance(const Vector3f wind, const float height, const bool print)
528 {
529  bool reverse = false;
530 
531  float course = radians(target_heading_deg);
532 
533  // a forward speed of 0 will result in a divide by 0
534  float forward_speed_ms = MAX(forward_speed, 0.1f);
535 
536  Vector2f wind_vec(wind.x, wind.y); // work with the 2D component of wind
537  float wind_length = MAX(wind_vec.length(), 0.05f); // always assume a slight wind to avoid divide by 0
538  Vector2f course_vec(cosf(course), sinf(course));
539 
540  float offset = course - atan2f(-wind.y, -wind.x);
541 
542  // estimator for how far the aircraft will travel while entering the stall
543  float stall_distance = slope_a * wind_length * cosf(offset) + slope_b;
544 
545  float theta = acosf(constrain_float((wind_vec * course_vec) / wind_length, -1.0f, 1.0f));
546  if ((course_vec % wind_vec) > 0) {
547  reverse = true;
548  theta *= -1;
549  }
550 
551  float cross_component = sinf(theta) * wind_length;
552  float estimated_crab_angle = asinf(constrain_float(cross_component / forward_speed_ms, -1.0f, 1.0f));
553  if (reverse) {
554  estimated_crab_angle *= -1;
555  }
556 
557  float estimated_forward = cosf(estimated_crab_angle) * forward_speed_ms + cosf(theta) * wind_length;
558 
559  if (is_positive(down_speed)) {
560  predicted_travel_distance = (estimated_forward * height / down_speed) + stall_distance;
561  } else {
562  // if we don't have a sane downward speed in a deepstall (IE not zero, and not
563  // an ascent) then just provide the stall_distance as a reasonable approximation
564  predicted_travel_distance = stall_distance;
565  }
566 
567  if(print) {
568  // allow printing the travel distances on the final entry as its used for tuning
569  gcs().send_text(MAV_SEVERITY_INFO, "Deepstall: Entry: %0.1f (m) Travel: %0.1f (m)",
570  (double)stall_distance, (double)predicted_travel_distance);
571  }
572 
573  return predicted_travel_distance;
574 }

577  const float height_error) const

579  const Vector2f location_delta = current_loc.get_distance_NE(target_loc);

587  return true;

592 float AP_Landing_Deepstall::update_steering()
593 {
594  Location current_loc;
595  if ((!landing.ahrs.get_position(current_loc) || !landing.ahrs.healthy()) && !hold_level) {
596  // panic if no position source is available
597  // continue the stall but target just holding the wings held level as deepstall should be a minimal
598  // energy configuration on the aircraft, and if a position isn't available aborting would be worse
599  gcs().send_text(MAV_SEVERITY_CRITICAL, "Deepstall: Invalid data from AHRS. Holding level");
600  hold_level = true;
601  }
602 
603  float desired_change = 0.0f;
604 
605  if (!hold_level) {
606  uint32_t time = AP_HAL::millis();
607  float dt = constrain_float(time - last_time, (uint32_t)10UL, (uint32_t)200UL) * 1e-3;
608  last_time = time;
609 
610  Vector2f ab = arc_exit.get_distance_NE(extended_approach);
611  ab.normalize();
612  const Vector2f a_air = arc_exit.get_distance_NE(current_loc);
613 
614  crosstrack_error = a_air % ab;
615  float sine_nu1 = constrain_float(crosstrack_error / MAX(L1_period, 0.1f), -0.7071f, 0.7107f);
616  float nu1 = asinf(sine_nu1);
617 
618  if (L1_i > 0) {
619  L1_xtrack_i += nu1 * L1_i / dt;
620  L1_xtrack_i = constrain_float(L1_xtrack_i, -0.5f, 0.5f);
621  nu1 += L1_xtrack_i;
622  }
623  desired_change = wrap_PI(radians(target_heading_deg) + nu1 - landing.ahrs.yaw) / time_constant;
624  }
625 
626  float yaw_rate = landing.ahrs.get_gyro().z;
627  float yaw_rate_limit_rps = radians(yaw_rate_limit);
628  float error = wrap_PI(constrain_float(desired_change, -yaw_rate_limit_rps, yaw_rate_limit_rps) - yaw_rate);
629 
630 #ifdef DEBUG_PRINTS
631  gcs().send_text(MAV_SEVERITY_INFO, "x: %f e: %f r: %f d: %f",
632  (double)crosstrack_error,
633  (double)error,
634  (double)degrees(yaw_rate),
635  (double)current_loc.get_distance(landing_point));
636 #endif // DEBUG_PRINTS
637 
638  return ds_PID.get_pid(error);
639 }
