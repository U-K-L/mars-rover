402 void AC_PosControl::get_stopping_point_z(Vector3f& stopping_point) const
403 {
404  const float curr_pos_z = _inav.get_altitude();
405  float curr_vel_z = _inav.get_velocity_z();
406 
407  float linear_distance; // half the distance we swap between linear and sqrt and the distance we offset sqrt
408  float linear_velocity; // the velocity we swap between linear and sqrt
409 
410  // if position controller is active add current velocity error to avoid sudden jump in acceleration
411  if (is_active_z()) {
412  curr_vel_z += _vel_error.z;
413  if (_flags.use_desvel_ff_z) {
414  curr_vel_z -= _vel_desired.z;
415  }
416  }
417 
418  // avoid divide by zero by using current position if kP is very low or acceleration is zero
419  if (_p_pos_z.kP() <= 0.0f || _accel_z_cms <= 0.0f) {
420  stopping_point.z = curr_pos_z;
421  return;
422  }
423 
424  // calculate the velocity at which we switch from calculating the stopping point using a linear function to a sqrt function
425  linear_velocity = _accel_z_cms / _p_pos_z.kP();
426 
427  if (fabsf(curr_vel_z) < linear_velocity) {
428  // if our current velocity is below the cross-over point we use a linear function
429  stopping_point.z = curr_pos_z + curr_vel_z / _p_pos_z.kP();
430  } else {
431  linear_distance = _accel_z_cms / (2.0f * _p_pos_z.kP() * _p_pos_z.kP());
432  if (curr_vel_z > 0) {
433  stopping_point.z = curr_pos_z + (linear_distance + curr_vel_z * curr_vel_z / (2.0f * _accel_z_cms));
434  } else {
435  stopping_point.z = curr_pos_z - (linear_distance + curr_vel_z * curr_vel_z / (2.0f * _accel_z_cms));
436  }
437  }
438  stopping_point.z = constrain_float(stopping_point.z, curr_pos_z - POSCONTROL_STOPPING_DIST_DOWN_MAX, curr_pos_z + POSCONTROL_STOPPING_DIST_UP_MAX);

697 void AC_PosControl::get_stopping_point_xy(Vector3f &stopping_point) const
698 {
699  const Vector3f curr_pos = _inav.get_position();
700  Vector3f curr_vel = _inav.get_velocity();
701  float linear_distance; // the distance at which we swap from a linear to sqrt response
702  float linear_velocity; // the velocity above which we swap from a linear to sqrt response
703  float stopping_dist; // the distance within the vehicle can stop
704  float kP = _p_pos_xy.kP();
705 
706  // add velocity error to current velocity
707  if (is_active_xy()) {
708  curr_vel.x += _vel_error.x;
709  curr_vel.y += _vel_error.y;
710  }
711 
712  // calculate current velocity
713  float vel_total = norm(curr_vel.x, curr_vel.y);
714 
715  // avoid divide by zero by using current position if the velocity is below 10cm/s, kP is very low or acceleration is zero
716  if (kP <= 0.0f || _accel_cms <= 0.0f || is_zero(vel_total)) {
717  stopping_point.x = curr_pos.x;
718  stopping_point.y = curr_pos.y;
719  return;
720  }
721 
722  // calculate point at which velocity switches from linear to sqrt
723  linear_velocity = _accel_cms / kP;
724 
725  // calculate distance within which we can stop
726  if (vel_total < linear_velocity) {
727  stopping_dist = vel_total / kP;
728  } else {
729  linear_distance = _accel_cms / (2.0f * kP * kP);
730  stopping_dist = linear_distance + (vel_total * vel_total) / (2.0f * _accel_cms);
731  }
732 
733  // constrain stopping distance
734  stopping_dist = constrain_float(stopping_dist, 0, _leash);
735 
736  // convert the stopping distance into a stopping point using velocity vector
737  stopping_point.x = curr_pos.x + (stopping_dist * curr_vel.x / vel_total);
738  stopping_point.y = curr_pos.y + (stopping_dist * curr_vel.y / vel_total);
739 }

1002 
1003  // avoid divide by zero
1004  if (kP <= 0.0f) {
1005  _vel_target.x = 0.0f;
1006  _vel_target.y = 0.0f;
1007  } else {
1008  // calculate distance error
1009  _pos_error.x = _pos_target.x - curr_pos.x;
1010  _pos_error.y = _pos_target.y - curr_pos.y;

1016  _pos_target.x = curr_pos.x + _pos_error.x;
1017  _pos_target.y = curr_pos.y + _pos_error.y;
1018  }
1019 
1020  _vel_target = sqrt_controller(_pos_error, kP, _accel_cms);
1021  }
1022 
1023  // add velocity feed-forward
1024  _vel_target.x += _vel_desired.x;
1025  _vel_target.y += _vel_desired.y;

1030 
1031  // check if vehicle velocity is being overridden

1033  _flags.vehicle_horiz_vel_override = false;

1046

1048  vel_xy_p = _pid_vel_xy.get_p();
1049

1051  // TODO: move limit handling into the PI and PID controller
1052  if (!_limit.accel_xy && !_motors.limit.throttle_upper) {
1053  vel_xy_i = _pid_vel_xy.get_i();

1059  vel_xy_d = _pid_vel_xy.get_d();

1063  accel_target.y = (vel_xy_p.y + vel_xy_i.y + vel_xy_d.y) * ekfNavVelGainScaler;

1066  if (_flags.reset_accel_to_lean_xy) {
1067  _accel_target_filter.reset(Vector2f(accel_target.x, accel_target.y));
1068  _flags.reset_accel_to_lean_xy = false;
1069  }

1072  _accel_target_filter.set_cutoff_frequency(MIN(_accel_xy_filt_hz, 5.0f * ekfNavVelGainScaler));
1073  _accel_target_filter.apply(accel_target, dt);

1076  _accel_target.x = _accel_target_filter.get().x;
1077  _accel_target.y = _accel_target_filter.get().y;

1084

1087  float accel_max = MIN(GRAVITY_MSS * 100.0f * tanf(ToRad(angle_max * 0.01f)), POSCONTROL_ACCEL_XY_MAX);
1088  _limit.accel_xy = limit_vector_length(_accel_target.x, _accel_target.y, accel_max);
1089

1091  accel_to_lean_angles(_accel_target.x, _accel_target.y, _roll_target, _pitch_target);
