33 /*
34  detect if a Lightware rangefinder is connected. We'll detect by
35  trying to take a reading on I2C. If we get a result the sensor is
36  there.
37 */
38 AP_RangeFinder_Backend *AP_RangeFinder_LightWareI2C::detect(RangeFinder::RangeFinder_State &_state, AP_RangeFinder_Params &_params, AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev)
39 {
40  if (!dev) {
41  return nullptr;
42  }
43 
44  AP_RangeFinder_LightWareI2C *sensor
45  = new AP_RangeFinder_LightWareI2C(_state, _params, std::move(dev));
46 
47  if (!sensor) {
48  delete sensor;
49  return nullptr;
50  }
51 
52  if (sensor->_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
53  uint16_t reading_cm;
54  if (!sensor->get_reading(reading_cm)) {
55  sensor->_dev->get_semaphore()->give();
56  delete sensor;
57  return nullptr;
58  }
59  sensor->_dev->get_semaphore()->give();
60  }
61 
62  sensor->init();
63 
64  return sensor;
65 }

74 // read - return last value measured by sensor
75 bool AP_RangeFinder_LightWareI2C::get_reading(uint16_t &reading_cm)
76 {
77  be16_t val;
78 
79  if (params.address == 0) {
80  return false;
81  }
82 
83  // read the high and low byte distance registers
84  bool ret = _dev->read((uint8_t *) &val, sizeof(val));
85  if (ret) {
86  // combine results into distance
87  reading_cm = be16toh(val);
88  }
89 
90  return ret;
91 }
