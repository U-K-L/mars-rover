38 static bool check_result(const char *name, bool bytes, const uint16_t *values, uint8_t nvalues)
39 {
40  char label[20];
41  snprintf(label, 20, "%s(%s)", name, bytes?"bytes":"pulses");
42  if (!rcprot->new_input()) {
43  printf("%s: No new input\n", label);
44  return false;
45  }
46  if (strcmp(rcprot->protocol_name(), name) != 0) {
47  printf("%s: wrong protocol detected %s\n", label, rcprot->protocol_name());
48  return false;
49  }
50  uint8_t n = rcprot->num_channels();
51  if (n != nvalues) {
52  printf("%s: wrong number of channels %u should be %u\n", label, n, nvalues);
53  return false;
54  }
55  for (uint8_t i=0; i<n; i++) {
56  uint16_t v = rcprot->read(i);
57  if (values[i] != v) {
58  printf("%s: chan %u wrong value %u should be %u\n", label, i+1, v, values[i]);
59  return false;
60  }
61  }
62  printf("%s OK\n", label);
63  return true;
64 }

87 static void send_bit(uint8_t bit, uint32_t baudrate)
88 {
89  static uint16_t bits_0, bits_1;
90  if (baudrate == 115200) {
91  // yes, this is backwards ...
92  bit = !bit;
93  }
94  if (bit == 0) {
95  if (bits_1 > 0) {
96  uint32_t w0=(bits_0 * (uint32_t)1000000) / baudrate;
97  uint32_t w1=(bits_1 * (uint32_t)1000000) / baudrate;
98  //printf("%u %u\n", w0, w1);
99  rcprot->process_pulse(w0, w1);
100  bits_0 = 1;
101  bits_1 = 0;
102  } else {
103  bits_0++;
104  }
105  } else {
106  bits_1++;
107  }
108 }

110 /*
111  call process_pulse() for a byte of input
112  */
113 static void send_byte(uint8_t b, uint32_t baudrate)
114 {
115  send_bit(0, baudrate); // start bit
116  uint8_t parity = 0;
117  for (uint8_t i=0; i<8; i++) {
118  uint8_t bit = (b & (1U<<i))?1:0;
119  send_bit(bit, baudrate);
120  if (bit) {
121  parity = !parity;
122  }
123  }
124  if (baudrate == 100000) {
125  // assume SBUS, send parity
126  send_bit(parity, baudrate);
127  }
128  send_bit(1, baudrate); // stop bit
129  if (baudrate == 100000) {
130  send_bit(1, baudrate); // 2nd stop bit
131  }
132 }
