89 void AP_IOMCU_FW::rcin_serial_update(void)
90 {
91  uint8_t b[16];
92  uint32_t n;
93 
94  // read from DSM port
95  if ((n = chnReadTimeout(&SD1, b, sizeof(b), TIME_IMMEDIATE)) > 0) {
96  n = MIN(n, sizeof(b));
97  rc_stats.num_dsm_bytes += n;
98  for (uint8_t i=0; i<n; i++) {
99  rcprotocol->process_byte(b[i], 115200);
100  }
101  //BLUE_TOGGLE();
102  }
103 
104  // read from SBUS port
105  if ((n = chnReadTimeout(&SD3, b, sizeof(b), TIME_IMMEDIATE)) > 0) {
106  eventflags_t flags;
107  if ((flags = chEvtGetAndClearFlags(&sd3_listener)) & SD_PARITY_ERROR) {
108  rc_stats.sbus_error = flags;
109  rc_stats.num_sbus_errors++;
110  } else {
111  n = MIN(n, sizeof(b));
112  rc_stats.num_sbus_bytes += n;
113  for (uint8_t i=0; i<n; i++) {
114  rcprotocol->process_byte(b[i], 100000);
115  }
116  }
117  }
118 }

129 /*
130  perform a DSM bind operation
131  */
132 void AP_IOMCU_FW::dsm_bind_step(void)
133 {
134  uint32_t now = last_ms;
135  switch (dsm_bind_state) {
136  case 1:
137  palSetLineMode(HAL_GPIO_PIN_SPEKTRUM_PWR_EN, PAL_MODE_OUTPUT_PUSHPULL);
138  SPEKTRUM_POWER(0);
139  palSetLineMode(HAL_GPIO_PIN_SPEKTRUM_OUT, PAL_MODE_OUTPUT_PUSHPULL);
140  SPEKTRUM_SET(1);
141  dsm_bind_state = 2;
142  last_dsm_bind_ms = now;
143  break;
144 
145  case 2:
146  if (now - last_dsm_bind_ms >= 500) {
147  SPEKTRUM_POWER(1);
148  dsm_bind_state = 3;
149  last_dsm_bind_ms = now;
150  }
151  break;
152 
153  case 3: {
154  if (now - last_dsm_bind_ms >= 72) {
155  // 9 pulses works with all satellite receivers, and supports the highest
156  // available protocol
157  delay_one_ms(now);
158  const uint8_t num_pulses = 9;
159  for (uint8_t i=0; i<num_pulses; i++) {
160  // the delay should be 120us, but we are running our
161  // clock at 1kHz, and 1ms works fine
162  delay_one_ms(now);
163  SPEKTRUM_SET(0);
164  delay_one_ms(now);
165  SPEKTRUM_SET(1);
166  }
167  last_dsm_bind_ms = now;
168  dsm_bind_state = 4;
169  }
170  break;
171  }
172 
173  case 4:
174  if (now - last_dsm_bind_ms >= 50) {
175  // set back as input pin
176  palSetLineMode(HAL_GPIO_PIN_SPEKTRUM_OUT, PAL_MODE_INPUT);
177  dsm_bind_state = 0;
178  }
179  break;
180 
181  default:
182  dsm_bind_state = 0;
183  break;
184  }
185 }
