167 void AP_MotorsHeli::init(motor_frame_class frame_class, motor_frame_type frame_type)
168 {
169  // remember frame class and type
170  _frame_type = frame_type;
171  _frame_class = frame_class;
172 
173  // set update rate
174  set_update_rate(_speed_hz);
175 
176  // load boot-up servo test cycles into counter to be consumed
177  _servo_test_cycle_counter = _servo_test;
178 
179  // ensure inputs are not passed through to servos on start-up
180  _servo_mode = SERVO_CONTROL_MODE_AUTOMATED;
181 
182  // initialise radio passthrough for collective to middle
183  _throttle_radio_passthrough = 0.5f;
184 
185  // initialise Servo/PWM ranges and endpoints
186  if (!init_outputs()) {
187  // don't set initialised_ok
188  return;
189  }
190 
191  // calculate all scalars
192  calculate_scalars();
193 
194  // record successful initialisation if what we setup was the desired frame_class
195  _flags.initialised_ok = (frame_class == MOTOR_FRAME_HELI);
196 
197  // set flag to true so targets are initialized once aircraft is armed for first time
198  _heliflags.init_targets_on_arming = true;
199 
200 }

269 // output_disarmed - sends commands to the motors
270 void AP_MotorsHeli::output_disarmed()
271 {
272  if (_servo_test_cycle_counter > 0){
273  // perform boot-up servo test cycle if enabled
274  servo_test();
275  } else {
276  // manual override (i.e. when setting up swash)
277  switch (_servo_mode) {
278  case SERVO_CONTROL_MODE_MANUAL_PASSTHROUGH:
279  // pass pilot commands straight through to swash
280  _roll_in = _roll_radio_passthrough;
281  _pitch_in = _pitch_radio_passthrough;
282  _throttle_filter.reset(_throttle_radio_passthrough);
283  _yaw_in = _yaw_radio_passthrough;
284  break;
285  case SERVO_CONTROL_MODE_MANUAL_CENTER:
286  // fixate mid collective
287  _roll_in = 0.0f;
288  _pitch_in = 0.0f;
289  _throttle_filter.reset(_collective_mid_pct);
290  _yaw_in = 0.0f;
291  break;
292  case SERVO_CONTROL_MODE_MANUAL_MAX:
293  // fixate max collective
294  _roll_in = 0.0f;
295  _pitch_in = 0.0f;
296  _throttle_filter.reset(1.0f);
297  if (_frame_class == MOTOR_FRAME_HELI_DUAL ||
298  _frame_class == MOTOR_FRAME_HELI_QUAD) {
299  _yaw_in = 0;
300  } else {
301  _yaw_in = 1;
302  }
303  break;
304  case SERVO_CONTROL_MODE_MANUAL_MIN:
305  // fixate min collective
306  _roll_in = 0.0f;
307  _pitch_in = 0.0f;
308  _throttle_filter.reset(0.0f);
309  if (_frame_class == MOTOR_FRAME_HELI_DUAL ||
310  _frame_class == MOTOR_FRAME_HELI_QUAD) {
311  _yaw_in = 0;
312  } else {
313  _yaw_in = -1;
314  }
315  break;
316  case SERVO_CONTROL_MODE_MANUAL_OSCILLATE:
317  // use servo_test function from child classes
318  servo_test();
319  break;
320  default:
321  // no manual override
322  break;
323  }
324  }
325 
326  // ensure swash servo endpoints haven't been moved
327  init_outputs();
328 
329  // continuously recalculate scalars to allow setup
330  calculate_scalars();
331 
332  // helicopters always run stabilizing flight controls
333  move_actuators(_roll_in, _pitch_in, get_throttle(), _yaw_in);
334 }

336 // run spool logic
337 void AP_MotorsHeli::output_logic()
338 {
339  // force desired and current spool mode if disarmed and armed with interlock enabled
340  if (_flags.armed) {
341  if (!_flags.interlock) {
342  _spool_desired = DesiredSpoolState::GROUND_IDLE;
343  } else {
344  _heliflags.init_targets_on_arming = false;
345  }
346  } else {
347  _heliflags.init_targets_on_arming = true;
348  _spool_desired = DesiredSpoolState::SHUT_DOWN;
349  _spool_state = SpoolState::SHUT_DOWN;
350  }
351 
352  switch (_spool_state) {
353  case SpoolState::SHUT_DOWN:
354  // Motors should be stationary.
355  // Servos set to their trim values or in a test condition.
356 
357  // make sure the motors are spooling in the correct direction
358  if (_spool_desired != DesiredSpoolState::SHUT_DOWN) {
359  _spool_state = SpoolState::GROUND_IDLE;
360  break;
361  }
362 
363  break;
364 
365  case SpoolState::GROUND_IDLE: {
366  // Motors should be stationary or at ground idle.
367  // Servos should be moving to correct the current attitude.
368  if (_spool_desired == DesiredSpoolState::SHUT_DOWN){
369  _spool_state = SpoolState::SHUT_DOWN;
370  } else if(_spool_desired == DesiredSpoolState::THROTTLE_UNLIMITED) {
371  _spool_state = SpoolState::SPOOLING_UP;
372  } else { // _spool_desired == GROUND_IDLE
373 
374  }
375 
376  break;
377  }
378  case SpoolState::SPOOLING_UP:
379  // Maximum throttle should move from minimum to maximum.
380  // Servos should exhibit normal flight behavior.
381 
382  // make sure the motors are spooling in the correct direction
383  if (_spool_desired != DesiredSpoolState::THROTTLE_UNLIMITED ){
384  _spool_state = SpoolState::SPOOLING_DOWN;
385  break;
386  }
387 
388  if (_heliflags.rotor_runup_complete){
389  _spool_state = SpoolState::THROTTLE_UNLIMITED;
390  }
391  break;
392 
393  case SpoolState::THROTTLE_UNLIMITED:
394  // Throttle should exhibit normal flight behavior.
395  // Servos should exhibit normal flight behavior.
396 
397  // make sure the motors are spooling in the correct direction
398  if (_spool_desired != DesiredSpoolState::THROTTLE_UNLIMITED) {
399  _spool_state = SpoolState::SPOOLING_DOWN;
400  break;
401  }
402 
403 
404  break;
405 
406  case SpoolState::SPOOLING_DOWN:
407  // Maximum throttle should move from maximum to minimum.
408  // Servos should exhibit normal flight behavior.
409 
410  // make sure the motors are spooling in the correct direction
411  if (_spool_desired == DesiredSpoolState::THROTTLE_UNLIMITED) {
412  _spool_state = SpoolState::SPOOLING_UP;
413  break;
414  }
415  if (!rotor_speed_above_critical()){
416  _spool_state = SpoolState::GROUND_IDLE;
417  }
418  break;
419  }
420 }

422 // parameter_check - check if helicopter specific parameters are sensible
423 bool AP_MotorsHeli::parameter_check(bool display_msg) const
424 {
425  // returns false if _rsc_setpoint is not higher than _rsc_critical as this would not allow rotor_runup_complete to ever return true
426  if (_rsc_critical >= _rsc_setpoint) {
427  if (display_msg) {
428  gcs().send_text(MAV_SEVERITY_CRITICAL, "PreArm: H_RSC_CRITICAL too large");
429  }
430  return false;
431  }
432 
433  // returns false if RSC Mode is not set to a valid control mode
434  if (_rsc_mode <= (int8_t)ROTOR_CONTROL_MODE_DISABLED || _rsc_mode > (int8_t)ROTOR_CONTROL_MODE_CLOSED_LOOP_POWER_OUTPUT) {
435  if (display_msg) {
436  gcs().send_text(MAV_SEVERITY_CRITICAL, "PreArm: H_RSC_MODE invalid");
437  }
438  return false;
439  }
440 
441  // returns false if RSC Runup Time is less than Ramp time as this could cause undesired behaviour of rotor speed estimate
442  if (_rsc_runup_time <= _rsc_ramp_time){
443  if (display_msg) {
444  gcs().send_text(MAV_SEVERITY_CRITICAL, "PreArm: H_RUNUP_TIME too small");
445  }
446  return false;
447  }
448 
449  // returns false if idle output is higher than critical rotor speed as this could block runup_complete from going false
450  if ( _rsc_idle_output >= _rsc_critical){
451  if (display_msg) {
452  gcs().send_text(MAV_SEVERITY_CRITICAL, "PreArm: H_RSC_IDLE too large");
453  }
454  return false;
455  }
456 
457  // all other cases parameters are OK
458  return true;
459 }
