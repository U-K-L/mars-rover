132 /*
133  process one byte of serial input for MSP protocol
134  */
135 bool AP_BLHeli::msp_process_byte(uint8_t c)
136 {
137  if (msp.state == MSP_IDLE) {
138  msp.escMode = PROTOCOL_NONE;
139  if (c == '$') {
140  msp.state = MSP_HEADER_START;
141  } else {
142  return false;
143  }
144  } else if (msp.state == MSP_HEADER_START) {
145  msp.state = (c == 'M') ? MSP_HEADER_M : MSP_IDLE;
146  } else if (msp.state == MSP_HEADER_M) {
147  msp.state = MSP_IDLE;
148  switch (c) {
149  case '<': // COMMAND
150  msp.packetType = MSP_PACKET_COMMAND;
151  msp.state = MSP_HEADER_ARROW;
152  break;
153  case '>': // REPLY
154  msp.packetType = MSP_PACKET_REPLY;
155  msp.state = MSP_HEADER_ARROW;
156  break;
157  default:
158  break;
159  }
160  } else if (msp.state == MSP_HEADER_ARROW) {
161  if (c > sizeof(msp.buf)) {
162  msp.state = MSP_IDLE;
163  } else {
164  msp.dataSize = c;
165  msp.offset = 0;
166  msp.checksum = 0;
167  msp.checksum ^= c;
168  msp.state = MSP_HEADER_SIZE;
169  }
170  } else if (msp.state == MSP_HEADER_SIZE) {
171  msp.cmdMSP = c;
172  msp.checksum ^= c;
173  msp.state = MSP_HEADER_CMD;
174  } else if (msp.state == MSP_HEADER_CMD && msp.offset < msp.dataSize) {
175  msp.checksum ^= c;
176  msp.buf[msp.offset++] = c;
177  } else if (msp.state == MSP_HEADER_CMD && msp.offset >= msp.dataSize) {
178  if (msp.checksum == c) {
179  msp.state = MSP_COMMAND_RECEIVED;
180  } else {
181  msp.state = MSP_IDLE;
182  }
183  }
184  return true;
185 }

195 /*
196  process one byte of serial input for blheli 4way protocol
197  */
198 bool AP_BLHeli::blheli_4way_process_byte(uint8_t c)
199 {
200  if (blheli.state == BLHELI_IDLE) {
201  if (c == cmd_Local_Escape) {
202  blheli.state = BLHELI_HEADER_START;
203  blheli.crc = 0;
204  blheli_crc_update(c);
205  } else {
206  return false;
207  }
208  } else if (blheli.state == BLHELI_HEADER_START) {
209  blheli.command = c;
210  blheli_crc_update(c);
211  blheli.state = BLHELI_HEADER_CMD;
212  } else if (blheli.state == BLHELI_HEADER_CMD) {
213  blheli.address = c<<8;
214  blheli.state = BLHELI_HEADER_ADDR_HIGH;
215  blheli_crc_update(c);
216  } else if (blheli.state == BLHELI_HEADER_ADDR_HIGH) {
217  blheli.address |= c;
218  blheli.state = BLHELI_HEADER_ADDR_LOW;
219  blheli_crc_update(c);
220  } else if (blheli.state == BLHELI_HEADER_ADDR_LOW) {
221  blheli.state = BLHELI_HEADER_LEN;
222  blheli.param_len = c?c:256;
223  blheli.offset = 0;
224  blheli_crc_update(c);
225  } else if (blheli.state == BLHELI_HEADER_LEN) {
226  blheli.buf[blheli.offset++] = c;
227  blheli_crc_update(c);
228  if (blheli.offset == blheli.param_len) {
229  blheli.state = BLHELI_CRC1;
230  }
231  } else if (blheli.state == BLHELI_CRC1) {
232  blheli.crc1 = c;
233  blheli.state = BLHELI_CRC2;
234  } else if (blheli.state == BLHELI_CRC2) {
235  uint16_t crc = blheli.crc1<<8 | c;
236  if (crc == blheli.crc) {
237  blheli.state = BLHELI_COMMAND_RECEIVED;
238  } else {
239  blheli.state = BLHELI_IDLE;
240  }
241  }
242  return true;
243 }

248 void AP_BLHeli::msp_send_reply(uint8_t cmd, const uint8_t *buf, uint8_t len)
249 {
250  uint8_t *b = &msp.buf[0];
251  *b++ = '$';
252  *b++ = 'M';
253  *b++ = '>';
254  *b++ = len;
255  *b++ = cmd;
256  memcpy(b, buf, len);
257  b += len;
258  uint8_t c = 0;
259  for (uint8_t i=0; i<len+2; i++) {
260  c ^= msp.buf[i+3];
261  }
262  *b++ = c;
263  uart->write_locked(&msp.buf[0], len+6, BLHELI_UART_LOCK_KEY);
264 }

292  process a MSP command from GCS
293  */
294 void AP_BLHeli::msp_process_command(void)
295 {
296  debug("MSP cmd %u len=%u", msp.cmdMSP, msp.dataSize);
297  switch (msp.cmdMSP) {
298  case MSP_API_VERSION: {
299  debug("MSP_API_VERSION");
300  uint8_t buf[3] = { MSP_PROTOCOL_VERSION, API_VERSION_MAJOR, API_VERSION_MINOR };
301  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
302  break;
303  }
304 
305  case MSP_FC_VARIANT:
306  debug("MSP_FC_VARIANT");
307  msp_send_reply(msp.cmdMSP, (const uint8_t *)ARDUPILOT_IDENTIFIER, FLIGHT_CONTROLLER_IDENTIFIER_LENGTH);
308  break;
309 
310  case MSP_FC_VERSION: {
311  debug("MSP_FC_VERSION");
312  uint8_t version[3] = { 3, 3, 0 };
313  msp_send_reply(msp.cmdMSP, version, sizeof(version));
314  break;
315  }
316  case MSP_BOARD_INFO: {
317  debug("MSP_BOARD_INFO");
318  // send a generic 'ArduPilot ChibiOS' board type
319  uint8_t buf[7] = { 'A', 'R', 'C', 'H', 0, 0, 0 };
320  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
321  break;
322  }
323 
324  case MSP_BUILD_INFO: {
325  debug("MSP_BUILD_INFO");
326  // build date, build time, git version
327  uint8_t buf[26] {
328  0x4d, 0x61, 0x72, 0x20, 0x31, 0x36, 0x20, 0x32, 0x30,
329  0x31, 0x38, 0x30, 0x38, 0x3A, 0x34, 0x32, 0x3a, 0x32, 0x39,
330  0x62, 0x30, 0x66, 0x66, 0x39, 0x32, 0x38};
331  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
332  break;
333  }
334 
335  case MSP_REBOOT:
336  debug("MSP: ignoring reboot command");
337  break;
338 
339  case MSP_UID:
340  // MCU identifer
341  debug("MSP_UID");
342  msp_send_reply(msp.cmdMSP, (const uint8_t *)UDID_START, 12);
343  break;
344 
345  case MSP_ADVANCED_CONFIG: {
346  debug("MSP_ADVANCED_CONFIG");
347  uint8_t buf[10];
348  buf[0] = 1; // gyro sync denom
349  buf[1] = 4; // pid process denom
350  buf[2] = 0; // use unsynced pwm
351  buf[3] = (uint8_t)PWM_TYPE_DSHOT150; // motor PWM protocol
352  putU16(&buf[4], 480); // motor PWM Rate
353  putU16(&buf[6], 450); // idle offset value
354  buf[8] = 0; // use 32kHz
355  buf[9] = 0; // motor PWM inversion
356  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
357  break;
358  }
359 
360  case MSP_FEATURE_CONFIG: {
361  debug("MSP_FEATURE_CONFIG");
362  uint8_t buf[4];
363  putU32(buf, 0); // from MSPFeatures enum
364  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
365  break;
366  }
367 
368  case MSP_STATUS: {
369  debug("MSP_STATUS");
370  uint8_t buf[21];
371  putU16(&buf[0], 1000); // loop time usec
372  putU16(&buf[2], 0); // i2c error count
373  putU16(&buf[4], 0x27); // available sensors
374  putU32(&buf[6], 0); // flight modes
375  buf[10] = 0; // pid profile index
376  putU16(&buf[11], 5); // system load percent
377  putU16(&buf[13], 0); // gyro cycle time
378  buf[15] = 0; // flight mode flags length
379  buf[16] = 18; // arming disable flags count
380  putU32(&buf[17], 0); // arming disable flags
381  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
382  break;
383  }
384 
385  case MSP_MOTOR_3D_CONFIG: {
386  debug("MSP_MOTOR_3D_CONFIG");
387  uint8_t buf[6];
388  putU16(&buf[0], 1406); // 3D deadband low
389  putU16(&buf[2], 1514); // 3D deadband high
390  putU16(&buf[4], 1460); // 3D neutral
391  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
392  break;
393  }
394 
395  case MSP_MOTOR_CONFIG: {
396  debug("MSP_MOTOR_CONFIG");
397  uint8_t buf[6];
398  putU16(&buf[0], 1030); // min throttle
399  putU16(&buf[2], 2000); // max throttle
400  putU16(&buf[4], 1000); // min command
401  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
402  break;
403  }
404 
405  case MSP_MOTOR: {
406  debug("MSP_MOTOR");
407  // get the output going to each motor
408  uint8_t buf[16] {};
409  for (uint8_t i = 0; i < num_motors; i++) {
410  uint16_t v = hal.rcout->read(motor_map[i]);
411  putU16(&buf[2*i], v);
412  }
413  msp_send_reply(msp.cmdMSP, buf, sizeof(buf));
414  break;
415  }
416 
417  case MSP_SET_MOTOR: {
418  debug("MSP_SET_MOTOR");
419  // set the output to each motor
420  uint8_t nmotors = msp.dataSize / 2;
421  debug("MSP_SET_MOTOR %u", nmotors);
422  SRV_Channels::set_disabled_channel_mask(0xFFFF);
423  motors_disabled = true;
424  hal.rcout->cork();
425  for (uint8_t i = 0; i < nmotors; i++) {
426  if (i >= num_motors) {
427  break;
428  }
429  uint16_t v = getU16(&msp.buf[i*2]);
430  debug("MSP_SET_MOTOR %u %u", i, v);
431  // map from a MSP value to a value in the range 1000 to 2000
432  uint16_t pwm = (v < 1000)?0:v;
433  hal.rcout->write(motor_map[i], pwm);
434  }
435  hal.rcout->push();
436  break;
437  }
438 
439  case MSP_SET_4WAY_IF: {
440  debug("MSP_SET_4WAY_IF");
441  if (msp.dataSize == 0) {
442  msp.escMode = PROTOCOL_4WAY;
443  } else if (msp.dataSize == 2) {
444  msp.escMode = (enum escProtocol)msp.buf[0];
445  msp.portIndex = msp.buf[1];
446  }
447  debug("escMode=%u portIndex=%u num_motors=%u", msp.escMode, msp.portIndex, num_motors);
448  uint8_t n = num_motors;
449  switch (msp.escMode) {
450  case PROTOCOL_4WAY:
451  break;
452  default:
453  n = 0;
454  hal.rcout->serial_end();
455  serial_start_ms = 0;
456  break;
457  }
458  msp_send_reply(msp.cmdMSP, &n, 1);
459  break;
460  }
461  default:
462  debug("Unknown MSP command %u", msp.cmdMSP);
463  break;
464  }
465 }

471 {
472  uint8_t *b = &blheli.buf[0];
473  *b++ = cmd_Remote_Escape;
474  *b++ = blheli.command;

476  *b++ = len==256?0:len;

479  *b++ = blheli.ack;

482  debug("OutB(%u) 0x%02x ack=0x%02x", len+8, (unsigned)blheli.command, blheli.ack);

485 /*
486  CRC used when talking to ESCs
487  */
488 uint16_t AP_BLHeli::BL_CRC(const uint8_t *buf, uint16_t len)
489 {
490  uint16_t crc = 0;
491  while (len--) {
492  uint8_t xb = *buf++;
493  for (uint8_t i = 0; i < 8; i++) {
494  if (((xb & 0x01) ^ (crc & 0x0001)) !=0 ) {
495  crc = crc >> 1;
496  crc = crc ^ 0xA001;
497  } else {
498  crc = crc >> 1;
499  }
500  xb = xb >> 1;
501  }
502  }
503  return crc;
504 }

518 /*
519  send a set of bytes to an RC output channel
520  */
521 bool AP_BLHeli::BL_SendBuf(const uint8_t *buf, uint16_t len)
522 {
523  bool send_crc = isMcuConnected();
524  if (blheli.chan >= num_motors) {
525  return false;
526  }
527  if (!hal.rcout->serial_setup_output(motor_map[blheli.chan], 19200, motor_mask)) {
528  blheli.ack = ACK_D_GENERAL_ERROR;
529  return false;
530  }
531  if (serial_start_ms == 0) {
532  serial_start_ms = AP_HAL::millis();
533  }
534  uint32_t now = AP_HAL::millis();
535  if (serial_start_ms == 0 || now - serial_start_ms < 1000) {
536  /*
537  we've just started the interface. We want it idle for at
538  least 1 second before we start sending serial data.
539  */
540  hal.scheduler->delay(1100);
541  }
542  memcpy(blheli.buf, buf, len);
543  uint16_t crc = BL_CRC(buf, len);
544  blheli.buf[len] = crc;
545  blheli.buf[len+1] = crc>>8;
546  if (!hal.rcout->serial_write_bytes(blheli.buf, len+(send_crc?2:0))) {
547  blheli.ack = ACK_D_GENERAL_ERROR;
548  return false;
549  }
550  return true;
551 }

553 /*
554  read bytes from the ESC connection
555  */
556 bool AP_BLHeli::BL_ReadBuf(uint8_t *buf, uint16_t len)
557 {
558  bool check_crc = isMcuConnected() && len > 0;
559  uint16_t req_bytes = len+(check_crc?3:1);
560  uint16_t n = hal.rcout->serial_read_bytes(blheli.buf, req_bytes);
561  debug("BL_ReadBuf %u -> %u", len, n);
562  if (req_bytes != n) {
563  debug("short read");
564  blheli.ack = ACK_D_GENERAL_ERROR;
565  return false;
566  }
567  if (check_crc) {
568  uint16_t crc = BL_CRC(blheli.buf, len);
569  if ((crc & 0xff) != blheli.buf[len] ||
570  (crc >> 8) != blheli.buf[len+1]) {
571  debug("bad CRC");
572  blheli.ack = ACK_D_GENERAL_ERROR;
573  return false;
574  }
575  if (blheli.buf[len+2] != brSUCCESS) {
576  debug("bad ACK 0x%02x", blheli.buf[len+2]);
577  blheli.ack = ACK_D_GENERAL_ERROR;
578  return false;
579  }
580  } else {
581  if (blheli.buf[len] != brSUCCESS) {
582  debug("bad ACK1 0x%02x", blheli.buf[len]);
583  blheli.ack = ACK_D_GENERAL_ERROR;
584  return false;
585  }
586  }
587  if (len > 0) {
588  memcpy(buf, blheli.buf, len);
589  }
590  return true;
591 }

594 {

649 bool AP_BLHeli::BL_ConnectEx(void)
650 {

652  debug("Using cached interface 0x%x for %u", blheli.interface_mode[blheli.chan], blheli.chan);
653  return true;

655  debug("BL_ConnectEx %u/%u at %u", blheli.chan, num_motors, motor_map[blheli.chan]);

658  if (!BL_SendBuf(BootInit, 21)) {
659  return false;
660  }
661

663  if (!BL_ReadBuf(BootInfo, 8)) {
664  return false;
665  }

669  blheli.ack = ACK_D_GENERAL_ERROR;

675  blheli.deviceInfo[blheli.chan][1] = BootInfo[4];
676  blheli.deviceInfo[blheli.chan][0] = BootInfo[5];

682  case 0x9307:
683  case 0x930A:
684  case 0x930F:
685  case 0x940B:
686  blheli.interface_mode[blheli.chan] = imATM_BLB;
687  debug("Interface type imATM_BLB");

689  case 0xF310:
690  case 0xF330:
691  case 0xF410:
692  case 0xF390:
693  case 0xF850:
694  case 0xE8B1:
695  case 0xE8B2:
696  blheli.interface_mode[blheli.chan] = imSIL_BLB;
697  debug("Interface type imSIL_BLB");
698  break;
699  case 0x1F06:
700  case 0x3306:
701  case 0x3406:
702  case 0x3506:
703  case 0x2B06:
704  case 0x4706:
705  blheli.interface_mode[blheli.chan] = imARM_BLB;
706  debug("Interface type imARM_BLB");
707  break;
708  default:
709  blheli.ack = ACK_D_GENERAL_ERROR;
710  debug("Unknown interface type 0x%04x", *devword);
711  break;
712  }
713  blheli.deviceInfo[blheli.chan][3] = blheli.interface_mode[blheli.chan];

820 /*
821  process a blheli 4way command from GCS
822  */
823 void AP_BLHeli::blheli_process_command(void)
824 {
825  debug("BLHeli cmd 0x%02x len=%u", blheli.command, blheli.param_len);
826  blheli.ack = ACK_OK;
827  switch (blheli.command) {
828  case cmd_InterfaceTestAlive: {
829  debug("cmd_InterfaceTestAlive");
830  BL_SendCMDKeepAlive();
831  if (blheli.ack != ACK_OK) {
832  setDisconnected();
833  }
834  uint8_t b = 0;
835  blheli_send_reply(&b, 1);
836  break;
837  }
838  case cmd_ProtocolGetVersion: {
839  debug("cmd_ProtocolGetVersion");
840  uint8_t buf[1];
841  buf[0] = SERIAL_4WAY_PROTOCOL_VER;
842  blheli_send_reply(buf, sizeof(buf));
843  break;
844  }
845  case cmd_InterfaceGetName: {
846  debug("cmd_InterfaceGetName");
847  uint8_t buf[5] = { 4, 'A', 'R', 'D', 'U' };
848  blheli_send_reply(buf, sizeof(buf));
849  break;
850  }
851  case cmd_InterfaceGetVersion: {
852  debug("cmd_InterfaceGetVersion");
853  uint8_t buf[2] = { SERIAL_4WAY_VERSION_HI, SERIAL_4WAY_VERSION_LO };
854  blheli_send_reply(buf, sizeof(buf));
855  break;
856  }
857  case cmd_InterfaceExit: {
858  debug("cmd_InterfaceExit");
859  msp.escMode = PROTOCOL_NONE;
860  uint8_t b = 0;
861  blheli_send_reply(&b, 1);
862  hal.rcout->serial_end();
863  serial_start_ms = 0;
864  if (motors_disabled) {
865  motors_disabled = false;
866  SRV_Channels::set_disabled_channel_mask(0);
867  }
868  if (uart_locked) {
869  debug("Unlocked UART");
870  uart->lock_port(0, 0);
871  uart_locked = false;
872  }
873  memset(blheli.connected, 0, sizeof(blheli.connected));
874  break;
875  }
876  case cmd_DeviceReset: {
877  debug("cmd_DeviceReset(%u)", unsigned(blheli.buf[0]));
878  if (blheli.buf[0] >= num_motors) {
879  debug("bad reset channel %u", blheli.buf[0]);
880  blheli.ack = ACK_D_GENERAL_ERROR;
881  blheli_send_reply(&blheli.buf[0], 1);
882  break;
883  }
884  blheli.chan = blheli.buf[0];
885  switch (blheli.interface_mode[blheli.chan]) {
886  case imSIL_BLB:
887  case imATM_BLB:
888  case imARM_BLB:
889  BL_SendCMDRunRestartBootloader();
890  break;
891  case imSK:
892  break;
893  }
894  blheli_send_reply(&blheli.chan, 1);
895  setDisconnected();
896  break;
897  }
898 
899  case cmd_DeviceInitFlash: {
900  debug("cmd_DeviceInitFlash(%u)", unsigned(blheli.buf[0]));
901  if (blheli.buf[0] >= num_motors) {
902  debug("bad channel %u", blheli.buf[0]);
903  break;
904  }
905  blheli.chan = blheli.buf[0];
906  blheli.ack = ACK_OK;
907  BL_ConnectEx();
908  uint8_t buf[4] = {blheli.deviceInfo[blheli.chan][0],
909  blheli.deviceInfo[blheli.chan][1],
910  blheli.deviceInfo[blheli.chan][2],
911  blheli.deviceInfo[blheli.chan][3]}; // device ID
912  blheli_send_reply(buf, sizeof(buf));
913  break;
914  }
915 
916  case cmd_InterfaceSetMode: {
917  debug("cmd_InterfaceSetMode(%u)", unsigned(blheli.buf[0]));
918  blheli.interface_mode[blheli.chan] = blheli.buf[0];
919  blheli_send_reply(&blheli.interface_mode[blheli.chan], 1);
920  break;
921  }
922 
923  case cmd_DeviceRead: {
924  uint16_t nbytes = blheli.buf[0]?blheli.buf[0]:256;
925  debug("cmd_DeviceRead(%u) n=%u", blheli.chan, nbytes);
926  uint8_t buf[nbytes];
927  uint8_t cmd = blheli.interface_mode[blheli.chan]==imATM_BLB?CMD_READ_FLASH_ATM:CMD_READ_FLASH_SIL;
928  if (!BL_ReadA(cmd, buf, nbytes)) {
929  nbytes = 1;
930  }
931  blheli_send_reply(buf, nbytes);
932  break;
933  }
934 
935  case cmd_DevicePageErase: {
936  uint8_t page = blheli.buf[0];
937  debug("cmd_DevicePageErase(%u) im=%u", page, blheli.interface_mode[blheli.chan]);
938  switch (blheli.interface_mode[blheli.chan]) {
939  case imSIL_BLB:
940  case imARM_BLB: {
941  if (blheli.interface_mode[blheli.chan] == imARM_BLB) {
942  // Address =Page * 1024
943  blheli.address = page << 10;
944  } else {
945  // Address =Page * 512
946  blheli.address = page << 9;
947  }
948  debug("ARM PageErase 0x%04x", blheli.address);
949  BL_PageErase();
950  blheli.address = 0;
951  blheli_send_reply(&page, 1);
952  break;
953  }
954  default:
955  blheli.ack = ACK_I_INVALID_CMD;
956  blheli_send_reply(&page, 1);
957  break;
958  }
959  break;
960  }
961 
962  case cmd_DeviceWrite: {
963  uint16_t nbytes = blheli.param_len;
964  debug("cmd_DeviceWrite n=%u im=%u", nbytes, blheli.interface_mode[blheli.chan]);
965  uint8_t buf[nbytes];
966  memcpy(buf, blheli.buf, nbytes);
967  switch (blheli.interface_mode[blheli.chan]) {
968  case imSIL_BLB:
969  case imATM_BLB:
970  case imARM_BLB: {
971  BL_WriteFlash(buf, nbytes);
972  break;
973  }
974  case imSK: {
975  debug("Unsupported flash mode imSK");
976  break;
977  }
978  }
979  uint8_t b=0;
980  blheli_send_reply(&b, 1);
981  break;
982  }
983 
984  case cmd_DeviceVerify: {
985  uint16_t nbytes = blheli.param_len;
986  debug("cmd_DeviceWrite n=%u im=%u", nbytes, blheli.interface_mode[blheli.chan]);
987  switch (blheli.interface_mode[blheli.chan]) {
988  case imARM_BLB: {
989  uint8_t buf[nbytes];
990  memcpy(buf, blheli.buf, nbytes);
991  BL_VerifyFlash(buf, nbytes);
992  break;
993  }
994  default:
995  blheli.ack = ACK_I_INVALID_CMD;
996  break;
997  }
998  uint8_t b=0;
999  blheli_send_reply(&b, 1);
1000  break;
1001  }
1002 
1003  case cmd_DeviceReadEEprom: {
1004  uint16_t nbytes = blheli.buf[0]?blheli.buf[0]:256;
1005  uint8_t buf[nbytes];
1006  debug("cmd_DeviceReadEEprom n=%u im=%u", nbytes, blheli.interface_mode[blheli.chan]);
1007  switch (blheli.interface_mode[blheli.chan]) {
1008  case imATM_BLB: {
1009  if (!BL_ReadA(CMD_READ_EEPROM, buf, nbytes)) {
1010  blheli.ack = ACK_D_GENERAL_ERROR;
1011  }
1012  break;
1013  }
1014  default:
1015  blheli.ack = ACK_I_INVALID_CMD;
1016  break;
1017  }
1018  if (blheli.ack != ACK_OK) {
1019  nbytes = 1;
1020  buf[0] = 0;
1021  }
1022  blheli_send_reply(buf, nbytes);
1023  break;
1024  }
1025 
1026  case cmd_DeviceWriteEEprom: {
1027  uint16_t nbytes = blheli.param_len;
1028  uint8_t buf[nbytes];
1029  memcpy(buf, blheli.buf, nbytes);
1030  debug("cmd_DeviceWriteEEprom n=%u im=%u", nbytes, blheli.interface_mode[blheli.chan]);
1031  switch (blheli.interface_mode[blheli.chan]) {
1032  case imATM_BLB:
1033  BL_WriteA(CMD_PROG_EEPROM, buf, nbytes, 1000);
1034  break;
1035  default:
1036  blheli.ack = ACK_D_GENERAL_ERROR;
1037  break;
1038  }
1039  uint8_t b = 0;
1040  blheli_send_reply(&b, 1);
1041  break;
1042  }
1043 
1044  case cmd_DeviceEraseAll:
1045  case cmd_DeviceC2CK_LOW:
1046  default:
1047  // ack=unknown command
1048  blheli.ack = ACK_I_INVALID_CMD;
1049  debug("Unknown BLHeli protocol 0x%02x", blheli.command);
1050  uint8_t b = 0;
1051  blheli_send_reply(&b, 1);
1052  break;
1053  }
1054 }

1056 /*
1057  process an input byte, return true if we have received a whole
1058  packet with correct CRC
1059  */
1060 bool AP_BLHeli::process_input(uint8_t b)
1061 {
1062  bool valid_packet = false;
1063 
1064  if (msp.escMode == PROTOCOL_4WAY && blheli.state == BLHELI_IDLE && b == '$') {
1065  debug("Change to MSP mode");
1066  msp.escMode = PROTOCOL_NONE;
1067  hal.rcout->serial_end();
1068  serial_start_ms = 0;
1069  }
1070  if (msp.escMode != PROTOCOL_4WAY && msp.state == MSP_IDLE && b == '/') {
1071  debug("Change to BLHeli mode");
1072  memset(blheli.connected, 0, sizeof(blheli.connected));
1073  msp.escMode = PROTOCOL_4WAY;
1074  }
1075  if (msp.escMode == PROTOCOL_4WAY) {
1076  blheli_4way_process_byte(b);
1077  } else {
1078  msp_process_byte(b);
1079  }
1080  if (msp.escMode == PROTOCOL_4WAY) {
1081  if (blheli.state == BLHELI_COMMAND_RECEIVED) {
1082  valid_packet = true;
1083  last_valid_ms = AP_HAL::millis();
1084  if (uart->lock_port(BLHELI_UART_LOCK_KEY, 0)) {
1085  uart_locked = true;
1086  }
1087  blheli_process_command();
1088  blheli.state = BLHELI_IDLE;
1089  msp.state = MSP_IDLE;
1090  }
1091  } else if (msp.state == MSP_COMMAND_RECEIVED) {
1092  if (msp.packetType == MSP_PACKET_COMMAND) {
1093  valid_packet = true;
1094  if (uart->lock_port(BLHELI_UART_LOCK_KEY, 0)) {
1095  uart_locked = true;
1096  }
1097  last_valid_ms = AP_HAL::millis();
1098  msp_process_command();
1099  }
1100  msp.state = MSP_IDLE;
1101  blheli.state = BLHELI_IDLE;
1102  }
1103 
1104  return valid_packet;
1105 }
