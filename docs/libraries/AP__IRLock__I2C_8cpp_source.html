52 /*
53  synchronise with frame start. We expect 0xAA55AA55 at the start of
54  a frame
55 */
56 bool AP_IRLock_I2C::sync_frame_start(void)
57 {
58  uint32_t sync_word;
59  if (!dev->transfer(nullptr, 0, (uint8_t *)&sync_word, 4)) {
60  return false;
61  }
62 
63  // record sensor successfully responded to I2C request
64  _last_read_ms = AP_HAL::millis();
65 
66  uint8_t count=40;
67  while (count-- && sync_word != IRLOCK_SYNC && sync_word != 0) {
68  uint8_t sync_byte;
69  if (!dev->transfer(nullptr, 0, &sync_byte, 1)) {
70  return false;
71  }
72  if (sync_byte == 0) {
73  break;
74  }
75  sync_word = (sync_word>>8) | (uint32_t(sync_byte)<<24);
76  }
77  return sync_word == IRLOCK_SYNC;
78 }

95 bool AP_IRLock_I2C::read_block(struct frame &irframe)
96 {
97  if (!dev->transfer(nullptr, 0, (uint8_t*)&irframe, sizeof(irframe))) {
98  return false;
99  }
100 
101  // record sensor successfully responded to I2C request
102  _last_read_ms = AP_HAL::millis();
103 
104  /* check crc */
105  uint32_t crc = irframe.signature + irframe.pixel_x + irframe.pixel_y + irframe.pixel_size_x + irframe.pixel_size_y;
106  if (crc != irframe.checksum) {
107  // printf("bad crc 0x%04x 0x%04x\n", crc, irframe.checksum);
108  return false;
109  }
110  return true;
111 }

113 void AP_IRLock_I2C::read_frames(void)
114 {
115  if (!sync_frame_start()) {
116  return;
117  }
118  struct frame irframe;
119 
120  if (!read_block(irframe)) {
121  return;
122  }
123 
124  int16_t corner1_pix_x = irframe.pixel_x - irframe.pixel_size_x/2;
125  int16_t corner1_pix_y = irframe.pixel_y - irframe.pixel_size_y/2;
126  int16_t corner2_pix_x = irframe.pixel_x + irframe.pixel_size_x/2;
127  int16_t corner2_pix_y = irframe.pixel_y + irframe.pixel_size_y/2;
128 
129  float corner1_pos_x, corner1_pos_y, corner2_pos_x, corner2_pos_y;
130  pixel_to_1M_plane(corner1_pix_x, corner1_pix_y, corner1_pos_x, corner1_pos_y);
131  pixel_to_1M_plane(corner2_pix_x, corner2_pix_y, corner2_pos_x, corner2_pos_y);
132 
133  {
134  WITH_SEMAPHORE(sem);
135 
136  /* convert to angles */
137  _target_info.timestamp = AP_HAL::millis();
138  _target_info.pos_x = 0.5f*(corner1_pos_x+corner2_pos_x);
139  _target_info.pos_y = 0.5f*(corner1_pos_y+corner2_pos_y);
140  _target_info.pos_z = 1.0f;
141  }
142 
143 #if 0
