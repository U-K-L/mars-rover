39 bool Buzzer::init()
40 {
41  if (pNotify->buzzer_enabled() == false) {
42  return false;
43  }
44 #if defined(HAL_BUZZER_PIN)
45  _pin = HAL_BUZZER_PIN;
46 #else
47  _pin = pNotify->get_buzz_pin();
48 #endif
49  if(!_pin) return false;
50 
51  // setup the pin and ensure it's off
52  hal.gpio->pinMode(_pin, HAL_GPIO_OUTPUT);
53  on(false);
54 
55  // set initial boot states. This prevents us issuing a arming
56  // warning in plane and rover on every boot
57  _flags.armed = AP_Notify::flags.armed;
58  _flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
59  return true;
60 }

69 void Buzzer::update_pattern_to_play()
70 {
71  // check for arming failed event
72  if (AP_Notify::events.arming_failed) {
73  // arming failed buzz
74  play_pattern(SINGLE_BUZZ);
75  return;
76  }
77 
78  if (AP_HAL::millis() - _pattern_start_time < _pattern_start_interval_time_ms) {
79  // do not interrupt playing patterns / enforce minumum separation
80  return;
81  }
82 
83  // check if armed status has changed
84  if (_flags.armed != AP_Notify::flags.armed) {
85  _flags.armed = AP_Notify::flags.armed;
86  if (_flags.armed) {
87  // double buzz when armed
88  play_pattern(ARMING_BUZZ);
89  }else{
90  // single buzz when disarmed
91  play_pattern(SINGLE_BUZZ);
92  }
93  return;
94  }
95 
96  // check ekf bad
97  if (_flags.ekf_bad != AP_Notify::flags.ekf_bad) {
98  _flags.ekf_bad = AP_Notify::flags.ekf_bad;
99  if (_flags.ekf_bad) {
100  // ekf bad warning buzz
101  play_pattern(EKF_BAD);
102  }
103  return;
104  }
105 
106  // if vehicle lost was enabled, starting beep
107  if (AP_Notify::flags.vehicle_lost) {
108  play_pattern(DOUBLE_BUZZ);
109  return;
110  }
111 
112  // if battery failsafe constantly single buzz
113  if (AP_Notify::flags.failsafe_battery) {
114  play_pattern(SINGLE_BUZZ);
115  return;
116  }
117 }

120 void Buzzer::update_playing_pattern()
121 {
122  if (_pattern == 0UL) {
123  return;
124  }
125 
126  const uint32_t now = AP_HAL::millis();
127  const uint32_t delta = now - _pattern_start_time;
128  if (delta >= 3200) {
129  // finished playing pattern
130  on(false);
131  _pattern = 0UL;
132  return;
133  }
134  const uint32_t bit = delta / 100UL; // each bit is 100ms
135  on(_pattern & (1U<<(31-bit)));
136 }
