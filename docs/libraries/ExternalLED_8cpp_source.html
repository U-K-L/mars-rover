43  main update function called at 50Hz
44  */
45 void ExternalLED::update(void)
46 {
47  // reduce update rate from 50hz to 10hz
48  _counter++;
49  if (_counter < 5) {
50  return;
51  }
52  _counter = 0;
53 
54  // internal counter used to control step of armed and gps led
55  _counter2++;
56  if (_counter2 >= 10) {
57  _counter2 = 0;
58  }
59 
60  // initialising
61  if (AP_Notify::flags.initialising) {
62  // blink arming and gps leds at 5hz
63  switch(_counter2) {
64  case 0:
65  case 2:
66  case 4:
67  case 6:
68  case 8:
69  armed_led(true);
70  gps_led(false);
71  break;
72  case 1:
73  case 3:
74  case 5:
75  case 7:
76  case 9:
77  armed_led(false);
78  gps_led(true);
79  break;
80  }
81  return;
82  }
83 
84  // arming led control
85  if (AP_Notify::flags.armed) {
86  armed_led(true);
87  }else{
88  // blink arming led at 2hz
89  switch(_counter2) {
90  case 0:
91  case 1:
92  case 2:
93  case 5:
94  case 6:
95  case 7:
96  armed_led(false);
97  break;
98  case 3:
99  case 4:
100  case 8:
101  case 9:
102  armed_led(true);
103  break;
104  }
105  }
106 
107  // GPS led control
108  switch (AP_Notify::flags.gps_status) {
109  case 0:
110  // no GPS attached
111  gps_led(false);
112  break;
113  case 1:
114  case 2:
115  // GPS attached but no lock, blink at 4Hz
116  switch(_counter2) { // Pattern: 3(off), 2(on), 3(off), 2(on), repeat
117  case 0:
118  case 1:
119  case 2:
120  case 5:
121  case 6:
122  case 7:
123  gps_led(false);
124  break;
125  case 3:
126  case 4:
127  case 8:
128  case 9:
129  gps_led(true);
130  break;
131  }
132  break;
133  case 3:
134  // solid blue on gps lock
135  gps_led(true);
136  break;
137  }
138 
139  // motor led control
140  // if we are displaying a pattern complete it
141  if (_pattern != NONE) {
142  _pattern_counter++;
143  switch(_pattern) {
144  case NONE:
145  // do nothing
146  break;
147  case FAST_FLASH:
148  switch(_pattern_counter) {
149  case 1:
150  case 3:
151  case 5:
152  case 7:
153  case 9:
154  motor_led1(true);
155  motor_led2(true);
156  break;
157  case 2:
158  case 4:
159  case 6:
160  case 8:
161  motor_led1(false);
162  motor_led2(false);
163  break;
164  case 10:
165  motor_led1(false);
166  motor_led2(false);
167  set_pattern(NONE);
168  break;
169  }
170  break;
171  case OSCILLATE:
172  switch(_pattern_counter) {
173  case 1:
174  motor_led1(true);
175  motor_led2(false);
176  break;
177  case 4:
178  motor_led1(false);
179  motor_led2(true);
180  break;
181  case 6:
182  set_pattern(NONE);
183  break;
184  }
185  break;
186  }
187  }else{
188  if (AP_Notify::flags.failsafe_battery || AP_Notify::flags.failsafe_radio) {
189  // radio or battery failsafe indicated by fast flashing
190  set_pattern(FAST_FLASH);
191  } else {
192  // otherwise do whatever the armed led is doing
193  motor_led1(_flags.armedled_on);
194  motor_led2(_flags.armedled_on);
195  }
196  }
197 }
