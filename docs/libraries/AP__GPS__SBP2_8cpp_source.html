94 //This attempts to reads all SBP messages from the incoming port.
95 //Returns true if a new message was read, false if we failed to read a message.
96 void
97 AP_GPS_SBP2::_sbp_process()
98 {
99  uint32_t nleft = port->available();
100  while (nleft > 0) {
101  nleft--;
102  uint8_t temp = port->read();
103  uint16_t crc;
104 
105  //This switch reads one character at a time,
106  //parsing it into buffers until a full message is dispatched
107  switch (parser_state.state) {
108  case sbp_parser_state_t::WAITING:
109  if (temp == SBP_PREAMBLE) {
110  parser_state.n_read = 0;
111  parser_state.state = sbp_parser_state_t::GET_TYPE;
112  }
113  break;
114 
115  case sbp_parser_state_t::GET_TYPE:
116  *((uint8_t*)&(parser_state.msg_type) + parser_state.n_read) = temp;
117  parser_state.n_read += 1;
118  if (parser_state.n_read >= 2) {
119  parser_state.n_read = 0;
120  parser_state.state = sbp_parser_state_t::GET_SENDER;
121  }
122  break;
123 
124  case sbp_parser_state_t::GET_SENDER:
125  *((uint8_t*)&(parser_state.sender_id) + parser_state.n_read) = temp;
126  parser_state.n_read += 1;
127  if (parser_state.n_read >= 2) {
128  parser_state.n_read = 0;
129  parser_state.state = sbp_parser_state_t::GET_LEN;
130  }
131  break;
132 
133  case sbp_parser_state_t::GET_LEN:
134  parser_state.msg_len = temp;
135  parser_state.n_read = 0;
136  parser_state.state = sbp_parser_state_t::GET_MSG;
137  break;
138 
139  case sbp_parser_state_t::GET_MSG:
140  *((uint8_t*)&(parser_state.msg_buff) + parser_state.n_read) = temp;
141  parser_state.n_read += 1;
142  if (parser_state.n_read >= parser_state.msg_len) {
143  parser_state.n_read = 0;
144  parser_state.state = sbp_parser_state_t::GET_CRC;
145  }
146  break;
147 
148  case sbp_parser_state_t::GET_CRC:
149  *((uint8_t*)&(parser_state.crc) + parser_state.n_read) = temp;
150  parser_state.n_read += 1;
151  if (parser_state.n_read >= 2) {
152  parser_state.state = sbp_parser_state_t::WAITING;
153 
154  crc = crc16_ccitt((uint8_t*)&(parser_state.msg_type), 2, 0);
155  crc = crc16_ccitt((uint8_t*)&(parser_state.sender_id), 2, crc);
156  crc = crc16_ccitt(&(parser_state.msg_len), 1, crc);
157  crc = crc16_ccitt(parser_state.msg_buff, parser_state.msg_len, crc);
158  if (parser_state.crc == crc) {
159  _sbp_process_message();
160  } else {
161  Debug("CRC Error Occurred!");
162  crc_error_counter += 1;
163  }
164  }
165  break;
166 
167  default:
168  parser_state.state = sbp_parser_state_t::WAITING;
169  break;
170  }
171  }
172 }

175 //INVARIANT: A fully received message with correct CRC is currently in parser_state
176 void
177 AP_GPS_SBP2::_sbp_process_message() {
178  //Here, we copy messages into local structs.
179  switch (parser_state.msg_type) {
180  case SBP_HEARTBEAT_MSGTYPE:
181  memcpy(&last_heartbeat, parser_state.msg_buff, sizeof(struct sbp_heartbeat_t));
182  last_heartbeat_received_ms = AP_HAL::millis();
183  break;
184 
185  case SBP_GPS_TIME_MSGTYPE:
186  memcpy(&last_gps_time, parser_state.msg_buff, sizeof(struct sbp_gps_time_t));
187  check_new_itow(last_gps_time.tow, parser_state.msg_len);
188  break;
189 
190  case SBP_VEL_NED_MSGTYPE:
191  memcpy(&last_vel_ned, parser_state.msg_buff, sizeof(struct sbp_vel_ned_t));
192  check_new_itow(last_vel_ned.tow, parser_state.msg_len);
193  break;
194 
195  case SBP_POS_LLH_MSGTYPE:
196  memcpy(&last_pos_llh, parser_state.msg_buff, sizeof(struct sbp_pos_llh_t));
197  check_new_itow(last_pos_llh.tow, parser_state.msg_len);
198  break;
199 
200  case SBP_DOPS_MSGTYPE:
201  memcpy(&last_dops, parser_state.msg_buff, sizeof(struct sbp_dops_t));
202  check_new_itow(last_dops.tow, parser_state.msg_len);
203  break;
204 
205  case SBP_EXT_EVENT_MSGTYPE:
206  memcpy(&last_event, parser_state.msg_buff, sizeof(struct sbp_ext_event_t));
207  check_new_itow(last_event.tow, parser_state.msg_len);
208  logging_ext_event();
209  break;
210 
211  default:
212  break;
213  }
214 
215  // send all messages we receive to log, even if it's an unsupported message,
216  // so we can do additional post-processing from logs.
217  // The log mask will be used to adjust or suppress logging
218  logging_log_raw_sbp(parser_state.msg_type, parser_state.sender_id, parser_state.msg_len, parser_state.msg_buff);
219 }

226 bool
227 AP_GPS_SBP2::_attempt_state_update()
228 {
229  if (last_heartbeat_received_ms == 0)
230  return false;
231 
232  uint32_t now = AP_HAL::millis();
233 
234  if (now - last_heartbeat_received_ms > SBP_TIMEOUT_HEARTBEAT) {
235 
236  state.status = AP_GPS::NO_FIX;
237  Info("No Heartbeats from Piksi! Status to NO_FIX.");
238  return false;
239 
240  } else if (last_heartbeat.protocol_major != 2) {
241 
242  state.status = AP_GPS::NO_FIX;
243  Info("Received a heartbeat from non-SBPv2 device. Current driver only supports SBPv2. Status to NO_FIX.");
244  return false;
245 
246  } else if (last_heartbeat.nap_error == 1 ||
247  last_heartbeat.io_error == 1 ||
248  last_heartbeat.sys_error == 1) {
249 
250  state.status = AP_GPS::NO_FIX;
251 
252  Info("Piksi reported an error. Status to NO_FIX.");
253  Debug(" ext_antenna: %d", last_heartbeat.ext_antenna);
254  Debug(" res2: %d", last_heartbeat.res2);
255  Debug(" protocol_major: %d", last_heartbeat.protocol_major);
256  Debug(" protocol_minor: %d", last_heartbeat.protocol_minor);
257  Debug(" res: %d", last_heartbeat.res);
258  Debug(" nap_error: %d", last_heartbeat.nap_error);
259  Debug(" io_error: %d", last_heartbeat.io_error);
260  Debug(" sys_error: %d", last_heartbeat.sys_error);
261 
262  return false;
263 
264  } else if (last_pos_llh.tow == last_vel_ned.tow
265  && (distMod(last_gps_time.tow, last_vel_ned.tow, AP_MSEC_PER_WEEK) < 10000)
266  && (distMod(last_dops.tow, last_vel_ned.tow, AP_MSEC_PER_WEEK) < 60000)
267  && (last_vel_ned.tow > last_full_update_tow || (last_gps_time.wn > last_full_update_wn && last_vel_ned.tow < last_full_update_tow))) {
268 
269  //We have an aligned VEL and LLH, and a recent DOPS and TIME.
270 
271  //
272  // Check Flags for Valid Messages
273  //
274  if (last_gps_time.flags.time_src == 0 ||
275  last_vel_ned.flags.vel_mode == 0 ||
276  last_pos_llh.flags.fix_mode == 0 ||
277  last_dops.flags.fix_mode == 0) {
278 
279  Debug("Message Marked as Invalid. NO FIX! Flags: {GPS_TIME: %d, VEL_NED: %d, POS_LLH: %d, DOPS: %d}",
280  last_gps_time.flags.time_src,
281  last_vel_ned.flags.vel_mode,
282  last_pos_llh.flags.fix_mode,
283  last_dops.flags.fix_mode);
284 
285  state.status = AP_GPS::NO_FIX;
286  return false;
287  }
288 
289  //
290  // Update external time and accuracy state
291  //
292  state.time_week = last_gps_time.wn;
293  state.time_week_ms = last_vel_ned.tow;
294  state.hdop = last_dops.hdop;
295  state.vdop = last_dops.vdop;
296  state.last_gps_time_ms = now;
297 
298  //
299  // Update velocity state
300  //
301  state.velocity[0] = (float)(last_vel_ned.n * 1.0e-3);
302  state.velocity[1] = (float)(last_vel_ned.e * 1.0e-3);
303  state.velocity[2] = (float)(last_vel_ned.d * 1.0e-3);
304 
305  float ground_vector_sq = state.velocity[0]*state.velocity[0] + state.velocity[1]*state.velocity[1];
306  state.ground_speed = safe_sqrt(ground_vector_sq);
307 
308  state.ground_course = wrap_360(degrees(atan2f(state.velocity[1], state.velocity[0])));
309 
310  state.speed_accuracy = safe_sqrt(
311  powf((float)last_vel_ned.h_accuracy * 1.0e-3f, 2) +
312  powf((float)last_vel_ned.v_accuracy * 1.0e-3f, 2));
313  state.horizontal_accuracy = (float) last_pos_llh.h_accuracy * 1.0e-3f;
314  state.vertical_accuracy = (float) last_pos_llh.v_accuracy * 1.0e-3f;
315 
316  //
317  // Set flags appropriately
318  //
319  state.have_vertical_velocity = true;
320  state.have_speed_accuracy = !is_zero(state.speed_accuracy);
321  state.have_horizontal_accuracy = !is_zero(state.horizontal_accuracy);
322  state.have_vertical_accuracy = !is_zero(state.vertical_accuracy);
323 
324  //
325  // Update position state
326  //
327  state.location.lat = (int32_t) (last_pos_llh.lat * (double)1e7);
328  state.location.lng = (int32_t) (last_pos_llh.lon * (double)1e7);
329  state.location.alt = (int32_t) (last_pos_llh.height * 100);
330  state.num_sats = last_pos_llh.n_sats;
331 
332  switch (last_pos_llh.flags.fix_mode) {
333  case 1:
334  state.status = AP_GPS::GPS_OK_FIX_3D;
335  break;
336  case 2:
337  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
338  break;
339  case 3:
340  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
341  break;
342  case 4:
343  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
344  break;
345  case 6:
346  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
347  break;
348  default:
349  state.status = AP_GPS::NO_FIX;
350  break;
351  }
352 
353  //
354  // Update Internal Timing
355  //
356  last_full_update_tow = last_vel_ned.tow;
357  last_full_update_wn = last_gps_time.wn;
358 
359  return true;
360  }
361  return false;
362 }

366 bool
367 AP_GPS_SBP2::_detect(struct SBP2_detect_state &state, uint8_t data)
368 {
369  // This switch reads one character at a time, if we find something that
370  // looks like our preamble we'll try to read the full message length,
371  // calculating the CRC. If the CRC matches, we have an SBP GPS!
372  switch (state.state) {
373  case SBP2_detect_state::WAITING:
374  if (data == SBP_PREAMBLE) {
375  state.n_read = 0;
376  state.crc_so_far = 0;
377  state.state = SBP2_detect_state::GET_TYPE;
378  }
379  break;
380 
381  case SBP2_detect_state::GET_TYPE:
382  *((uint8_t*)&(state.msg_type) + state.n_read) = data;
383  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
384  state.n_read += 1;
385  if (state.n_read >= 2) {
386  state.n_read = 0;
387  state.state = SBP2_detect_state::GET_SENDER;
388  }
389  break;
390 
391  case SBP2_detect_state::GET_SENDER:
392  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
393  state.n_read += 1;
394  if (state.n_read >= 2) {
395  state.n_read = 0;
396  state.state = SBP2_detect_state::GET_LEN;
397  }
398  break;
399 
400  case SBP2_detect_state::GET_LEN:
401  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
402  state.msg_len = data;
403  state.n_read = 0;
404  state.state = SBP2_detect_state::GET_MSG;
405  break;
406 
407  case SBP2_detect_state::GET_MSG:
408  if (state.msg_type == SBP_HEARTBEAT_MSGTYPE && state.n_read < 4) {
409  *((uint8_t*)&(state.heartbeat_buff) + state.n_read) = data;
410  }
411  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
412  state.n_read += 1;
413  if (state.n_read >= state.msg_len) {
414  state.n_read = 0;
415  state.state = SBP2_detect_state::GET_CRC;
416  }
417  break;
418 
419  case SBP2_detect_state::GET_CRC:
420  *((uint8_t*)&(state.crc) + state.n_read) = data;
421  state.n_read += 1;
422  if (state.n_read >= 2) {
423  state.state = SBP2_detect_state::WAITING;
424  if (state.crc == state.crc_so_far
425  && state.msg_type == SBP_HEARTBEAT_MSGTYPE) {
426  struct sbp_heartbeat_t* heartbeat = ((struct sbp_heartbeat_t*)state.heartbeat_buff);
427  return heartbeat->protocol_major == 2;
428  }
429  return false;
430  }
431  break;
432 
433  default:
434  state.state = SBP2_detect_state::WAITING;
435  break;
436  }
437  return false;
438 }

460 void
461 AP_GPS_SBP2::logging_log_raw_sbp(uint16_t msg_type,
462  uint16_t sender_id,
463  uint8_t msg_len,
464  uint8_t *msg_buff) {
465  if (!should_log()) {
466  return;
467  }
468 
469  //MASK OUT MESSAGES WE DON'T WANT TO LOG
470  if (( ((uint16_t) gps._sbp_logmask) & msg_type) == 0) {
471  return;
472  }
473 
474  uint64_t time_us = AP_HAL::micros64();
475  uint8_t pages = 1;
476 
477  if (msg_len > 48) {
478  pages += (msg_len - 48) / 104 + 1;
479  }
480 
481  struct log_SbpRAWH pkt = {
482  LOG_PACKET_HEADER_INIT(LOG_MSG_SBPRAWH),
483  time_us : time_us,
484  msg_type : msg_type,
485  sender_id : sender_id,
486  index : 1,
487  pages : pages,
488  msg_len : msg_len,
489  };
490  memcpy(pkt.data, msg_buff, MIN(msg_len, 48));
491  AP::logger().WriteBlock(&pkt, sizeof(pkt));
492 
493  for (uint8_t i = 0; i < pages - 1; i++) {
494  struct log_SbpRAWM pkt2 = {
495  LOG_PACKET_HEADER_INIT(LOG_MSG_SBPRAWM),
496  time_us : time_us,
497  msg_type : msg_type,
498  sender_id : sender_id,
499  index : uint8_t(i + 2),
500  pages : pages,
501  msg_len : msg_len,
502  };
503  memcpy(pkt2.data, &msg_buff[48 + i * 104], MIN(msg_len - (48 + i * 104), 104));
504  AP::logger().WriteBlock(&pkt2, sizeof(pkt2));
505  }
