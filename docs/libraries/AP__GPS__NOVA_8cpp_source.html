56 bool
57 AP_GPS_NOVA::read(void)
58 {
59  uint32_t now = AP_HAL::millis();
60 
61  if (_init_blob_index < (sizeof(_initialisation_blob) / sizeof(_initialisation_blob[0]))) {
62  const char *init_str = _initialisation_blob[_init_blob_index];
63 
64  if (now > _init_blob_time) {
65  port->write((const uint8_t*)init_str, strlen(init_str));
66  _init_blob_time = now + 200;
67  _init_blob_index++;
68  }
69  }
70 
71  bool ret = false;
72  while (port->available() > 0) {
73  uint8_t temp = port->read();
74  ret |= parse(temp);
75  }
76 
77  return ret;
78 }

80 bool
81 AP_GPS_NOVA::parse(uint8_t temp)
82 {
83  switch (nova_msg.nova_state)
84  {
85  default:
86  case nova_msg_parser::PREAMBLE1:
87  if (temp == NOVA_PREAMBLE1)
88  nova_msg.nova_state = nova_msg_parser::PREAMBLE2;
89  nova_msg.read = 0;
90  break;
91  case nova_msg_parser::PREAMBLE2:
92  if (temp == NOVA_PREAMBLE2)
93  {
94  nova_msg.nova_state = nova_msg_parser::PREAMBLE3;
95  }
96  else
97  {
98  nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
99  }
100  break;
101  case nova_msg_parser::PREAMBLE3:
102  if (temp == NOVA_PREAMBLE3)
103  {
104  nova_msg.nova_state = nova_msg_parser::HEADERLENGTH;
105  }
106  else
107  {
108  nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
109  }
110  break;
111  case nova_msg_parser::HEADERLENGTH:
112  Debug("NOVA HEADERLENGTH\n");
113  nova_msg.header.data[0] = NOVA_PREAMBLE1;
114  nova_msg.header.data[1] = NOVA_PREAMBLE2;
115  nova_msg.header.data[2] = NOVA_PREAMBLE3;
116  nova_msg.header.data[3] = temp;
117  nova_msg.header.nova_headeru.headerlength = temp;
118  nova_msg.nova_state = nova_msg_parser::HEADERDATA;
119  nova_msg.read = 4;
120  break;
121  case nova_msg_parser::HEADERDATA:
122  if (nova_msg.read >= sizeof(nova_msg.header.data)) {
123  Debug("parse header overflow length=%u\n", (unsigned)nova_msg.read);
124  nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
125  break;
126  }
127  nova_msg.header.data[nova_msg.read] = temp;
128  nova_msg.read++;
129  if (nova_msg.read >= nova_msg.header.nova_headeru.headerlength)
130  {
131  nova_msg.nova_state = nova_msg_parser::DATA;
132  }
133  break;
134  case nova_msg_parser::DATA:
135  if (nova_msg.read >= sizeof(nova_msg.data)) {
136  Debug("parse data overflow length=%u msglength=%u\n", (unsigned)nova_msg.read,nova_msg.header.nova_headeru.messagelength);
137  nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
138  break;
139  }
140  nova_msg.data.bytes[nova_msg.read - nova_msg.header.nova_headeru.headerlength] = temp;
141  nova_msg.read++;
142  if (nova_msg.read >= (nova_msg.header.nova_headeru.messagelength + nova_msg.header.nova_headeru.headerlength))
143  {
144  Debug("NOVA DATA exit\n");
145  nova_msg.nova_state = nova_msg_parser::CRC1;
146  }
147  break;
148  case nova_msg_parser::CRC1:
149  nova_msg.crc = (uint32_t) (temp << 0);
150  nova_msg.nova_state = nova_msg_parser::CRC2;
151  break;
152  case nova_msg_parser::CRC2:
153  nova_msg.crc += (uint32_t) (temp << 8);
154  nova_msg.nova_state = nova_msg_parser::CRC3;
155  break;
156  case nova_msg_parser::CRC3:
157  nova_msg.crc += (uint32_t) (temp << 16);
158  nova_msg.nova_state = nova_msg_parser::CRC4;
159  break;
160  case nova_msg_parser::CRC4:
161  nova_msg.crc += (uint32_t) (temp << 24);
162  nova_msg.nova_state = nova_msg_parser::PREAMBLE1;
163 
164  uint32_t crc = CalculateBlockCRC32((uint32_t)nova_msg.header.nova_headeru.headerlength, (uint8_t *)&nova_msg.header.data, (uint32_t)0);
165  crc = CalculateBlockCRC32((uint32_t)nova_msg.header.nova_headeru.messagelength, (uint8_t *)&nova_msg.data, crc);
166 
167  if (nova_msg.crc == crc)
168  {
169  return process_message();
170  }
171  else
172  {
173  Debug("crc failed");
174  crc_error_counter++;
175  }
176  break;
177  }
178 
179  return false;
180 }

182 bool
183 AP_GPS_NOVA::process_message(void)
184 {
185  uint16_t messageid = nova_msg.header.nova_headeru.messageid;
186 
187  Debug("NOVA process_message messid=%u\n",messageid);
188 
189  check_new_itow(nova_msg.header.nova_headeru.tow, nova_msg.header.nova_headeru.messagelength + nova_msg.header.nova_headeru.headerlength);
190 
191  if (messageid == 42) // bestpos
192  {
193  const bestpos &bestposu = nova_msg.data.bestposu;
194 
195  state.time_week = nova_msg.header.nova_headeru.week;
196  state.time_week_ms = (uint32_t) nova_msg.header.nova_headeru.tow;
197  state.last_gps_time_ms = AP_HAL::millis();
198 
199  state.location.lat = (int32_t) (bestposu.lat * (double)1e7);
200  state.location.lng = (int32_t) (bestposu.lng * (double)1e7);
201  state.location.alt = (int32_t) (bestposu.hgt * 100);
202 
203  state.num_sats = bestposu.svsused;
204 
205  state.horizontal_accuracy = (float) ((bestposu.latsdev + bestposu.lngsdev)/2);
206  state.vertical_accuracy = (float) bestposu.hgtsdev;
207  state.have_horizontal_accuracy = true;
208  state.have_vertical_accuracy = true;
209  state.rtk_age_ms = bestposu.diffage * 1000;
210  state.rtk_num_sats = bestposu.svsused;
211 
212  if (bestposu.solstat == 0) // have a solution
213  {
214  switch (bestposu.postype)
215  {
216  case 16:
217  state.status = AP_GPS::GPS_OK_FIX_3D;
218  break;
219  case 17: // psrdiff
220  case 18: // waas
221  case 20: // omnistar
222  case 68: // ppp_converg
223  case 69: // ppp
224  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
225  break;
226  case 32: // l1 float
227  case 33: // iono float
228  case 34: // narrow float
229  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
230  break;
231  case 48: // l1 int
232  case 50: // narrow int
233  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
234  break;
235  case 0: // NONE
236  case 1: // FIXEDPOS
237  case 2: // FIXEDHEIGHT
238  default:
239  state.status = AP_GPS::NO_FIX;
240  break;
241  }
242  }
243  else
244  {
245  state.status = AP_GPS::NO_FIX;
246  }
247 
248  _new_position = true;
249  }
250 
251  if (messageid == 99) // bestvel
252  {
253  const bestvel &bestvelu = nova_msg.data.bestvelu;
254 
255  state.ground_speed = (float) bestvelu.horspd;
256  state.ground_course = (float) bestvelu.trkgnd;
257  fill_3d_velocity();
258  state.velocity.z = -(float) bestvelu.vertspd;
259  state.have_vertical_velocity = true;
260 
261  _last_vel_time = (uint32_t) nova_msg.header.nova_headeru.tow;
262  _new_speed = true;
263  }
264 
265  if (messageid == 174) // psrdop
266  {
267  const psrdop &psrdopu = nova_msg.data.psrdopu;
268 
269  state.hdop = (uint16_t) (psrdopu.hdop*100);
270  state.vdop = (uint16_t) (psrdopu.htdop*100);
271  return false;
272  }
273 
274  // ensure out position and velocity stay insync
275  if (_new_position && _new_speed && _last_vel_time == state.time_week_ms) {
276  _new_speed = _new_position = false;
277 
278  return true;
279  }
280 
281  return false;
282 }

295 #define CRC32_POLYNOMIAL 0xEDB88320L
296 uint32_t AP_GPS_NOVA::CRC32Value(uint32_t icrc)
297 {
298  int i;
299  uint32_t crc = icrc;
300  for ( i = 8 ; i > 0; i-- )
301  {
302  if ( crc & 1 )
303  crc = ( crc >> 1 ) ^ CRC32_POLYNOMIAL;
304  else
305  crc >>= 1;
306  }
307  return crc;
308 }

310 uint32_t AP_GPS_NOVA::CalculateBlockCRC32(uint32_t length, uint8_t *buffer, uint32_t crc)
311 {
312  while ( length-- != 0 )
313  {
314  crc = ((crc >> 8) & 0x00FFFFFFL) ^ (CRC32Value(((uint32_t) crc ^ *buffer++) & 0xff));
315  }
316  return( crc );
