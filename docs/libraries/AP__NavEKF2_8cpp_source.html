617 // Initialise the filter
618 bool NavEKF2::InitialiseFilter(void)
619 {
620  if (_enable == 0) {
621  return false;
622  }
623  const AP_InertialSensor &ins = AP::ins();
624 
625  imuSampleTime_us = AP_HAL::micros64();
626 
627  // remember expected frame time
628  _frameTimeUsec = 1e6 / ins.get_sample_rate();
629 
630  // expected number of IMU frames per prediction
631  _framesPerPrediction = uint8_t((EKF_TARGET_DT / (_frameTimeUsec * 1.0e-6) + 0.5));
632 
633  // see if we will be doing logging
634  AP_Logger *logger = AP_Logger::get_singleton();
635  if (logger != nullptr) {
636  logging.enabled = logger->log_replay();
637  }
638 
639  if (core == nullptr) {
640 
641  // don't run multiple filters for 1 IMU
642  uint8_t mask = (1U<<ins.get_accel_count())-1;
643  _imuMask.set(_imuMask.get() & mask);
644 
645  // count IMUs from mask
646  num_cores = 0;
647  for (uint8_t i=0; i<7; i++) {
648  if (_imuMask & (1U<<i)) {
649  num_cores++;
650  }
651  }
652 
653  // check if there is enough memory to create the EKF cores
654  if (hal.util->available_memory() < sizeof(NavEKF2_core)*num_cores + 4096) {
655  gcs().send_text(MAV_SEVERITY_CRITICAL, "NavEKF2: not enough memory");
656  _enable.set(0);
657  return false;
658  }
659 
660  // try to allocate from CCM RAM, fallback to Normal RAM if not available or full
661  core = (NavEKF2_core*)hal.util->malloc_type(sizeof(NavEKF2_core)*num_cores, AP_HAL::Util::MEM_FAST);
662  if (core == nullptr) {
663  _enable.set(0);
664  gcs().send_text(MAV_SEVERITY_CRITICAL, "NavEKF2: allocation failed");
665  return false;
666  }
667  for (uint8_t i = 0; i < num_cores; i++) {
668  //Call Constructors
669  new (&core[i]) NavEKF2_core();
670  }
671 
672  // set the IMU index for the cores
673  num_cores = 0;
674  for (uint8_t i=0; i<7; i++) {
675  if (_imuMask & (1U<<i)) {
676  if(!core[num_cores].setup_core(this, i, num_cores)) {
677  return false;
678  }
679  num_cores++;
680  }
681  }
682 
683  // Set the primary initially to be the lowest index
684  primary = 0;
685  }
686 
687  // initialise the cores. We return success only if all cores
688  // initialise successfully
689  bool ret = true;
690  for (uint8_t i=0; i<num_cores; i++) {
691  ret &= core[i].InitialiseFilterBootstrap();
692  }
693 
694  // zero the structs used capture reset events
695  memset(&yaw_reset_data, 0, sizeof(yaw_reset_data));
696  memset((void *)&pos_reset_data, 0, sizeof(pos_reset_data));
697  memset(&pos_down_reset_data, 0, sizeof(pos_down_reset_data));
698 
699  check_log_write();
700  return ret;
701 }

703 // Update Filter States - this should be called whenever new IMU data is available
704 void NavEKF2::UpdateFilter(void)
705 {
706  if (!core) {
707  return;
708  }
709 
710  imuSampleTime_us = AP_HAL::micros64();
711 
712  const AP_InertialSensor &ins = AP::ins();
713 
714  bool statePredictEnabled[num_cores];
715  for (uint8_t i=0; i<num_cores; i++) {
716  // if we have not overrun by more than 3 IMU frames, and we
717  // have already used more than 1/3 of the CPU budget for this
718  // loop then suppress the prediction step. This allows
719  // multiple EKF instances to cooperate on scheduling
720  if (core[i].getFramesSincePredict() < (_framesPerPrediction+3) &&
721  (AP_HAL::micros() - ins.get_last_update_usec()) > _frameTimeUsec/3) {
722  statePredictEnabled[i] = false;
723  } else {
724  statePredictEnabled[i] = true;
725  }
726  core[i].UpdateFilter(statePredictEnabled[i]);
727  }
728 
729  // If the current core selected has a bad error score or is unhealthy, switch to a healthy core with the lowest fault score
730  // Don't start running the check until the primary core has started returned healthy for at least 10 seconds to avoid switching
731  // due to initial alignment fluctuations and race conditions
732  if (!runCoreSelection) {
733  static uint64_t lastUnhealthyTime_us = 0;
734  if (!core[primary].healthy() || lastUnhealthyTime_us == 0) {
735  lastUnhealthyTime_us = imuSampleTime_us;
736  }
737  runCoreSelection = (imuSampleTime_us - lastUnhealthyTime_us) > 1E7;
738  }
739  float primaryErrorScore = core[primary].errorScore();
740  if ((primaryErrorScore > 1.0f || !core[primary].healthy()) && runCoreSelection) {
741  float lowestErrorScore = 0.67f * primaryErrorScore;
742  uint8_t newPrimaryIndex = primary; // index for new primary
743  for (uint8_t coreIndex=0; coreIndex<num_cores; coreIndex++) {
744 
745  if (coreIndex != primary) {
746  // an alternative core is available for selection only if healthy and if states have been updated on this time step
747  bool altCoreAvailable = core[coreIndex].healthy() && statePredictEnabled[coreIndex];
748 
749  // If the primary core is unhealthy and another core is available, then switch now
750  // If the primary core is still healthy,then switching is optional and will only be done if
751  // a core with a significantly lower error score can be found
752  float altErrorScore = core[coreIndex].errorScore();
753  if (altCoreAvailable && (!core[newPrimaryIndex].healthy() || altErrorScore < lowestErrorScore)) {
754  newPrimaryIndex = coreIndex;
755  lowestErrorScore = altErrorScore;
756  }
757  }
758  }
759  // update the yaw and position reset data to capture changes due to the lane switch
760  if (newPrimaryIndex != primary) {
761  updateLaneSwitchYawResetData(newPrimaryIndex, primary);
762  updateLaneSwitchPosResetData(newPrimaryIndex, primary);
763  updateLaneSwitchPosDownResetData(newPrimaryIndex, primary);
764  primary = newPrimaryIndex;
765  lastLaneSwitch_ms = AP_HAL::millis();
766  }
767  }
768 
769  if (primary != 0 && core[0].healthy() && !hal.util->get_soft_armed()) {
770  // when on the ground and disarmed force the first lane. This
771  // avoids us ending with with a lottery for which IMU is used
772  // in each flight. Otherwise the alignment of the timing of
773  // the lane updates with the timing of GPS updates can lead to
774  // a lane other than the first one being used as primary for
775  // some flights. As different IMUs may have quite different
776  // noise characteristics this leads to inconsistent
777  // performance
778  primary = 0;
779  }
780 
781  check_log_write();
782 }

784 /*
785  check if switching lanes will reduce the normalised
786  innovations. This is called when the vehicle code is about to
787  trigger an EKF failsafe, and it would like to avoid that by
788  using a different EKF lane
789 */
790 void NavEKF2::checkLaneSwitch(void)
791 {
792  uint32_t now = AP_HAL::millis();
793  if (lastLaneSwitch_ms != 0 && now - lastLaneSwitch_ms < 5000) {
794  // don't switch twice in 5 seconds
795  return;
796  }
797  float primaryErrorScore = core[primary].errorScore();
798  float lowestErrorScore = primaryErrorScore;
799  uint8_t newPrimaryIndex = primary;
800  for (uint8_t coreIndex=0; coreIndex<num_cores; coreIndex++) {
801  if (coreIndex != primary) {
802  // an alternative core is available for selection only if healthy and if states have been updated on this time step
803  bool altCoreAvailable = core[coreIndex].healthy();
804  float altErrorScore = core[coreIndex].errorScore();
805  if (altCoreAvailable &&
806  altErrorScore < lowestErrorScore &&
807  altErrorScore < 0.9) {
808  newPrimaryIndex = coreIndex;
809  lowestErrorScore = altErrorScore;
810  }
811  }
812  }
813 
814  // update the yaw and position reset data to capture changes due to the lane switch
815  if (newPrimaryIndex != primary) {
816  updateLaneSwitchYawResetData(newPrimaryIndex, primary);
817  updateLaneSwitchPosResetData(newPrimaryIndex, primary);
818  updateLaneSwitchPosDownResetData(newPrimaryIndex, primary);
819  primary = newPrimaryIndex;
820  lastLaneSwitch_ms = now;
821  gcs().send_text(MAV_SEVERITY_CRITICAL, "NavEKF2: lane switch %u", primary);
822  }
823 }
