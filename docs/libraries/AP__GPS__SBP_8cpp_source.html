92 //This attempts to reads all SBP messages from the incoming port.
93 //Returns true if a new message was read, false if we failed to read a message.
94 void
95 AP_GPS_SBP::_sbp_process()
96 {
97 
98  while (port->available() > 0) {
99  uint8_t temp = port->read();
100  uint16_t crc;
101 
102 
103  //This switch reads one character at a time,
104  //parsing it into buffers until a full message is dispatched
105  switch (parser_state.state) {
106  case sbp_parser_state_t::WAITING:
107  if (temp == SBP_PREAMBLE) {
108  parser_state.n_read = 0;
109  parser_state.state = sbp_parser_state_t::GET_TYPE;
110  }
111  break;
112 
113  case sbp_parser_state_t::GET_TYPE:
114  *((uint8_t*)&(parser_state.msg_type) + parser_state.n_read) = temp;
115  parser_state.n_read += 1;
116  if (parser_state.n_read >= 2) {
117  parser_state.n_read = 0;
118  parser_state.state = sbp_parser_state_t::GET_SENDER;
119  }
120  break;
121 
122  case sbp_parser_state_t::GET_SENDER:
123  *((uint8_t*)&(parser_state.sender_id) + parser_state.n_read) = temp;
124  parser_state.n_read += 1;
125  if (parser_state.n_read >= 2) {
126  parser_state.n_read = 0;
127  parser_state.state = sbp_parser_state_t::GET_LEN;
128  }
129  break;
130 
131  case sbp_parser_state_t::GET_LEN:
132  parser_state.msg_len = temp;
133  parser_state.n_read = 0;
134  parser_state.state = sbp_parser_state_t::GET_MSG;
135  break;
136 
137  case sbp_parser_state_t::GET_MSG:
138  *((uint8_t*)&(parser_state.msg_buff) + parser_state.n_read) = temp;
139  parser_state.n_read += 1;
140  if (parser_state.n_read >= parser_state.msg_len) {
141  parser_state.n_read = 0;
142  parser_state.state = sbp_parser_state_t::GET_CRC;
143  }
144  break;
145 
146  case sbp_parser_state_t::GET_CRC:
147  *((uint8_t*)&(parser_state.crc) + parser_state.n_read) = temp;
148  parser_state.n_read += 1;
149  if (parser_state.n_read >= 2) {
150  parser_state.state = sbp_parser_state_t::WAITING;
151 
152  crc = crc16_ccitt((uint8_t*)&(parser_state.msg_type), 2, 0);
153  crc = crc16_ccitt((uint8_t*)&(parser_state.sender_id), 2, crc);
154  crc = crc16_ccitt(&(parser_state.msg_len), 1, crc);
155  crc = crc16_ccitt(parser_state.msg_buff, parser_state.msg_len, crc);
156  if (parser_state.crc == crc) {
157  _sbp_process_message();
158  } else {
159  Debug("CRC Error Occurred!");
160  crc_error_counter += 1;
161  }
162 
163  parser_state.state = sbp_parser_state_t::WAITING;
164  }
165  break;
166 
167  default:
168  parser_state.state = sbp_parser_state_t::WAITING;
169  break;
170  }
171  }
172 }

175 //INVARIANT: A fully received message with correct CRC is currently in parser_state
176 void
177 AP_GPS_SBP::_sbp_process_message() {
178  switch (parser_state.msg_type) {
179  case SBP_HEARTBEAT_MSGTYPE:
180  last_heatbeat_received_ms = AP_HAL::millis();
181  break;
182 
183  case SBP_GPS_TIME_MSGTYPE:
184  memcpy(&last_gps_time, parser_state.msg_buff, sizeof(last_gps_time));
185  check_new_itow(last_gps_time.tow, parser_state.msg_len);
186  break;
187 
188  case SBP_VEL_NED_MSGTYPE:
189  memcpy(&last_vel_ned, parser_state.msg_buff, sizeof(last_vel_ned));
190  check_new_itow(last_vel_ned.tow, parser_state.msg_len);
191  break;
192 
193  case SBP_POS_LLH_MSGTYPE: {
194  struct sbp_pos_llh_t *pos_llh = (struct sbp_pos_llh_t*)parser_state.msg_buff;
195  check_new_itow(pos_llh->tow, parser_state.msg_len);
196  // Check if this is a single point or RTK solution
197  // flags = 0 -> single point
198  if (pos_llh->flags == 0) {
199  last_pos_llh_spp = *pos_llh;
200  } else if (pos_llh->flags == 1 || pos_llh->flags == 2) {
201  last_pos_llh_rtk = *pos_llh;
202  }
203  break;
204  }
205 
206  case SBP_DOPS_MSGTYPE:
207  memcpy(&last_dops, parser_state.msg_buff, sizeof(last_dops));
208  check_new_itow(last_dops.tow, parser_state.msg_len);
209  break;
210 
211  case SBP_TRACKING_STATE_MSGTYPE:
212  //INTENTIONALLY BLANK
213  //Currently unhandled, but logged after switch statement.
214  break;
215 
216  case SBP_IAR_STATE_MSGTYPE: {
217  sbp_iar_state_t *iar = (struct sbp_iar_state_t*)parser_state.msg_buff;
218  last_iar_num_hypotheses = iar->num_hypotheses;
219  break;
220  }
221 
222  default:
223  // log anyway if it's an unsupported message.
224  // The log mask will be used to adjust or suppress logging
225  break;
226  }
227 
228  logging_log_raw_sbp(parser_state.msg_type, parser_state.sender_id, parser_state.msg_len, parser_state.msg_buff);
229 }

231 bool
232 AP_GPS_SBP::_attempt_state_update()
233 {
234 
235  // If we currently have heartbeats
236  // - NO FIX
237  //
238  // If we have a full update available, save it
239  //
240  uint32_t now = AP_HAL::millis();
241  bool ret = false;
242 
243  if (now - last_heatbeat_received_ms > SBP_TIMEOUT_HEATBEAT) {
244 
245  state.status = AP_GPS::NO_FIX;
246  Debug("No Heartbeats from Piksi! Driver Ready to Die!");
247 
248  } else if (last_pos_llh_rtk.tow == last_vel_ned.tow
249  && abs((int32_t) (last_gps_time.tow - last_vel_ned.tow)) < 10000
250  && abs((int32_t) (last_dops.tow - last_vel_ned.tow)) < 60000
251  && last_vel_ned.tow > last_full_update_tow) {
252 
253  // Use the RTK position
254  sbp_pos_llh_t *pos_llh = &last_pos_llh_rtk;
255 
256  // Update time state
257  state.time_week = last_gps_time.wn;
258  state.time_week_ms = last_vel_ned.tow;
259 
260  state.hdop = last_dops.hdop;
261 
262  // Update velocity state
263  state.velocity[0] = (float)(last_vel_ned.n * 1.0e-3);
264  state.velocity[1] = (float)(last_vel_ned.e * 1.0e-3);
265  state.velocity[2] = (float)(last_vel_ned.d * 1.0e-3);
266  state.have_vertical_velocity = true;
267 
268  float ground_vector_sq = state.velocity[0]*state.velocity[0] + state.velocity[1]*state.velocity[1];
269  state.ground_speed = safe_sqrt(ground_vector_sq);
270 
271  state.ground_course = wrap_360(degrees(atan2f(state.velocity[1], state.velocity[0])));
272 
273  // Update position state
274 
275  state.location.lat = (int32_t) (pos_llh->lat * (double)1e7);
276  state.location.lng = (int32_t) (pos_llh->lon * (double)1e7);
277  state.location.alt = (int32_t) (pos_llh->height * 100);
278  state.num_sats = pos_llh->n_sats;
279 
280  if (pos_llh->flags == 0) {
281  state.status = AP_GPS::GPS_OK_FIX_3D;
282  } else if (pos_llh->flags == 2) {
283  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FLOAT;
284  } else if (pos_llh->flags == 1) {
285  state.status = AP_GPS::GPS_OK_FIX_3D_RTK_FIXED;
286  }
287 
288  last_full_update_tow = last_vel_ned.tow;
289  last_full_update_cpu_ms = now;
290  state.rtk_iar_num_hypotheses = last_iar_num_hypotheses;
291 
292  logging_log_full_update();
293  ret = true;
294 
295  } else if (now - last_full_update_cpu_ms > SBP_TIMEOUT_PVT) {
296 
297  //INVARIANT: If we currently have a fix, ONLY return true after a full update.
298 
299  state.status = AP_GPS::NO_FIX;
300  ret = true;
301 
302  } else {
303 
304  //No timeouts yet, no data yet, nothing has happened.
305 
306  }
307 
308  return ret;
309 
310 }

314 bool
315 AP_GPS_SBP::_detect(struct SBP_detect_state &state, uint8_t data)
316 {
317  // This switch reads one character at a time, if we find something that
318  // looks like our preamble we'll try to read the full message length,
319  // calculating the CRC. If the CRC matches, we have an SBP GPS!
320 
321  switch (state.state) {
322  case SBP_detect_state::WAITING:
323  if (data == SBP_PREAMBLE) {
324  state.n_read = 0;
325  state.crc_so_far = 0;
326  state.state = SBP_detect_state::GET_TYPE;
327  }
328  break;
329 
330  case SBP_detect_state::GET_TYPE:
331  *((uint8_t*)&(state.msg_type) + state.n_read) = data;
332  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
333  state.n_read += 1;
334  if (state.n_read >= 2) {
335  state.n_read = 0;
336  state.state = SBP_detect_state::GET_SENDER;
337  }
338  break;
339 
340  case SBP_detect_state::GET_SENDER:
341  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
342  state.n_read += 1;
343  if (state.n_read >= 2) {
344  state.n_read = 0;
345  state.state = SBP_detect_state::GET_LEN;
346  }
347  break;
348 
349  case SBP_detect_state::GET_LEN:
350  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
351  state.msg_len = data;
352  state.n_read = 0;
353  state.state = SBP_detect_state::GET_MSG;
354  break;
355 
356  case SBP_detect_state::GET_MSG:
357  if (state.msg_type == SBP_HEARTBEAT_MSGTYPE && state.n_read < 4) {
358  *((uint8_t*)&(state.heartbeat_buff) + state.n_read) = data;
359  }
360  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
361  state.n_read += 1;
362  if (state.n_read >= state.msg_len) {
363  state.n_read = 0;
364  state.state = SBP_detect_state::GET_CRC;
365  }
366  break;
367 
368  case SBP_detect_state::GET_CRC:
369  *((uint8_t*)&(state.crc) + state.n_read) = data;
370  state.n_read += 1;
371  if (state.n_read >= 2) {
372  state.state = SBP_detect_state::WAITING;
373  if (state.crc == state.crc_so_far
374  && state.msg_type == SBP_HEARTBEAT_MSGTYPE) {
375  struct sbp_heartbeat_t* heartbeat = ((struct sbp_heartbeat_t*)state.heartbeat_buff);
376  return heartbeat->protocol_major == 0;
377  }
378  return false;
379  }
380  break;
381 
382  default:
383  state.state = SBP_detect_state::WAITING;
384  break;
385  }
386  return false;
387 }

428  pages += (msg_len - 48) / 104 + 1;

432  LOG_PACKET_HEADER_INIT(LOG_MSG_SBPRAWH),
433  time_us : time_us,
434  msg_type : msg_type,
435  sender_id : sender_id,
436  index : 1,
437  pages : pages,
438  msg_len : msg_len,
439  };
440  memcpy(pkt.data, msg_buff, MIN(msg_len, 48));
441  AP::logger().WriteBlock(&pkt, sizeof(pkt));
442 
443  for (uint8_t i = 0; i < pages - 1; i++) {
444  struct log_SbpRAWM pkt2 = {
445  LOG_PACKET_HEADER_INIT(LOG_MSG_SBPRAWM),

449  index : uint8_t(i + 2),
450  pages : pages,
451  msg_len : msg_len,
452  };
453  memcpy(pkt2.data, &msg_buff[48 + i * 104], MIN(msg_len - (48 + i * 104), 104));
454  AP::logger().WriteBlock(&pkt2, sizeof(pkt2));
455  }
