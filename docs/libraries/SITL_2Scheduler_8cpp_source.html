150 void Scheduler::_run_timer_procs()
151 {
152  if (_in_timer_proc) {
153  // the timer calls took longer than the period of the
154  // timer. This is bad, and may indicate a serious
155  // driver failure. We can't just call the drivers
156  // again, as we could run out of stack. So we only
157  // call the _failsafe call. It's job is to detect if
158  // the drivers or the main loop are indeed dead and to
159  // activate whatever failsafe it thinks may help if
160  // need be. We assume the failsafe code can't
161  // block. If it does then we will recurse and die when
162  // we run out of stack
163  if (_failsafe != nullptr) {
164  _failsafe();
165  }
166  return;
167  }
168  _in_timer_proc = true;
169 
170  // now call the timer based drivers
171  for (int i = 0; i < _num_timer_procs; i++) {
172  if (_timer_proc[i]) {
173  _timer_proc[i]();
174  }
175  }
176 
177  // and the failsafe, if one is setup
178  if (_failsafe != nullptr) {
179  _failsafe();
180  }
181 
182  _in_timer_proc = false;
183 }
