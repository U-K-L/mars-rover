172 {

194 static int16_t
195 _atob (uint32_t *vp, char *p, int base)
196 {
197  uint32_t value, v1, v2;
198  char *q, tmp[20];
199  int digit;
200 
201  if (p[0] == '0' && (p[1] == 'x' || p[1] == 'X')) {
202  base = 16;
203  p += 2;
204  }
205 
206  if (base == 16 && (q = strchr (p, '.')) != 0) {
207  if ((unsigned)(q - p) > (unsigned)(sizeof(tmp) - 1))
208  return (0);
209 
210  strncpy (tmp, p, q - p);
211  tmp[q - p] = '\0';
212  if (!_atob (&v1, tmp, 16))
213  return (0);
214 
215  q++;
216  if (strchr (q, '.'))
217  return (0);
218 
219  if (!_atob (&v2, q, 16))
220  return (0);
221  *vp = (v1 << 16) + v2;
222  return (1);
223  }
224 
225  value = *vp = 0;
226  for (; *p; p++) {
227  if (*p >= '0' && *p <= '9')
228  digit = *p - '0';
229  else if (*p >= 'a' && *p <= 'f')
230  digit = *p - 'a' + 10;
231  else if (*p >= 'A' && *p <= 'F')
232  digit = *p - 'A' + 10;
233  else
234  return (0);
235 
236  if (digit >= base)
237  return (0);
238  value *= base;
239  value += digit;
240  }
241  *vp = value;
242  return (1);
243 }

246  * atob(vp,p,base)
247  * converts p to binary result in vp, rtn 1 on success
248  */
249 static int16_t atob(uint32_t *vp, char *p, int base)
250 {
251  uint32_t v;
252 
253  if (base == 0)
254  p = _getbase (p, &base);
255  if (_atob (&v, p, base)) {
256  *vp = v;
257  return (1);
258  }
259  return (0);

268 vsscanf (const char *buf, const char *s, va_list ap)
269 {
270  int count, noassign, base=0, lflag;
271  uint32_t width;
272  const char *tc;
273  char *t, tmp[MAXLN];
274 
275  count = noassign = width = lflag = 0;
276  while (*s && *buf) {
277  while (isspace ((unsigned char)(*s)))
278  s++;
279  if (*s == '%') {
280  s++;
281  for (; *s; s++) {
282  if (strchr ("dibouxcsefg%", *s))
283  break;
284  if (*s == '*')
285  noassign = 1;
286  else if (*s == 'l' || *s == 'L')
287  lflag = 1;
288  else if (*s >= '1' && *s <= '9') {
289  for (tc = s; isdigit ((unsigned)(*s)); s++);
290  strncpy (tmp, tc, s - tc);
291  tmp[s - tc] = '\0';
292  atob (&width, tmp, 10);
293  s--;
294  }
295  }
296  if (*s == 's') {
297  while (isspace ((unsigned char)(*buf)))
298  buf++;
299  if (!width)
300  width = strcspn (buf, ISSPACE);
301  if (!noassign) {
302  strncpy (t = va_arg (ap, char *), buf, width);
303  t[width] = '\0';
304  }
305  buf += width;
306  } else if (*s == 'c') {
307  if (!width)
308  width = 1;
309  if (!noassign) {
310  strncpy (t = va_arg (ap, char *), buf, width);
311  t[width] = '\0';
312  }
313  buf += width;
314  } else if (strchr ("dobxu", *s)) {
315  while (isspace ((unsigned char)(*buf)))
316  buf++;
317  if (*s == 'd' || *s == 'u')
318  base = 10;
319  else if (*s == 'x')
320  base = 16;
321  else if (*s == 'o')
322  base = 8;
323  else if (*s == 'b')
324  base = 2;
325  if (!width) {
326  if (isspace ((unsigned char)(*(s + 1))) || *(s + 1) == 0)
327  width = strcspn (buf, ISSPACE);
328  else
329  width = strchr (buf, *(s + 1)) - buf;
330  }
331  strncpy (tmp, buf, width);
332  tmp[width] = '\0';
333  buf += width;
334  if (!noassign)
335  atob (va_arg (ap, uint32_t *), tmp, base);
336  }
337  if (!noassign)
338  count++;
339  width = noassign = lflag = 0;
340  s++;
341  } else {
342  while (isspace ((unsigned char)(*buf)))
343  buf++;
344  if (*s != *buf)
345  break;
346  else
347  s++, buf++;
348  }
349  }
350  return (count);
351 }
