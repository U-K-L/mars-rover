181 void AP_OSD::init()
182 {
183  switch ((enum osd_types)osd_type.get()) {
184  case OSD_NONE:
185  default:
186  break;
187 
188  case OSD_MAX7456: {
189  AP_HAL::OwnPtr<AP_HAL::Device> spi_dev = std::move(hal.spi->get_device("osd"));
190  if (!spi_dev) {
191  break;
192  }
193  backend = AP_OSD_MAX7456::probe(*this, std::move(spi_dev));
194  if (backend == nullptr) {
195  break;
196  }
197  hal.console->printf("Started MAX7456 OSD\n");
198  break;
199  }
200 
201 #ifdef WITH_SITL_OSD
202  case OSD_SITL: {
203  backend = AP_OSD_SITL::probe(*this);
204  if (backend == nullptr) {
205  break;
206  }
207  hal.console->printf("Started SITL OSD\n");
208  break;
209  }
210 #endif
211  }
212  if (backend != nullptr) {
213  // create thread as higher priority than IO
214  hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_OSD::osd_thread, void), "OSD", 1024, AP_HAL::Scheduler::PRIORITY_IO, 1);
215  }
216 }

238 //update maximums and totals
239 void AP_OSD::stats()
240 {
241  uint32_t now = AP_HAL::millis();
242  if (!AP_Notify::flags.armed) {
243  last_update_ms = now;
244  return;
245  }
246 
247  // flight distance
248  uint32_t delta_ms = now - last_update_ms;
249  last_update_ms = now;
250 
251  AP_AHRS &ahrs = AP::ahrs();
252  Vector2f v = ahrs.groundspeed_vector();
253  float speed = v.length();
254  if (speed < 2.0) {
255  speed = 0.0;
256  }
257  float dist_m = (speed * delta_ms)*0.001;
258  last_distance_m += dist_m;
259 
260  // maximum ground speed
261  max_speed_mps = fmaxf(max_speed_mps,speed);
262 
263  // maximum distance
264  Location loc;
265  if (ahrs.get_position(loc) && ahrs.home_is_set()) {
266  const Location &home_loc = ahrs.get_home();
267  float distance = home_loc.get_distance(loc);
268  max_dist_m = fmaxf(max_dist_m, distance);
269  }
270 
271  // maximum altitude
272  float alt;
273  AP::ahrs().get_relative_position_D_home(alt);
274  alt = -alt;
275  max_alt_m = fmaxf(max_alt_m, alt);
276  // maximum current
277  AP_BattMonitor &battery = AP::battery();
278  float amps = battery.current_amps();
279  max_current_a = fmaxf(max_current_a, amps);
280 }

283 //Thanks to minimosd authors for the multiple osd screen idea
284 void AP_OSD::update_current_screen()
285 {
286  // Switch on ARM/DISARM event
287  if (AP_Notify::flags.armed){
288  if (!was_armed && arm_scr > 0 && arm_scr <= AP_OSD_NUM_SCREENS && screen[arm_scr-1].enabled){
289  current_screen = arm_scr-1;
290  }
291  was_armed = true;
292  } else if (was_armed) {
293  if (disarm_scr > 0 && disarm_scr <= AP_OSD_NUM_SCREENS && screen[disarm_scr-1].enabled){
294  current_screen = disarm_scr-1;
295  }
296  was_armed = false;
297  }
298 
299  // Switch on failsafe event
300  if (AP_Notify::flags.failsafe_radio || AP_Notify::flags.failsafe_battery) {
301  if (!was_failsafe && failsafe_scr > 0 && failsafe_scr <= AP_OSD_NUM_SCREENS && screen[failsafe_scr-1].enabled){
302  pre_fs_screen = current_screen;
303  current_screen = failsafe_scr-1;
304  }
305  was_failsafe = true;
306  } else if (was_failsafe) {
307  if (screen[pre_fs_screen].enabled){
308  current_screen = pre_fs_screen;
309  }
310  was_failsafe = false;
311  }
312 
313  if (rc_channel == 0) {
314  return;
315  }
316 
317  RC_Channel *channel = RC_Channels::rc_channel(rc_channel-1);
318  if (channel == nullptr) {
319  return;
320  }
321 
322  int16_t channel_value = channel->get_radio_in();
323  switch (sw_method) {
324  //switch to next screen if channel value was changed
325  default:
326  case TOGGLE:
327  if (previous_channel_value == 0) {
328  //do not switch to the next screen just after initialization
329  previous_channel_value = channel_value;
330  }
331  if (abs(channel_value-previous_channel_value) > 200) {
332  if (switch_debouncer) {
333  next_screen();
334  previous_channel_value = channel_value;
335  } else {
336  switch_debouncer = true;
337  return;
338  }
339  }
340  break;
341  //select screen based on pwm ranges specified
342  case PWM_RANGE:
343  for (int i=0; i<AP_OSD_NUM_SCREENS; i++) {
344  if (screen[i].enabled && screen[i].channel_min <= channel_value && screen[i].channel_max > channel_value && previous_pwm_screen != i) {
345  current_screen = previous_pwm_screen = i;
346  break;
347  }
348  }
349  break;
350  //switch to next screen after low to high transition and every 1s while channel value is high
351  case AUTO_SWITCH:
352  if (channel_value > channel->get_radio_trim()) {
353  if (switch_debouncer) {
354  uint32_t now = AP_HAL::millis();
355  if (now - last_switch_ms > 1000) {
356  next_screen();
357  last_switch_ms = now;
358  }
359  } else {
360  switch_debouncer = true;
361  return;
362  }
363  } else {
364  last_switch_ms = 0;
365  }
366  break;
367  }
368  switch_debouncer = false;
369 }

371 //select next avaliable screen, do nothing if all screens disabled
372 void AP_OSD::next_screen()
373 {
374  uint8_t t = current_screen;
375  do {
376  t = (t + 1)%AP_OSD_NUM_SCREENS;
377  } while (t != current_screen && !screen[t].enabled);
378  current_screen = t;
379 }
