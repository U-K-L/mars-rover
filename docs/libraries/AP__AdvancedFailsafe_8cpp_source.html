155 // check for Failsafe conditions. This is called at 10Hz by the main
156 // ArduPlane code
157 void
158 AP_AdvancedFailsafe::check(uint32_t last_heartbeat_ms, bool geofence_breached, uint32_t last_valid_rc_ms)
159 {
160  if (!_enable) {
161  return;
162  }
163 
164  // we always check for fence breach
165  if(_enable_geofence_fs) {
166  if (geofence_breached || check_altlimit()) {
167  if (!_terminate) {
168  gcs().send_text(MAV_SEVERITY_CRITICAL, "Terminating due to fence breach");
169  _terminate.set_and_notify(1);
170  }
171  }
172  }
173 
174  enum control_mode mode = afs_mode();
175 
176  // check if RC termination is enabled
177  // check for RC failure in manual mode or RC failure when AFS_RC_MANUAL is 0
178  if (_state != STATE_PREFLIGHT && !_terminate && _enable_RC_fs &&
179  (mode == AFS_MANUAL || mode == AFS_STABILIZED || !_rc_term_manual_only) &&
180  _rc_fail_time_seconds > 0 &&
181  (AP_HAL::millis() - last_valid_rc_ms) > (_rc_fail_time_seconds * 1000.0f)) {
182  gcs().send_text(MAV_SEVERITY_CRITICAL, "Terminating due to RC failure");
183  _terminate.set_and_notify(1);
184  }
185 
186  // tell the failsafe board if we are in manual control
187  // mode. This tells it to pass through controls from the
188  // receiver
189  if (_manual_pin != -1) {
190  hal.gpio->pinMode(_manual_pin, HAL_GPIO_OUTPUT);
191  hal.gpio->write(_manual_pin, mode==AFS_MANUAL);
192  }
193 
194  uint32_t now = AP_HAL::millis();
195  bool gcs_link_ok = ((now - last_heartbeat_ms) < 10000);
196  bool gps_lock_ok = ((now - AP::gps().last_fix_time_ms()) < 3000);
197 
198  switch (_state) {
199  case STATE_PREFLIGHT:
200  // we startup in preflight mode. This mode ends when
201  // we first enter auto.
202  if (mode == AFS_AUTO) {
203  gcs().send_text(MAV_SEVERITY_DEBUG, "AFS State: AFS_AUTO");
204  _state = STATE_AUTO;
205  }
206  break;
207 
208  case STATE_AUTO:
209  // this is the normal mode.
210  if (!gcs_link_ok) {
211  gcs().send_text(MAV_SEVERITY_DEBUG, "AFS State: DATA_LINK_LOSS");
212  _state = STATE_DATA_LINK_LOSS;
213  if (_wp_comms_hold) {
214  _saved_wp = mission.get_current_nav_cmd().index;
215  mission.set_current_cmd(_wp_comms_hold);
216  }
217  // if two events happen within 30s we consider it to be part of the same event
218  if (now - _last_comms_loss_ms > 30*1000UL) {
219  _comms_loss_count++;
220  _last_comms_loss_ms = now;
221  }
222  break;
223  }
224  if (!gps_lock_ok) {
225  gcs().send_text(MAV_SEVERITY_DEBUG, "AFS State: GPS_LOSS");
226  _state = STATE_GPS_LOSS;
227  if (_wp_gps_loss) {
228  _saved_wp = mission.get_current_nav_cmd().index;
229  mission.set_current_cmd(_wp_gps_loss);
230  }
231  // if two events happen within 30s we consider it to be part of the same event
232  if (now - _last_gps_loss_ms > 30*1000UL) {
233  _gps_loss_count++;
234  _last_gps_loss_ms = now;
235  }
236  break;
237  }
238  break;
239 
240  case STATE_DATA_LINK_LOSS:
241  if (!gps_lock_ok) {
242  // losing GPS lock when data link is lost
243  // leads to termination if AFS_DUAL_LOSS is 1
244  if(_enable_dual_loss) {
245  if (!_terminate) {
246  gcs().send_text(MAV_SEVERITY_CRITICAL, "Terminating due to dual loss");
247  _terminate.set_and_notify(1);
248  }
249  }
250  } else if (gcs_link_ok) {
251  _state = STATE_AUTO;
252  gcs().send_text(MAV_SEVERITY_DEBUG, "AFS State: AFS_AUTO, GCS now OK");
253  // we only return to the mission if we have not exceeded AFS_MAX_COM_LOSS
254  if (_saved_wp != 0 &&
255  (_max_comms_loss <= 0 ||
256  _comms_loss_count <= _max_comms_loss)) {
257  mission.set_current_cmd(_saved_wp);
258  _saved_wp = 0;
259  }
260  }
261  break;
262 
263  case STATE_GPS_LOSS:
264  if (!gcs_link_ok) {
265  // losing GCS link when GPS lock lost
266  // leads to termination if AFS_DUAL_LOSS is 1
267  if (!_terminate && _enable_dual_loss) {
268  gcs().send_text(MAV_SEVERITY_CRITICAL, "Terminating due to dual loss");
269  _terminate.set_and_notify(1);
270  }
271  } else if (gps_lock_ok) {
272  gcs().send_text(MAV_SEVERITY_DEBUG, "AFS State: AFS_AUTO, GPS now OK");
273  _state = STATE_AUTO;
274  // we only return to the mission if we have not exceeded AFS_MAX_GPS_LOSS
275  if (_saved_wp != 0 &&
276  (_max_gps_loss <= 0 || _gps_loss_count <= _max_gps_loss)) {
277  mission.set_current_cmd(_saved_wp);
278  _saved_wp = 0;
279  }
280  }
281  break;
282  }
283 
284  // if we are not terminating or if there is a separate terminate
285  // pin configured then toggle the heartbeat pin at 10Hz
286  if (_heartbeat_pin != -1 && (_terminate_pin != -1 || !_terminate)) {
287  _heartbeat_pin_value = !_heartbeat_pin_value;
288  hal.gpio->pinMode(_heartbeat_pin, HAL_GPIO_OUTPUT);
289  hal.gpio->write(_heartbeat_pin, _heartbeat_pin_value);
290  }
291 
292  // set the terminate pin
293  if (_terminate_pin != -1) {
294  hal.gpio->pinMode(_terminate_pin, HAL_GPIO_OUTPUT);
295  hal.gpio->write(_terminate_pin, _terminate?1:0);
296  }
297 }

317 // check for altitude limit breach
318 bool
319 AP_AdvancedFailsafe::check_altlimit(void)
320 {
321  if (!_enable) {
322  return false;
323  }
324  if (_amsl_limit == 0 || _qnh_pressure <= 0) {
325  // no limit set
326  return false;
327  }
328 
329  // see if the barometer is dead
330  const AP_Baro &baro = AP::baro();
331  const AP_GPS &gps = AP::gps();
332  if (AP_HAL::millis() - baro.get_last_update() > 5000) {
333  // the barometer has been unresponsive for 5 seconds. See if we can switch to GPS
334  if (_amsl_margin_gps != -1 &&
335  gps.status() >= AP_GPS::GPS_OK_FIX_3D &&
336  gps.location().alt*0.01f <= _amsl_limit - _amsl_margin_gps) {
337  // GPS based altitude OK
338  return false;
339  }
340  // no barometer - immediate termination
341  return true;
342  }
343 
344  float alt_amsl = baro.get_altitude_difference(_qnh_pressure*100, baro.get_pressure());
345  if (alt_amsl > _amsl_limit) {
346  // pressure altitude breach
347  return true;
348  }
349 
350  // all OK
351  return false;
352 }

354 /*
355  return true if we should crash the vehicle
356  */
357 bool AP_AdvancedFailsafe::should_crash_vehicle(void)
358 {
359  if (!_enable) {
360  return false;
361  }
362  // ensure failsafe values are setup for if FMU crashes on PX4/Pixhawk
363  if (!_failsafe_setup) {
364  _failsafe_setup = true;
365  setup_IO_failsafe();
366  }
367 
368  // determine if the vehicle should be crashed
369  // only possible if FS_TERM_ACTION is 42 and _terminate is non zero
370  // _terminate may be set via parameters, or a mavlink command
371  if (_terminate &&
372  (_terminate_action == TERMINATE_ACTION_TERMINATE ||
373  _terminate_action == TERMINATE_ACTION_LAND)) {
374  // we are terminating
375  return true;
376  }
377 
378  // continue flying
379  return false;
380 }

382 // update GCS based termination
383 // returns true if AFS is in the desired termination state
384 bool AP_AdvancedFailsafe::gcs_terminate(bool should_terminate, const char *reason) {
385  if (!_enable) {
386  gcs().send_text(MAV_SEVERITY_INFO, "AFS not enabled, can't terminate the vehicle");
387  return false;
388  }
389 
390  _terminate.set_and_notify(should_terminate ? 1 : 0);
391 
392  // evaluate if we will crash or not, as AFS must be enabled, and TERM_ACTION has to be correct
393  bool is_terminating = should_crash_vehicle();
394 
395  if(should_terminate == is_terminating) {
396  if (is_terminating) {
397  gcs().send_text(MAV_SEVERITY_INFO, "Terminating due to %s", reason);
398  } else {
399  gcs().send_text(MAV_SEVERITY_INFO, "Aborting termination due to %s", reason);
400  }
401  return true;
402  } else if (should_terminate && _terminate_action != TERMINATE_ACTION_TERMINATE) {
403  gcs().send_text(MAV_SEVERITY_INFO, "Unable to terminate, termination is not configured");
404  }
405  return false;
406 }
