48 /*
49  update the gimbal state
50 */
51 void Gimbal::update(void)
52 {
53  // calculate delta time in seconds
54  uint32_t now_us = AP_HAL::micros();
55 
56  float delta_t = (now_us - last_update_us) * 1.0e-6f;
57  last_update_us = now_us;
58 
59  Matrix3f vehicle_dcm;
60  fdm.quaternion.rotation_matrix(vehicle_dcm);
61 
62  const Vector3f &vehicle_gyro = AP::ins().get_gyro();
63  const Vector3f &vehicle_accel_body = AP::ins().get_accel();
64 
65  // take a copy of the demanded rates to bypass the limiter function for testing
66  Vector3f demRateRaw = demanded_angular_rate;
67 
68  // 1) Rotate the copters rotation rates into the gimbals frame of reference
69  // copterAngRate_G = transpose(DCMgimbal)*DCMcopter*copterAngRate
70  Vector3f copterAngRate_G = dcm.transposed()*vehicle_dcm*vehicle_gyro;
71 
72  // 2) Subtract the copters body rates to obtain a copter relative rotational
73  // rate vector (X,Y,Z) in gimbal sensor frame
74  // relativeGimbalRate(X,Y,Z) = gimbalRateDemand - copterAngRate_G
75  Vector3f relativeGimbalRate = demanded_angular_rate - copterAngRate_G;
76 
77  // calculate joint angles (euler312 order)
78  // calculate copter -> gimbal rotation matrix
79  Matrix3f rotmat_copter_gimbal = dcm.transposed() * vehicle_dcm;
80 
81  joint_angles = rotmat_copter_gimbal.transposed().to_euler312();
82 
83  /* 4) For each of the three joints, calculate upper and lower rate limits
84  from the corresponding angle limits and current joint angles
85 
86  upperRatelimit = (jointAngle - lowerAngleLimit) * travelLimitGain
87  lowerRatelimit = (jointAngle - upperAngleLimit) * travelLimitGain
88 
89  travelLimitGain is equal to the inverse of the bump stop time constant and
90  should be set to something like 20 initially. If set too high it can cause
91  the rates to 'ring' when they the limiter is in force, particularly given
92  we are using a first order numerical integration.
93  */
94  Vector3f upperRatelimit = -(joint_angles - upper_joint_limits) * travelLimitGain;
95  Vector3f lowerRatelimit = -(joint_angles - lower_joint_limits) * travelLimitGain;
96 
97  /*
98  5) Calculate the gimbal joint rates (roll, elevation, azimuth)
99 
100  gimbalJointRates(roll, elev, azimuth) = Matrix*relativeGimbalRate(X,Y,Z)
101 
102  where matrix =
103  +- -+
104  | cos(elevAngle), 0, sin(elevAngle) |
105  | |
106  | sin(elevAngle) tan(rollAngle), 1, -cos(elevAngle) tan(rollAngle) |
107  | |
108  | sin(elevAngle) cos(elevAngle) |
109  | - --------------, 0, -------------- |
110  | cos(rollAngle) cos(rollAngle) |
111  +- -+
112  */
113  float rollAngle = joint_angles.x;
114  float elevAngle = joint_angles.y;
115  Matrix3f matrix = Matrix3f(Vector3f(cosf(elevAngle), 0, sinf(elevAngle)),
116  Vector3f(sinf(elevAngle)*tanf(rollAngle), 1, -cosf(elevAngle)*tanf(rollAngle)),
117  Vector3f(-sinf(elevAngle)/cosf(rollAngle), 0, cosf(elevAngle)/cosf(rollAngle)));
118  Vector3f gimbalJointRates = matrix * relativeGimbalRate;
119 
120  // 6) Apply the rate limits from 4)
121  gimbalJointRates.x = constrain_float(gimbalJointRates.x, lowerRatelimit.x, upperRatelimit.x);
122  gimbalJointRates.y = constrain_float(gimbalJointRates.y, lowerRatelimit.y, upperRatelimit.y);
123  gimbalJointRates.z = constrain_float(gimbalJointRates.z, lowerRatelimit.z, upperRatelimit.z);
124  /*
125  7) Convert the modified gimbal joint rates to body rates (still copter
126  relative)
127  relativeGimbalRate(X,Y,Z) = Matrix * gimbalJointRates(roll, elev, azimuth)
128 
129  where Matrix =
130 
131  +- -+
132  | cos(elevAngle), 0, -cos(rollAngle) sin(elevAngle) |
133  | |
134  | 0, 1, sin(rollAngle) |
135  | |
136  | sin(elevAngle), 0, cos(elevAngle) cos(rollAngle) |
137  +- -+
138  */
139  matrix = Matrix3f(Vector3f(cosf(elevAngle), 0, -cosf(rollAngle)*sinf(elevAngle)),
140  Vector3f(0, 1, sinf(rollAngle)),
141  Vector3f(sinf(elevAngle), 0, cosf(elevAngle)*cosf(rollAngle)));
142  relativeGimbalRate = matrix * gimbalJointRates;
143 
144  // 8) Add to the result from step 1) to obtain the demanded gimbal body rates
145  // in an inertial frame of reference
146  // demandedGimbalRatesInertial(X,Y,Z) = relativeGimbalRate(X,Y,Z) + copterAngRate_G
147  // Vector3f demandedGimbalRatesInertial = relativeGimbalRate + copterAngRate_G;
148 
149  // for the moment we will set gyros equal to demanded_angular_rate
150  gimbal_angular_rate = demRateRaw; // demandedGimbalRatesInertial + true_gyro_bias - supplied_gyro_bias
151 
152  // update rotation of the gimbal
153  dcm.rotate(gimbal_angular_rate*delta_t);
154  dcm.normalize();
155 
156  // calculate copter -> gimbal rotation matrix
157  rotmat_copter_gimbal = dcm.transposed() * vehicle_dcm;
158 
159  // calculate joint angles (euler312 order)
160  joint_angles = rotmat_copter_gimbal.transposed().to_euler312();
161 
162  // update observed gyro
163  gyro = gimbal_angular_rate + true_gyro_bias;
164 
165  // update delta_angle (integrate)
166  delta_angle += gyro * delta_t;
167 
168  // calculate accel in gimbal body frame
169  Vector3f copter_accel_earth = vehicle_dcm * vehicle_accel_body;
170  Vector3f accel = dcm.transposed() * copter_accel_earth;
171 
172  // integrate velocity
173  delta_velocity += accel * delta_t;
174 
175  // see if we should do a report
176  send_report();
177 }

219 void Gimbal::param_send(const struct gimbal_param *p)
220 {
221  mavlink_message_t msg;
222  mavlink_param_value_t param_value;
223  strncpy(param_value.param_id, p->name, sizeof(param_value.param_id));
224  param_value.param_value = p->value;
225  param_value.param_count = 0;
226  param_value.param_index = 0;
227  param_value.param_type = MAV_PARAM_TYPE_REAL32;
228 
229  mavlink_status_t *chan0_status = mavlink_get_channel_status(MAVLINK_COMM_0);
230  uint8_t saved_seq = chan0_status->current_tx_seq;
231  chan0_status->current_tx_seq = mavlink.seq;
232  uint16_t len = mavlink_msg_param_value_encode(vehicle_system_id,
233  vehicle_component_id,
234  &msg, &param_value);
235  chan0_status->current_tx_seq = saved_seq;
236 
237  uint8_t msgbuf[len];
238  len = mavlink_msg_to_send_buffer(msgbuf, &msg);
239  if (len > 0) {
240  mav_socket.send(msgbuf, len);
241  }

245 /*
246  send a report to the vehicle control code over MAVLink
247 */
248 void Gimbal::send_report(void)
249 {
250  uint32_t now = AP_HAL::millis();
251  if (now < 10000) {
252  // don't send gimbal reports until 10s after startup. This
253  // avoids a windows threading issue with non-blocking sockets
254  // and the initial wait on uartA
255  return;
256  }
257  if (!mavlink.connected && mav_socket.connect(target_address, target_port)) {
258  ::printf("Gimbal connected to %s:%u\n", target_address, (unsigned)target_port);
259  mavlink.connected = true;
260  }
261  if (!mavlink.connected) {
262  return;
263  }
264 
265  if (param_send_last_ms && now - param_send_last_ms > 100) {
266  param_send(&gimbal_params[param_send_idx]);
267  if (++param_send_idx == ARRAY_SIZE(gimbal_params)) {
268  printf("Finished sending parameters\n");
269  param_send_last_ms = 0;
270  }
271  }
272 
273  // check for incoming MAVLink messages
274  uint8_t buf[100];
275  ssize_t ret;
276 
277  while ((ret=mav_socket.recv(buf, sizeof(buf), 0)) > 0) {
278  for (uint8_t i=0; i<ret; i++) {
279  mavlink_message_t msg;
280  mavlink_status_t status;
281  if (mavlink_frame_char_buffer(&mavlink.rxmsg, &mavlink.status,
282  buf[i],
283  &msg, &status) == MAVLINK_FRAMING_OK) {
284  switch (msg.msgid) {
285  case MAVLINK_MSG_ID_HEARTBEAT: {
286  mavlink_heartbeat_t pkt;
287  mavlink_msg_heartbeat_decode(&msg, &pkt);
288  printf("Gimbal: got HB type=%u autopilot=%u base_mode=0x%x\n", pkt.type, pkt.autopilot, pkt.base_mode);
289  if (!seen_heartbeat) {
290  seen_heartbeat = true;
291  vehicle_component_id = msg.compid;
292  vehicle_system_id = msg.sysid;
293  ::printf("Gimbal using srcSystem %u\n", (unsigned)vehicle_system_id);
294  }
295  break;
296  }
297  case MAVLINK_MSG_ID_GIMBAL_CONTROL: {
298  static uint32_t counter;
299  if (counter++ % 100 == 0) {
300  printf("GIMBAL_CONTROL %u\n", counter);
301  }
302  mavlink_gimbal_control_t pkt;
303  mavlink_msg_gimbal_control_decode(&msg, &pkt);
304  demanded_angular_rate = Vector3f(pkt.demanded_rate_x,
305  pkt.demanded_rate_y,
306  pkt.demanded_rate_z);
307  // no longer supply a bias
308  supplied_gyro_bias.zero();
309  seen_gimbal_control = true;
310  break;
311  }
312  case MAVLINK_MSG_ID_PARAM_SET: {
313  mavlink_param_set_t pkt;
314  mavlink_msg_param_set_decode(&msg, &pkt);
315  printf("Gimbal got PARAM_SET %.16s %f\n", pkt.param_id, pkt.param_value);
316 
317  struct gimbal_param *p = param_find(pkt.param_id);
318  if (p) {
319  p->value = pkt.param_value;
320  param_send(p);
321  }
322 
323  break;
324  }
325  case MAVLINK_MSG_ID_PARAM_REQUEST_LIST: {
326  mavlink_param_request_list_t pkt;
327  mavlink_msg_param_request_list_decode(&msg, &pkt);
328  if (pkt.target_system == 0 && pkt.target_component == MAV_COMP_ID_GIMBAL) {
329  // start param send
330  param_send_idx = 0;
331  param_send_last_ms = AP_HAL::millis();
332  }
333  printf("Gimbal sending %u parameters\n", (unsigned)ARRAY_SIZE(gimbal_params));
334  break;
335  }
336  default:
337  printf("Gimbal got unexpected msg %u\n", msg.msgid);
338  break;
339  }
340  }
341  }
342  }
343 
344  if (!seen_heartbeat) {
345  return;
346  }
347  mavlink_message_t msg;
348  uint16_t len;
349 
350  if (now - last_heartbeat_ms >= 1000) {
351  mavlink_heartbeat_t heartbeat;
352  heartbeat.type = MAV_TYPE_GIMBAL;
353  heartbeat.autopilot = MAV_AUTOPILOT_ARDUPILOTMEGA;
354  heartbeat.base_mode = 0;
355  heartbeat.system_status = 0;
356  heartbeat.mavlink_version = 0;
357  heartbeat.custom_mode = 0;
358 
359  /*
360  save and restore sequence number for chan0, as it is used by
361  generated encode functions
362  */
363  mavlink_status_t *chan0_status = mavlink_get_channel_status(MAVLINK_COMM_0);
364  uint8_t saved_seq = chan0_status->current_tx_seq;
365  chan0_status->current_tx_seq = mavlink.seq;
366  len = mavlink_msg_heartbeat_encode(vehicle_system_id,
367  vehicle_component_id,
368  &msg, &heartbeat);
369  chan0_status->current_tx_seq = saved_seq;
370 
371  mav_socket.send(&msg.magic, len);
372  last_heartbeat_ms = now;
373  }
374 
375  /*
376  send a GIMBAL_REPORT message
377  */
378  uint32_t now_us = AP_HAL::micros();
379  if (now_us - last_report_us > reporting_period_ms*1000UL) {
380  mavlink_gimbal_report_t gimbal_report;
381  float delta_time = (now_us - last_report_us) * 1.0e-6f;
382  last_report_us = now_us;
383  gimbal_report.target_system = vehicle_system_id;
384  gimbal_report.target_component = vehicle_component_id;
385  gimbal_report.delta_time = delta_time;
386  gimbal_report.delta_angle_x = delta_angle.x;
387  gimbal_report.delta_angle_y = delta_angle.y;
388  gimbal_report.delta_angle_z = delta_angle.z;
389  gimbal_report.delta_velocity_x = delta_velocity.x;
390  gimbal_report.delta_velocity_y = delta_velocity.y;
391  gimbal_report.delta_velocity_z = delta_velocity.z;
392  gimbal_report.joint_roll = joint_angles.x;
393  gimbal_report.joint_el = joint_angles.y;
394  gimbal_report.joint_az = joint_angles.z;
395 
396  mavlink_status_t *chan0_status = mavlink_get_channel_status(MAVLINK_COMM_0);
397  uint8_t saved_seq = chan0_status->current_tx_seq;
398  chan0_status->current_tx_seq = mavlink.seq;
399  len = mavlink_msg_gimbal_report_encode(vehicle_system_id,
400  vehicle_component_id,
401  &msg, &gimbal_report);
402  chan0_status->current_tx_seq = saved_seq;
403 
404  uint8_t msgbuf[len];
405  len = mavlink_msg_to_send_buffer(msgbuf, &msg);
406  if (len > 0) {
407  mav_socket.send(msgbuf, len);
408  }
409 
410  delta_velocity.zero();
411  delta_angle.zero();
412  }
413 }
