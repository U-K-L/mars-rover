62 /*
63  thread for handling UART send/receive
64 
65  We use events indexed by serial_num to trigger a more rapid send for
66  unbuffered_write uarts, and run at 1kHz for general UART handling
67  */
68 void UARTDriver::uart_thread(void* arg)
69 {
70  uint32_t last_thread_run_us = 0; // last time we did a 1kHz run of uarts
71 
72  uart_thread_ctx = chThdGetSelfX();
73  while (true) {
74  eventmask_t mask = chEvtWaitAnyTimeout(~0, chTimeMS2I(1));
75  uint32_t now = AP_HAL::micros();
76  if (now - last_thread_run_us >= 1000) {
77  // run them all if it's been more than 1ms since we ran
78  // them all
79  mask = ~0;
80  last_thread_run_us = now;
81  }
82  for (uint8_t i=0; i<UART_MAX_DRIVERS; i++) {
83  if (uart_drivers[i] == nullptr) {
84  continue;
85  }
86  if ((mask & EVENT_MASK(i)) &&
87  uart_drivers[i]->_initialised) {
88  uart_drivers[i]->_timer_tick();
89  }
90  }
91  }
92 }

124 void UARTDriver::begin(uint32_t b, uint16_t rxS, uint16_t txS)
125 {
126  thread_init();
127 
128  if (sdef.serial == nullptr) {
129  return;
130  }
131  uint16_t min_tx_buffer = 1024;
132  uint16_t min_rx_buffer = 512;
133 
134  if (sdef.is_usb) {
135  // give more buffer space for log download on USB
136  min_tx_buffer *= 4;
137  }
138 
139  // on PX4 we have enough memory to have a larger transmit and
140  // receive buffer for all ports. This means we don't get delays
141  // while waiting to write GPS config packets
142  if (txS < min_tx_buffer) {
143  txS = min_tx_buffer;
144  }
145  if (rxS < min_rx_buffer) {
146  rxS = min_rx_buffer;
147  }
148 
149  /*
150  allocate the read buffer
151  we allocate buffers before we successfully open the device as we
152  want to allocate in the early stages of boot, and cause minimum
153  thrashing of the heap once we are up. The ttyACM0 driver may not
154  connect for some time after boot
155  */
156  while (_in_timer) {
157  hal.scheduler->delay(1);
158  }
159  if (rxS != _readbuf.get_size()) {
160  _initialised = false;
161  _readbuf.set_size(rxS);
162  }
163 
164  bool clear_buffers = false;
165  if (b != 0) {
166  // clear buffers on baudrate change, but not on the console (which is usually USB)
167  if (_baudrate != b && hal.console != this) {
168  clear_buffers = true;
169  }
170  _baudrate = b;
171  }
172 
173  if (clear_buffers) {
174  _readbuf.clear();
175  }
176 
177  if (rx_bounce_buf == nullptr) {
178  rx_bounce_buf = (uint8_t *)hal.util->malloc_type(RX_BOUNCE_BUFSIZE, AP_HAL::Util::MEM_DMA_SAFE);
179  }
180  if (tx_bounce_buf == nullptr) {
181  tx_bounce_buf = (uint8_t *)hal.util->malloc_type(TX_BOUNCE_BUFSIZE, AP_HAL::Util::MEM_DMA_SAFE);
182  chVTObjectInit(&tx_timeout);
183  tx_bounce_buf_ready = true;
184  }
185 
186  /*
187  allocate the write buffer
188  */
189  while (_in_timer) {
190  hal.scheduler->delay(1);
191  }
192  if (txS != _writebuf.get_size()) {
193  _initialised = false;
194  _writebuf.set_size(txS);
195  }
196 
197  if (clear_buffers) {
198  _writebuf.clear();
199  }
200 
201  if (sdef.is_usb) {
202 #ifdef HAVE_USB_SERIAL
203  /*
204  * Initializes a serial-over-USB CDC driver.
205  */
206  if (!_device_initialised) {
207  sduObjectInit((SerialUSBDriver*)sdef.serial);
208  sduStart((SerialUSBDriver*)sdef.serial, &serusbcfg);
209  /*
210  * Activates the USB driver and then the USB bus pull-up on D+.
211  * Note, a delay is inserted in order to not have to disconnect the cable
212  * after a reset.
213  */
214  usbDisconnectBus(serusbcfg.usbp);
215  hal.scheduler->delay_microseconds(1500);
216  usbStart(serusbcfg.usbp, &usbcfg);
217  usbConnectBus(serusbcfg.usbp);
218  _device_initialised = true;
219  }
220 #endif
221  } else {
222 #if HAL_USE_SERIAL == TRUE
223  if (_baudrate != 0) {
224  bool was_initialised = _device_initialised;
225  //setup Rx DMA
226  if(!_device_initialised) {
227  if(sdef.dma_rx) {
228  osalDbgAssert(rxdma == nullptr, "double DMA allocation");
229  chSysLock();
230  rxdma = dmaStreamAllocI(sdef.dma_rx_stream_id,
231  12, //IRQ Priority
232  (stm32_dmaisr_t)rxbuff_full_irq,
233  (void *)this);
234  osalDbgAssert(rxdma, "stream alloc failed");
235  chSysUnlock();
236 #if defined(STM32F7) || defined(STM32H7)
237  dmaStreamSetPeripheral(rxdma, &((SerialDriver*)sdef.serial)->usart->RDR);
238 #else
239  dmaStreamSetPeripheral(rxdma, &((SerialDriver*)sdef.serial)->usart->DR);
240 #endif // STM32F7
241 #if STM32_DMA_SUPPORTS_DMAMUX
242  dmaSetRequestSource(rxdma, sdef.dma_rx_channel_id);
243 #endif
244  }
245  if (sdef.dma_tx) {
246  // we only allow for sharing of the TX DMA channel, not the RX
247  // DMA channel, as the RX side is active all the time, so
248  // cannot be shared
249  dma_handle = new Shared_DMA(sdef.dma_tx_stream_id,
250  SHARED_DMA_NONE,
251  FUNCTOR_BIND_MEMBER(&UARTDriver::dma_tx_allocate, void, Shared_DMA *),
252  FUNCTOR_BIND_MEMBER(&UARTDriver::dma_tx_deallocate, void, Shared_DMA *));
253  }
254  _device_initialised = true;
255  }
256  sercfg.speed = _baudrate;
257 
258  // start with options from set_options()
259  sercfg.cr1 = 0;
260  sercfg.cr2 = _cr2_options;
261  sercfg.cr3 = _cr3_options;
262 
263  if (!sdef.dma_tx && !sdef.dma_rx) {
264  } else {
265  if (sdef.dma_rx) {
266  sercfg.cr1 |= USART_CR1_IDLEIE;
267  sercfg.cr3 |= USART_CR3_DMAR;
268  }
269  if (sdef.dma_tx) {
270  sercfg.cr3 |= USART_CR3_DMAT;
271  }
272  }
273  sercfg.cr2 |= USART_CR2_STOP1_BITS;
274  sercfg.irq_cb = rx_irq_cb;
275  sercfg.ctx = (void*)this;
276 
277  sdStart((SerialDriver*)sdef.serial, &sercfg);
278 
279  if(sdef.dma_rx) {
280  //Configure serial driver to skip handling RX packets
281  //because we will handle them via DMA
282  ((SerialDriver*)sdef.serial)->usart->CR1 &= ~USART_CR1_RXNEIE;
283  //Start DMA
284  if(!was_initialised) {
285  uint32_t dmamode = STM32_DMA_CR_DMEIE | STM32_DMA_CR_TEIE;
286  dmamode |= STM32_DMA_CR_CHSEL(sdef.dma_rx_channel_id);
287  dmamode |= STM32_DMA_CR_PL(0);
288  dmaStreamSetMemory0(rxdma, rx_bounce_buf);
289  dmaStreamSetTransactionSize(rxdma, RX_BOUNCE_BUFSIZE);
290  dmaStreamSetMode(rxdma, dmamode | STM32_DMA_CR_DIR_P2M |
291  STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE);
292  dmaStreamEnable(rxdma);
293  }
294  }
295  }
296 #endif // HAL_USE_SERIAL
297  }
298 
299  if (_writebuf.get_size() && _readbuf.get_size()) {
300  _initialised = true;
301  }
302  _uart_owner_thd = chThdGetSelfX();
303 
304  // setup flow control
305  set_flow_control(_flow_control);
306 
307  if (serial_num == 0 && _initialised) {
308 #ifndef HAL_STDOUT_SERIAL
309  // setup hal.console to take printf() output
310  vprintf_console_hook = hal_console_vprintf;
311 #endif
312  }
313 }

347 /*
348  DMA transmit complettion interrupt handler
349  */
350 void UARTDriver::tx_complete(void* self, uint32_t flags)
351 {
352  UARTDriver* uart_drv = (UARTDriver*)self;
353  chSysLockFromISR();
354  if (!uart_drv->tx_bounce_buf_ready) {
355  // reset timeout
356  chVTResetI(&uart_drv->tx_timeout);
357 
358  uart_drv->_last_write_completed_us = AP_HAL::micros();
359  uart_drv->tx_bounce_buf_ready = true;
360  if (uart_drv->unbuffered_writes && uart_drv->_writebuf.available()) {
361  // trigger a rapid send of next bytes
362  chEvtSignalI(uart_thread_ctx, EVENT_MASK(uart_drv->serial_num));
363  }
364  uart_drv->dma_handle->unlock_from_IRQ();
365  }
366  chSysUnlockFromISR();
367 }

371 {
372 #if HAL_USE_SERIAL == TRUE
373  UARTDriver* uart_drv = (UARTDriver*)self;
374  if (!uart_drv->sdef.dma_rx) {

382  if(sr & USART_SR_IDLE) {
383  volatile uint16_t dr = ((SerialDriver*)(uart_drv->sdef.serial))->usart->DR;
384  (void)dr;
385  //disable dma, triggering DMA transfer complete interrupt
386  uart_drv->rxdma->stream->CR &= ~STM32_DMA_CR_EN;
387  }
388 #endif // STM32F7
389 #endif // HAL_USE_SERIAL

392 void UARTDriver::rxbuff_full_irq(void* self, uint32_t flags)
393 {
394 #if HAL_USE_SERIAL == TRUE
395  UARTDriver* uart_drv = (UARTDriver*)self;
396  if (uart_drv->_lock_rx_in_timer_tick) {
397  return;
398  }
399  if (!uart_drv->sdef.dma_rx) {
400  return;
401  }
402  cacheBufferInvalidate(uart_drv->rx_bounce_buf, RX_BOUNCE_BUFSIZE);
403  uint8_t len = RX_BOUNCE_BUFSIZE - dmaStreamGetTransactionSize(uart_drv->rxdma);
404  if (len > 0) {
405  if (uart_drv->half_duplex) {
406  uint32_t now = AP_HAL::micros();
407  if (now - uart_drv->hd_write_us < uart_drv->hd_read_delay_us) {
408  len = 0;
409  }
410  }
411 
412  cacheBufferInvalidate(uart_drv->rx_bounce_buf, len);
413  uart_drv->_readbuf.write(uart_drv->rx_bounce_buf, len);
414 
415  uart_drv->receive_timestamp_update();
416  }
417 
418  //restart the DMA transfers
419  dmaStreamSetMemory0(uart_drv->rxdma, uart_drv->rx_bounce_buf);
420  dmaStreamSetTransactionSize(uart_drv->rxdma, RX_BOUNCE_BUFSIZE);
421  dmaStreamEnable(uart_drv->rxdma);
422  if (uart_drv->_wait.thread_ctx && uart_drv->_readbuf.available() >= uart_drv->_wait.n) {
423  chSysLockFromISR();
424  chEvtSignalI(uart_drv->_wait.thread_ctx, EVT_DATA);
425  chSysUnlockFromISR();
426  }
427  if (uart_drv->_rts_is_active) {
428  uart_drv->update_rts_line();
429  }
430 #endif // HAL_USE_SERIAL
431 }

544 size_t UARTDriver::write(uint8_t c)
545 {
546  if (lock_write_key != 0 || !_write_mutex.take_nonblocking()) {
547  return 0;

549

551  _write_mutex.give();
552  return 0;

554

558  return 0;

560  hal.scheduler->delay(1);
561  }
562  size_t ret = _writebuf.write(&c, 1);
563  if (unbuffered_writes) {
564  write_pending_bytes();
565  }
566  _write_mutex.give();

570 size_t UARTDriver::write(const uint8_t *buffer, size_t size)
571 {
572  if (!_initialised || lock_write_key != 0) {
573  return 0;
574  }
575 
576  if (!_write_mutex.take_nonblocking()) {
577  return 0;
578  }
579 
580  if (_blocking_writes && !unbuffered_writes) {
581  /*
582  use the per-byte delay loop in write() above for blocking writes
583  */
584  _write_mutex.give();
585  size_t ret = 0;
586  while (size--) {
587  if (write(*buffer++) != 1) break;
588  ret++;
589  }
590  return ret;
591  }
592 
593  size_t ret = _writebuf.write(buffer, size);
594  if (unbuffered_writes) {
595  write_pending_bytes();
596  }
597  _write_mutex.give();
598  return ret;

694 {
695  check_dma_tx_completion();
696 
697  if (!tx_bounce_buf_ready) {
698  return;
699  }
700 
701  /* TX DMA channel preparation.*/
702  _total_written += tx_len;
703  _writebuf.advance(tx_len);
704  tx_len = _writebuf.peekbytes(tx_bounce_buf, MIN(n, TX_BOUNCE_BUFSIZE));
705  if (tx_len == 0) {
706  return;
707  }
708  if (!dma_handle->lock_nonblock()) {
709  tx_len = 0;
710  return;
711  }
712  if (dma_handle->has_contention()) {
713  /*
714  someone else is using this same DMA channel. To reduce
715  latency we will drop the TX size with DMA on this UART to
716  keep TX times below 250us. This can still suffer from long
717  times due to CTS blockage
718  */
719  uint32_t max_tx_bytes = 1 + (_baudrate * 250UL / 1000000UL);
720  if (tx_len > max_tx_bytes) {
721  tx_len = max_tx_bytes;
722  }
723  }
724 
725  if (half_duplex) {
726  half_duplex_setup_delay(tx_len);
727  }
728 
729  tx_bounce_buf_ready = false;
730  osalDbgAssert(txdma != nullptr, "UART TX DMA allocation failed");
731  cacheBufferFlush(tx_bounce_buf, tx_len);
732  dmaStreamSetMemory0(txdma, tx_bounce_buf);
733  dmaStreamSetTransactionSize(txdma, tx_len);
734  uint32_t dmamode = STM32_DMA_CR_DMEIE | STM32_DMA_CR_TEIE;
735  dmamode |= STM32_DMA_CR_CHSEL(sdef.dma_tx_channel_id);
736  dmamode |= STM32_DMA_CR_PL(0);
737  dmaStreamSetMode(txdma, dmamode | STM32_DMA_CR_DIR_M2P |
738  STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE);
739  dmaStreamEnable(txdma);
740  uint32_t timeout_us = ((1000000UL * (tx_len+2) * 10) / _baudrate) + 500;
741  chVTSet(&tx_timeout, chTimeUS2I(timeout_us), handle_tx_timeout, this);
742 }

787 /*
788  write any pending bytes to the device
789  */
790 void UARTDriver::write_pending_bytes(void)
791 {
792  uint32_t n;
793 
794  if (sdef.dma_tx) {
795  check_dma_tx_completion();
796  }
797 
798  // write any pending bytes
799  n = _writebuf.available();
800  if (n <= 0) {
801  return;
802  }
803 
804  if (!sdef.dma_tx) {
805  write_pending_bytes_NODMA(n);
806  } else {
807  write_pending_bytes_DMA(n);
808  }
809 
810  // handle AUTO flow control mode
811  if (_flow_control == FLOW_CONTROL_AUTO) {
812  if (_first_write_started_us == 0) {
813  _first_write_started_us = AP_HAL::micros();
814  }
815  if (sdef.dma_tx) {
816  // when we are using DMA we have a reliable indication that a write
817  // has completed from the DMA completion interrupt
818  if (_last_write_completed_us != 0) {
819  _flow_control = FLOW_CONTROL_ENABLE;
820  return;
821  }
822  } else {
823  // without DMA we need to look at the number of bytes written into the queue versus the
824  // remaining queue space
825  uint32_t space = qSpaceI(&((SerialDriver*)sdef.serial)->oqueue);
826  uint32_t used = SERIAL_BUFFERS_SIZE - space;
827  // threshold is 8 for the GCS_Common code to unstick SiK radios, which
828  // sends 6 bytes with flow control disabled
829  const uint8_t threshold = 8;
830  if (_total_written > used && _total_written - used > threshold) {
831  _flow_control = FLOW_CONTROL_ENABLE;
832  return;
833  }
834  }
835  if (AP_HAL::micros() - _first_write_started_us > 500*1000UL) {
836  // it doesn't look like hw flow control is working
837  hal.console->printf("disabling flow control on serial %u\n", sdef.get_index());
838  set_flow_control(FLOW_CONTROL_DISABLE);
839  }
840  }
841 }

856 /*
857  push any pending bytes to/from the serial port. This is called at
858  1kHz in the timer thread. Doing it this way reduces the system call
859  overhead in the main task enormously.
860  */
861 void UARTDriver::_timer_tick(void)
862 {
863  if (!_initialised) return;
864 
865  if (sdef.dma_rx && rxdma) {
866  _lock_rx_in_timer_tick = true;
867  //Check if DMA is enabled
868  //if not, it might be because the DMA interrupt was silenced
869  //let's handle that here so that we can continue receiving
870  if (!(rxdma->stream->CR & STM32_DMA_CR_EN)) {
871  uint8_t len = RX_BOUNCE_BUFSIZE - dmaStreamGetTransactionSize(rxdma);
872  if (len != 0) {
873  cacheBufferInvalidate(rx_bounce_buf, len);
874  _readbuf.write(rx_bounce_buf, len);
875 
876  receive_timestamp_update();
877  if (_rts_is_active) {
878  update_rts_line();
879  }
880  }
881  //DMA disabled by idle interrupt never got a chance to be handled
882  //we will enable it here
883  dmaStreamSetMemory0(rxdma, rx_bounce_buf);
884  dmaStreamSetTransactionSize(rxdma, RX_BOUNCE_BUFSIZE);
885  dmaStreamEnable(rxdma);
886  }
887  _lock_rx_in_timer_tick = false;
888  }
889 
890  // don't try IO on a disconnected USB port
891  if (sdef.is_usb) {
892 #ifdef HAVE_USB_SERIAL
893  if (((SerialUSBDriver*)sdef.serial)->config->usbp->state != USB_ACTIVE) {
894  return;
895  }
896 #endif
897  }
898  if(sdef.is_usb) {
899 #ifdef HAVE_USB_SERIAL
900  ((GPIO *)hal.gpio)->set_usb_connected();
901 #endif
902  }
903  _in_timer = true;
904 
905  if (!sdef.dma_rx) {
906  // try to fill the read buffer
907  ByteBuffer::IoVec vec[2];
908 
909  const auto n_vec = _readbuf.reserve(vec, _readbuf.space());
910  for (int i = 0; i < n_vec; i++) {
911  int ret = 0;
912  //Do a non-blocking read
913  if (sdef.is_usb) {
914  #ifdef HAVE_USB_SERIAL
915  ret = chnReadTimeout((SerialUSBDriver*)sdef.serial, vec[i].data, vec[i].len, TIME_IMMEDIATE);
916  #endif
917  } else {
918 #if HAL_USE_SERIAL == TRUE
919  ret = chnReadTimeout((SerialDriver*)sdef.serial, vec[i].data, vec[i].len, TIME_IMMEDIATE);
920 #endif
921  }
922  if (ret < 0) {
923  break;
924  }
925  if (half_duplex) {
926  uint32_t now = AP_HAL::micros();
927  if (now - hd_write_us < hd_read_delay_us) {
928  break;
929  }
930  }
931  _readbuf.commit((unsigned)ret);
932 
933  receive_timestamp_update();
934 
935  /* stop reading as we read less than we asked for */
936  if ((unsigned)ret < vec[i].len) {
937  break;
938  }
939  }
940  }
941  if (_wait.thread_ctx && _readbuf.available() >= _wait.n) {
942  chEvtSignal(_wait.thread_ctx, EVT_DATA);
943  }
944  if (unbuffered_writes) {
945  // now send pending bytes. If we are doing "unbuffered" writes
946  // then the send normally happens as soon as the bytes are
947  // provided by the write() call, but if the write is larger
948  // than the DMA buffer size then there can be extra bytes to
949  // send here, and they must be sent with the write lock held
950  _write_mutex.take(HAL_SEMAPHORE_BLOCK_FOREVER);
951  write_pending_bytes();
952  _write_mutex.give();
953  } else {
954  write_pending_bytes();
955  }
956 
957  _in_timer = false;
958 }

960 /*
961  change flow control mode for port
962  */
963 void UARTDriver::set_flow_control(enum flow_control flowcontrol)
964 {
965  if (sdef.rts_line == 0 || sdef.is_usb) {
966  // no hw flow control available
967  return;
968  }
969 #if HAL_USE_SERIAL == TRUE
970  SerialDriver *sd = (SerialDriver*)(sdef.serial);
971  _flow_control = flowcontrol;
972  if (!_initialised) {
973  // not ready yet, we just set variable for when we call begin
974  return;
975  }
976  switch (_flow_control) {
977 
978  case FLOW_CONTROL_DISABLE:
979  // force RTS active when flow disabled
980  palSetLineMode(sdef.rts_line, 1);
981  palClearLine(sdef.rts_line);
982  _rts_is_active = true;
983  // disable hardware CTS support
984  chSysLock();
985  if ((sd->usart->CR3 & (USART_CR3_CTSE | USART_CR3_RTSE)) != 0) {
986  sd->usart->CR1 &= ~USART_CR1_UE;
987  sd->usart->CR3 &= ~(USART_CR3_CTSE | USART_CR3_RTSE);
988  sd->usart->CR1 |= USART_CR1_UE;
989  }
990  chSysUnlock();
991  break;
992 
993  case FLOW_CONTROL_AUTO:
994  // reset flow control auto state machine
995  _first_write_started_us = 0;
996  _last_write_completed_us = 0;
997  FALLTHROUGH;
998 
999  case FLOW_CONTROL_ENABLE:
1000  // we do RTS in software as STM32 hardware RTS support toggles
1001  // the pin for every byte which loses a lot of bandwidth
1002  palSetLineMode(sdef.rts_line, 1);
1003  palClearLine(sdef.rts_line);
1004  _rts_is_active = true;
1005  // enable hardware CTS support, disable RTS support as we do that in software
1006  chSysLock();
1007  if ((sd->usart->CR3 & (USART_CR3_CTSE | USART_CR3_RTSE)) != USART_CR3_CTSE) {
1008  // CTSE and RTSE can only be written when uart is disabled
1009  sd->usart->CR1 &= ~USART_CR1_UE;
1010  sd->usart->CR3 |= USART_CR3_CTSE;
1011  sd->usart->CR3 &= ~USART_CR3_RTSE;
1012  sd->usart->CR1 |= USART_CR1_UE;
1013  }
1014  chSysUnlock();
1015  break;
1016  }
1017 #endif // HAL_USE_SERIAL
1018 }
