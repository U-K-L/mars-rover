56  change what data is requested from XPlane. This saves the user from

59 void XPlane::select_data(uint64_t usel_mask, uint64_t sel_mask)
60 {
61  struct PACKED {
62  uint8_t marker[5] { 'D', 'S', 'E', 'L', '0' };
63  uint32_t data[8] {};
64  } dsel;
65  uint8_t count = 0;
66  for (uint8_t i=0; i<64 && count<8; i++) {
67  if ((((uint64_t)1)<<i) & sel_mask) {
68  dsel.data[count++] = i;
69  printf("i=%u\n", (unsigned)i);
70  }
71  }
72  if (count != 0) {
73  socket_out.send(&dsel, sizeof(dsel));
74  printf("Selecting %u data types 0x%llx\n", (unsigned)count, (unsigned long long)sel_mask);
75  }
76 
77  struct PACKED {
78  uint8_t marker[5] { 'U', 'S', 'E', 'L', '0' };
79  uint32_t data[8] {};
80  } usel;
81  count = 0;
82 
83  // only de-select an output once, so we don't fight the user
84  usel_mask &= ~unselected_mask;
85  unselected_mask |= usel_mask;
86 
87  for (uint8_t i=0; i<64 && count<8; i++) {
88  if ((((uint64_t)1)<<i) & usel_mask) {
89  usel.data[count++] = i;
90  }
91  }
92  if (count != 0) {
93  socket_out.send(&usel, sizeof(usel));
94  printf("De-selecting %u data types 0x%llx\n", (unsigned)count, (unsigned long long)usel_mask);
95  }
96 }

98 /*
99  receive data from X-Plane via UDP
100 */
101 bool XPlane::receive_data(void)
102 {
103  uint8_t pkt[10000];
104  uint8_t *p = &pkt[5];
105  const uint8_t pkt_len = 36;
106  uint64_t data_mask = 0;
107  const uint64_t one = 1U;
108  const uint64_t required_mask = (one<<Times | one<<LatLonAlt | one<<Speed | one<<PitchRollHeading |
109  one<<LocVelDistTraveled | one<<AngularVelocities | one<<Gload |
110  one << Joystick1 | one << ThrottleCommand | one << Trim |
111  one << PropPitch | one << EngineRPM | one << PropRPM | one << Generator |
112  one << Mixture);
113  Location loc {};
114  Vector3f pos;
115  uint32_t wait_time_ms = 1;
116  uint32_t now = AP_HAL::millis();
117 
118  // if we are about to get another frame from X-Plane then wait longer
119  if (xplane_frame_time > wait_time_ms &&
120  now+1 >= last_data_time_ms + xplane_frame_time) {
121  wait_time_ms = 10;
122  }
123  ssize_t len = socket_in.recv(pkt, sizeof(pkt), wait_time_ms);
124 
125  if (len < pkt_len+5 || memcmp(pkt, "DATA", 4) != 0) {
126  // not a data packet we understand
127  goto failed;
128  }
129  len -= 5;
130 
131  if (!connected) {
132  // we now know the IP X-Plane is using
133  uint16_t port;
134  socket_in.last_recv_address(xplane_ip, port);
135  socket_out.connect(xplane_ip, xplane_port);
136  connected = true;
137  printf("Connected to %s:%u\n", xplane_ip, (unsigned)xplane_port);
138  }
139 
140  while (len >= pkt_len) {
141  const float *data = (const float *)p;
142  uint8_t code = p[0];
143  // keep a mask of what codes we have received
144  if (code < 64) {
145  data_mask |= (((uint64_t)1) << code);
146  }
147  switch (code) {
148  case Times: {
149  uint64_t tus = data[3] * 1.0e6f;
150  if (tus + time_base_us <= time_now_us) {
151  uint64_t tdiff = time_now_us - (tus + time_base_us);
152  if (tdiff > 1e6f) {
153  printf("X-Plane time reset %lu\n", (unsigned long)tdiff);
154  }
155  time_base_us = time_now_us - tus;
156  }
157  uint64_t tnew = time_base_us + tus;
158  //uint64_t dt = tnew - time_now_us;
159  //printf("dt %u\n", (unsigned)dt);
160  time_now_us = tnew;
161  break;
162  }
163 
164  case LatLonAlt: {
165  loc.lat = data[1] * 1e7;
166  loc.lng = data[2] * 1e7;
167  loc.alt = data[3] * FEET_TO_METERS * 100.0f;
168  const float altitude_above_ground = data[4] * FEET_TO_METERS;
169  ground_level = loc.alt * 0.01f - altitude_above_ground;
170  break;
171  }
172 
173  case Speed:
174  airspeed = data[2] * KNOTS_TO_METERS_PER_SECOND;
175  airspeed_pitot = airspeed;
176  break;
177 
178  case AoA:
179  // ignored
180  break;
181 
182  case Trim:
183  if (heli_frame) {
184  // use flaps for collective as no direct collective data input
185  rcin[2] = data[4];
186  }
187  break;
188 
189  case PitchRollHeading: {
190  float roll, pitch, yaw;
191  pitch = radians(data[1]);
192  roll = radians(data[2]);
193  yaw = radians(data[3]);
194  dcm.from_euler(roll, pitch, yaw);
195  break;
196  }
197 
198  case AtmosphereWeather:
199  // ignored
200  break;
201 
202  case LocVelDistTraveled:
203  pos.y = data[1];
204  pos.z = -data[2];
205  pos.x = -data[3];
206  velocity_ef.y = data[4];
207  velocity_ef.z = -data[5];
208  velocity_ef.x = -data[6];
209  break;
210 
211  case AngularVelocities:
212  gyro.y = data[1];
213  gyro.x = data[2];
214  gyro.z = data[3];
215  break;
216 
217  case Gload:
218  accel_body.z = -data[5] * GRAVITY_MSS;
219  accel_body.x = data[6] * GRAVITY_MSS;
220  accel_body.y = data[7] * GRAVITY_MSS;
221  break;
222 
223  case Joystick1:
224  rcin_chan_count = 4;
225  rcin[0] = (data[2] + 1)*0.5f;
226  rcin[1] = (data[1] + 1)*0.5f;
227  rcin[3] = (data[3] + 1)*0.5f;
228  break;
229 
230  case ThrottleCommand: {
231  if (!heli_frame) {
232  /* getting joystick throttle input is very weird. The
233  * problem is that XPlane sends the ThrottleCommand packet
234  * both for joystick throttle input and for throttle that
235  * we have provided over the link. So we need some way to
236  * detect when we get our own values back. The trick used
237  * is to add throttle_magic to the values we send, then
238  * detect this offset in the data coming back. Very ugly,
239  * but I can't find a better way of allowing joystick
240  * input from XPlane10
241  */
242  bool has_magic = ((uint32_t)(data[1] * throttle_magic_scale) % 1000U) == (uint32_t)(throttle_magic * throttle_magic_scale);
243 #pragma GCC diagnostic push
244 #pragma GCC diagnostic ignored "-Wfloat-equal"
245  if (data[1] < 0 ||
246  data[1] == throttle_sent ||
247  has_magic) {
248  break;
249  }
250 #pragma GCC diagnostic pop
251  rcin[2] = data[1];
252  }
253  break;
254  }
255 
256  case PropPitch: {
257  break;
258  }
259 
260  case EngineRPM:
261  rpm1 = data[1];
262  break;
263 
264  case PropRPM:
265  rpm2 = data[1];
266  break;
267 
268  case Joystick2:
269  break;
270 
271  case Generator:
272  /*
273  in order to get interlock switch on helis we map the
274  "generator1 on/off" function of XPlane 10 to channel 8.
275  */
276  rcin_chan_count = 8;
277  rcin[7] = data[1];
278  break;
279 
280  case Mixture:
281  // map channel 6 and 7 from Mixture3 and Mixture4 for extra channels
282  rcin_chan_count = MAX(7, rcin_chan_count);
283  rcin[5] = data[3];
284  rcin[6] = data[4];
285  break;
286  }
287  len -= pkt_len;
288  p += pkt_len;
289  }
290 
291  if (data_mask != required_mask) {
292  // ask XPlane to change what data it sends
293  uint64_t usel = data_mask & ~required_mask;
294  uint64_t sel = required_mask & ~data_mask;
295  usel &= ~unselected_mask;
296  if (usel || sel) {
297  select_data(usel, sel);
298  goto failed;
299  }
300  }
301  position = pos + position_zero;
302  update_position();
303  time_advance();
304 
305  accel_earth = dcm * accel_body;
306  accel_earth.z += GRAVITY_MSS;
307 
308  // the position may slowly deviate due to float accuracy and longitude scaling
309  if (loc.get_distance(location) > 4 || abs(loc.alt - location.alt)*0.01f > 2.0f) {
310  printf("X-Plane home reset dist=%f alt=%.1f/%.1f\n",
311  loc.get_distance(location), loc.alt*0.01f, location.alt*0.01f);
312  // reset home location
313  position_zero(-pos.x, -pos.y, -pos.z);
314  home.lat = loc.lat;
315  home.lng = loc.lng;
316  home.alt = loc.alt;
317  position.x = 0;
318  position.y = 0;
319  position.z = 0;
320  update_position();
321  time_advance();
322  }
323 
324  update_mag_field_bf();
325 
326  if (now > last_data_time_ms && now - last_data_time_ms < 100) {
327  xplane_frame_time = now - last_data_time_ms;
328  }
329  last_data_time_ms = AP_HAL::millis();
330 
331  report.data_count++;
332  report.frame_count++;
333  return true;
334 
335 failed:
336  if (AP_HAL::millis() - last_data_time_ms > 200) {
337  // don't extrapolate beyond 0.2s
338  return false;
339  }
340 
341  // advance time by 1ms
342  frame_time_us = 1000;
343  float delta_time = frame_time_us * 1e-6f;
344 
345  time_now_us += frame_time_us;
346 
347  extrapolate_sensors(delta_time);
348 
349  update_position();
350  time_advance();
351  update_mag_field_bf();
352  report.frame_count++;
353  return false;
354 }

357 /*
358  send data to X-Plane via UDP
359 */
360 void XPlane::send_data(const struct sitl_input &input)
361 {
362  float aileron = (input.servos[0]-1500)/500.0f;
363  float elevator = (input.servos[1]-1500)/500.0f;
364  float throttle = (input.servos[2]-1000)/1000.0;
365  float rudder = (input.servos[3]-1500)/500.0f;
366  struct PACKED {
367  uint8_t marker[5] { 'D', 'A', 'T', 'A', '0' };
368  uint32_t code;
369  float data[8];
370  } d {};
371 
372  if (input.servos[0] == 0) {
373  aileron = 0;
374  }
375  if (input.servos[1] == 0) {
376  elevator = 0;
377  }
378  if (input.servos[2] == 0) {
379  throttle = 0;
380  }
381  if (input.servos[3] == 0) {
382  rudder = 0;
383  }
384 
385  // we add the throttle_magic to the throttle value we send so we
386  // can detect when we get it back
387  throttle = ((uint32_t)(throttle * 1000)) * 1.0e-3f + throttle_magic;
388 
389  uint8_t flap_chan;
390  if (SRV_Channels::find_channel(SRV_Channel::k_flap, flap_chan) ||
391  SRV_Channels::find_channel(SRV_Channel::k_flap_auto, flap_chan)) {
392  float flap = (input.servos[flap_chan]-1000)/1000.0;
393  if (!is_equal(flap, last_flap)) {
394  send_dref("sim/flightmodel/controls/flaprqst", flap);
395  send_dref("sim/aircraft/overflow/acf_flap_arm", flap>0?1:0);
396  }
397  }
398 
399  d.code = FlightCon;
400  d.data[0] = elevator;
401  d.data[1] = aileron;
402  d.data[2] = rudder;
403  d.data[4] = rudder;
404  socket_out.send(&d, sizeof(d));
405 
406  if (!heli_frame) {
407  d.code = ThrottleCommand;
408  d.data[0] = throttle;
409  d.data[1] = throttle;
410  d.data[2] = throttle;
411  d.data[3] = throttle;
412  d.data[4] = 0;
413  socket_out.send(&d, sizeof(d));
414  } else {
415  // send chan3 as collective pitch, on scale from -10 to +10
416  float collective = 10*(input.servos[2]-1500)/500.0;
417 
418  // and send throttle from channel 8
419  throttle = (input.servos[7]-1000)/1000.0;
420 
421  // allow for extra throttle outputs for special aircraft
422  float throttle2 = (input.servos[5]-1000)/1000.0;
423  float throttle3 = (input.servos[6]-1000)/1000.0;
424 
425  d.code = PropPitch;
426  d.data[0] = collective;
427  d.data[1] = -rudder*15; // reverse sense of rudder, 15 degrees pitch range
428  d.data[2] = 0;
429  d.data[3] = 0;
430  d.data[4] = 0;
431  socket_out.send(&d, sizeof(d));
432 
433  d.code = ThrottleCommand;
434  d.data[0] = throttle;
435  d.data[1] = throttle;
436  d.data[2] = throttle2;
437  d.data[3] = throttle3;
438  d.data[4] = 0;
439  socket_out.send(&d, sizeof(d));
440  }
441 
442  throttle_sent = throttle;
443 }

461 /*
462  update the XPlane simulation by one time step
463  */
464 void XPlane::update(const struct sitl_input &input)
465 {
466  if (receive_data()) {
467  send_data(input);
468  }
469 
470  uint32_t now = AP_HAL::millis();
471  if (report.last_report_ms == 0) {
472  report.last_report_ms = now;
473  }
474  if (now - report.last_report_ms > 5000) {
475  float dt = (now - report.last_report_ms) * 1.0e-3f;
476  printf("Data rate: %.1f FPS Frame rate: %.1f FPS\n",
477  report.data_count/dt, report.frame_count/dt);
478  report.last_report_ms = now;
479  report.data_count = 0;
480  report.frame_count = 0;
481  }
482 }
