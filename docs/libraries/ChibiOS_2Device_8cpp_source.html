39 /*
40  per-bus callback thread
41 */
42 void DeviceBus::bus_thread(void *arg)
43 {
44  struct DeviceBus *binfo = (struct DeviceBus *)arg;
45 
46  while (true) {
47  uint64_t now = AP_HAL::micros64();
48  DeviceBus::callback_info *callback;
49 
50  // find a callback to run
51  for (callback = binfo->callbacks; callback; callback = callback->next) {
52  if (now >= callback->next_usec) {
53  while (now >= callback->next_usec) {
54  callback->next_usec += callback->period_usec;
55  }
56  // call it with semaphore held
57  if (binfo->semaphore.take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
58  callback->cb();
59  binfo->semaphore.give();
60  }
61  }
62  }
63 
64  // work out when next loop is needed
65  uint64_t next_needed = 0;
66  now = AP_HAL::micros64();
67 
68  for (callback = binfo->callbacks; callback; callback = callback->next) {
69  if (next_needed == 0 ||
70  callback->next_usec < next_needed) {
71  next_needed = callback->next_usec;
72  if (next_needed < now) {
73  next_needed = now;
74  }
75  }
76  }
77 
78  // delay for at most 50ms, to handle newly added callbacks
79  uint32_t delay = 50000;
80  if (next_needed >= now && next_needed - now < delay) {
81  delay = next_needed - now;
82  }
83  // don't delay for less than 100usec, so one thread doesn't
84  // completely dominate the CPU
85  if (delay < 100) {
86  delay = 100;
87  }
88  hal.scheduler->delay_microseconds(delay);
89  }
90  return;
91 }

94 AP_HAL::Device::PeriodicHandle DeviceBus::register_periodic_callback(uint32_t period_usec, AP_HAL::Device::PeriodicCb cb, AP_HAL::Device *_hal_device)
95 {
96  if (!thread_started) {
97  thread_started = true;
98 
99  hal_device = _hal_device;
100  // setup a name for the thread
101  const uint8_t name_len = 7;
102  char *name = (char *)malloc(name_len);
103  switch (hal_device->bus_type()) {
104  case AP_HAL::Device::BUS_TYPE_I2C:
105  snprintf(name, name_len, "I2C:%u",
106  hal_device->bus_num());
107  break;
108 
109  case AP_HAL::Device::BUS_TYPE_SPI:
110  snprintf(name, name_len, "SPI:%u",
111  hal_device->bus_num());
112  break;
113  default:
114  break;
115  }
116 
117  thread_ctx = thread_create_alloc(THD_WORKING_AREA_SIZE(1024),
118  name,
119  thread_priority, /* Initial priority. */
120  DeviceBus::bus_thread, /* Thread function. */
121  this); /* Thread parameter. */
122  if (thread_ctx == nullptr) {
123  AP_HAL::panic("Failed to create bus thread %s", name);
124  }
125  }
126  DeviceBus::callback_info *callback = new DeviceBus::callback_info;
127  if (callback == nullptr) {
128  return nullptr;
129  }
130  callback->cb = cb;
131  callback->period_usec = period_usec;
132  callback->next_usec = AP_HAL::micros64() + period_usec;
133 
134  // add to linked list of callbacks on thread
135  callback->next = callbacks;
136  callbacks = callback;
137 
138  return callback;
139 }

147 {

151 
152  DeviceBus::callback_info *callback = static_cast<DeviceBus::callback_info *>(h);
