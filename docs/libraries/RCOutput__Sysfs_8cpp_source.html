42 void RCOutput_Sysfs::init()
43 {
44  for (uint8_t i = 0; i < _channel_count; i++) {
45 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_DISCO
46  _pwm_channels[i] = new PWM_Sysfs_Bebop(_channel_base+i);

48  _pwm_channels[i] = new PWM_Sysfs(_chip+i, 0);
49 #else
50  _pwm_channels[i] = new PWM_Sysfs(_chip, _channel_base+i);
51 #endif
52  if (!_pwm_channels[i]) {
53  AP_HAL::panic("RCOutput_Sysfs_PWM: Unable to setup PWM pin.");
54  }
55  _pwm_channels[i]->init();

57 
58  /* Set the initial frequency */
