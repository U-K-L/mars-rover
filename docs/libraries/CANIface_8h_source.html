410  template <typename DelayCallable>
411  int init(DelayCallable delay_callable, uavcan::uint32_t& inout_bitrate = BitRateAutoDetect)
412  {
413  if (inout_bitrate > 0) {
414  return driver.init(inout_bitrate, CanIface::NormalMode);
415  }
416  else {
417  static const uavcan::uint32_t StandardBitRates[] = {
418  1000000,
419  500000,
420  250000,
421  125000
422  };
423 
424  for (uavcan::uint8_t br = 0; br < sizeof(StandardBitRates) / sizeof(StandardBitRates[0]); br++) {
425  inout_bitrate = StandardBitRates[br];
426 
427  const int res = driver.init(inout_bitrate, CanIface::SilentMode);
428 
429  delay_callable();
430 
431  if (res >= 0) {
432  for (uavcan::uint8_t iface = 0; iface < driver.getNumIfaces(); iface++) {
433  if (!driver.getIface(iface)->isRxBufferEmpty()) {
434  // Re-initializing in normal mode
435  return driver.init(inout_bitrate, CanIface::NormalMode);
436  }
437  }
438  }
439  }
440 
441  return -ErrBitRateNotDetected;
442  }
443  }
