127 void SoloGimbal_Parameters::update()
128 {
129  uint32_t tnow_ms = AP_HAL::millis();
130 
131  // retry initial param retrieval
132  if(!received_all() && ((tnow_ms - _last_request_ms) > _retry_period) &&
133  (HAVE_PAYLOAD_SPACE(_chan, PARAM_REQUEST_LIST))) {
134  _last_request_ms = tnow_ms;
135  mavlink_msg_param_request_list_send(_chan, 0, MAV_COMP_ID_GIMBAL);
136 
137  for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
138  if (!_params[i].seen) {
139  _params[i].fetch_attempts++;
140  }
141  }
142  }
143 
144  // retry param_set
145  for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
146  if (!HAVE_PAYLOAD_SPACE(_chan, PARAM_SET)) {
147  break;
148  }
149 
150  if ((_params[i].state == GMB_PARAMSTATE_ATTEMPTING_TO_SET) && (tnow_ms - _last_set_ms > _retry_period)) {
151  mavlink_msg_param_set_send(_chan, 0, MAV_COMP_ID_GIMBAL, get_param_name((gmb_param_t)i), _params[i].value, MAV_PARAM_TYPE_REAL32);
152  _last_set_ms = AP_HAL::millis();
153 
154  if (!_params[i].seen) {
155  _params[i].fetch_attempts++;
156  }
157  }
158  }
159 
160  // check for nonexistent parameters
161  for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
162  if (!_params[i].seen && _params[i].fetch_attempts > _max_fetch_attempts) {
163  _params[i].state = GMB_PARAMSTATE_NONEXISTANT;
164  hal.console->printf("Gimbal parameter %s timed out\n", get_param_name((gmb_param_t)i));
165  }
166  }
167 
168  if(_flashing_step == GMB_PARAM_FLASHING_WAITING_FOR_SET) {
169  bool done = true;
170  for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
171  if (_params[i].state == GMB_PARAMSTATE_ATTEMPTING_TO_SET) {
172  done = false;
173  break;
174  }
175  }
176 
177  if (done) {
178  _flashing_step = GMB_PARAM_FLASHING_WAITING_FOR_ACK;
179  set_param(GMB_PARAM_GMB_FLASH,69.0f);
180  }

184 void SoloGimbal_Parameters::handle_param_value(const mavlink_message_t *msg)
185 {
186  mavlink_param_value_t packet;
187  mavlink_msg_param_value_decode(msg, &packet);
188 
189  AP_Logger *logger = AP_Logger::get_singleton();
190  if (logger != nullptr) {
191  logger->Write_Parameter(packet.param_id, packet.param_value);
192  }
193 
194  for(uint8_t i=0; i<MAVLINK_GIMBAL_NUM_TRACKED_PARAMS; i++) {
195  if (!strcmp(packet.param_id, get_param_name((gmb_param_t)i))) {
196  _params[i].seen = true;
197  switch(_params[i].state) {
198  case GMB_PARAMSTATE_NONEXISTANT:
199  case GMB_PARAMSTATE_NOT_YET_READ:
200  case GMB_PARAMSTATE_FETCH_AGAIN:
201  _params[i].value = packet.param_value;
202  _params[i].state = GMB_PARAMSTATE_CONSISTENT;
203  break;
204  case GMB_PARAMSTATE_CONSISTENT:
205  _params[i].value = packet.param_value;
206  break;
207  case GMB_PARAMSTATE_ATTEMPTING_TO_SET:
208  if (i == GMB_PARAM_GMB_FLASH) {
209  if (_flashing_step == GMB_PARAM_FLASHING_WAITING_FOR_ACK && (int)packet.param_value == 1) {
210  _flashing_step = GMB_PARAM_NOT_FLASHING;
211  }
212  _params[i].value = 0;
213  _params[i].state = GMB_PARAMSTATE_CONSISTENT;
214  } else if (is_equal(packet.param_value,_params[i].value)) {
215  _params[i].state = GMB_PARAMSTATE_CONSISTENT;
216  }
217  break;
218  }
219  break;
220  }
221  }
