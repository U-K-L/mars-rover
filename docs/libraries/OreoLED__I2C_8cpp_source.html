71 // UPDATE device according to timed_updated. Called at 50Hz
72 void OreoLED_I2C::update()
73 {
74  if (slow_counter()) {
75  return; // slow rate from 50hz to 10hz
76  }
77 
78  if (mode_firmware_update()) {
79  return; // don't go any further if the Pixhawk is in firmware update
80  }
81 
82  if (mode_init()) {
83  return; // don't go any further if the Pixhawk is initializing
84  }
85 
86  if (mode_failsafe_radio()) {
87  return; // don't go any further if the Pixhawk is is in radio failsafe
88  }
89 
90  set_standard_colors();
91 
92  if (mode_failsafe_batt()) {
93  return; // stop here if the battery is low.
94  }
95 
96  if (_pattern_override) {
97  return; // stop here if in mavlink LED control override.
98  }
99 
100  if (mode_auto_flight()) {
101  return; // stop here if in an autopilot mode.
102  }
103 
104  mode_pilot_flight(); // stop here if in an pilot controlled mode.
105 }

314  if (!(_state_desired[i] == _state_sent[i])) {

318  } else if (instance < OREOLED_NUM_LEDS) {
319  // store desired rgb for one LED
320  _state_desired[instance].set_rgb(pattern, red, green, blue);
321  if (!(_state_desired[instance] == _state_sent[instance])) {
322  _send_required = true;
323  }

389 /*
390  send a command onto the I2C bus
391  */
392 bool OreoLED_I2C::command_send(oreoled_cmd_t &cmd)
393 {
394  //printf("sending %u\n", cmd.num_bytes);
395  _dev->set_address(OREOLED_BASE_I2C_ADDR + cmd.led_num);
396 
397  /* Calculate XOR CRC and append to the i2c write data */
398  uint8_t cmd_xor = OREOLED_BASE_I2C_ADDR + cmd.led_num;
399 
400  for (uint8_t i = 0; i < cmd.num_bytes; i++) {
401  cmd_xor ^= cmd.buff[i];
402  }
403  cmd.buff[cmd.num_bytes++] = cmd_xor;
404 
405  uint8_t reply[3] {};
406  bool ret = _dev->transfer(cmd.buff, cmd.num_bytes, reply, sizeof(reply));
407  //printf("command[%u] %02x %02x %02x %s -> %02x %02x %02x\n", cmd.led_num, ret?"OK":"fail", reply[0], reply[1], reply[2]);
408  return ret;
409 }

411 /*
412  send boot command to all LEDs
413  */
414 void OreoLED_I2C::boot_leds(void)
415 {
416  oreoled_cmd_t cmd;
417  for (uint8_t i=0; i<OREOLED_NUM_LEDS; i++) {
418  cmd.led_num = i;
419  cmd.buff[0] = OREOLED_BOOT_CMD_BOOT_APP;
420  cmd.buff[1] = OREOLED_BOOT_CMD_BOOT_NONCE;
421  cmd.buff[2] = OREOLED_BASE_I2C_ADDR + i;
422  cmd.num_bytes = 3;
423  command_send(cmd);

427 // update_timer - called by scheduler and updates driver with commands
428 void OreoLED_I2C::update_timer(void)
429 {
430  WITH_SEMAPHORE(_sem);
431 
432  uint32_t now = AP_HAL::millis();
433 
434  if (_boot_count < 20 &&
435  now - _last_boot_ms > 100) {
436  // send boot command 20 times
437  _boot_count++;
438  _last_boot_ms = now;
439  boot_leds();
440  }
441 
442  // send a sync every 4.1s. The driver uses 4ms, but using
443  // exactly 4ms does not work. It seems that the oreoled firmware
444  // relies on the inaccuracy of the NuttX scheduling for this to
445  // work, and exactly 4ms from ChibiOS causes syncronisation to
446  // fail
447  if (now - _last_sync_ms > 4100) {
448  _last_sync_ms = now;
449  send_sync();
450  }
451 
452  // exit immediately if send not required, or state is being updated
453  if (!_send_required) {
454  return;
455  }
456 
457  // for each LED
458  for (uint8_t i=0; i<OREOLED_NUM_LEDS; i++) {
459 
460  // check for state change
461  if (true) {
462  switch (_state_desired[i].mode) {
463  case OREOLED_MODE_MACRO: {
464  oreoled_cmd_t cmd {};
465  cmd.led_num = i;
466  cmd.buff[0] = OREOLED_PATTERN_PARAMUPDATE;
467  cmd.buff[1] = OREOLED_PARAM_MACRO;
468  cmd.buff[2] = _state_desired[i].macro;
469  cmd.num_bytes = 3;
470  command_send(cmd);
471  break;
472  }
473 
474  case OREOLED_MODE_RGB: {
475  oreoled_cmd_t cmd {};
476  cmd.led_num = i;
477  cmd.buff[0] = _state_desired[i].pattern;
478  cmd.buff[1] = OREOLED_PARAM_BIAS_RED;
479  cmd.buff[2] = _state_desired[i].red;
480  cmd.buff[3] = OREOLED_PARAM_BIAS_GREEN;
481  cmd.buff[4] = _state_desired[i].green;
482  cmd.buff[5] = OREOLED_PARAM_BIAS_BLUE;
483  cmd.buff[6] = _state_desired[i].blue;
484  cmd.num_bytes = 7;
485  command_send(cmd);
486  break;
487  }
488 
489  case OREOLED_MODE_RGB_EXTENDED: {
490  oreoled_cmd_t cmd {};
491  cmd.led_num = i;
492  cmd.buff[0] = _state_desired[i].pattern;
493  cmd.buff[1] = OREOLED_PARAM_BIAS_RED;
494  cmd.buff[2] = _state_desired[i].red;
495  cmd.buff[3] = OREOLED_PARAM_BIAS_GREEN;
496  cmd.buff[4] = _state_desired[i].green;
497  cmd.buff[5] = OREOLED_PARAM_BIAS_BLUE;
498  cmd.buff[6] = _state_desired[i].blue;
499  cmd.buff[7] = OREOLED_PARAM_AMPLITUDE_RED;
500  cmd.buff[8] = _state_desired[i].amplitude_red;
501  cmd.buff[9] = OREOLED_PARAM_AMPLITUDE_GREEN;
502  cmd.buff[10] = _state_desired[i].amplitude_green;
503  cmd.buff[11] = OREOLED_PARAM_AMPLITUDE_BLUE;
504  cmd.buff[12] = _state_desired[i].amplitude_blue;
505  // Note: The Oreo LED controller expects to receive uint16 values
506  // in little endian order
507  cmd.buff[13] = OREOLED_PARAM_PERIOD;
508  cmd.buff[14] = (_state_desired[i].period & 0xFF00) >> 8;
509  cmd.buff[15] = (_state_desired[i].period & 0x00FF);
510  cmd.buff[16] = OREOLED_PARAM_PHASEOFFSET;
511  cmd.buff[17] = (_state_desired[i].phase_offset & 0xFF00) >> 8;
512  cmd.buff[18] = (_state_desired[i].phase_offset & 0x00FF);
513  cmd.num_bytes = 19;
514  command_send(cmd);
515  break;
516  }
517 
518  default:
519  break;
520  };
521  // save state change
522  _state_sent[i] = _state_desired[i];
523  }
524  }
525 
526  // flag updates sent
527  _send_required = false;
528 }

530 void OreoLED_I2C::send_sync(void)
531 {
532  /* set I2C address to zero */

546 // Handle an LED_CONTROL mavlink message
547 void OreoLED_I2C::handle_led_control(mavlink_message_t *msg)
548 {
549  // decode mavlink message
550  mavlink_led_control_t packet;
551  mavlink_msg_led_control_decode(msg, &packet);
552 
553  // exit immediately if instance is invalid
554  if (packet.instance >= OREOLED_NUM_LEDS && packet.instance != OREOLED_INSTANCE_ALL) {
555  return;
556  }
557 
558  // if pattern is OFF, we clear pattern override so normal lighting should resume
559  if (packet.pattern == LED_CONTROL_PATTERN_OFF) {
560  _pattern_override = 0;
561  clear_state();
562  return;
563  }
564 
565  if (packet.pattern == LED_CONTROL_PATTERN_CUSTOM) {
566  // Here we handle two different "sub commands",
567  // depending on the bytes in the first CUSTOM_HEADER_LENGTH
568  // of the custom pattern byte buffer
569 
570  // Return if we don't have at least CUSTOM_HEADER_LENGTH bytes
571  if (packet.custom_len < CUSTOM_HEADER_LENGTH) {
572  return;
573  }
574 
575  // check for the RGB0 sub-command
576  if (memcmp(packet.custom_bytes, "RGB0", CUSTOM_HEADER_LENGTH) == 0) {
577  // check to make sure the total length matches the length of the RGB0 command + data values
578  if (packet.custom_len != CUSTOM_HEADER_LENGTH + 4) {
579  return;
580  }
581 
582  // check for valid pattern id
583  if (packet.custom_bytes[CUSTOM_HEADER_LENGTH] >= OREOLED_PATTERN_ENUM_COUNT) {
584  return;
585  }
586 
587  // convert the first byte after the command to a oreoled_pattern
588  oreoled_pattern pattern = (oreoled_pattern)packet.custom_bytes[CUSTOM_HEADER_LENGTH];
589 
590  // call the set_rgb function, using the rest of the bytes as the RGB values
591  set_rgb(packet.instance, pattern, packet.custom_bytes[CUSTOM_HEADER_LENGTH + 1], packet.custom_bytes[CUSTOM_HEADER_LENGTH + 2], packet.custom_bytes[CUSTOM_HEADER_LENGTH + 3]);
592 
593  } else if (memcmp(packet.custom_bytes, "RGB1", CUSTOM_HEADER_LENGTH) == 0) { // check for the RGB1 sub-command
594 
595  // check to make sure the total length matches the length of the RGB1 command + data values
596  if (packet.custom_len != CUSTOM_HEADER_LENGTH + 11) {
597  return;
598  }
599 
600  // check for valid pattern id
601  if (packet.custom_bytes[CUSTOM_HEADER_LENGTH] >= OREOLED_PATTERN_ENUM_COUNT) {
602  return;
603  }
604 
605  // convert the first byte after the command to a oreoled_pattern
606  oreoled_pattern pattern = (oreoled_pattern)packet.custom_bytes[CUSTOM_HEADER_LENGTH];
607 
608  // uint16_t values are stored in custom_bytes in little endian order
609  // assume the flight controller is little endian when decoding values
610  uint16_t period =
611  ((0x00FF & (uint16_t)packet.custom_bytes[CUSTOM_HEADER_LENGTH + 7]) << 8) |
612  (0x00FF & (uint16_t)packet.custom_bytes[CUSTOM_HEADER_LENGTH + 8]);
613  uint16_t phase_offset =
614  ((0x00FF & (uint16_t)packet.custom_bytes[CUSTOM_HEADER_LENGTH + 9]) << 8) |
615  (0x00FF & (uint16_t)packet.custom_bytes[CUSTOM_HEADER_LENGTH + 10]);
616 
617  // call the set_rgb function, using the rest of the bytes as the RGB values
618  set_rgb(packet.instance, pattern, packet.custom_bytes[CUSTOM_HEADER_LENGTH + 1], packet.custom_bytes[CUSTOM_HEADER_LENGTH + 2],
619  packet.custom_bytes[CUSTOM_HEADER_LENGTH + 3], packet.custom_bytes[CUSTOM_HEADER_LENGTH + 4], packet.custom_bytes[CUSTOM_HEADER_LENGTH + 5],
620  packet.custom_bytes[CUSTOM_HEADER_LENGTH + 6], period, phase_offset);
621  } else { // unrecognized command
622  return;
623  }
624  } else {
625  // other patterns sent as macro
626  set_macro(packet.instance, (oreoled_macro)packet.pattern);
627  }
628  _pattern_override = packet.pattern;
629 }
