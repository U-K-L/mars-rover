50 // read - return last value measured by sensor
51 bool AP_RangeFinder_LightWareSerial::get_reading(uint16_t &reading_cm)
52 {
53  if (uart == nullptr) {
54  return false;
55  }
56 
57  // read any available lines from the lidar
58  float sum = 0;
59  uint16_t count = 0;
60  int16_t nbytes = uart->available();
61  while (nbytes-- > 0) {
62  char c = uart->read();
63  if (c == '\r') {
64  linebuf[linebuf_len] = 0;
65  sum += (float)atof(linebuf);
66  count++;
67  linebuf_len = 0;
68  } else if (isdigit(c) || c == '.') {
69  linebuf[linebuf_len++] = c;
70  if (linebuf_len == sizeof(linebuf)) {
71  // too long, discard the line
72  linebuf_len = 0;
73  }
74  }
75  }
76 
77  uint32_t now = AP_HAL::millis();
78  if (last_init_ms == 0 ||
79  (now - last_init_ms > 1000 &&
80  now - state.last_reading_ms > 1000)) {
81  // send enough serial transitions to trigger LW20 into serial
82  // mode. It starts in dual I2C/serial mode, and wants to see
83  // enough transitions to switch into serial mode.
84  uart->write("www\r\n");
85  last_init_ms = now;
86  } else {
87  uart->write('d');
88  }
89 
90  if (count == 0) {
91  return false;
92  }
93  reading_cm = 100 * sum / count;
94  return true;
95 }
