31 // format of serial packets received from benewake lidar
32 //
33 // Data Bit Definition Description
34 // ------------------------------------------------
35 // byte 0 Frame header 0x59
36 // byte 1 Frame header 0x59
37 // byte 2 DIST_L Distance (in cm) low 8 bits
38 // byte 3 DIST_H Distance (in cm) high 8 bits
39 // byte 4 STRENGTH_L Strength low 8 bits
40 // byte 5 STRENGTH_H Strength high 8 bits
41 // byte 6 (TF02) SIG Reliability in 8 levels, 7 & 8 means reliable
42 // byte 6 (TFmini) Distance Mode 0x02 for short distance (mm), 0x07 for long distance (cm)
43 // byte 7 (TF02 only) TIME Exposure time in two levels 0x03 and 0x06
44 // byte 8 Checksum Checksum byte, sum of bytes 0 to bytes 7

75 // distance returned in reading_cm, signal_ok is set to true if sensor reports a strong signal
76 bool AP_RangeFinder_Benewake::get_reading(uint16_t &reading_cm)
77 {
78  if (uart == nullptr) {
79  return false;
80  }
81 
82  float sum_cm = 0;
83  uint16_t count = 0;
84  uint16_t count_out_of_range = 0;
85 
86  // read any available lines from the lidar
87  int16_t nbytes = uart->available();
88  while (nbytes-- > 0) {
89  int16_t r = uart->read();
90  if (r < 0) {
91  continue;
92  }
93  uint8_t c = (uint8_t)r;
94  // if buffer is empty and this byte is 0x59, add to buffer
95  if (linebuf_len == 0) {
96  if (c == BENEWAKE_FRAME_HEADER) {
97  linebuf[linebuf_len++] = c;
98  }
99  } else if (linebuf_len == 1) {
100  // if buffer has 1 element and this byte is 0x59, add it to buffer
101  // if not clear the buffer
102  if (c == BENEWAKE_FRAME_HEADER) {
103  linebuf[linebuf_len++] = c;
104  } else {
105  linebuf_len = 0;
106  }
107  } else {
108  // add character to buffer
109  linebuf[linebuf_len++] = c;
110  // if buffer now has 9 items try to decode it
111  if (linebuf_len == BENEWAKE_FRAME_LENGTH) {
112  // calculate checksum
113  uint8_t checksum = 0;
114  for (uint8_t i=0; i<BENEWAKE_FRAME_LENGTH-1; i++) {
115  checksum += linebuf[i];
116  }
117  // if checksum matches extract contents
118  if (checksum == linebuf[BENEWAKE_FRAME_LENGTH-1]) {
119  // calculate distance
120  uint16_t dist = ((uint16_t)linebuf[3] << 8) | linebuf[2];
121  if (dist >= BENEWAKE_DIST_MAX_CM) {
122  // this reading is out of range
123  count_out_of_range++;
124  } else if (model_type == BENEWAKE_TFmini) {
125  // no signal byte from TFmini so add distance to sum
126  sum_cm += dist;
127  count++;
128  } else {
129  // TF02 provides signal reliability (good = 7 or 8)
130  if (linebuf[6] >= 7) {
131  // add distance to sum
132  sum_cm += dist;
133  count++;
134  } else {
135  // this reading is out of range
136  count_out_of_range++;
137  }
138  }
139  }
140  // clear buffer
141  linebuf_len = 0;
142  }
143  }
144  }
145 
146  if (count > 0) {
147  // return average distance of readings
148  reading_cm = sum_cm / count;
149  return true;
150  }
151 
152  if (count_out_of_range > 0) {
153  // if only out of range readings return larger of
154  // driver defined maximum range for the model and user defined max range + 1m
155  float model_dist_max_cm = (model_type == BENEWAKE_TFmini) ? BENEWAKE_TFMINI_OUT_OF_RANGE_CM : BENEWAKE_TF02_OUT_OF_RANGE_CM;
156  reading_cm = MAX(model_dist_max_cm, max_distance_cm() + BENEWAKE_OUT_OF_RANGE_ADD_CM);
157  return true;
158  }
159 
160  // no readings so return false
161  return false;
162 }
