55 // update status based on distance measurement
56 void AP_RangeFinder_Backend::update_status()
57 {
58  // check distance
59  if ((int16_t)state.distance_cm > params.max_distance_cm) {
60  set_status(RangeFinder::RangeFinder_OutOfRangeHigh);
61  } else if ((int16_t)state.distance_cm < params.min_distance_cm) {
62  set_status(RangeFinder::RangeFinder_OutOfRangeLow);
63  } else {
64  set_status(RangeFinder::RangeFinder_Good);
65  }
