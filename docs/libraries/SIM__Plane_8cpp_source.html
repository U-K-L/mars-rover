92 float Plane::liftCoeff(float alpha) const
93 {
94  const float alpha0 = coefficient.alpha_stall;
95  const float M = coefficient.mcoeff;
96  const float c_lift_0 = coefficient.c_lift_0;
97  const float c_lift_a0 = coefficient.c_lift_a;
98 
99  // clamp the value of alpha to avoid exp(90) in calculation of sigmoid
100  const float max_alpha_delta = 0.8f;
101  if (alpha-alpha0 > max_alpha_delta) {
102  alpha = alpha0 + max_alpha_delta;
103  } else if (alpha0-alpha > max_alpha_delta) {
104  alpha = alpha0 - max_alpha_delta;
105  }
106  double sigmoid = ( 1+exp(-M*(alpha-alpha0))+exp(M*(alpha+alpha0)) ) / (1+exp(-M*(alpha-alpha0))) / (1+exp(M*(alpha+alpha0)));
107  double linear = (1.0-sigmoid) * (c_lift_0 + c_lift_a0*alpha); //Lift at small AoA
108  double flatPlate = sigmoid*(2*copysign(1,alpha)*pow(sin(alpha),2)*cos(alpha)); //Lift beyond stall
109 
110  float result = linear+flatPlate;
111  return result;
112 }

114 float Plane::dragCoeff(float alpha) const
115 {
116  const float b = coefficient.b;
117  const float s = coefficient.s;
118  const float c_drag_p = coefficient.c_drag_p;
119  const float c_lift_0 = coefficient.c_lift_0;
120  const float c_lift_a0 = coefficient.c_lift_a;
121  const float oswald = coefficient.oswald;
122 
123  double AR = pow(b,2)/s;
124  double c_drag_a = c_drag_p + pow(c_lift_0+c_lift_a0*alpha,2)/(M_PI*oswald*AR);
125 
126  return c_drag_a;
127 }

129 // Torque calculation function
130 Vector3f Plane::getTorque(float inputAileron, float inputElevator, float inputRudder, float inputThrust, const Vector3f &force) const
131 {
132  float alpha = angle_of_attack;
133 
134  //calculate aerodynamic torque
135  float effective_airspeed = airspeed;
136 
137  if (tailsitter) {
138  /*
139  tailsitters get airspeed from prop-wash
140  */
141  effective_airspeed += inputThrust * 20;
142 
143  // reduce effective angle of attack as thrust increases
144  alpha *= constrain_float(1 - inputThrust, 0, 1);
145  }
146 
147  const float s = coefficient.s;
148  const float c = coefficient.c;
149  const float b = coefficient.b;
150  const float c_l_0 = coefficient.c_l_0;
151  const float c_l_b = coefficient.c_l_b;
152  const float c_l_p = coefficient.c_l_p;
153  const float c_l_r = coefficient.c_l_r;
154  const float c_l_deltaa = coefficient.c_l_deltaa;
155  const float c_l_deltar = coefficient.c_l_deltar;
156  const float c_m_0 = coefficient.c_m_0;
157  const float c_m_a = coefficient.c_m_a;
158  const float c_m_q = coefficient.c_m_q;
159  const float c_m_deltae = coefficient.c_m_deltae;
160  const float c_n_0 = coefficient.c_n_0;
161  const float c_n_b = coefficient.c_n_b;
162  const float c_n_p = coefficient.c_n_p;
163  const float c_n_r = coefficient.c_n_r;
164  const float c_n_deltaa = coefficient.c_n_deltaa;
165  const float c_n_deltar = coefficient.c_n_deltar;
166  const Vector3f &CGOffset = coefficient.CGOffset;
167 
168  float rho = air_density;
169 
170  //read angular rates
171  double p = gyro.x;
172  double q = gyro.y;
173  double r = gyro.z;
174 
175  double qbar = 1.0/2.0*rho*pow(effective_airspeed,2)*s; //Calculate dynamic pressure
176  double la, na, ma;
177  if (is_zero(effective_airspeed))
178  {
179  la = 0;
180  ma = 0;
181  na = 0;
182  }
183  else
184  {
185  la = qbar*b*(c_l_0 + c_l_b*beta + c_l_p*b*p/(2*effective_airspeed) + c_l_r*b*r/(2*effective_airspeed) + c_l_deltaa*inputAileron + c_l_deltar*inputRudder);
186  ma = qbar*c*(c_m_0 + c_m_a*alpha + c_m_q*c*q/(2*effective_airspeed) + c_m_deltae*inputElevator);
187  na = qbar*b*(c_n_0 + c_n_b*beta + c_n_p*b*p/(2*effective_airspeed) + c_n_r*b*r/(2*effective_airspeed) + c_n_deltaa*inputAileron + c_n_deltar*inputRudder);
188  }
189 
190 
191  // Add torque to to force misalignment with CG
192  // r x F, where r is the distance from CoG to CoL
193  la += CGOffset.y * force.z - CGOffset.z * force.y;
194  ma += -CGOffset.x * force.z + CGOffset.z * force.x;
195  na += -CGOffset.y * force.x + CGOffset.x * force.y;
196 
197  return Vector3f(la, ma, na);
198 }

200 // Force calculation function from last_letter
201 Vector3f Plane::getForce(float inputAileron, float inputElevator, float inputRudder) const
202 {
203  const float alpha = angle_of_attack;
204  const float c_drag_q = coefficient.c_drag_q;
205  const float c_lift_q = coefficient.c_lift_q;
206  const float s = coefficient.s;
207  const float c = coefficient.c;
208  const float b = coefficient.b;
209  const float c_drag_deltae = coefficient.c_drag_deltae;
210  const float c_lift_deltae = coefficient.c_lift_deltae;
211  const float c_y_0 = coefficient.c_y_0;
212  const float c_y_b = coefficient.c_y_b;
213  const float c_y_p = coefficient.c_y_p;
214  const float c_y_r = coefficient.c_y_r;
215  const float c_y_deltaa = coefficient.c_y_deltaa;
216  const float c_y_deltar = coefficient.c_y_deltar;
217 
218  float rho = air_density;
219 
220  //request lift and drag alpha-coefficients from the corresponding functions
221  double c_lift_a = liftCoeff(alpha);
222  double c_drag_a = dragCoeff(alpha);
223 
224  //convert coefficients to the body frame
225  double c_x_a = -c_drag_a*cos(alpha)+c_lift_a*sin(alpha);
226  double c_x_q = -c_drag_q*cos(alpha)+c_lift_q*sin(alpha);
227  double c_z_a = -c_drag_a*sin(alpha)-c_lift_a*cos(alpha);
228  double c_z_q = -c_drag_q*sin(alpha)-c_lift_q*cos(alpha);
229 
230  //read angular rates
231  double p = gyro.x;
232  double q = gyro.y;
233  double r = gyro.z;
234 
235  //calculate aerodynamic force
236  double qbar = 1.0/2.0*rho*pow(airspeed,2)*s; //Calculate dynamic pressure
237  double ax, ay, az;
238  if (is_zero(airspeed))
239  {
240  ax = 0;
241  ay = 0;
242  az = 0;
243  }
244  else
245  {
246  ax = qbar*(c_x_a + c_x_q*c*q/(2*airspeed) - c_drag_deltae*cos(alpha)*fabs(inputElevator) + c_lift_deltae*sin(alpha)*inputElevator);
247  // split c_x_deltae to include "abs" term
248  ay = qbar*(c_y_0 + c_y_b*beta + c_y_p*b*p/(2*airspeed) + c_y_r*b*r/(2*airspeed) + c_y_deltaa*inputAileron + c_y_deltar*inputRudder);
249  az = qbar*(c_z_a + c_z_q*c*q/(2*airspeed) - c_drag_deltae*sin(alpha)*fabs(inputElevator) - c_lift_deltae*cos(alpha)*inputElevator);
250  // split c_z_deltae to include "abs" term
251  }
252  return Vector3f(ax, ay, az);
253 }

255 void Plane::calculate_forces(const struct sitl_input &input, Vector3f &rot_accel, Vector3f &body_accel)
256 {
257  float aileron = filtered_servo_angle(input, 0);
258  float elevator = filtered_servo_angle(input, 1);
259  float rudder = filtered_servo_angle(input, 3);
260  bool launch_triggered = input.servos[6] > 1700;
261  float throttle;
262  if (reverse_elevator_rudder) {
263  elevator = -elevator;
264  rudder = -rudder;
265  }
266  if (elevons) {
267  // fake an elevon plane
268  float ch1 = aileron;
269  float ch2 = elevator;
270  aileron = (ch2-ch1)/2.0f;
271  // the minus does away with the need for RC2_REV=-1
272  elevator = -(ch2+ch1)/2.0f;
273 
274  // assume no rudder
275  rudder = 0;
276  } else if (vtail) {
277  // fake a vtail plane
278  float ch1 = elevator;
279  float ch2 = rudder;
280  // this matches VTAIL_OUTPUT==2
281  elevator = (ch2-ch1)/2.0f;
282  rudder = (ch2+ch1)/2.0f;
283  } else if (dspoilers) {
284  // fake a differential spoiler plane. Use outputs 1, 2, 4 and 5
285  float dspoiler1_left = filtered_servo_angle(input, 0);
286  float dspoiler1_right = filtered_servo_angle(input, 1);
287  float dspoiler2_left = filtered_servo_angle(input, 3);
288  float dspoiler2_right = filtered_servo_angle(input, 4);
289  float elevon_left = (dspoiler1_left + dspoiler2_left)/2;
290  float elevon_right = (dspoiler1_right + dspoiler2_right)/2;
291  aileron = (elevon_right-elevon_left)/2;
292  elevator = (elevon_left+elevon_right)/2;
293  rudder = fabsf(dspoiler1_right - dspoiler2_right)/2 - fabsf(dspoiler1_left - dspoiler2_left)/2;
294  }
295  //printf("Aileron: %.1f elevator: %.1f rudder: %.1f\n", aileron, elevator, rudder);
296 
297  if (reverse_thrust) {
298  throttle = filtered_servo_angle(input, 2);
299  } else {
300  throttle = filtered_servo_range(input, 2);
301  }
302 
303  float thrust = throttle;
304 
305  if (ice_engine) {
306  thrust = icengine.update(input);
307  }
308 
309  // calculate angle of attack
310  angle_of_attack = atan2f(velocity_air_bf.z, velocity_air_bf.x);
311  beta = atan2f(velocity_air_bf.y,velocity_air_bf.x);
312 
313  if (tailsitter) {
314  /*
315  tailsitters get 4x the control surfaces
316  */
317  aileron *= 4;
318  elevator *= 4;
319  rudder *= 4;
320  }
321 
322  Vector3f force = getForce(aileron, elevator, rudder);
323  rot_accel = getTorque(aileron, elevator, rudder, thrust, force);
324 
325  if (have_launcher) {
326  /*
327  simple simulation of a launcher
328  */
329  if (launch_triggered) {
330  uint64_t now = AP_HAL::millis64();
331  if (launch_start_ms == 0) {
332  launch_start_ms = now;
333  }
334  if (now - launch_start_ms < launch_time*1000) {
335  force.x += launch_accel;
336  force.z += launch_accel/3;
337  }
338  } else {
339  // allow reset of catapult
340  launch_start_ms = 0;
341  }
342  }
343 
344  // simulate engine RPM
345  rpm1 = thrust * 7000;
346 
347  // scale thrust to newtons
348  thrust *= thrust_scale;
349 
350  accel_body = Vector3f(thrust, 0, 0) + force;
351  accel_body /= mass;
352 
353  // add some noise
354  if (thrust_scale > 0) {
355  add_noise(fabsf(thrust) / thrust_scale);
356  }
357 
358  if (on_ground() && !tailsitter) {
359  // add some ground friction
360  Vector3f vel_body = dcm.transposed() * velocity_ef;
361  accel_body.x -= vel_body.x * 0.3f;
362  }
363 }
