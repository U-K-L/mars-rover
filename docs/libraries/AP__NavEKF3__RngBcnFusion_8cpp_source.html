494  receiverPosCov[j][i] = temp;

498  }

514 /*
515 Run a single state Kalman filter to estimate the vertical position offset of the range beacon constellation
516 Calculate using a high and low hypothesis and select the hypothesis with the lowest innovation sequence
517 */
518 void NavEKF3_core::CalcRangeBeaconPosDownOffset(float obsVar, Vector3f &vehiclePosNED, bool aligning)
519 {
520  // Handle height offsets between the primary height source and the range beacons by estimating
521  // the beacon systems global vertical position offset using a single state Kalman filter
522  // The estimated offset is used to correct the beacon height when calculating innovations
523  // A high and low estimate is calculated to handle the ambiguity in height associated with beacon positions that are co-planar
524  // The main filter then uses the offset with the smaller innovations
525 
526  float innov; // range measurement innovation (m)
527  float innovVar; // range measurement innovation variance (m^2)
528  float gain; // Kalman gain
529  float obsDeriv; // derivative of observation relative to state
530 
531  const float stateNoiseVar = 0.1f; // State process noise variance
532  const float filtAlpha = 0.1f; // LPF constant
533  const float innovGateWidth = 5.0f; // width of innovation consistency check gate in std
534 
535  // estimate upper value for offset
536 
537  // calculate observation derivative
538  float t2 = rngBcnDataDelayed.beacon_posNED.z - vehiclePosNED.z + bcnPosDownOffsetMax;
539  float t3 = rngBcnDataDelayed.beacon_posNED.y - vehiclePosNED.y;
540  float t4 = rngBcnDataDelayed.beacon_posNED.x - vehiclePosNED.x;
541  float t5 = t2*t2;
542  float t6 = t3*t3;
543  float t7 = t4*t4;
544  float t8 = t5+t6+t7;
545  float t9;
546  if (t8 > 0.1f) {
547  t9 = 1.0f/sqrtf(t8);
548  obsDeriv = t2*t9;
549 
550  // Calculate innovation
551  innov = sqrtf(t8) - rngBcnDataDelayed.rng;
552 
553  // covariance prediction
554  bcnPosOffsetMaxVar += stateNoiseVar;
555 
556  // calculate the innovation variance
557  innovVar = obsDeriv * bcnPosOffsetMaxVar * obsDeriv + obsVar;
558  innovVar = MAX(innovVar, obsVar);
559 
560  // calculate the Kalman gain
561  gain = (bcnPosOffsetMaxVar * obsDeriv) / innovVar;
562 
563  // calculate a filtered state change magnitude to be used to select between the high or low offset
564  float stateChange = innov * gain;
565  maxOffsetStateChangeFilt = (1.0f - filtAlpha) * maxOffsetStateChangeFilt + fminf(fabsf(filtAlpha * stateChange) , 1.0f);
566 
567  // Reject range innovation spikes using a 5-sigma threshold unless aligning
568  if ((sq(innov) < sq(innovGateWidth) * innovVar) || aligning) {
569 
570  // state update
571  bcnPosDownOffsetMax -= stateChange;
572 
573  // covariance update
574  bcnPosOffsetMaxVar -= gain * obsDeriv * bcnPosOffsetMaxVar;
575  bcnPosOffsetMaxVar = MAX(bcnPosOffsetMaxVar, 0.0f);
576  }
577  }
578 
579  // estimate lower value for offset
580 
581  // calculate observation derivative
582  t2 = rngBcnDataDelayed.beacon_posNED.z - vehiclePosNED.z + bcnPosDownOffsetMin;
583  t5 = t2*t2;
584  t8 = t5+t6+t7;
585  if (t8 > 0.1f) {
586  t9 = 1.0f/sqrtf(t8);
587  obsDeriv = t2*t9;
588 
589  // Calculate innovation
590  innov = sqrtf(t8) - rngBcnDataDelayed.rng;
591 
592  // covariance prediction
593  bcnPosOffsetMinVar += stateNoiseVar;
594 
595  // calculate the innovation variance
596  innovVar = obsDeriv * bcnPosOffsetMinVar * obsDeriv + obsVar;
597  innovVar = MAX(innovVar, obsVar);
598 
599  // calculate the Kalman gain
600  gain = (bcnPosOffsetMinVar * obsDeriv) / innovVar;
601 
602  // calculate a filtered state change magnitude to be used to select between the high or low offset
603  float stateChange = innov * gain;
604  minOffsetStateChangeFilt = (1.0f - filtAlpha) * minOffsetStateChangeFilt + fminf(fabsf(filtAlpha * stateChange) , 1.0f);
605 
606  // Reject range innovation spikes using a 5-sigma threshold unless aligning
607  if ((sq(innov) < sq(innovGateWidth) * innovVar) || aligning) {
608 
609  // state update
610  bcnPosDownOffsetMin -= stateChange;
611 
612  // covariance update
613  bcnPosOffsetMinVar -= gain * obsDeriv * bcnPosOffsetMinVar;
614  bcnPosOffsetMinVar = MAX(bcnPosOffsetMinVar, 0.0f);
615  }
616  }
617 
618  // calculate the mid vertical position of all beacons
619  float bcnMidPosD = 0.5f * (minBcnPosD + maxBcnPosD);
620 
621  // ensure the two beacon vertical offset hypothesis place the mid point of the beacons below and above the flight vehicle
622  bcnPosDownOffsetMax = MAX(bcnPosDownOffsetMax, vehiclePosNED.z - bcnMidPosD + 0.5f);
623  bcnPosDownOffsetMin = MIN(bcnPosDownOffsetMin, vehiclePosNED.z - bcnMidPosD - 0.5f);
624 
625  // calculate the innovation for the main filter using the offset that is most stable
626  // apply hysteresis to prevent rapid switching
627  if (!usingMinHypothesis && (minOffsetStateChangeFilt < (0.8f * maxOffsetStateChangeFilt))) {
628  usingMinHypothesis = true;
629  } else if (usingMinHypothesis && (maxOffsetStateChangeFilt < (0.8f * minOffsetStateChangeFilt))) {
630  usingMinHypothesis = false;
631  }
632  if (usingMinHypothesis) {
633  bcnPosOffsetNED.z = bcnPosDownOffsetMin;
634  } else {
635  bcnPosOffsetNED.z = bcnPosDownOffsetMax;
636  }
637 
638  // apply the vertical offset to the beacon positions
639  rngBcnDataDelayed.beacon_posNED.z += bcnPosOffsetNED.z;
640 }
