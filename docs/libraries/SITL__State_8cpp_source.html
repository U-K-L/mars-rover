382 void SITL_State::_simulator_servos(struct sitl_input &input)
383 {
384  static uint32_t last_update_usec;
385 
386  /* this maps the registers used for PWM outputs. The RC
387  * driver updates these whenever it wants the channel output
388  * to change */
389  uint8_t i;
390 
391  if (last_update_usec == 0 || !output_ready) {
392  for (i=0; i<SITL_NUM_CHANNELS; i++) {
393  pwm_output[i] = 1000;
394  }
395  if (_vehicle == ArduPlane) {
396  pwm_output[0] = pwm_output[1] = pwm_output[3] = 1500;
397  }
398  if (_vehicle == APMrover2) {
399  pwm_output[0] = pwm_output[1] = pwm_output[2] = pwm_output[3] = 1500;
400  }
401  }
402 
403  // output at chosen framerate
404  uint32_t now = AP_HAL::micros();
405  last_update_usec = now;
406 
407  float altitude = _barometer?_barometer->get_altitude():0;
408  float wind_speed = 0;
409  float wind_direction = 0;
410  float wind_dir_z = 0;
411 
412  // give 5 seconds to calibrate airspeed sensor at 0 wind speed
413  if (wind_start_delay_micros == 0) {
414  wind_start_delay_micros = now;
415  } else if (_sitl && (now - wind_start_delay_micros) > 5000000 ) {
416  // The EKF does not like step inputs so this LPF keeps it happy.
417  wind_speed = _sitl->wind_speed_active = (0.95f*_sitl->wind_speed_active) + (0.05f*_sitl->wind_speed);
418  wind_direction = _sitl->wind_direction_active = (0.95f*_sitl->wind_direction_active) + (0.05f*_sitl->wind_direction);
419  wind_dir_z = _sitl->wind_dir_z_active = (0.95f*_sitl->wind_dir_z_active) + (0.05f*_sitl->wind_dir_z);
420 
421  // pass wind into simulators using different wind types via param SIM_WIND_T*.
422  switch (_sitl->wind_type) {
423  case SITL::SITL::WIND_TYPE_SQRT:
424  if (altitude < _sitl->wind_type_alt) {
425  wind_speed *= sqrtf(MAX(altitude / _sitl->wind_type_alt, 0));
426  }
427  break;
428 
429  case SITL::SITL::WIND_TYPE_COEF:
430  wind_speed += (altitude - _sitl->wind_type_alt) * _sitl->wind_type_coef;
431  break;
432 
433  case SITL::SITL::WIND_TYPE_NO_LIMIT:
434  default:
435  break;
436  }
437 
438  // never allow negative wind velocity
439  wind_speed = MAX(wind_speed, 0);
440  }
441 
442  input.wind.speed = wind_speed;
443  input.wind.direction = wind_direction;
444  input.wind.turbulence = _sitl?_sitl->wind_turbulance:0;
445  input.wind.dir_z = wind_dir_z;
446 
447  for (i=0; i<SITL_NUM_CHANNELS; i++) {
448  if (pwm_output[i] == 0xFFFF) {
449  input.servos[i] = 0;
450  } else {
451  input.servos[i] = pwm_output[i];
452  }
453  }
454 
455  float engine_mul = _sitl?_sitl->engine_mul.get():1;
456  uint8_t engine_fail = _sitl?_sitl->engine_fail.get():0;
457  bool motors_on = false;
458 
459  if (engine_fail >= ARRAY_SIZE(input.servos)) {
460  engine_fail = 0;
461  }
462  // apply engine multiplier to motor defined by the SIM_ENGINE_FAIL parameter
463  if (_vehicle != APMrover2) {
464  input.servos[engine_fail] = ((input.servos[engine_fail]-1000) * engine_mul) + 1000;
465  } else {
466  input.servos[engine_fail] = static_cast<uint16_t>(((input.servos[engine_fail] - 1500) * engine_mul) + 1500);
467  }
468 
469  if (_vehicle == ArduPlane) {
470  motors_on = ((input.servos[2] - 1000) / 1000.0f) > 0;
471  } else if (_vehicle == APMrover2) {
472  input.servos[2] = static_cast<uint16_t>(constrain_int16(input.servos[2], 1000, 2000));
473  input.servos[0] = static_cast<uint16_t>(constrain_int16(input.servos[0], 1000, 2000));
474  motors_on = !is_zero(((input.servos[2] - 1500) / 500.0f));
475  } else {
476  motors_on = false;
477  // run checks on each motor
478  for (i=0; i<4; i++) {
479  // check motors do not exceed their limits
480  if (input.servos[i] > 2000) input.servos[i] = 2000;
481  if (input.servos[i] < 1000) input.servos[i] = 1000;
482  // update motor_on flag
483  if ((input.servos[i]-1000)/1000.0f > 0) {
484  motors_on = true;
485  }
486  }
487  }
488  if (_sitl) {
489  _sitl->motors_on = motors_on;
490  }
491 
492  float voltage = 0;
493  _current = 0;
494 
495  if (_sitl != nullptr) {
496  if (_sitl->state.battery_voltage <= 0) {
497  if (_vehicle == ArduSub) {
498  voltage = _sitl->batt_voltage;
499  for (i = 0; i < 6; i++) {
500  float pwm = input.servos[i];
501  //printf("i: %d, pwm: %.2f\n", i, pwm);
502  float fraction = fabsf((pwm - 1500) / 500.0f);
503 
504  voltage -= fraction * 0.5f;
505 
506  float draw = fraction * 15;
507  _current += draw;
508  }
509  } else {
510  // simulate simple battery setup
511  float throttle;
512  if (_vehicle == APMrover2) {
513  throttle = motors_on ? (input.servos[2] - 1500) / 500.0f : 0;
514  } else {
515  throttle = motors_on ? (input.servos[2] - 1000) / 1000.0f : 0;
516  }
517  // lose 0.7V at full throttle
518  voltage = _sitl->batt_voltage - 0.7f*fabsf(throttle);
519 
520  // assume 50A at full throttle
521  _current = 50.0f * fabsf(throttle);
522  }
523  } else {
524  // FDM provides voltage and current
525  voltage = _sitl->state.battery_voltage;
526  _current = _sitl->state.battery_current;
527  }
528  }
529 
530  // assume 3DR power brick
531  voltage_pin_value = ((voltage / 10.1f) / 5.0f) * 1024;
532  current_pin_value = ((_current / 17.0f) / 5.0f) * 1024;
533  // fake battery2 as just a 25% gain on the first one
534  voltage2_pin_value = ((voltage * 0.25f / 10.1f) / 5.0f) * 1024;
535  current2_pin_value = ((_current * 0.25f / 17.0f) / 5.0f) * 1024;
