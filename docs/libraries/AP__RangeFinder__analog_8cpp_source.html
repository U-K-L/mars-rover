83 /*
84  update distance_cm
85  */
86 void AP_RangeFinder_analog::update(void)
87 {
88  update_voltage();
89  float v = state.voltage_mv * 0.001f;
90  float dist_m = 0;
91  float scaling = params.scaling;
92  float offset = params.offset;
93  RangeFinder::RangeFinder_Function function = (RangeFinder::RangeFinder_Function)params.function.get();
94  int16_t _max_distance_cm = params.max_distance_cm;
95 
96  switch (function) {
97  case RangeFinder::FUNCTION_LINEAR:
98  dist_m = (v - offset) * scaling;
99  break;
100 
101  case RangeFinder::FUNCTION_INVERTED:
102  dist_m = (offset - v) * scaling;
103  break;
104 
105  case RangeFinder::FUNCTION_HYPERBOLA:
106  if (v <= offset) {
107  dist_m = 0;
108  } else {
109  dist_m = scaling / (v - offset);
110  }
111  if (dist_m > _max_distance_cm * 0.01f) {
112  dist_m = _max_distance_cm * 0.01f;
113  }
114  break;
115  }
116  if (dist_m < 0) {
117  dist_m = 0;
118  }
119  state.distance_cm = dist_m * 100.0f;
120  state.last_reading_ms = AP_HAL::millis();
121 
122  // update range_valid state based on distance measured
123  update_status();
124 }
