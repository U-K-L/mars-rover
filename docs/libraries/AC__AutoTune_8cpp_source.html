5 /*
6  * autotune support for multicopters
7  *
8  * Instructions:
9  * 1) Set up one flight mode switch position to be AltHold.
10  * 2) Set the Ch7 Opt or Ch8 Opt to AutoTune to allow you to turn the auto tuning on/off with the ch7 or ch8 switch.
11  * 3) Ensure the ch7 or ch8 switch is in the LOW position.
12  * 4) Wait for a calm day and go to a large open area.
13  * 5) Take off and put the vehicle into AltHold mode at a comfortable altitude.
14  * 6) Set the ch7/ch8 switch to the HIGH position to engage auto tuning:
15  * a) You will see it twitch about 20 degrees left and right for a few minutes, then it will repeat forward and back.
16  * b) Use the roll and pitch stick at any time to reposition the copter if it drifts away (it will use the original PID gains during repositioning and between tests).
17  * When you release the sticks it will continue auto tuning where it left off.
18  * c) Move the ch7/ch8 switch into the LOW position at any time to abandon the autotuning and return to the origin PIDs.
19  * d) Make sure that you do not have any trim set on your transmitter or the autotune may not get the signal that the sticks are centered.
20  * 7) When the tune completes the vehicle will change back to the original PID gains.
21  * 8) Put the ch7/ch8 switch into the LOW position then back to the HIGH position to test the tuned PID gains.
22  * 9) Put the ch7/ch8 switch into the LOW position to fly using the original PID gains.
23  * 10) If you are happy with the autotuned PID gains, leave the ch7/ch8 switch in the HIGH position, land and disarm to save the PIDs permanently.
24  * If you DO NOT like the new PIDS, switch ch7/ch8 LOW to return to the original PIDs. The gains will not be saved when you disarm
25  *
26  * What it's doing during each "twitch":
27  * a) invokes 90 deg/sec rate request
28  * b) records maximum "forward" roll rate and bounce back rate
29  * c) when copter reaches 20 degrees or 1 second has passed, it commands level
30  * d) tries to keep max rotation rate between 80% ~ 100% of requested rate (90deg/sec) by adjusting rate P
31  * e) increases rate D until the bounce back becomes greater than 10% of requested rate (90deg/sec)
32  * f) decreases rate D until the bounce back becomes less than 10% of requested rate (90deg/sec)
33  * g) increases rate P until the max rotate rate becomes greater than the request rate (90deg/sec)
34  * h) invokes a 20deg angle request on roll or pitch
35  * i) increases stab P until the maximum angle becomes greater than 110% of the requested angle (20deg)
36  * j) decreases stab P by 25%
37  *
38  */

526  target_max_rate = MAX(AUTOTUNE_TARGET_MIN_RATE_RLLPIT_CDS, step_scaler*AUTOTUNE_TARGET_RATE_RLLPIT_CDS);
527  target_rate = constrain_float(ToDeg(attitude_control->max_rate_step_bf_roll())*100.0f, AUTOTUNE_TARGET_MIN_RATE_RLLPIT_CDS, target_max_rate);
528  target_angle = constrain_float(ToDeg(attitude_control->max_angle_step_bf_roll())*100.0f, AUTOTUNE_TARGET_MIN_ANGLE_RLLPIT_CD, AUTOTUNE_TARGET_ANGLE_RLLPIT_CD);
529  abort_angle = AUTOTUNE_TARGET_ANGLE_RLLPIT_CD;
530  start_rate = ToDeg(ahrs_view->get_gyro().x) * 100.0f;
531  start_angle = ahrs_view->roll_sensor;
532  rotation_rate_filt.set_cutoff_frequency(attitude_control->get_rate_roll_pid().filt_hz()*2.0f);
533  break;
534  case PITCH:
535  target_max_rate = MAX(AUTOTUNE_TARGET_MIN_RATE_RLLPIT_CDS, step_scaler*AUTOTUNE_TARGET_RATE_RLLPIT_CDS);
536  target_rate = constrain_float(ToDeg(attitude_control->max_rate_step_bf_pitch())*100.0f, AUTOTUNE_TARGET_MIN_RATE_RLLPIT_CDS, target_max_rate);
537  target_angle = constrain_float(ToDeg(attitude_control->max_angle_step_bf_pitch())*100.0f, AUTOTUNE_TARGET_MIN_ANGLE_RLLPIT_CD, AUTOTUNE_TARGET_ANGLE_RLLPIT_CD);
538  abort_angle = AUTOTUNE_TARGET_ANGLE_RLLPIT_CD;
539  start_rate = ToDeg(ahrs_view->get_gyro().y) * 100.0f;
540  start_angle = ahrs_view->pitch_sensor;
541  rotation_rate_filt.set_cutoff_frequency(attitude_control->get_rate_pitch_pid().filt_hz()*2.0f);
542  break;
543  case YAW:
544  target_max_rate = MAX(AUTOTUNE_TARGET_MIN_RATE_RLLPIT_CDS, step_scaler*AUTOTUNE_TARGET_RATE_YAW_CDS);
545  target_rate = constrain_float(ToDeg(attitude_control->max_rate_step_bf_yaw()*0.75f)*100.0f, AUTOTUNE_TARGET_MIN_RATE_YAW_CDS, target_max_rate);
546  target_angle = constrain_float(ToDeg(attitude_control->max_angle_step_bf_yaw()*0.75f)*100.0f, AUTOTUNE_TARGET_MIN_ANGLE_YAW_CD, AUTOTUNE_TARGET_ANGLE_YAW_CD);
547  abort_angle = AUTOTUNE_TARGET_ANGLE_YAW_CD;
548  start_rate = ToDeg(ahrs_view->get_gyro().z) * 100.0f;
549  start_angle = ahrs_view->yaw_sensor;
550  rotation_rate_filt.set_cutoff_frequency(AUTOTUNE_Y_FILT_FREQ);
551  break;
552  }
553  if ((tune_type == SP_DOWN) || (tune_type == SP_UP)) {
554  rotation_rate_filt.reset(start_rate);
555  } else {
556  rotation_rate_filt.reset(0);
557  }
558  break;
559  }
560 
561  case TWITCHING: {
562  // Run the twitching step
563  load_gains(GAIN_TWITCH);
564 
565  // disable rate limits
566  attitude_control->use_sqrt_controller(false);
567  // hold current attitude
568  attitude_control->input_rate_bf_roll_pitch_yaw(0.0f, 0.0f, 0.0f);
569 
570  if ((tune_type == SP_DOWN) || (tune_type == SP_UP)) {
571  // step angle targets on first iteration
572  if (twitch_first_iter) {
573  twitch_first_iter = false;
574  // Testing increasing stabilize P gain so will set lean angle target
575  switch (axis) {
576  case ROLL:
577  // request roll to 20deg
578  attitude_control->input_angle_step_bf_roll_pitch_yaw(direction_sign * target_angle, 0.0f, 0.0f);
579  break;
580  case PITCH:
581  // request pitch to 20deg
582  attitude_control->input_angle_step_bf_roll_pitch_yaw(0.0f, direction_sign * target_angle, 0.0f);
583  break;
584  case YAW:
585  // request pitch to 20deg
586  attitude_control->input_angle_step_bf_roll_pitch_yaw(0.0f, 0.0f, direction_sign * target_angle);
587  break;
588  }
589  }
590  } else {
591  // Testing rate P and D gains so will set body-frame rate targets.
592  // Rate controller will use existing body-frame rates and convert to motor outputs
593  // for all axes except the one we override here.
594  switch (axis) {
595  case ROLL:
596  // override body-frame roll rate
597  attitude_control->rate_bf_roll_target(direction_sign * target_rate + start_rate);
598  break;
599  case PITCH:
600  // override body-frame pitch rate
601  attitude_control->rate_bf_pitch_target(direction_sign * target_rate + start_rate);
602  break;
603  case YAW:
604  // override body-frame yaw rate
605  attitude_control->rate_bf_yaw_target(direction_sign * target_rate + start_rate);
606  break;
607  }
608  }
609 
610  // capture this iterations rotation rate and lean angle
611  float gyro_reading = 0;
612  switch (axis) {
613  case ROLL:
614  gyro_reading = ahrs_view->get_gyro().x;
615  lean_angle = direction_sign * (ahrs_view->roll_sensor - (int32_t)start_angle);
616  break;
617  case PITCH:
618  gyro_reading = ahrs_view->get_gyro().y;
619  lean_angle = direction_sign * (ahrs_view->pitch_sensor - (int32_t)start_angle);
620  break;
621  case YAW:
622  gyro_reading = ahrs_view->get_gyro().z;
623  lean_angle = direction_sign * wrap_180_cd(ahrs_view->yaw_sensor-(int32_t)start_angle);
624  break;
625  }
626 
627  // Add filter to measurements
628  float filter_value;
629  switch (tune_type) {
630  case SP_DOWN:
631  case SP_UP:
632  filter_value = direction_sign * (ToDeg(gyro_reading) * 100.0f);
633  break;
634  default:
635  filter_value = direction_sign * (ToDeg(gyro_reading) * 100.0f - start_rate);
636  break;
637  }
638  rotation_rate = rotation_rate_filt.apply(filter_value,
639  AP::scheduler().get_loop_period_s());
640 
641  switch (tune_type) {
642  case RD_UP:
643  case RD_DOWN:
644  twitching_test_rate(rotation_rate, target_rate, test_rate_min, test_rate_max);
645  twitching_measure_acceleration(test_accel_max, rotation_rate, rate_max);
646  twitching_abort_rate(lean_angle, rotation_rate, abort_angle, test_rate_min);
647  if (lean_angle >= target_angle) {
648  step = UPDATE_GAINS;
649  }
650  break;
651  case RP_UP:
652  twitching_test_rate(rotation_rate, target_rate*(1+0.5f*aggressiveness), test_rate_min, test_rate_max);
653  twitching_measure_acceleration(test_accel_max, rotation_rate, rate_max);
654  twitching_abort_rate(lean_angle, rotation_rate, abort_angle, test_rate_min);
655  break;
656  case SP_DOWN:
657  case SP_UP:
658  twitching_test_angle(lean_angle, rotation_rate, target_angle*(1+0.5f*aggressiveness), test_angle_min, test_angle_max, test_rate_min, test_rate_max);
659  twitching_measure_acceleration(test_accel_max, rotation_rate - direction_sign * start_rate, rate_max);
660  break;
661  }
662 
663  // log this iterations lean angle and rotation rate
664  Log_Write_AutoTuneDetails(lean_angle, rotation_rate);
665  AP::logger().Write_Rate(ahrs_view, *motors, *attitude_control, *pos_control);
666  log_pids();
667  break;
668  }
669 
670  case UPDATE_GAINS:
671 
672  // re-enable rate limits
673  attitude_control->use_sqrt_controller(true);
674 
675  // log the latest gains
676  if ((tune_type == SP_DOWN) || (tune_type == SP_UP)) {
677  switch (axis) {
678  case ROLL:
679  Log_Write_AutoTune(axis, tune_type, target_angle, test_angle_min, test_angle_max, tune_roll_rp, tune_roll_rd, tune_roll_sp, test_accel_max);
680  break;
681  case PITCH:
682  Log_Write_AutoTune(axis, tune_type, target_angle, test_angle_min, test_angle_max, tune_pitch_rp, tune_pitch_rd, tune_pitch_sp, test_accel_max);
683  break;
684  case YAW:
685  Log_Write_AutoTune(axis, tune_type, target_angle, test_angle_min, test_angle_max, tune_yaw_rp, tune_yaw_rLPF, tune_yaw_sp, test_accel_max);
686  break;
687  }
688  } else {
689  switch (axis) {
690  case ROLL:
691  Log_Write_AutoTune(axis, tune_type, target_rate, test_rate_min, test_rate_max, tune_roll_rp, tune_roll_rd, tune_roll_sp, test_accel_max);
692  break;
693  case PITCH:
694  Log_Write_AutoTune(axis, tune_type, target_rate, test_rate_min, test_rate_max, tune_pitch_rp, tune_pitch_rd, tune_pitch_sp, test_accel_max);
695  break;
696  case YAW:
697  Log_Write_AutoTune(axis, tune_type, target_rate, test_rate_min, test_rate_max, tune_yaw_rp, tune_yaw_rLPF, tune_yaw_sp, test_accel_max);
698  break;
699  }
700  }
701 
702  // Check results after mini-step to increase rate D gain
703  switch (tune_type) {
704  case RD_UP:
705  switch (axis) {
706  case ROLL:
707  updating_rate_d_up(tune_roll_rd, min_d, AUTOTUNE_RD_MAX, AUTOTUNE_RD_STEP, tune_roll_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
708  break;
709  case PITCH:
710  updating_rate_d_up(tune_pitch_rd, min_d, AUTOTUNE_RD_MAX, AUTOTUNE_RD_STEP, tune_pitch_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
711  break;
712  case YAW:
713  updating_rate_d_up(tune_yaw_rLPF, AUTOTUNE_RLPF_MIN, AUTOTUNE_RLPF_MAX, AUTOTUNE_RD_STEP, tune_yaw_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
714  break;
715  }
716  break;
717  // Check results after mini-step to decrease rate D gain
718  case RD_DOWN:
719  switch (axis) {
720  case ROLL:
721  updating_rate_d_down(tune_roll_rd, min_d, AUTOTUNE_RD_STEP, tune_roll_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
722  break;
723  case PITCH:
724  updating_rate_d_down(tune_pitch_rd, min_d, AUTOTUNE_RD_STEP, tune_pitch_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
725  break;
726  case YAW:
727  updating_rate_d_down(tune_yaw_rLPF, AUTOTUNE_RLPF_MIN, AUTOTUNE_RD_STEP, tune_yaw_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
728  break;
729  }
730  break;
731  // Check results after mini-step to increase rate P gain
732  case RP_UP:
733  switch (axis) {
734  case ROLL:
735  updating_rate_p_up_d_down(tune_roll_rd, min_d, AUTOTUNE_RD_STEP, tune_roll_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
736  break;
737  case PITCH:
738  updating_rate_p_up_d_down(tune_pitch_rd, min_d, AUTOTUNE_RD_STEP, tune_pitch_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
739  break;
740  case YAW:
741  updating_rate_p_up_d_down(tune_yaw_rLPF, AUTOTUNE_RLPF_MIN, AUTOTUNE_RD_STEP, tune_yaw_rp, AUTOTUNE_RP_MIN, AUTOTUNE_RP_MAX, AUTOTUNE_RP_STEP, target_rate, test_rate_min, test_rate_max);
742  break;
743  }
744  break;
745  // Check results after mini-step to increase stabilize P gain
746  case SP_DOWN:
747  switch (axis) {
748  case ROLL:
749  updating_angle_p_down(tune_roll_sp, AUTOTUNE_SP_MIN, AUTOTUNE_SP_STEP, target_angle, test_angle_max, test_rate_min, test_rate_max);
750  break;
751  case PITCH:
752  updating_angle_p_down(tune_pitch_sp, AUTOTUNE_SP_MIN, AUTOTUNE_SP_STEP, target_angle, test_angle_max, test_rate_min, test_rate_max);
753  break;
754  case YAW:
755  updating_angle_p_down(tune_yaw_sp, AUTOTUNE_SP_MIN, AUTOTUNE_SP_STEP, target_angle, test_angle_max, test_rate_min, test_rate_max);
756  break;
757  }
758  break;
759  // Check results after mini-step to increase stabilize P gain
760  case SP_UP:
761  switch (axis) {
762  case ROLL:
763  updating_angle_p_up(tune_roll_sp, AUTOTUNE_SP_MAX, AUTOTUNE_SP_STEP, target_angle, test_angle_max, test_rate_min, test_rate_max);
764  break;
765  case PITCH:
766  updating_angle_p_up(tune_pitch_sp, AUTOTUNE_SP_MAX, AUTOTUNE_SP_STEP, target_angle, test_angle_max, test_rate_min, test_rate_max);
767  break;
768  case YAW:
769  updating_angle_p_up(tune_yaw_sp, AUTOTUNE_SP_MAX, AUTOTUNE_SP_STEP, target_angle, test_angle_max, test_rate_min, test_rate_max);
770  break;
771  }
772  break;
773  }
774 
775  // we've complete this step, finalize pids and move to next step
776  if (counter >= AUTOTUNE_SUCCESS_COUNT) {
777 
778  // reset counter
779  counter = 0;
780 
781  // reset scaling factor
782  step_scaler = 1;
783 
784  // move to the next tuning type
785  switch (tune_type) {
786  case RD_UP:
787  tune_type = TuneType(tune_type + 1);
788  break;
789  case RD_DOWN:
790  tune_type = TuneType(tune_type + 1);
791  switch (axis) {
792  case ROLL:
793  tune_roll_rd = MAX(min_d, tune_roll_rd * AUTOTUNE_RD_BACKOFF);
794  tune_roll_rp = MAX(AUTOTUNE_RP_MIN, tune_roll_rp * AUTOTUNE_RD_BACKOFF);
795  break;
796  case PITCH:
797  tune_pitch_rd = MAX(min_d, tune_pitch_rd * AUTOTUNE_RD_BACKOFF);
798  tune_pitch_rp = MAX(AUTOTUNE_RP_MIN, tune_pitch_rp * AUTOTUNE_RD_BACKOFF);
799  break;
800  case YAW:
801  tune_yaw_rLPF = MAX(AUTOTUNE_RLPF_MIN, tune_yaw_rLPF * AUTOTUNE_RD_BACKOFF);
802  tune_yaw_rp = MAX(AUTOTUNE_RP_MIN, tune_yaw_rp * AUTOTUNE_RD_BACKOFF);
803  break;
804  }
805  break;
806  case RP_UP:
807  tune_type = TuneType(tune_type + 1);
808  switch (axis) {
809  case ROLL:
810  tune_roll_rp = MAX(AUTOTUNE_RP_MIN, tune_roll_rp * AUTOTUNE_RP_BACKOFF);
811  break;
812  case PITCH:
813  tune_pitch_rp = MAX(AUTOTUNE_RP_MIN, tune_pitch_rp * AUTOTUNE_RP_BACKOFF);
814  break;
815  case YAW:
816  tune_yaw_rp = MAX(AUTOTUNE_RP_MIN, tune_yaw_rp * AUTOTUNE_RP_BACKOFF);
817  break;
818  }
819  break;
820  case SP_DOWN:
821  tune_type = TuneType(tune_type + 1);
822  break;
823  case SP_UP:
824  // we've reached the end of a D-up-down PI-up-down tune type cycle
825  tune_type = RD_UP;
826 
827  // advance to the next axis
828  bool complete = false;
829  switch (axis) {
830  case ROLL:
831  axes_completed |= AUTOTUNE_AXIS_BITMASK_ROLL;
832  tune_roll_sp = MAX(AUTOTUNE_SP_MIN, tune_roll_sp * AUTOTUNE_SP_BACKOFF);
833  tune_roll_accel = MAX(AUTOTUNE_RP_ACCEL_MIN, test_accel_max * AUTOTUNE_ACCEL_RP_BACKOFF);
834  if (pitch_enabled()) {
835  axis = PITCH;
836  } else if (yaw_enabled()) {
837  axis = YAW;
838  } else {
839  complete = true;
840  }
841  break;
842  case PITCH:
843  axes_completed |= AUTOTUNE_AXIS_BITMASK_PITCH;
844  tune_pitch_sp = MAX(AUTOTUNE_SP_MIN, tune_pitch_sp * AUTOTUNE_SP_BACKOFF);
845  tune_pitch_accel = MAX(AUTOTUNE_RP_ACCEL_MIN, test_accel_max * AUTOTUNE_ACCEL_RP_BACKOFF);
846  if (yaw_enabled()) {
847  axis = YAW;
848  } else {
849  complete = true;
850  }
851  break;
852  case YAW:
853  axes_completed |= AUTOTUNE_AXIS_BITMASK_YAW;
854  tune_yaw_sp = MAX(AUTOTUNE_SP_MIN, tune_yaw_sp * AUTOTUNE_SP_BACKOFF);
855  tune_yaw_accel = MAX(AUTOTUNE_Y_ACCEL_MIN, test_accel_max * AUTOTUNE_ACCEL_Y_BACKOFF);
856  complete = true;
857  break;
858  }
859 
860  // if we've just completed all axes we have successfully completed the autotune
861  // change to TESTING mode to allow user to fly with new gains
862  if (complete) {
863  mode = SUCCESS;
864  update_gcs(AUTOTUNE_MESSAGE_SUCCESS);
865  Log_Write_Event(EVENT_AUTOTUNE_SUCCESS);
866  AP_Notify::events.autotune_complete = true;
867  } else {
868  AP_Notify::events.autotune_next_axis = true;
869  }
870  break;
871  }
872  }
873 
874  // reverse direction
875  positive_direction = !positive_direction;
876 
877  if (axis == YAW) {
878  attitude_control->input_euler_angle_roll_pitch_yaw(0.0f, 0.0f, ahrs_view->yaw_sensor, false);
879  }
880 
881  // set gains to their intra-test values (which are very close to the original gains)
882  load_gains(GAIN_INTRA_TEST);
883 
884  // reset testing step
885  step = WAITING_FOR_LEVEL;
886  step_start_time_ms = now;
887  level_start_time_ms = step_start_time_ms;
888  step_time_limit_ms = AUTOTUNE_REQUIRED_LEVEL_TIME_MS;
889  break;
890  }
891 }

1625

1627  // we are not trying to hold position
1628  return;
1629  }

1632  if (!position_ok()) {
1633  return;
1634  }
1635

1637  have_position = true;
1638  start_position = inertial_nav->get_position();
1639  }
1640 
1641  // don't go past 10 degrees, as autotune result would deteriorate too much
1642  const float angle_max_cd = 1000;
1643 
1644  // hit the 10 degree limit at 20 meters position error
1645  const float dist_limit_cm = 2000;

1648  // target position. That corresponds to a lean angle of 2.5 degrees
1649  const float yaw_dist_limit_cm = 500;

1656  return;

1660  very simple linear controller
