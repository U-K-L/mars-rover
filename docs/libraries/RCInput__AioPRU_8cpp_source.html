49 /*
50  called at 1kHz to check for new pulse capture data from the PRU
51  */
52 void RCInput_AioPRU::_timer_tick()
53 {
54  while (ring_buffer->ring_head != ring_buffer->ring_tail) {
55  if (ring_buffer->ring_tail >= NUM_RING_ENTRIES) {
56  // invalid ring_tail from PRU - ignore RC input
57  return;
58  }
59  _process_rc_pulse((ring_buffer->buffer[ring_buffer->ring_head].s1_t) / TICK_PER_US,
60  (ring_buffer->buffer[ring_buffer->ring_head].s0_t) / TICK_PER_US);
61  // move to the next ring buffer entry
62  ring_buffer->ring_head = (ring_buffer->ring_head + 1) % NUM_RING_ENTRIES;
63  }
64 }
