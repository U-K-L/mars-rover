41 void AP_OpticalFlow_Onboard::update()
42 {
43  AP_HAL::OpticalFlow::Data_Frame data_frame;
44  // read at maximum 10Hz
45  uint32_t now = AP_HAL::millis();
46  if (now - _last_read_ms < 100) {
47  return;
48  }
49  _last_read_ms = now;
50 
51  if (!hal.opticalflow->read(data_frame)) {
52  return;
53  }
54 
55  struct OpticalFlow::OpticalFlow_state state;
56  state.surface_quality = data_frame.quality;
57  if (data_frame.delta_time > 0) {
58  const Vector2f flowScaler = _flowScaler();
59  float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
60  float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
61 
62  // delta_time is in microseconds and flow is in milliradians
63  // per second, so multiply by 1000
64  state.flowRate.x = flowScaleFactorX * 1000.0f /
65  float(data_frame.delta_time) *
66  data_frame.pixel_flow_x_integral;
67 
68  state.flowRate.y = flowScaleFactorY * 1000.0f /
69  float(data_frame.delta_time) *
70  data_frame.pixel_flow_y_integral;
71 
72  // delta_time is in microseconds so multiply to get back to seconds
73  state.bodyRate.x = 1000000.0f / float(data_frame.delta_time) *
74  data_frame.gyro_x_integral;
75 
76  state.bodyRate.y = 1000000.0f / float(data_frame.delta_time) *
77  data_frame.gyro_y_integral;
78 
79  _applyYaw(state.flowRate);
80  } else {
81  state.flowRate.zero();
82  state.bodyRate.zero();
83  }
84 
85  // copy results to front end
86  _update_frontend(state);
87 
88 #if OPTICALFLOW_ONBOARD_DEBUG
89  hal.console->printf("FLOW_ONBOARD qual:%u FlowRateX:%4.2f Y:%4.2f"
90  "BodyRateX:%4.2f Y:%4.2f, delta_time = %u\n",
91  (unsigned)state.surface_quality,
92  (double)state.flowRate.x,
93  (double)state.flowRate.y,
94  (double)state.bodyRate.x,
95  (double)state.bodyRate.y,
96  data_frame.delta_time);
97 #endif
98 }
