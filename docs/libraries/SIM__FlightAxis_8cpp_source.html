147 /*
148  extremely primitive SOAP parser that assumes the format used by FlightAxis
149 */
150 void FlightAxis::parse_reply(const char *reply)
151 {
152  const char *reply0 = reply;
153  for (uint16_t i=0; i<num_keys; i++) {
154  const char *p = strstr(reply, keytable[i].key);
155  if (p == nullptr) {
156  p = strstr(reply0, keytable[i].key);
157  }
158  if (p == nullptr) {
159  printf("Failed to find key %s\n", keytable[i].key);
160  controller_started = false;
161  break;
162  }
163  p += strlen(keytable[i].key) + 1;
164  double v;
165  if (strncmp(p, "true", 4) == 0) {
166  v = 1;
167  } else if (strncmp(p, "false", 5) == 0) {
168  v = 0;
169  } else {
170  v = atof(p);
171  }
172  keytable[i].ref = v;
173  // this assumes key order and allows us to decode arrays
174  p = strchr(p, '>');
175  if (p != nullptr) {
176  reply = p;
177  }
178  }
179 }

182 /*
183  make a SOAP request, returning body of reply
184  */
185 char *FlightAxis::soap_request(const char *action, const char *fmt, ...)
186 {
187  va_list ap;
188  char *req1;
189 
190  va_start(ap, fmt);
191  vasprintf(&req1, fmt, ap);
192  va_end(ap);
193 
194  //printf("%s\n", req1);
195 
196  // open SOAP socket to FlightAxis
197  SocketAPM sock(false);
198  if (!sock.connect(controller_ip, controller_port)) {
199  free(req1);
200  return nullptr;
201  }
202  sock.set_blocking(false);
203 
204  char *req;
205  asprintf(&req, R"(POST / HTTP/1.1
206 soapaction: '%s'
207 content-length: %u
208 content-type: text/xml;charset='UTF-8'
209 Connection: Keep-Alive
210 
211 %s)",
212  action,
213  (unsigned)strlen(req1), req1);
214  sock.send(req, strlen(req));
215  free(req1);
216  free(req);
217  char reply[10000];
218  memset(reply, 0, sizeof(reply));
219  ssize_t ret = sock.recv(reply, sizeof(reply)-1, 1000);
220  if (ret <= 0) {
221  printf("No data\n");
222  return nullptr;
223  }
224  char *p = strstr(reply, "Content-Length: ");
225  if (p == nullptr) {
226  printf("No Content-Length\n");
227  return nullptr;
228  }
229 
230  // get the content length
231  uint32_t content_length = strtoul(p+16, nullptr, 10);
232  char *body = strstr(p, "\r\n\r\n");
233  if (body == nullptr) {
234  printf("No body\n");
235  return nullptr;
236  }
237  body += 4;
238 
239  // get the rest of the body
240  int32_t expected_length = content_length + (body - reply);
241  if (expected_length >= (int32_t)sizeof(reply)) {
242  printf("Reply too large %i\n", expected_length);
243  return nullptr;
244  }
245  while (ret < expected_length) {
246  ssize_t ret2 = sock.recv(&reply[ret], sizeof(reply)-(1+ret), 100);
247  if (ret2 <= 0) {
248  return nullptr;
249  }
250  // nul terminate
251  reply[ret+ret2] = 0;
252  ret += ret2;
253  }
254  return strdup(reply);
255 }

259 void FlightAxis::exchange_data(const struct sitl_input &input)
260 {
261  if (!controller_started ||
262  is_zero(state.m_flightAxisControllerIsActive) ||
263  !is_zero(state.m_resetButtonHasBeenPressed)) {
264  printf("Starting controller at %s\n", controller_ip);
265  // call a restore first. This allows us to connect after the aircraft is changed in RealFlight
266  char *reply = soap_request("RestoreOriginalControllerDevice", R"(<?xml version='1.0' encoding='UTF-8'?>
267 <soap:Envelope xmlns:soap='http://schemas.xmlsoap.org/soap/envelope/' xmlns:xsd='http://www.w3.org/2001/XMLSchema' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>
268 <soap:Body>
269 <RestoreOriginalControllerDevice><a>1</a><b>2</b></RestoreOriginalControllerDevice>
270 </soap:Body>
271 </soap:Envelope>)");
272  free(reply);
273  reply = soap_request("InjectUAVControllerInterface", R"(<?xml version='1.0' encoding='UTF-8'?>
274 <soap:Envelope xmlns:soap='http://schemas.xmlsoap.org/soap/envelope/' xmlns:xsd='http://www.w3.org/2001/XMLSchema' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>
275 <soap:Body>
276 <InjectUAVControllerInterface><a>1</a><b>2</b></InjectUAVControllerInterface>
277 </soap:Body>
278 </soap:Envelope>)");
279  free(reply);
280  activation_frame_counter = frame_counter;
281  controller_started = true;
282  }
283 
284  // maximum number of servos to send is 12 with new FlightAxis
285  float scaled_servos[12];
286  for (uint8_t i=0; i<ARRAY_SIZE(scaled_servos); i++) {
287  scaled_servos[i] = (input.servos[i] - 1000) / 1000.0f;
288  }
289 
290  if (rev4_servos) {
291  // swap first 4 and last 4 servos, for quadplane testing
292  float saved[4];
293  memcpy(saved, &scaled_servos[0], sizeof(saved));
294  memcpy(&scaled_servos[0], &scaled_servos[4], sizeof(saved));
295  memcpy(&scaled_servos[4], saved, sizeof(saved));
296  }
297 
298  if (heli_demix) {
299  // FlightAxis expects "roll/pitch/collective/yaw" input
300  float swash1 = scaled_servos[0];
301  float swash2 = scaled_servos[1];
302  float swash3 = scaled_servos[2];
303 
304  float roll_rate = swash1 - swash2;
305  float pitch_rate = -((swash1+swash2) / 2.0f - swash3);
306 
307  scaled_servos[0] = constrain_float(roll_rate + 0.5, 0, 1);
308  scaled_servos[1] = constrain_float(pitch_rate + 0.5, 0, 1);
309  }
310 
311 
312  char *reply = soap_request("ExchangeData", R"(<?xml version='1.0' encoding='UTF-8'?><soap:Envelope xmlns:soap='http://schemas.xmlsoap.org/soap/envelope/' xmlns:xsd='http://www.w3.org/2001/XMLSchema' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>
313 <soap:Body>
314 <ExchangeData>
315 <pControlInputs>
316 <m-selectedChannels>4095</m-selectedChannels>
317 <m-channelValues-0to1>
318 <item>%.4f</item>
319 <item>%.4f</item>
320 <item>%.4f</item>
321 <item>%.4f</item>
322 <item>%.4f</item>
323 <item>%.4f</item>
324 <item>%.4f</item>
325 <item>%.4f</item>
326 <item>%.4f</item>
327 <item>%.4f</item>
328 <item>%.4f</item>
329 <item>%.4f</item>
330 </m-channelValues-0to1>
331 </pControlInputs>
332 </ExchangeData>
333 </soap:Body>
334 </soap:Envelope>)",
335  scaled_servos[0],
336  scaled_servos[1],
337  scaled_servos[2],
338  scaled_servos[3],
339  scaled_servos[4],
340  scaled_servos[5],
341  scaled_servos[6],
342  scaled_servos[7],
343  scaled_servos[8],
344  scaled_servos[9],
345  scaled_servos[10],
346  scaled_servos[11]);
347 
348  if (reply) {
349  WITH_SEMAPHORE(mutex);
350  double lastt_s = state.m_currentPhysicsTime_SEC;
351  parse_reply(reply);
352  double dt = state.m_currentPhysicsTime_SEC - lastt_s;
353  if (dt > 0 && dt < 0.1) {
354  if (average_frame_time_s < 1.0e-6) {
355  average_frame_time_s = dt;
356  }
357  average_frame_time_s = average_frame_time_s * 0.98 + dt * 0.02;
358  }
359  socket_frame_counter++;
360  free(reply);
361  }

365 /*
366  update the FlightAxis simulation by one time step
367  */
368 void FlightAxis::update(const struct sitl_input &input)
369 {
370  WITH_SEMAPHORE(mutex);
371 
372  last_input = input;
373 
374  double dt_seconds = state.m_currentPhysicsTime_SEC - last_time_s;
375  if (dt_seconds < 0) {
376  // cope with restarting RealFlight while connected
377  initial_time_s = time_now_us * 1.0e-6f;
378  last_time_s = state.m_currentPhysicsTime_SEC;
379  position_offset.zero();
380  return;
381  }
382  if (dt_seconds < 0.00001f) {
383  float delta_time = 0.001;
384  // don't go past the next expected frame
385  if (delta_time + extrapolated_s > average_frame_time_s) {
386  delta_time = average_frame_time_s - extrapolated_s;
387  }
388  if (delta_time <= 0) {
389  usleep(1000);
390  return;
391  }
392  time_now_us += delta_time * 1.0e6;
393  extrapolate_sensors(delta_time);
394  update_position();
395  update_mag_field_bf();
396  usleep(delta_time*1.0e6);
397  extrapolated_s += delta_time;
398  report_FPS();
399  return;
400  }
401 
402  extrapolated_s = 0;
403 
404  if (initial_time_s <= 0) {
405  dt_seconds = 0.001f;
406  initial_time_s = state.m_currentPhysicsTime_SEC - dt_seconds;
407  }
408 
409  /*
410  the queternion convention in realflight seems to have Z negative
411  */
412  Quaternion quat(state.m_orientationQuaternion_W,
413  state.m_orientationQuaternion_Y,
414  state.m_orientationQuaternion_X,
415  -state.m_orientationQuaternion_Z);
416  quat.rotation_matrix(dcm);
417 
418  gyro = Vector3f(radians(constrain_float(state.m_rollRate_DEGpSEC, -2000, 2000)),
419  radians(constrain_float(state.m_pitchRate_DEGpSEC, -2000, 2000)),
420  -radians(constrain_float(state.m_yawRate_DEGpSEC, -2000, 2000))) * target_speedup;
421 
422  velocity_ef = Vector3f(state.m_velocityWorldU_MPS,
423  state.m_velocityWorldV_MPS,
424  state.m_velocityWorldW_MPS);
425  position = Vector3f(state.m_aircraftPositionY_MTR,
426  state.m_aircraftPositionX_MTR,
427  -state.m_altitudeASL_MTR - home.alt*0.01);
428 
429  accel_body(state.m_accelerationBodyAX_MPS2,
430  state.m_accelerationBodyAY_MPS2,
431  state.m_accelerationBodyAZ_MPS2);
432 
433  // accel on the ground is nasty in realflight, and prevents helicopter disarm
434  if (!is_zero(state.m_isTouchingGround)) {
435  Vector3f accel_ef = (velocity_ef - last_velocity_ef) / dt_seconds;
436  accel_ef.z -= GRAVITY_MSS;
437  accel_body = dcm.transposed() * accel_ef;
438  }
439 
440  // limit to 16G to match pixhawk
441  float a_limit = GRAVITY_MSS*16;
442  accel_body.x = constrain_float(accel_body.x, -a_limit, a_limit);
443  accel_body.y = constrain_float(accel_body.y, -a_limit, a_limit);
444  accel_body.z = constrain_float(accel_body.z, -a_limit, a_limit);
445 
446  // offset based on first position to account for offset in RF world
447  if (position_offset.is_zero() || !is_zero(state.m_resetButtonHasBeenPressed)) {
448  position_offset = position;
449  }
450  position -= position_offset;
451 
452  airspeed = state.m_airspeed_MPS;
453 
454  /* for pitot airspeed we need the airspeed along the X axis. We
455  can't get that from m_airspeed_MPS, so instead we canculate it
456  from wind vector and ground speed
457  */
458  Vector3f m_wind_ef(-state.m_windY_MPS,-state.m_windX_MPS,-state.m_windZ_MPS);
459  Vector3f airspeed_3d_ef = m_wind_ef + velocity_ef;
460  Vector3f airspeed3d = dcm.mul_transpose(airspeed_3d_ef);
461 
462  airspeed_pitot = MAX(airspeed3d.x,0);
463 
464 #if 0
465  printf("WIND: %.1f %.1f %.1f AS3D %.1f %.1f %.1f\n",
466  state.m_windX_MPS,
467  state.m_windY_MPS,
468  state.m_windZ_MPS,
469  airspeed3d.x,
470  airspeed3d.y,
471  airspeed3d.z);
472 #endif
473 
474  battery_voltage = state.m_batteryVoltage_VOLTS;
475  battery_current = state.m_batteryCurrentDraw_AMPS;
476  rpm1 = state.m_heliMainRotorRPM;
477  rpm2 = state.m_propRPM;
478 
479  /*
480  the interlink interface supports 8 input channels
481  */
482  rcin_chan_count = 8;
483  for (uint8_t i=0; i<rcin_chan_count; i++) {
484  rcin[i] = state.rcin[i];
485  }
486 
487  update_position();
488  time_advance();
489  uint64_t new_time_us = (state.m_currentPhysicsTime_SEC - initial_time_s)*1.0e6;
490  if (new_time_us < time_now_us) {
491  uint64_t dt_us = time_now_us - new_time_us;
492  if (dt_us > 500000) {
493  // time going backwards
494  time_now_us = new_time_us;
495  }
496  } else {
497  time_now_us = new_time_us;
498  }
499 
500  last_time_s = state.m_currentPhysicsTime_SEC;
501 
502  last_velocity_ef = velocity_ef;
503 
504  // update magnetic field
505  update_mag_field_bf();
506 
507  report_FPS();
508 }
