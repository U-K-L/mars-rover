156 void AP_InertialSensor_Invensensev2::start()
157 {
158  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
159  return;
160  }
161 
162  // initially run the bus at low speed
163  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
164 
165  // only used for wake-up in accelerometer only low power mode
166  _register_write(INV2REG_PWR_MGMT_2, 0x00);
167  hal.scheduler->delay(1);
168 
169  // always use FIFO
170  _fifo_reset();
171 
172  // grab the used instances
173  enum DevTypes gdev, adev;
174  switch (_inv2_type) {
175  case Invensensev2_ICM20648:
176  gdev = DEVTYPE_INS_ICM20648;
177  adev = DEVTYPE_INS_ICM20648;
178  // using 16g full range, 2048 LSB/g
179  _accel_scale = (GRAVITY_MSS / 2048);
180  break;
181  case Invensensev2_ICM20649:
182  // 20649 is setup for 30g full scale, 1024 LSB/g
183  gdev = DEVTYPE_INS_ICM20649;
184  adev = DEVTYPE_INS_ICM20649;
185  _accel_scale = (GRAVITY_MSS / 1024);
186  break;
187  case Invensensev2_ICM20948:
188  default:
189  gdev = DEVTYPE_INS_ICM20948;
190  adev = DEVTYPE_INS_ICM20948;
191  // using 16g full range, 2048 LSB/g
192  _accel_scale = (GRAVITY_MSS / 2048);
193  break;
194  }
195 
196  _gyro_instance = _imu.register_gyro(1125, _dev->get_bus_id_devtype(gdev));
197  _accel_instance = _imu.register_accel(1125, _dev->get_bus_id_devtype(adev));
198 
199  // setup on-sensor filtering and scaling
200  _set_filter_and_scaling();
201 #if INVENSENSE_EXT_SYNC_ENABLE
202  _register_write(INV2REG_FSYNC_CONFIG, FSYNC_CONFIG_EXT_SYNC_AZ, true);
203 #endif
204  // update backend sample rate
205  _set_accel_raw_sample_rate(_accel_instance, _backend_rate_hz);
206  _set_gyro_raw_sample_rate(_gyro_instance, _backend_rate_hz);
207 
208  // indicate what multiplier is appropriate for the sensors'
209  // readings to fit them into an int16_t:
210  _set_raw_sample_accel_multiplier(_accel_instance, multiplier_accel);
211 
212  if (_fast_sampling) {
213  hal.console->printf("INV2[%u]: enabled fast sampling rate %uHz/%uHz\n",
214  _accel_instance, _backend_rate_hz*_fifo_downsample_rate, _backend_rate_hz);
215  }
216 
217  // set sample rate to 1.125KHz
218  _register_write(INV2REG_GYRO_SMPLRT_DIV, 0, true);
219  hal.scheduler->delay(1);
220 
221  // configure interrupt to fire when new data arrives
222  _register_write(INV2REG_INT_ENABLE_1, 0x01);
223  hal.scheduler->delay(1);
224 
225  // now that we have initialised, we set the bus speed to high
226  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
227 
228  _dev->get_semaphore()->give();
229 
230  // setup sensor rotations from probe()
231  set_gyro_orientation(_gyro_instance, _rotation);
232  set_accel_orientation(_accel_instance, _rotation);
233 
234  // setup scale factors for fifo data after downsampling
235  _fifo_accel_scale = _accel_scale / (MAX(_fifo_downsample_rate,2)/2);
236  _fifo_gyro_scale = GYRO_SCALE / _fifo_downsample_rate;
237 
238  // allocate fifo buffer
239  _fifo_buffer = (uint8_t *)hal.util->malloc_type(INV2_FIFO_BUFFER_LEN * INV2_SAMPLE_SIZE, AP_HAL::Util::MEM_DMA_SAFE);
240  if (_fifo_buffer == nullptr) {
241  AP_HAL::panic("Invensense: Unable to allocate FIFO buffer");
242  }
243 
244  // start the timer process to read samples
245  _dev->register_periodic_callback(1265625UL / _backend_rate_hz, FUNCTOR_BIND_MEMBER(&AP_InertialSensor_Invensensev2::_poll_data, void));
246 }

306 bool AP_InertialSensor_Invensensev2::_accumulate(uint8_t *samples, uint8_t n_samples)
307 {
308  for (uint8_t i = 0; i < n_samples; i++) {
309  const uint8_t *data = samples + INV2_SAMPLE_SIZE * i;
310  Vector3f accel, gyro;
311  bool fsync_set = false;
312 
313 #if INVENSENSE_EXT_SYNC_ENABLE
314  fsync_set = (int16_val(data, 2) & 1U) != 0;
315 #endif
316 
317  accel = Vector3f(int16_val(data, 1),
318  int16_val(data, 0),
319  -int16_val(data, 2));
320  accel *= _accel_scale;
321 
322  int16_t t2 = int16_val(data, 6);
323  if (!_check_raw_temp(t2)) {
324  debug("temp reset IMU[%u] %d %d", _accel_instance, _raw_temp, t2);
325  _fifo_reset();
326  return false;
327  }
328  float temp = t2 * temp_sensitivity + temp_zero;
329 
330  gyro = Vector3f(int16_val(data, 4),
331  int16_val(data, 3),
332  -int16_val(data, 5));
333  gyro *= GYRO_SCALE;
334 
335  _rotate_and_correct_accel(_accel_instance, accel);
336  _rotate_and_correct_gyro(_gyro_instance, gyro);
337 
338  _notify_new_accel_raw_sample(_accel_instance, accel, 0, fsync_set);
339  _notify_new_gyro_raw_sample(_gyro_instance, gyro);
340 
341  _temp_filtered = _temp_filter.apply(temp);
342  }
343  return true;
344 }

346 /*
347  when doing fast sampling the sensor gives us 9k samples/second. Every 2nd accel sample is a duplicate.
348 
349  To filter this we first apply a 1p low pass filter at 188Hz, then we
350  average over 8 samples to bring the data rate down to 1kHz. This
351  gives very good aliasing rejection at frequencies well above what
352  can be handled with 1kHz sample rates.
353  */
354 bool AP_InertialSensor_Invensensev2::_accumulate_sensor_rate_sampling(uint8_t *samples, uint8_t n_samples)
355 {
356  int32_t tsum = 0;
357  int32_t unscaled_clip_limit = _clip_limit / _accel_scale;
358  bool clipped = false;
359  bool ret = true;
360 
361  for (uint8_t i = 0; i < n_samples; i++) {
362  const uint8_t *data = samples + INV2_SAMPLE_SIZE * i;
363 
364  // use temperature to detect FIFO corruption
365  int16_t t2 = int16_val(data, 6);
366  if (!_check_raw_temp(t2)) {
367  debug("temp reset IMU[%u] %d %d", _accel_instance, _raw_temp, t2);
368  _fifo_reset();
369  ret = false;
370  break;
371  }
372  tsum += t2;
373  // accel data is at 4kHz
374  if ((_accum.count & 1) == 0) {
375  Vector3f a(int16_val(data, 1),
376  int16_val(data, 0),
377  -int16_val(data, 2));
378  if (fabsf(a.x) > unscaled_clip_limit ||
379  fabsf(a.y) > unscaled_clip_limit ||
380  fabsf(a.z) > unscaled_clip_limit) {
381  clipped = true;
382  }
383  _accum.accel += _accum.accel_filter.apply(a);
384  Vector3f a2 = a * _accel_scale;
385  _notify_new_accel_sensor_rate_sample(_accel_instance, a2);
386  }
387 
388  Vector3f g(int16_val(data, 4),
389  int16_val(data, 3),
390  -int16_val(data, 5));
391 
392  Vector3f g2 = g * GYRO_SCALE;
393  _notify_new_gyro_sensor_rate_sample(_gyro_instance, g2);
394 
395  _accum.gyro += _accum.gyro_filter.apply(g);
396  _accum.count++;
397 
398  if (_accum.count == _fifo_downsample_rate) {
399  _accum.accel *= _fifo_accel_scale;
400  _accum.gyro *= _fifo_gyro_scale;
401  _rotate_and_correct_accel(_accel_instance, _accum.accel);
402  _rotate_and_correct_gyro(_gyro_instance, _accum.gyro);
403 
404  _notify_new_accel_raw_sample(_accel_instance, _accum.accel, 0, false);
405  _notify_new_gyro_raw_sample(_gyro_instance, _accum.gyro);
406 
407  _accum.accel.zero();
408  _accum.gyro.zero();
409  _accum.count = 0;
410  }
411  }
412 
413  if (clipped) {
414  increment_clip_count(_accel_instance);
415  }
416 
417  if (ret) {
418  float temp = (static_cast<float>(tsum)/n_samples)*temp_sensitivity + temp_zero;
419  _temp_filtered = _temp_filter.apply(temp);
420  }
421 
422  return ret;
423 }

425 void AP_InertialSensor_Invensensev2::_read_fifo()
426 {
427  uint8_t n_samples;
428  uint16_t bytes_read;
429  uint8_t *rx = _fifo_buffer;
430  bool need_reset = false;
431 
432  if (!_block_read(INV2REG_FIFO_COUNTH, rx, 2)) {
433  goto check_registers;
434  }
435 
436  bytes_read = uint16_val(rx, 0);
437  n_samples = bytes_read / INV2_SAMPLE_SIZE;
438 
439  if (n_samples == 0) {
440  /* Not enough data in FIFO */
441  goto check_registers;
442  }
443 
444  /*
445  testing has shown that if we have more than 32 samples in the
446  FIFO then some of those samples will be corrupt. It always is
447  the ones at the end of the FIFO, so clear those with a reset
448  once we've read the first 24. Reading 24 gives us the normal
449  number of samples for fast sampling at 400Hz
450 
451  On I2C with the much lower clock rates we need a lower threshold
452  or we may never catch up
453  */
454  if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C) {
455  if (n_samples > 4) {
456  need_reset = true;
457  n_samples = 4;
458  }
459  } else {
460  if (n_samples > 32) {
461  need_reset = true;
462  n_samples = 24;
463  }
464  }
465  while (n_samples > 0) {
466  uint8_t n = MIN(n_samples, INV2_FIFO_BUFFER_LEN);
467  if (!_dev->set_chip_select(true)) {
468  if (!_block_read(INV2REG_FIFO_R_W, rx, n * INV2_SAMPLE_SIZE)) {
469  goto check_registers;
470  }
471  } else {
472  // this ensures we keep things nicely setup for DMA
473  _select_bank(GET_BANK(INV2REG_FIFO_R_W));
474  uint8_t reg = GET_REG(INV2REG_FIFO_R_W) | 0x80;
475  if (!_dev->transfer(&reg, 1, nullptr, 0)) {
476  _dev->set_chip_select(false);
477  goto check_registers;
478  }
479  memset(rx, 0, n * INV2_SAMPLE_SIZE);
480  if (!_dev->transfer(rx, n * INV2_SAMPLE_SIZE, rx, n * INV2_SAMPLE_SIZE)) {
481  hal.console->printf("INV2: error in fifo read %u bytes\n", n * INV2_SAMPLE_SIZE);
482  _dev->set_chip_select(false);
483  goto check_registers;
484  }
485  _dev->set_chip_select(false);
486  }
487 
488  if (_fast_sampling) {
489  if (!_accumulate_sensor_rate_sampling(rx, n)) {
490  debug("IMU[%u] stop at %u of %u", _accel_instance, n_samples, bytes_read/INV2_SAMPLE_SIZE);
491  break;
492  }
493  } else {
494  if (!_accumulate(rx, n)) {
495  break;
496  }
497  }
498  n_samples -= n;
499  }
500 
501  if (need_reset) {
502  //debug("fifo reset n_samples %u", bytes_read/INV2_SAMPLE_SIZE);
503  _fifo_reset();
504  }
505 
506 check_registers:
507  // check next register value for correctness
508  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
509  if (!_dev->check_next_register()) {
510  _inc_gyro_error_count(_gyro_instance);
511  _inc_accel_error_count(_accel_instance);
512  }
513  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
514 }

562 /*
563  set the DLPF filter frequency and Gyro Accel Scaling. Assumes caller has taken semaphore
564  */
565 void AP_InertialSensor_Invensensev2::_set_filter_and_scaling(void)
566 {
567  uint8_t gyro_config = (_inv2_type == Invensensev2_ICM20649)?BITS_GYRO_FS_2000DPS_20649 : BITS_GYRO_FS_2000DPS;
568  uint8_t accel_config = (_inv2_type == Invensensev2_ICM20649)?BITS_ACCEL_FS_30G_20649:BITS_ACCEL_FS_16G;
569 
570  // assume 1.125kHz sampling to start
571  _fifo_downsample_rate = 1;
572  _backend_rate_hz = 1125;
573 
574  if (enable_fast_sampling(_accel_instance)) {
575  _fast_sampling = _dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI;
576  if (_fast_sampling) {
577  if (get_sample_rate_hz() <= 1125) {
578  _fifo_downsample_rate = 8;
579  } else if (get_sample_rate_hz() <= 2250) {
580  _fifo_downsample_rate = 4;
581  } else {
582  _fifo_downsample_rate = 2;
583  }
584  // calculate rate we will be giving samples to the backend
585  _backend_rate_hz *= (8 / _fifo_downsample_rate);
586 
587  // for logging purposes set the oversamping rate
588  _set_accel_oversampling(_accel_instance, _fifo_downsample_rate/2);
589  _set_gyro_oversampling(_gyro_instance, _fifo_downsample_rate);
590 
591  _set_accel_sensor_rate_sampling_enabled(_accel_instance, true);
592  _set_gyro_sensor_rate_sampling_enabled(_gyro_instance, true);
593 
594  /* set divider for internal sample rate to 0x1F when fast
595  sampling enabled. This reduces the impact of the slave
596  sensor on the sample rate.
597  */

637 bool AP_InertialSensor_Invensensev2::_hardware_init(void)
638 {
639  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
640  return false;
641  }
642 
643  // disabled setup of checked registers as it can't cope with bank switching
644  // _dev->setup_checked_registers(7, _dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C?200:20);
645 
646  // initially run the bus at low speed
647  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
648 
649  if (!_check_whoami()) {
650  _dev->get_semaphore()->give();
651  return false;
652  }
653 
654  // Chip reset
655  uint8_t tries;
656  for (tries = 0; tries < 5; tries++) {
657  _last_stat_user_ctrl = _register_read(INV2REG_USER_CTRL);
658 
659  /* First disable the master I2C to avoid hanging the slaves on the
660  * aulixiliar I2C bus - it will be enabled again if the AuxiliaryBus
661  * is used */
662  if (_last_stat_user_ctrl & BIT_USER_CTRL_I2C_MST_EN) {
663  _last_stat_user_ctrl &= ~BIT_USER_CTRL_I2C_MST_EN;
664  _register_write(INV2REG_USER_CTRL, _last_stat_user_ctrl);
665  hal.scheduler->delay(10);
666  }
667 
668  /* reset device */
669  _register_write(INV2REG_PWR_MGMT_1, BIT_PWR_MGMT_1_DEVICE_RESET);
670  hal.scheduler->delay(100);
671 
672  /* bus-dependent initialization */
673  if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI) {
674  /* Disable I2C bus if SPI selected (Recommended in Datasheet to be
675  * done just after the device is reset) */
676  _last_stat_user_ctrl |= BIT_USER_CTRL_I2C_IF_DIS;
677  _register_write(INV2REG_USER_CTRL, _last_stat_user_ctrl);
678  }
679 
680  // Wake up device and select Auto clock. Note that the
681  // Invensense starts up in sleep mode, and it can take some time
682  // for it to come out of sleep
683  _register_write(INV2REG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_AUTO);
684  hal.scheduler->delay(5);
685 
686  // check it has woken up
687  if (_register_read(INV2REG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_AUTO) {
688  break;
689  }
690 
691  hal.scheduler->delay(10);
692  if (_data_ready()) {
693  break;
694  }
695  }
696 
697  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
698 
699  if (tries == 5) {
700  hal.console->printf("Failed to boot Invensense 5 times\n");
701  _dev->get_semaphore()->give();
702  return false;
703  }
704 
705  if (_inv2_type == Invensensev2_ICM20649) {
706  _clip_limit = 29.5f * GRAVITY_MSS;
707  }
708  _dev->get_semaphore()->give();
709 
710  return true;
711 }
