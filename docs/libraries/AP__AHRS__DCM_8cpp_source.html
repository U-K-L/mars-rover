134 // update the DCM matrix using only the gyros
135 void
136 AP_AHRS_DCM::matrix_update(float _G_Dt)
137 {
138  // note that we do not include the P terms in _omega. This is
139  // because the spin_rate is calculated from _omega.length(),
140  // and including the P terms would give positive feedback into
141  // the _P_gain() calculation, which can lead to a very large P
142  // value
143  _omega.zero();
144 
145  // average across first two healthy gyros. This reduces noise on
146  // systems with more than one gyro. We don't use the 3rd gyro
147  // unless another is unhealthy as 3rd gyro on PH2 has a lot more
148  // noise
149  uint8_t healthy_count = 0;
150  Vector3f delta_angle;
151  const AP_InertialSensor &_ins = AP::ins();
152  for (uint8_t i=0; i<_ins.get_gyro_count(); i++) {
153  if (_ins.use_gyro(i) && healthy_count < 2) {
154  Vector3f dangle;
155  if (_ins.get_delta_angle(i, dangle)) {
156  healthy_count++;
157  delta_angle += dangle;
158  }
159  }
160  }
161  if (healthy_count > 1) {
162  delta_angle /= healthy_count;
163  }
164  if (_G_Dt > 0) {
165  _omega = delta_angle / _G_Dt;
166  _omega += _omega_I;
167  _dcm_matrix.rotate((_omega + _omega_P + _omega_yaw_P) * _G_Dt);
168  }
169 }

172 /*
173  * reset the DCM matrix and omega. Used on ground start, and on
174  * extreme errors in the matrix
175  */
176 void
177 AP_AHRS_DCM::reset(bool recover_eulers)
178 {
179  // support locked access functions to AHRS data
180  WITH_SEMAPHORE(_rsem);
181 
182  // reset the integration terms
183  _omega_I.zero();
184  _omega_P.zero();
185  _omega_yaw_P.zero();
186  _omega.zero();
187 
188  // if the caller wants us to try to recover to the current
189  // attitude then calculate the dcm matrix from the current
190  // roll/pitch/yaw values
191  if (hal.util->was_watchdog_reset() && AP_HAL::millis() < 10000) {
192  const AP_HAL::Util::PersistentData &pd = hal.util->persistent_data;
193  roll = pd.roll_rad;
194  pitch = pd.pitch_rad;
195  yaw = pd.yaw_rad;
196  _dcm_matrix.from_euler(roll, pitch, yaw);
197  gcs().send_text(MAV_SEVERITY_INFO, "Restored watchdog attitude %.0f %.0f %.0f",
198  degrees(roll), degrees(pitch), degrees(yaw));
199  } else if (recover_eulers && !isnan(roll) && !isnan(pitch) && !isnan(yaw)) {
200  _dcm_matrix.from_euler(roll, pitch, yaw);
201  } else {
202 
203  // Use the measured accel due to gravity to calculate an initial
204  // roll and pitch estimate
205 
206  AP_InertialSensor &_ins = AP::ins();
207 
208  // Get body frame accel vector
209  Vector3f initAccVec = _ins.get_accel();
210  uint8_t counter = 0;
211 
212  // the first vector may be invalid as the filter starts up
213  while ((initAccVec.length() < 9.0f || initAccVec.length() > 11) && counter++ < 20) {
214  _ins.wait_for_sample();
215  _ins.update();
216  initAccVec = _ins.get_accel();
217  }
218 
219  // normalise the acceleration vector
220  if (initAccVec.length() > 5.0f) {
221  // calculate initial pitch angle
222  pitch = atan2f(initAccVec.x, norm(initAccVec.y, initAccVec.z));
223  // calculate initial roll angle
224  roll = atan2f(-initAccVec.y, -initAccVec.z);
225  } else {
226  // If we can't use the accel vector, then align flat
227  roll = 0.0f;
228  pitch = 0.0f;
229  }
230  _dcm_matrix.from_euler(roll, pitch, 0);
231 
232  }
233 
234  if (_last_startup_ms == 0) {
235  load_watchdog_home();
236  }
237  _last_startup_ms = AP_HAL::millis();
238 }

277 // renormalise one vector component of the DCM matrix
278 // this will return false if renormalization fails
279 bool
280 AP_AHRS_DCM::renorm(Vector3f const &a, Vector3f &result)
281 {
282  // numerical errors will slowly build up over time in DCM,
283  // causing inaccuracies. We can keep ahead of those errors
284  // using the renormalization technique from the DCM IMU paper
285  // (see equations 18 to 21).
286 
287  // For APM we don't bother with the taylor expansion
288  // optimisation from the paper as on our 2560 CPU the cost of
289  // the sqrt() is 44 microseconds, and the small time saving of
290  // the taylor expansion is not worth the potential of
291  // additional error buildup.
292 
293  // Note that we can get significant renormalisation values
294  // when we have a larger delta_t due to a glitch eleswhere in
295  // APM, such as a I2c timeout or a set of EEPROM writes. While
296  // we would like to avoid these if possible, if it does happen
297  // we don't want to compound the error by making DCM less
298  // accurate.
299 
300  const float renorm_val = 1.0f / a.length();
301 
302  // keep the average for reporting
303  _renorm_val_sum += renorm_val;
304  _renorm_val_count++;
305 
306  if (!(renorm_val < 2.0f && renorm_val > 0.5f)) {
307  // this is larger than it should get - log it as a warning
308  if (!(renorm_val < 1.0e6f && renorm_val > 1.0e-6f)) {
309  // we are getting values which are way out of
310  // range, we will reset the matrix and hope we
311  // can recover our attitude using drift
312  // correction before we hit the ground!
313  //Serial.printf("ERROR: DCM renormalisation error. renorm_val=%f\n",
314  // renorm_val);
315  return false;
316  }
317  }
318 
319  result = a * renorm_val;
320  return true;
321 }

323 /*************************************************
324  * Direction Cosine Matrix IMU: Theory
325  * William Premerlani and Paul Bizard
326  *
327  * Numerical errors will gradually reduce the orthogonality conditions expressed by equation 5
328  * to approximations rather than identities. In effect, the axes in the two frames of reference no
329  * longer describe a rigid body. Fortunately, numerical error accumulates very slowly, so it is a
330  * simple matter to stay ahead of it.
331  * We call the process of enforcing the orthogonality conditions: renormalization.
332  */
333 void
334 AP_AHRS_DCM::normalize(void)
335 {
336  const float error = _dcm_matrix.a * _dcm_matrix.b; // eq.18
337 
338  const Vector3f t0 = _dcm_matrix.a - (_dcm_matrix.b * (0.5f * error)); // eq.19
339  const Vector3f t1 = _dcm_matrix.b - (_dcm_matrix.a * (0.5f * error)); // eq.19
340  const Vector3f t2 = t0 % t1; // c= a x b // eq.20
341 
342  if (!renorm(t0, _dcm_matrix.a) ||
343  !renorm(t1, _dcm_matrix.b) ||
344  !renorm(t2, _dcm_matrix.c)) {
345  // Our solution is blowing up and we will force back
346  // to last euler angles
347  _last_failure_ms = AP_HAL::millis();
348  AP_AHRS_DCM::reset(true);
349  }
350 }

353 // produce a yaw error value. The returned value is proportional
354 // to sin() of the current heading error in earth frame
355 float
356 AP_AHRS_DCM::yaw_error_compass(void)
357 {
358  const Vector3f &mag = _compass->get_field();
359  // get the mag vector in the earth frame
360  Vector2f rb = _dcm_matrix.mulXY(mag);
361 
362  if (rb.length() < FLT_EPSILON) {
363  return 0.0f;
364  }
365 
366  rb.normalize();
367  if (rb.is_inf()) {
368  // not a valid vector
369  return 0.0f;
370  }
371 
372  // update vector holding earths magnetic field (if required)
373  if( !is_equal(_last_declination,_compass->get_declination()) ) {
374  _last_declination = _compass->get_declination();
375  _mag_earth.x = cosf(_last_declination);
376  _mag_earth.y = sinf(_last_declination);
377  }
378 
379  // calculate the error term in earth frame
380  // calculate the Z component of the cross product of rb and _mag_earth
381  return rb % _mag_earth;
382 }

476 // yaw drift correction using the compass or GPS
477 // this function prodoces the _omega_yaw_P vector, and also
478 // contributes to the _omega_I.z long term yaw drift estimate
479 void
480 AP_AHRS_DCM::drift_correction_yaw(void)
481 {
482  bool new_value = false;
483  float yaw_error;
484  float yaw_deltat;
485 
486  const AP_GPS &_gps = AP::gps();
487 
488  if (_compass && _compass->is_calibrating()) {
489  // don't do any yaw correction while calibrating
490  return;
491  }
492 
493  if (AP_AHRS_DCM::use_compass()) {
494  /*
495  we are using compass for yaw
496  */
497  if (_compass->last_update_usec() != _compass_last_update) {
498  yaw_deltat = (_compass->last_update_usec() - _compass_last_update) * 1.0e-6f;
499  _compass_last_update = _compass->last_update_usec();
500  // we force an additional compass read()
501  // here. This has the effect of throwing away
502  // the first compass value, which can be bad
503  if (!_flags.have_initial_yaw && _compass->read()) {
504  const float heading = _compass->calculate_heading(_dcm_matrix);
505  _dcm_matrix.from_euler(roll, pitch, heading);
506  _omega_yaw_P.zero();
507  _flags.have_initial_yaw = true;
508  }
509  new_value = true;
510  yaw_error = yaw_error_compass();
511 
512  // also update the _gps_last_update, so if we later
513  // disable the compass due to significant yaw error we
514  // don't suddenly change yaw with a reset
515  _gps_last_update = _gps.last_fix_time_ms();
516  }
517  } else if (_flags.fly_forward && have_gps()) {
518  /*
519  we are using GPS for yaw
520  */
521  if (_gps.last_fix_time_ms() != _gps_last_update &&
522  _gps.ground_speed() >= GPS_SPEED_MIN) {
523  yaw_deltat = (_gps.last_fix_time_ms() - _gps_last_update) * 1.0e-3f;
524  _gps_last_update = _gps.last_fix_time_ms();
525  new_value = true;
526  const float gps_course_rad = ToRad(_gps.ground_course_cd() * 0.01f);
527  const float yaw_error_rad = wrap_PI(gps_course_rad - yaw);
528  yaw_error = sinf(yaw_error_rad);
529 
530  /* reset yaw to match GPS heading under any of the
531  following 3 conditions:
532 
533  1) if we have reached GPS_SPEED_MIN and have never had
534  yaw information before
535 
536  2) if the last time we got yaw information from the GPS
537  is more than 20 seconds ago, which means we may have
538  suffered from considerable gyro drift
539 
540  3) if we are over 3*GPS_SPEED_MIN (which means 9m/s)
541  and our yaw error is over 60 degrees, which means very
542  poor yaw. This can happen on bungee launch when the
543  operator pulls back the plane rapidly enough then on
544  release the GPS heading changes very rapidly
545  */
546  if (!_flags.have_initial_yaw ||
547  yaw_deltat > 20 ||
548  (_gps.ground_speed() >= 3*GPS_SPEED_MIN && fabsf(yaw_error_rad) >= 1.047f)) {
549  // reset DCM matrix based on current yaw
550  _dcm_matrix.from_euler(roll, pitch, gps_course_rad);
551  _omega_yaw_P.zero();
552  _flags.have_initial_yaw = true;
553  yaw_error = 0;
554  }
555  }
556  }
557 
558  if (!new_value) {
559  // we don't have any new yaw information
560  // slowly decay _omega_yaw_P to cope with loss
561  // of our yaw source
562  _omega_yaw_P *= 0.97f;
563  return;
564  }
565 
566  // convert the error vector to body frame
567  const float error_z = _dcm_matrix.c.z * yaw_error;
568 
569  // the spin rate changes the P gain, and disables the
570  // integration at higher rates
571  const float spin_rate = _omega.length();
572 
573  // sanity check _kp_yaw
574  if (_kp_yaw < AP_AHRS_YAW_P_MIN) {
575  _kp_yaw = AP_AHRS_YAW_P_MIN;
576  }
577 
578  // update the proportional control to drag the
579  // yaw back to the right value. We use a gain
580  // that depends on the spin rate. See the fastRotations.pdf
581  // paper from Bill Premerlani
582  // We also adjust the gain depending on the rate of change of horizontal velocity which
583  // is proportional to how observable the heading is from the acceerations and GPS velocity
584  // The accelration derived heading will be more reliable in turns than compass or GPS
585 
586  _omega_yaw_P.z = error_z * _P_gain(spin_rate) * _kp_yaw * _yaw_gain();
587  if (use_fast_gains()) {
588  _omega_yaw_P.z *= 8;
589  }
590 
591  // don't update the drift term if we lost the yaw reference
592  // for more than 2 seconds
593  if (yaw_deltat < 2.0f && spin_rate < ToRad(SPIN_RATE_LIMIT)) {
594  // also add to the I term
595  _omega_I_sum.z += error_z * _ki_yaw * yaw_deltat;
596  }
597 
598  _error_yaw = 0.8f * _error_yaw + 0.2f * fabsf(yaw_error);
599 }

619 // perform drift correction. This function aims to update _omega_P and
620 // _omega_I with our best estimate of the short term and long term
621 // gyro error. The _omega_P value is what pulls our attitude solution
622 // back towards the reference vector quickly. The _omega_I term is an
623 // attempt to learn the long term drift rate of the gyros.
624 //
625 // This drift correction implementation is based on a paper
626 // by Bill Premerlani from here:
627 // http://gentlenav.googlecode.com/files/RollPitchDriftCompensation.pdf
628 void
629 AP_AHRS_DCM::drift_correction(float deltat)
630 {

632  uint32_t last_correction_time;
633 
634  // perform yaw drift correction if we have a new yaw reference
635  // vector

925  // short term errors don't cause a buildup of omega_I
926  // beyond the physical limits of the device

938  _ra_deltat = 0;
939  _ra_sum_start = last_correction_time;

942  _last_velocity = velocity;

946 // update our wind speed estimate
947 void AP_AHRS_DCM::estimate_wind(void)
948 {
949  if (!_flags.wind_estimation) {
950  return;
951  }
952  const Vector3f &velocity = _last_velocity;
953 
954  // this is based on the wind speed estimation code from MatrixPilot by
955  // Bill Premerlani. Adaption for ArduPilot by Jon Challinger
956  // See http://gentlenav.googlecode.com/files/WindEstimation.pdf
957  const Vector3f fuselageDirection = _dcm_matrix.colx();
958  const Vector3f fuselageDirectionDiff = fuselageDirection - _last_fuse;
959  const uint32_t now = AP_HAL::millis();
960 
961  // scrap our data and start over if we're taking too long to get a direction change
962  if (now - _last_wind_time > 10000) {
963  _last_wind_time = now;
964  _last_fuse = fuselageDirection;
965  _last_vel = velocity;
966  return;
967  }
968 
969  float diff_length = fuselageDirectionDiff.length();
970  if (diff_length > 0.2f) {
971  // when turning, use the attitude response to estimate
972  // wind speed
973  float V;
974  const Vector3f velocityDiff = velocity - _last_vel;
975 
976  // estimate airspeed it using equation 6
977  V = velocityDiff.length() / diff_length;
978 
979  const Vector3f fuselageDirectionSum = fuselageDirection + _last_fuse;
980  const Vector3f velocitySum = velocity + _last_vel;
981 
982  _last_fuse = fuselageDirection;
983  _last_vel = velocity;
984 
985  const float theta = atan2f(velocityDiff.y, velocityDiff.x) - atan2f(fuselageDirectionDiff.y, fuselageDirectionDiff.x);
986  const float sintheta = sinf(theta);
987  const float costheta = cosf(theta);
988 
989  Vector3f wind = Vector3f();
990  wind.x = velocitySum.x - V * (costheta * fuselageDirectionSum.x - sintheta * fuselageDirectionSum.y);
991  wind.y = velocitySum.y - V * (sintheta * fuselageDirectionSum.x + costheta * fuselageDirectionSum.y);
992  wind.z = velocitySum.z - V * fuselageDirectionSum.z;
993  wind *= 0.5f;
994 
995  if (wind.length() < _wind.length() + 20) {
996  _wind = _wind * 0.95f + wind * 0.05f;
997  }
998 
999  _last_wind_time = now;
1000  } else if (now - _last_wind_time > 2000 && airspeed_sensor_enabled()) {
1001  // when flying straight use airspeed to get wind estimate if available
1002  const Vector3f airspeed = _dcm_matrix.colx() * _airspeed->get_airspeed();
1003  const Vector3f wind = velocity - (airspeed * get_EAS2TAS());
1004  _wind = _wind * 0.92f + wind * 0.08f;
1005  }
1006 }
