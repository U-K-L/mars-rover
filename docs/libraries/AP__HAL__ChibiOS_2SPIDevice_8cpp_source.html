167 /*
168  low level transfer function
169  */
170 bool SPIDevice::do_transfer(const uint8_t *send, uint8_t *recv, uint32_t len)
171 {
172  bool old_cs_forced = cs_forced;
173 
174  if (!set_chip_select(true)) {
175  return false;
176  }
177 
178  bool ret = true;
179 
180 #if defined(HAL_SPI_USE_POLLED)
181  for (uint16_t i=0; i<len; i++) {
182  uint8_t ret = spiPolledExchange(spi_devices[device_desc.bus].driver, send?send[i]:0);
183  if (recv) {
184  recv[i] = ret;
185  }
186  }
187 #else
188  bus.bouncebuffer_setup(send, len, recv, len);
189  osalSysLock();
190  hal.util->persistent_data.spi_count++;
191  if (send == nullptr) {
192  spiStartReceiveI(spi_devices[device_desc.bus].driver, len, recv);
193  } else if (recv == nullptr) {
194  spiStartSendI(spi_devices[device_desc.bus].driver, len, send);
195  } else {
196  spiStartExchangeI(spi_devices[device_desc.bus].driver, len, send, recv);
197  }
198  // we allow SPI transfers to take a maximum of 20ms plus 32us per
199  // byte. This covers all use cases in ArduPilot. We don't ever
200  // expect this timeout to trigger unless there is a severe MCU
201  // error
202  const uint32_t timeout_us = 20000U + len * 32U;
203  msg_t msg = osalThreadSuspendTimeoutS(&spi_devices[device_desc.bus].driver->thread, TIME_MS2I(timeout_us));
204  osalSysUnlock();
205  if (msg == MSG_TIMEOUT) {
206  ret = false;
207  AP::internalerror().error(AP_InternalError::error_t::spi_fail);
208  spiAbort(spi_devices[device_desc.bus].driver);
209  }
210  bus.bouncebuffer_finish(send, recv, len);
211 #endif
212  set_chip_select(old_cs_forced);
213  return ret;
214 }

217 {

232 uint32_t SPIDevice::derive_freq_flag_bus(uint8_t busid, uint32_t _frequency)
233 {
234  uint32_t spi_clock_freq = SPI1_CLOCK;
235  if (busid > 0 && uint8_t(busid-1) < ARRAY_SIZE(bus_clocks)) {
236  spi_clock_freq = bus_clocks[busid-1] / 2;
237  }
238 
239  // find first divisor that brings us below the desired SPI clock
240  uint32_t i = 0;
241  while (spi_clock_freq > _frequency && i<7) {
242  spi_clock_freq >>= 1;
243  i++;
244  }
245 
246  // assuming the bitrate bits are consecutive in the CR1 register,
247  // we can just multiply by BR_0 to get the right bits for the desired
248  // scaling
249 #if defined(STM32H7)
250  return (i * SPI_CFG1_MBR_0) | SPI_CFG1_DSIZE_VALUE(7); // 8 bit transfers
251 #else
252  return i * SPI_CR1_BR_0;
253 #endif
254 }

262 bool SPIDevice::transfer(const uint8_t *send, uint32_t send_len,
263  uint8_t *recv, uint32_t recv_len)
264 {
265  if (!bus.semaphore.check_owner()) {
266  hal.console->printf("SPI: not owner of 0x%x\n", unsigned(get_bus_id()));
267  return false;
268  }
269  if ((send_len == recv_len && send == recv) || !send || !recv) {
270  // simplest cases, needed for DMA
271  return do_transfer(send, recv, recv_len?recv_len:send_len);
272  }
273  uint8_t buf[send_len+recv_len];
274  if (send_len > 0) {
275  memcpy(buf, send, send_len);
276  }
277  if (recv_len > 0) {
278  memset(&buf[send_len], 0, recv_len);
279  }
280  bool ret = do_transfer(buf, buf, send_len+recv_len);
281  if (ret && recv_len > 0) {
282  memcpy(recv, &buf[send_len], recv_len);
283  }
284  return ret;
285 }

315 /*
316  used to acquire bus and (optionally) assert cs
317 */
318 bool SPIDevice::acquire_bus(bool set, bool skip_cs)
319 {
320  bus.semaphore.assert_owner();
321  if (set && cs_forced) {
322  return true;
323  }
324  if (!set && !cs_forced) {
325  return false;
326  }
327  if (!set && cs_forced) {
328  if(!skip_cs) {
329  spiUnselectI(spi_devices[device_desc.bus].driver); /* Slave Select de-assertion. */
330  }
331  spiReleaseBus(spi_devices[device_desc.bus].driver); /* Ownership release. */
332  cs_forced = false;
333  bus.dma_handle->unlock();
334  } else {
335  bus.dma_handle->lock();
336  spiAcquireBus(spi_devices[device_desc.bus].driver); /* Acquire ownership of the bus. */
337  bus.spicfg.end_cb = nullptr;
338  bus.spicfg.ssport = PAL_PORT(device_desc.pal_line);
339  bus.spicfg.sspad = PAL_PAD(device_desc.pal_line);
340 #if defined(STM32H7)
341  bus.spicfg.cfg1 = freq_flag;
342  bus.spicfg.cfg2 = device_desc.mode;
343  if (bus.spicfg.dummytx == nullptr) {
344  bus.spicfg.dummytx = (uint32_t *)malloc_dma(4);
345  memset(bus.spicfg.dummytx, 0xFF, 4);
346  }
347  if (bus.spicfg.dummyrx == nullptr) {
348  bus.spicfg.dummyrx = (uint32_t *)malloc_dma(4);
349  }
350 #else
351  bus.spicfg.cr1 = (uint16_t)(freq_flag | device_desc.mode);
352  bus.spicfg.cr2 = 0;
353 #endif
354  if (bus.spi_started) {
355  spiStop(spi_devices[device_desc.bus].driver);
356  bus.spi_started = false;
357  }
358  spiStart(spi_devices[device_desc.bus].driver, &bus.spicfg); /* Setup transfer parameters. */
359  bus.spi_started = true;
360  if(!skip_cs) {
361  spiSelectI(spi_devices[device_desc.bus].driver); /* Slave Select assertion. */
362  }
363  cs_forced = true;
364  }
365  return true;
366 }

378 AP_HAL::OwnPtr<AP_HAL::SPIDevice>
379 SPIDeviceManager::get_device(const char *name)
380 {
381  /* Find the bus description in the table */
382  uint8_t i;
383  for (i = 0; i<ARRAY_SIZE(device_table); i++) {
384  if (strcmp(device_table[i].name, name) == 0) {
385  break;
386  }
387  }
388  if (i == ARRAY_SIZE(device_table)) {
389  return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(nullptr);
390  }
391 
392  SPIDesc &desc = device_table[i];
393 
394  // find the bus
395  SPIBus *busp;
396  for (busp = buses; busp; busp = (SPIBus *)busp->next) {
397  if (busp->bus == desc.bus) {
398  break;
399  }
400  }
401  if (busp == nullptr) {
402  // create a new one
403  busp = new SPIBus(desc.bus);
404  if (busp == nullptr) {
405  return nullptr;
406  }
407  busp->next = buses;
408  busp->bus = desc.bus;
409 
410  buses = busp;
411  }
412 
413  return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(new SPIDevice(*busp, desc));
414 }

418 /*
419  test clock frequencies. This measures the actual SPI clock
420  frequencies on all configured SPI buses. Used during board bringup
421  to validate clock configuration
422  */
423 void SPIDevice::test_clock_freq(void)
424 {
425  // delay for USB to come up
426  hal.console->printf("Waiting for USB\n");
427  for (uint8_t i=0; i<3; i++) {
428  hal.scheduler->delay(1000);
429  hal.console->printf("Waiting %u\n", AP_HAL::millis());
430  }
431  hal.console->printf("CLOCKS=\n");
432  for (uint8_t i=0; i<ARRAY_SIZE(bus_clocks); i++) {
433  hal.console->printf("%u:%u ", i+1, bus_clocks[i]);
434  }
435  hal.console->printf("\n");
436 
437  // we will send 1024 bytes without any CS asserted and measure the
438  // time it takes to do the transfer
439  uint16_t len = 1024;
440  uint8_t *buf1 = (uint8_t *)hal.util->malloc_type(len, AP_HAL::Util::MEM_DMA_SAFE);
441  uint8_t *buf2 = (uint8_t *)hal.util->malloc_type(len, AP_HAL::Util::MEM_DMA_SAFE);
442  for (uint8_t i=0; i<ARRAY_SIZE(spi_devices); i++) {
443  SPIConfig spicfg {};
444  const uint32_t target_freq = 2000000UL;
445  // use a clock divisor of 256 for maximum resolution
446 #if defined(STM32H7)
447  spicfg.cfg1 = derive_freq_flag_bus(spi_devices[i].busid, target_freq);
448 #else
449  spicfg.cr1 = derive_freq_flag_bus(spi_devices[i].busid, target_freq);
450 #endif
451  spiAcquireBus(spi_devices[i].driver);
452  spiStart(spi_devices[i].driver, &spicfg);
453  uint32_t t0 = AP_HAL::micros();
454  spiStartExchange(spi_devices[i].driver, len, buf1, buf2);
455  chSysLock();
456  msg_t msg = osalThreadSuspendTimeoutS(&spi_devices[i].driver->thread, TIME_MS2I(100));
457  chSysUnlock();
458  if (msg == MSG_TIMEOUT) {
459  spiAbort(spi_devices[i].driver);
460  hal.console->printf("SPI[%u] FAIL %p %p\n", spi_devices[i].busid, buf1, buf2);
461  spiStop(spi_devices[i].driver);
462  spiReleaseBus(spi_devices[i].driver);
463  continue;
464  }
465  uint32_t t1 = AP_HAL::micros();
466  spiStop(spi_devices[i].driver);
467  spiReleaseBus(spi_devices[i].driver);
468  hal.console->printf("SPI[%u] clock=%u\n", spi_devices[i].busid, unsigned(1000000ULL * len * 8ULL / uint64_t(t1 - t0)));
469  }
470  hal.util->free_type(buf1, len, AP_HAL::Util::MEM_DMA_SAFE);
471  hal.util->free_type(buf2, len, AP_HAL::Util::MEM_DMA_SAFE);
472 }
