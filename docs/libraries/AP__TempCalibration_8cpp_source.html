74 /*
75  calculate the correction given an exponent and a temperature
76 
77  This one parameter correction is deliberately chosen to be very
78  robust for extrapolation. It fits the characteristics of the
79  ICM-20789 barometer nicely.
80  */
81 float AP_TempCalibration::calculate_correction(float temp, float exponent) const
82 {
83  return powf(MAX(temp - Tzero, 0), exponent);
84 }

105 /*
106  calculate the sum of squares range of pressure values we get with
107  the current data. This is the function we try to minimise in the
108  calibration
109  */
110 float AP_TempCalibration::calculate_p_range(float baro_factor) const
111 {
112  float sum = 0;
113  float P0 = learn_values[0] + calculate_correction(learn_temp_start, baro_factor);
114  for (uint16_t i=0; i<learn_i; i++) {
115  if (is_zero(learn_values[i])) {
116  // gap in the data
117  continue;
118  }
119  float temp = learn_temp_start + learn_temp_step*i;
120  float correction = calculate_correction(temp, baro_factor);
121  float P = learn_values[i] + correction;
122  sum += sq(P - P0);
123  }
124  return sum / learn_i;
125 }

127 /*
128  calculate a calibration value
129 
130  This fits a simple single value power function to the baro data to
131  find the calibration exponent.
132  */
133 void AP_TempCalibration::calculate_calibration(void)
134 {
135  float current_err = calculate_p_range(baro_exponent);
136  float test_exponent = baro_exponent + learn_delta;
137  float test_err = calculate_p_range(test_exponent);
138  if (test_err >= current_err) {
139  test_exponent = baro_exponent - learn_delta;
140  test_err = calculate_p_range(test_exponent);
141  }
142  if (test_exponent <= exp_limit_max &&
143  test_exponent >= exp_limit_min &&
144  test_err < current_err) {
145  // move to new value
146  debug("CAL: %.2f\n", test_exponent);
147  if (!is_equal(test_exponent, baro_exponent.get())) {
148  baro_exponent.set_and_save(test_exponent);
149  }
150  temp_min.set_and_save_ifchanged(learn_temp_start);
151  temp_max.set_and_save_ifchanged(learn_temp_start + learn_i*learn_temp_step);
152  }
153 }

155 /*
156  update calibration learning
157  */
158 void AP_TempCalibration::learn_calibration(void)
159 {
160  // just for first baro now
161  const AP_Baro &baro = AP::baro();
162  if (!baro.healthy(0) ||
163  hal.util->get_soft_armed() ||
164  baro.get_temperature(0) < Tzero) {
165  return;
166  }
167 
168  // if we have any movement then we reset learning
169  if (learn_values == nullptr ||
170  !AP::ins().is_still()) {
171  debug("learn reset\n");
172  setup_learning();
173  if (learn_values == nullptr) {
174  return;
175  }
176  }
177  float temp = baro.get_temperature(0);
178  float P = baro.get_pressure(0);
179  uint16_t idx = (temp - learn_temp_start) / learn_temp_step;
180  if (idx >= learn_count) {
181  // could change learn_temp_step here
182  return;
183  }
184  if (is_zero(learn_values[idx])) {
185  learn_values[idx] = P;
186  debug("learning %u %.2f at %.2f\n", idx, learn_values[idx], temp);
187  } else {
188  // filter in new value
189  learn_values[idx] = 0.9 * learn_values[idx] + 0.1 * P;
190  }
191  learn_i = MAX(learn_i, idx);
192 
193  uint32_t now = AP_HAL::millis();
194  if (now - last_learn_ms > 100 &&
195  idx*learn_temp_step > min_learn_temp_range &&
196  temp - learn_temp_start > temp_max - temp_min) {
197  last_learn_ms = now;
198  // run estimation and update parameters
199  calculate_calibration();
200  }
201 }

206 void AP_TempCalibration::apply_calibration(void)
207 {
208  AP_Baro &baro = AP::baro();
209  // just for first baro now
210  if (!baro.healthy(0)) {
211  return;

214  float correction = calculate_correction(temp, baro_exponent);
215  baro.set_pressure_correction(0, correction);
