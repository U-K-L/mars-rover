104 /*
105  Function returns an equivalent elevator deflection in centi-degrees in the range from -4500 to 4500
106  A positive demand is up
107  Inputs are:
108  1) demanded pitch rate in degrees/second
109  2) control gain scaler = scaling_speed / aspeed
110  3) boolean which is true when stabilise mode is active
111  4) minimum FBW airspeed (metres/sec)
112  5) maximum FBW airspeed (metres/sec)
113 */
114 int32_t AP_PitchController::_get_rate_out(float desired_rate, float scaler, bool disable_integrator, float aspeed)
115 {
116  uint32_t tnow = AP_HAL::millis();
117  uint32_t dt = tnow - _last_t;
118 
119  if (_last_t == 0 || dt > 1000) {
120  dt = 0;
121  }
122  _last_t = tnow;
123 
124  float delta_time = (float)dt * 0.001f;
125 
126  // Get body rate vector (radians/sec)
127  float omega_y = _ahrs.get_gyro().y;
128 
129  // Calculate the pitch rate error (deg/sec) and scale
130  float achieved_rate = ToDeg(omega_y);
131  float rate_error = (desired_rate - achieved_rate) * scaler;
132 
133  // Multiply pitch rate error by _ki_rate and integrate
134  // Scaler is applied before integrator so that integrator state relates directly to elevator deflection
135  // This means elevator trim offset doesn't change as the value of scaler changes with airspeed
136  // Don't integrate if in stabilise mode as the integrator will wind up against the pilots inputs
137  if (!disable_integrator && gains.I > 0) {
138  float k_I = gains.I;
139  if (is_zero(gains.FF)) {
140  /*
141  if the user hasn't set a direct FF then assume they are
142  not doing sophisticated tuning. Set a minimum I value of
143  0.15 to ensure that the time constant for trimming in
144  pitch is not too long. We have had a lot of user issues
145  with very small I value leading to very slow pitch
146  trimming, which causes a lot of problems for TECS. A
147  value of 0.15 is still quite small, but a lot better
148  than what many users are running.
149  */
150  k_I = MAX(k_I, 0.15f);
151  }
152  float ki_rate = k_I * gains.tau;
153  //only integrate if gain and time step are positive and airspeed above min value.
154  if (dt > 0 && aspeed > 0.5f*float(aparm.airspeed_min)) {
155  float integrator_delta = rate_error * ki_rate * delta_time * scaler;
156  if (_last_out < -45) {
157  // prevent the integrator from increasing if surface defln demand is above the upper limit
158  integrator_delta = MAX(integrator_delta , 0);
159  } else if (_last_out > 45) {
160  // prevent the integrator from decreasing if surface defln demand is below the lower limit
161  integrator_delta = MIN(integrator_delta , 0);
162  }
163  _pid_info.I += integrator_delta;
164  }
165  } else {
166  _pid_info.I = 0;
167  }
168 
169  // Scale the integration limit
170  float intLimScaled = gains.imax * 0.01f;
171 
172  // Constrain the integrator state
173  _pid_info.I = constrain_float(_pid_info.I, -intLimScaled, intLimScaled);
174 
175  // Calculate equivalent gains so that values for K_P and K_I can be taken across from the old PID law
176  // No conversion is required for K_D
177  float eas2tas = _ahrs.get_EAS2TAS();
178  float kp_ff = MAX((gains.P - gains.I * gains.tau) * gains.tau - gains.D , 0) / eas2tas;
179  float k_ff = gains.FF / eas2tas;
180 
181  // Calculate the demanded control surface deflection
182  // Note the scaler is applied again. We want a 1/speed scaler applied to the feed-forward
183  // path, but want a 1/speed^2 scaler applied to the rate error path.
184  // This is because acceleration scales with speed^2, but rate scales with speed.
185  _pid_info.P = desired_rate * kp_ff * scaler;
186  _pid_info.FF = desired_rate * k_ff * scaler;
187  _pid_info.D = rate_error * gains.D * scaler;
188  _last_out = _pid_info.D + _pid_info.FF + _pid_info.P;
189  _pid_info.desired = desired_rate;
190  _pid_info.actual = achieved_rate;
191 
192  if (autotune.running && aspeed > aparm.airspeed_min) {
193  // let autotune have a go at the values
194  // Note that we don't pass the integrator component so we get
195  // a better idea of how much the base PD controller
196  // contributed
197  autotune.update(desired_rate, achieved_rate, _last_out);
198 
199  // set down rate to rate up when auto-tuning
200  _max_rate_neg.set_and_save_ifchanged(gains.rmax);
201  }
202 
203  _last_out += _pid_info.I;
204 
205  /*
206  when we are past the users defined roll limit for the
207  aircraft our priority should be to bring the aircraft back
208  within the roll limit. Using elevator for pitch control at
209  large roll angles is ineffective, and can be counter
210  productive as it induces earth-frame yaw which can reduce
211  the ability to roll. We linearly reduce elevator input when
212  beyond the configured roll limit, reducing to zero at 90
213  degrees
214  */
215  float roll_wrapped = labs(_ahrs.roll_sensor);
216  if (roll_wrapped > 9000) {
217  roll_wrapped = 18000 - roll_wrapped;
218  }
219  if (roll_wrapped > aparm.roll_limit_cd + 500 && aparm.roll_limit_cd < 8500 &&
220  labs(_ahrs.pitch_sensor) < 7000) {
221  float roll_prop = (roll_wrapped - (aparm.roll_limit_cd+500)) / (float)(9000 - aparm.roll_limit_cd);
222  _last_out *= (1 - roll_prop);
223  }
224 
225  // Convert to centi-degrees and constrain
226  return constrain_float(_last_out * 100, -4500, 4500);
227 }

249 /*
250  get the rate offset in degrees/second needed for pitch in body frame
251  to maintain height in a coordinated turn.
252 
253  Also returns the inverted flag and the estimated airspeed in m/s for
254  use by the rest of the pitch controller
255  */
256 float AP_PitchController::_get_coordination_rate_offset(float &aspeed, bool &inverted) const
257 {
258  float rate_offset;
259  float bank_angle = _ahrs.roll;
260 
261  // limit bank angle between +- 80 deg if right way up
262  if (fabsf(bank_angle) < radians(90)) {
263  bank_angle = constrain_float(bank_angle,-radians(80),radians(80));
264  inverted = false;
265  } else {
266  inverted = true;
267  if (bank_angle > 0.0f) {
268  bank_angle = constrain_float(bank_angle,radians(100),radians(180));
269  } else {
270  bank_angle = constrain_float(bank_angle,-radians(180),-radians(100));
271  }
272  }
273  if (!_ahrs.airspeed_estimate(&aspeed)) {
274  // If no airspeed available use average of min and max
275  aspeed = 0.5f*(float(aparm.airspeed_min) + float(aparm.airspeed_max));
276  }
277  if (abs(_ahrs.pitch_sensor) > 7000) {
278  // don't do turn coordination handling when at very high pitch angles
279  rate_offset = 0;
280  } else {
281  rate_offset = cosf(_ahrs.pitch)*fabsf(ToDeg((GRAVITY_MSS / MAX((aspeed * _ahrs.get_EAS2TAS()) , float(aparm.airspeed_min))) * tanf(bank_angle) * sinf(bank_angle))) * _roll_ff;
282  }
283  if (inverted) {
284  rate_offset = -rate_offset;
285  }
286  return rate_offset;
287 }

290 // A positive demand is up
291 // Inputs are:
292 // 1) demanded pitch angle in centi-degrees
293 // 2) control gain scaler = scaling_speed / aspeed
294 // 3) boolean which is true when stabilise mode is active
295 // 4) minimum FBW airspeed (metres/sec)
296 // 5) maximum FBW airspeed (metres/sec)
297 //
298 int32_t AP_PitchController::get_servo_out(int32_t angle_err, float scaler, bool disable_integrator)
299 {
300  // Calculate offset to pitch rate demand required to maintain pitch angle whilst banking
301  // Calculate ideal turn rate from bank angle and airspeed assuming a level coordinated turn
302  // Pitch rate offset is the component of turn rate about the pitch axis
303  float aspeed;
304  float rate_offset;
305  bool inverted;
306 
307  if (gains.tau < 0.1f) {
308  gains.tau.set(0.1f);
309  }
310 
311  rate_offset = _get_coordination_rate_offset(aspeed, inverted);
312 
313  // Calculate the desired pitch rate (deg/sec) from the angle error
314  float desired_rate = angle_err * 0.01f / gains.tau;
315 
316  // limit the maximum pitch rate demand. Don't apply when inverted
317  // as the rates will be tuned when upright, and it is common that
318  // much higher rates are needed inverted
319  if (!inverted) {
320  if (_max_rate_neg && desired_rate < -_max_rate_neg) {
321  desired_rate = -_max_rate_neg;
322  } else if (gains.rmax && desired_rate > gains.rmax) {
323  desired_rate = gains.rmax;
324  }
325  }
326 
327  if (inverted) {
328  desired_rate = -desired_rate;
329  }
330 
331  // Apply the turn correction offset
332  desired_rate = desired_rate + rate_offset;
333 
334  return _get_rate_out(desired_rate, scaler, disable_integrator, aspeed);
335 }
