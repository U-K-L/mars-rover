265 bool SPIDevice::transfer(const uint8_t *send, uint32_t send_len,
266  uint8_t *recv, uint32_t recv_len)
267 {
268  struct spi_ioc_transfer msgs[2] = { };
269  unsigned nmsgs = 0;
270  int fd = _bus.fd[_desc.subdev];
271 
272  assert(fd >= 0);
273 
274  if (send && send_len != 0) {
275  msgs[nmsgs].tx_buf = (uint64_t) send;
276  msgs[nmsgs].rx_buf = 0;
277  msgs[nmsgs].len = send_len;
278  msgs[nmsgs].speed_hz = _speed;
279  msgs[nmsgs].delay_usecs = 0;
280  msgs[nmsgs].bits_per_word = _desc.bits_per_word;
281  msgs[nmsgs].cs_change = 0;
282  nmsgs++;
283  }
284 
285  if (recv && recv_len != 0) {
286  msgs[nmsgs].tx_buf = 0;
287  msgs[nmsgs].rx_buf = (uint64_t) recv;
288  msgs[nmsgs].len = recv_len;
289  msgs[nmsgs].speed_hz = _speed;
290  msgs[nmsgs].delay_usecs = 0;
291  msgs[nmsgs].bits_per_word = _desc.bits_per_word;
292  msgs[nmsgs].cs_change = 0;
293  nmsgs++;
294  }
295 
296  if (!nmsgs) {
297  return false;
298  }
299 
300 #if DEBUG
301  if (_desc.mode == _bus.last_mode) {
302  /*
303  the mode in the kernel is not tied to the file descriptor,
304  so there is a chance some other process has changed it since
305  we last used the bus. We want to report when this happens so
306  the user has a chance of figuring out when there is
307  conflicted use of the SPI bus. Unfortunately this costs us
308  an extra syscall per transfer.
309  */
310  uint8_t current_mode;
311  if (ioctl(fd, SPI_IOC_RD_MODE, &current_mode) < 0) {
312  hal.console->printf("SPIDevice: error on getting mode fd=%d (%s)\n",
313  fd, strerror(errno));
314  _bus.last_mode = -1;
315  } else if (current_mode != _bus.last_mode) {
316  hal.console->printf("SPIDevice: bus mode conflict fd=%d mode=%u/%u\n",
317  fd, (unsigned)_bus.last_mode, (unsigned)current_mode);
318  _bus.last_mode = -1;
319  }
320  }
321 #endif
322 
323  int r;
324  if (_desc.mode != _bus.last_mode) {
325  r = ioctl(fd, SPI_IOC_WR_MODE, &_desc.mode);
326  if (r < 0) {
327  hal.console->printf("SPIDevice: error on setting mode fd=%d (%s)\n",
328  fd, strerror(errno));
329  return false;
330  }
331  _bus.last_mode = _desc.mode;
332  }
333 
334  _cs_assert();
335  r = ioctl(fd, SPI_IOC_MESSAGE(nmsgs), &msgs);
336  _cs_release();
337 
338  if (r == -1) {
339  hal.console->printf("SPIDevice: error transferring data fd=%d (%s)\n",
340  fd, strerror(errno));
341  return false;
342  }
343 
344  return true;
345 }

347 bool SPIDevice::transfer_fullduplex(const uint8_t *send, uint8_t *recv,
348  uint32_t len)
349 {
350  struct spi_ioc_transfer msgs[1] = { };
351  int fd = _bus.fd[_desc.subdev];
352 
353  assert(fd >= 0);
354 
355  if (!send || !recv || len == 0) {
356  return false;
357  }
358 
359  msgs[0].tx_buf = (uint64_t) send;
360  msgs[0].rx_buf = (uint64_t) recv;
361  msgs[0].len = len;
362  msgs[0].speed_hz = _speed;
363  msgs[0].delay_usecs = 0;
364  msgs[0].bits_per_word = _desc.bits_per_word;
365  msgs[0].cs_change = 0;
366 
367  int r = ioctl(fd, SPI_IOC_WR_MODE, &_desc.mode);
368  if (r < 0) {
369  hal.console->printf("SPIDevice: error on setting mode fd=%d (%s)\n",
370  fd, strerror(errno));
371  return false;
372  }
373 
374  _cs_assert();
375  r = ioctl(fd, SPI_IOC_MESSAGE(1), &msgs);
376  _cs_release();
377 
378  if (r == -1) {
379  hal.console->printf("SPIDevice: error transferring data fd=%d (%s)\n",
380  fd, strerror(errno));
381  return false;
382  }
383 
384  return true;

438 AP_HAL::OwnPtr<AP_HAL::SPIDevice>
439 SPIDeviceManager::get_device(const char *name)
440 {
441  SPIDesc *desc = nullptr;
442 
443  /* Find the bus description in the table */
444  for (uint8_t i = 0; i < _n_device_desc; i++) {
445  if (!strcmp(_device[i].name, name)) {
446  desc = &_device[i];
447  break;
448  }
449  }
450 
451  if (!desc) {
452  return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(nullptr);
453  }
454 
455  /* Find if bus already exists */
456  for (uint8_t i = 0, n = _buses.size(); i < n; i++) {
457  if (_buses[i]->bus == desc->bus) {
458  return _create_device(*_buses[i], *desc);
459  }
460  }
461 
462  /* Bus not found for this device, create a new one */
463  AP_HAL::OwnPtr<SPIBus> b{new SPIBus(desc->bus)};
464  if (!b) {
465  return nullptr;
466  }
467 
468  auto dev = _create_device(*b, *desc);
469  if (!dev) {
470  return nullptr;
471  }
472 
473  _buses.push_back(b.leak());
474 
475  return dev;
476 }
