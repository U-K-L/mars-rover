57 static void dma_rx_end_cb(UARTDriver *uart)
58 {
59  osalSysLockFromISR();
60  uart->usart->CR3 &= ~(USART_CR3_DMAT | USART_CR3_DMAR);
61 
62  (void)uart->usart->SR;
63  (void)uart->usart->DR;
64  (void)uart->usart->DR;
65  dmaStreamDisable(uart->dmarx);
66  dmaStreamDisable(uart->dmatx);
67 
68  iomcu.process_io_packet();
69  stats.num_total_rx++;
70  stats.num_dma_complete_rx = stats.num_total_rx - stats.num_idle_rx;
71 
72  dmaStreamSetMemory0(uart->dmarx, &iomcu.rx_io_packet);
73  dmaStreamSetTransactionSize(uart->dmarx, sizeof(iomcu.rx_io_packet));
74  dmaStreamSetMode(uart->dmarx, uart->dmamode | STM32_DMA_CR_DIR_P2M |
75  STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE);
76  dmaStreamEnable(uart->dmarx);
77  uart->usart->CR3 |= USART_CR3_DMAR;
78 
79  dmaStreamSetMemory0(uart->dmatx, &iomcu.tx_io_packet);
80  dmaStreamSetTransactionSize(uart->dmatx, iomcu.tx_io_packet.get_size());
81  dmaStreamSetMode(uart->dmatx, uart->dmamode | STM32_DMA_CR_DIR_M2P |
82  STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE);
83  dmaStreamEnable(uart->dmatx);
84  uart->usart->CR3 |= USART_CR3_DMAT;
85  osalSysUnlockFromISR();
86 }

88 static void idle_rx_handler(UARTDriver *uart)
89 {
90  volatile uint16_t sr = uart->usart->SR;
91 
92  if (sr & (USART_SR_LBD | USART_SR_ORE | /* overrun error - packet was too big for DMA or DMA was too slow */
93  USART_SR_NE | /* noise error - we have lost a byte due to noise */
94  USART_SR_FE |
95  USART_SR_PE)) { /* framing error - start/stop bit lost or line break */
96  /* send a line break - this will abort transmission/reception on the other end */
97  osalSysLockFromISR();
98  uart->usart->SR = ~USART_SR_LBD;
99  uart->usart->CR1 |= USART_CR1_SBK;
100  stats.num_rx_error++;
101  uart->usart->CR3 &= ~(USART_CR3_DMAT | USART_CR3_DMAR);
102  (void)uart->usart->SR;
103  (void)uart->usart->DR;
104  (void)uart->usart->DR;
105  dmaStreamDisable(uart->dmarx);
106  dmaStreamDisable(uart->dmatx);
107 
108  dmaStreamSetMemory0(uart->dmarx, &iomcu.rx_io_packet);
109  dmaStreamSetTransactionSize(uart->dmarx, sizeof(iomcu.rx_io_packet));
110  dmaStreamSetMode(uart->dmarx, uart->dmamode | STM32_DMA_CR_DIR_P2M |
111  STM32_DMA_CR_MINC | STM32_DMA_CR_TCIE);
112  dmaStreamEnable(uart->dmarx);
113  uart->usart->CR3 |= USART_CR3_DMAR;
114  osalSysUnlockFromISR();
115  return;
116  }
117 
118  if (sr & USART_SR_IDLE) {
119  dma_rx_end_cb(uart);
120  stats.num_idle_rx++;
121  }
122 }

161 void AP_IOMCU_FW::init()
162 {
163  // the first protocol version must be 4 to allow downgrade to
164  // old NuttX based firmwares
165  config.protocol_version = IOMCU_PROTOCOL_VERSION;
166  config.protocol_version2 = IOMCU_PROTOCOL_VERSION2;
167 
168  thread_ctx = chThdGetSelfX();
169 
170  if (palReadLine(HAL_GPIO_PIN_IO_HW_DETECT1) == 1 && palReadLine(HAL_GPIO_PIN_IO_HW_DETECT2) == 0) {
171  has_heater = true;
172  }
173 
174  //Set Heater PWM Polarity, 0 for Active Low and 1 for Active High
175  heater_pwm_polarity = !palReadLine(HAL_GPIO_PIN_HEATER);
176  //Set Heater pin mode
177  if (heater_pwm_polarity) {
178  palSetLineMode(HAL_GPIO_PIN_HEATER, PAL_MODE_OUTPUT_PUSHPULL);
179  } else {
180  palSetLineMode(HAL_GPIO_PIN_HEATER, PAL_MODE_OUTPUT_OPENDRAIN);
181  }
182 
183  adc_init();
184  rcin_serial_init();
185 
186  // power on spektrum port
187  palSetLineMode(HAL_GPIO_PIN_SPEKTRUM_PWR_EN, PAL_MODE_OUTPUT_PUSHPULL);
188  SPEKTRUM_POWER(1);
189 
190  // we do no allocations after setup completes
191  reg_status.freemem = hal.util->available_memory();
192 
193  if (hal.util->was_watchdog_safety_off()) {
194  hal.rcout->force_safety_off();
195  reg_status.flag_safety_off = true;
196  }

200 void AP_IOMCU_FW::update()
201 {
202  // we are not running any other threads, so we can use an
203  // immediate timeout here for lowest latency
204  eventmask_t mask = chEvtWaitAnyTimeout(~0, TIME_IMMEDIATE);
205 
206  // we get the timestamp once here, and avoid fetching it
207  // within the DMA callbacks
208  last_ms = AP_HAL::millis();
209  loop_counter++;
210 
211  if (do_reboot && (last_ms > reboot_time)) {
212  hal.scheduler->reboot(true);
213  while (true) {}
214  }
215 
216  if ((mask & EVENT_MASK(IOEVENT_PWM)) ||
217  (last_safety_off != reg_status.flag_safety_off)) {
218  last_safety_off = reg_status.flag_safety_off;
219  pwm_out_update();
220  }
221 
222  uint32_t now = last_ms;
223  reg_status.timestamp_ms = last_ms;
224 
225  // output SBUS if enabled
226  if ((reg_setup.features & P_SETUP_FEATURES_SBUS1_OUT) &&
227  reg_status.flag_safety_off &&
228  now - sbus_last_ms >= sbus_interval_ms) {
229  // output a new SBUS frame
230  sbus_last_ms = now;
231  sbus_out_write(reg_servo.pwm, IOMCU_MAX_CHANNELS);
232  }
233 
234  // handle FMU failsafe
235  if (now - fmu_data_received_time > 200) {
236  // we are not getting input from the FMU. Fill in failsafe values at 100Hz
237  if (now - last_failsafe_ms > 10) {
238  fill_failsafe_pwm();
239  chEvtSignal(thread_ctx, EVENT_MASK(IOEVENT_PWM));
240  last_failsafe_ms = now;
241  }
242  // turn amber on
243  AMBER_SET(1);
244  } else {
245  last_failsafe_ms = now;
246  // turn amber off
247  AMBER_SET(0);
248  }
249 
250  // update status page at 20Hz
251  if (now - last_status_ms > 50) {
252  last_status_ms = now;
253  page_status_update();
254  }
255 
256  // run remaining functions at 1kHz
257  if (now != last_loop_ms) {
258  last_loop_ms = now;
259  heater_update();
260  rcin_update();
261  safety_update();
262  rcout_mode_update();
263  rcin_serial_update();
264  hal.rcout->timer_tick();
265  if (dsm_bind_state) {
266  dsm_bind_step();
267  }
268  }
269 }

271 void AP_IOMCU_FW::pwm_out_update()
272 {
273  memcpy(reg_servo.pwm, reg_direct_pwm.pwm, sizeof(reg_direct_pwm));
274  hal.rcout->cork();
275  for (uint8_t i = 0; i < SERVO_COUNT; i++) {
276  if (reg_status.flag_safety_off || (reg_setup.ignore_safety & (1U<<i))) {
277  hal.rcout->write(i, reg_servo.pwm[i]);
278  } else {
279  hal.rcout->write(i, 0);
280  }
281  }
282  hal.rcout->push();
283 }

285 void AP_IOMCU_FW::heater_update()
286 {
287  uint32_t now = last_ms;
288  if (!has_heater) {
289  // use blue LED as heartbeat, run it 4x faster when override active
290  if (now - last_blue_led_ms > (override_active?125:500)) {
291  BLUE_TOGGLE();
292  last_blue_led_ms = now;
293  }
294  } else if (reg_setup.heater_duty_cycle == 0 || (now - last_heater_ms > 3000UL)) {
295  // turn off the heater
296  HEATER_SET(!heater_pwm_polarity);
297  } else {
298  uint8_t cycle = ((now / 10UL) % 100U);
299  //Turn off heater when cycle is greater than specified duty cycle
300  HEATER_SET((cycle >= reg_setup.heater_duty_cycle) ? !heater_pwm_polarity : heater_pwm_polarity);
301  }
302 }

304 void AP_IOMCU_FW::rcin_update()
305 {
306  ((ChibiOS::RCInput *)hal.rcin)->_timer_tick();
307  if (hal.rcin->new_input()) {
308  rc_input.count = hal.rcin->num_channels();
309  rc_input.flags_rc_ok = true;
310  for (uint8_t i = 0; i < IOMCU_MAX_CHANNELS; i++) {
311  rc_input.pwm[i] = hal.rcin->read(i);
312  }
313  rc_input.last_input_ms = last_ms;
314  rc_input.data = (uint16_t)rcprotocol->protocol_detected();
315  } else if (last_ms - rc_input.last_input_ms > 200U) {
316  rc_input.flags_rc_ok = false;
317  }
318  if (update_rcout_freq) {
319  hal.rcout->set_freq(reg_setup.pwm_rates, reg_setup.pwm_altrate);
320  update_rcout_freq = false;
321  }
322  if (update_default_rate) {
323  hal.rcout->set_default_rate(reg_setup.pwm_defaultrate);
324  }
325 
326  bool old_override = override_active;
327 
328  // check for active override channel
329  if (mixing.enabled &&
330  mixing.rc_chan_override > 0 &&
331  rc_input.flags_rc_ok &&
332  mixing.rc_chan_override <= IOMCU_MAX_CHANNELS) {
333  override_active = (rc_input.pwm[mixing.rc_chan_override-1] >= 1750);
334  } else {
335  override_active = false;
336  }
337  if (old_override != override_active) {
338  if (override_active) {
339  fill_failsafe_pwm();
340  }
341  chEvtSignal(thread_ctx, EVENT_MASK(IOEVENT_PWM));
342  }

345 void AP_IOMCU_FW::process_io_packet()
346 {
347  uint8_t rx_crc = rx_io_packet.crc;
348  uint8_t calc_crc;
349  rx_io_packet.crc = 0;
350  uint8_t pkt_size = rx_io_packet.get_size();
351  if (rx_io_packet.code == CODE_READ) {
352  // allow for more bandwidth efficient read packets
353  calc_crc = crc_crc8((const uint8_t *)&rx_io_packet, 4);
354  if (calc_crc != rx_crc) {
355  calc_crc = crc_crc8((const uint8_t *)&rx_io_packet, pkt_size);
356  }
357  } else {
358  calc_crc = crc_crc8((const uint8_t *)&rx_io_packet, pkt_size);
359  }
360  if (rx_crc != calc_crc || rx_io_packet.count > PKT_MAX_REGS) {
361  tx_io_packet.count = 0;
362  tx_io_packet.code = CODE_CORRUPT;
363  tx_io_packet.crc = 0;
364  tx_io_packet.page = 0;
365  tx_io_packet.offset = 0;
366  tx_io_packet.crc = crc_crc8((const uint8_t *)&tx_io_packet, tx_io_packet.get_size());
367  stats.num_bad_crc++;
368  return;
369  }
370  switch (rx_io_packet.code) {
371  case CODE_READ: {
372  stats.num_code_read++;
373  if (!handle_code_read()) {
374  tx_io_packet.count = 0;
375  tx_io_packet.code = CODE_ERROR;
376  tx_io_packet.crc = 0;
377  tx_io_packet.page = 0;
378  tx_io_packet.offset = 0;
379  tx_io_packet.crc = crc_crc8((const uint8_t *)&tx_io_packet, tx_io_packet.get_size());
380  }
381  }
382  break;
383  case CODE_WRITE: {
384  stats.num_write_pkt++;
385  if (!handle_code_write()) {
386  tx_io_packet.count = 0;
387  tx_io_packet.code = CODE_ERROR;
388  tx_io_packet.crc = 0;
389  tx_io_packet.page = 0;
390  tx_io_packet.offset = 0;
391  tx_io_packet.crc = crc_crc8((const uint8_t *)&tx_io_packet, tx_io_packet.get_size());
392  }
393  }
394  break;
395  default: {
396  stats.num_unknown_pkt++;
397  }
398  break;
399  }
400 }

416 bool AP_IOMCU_FW::handle_code_read()
417 {
418  uint16_t *values = nullptr;
419 #define COPY_PAGE(_page_name) \
420  do { \
421  values = (uint16_t *)&_page_name; \
422  tx_io_packet.count = sizeof(_page_name) / sizeof(uint16_t); \
423  } while(0);
424 
425  switch (rx_io_packet.page) {
426  case PAGE_CONFIG:
427  COPY_PAGE(config);
428  break;
429  case PAGE_SETUP:
430  COPY_PAGE(reg_setup);
431  break;
432  case PAGE_RAW_RCIN:
433  COPY_PAGE(rc_input);
434  break;
435  case PAGE_STATUS:
436  COPY_PAGE(reg_status);
437  break;
438  case PAGE_SERVOS:
439  COPY_PAGE(reg_servo);
440  break;
441  default:
442  return false;
443  }
444 
445  /* if the offset is at or beyond the end of the page, we have no data */
446  if (rx_io_packet.offset + rx_io_packet.count > tx_io_packet.count) {
447  return false;
448  }
449 
450  /* correct the data pointer and count for the offset */
451  values += rx_io_packet.offset;
452  tx_io_packet.page = rx_io_packet.page;
453  tx_io_packet.offset = rx_io_packet.offset;
454  tx_io_packet.count -= rx_io_packet.offset;
455  tx_io_packet.count = MIN(tx_io_packet.count, rx_io_packet.count);
456  tx_io_packet.count = MIN(tx_io_packet.count, PKT_MAX_REGS);
457  tx_io_packet.code = CODE_SUCCESS;
458  memcpy(tx_io_packet.regs, values, sizeof(uint16_t)*tx_io_packet.count);
459  tx_io_packet.crc = 0;
460  tx_io_packet.crc = crc_crc8((const uint8_t *)&tx_io_packet, tx_io_packet.get_size());
461  return true;
462 }

464 bool AP_IOMCU_FW::handle_code_write()
465 {
466  switch (rx_io_packet.page) {
467  case PAGE_SETUP:
468  switch (rx_io_packet.offset) {
469  case PAGE_REG_SETUP_ARMING:
470  reg_setup.arming = rx_io_packet.regs[0];
471  break;
472  case PAGE_REG_SETUP_FORCE_SAFETY_OFF:
473  if (rx_io_packet.regs[0] == FORCE_SAFETY_MAGIC) {
474  hal.rcout->force_safety_off();
475  reg_status.flag_safety_off = true;
476  } else {
477  return false;
478  }
479  break;
480  case PAGE_REG_SETUP_FORCE_SAFETY_ON:
481  if (rx_io_packet.regs[0] == FORCE_SAFETY_MAGIC) {
482  hal.rcout->force_safety_on();
483  reg_status.flag_safety_off = false;
484  } else {
485  return false;
486  }
487  break;
488  case PAGE_REG_SETUP_ALTRATE:
489  reg_setup.pwm_altrate = rx_io_packet.regs[0];
490  update_rcout_freq = true;
491  break;
492  case PAGE_REG_SETUP_PWM_RATE_MASK:
493  reg_setup.pwm_rates = rx_io_packet.regs[0];
494  update_rcout_freq = true;
495  break;
496  case PAGE_REG_SETUP_DEFAULTRATE:
497  if (rx_io_packet.regs[0] < 25 && reg_setup.pwm_altclock == 1) {
498  rx_io_packet.regs[0] = 25;
499  }
500 
501  if (rx_io_packet.regs[0] > 400 && reg_setup.pwm_altclock == 1) {
502  rx_io_packet.regs[0] = 400;
503  }
504  reg_setup.pwm_defaultrate = rx_io_packet.regs[0];
505  update_default_rate = true;
506  break;
507  case PAGE_REG_SETUP_SBUS_RATE:
508  reg_setup.sbus_rate = rx_io_packet.regs[0];
509  sbus_interval_ms = MAX(1000U / reg_setup.sbus_rate,3);
510  break;
511  case PAGE_REG_SETUP_FEATURES:
512  reg_setup.features = rx_io_packet.regs[0];
513  /* disable the conflicting options with SBUS 1 */
514  if (reg_setup.features & (P_SETUP_FEATURES_SBUS1_OUT)) {
515  reg_setup.features &= ~(P_SETUP_FEATURES_PWM_RSSI |
516  P_SETUP_FEATURES_ADC_RSSI |
517  P_SETUP_FEATURES_SBUS2_OUT);
518 
519  // enable SBUS output at specified rate
520  sbus_interval_ms = MAX(1000U / reg_setup.sbus_rate,3);
521 
522  // we need to release the JTAG reset pin to be used as a GPIO, otherwise we can't enable
523  // or disable SBUS out
524  AFIO->MAPR = AFIO_MAPR_SWJ_CFG_NOJNTRST;
525 
526  palClearLine(HAL_GPIO_PIN_SBUS_OUT_EN);
527  } else {
528  palSetLine(HAL_GPIO_PIN_SBUS_OUT_EN);
529  }
530  break;
531 
532  case PAGE_REG_SETUP_HEATER_DUTY_CYCLE:
533  reg_setup.heater_duty_cycle = rx_io_packet.regs[0];
534  last_heater_ms = last_ms;
535  break;
536 
537  case PAGE_REG_SETUP_REBOOT_BL:
538  if (reg_status.flag_safety_off) {
539  // don't allow reboot while armed
540  return false;
541  }
542 
543  // check the magic value
544  if (rx_io_packet.regs[0] != REBOOT_BL_MAGIC) {
545  return false;
546  }
547  schedule_reboot(100);
548  break;
549 
550  case PAGE_REG_SETUP_IGNORE_SAFETY:
551  reg_setup.ignore_safety = rx_io_packet.regs[0];
552  ((ChibiOS::RCOutput *)hal.rcout)->set_safety_mask(reg_setup.ignore_safety);
553  break;
554 
555  case PAGE_REG_SETUP_DSM_BIND:
556  if (dsm_bind_state == 0) {
557  dsm_bind_state = 1;
558  }
559  break;
560 
561  default:
562  break;
563  }
564  break;
565 
566  case PAGE_DIRECT_PWM: {
567  if (override_active) {
568  // no input when override is active
569  break;
570  }
571  /* copy channel data */
572  uint16_t i = 0, offset = rx_io_packet.offset, num_values = rx_io_packet.count;
573  if (offset + num_values > sizeof(reg_direct_pwm.pwm)/2) {
574  return false;
575  }
576  while ((offset < IOMCU_MAX_CHANNELS) && (num_values > 0)) {
577  /* XXX range-check value? */
578  if (rx_io_packet.regs[i] != PWM_IGNORE_THIS_CHANNEL) {
579  reg_direct_pwm.pwm[offset] = rx_io_packet.regs[i];
580  }
581 
582  offset++;
583  num_values--;
584  i++;
585  }
586  fmu_data_received_time = last_ms;
587  reg_status.flag_fmu_ok = true;
588  reg_status.flag_raw_pwm = true;
589  chEvtSignalI(thread_ctx, EVENT_MASK(IOEVENT_PWM));
590  break;
591  }
592 
593  case PAGE_MIXING: {
594  uint16_t offset = rx_io_packet.offset, num_values = rx_io_packet.count;
595  if (offset + num_values > sizeof(mixing)/2) {
596  return false;
597  }
598  memcpy(((uint16_t *)&mixing)+offset, &rx_io_packet.regs[0], num_values*2);
599  break;
600  }
601 
602  case PAGE_SAFETY_PWM: {
603  uint16_t offset = rx_io_packet.offset, num_values = rx_io_packet.count;
604  if (offset + num_values > sizeof(reg_safety_pwm.pwm)/2) {
605  return false;
606  }
607  memcpy((&reg_safety_pwm.pwm[0])+offset, &rx_io_packet.regs[0], num_values*2);
608  break;
609  }
610 
611  case PAGE_FAILSAFE_PWM: {
612  uint16_t offset = rx_io_packet.offset, num_values = rx_io_packet.count;
613  if (offset + num_values > sizeof(reg_failsafe_pwm.pwm)/2) {
614  return false;
615  }
616  memcpy((&reg_failsafe_pwm.pwm[0])+offset, &rx_io_packet.regs[0], num_values*2);
617  break;
618  }
619 
620  default:
621  break;
622  }
623  tx_io_packet.count = 0;
624  tx_io_packet.code = CODE_SUCCESS;
625  tx_io_packet.crc = 0;
626  tx_io_packet.page = 0;
627  tx_io_packet.offset = 0;
628  tx_io_packet.crc = crc_crc8((const uint8_t *)&tx_io_packet, tx_io_packet.get_size());
629  return true;
630 }

638 void AP_IOMCU_FW::calculate_fw_crc(void)
639 {
640 #define APP_SIZE_MAX 0xf000
641 #define APP_LOAD_ADDRESS 0x08001000
642  // compute CRC of the current firmware
643  uint32_t sum = 0;
644 
645  for (unsigned p = 0; p < APP_SIZE_MAX; p += 4) {
646  uint32_t bytes = *(uint32_t *)(p + APP_LOAD_ADDRESS);
647  sum = crc_crc32(sum, (const uint8_t *)&bytes, sizeof(bytes));
648  }
649 
650  reg_setup.crc[0] = sum & 0xFFFF;
651  reg_setup.crc[1] = sum >> 16;
652 }

655 /*
656  update safety state
657  */
658 void AP_IOMCU_FW::safety_update(void)
659 {
660  uint32_t now = last_ms;
661  if (now - safety_update_ms < 100) {
662  // update safety at 10Hz
663  return;
664  }
665  safety_update_ms = now;
666 
667  bool safety_pressed = palReadLine(HAL_GPIO_PIN_SAFETY_INPUT);
668  if (safety_pressed) {
669  if (reg_status.flag_safety_off && (reg_setup.arming & P_SETUP_ARMING_SAFETY_DISABLE_ON)) {
670  safety_pressed = false;
671  } else if ((!reg_status.flag_safety_off) && (reg_setup.arming & P_SETUP_ARMING_SAFETY_DISABLE_OFF)) {
672  safety_pressed = false;
673  }
674  }
675  if (safety_pressed) {
676  safety_button_counter++;
677  } else {
678  safety_button_counter = 0;
679  }
680  if (safety_button_counter == 10) {
681  // safety has been pressed for 1 second, change state
682  reg_status.flag_safety_off = !reg_status.flag_safety_off;
683  if (reg_status.flag_safety_off) {
684  hal.rcout->force_safety_off();
685  } else {
686  hal.rcout->force_safety_on();
687  }
688  }
689 
690 #if IOMCU_ENABLE_WATCHDOG_TEST
691  if (safety_button_counter == 50) {
692  // deliberate lockup of IOMCU on 5s button press, for testing
693  // watchdog
694  while (true) {
695  hal.scheduler->delay(50);
696  palToggleLine(HAL_GPIO_PIN_SAFETY_LED);
697  if (palReadLine(HAL_GPIO_PIN_SAFETY_INPUT)) {
698  // only trigger watchdog on button release, so we
699  // don't end up stuck in the bootloader
700  stm32_watchdog_pat();
701  }
702  }
703  }
704 #endif
705 
706  led_counter = (led_counter+1) % 16;
707  const uint16_t led_pattern = reg_status.flag_safety_off?0xFFFF:0x5500;
708  palWriteLine(HAL_GPIO_PIN_SAFETY_LED, (led_pattern & (1U << led_counter))?0:1);
709 }
