224 /*
225  detect if a VL53L0X rangefinder is connected. We'll detect by
226  trying to take a reading on I2C. If we get a result the sensor is
227  there.
228 */
229 AP_RangeFinder_Backend *AP_RangeFinder_VL53L0X::detect(RangeFinder::RangeFinder_State &_state, AP_RangeFinder_Params &_params, AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev)
230 {
231  if(!dev){
232  return nullptr;
233  }
234  AP_RangeFinder_VL53L0X *sensor
235  = new AP_RangeFinder_VL53L0X(_state, _params, std::move(dev));
236 
237  if (!sensor) {
238  delete sensor;
239  return nullptr;
240  }
241 
242  sensor->dev->get_semaphore()->take_blocking();
243 
244  if (!sensor->check_id() || !sensor->init()) {
245  sensor->dev->get_semaphore()->give();
246  delete sensor;
247  return nullptr;
248  }
249 
250  sensor->dev->get_semaphore()->give();
251 
252  return sensor;
253 }

269 // Get reference SPAD (single photon avalanche diode) count and type
270 // based on VL53L0X_get_info_from_device(),
271 // but only gets reference SPAD count and type
272 bool AP_RangeFinder_VL53L0X::get_SPAD_info(uint8_t * count, bool *type_is_aperture)
273 {
274  uint8_t tmp;
275 
276  write_register(0x80, 0x01);
277  write_register(0xFF, 0x01);
278  write_register(0x00, 0x00);
279 
280  write_register(0xFF, 0x06);
281  write_register(0x83, read_register(0x83) | 0x04);
282  write_register(0xFF, 0x07);
283  write_register(0x81, 0x01);
284 
285  write_register(0x80, 0x01);
286 
287  write_register(0x94, 0x6b);
288  write_register(0x83, 0x00);
289 
290  uint8_t tries = 50;
291  while (read_register(0x83) == 0x00) {
292  tries--;
293  if (tries == 0) {
294  return false;
295  }
296  hal.scheduler->delay(1);
297  }
298  write_register(0x83, 0x01);
299  tmp = read_register(0x92);
300 
301  *count = tmp & 0x7f;
302  *type_is_aperture = (tmp >> 7) & 0x01;
303 
304  write_register(0x81, 0x00);
305  write_register(0xFF, 0x06);
306  write_register(0x83, read_register(0x83) & ~0x04);
307  write_register(0xFF, 0x01);
308  write_register(0x00, 0x01);
309 
310  write_register(0xFF, 0x00);
311  write_register(0x80, 0x00);
312 
313  return true;
314 }

329 // Get the VCSEL pulse period in PCLKs for the given period type.
330 // based on VL53L0X_get_vcsel_pulse_period()
331 uint8_t AP_RangeFinder_VL53L0X::getVcselPulsePeriod(vcselPeriodType _type)
332 {
333 #define decodeVcselPeriod(reg_val) (((reg_val) + 1) << 1)
334  if (_type == VcselPeriodPreRange) {
335  return decodeVcselPeriod(read_register(PRE_RANGE_CONFIG_VCSEL_PERIOD));

342 // Convert sequence step timeout from MCLKs to microseconds with given VCSEL period in PCLKs
343 // based on VL53L0X_calc_timeout_us()
344 uint32_t AP_RangeFinder_VL53L0X::timeoutMclksToMicroseconds(uint16_t timeout_period_mclks, uint8_t vcsel_period_pclks)
345 {
346 #define calcMacroPeriod(vcsel_period_pclks) ((((uint32_t)2304 * (vcsel_period_pclks) * 1655) + 500) / 1000)
347  uint32_t macro_period_ns = calcMacroPeriod(vcsel_period_pclks);
348 
349  return ((timeout_period_mclks * macro_period_ns) + (macro_period_ns / 2)) / 1000;
350 }

352 // Decode sequence step timeout in MCLKs from register value
353 // based on VL53L0X_decode_timeout()
354 // Note: the original function returned a uint32_t, but the return value is
355 // always stored in a uint16_t.
356 uint16_t AP_RangeFinder_VL53L0X::decodeTimeout(uint16_t reg_val)
357 {
358  // format: "(LSByte * 2^MSByte) + 1"
359  return (uint16_t)((reg_val & 0x00FF) <<
360  (uint16_t)((reg_val & 0xFF00) >> 8)) + 1;
361 }

363 // Get sequence step timeouts
364 // based on get_sequence_step_timeout(),
365 // but gets all timeouts instead of just the requested one, and also stores
366 // intermediate values
367 void AP_RangeFinder_VL53L0X::getSequenceStepTimeouts(SequenceStepEnables const * enables, SequenceStepTimeouts * timeouts)
368 {
369  timeouts->pre_range_vcsel_period_pclks = getVcselPulsePeriod(VcselPeriodPreRange);
370 
371  timeouts->msrc_dss_tcc_mclks = read_register(MSRC_CONFIG_TIMEOUT_MACROP) + 1;
372  timeouts->msrc_dss_tcc_us =
373  timeoutMclksToMicroseconds(timeouts->msrc_dss_tcc_mclks,
374  timeouts->pre_range_vcsel_period_pclks);
375 
376  timeouts->pre_range_mclks =
377  decodeTimeout(read_register16(PRE_RANGE_CONFIG_TIMEOUT_MACROP_HI));
378  timeouts->pre_range_us =
379  timeoutMclksToMicroseconds(timeouts->pre_range_mclks,
380  timeouts->pre_range_vcsel_period_pclks);
381 
382  timeouts->final_range_vcsel_period_pclks = getVcselPulsePeriod(VcselPeriodFinalRange);
383 
384  timeouts->final_range_mclks =
385  decodeTimeout(read_register16(FINAL_RANGE_CONFIG_TIMEOUT_MACROP_HI));
386 
387  if (enables->pre_range) {
388  timeouts->final_range_mclks -= timeouts->pre_range_mclks;
389  }
390 
391  timeouts->final_range_us =
392  timeoutMclksToMicroseconds(timeouts->final_range_mclks,
393  timeouts->final_range_vcsel_period_pclks);
394 }

397 // Get the measurement timing budget in microseconds
398 // based on VL53L0X_get_measurement_timing_budget_micro_seconds()
399 // in us
400 uint32_t AP_RangeFinder_VL53L0X::getMeasurementTimingBudget(void)
401 {
402  SequenceStepEnables enables;
403  SequenceStepTimeouts timeouts;
404 
405  uint16_t const StartOverhead = 1910; // note that this is different than the value in set_
406  uint16_t const EndOverhead = 960;
407  uint16_t const MsrcOverhead = 660;
408  uint16_t const TccOverhead = 590;
409  uint16_t const DssOverhead = 690;
410  uint16_t const PreRangeOverhead = 660;
411  uint16_t const FinalRangeOverhead = 550;
412 
413  // "Start and end overhead times always present"
414  uint32_t budget_us = StartOverhead + EndOverhead;
415 
416  getSequenceStepEnables(&enables);
417  getSequenceStepTimeouts(&enables, &timeouts);
418 
419  if (enables.tcc) {
420  budget_us += (timeouts.msrc_dss_tcc_us + TccOverhead);
421  }
422 
423  if (enables.dss) {
424  budget_us += 2 * (timeouts.msrc_dss_tcc_us + DssOverhead);
425  } else if (enables.msrc) {
426  budget_us += (timeouts.msrc_dss_tcc_us + MsrcOverhead);
427  }
428 
429  if (enables.pre_range) {
430  budget_us += (timeouts.pre_range_us + PreRangeOverhead);
431  }
432 
433  if (enables.final_range) {
434  budget_us += (timeouts.final_range_us + FinalRangeOverhead);
435  }
436 
437  measurement_timing_budget_us = budget_us; // store for internal reuse
438  return budget_us;
439 }

450 // Encode sequence step timeout register value from timeout in MCLKs
451 // based on VL53L0X_encode_timeout()
452 // Note: the original function took a uint16_t, but the argument passed to it
453 // is always a uint16_t.
454 uint16_t AP_RangeFinder_VL53L0X::encodeTimeout(uint16_t timeout_mclks)
455 {
456  // format: "(LSByte * 2^MSByte) + 1"
457 
458  uint32_t ls_byte = 0;
459  uint16_t ms_byte = 0;
460 
461  if (timeout_mclks > 0) {
462  ls_byte = timeout_mclks - 1;
463 
464  while ((ls_byte & 0xFFFFFF00) > 0) {
465  ls_byte >>= 1;
466  ms_byte++;
467  }
468 
469  return (ms_byte << 8) | (ls_byte & 0xFF);
470  }
471  return 0;
472 }

474 // Set the measurement timing budget in microseconds, which is the time allowed
475 // for one measurement; the ST API and this library take care of splitting the
476 // timing budget among the sub-steps in the ranging sequence. A longer timing
477 // budget allows for more accurate measurements. Increasing the budget by a
478 // factor of N decreases the range measurement standard deviation by a factor of
479 // sqrt(N). Defaults to about 33 milliseconds; the minimum is 20 ms.
480 // based on VL53L0X_set_measurement_timing_budget_micro_seconds()
481 bool AP_RangeFinder_VL53L0X::setMeasurementTimingBudget(uint32_t budget_us)
482 {
483  SequenceStepEnables enables;
484  SequenceStepTimeouts timeouts;
485 
486  uint16_t const StartOverhead = 1320; // note that this is different than the value in get_
487  uint16_t const EndOverhead = 960;
488  uint16_t const MsrcOverhead = 660;
489  uint16_t const TccOverhead = 590;
490  uint16_t const DssOverhead = 690;
491  uint16_t const PreRangeOverhead = 660;
492  uint16_t const FinalRangeOverhead = 550;
493 
494  uint32_t const MinTimingBudget = 20000;
495 
496  if (budget_us < MinTimingBudget) { return false; }
497 
498  uint32_t used_budget_us = StartOverhead + EndOverhead;
499 
500  getSequenceStepEnables(&enables);
501  getSequenceStepTimeouts(&enables, &timeouts);
502 
503  if (enables.tcc) {
504  used_budget_us += (timeouts.msrc_dss_tcc_us + TccOverhead);
505  }
506 
507  if (enables.dss) {
508  used_budget_us += 2 * (timeouts.msrc_dss_tcc_us + DssOverhead);
509  } else if (enables.msrc) {
510  used_budget_us += (timeouts.msrc_dss_tcc_us + MsrcOverhead);
511  }
512 
513  if (enables.pre_range) {
514  used_budget_us += (timeouts.pre_range_us + PreRangeOverhead);
515  }
516 
517  if (enables.final_range) {
518  used_budget_us += FinalRangeOverhead;
519 
520  // "Note that the final range timeout is determined by the timing
521  // budget and the sum of all other timeouts within the sequence.
522  // If there is no room for the final range timeout, then an error
523  // will be set. Otherwise the remaining time will be applied to
524  // the final range."
525 
526  if (used_budget_us > budget_us) {
527  // "Requested timeout too big."
528  return false;
529  }
530 
531  uint32_t final_range_timeout_us = budget_us - used_budget_us;
532 
533  // set_sequence_step_timeout() begin
534  // (SequenceStepId == VL53L0X_SEQUENCESTEP_FINAL_RANGE)
535 
536  // "For the final range timeout, the pre-range timeout
537  // must be added. To do this both final and pre-range
538  // timeouts must be expressed in macro periods MClks
539  // because they have different vcsel periods."
540 
541  uint16_t final_range_timeout_mclks =
542  timeoutMicrosecondsToMclks(final_range_timeout_us,
543  timeouts.final_range_vcsel_period_pclks);
544 
545  if (enables.pre_range) {
546  final_range_timeout_mclks += timeouts.pre_range_mclks;
547  }
548 
549  write_register16(FINAL_RANGE_CONFIG_TIMEOUT_MACROP_HI,
550  encodeTimeout(final_range_timeout_mclks));
551 
552  // set_sequence_step_timeout() end
553  measurement_timing_budget_us = budget_us; // store for internal reuse
554  }
555  return true;
556 }

583  uint8_t spad_count;
584  bool spad_type_is_aperture;
585  if (!get_SPAD_info(&spad_count, &spad_type_is_aperture)) {
586  printf("VL53L0X: Failed to get SPAD info\n");
587  return false;
588  }
589 
590  // The SPAD map (RefGoodSpadMap) is read by VL53L0X_get_info_from_device() in
591  // the API, but the same data seems to be more easily readable from
592  // GLOBAL_CONFIG_SPAD_ENABLES_REF_0 through _6, so read it from there
593  uint8_t ref_spad_map[6];
594  if (!dev->read_registers(GLOBAL_CONFIG_SPAD_ENABLES_REF_0, ref_spad_map, 6)) {
595  printf("VL53L0X: Failed to read SPAD map\n");
596  return false;
597  }
598 
599  // -- VL53L0X_set_reference_spads() begin (assume NVM values are valid)
600  write_register(0xFF, 0x01);
601  write_register(DYNAMIC_SPAD_REF_EN_START_OFFSET, 0x00);
602  write_register(DYNAMIC_SPAD_NUM_REQUESTED_REF_SPAD, 0x2C);
603  write_register(0xFF, 0x00);
604  write_register(GLOBAL_CONFIG_REF_EN_START_SELECT, 0xB4);
605 
606  uint8_t first_spad_to_enable = spad_type_is_aperture ? 12 : 0; // 12 is the first aperture spad
607  uint8_t spads_enabled = 0;
608 
609  for (uint8_t i = 0; i < 48; i++) {
610  if (i < first_spad_to_enable || spads_enabled == spad_count) {
611  // This bit is lower than the first one that should be enabled, or
612  // (reference_spad_count) bits have already been enabled, so zero this bit
613  ref_spad_map[i / 8] &= ~(1 << (i % 8));
614  } else if ((ref_spad_map[i / 8] >> (i % 8)) & 0x1) {
615  spads_enabled++;
616  }
617  }
618 
619  uint8_t reg_spad_map[7] = { GLOBAL_CONFIG_SPAD_ENABLES_REF_0, };
620  memcpy(&reg_spad_map[1], ref_spad_map, 6);
621  dev->transfer(reg_spad_map, 7, nullptr, 0);
622 
623  for (uint16_t i=0; i<ARRAY_SIZE(tuning_data); i++) {
624  write_register(tuning_data[i].reg, tuning_data[i].value);
625  }
626 
627  write_register(SYSTEM_INTERRUPT_CONFIG_GPIO, 0x04);
628  write_register(GPIO_HV_MUX_ACTIVE_HIGH, read_register(GPIO_HV_MUX_ACTIVE_HIGH) & ~0x10); // active low
629  write_register(SYSTEM_INTERRUPT_CLEAR, 0x01);
630 
631  // -- VL53L0X_SetGpioConfig() end
632 
633  measurement_timing_budget_us = getMeasurementTimingBudget();
634 
635  // "Disable MSRC and TCC by default"
636  // MSRC = Minimum Signal Rate Check
637  // TCC = Target CentreCheck
638  // -- VL53L0X_SetSequenceStepEnable() begin
639 
640  write_register(SYSTEM_SEQUENCE_CONFIG, 0xE8);
641 
642  // -- VL53L0X_SetSequenceStepEnable() end
643 
644  // "Recalculate timing budget"
645  setMeasurementTimingBudget(measurement_timing_budget_us);
646 
647  // VL53L0X_StaticInit() end
648 
649  // VL53L0X_PerformRefCalibration() begin (VL53L0X_perform_ref_calibration())
650 
651  // -- VL53L0X_perform_vhv_calibration() begin
652 
653  write_register(SYSTEM_SEQUENCE_CONFIG, 0x01);
654  if (!performSingleRefCalibration(0x40)) {
655  printf("VL53L0X: Failed SingleRefCalibration1\n");
656  return false;
657  }
658 
659  // -- VL53L0X_perform_vhv_calibration() end
660 
661  // -- VL53L0X_perform_phase_calibration() begin
662 
663  write_register(SYSTEM_SEQUENCE_CONFIG, 0x02);
664  if (!performSingleRefCalibration(0x00)) {
665  printf("VL53L0X: Failed SingleRefCalibration2\n");
666  return false;
667  }
668 
669  // -- VL53L0X_perform_phase_calibration() end
670 
671  // "restore the previous Sequence Config"
672  write_register(SYSTEM_SEQUENCE_CONFIG, 0xE8);
673 
674  start_continuous();
675 
676  // call timer() every 33ms. We expect new data to be available every 33ms
677  dev->register_periodic_callback(33000,
678  FUNCTOR_BIND_MEMBER(&AP_RangeFinder_VL53L0X::timer, void));
679  return true;
680 }
