77 void AP_GPS_Backend::make_gps_time(uint32_t bcd_date, uint32_t bcd_milliseconds)
78 {
79  uint8_t year, mon, day, hour, min, sec;
80  uint16_t msec;
81 
82  year = bcd_date % 100;
83  mon = (bcd_date / 100) % 100;
84  day = bcd_date / 10000;
85 
86  uint32_t v = bcd_milliseconds;
87  msec = v % 1000; v /= 1000;
88  sec = v % 100; v /= 100;
89  min = v % 100; v /= 100;
90  hour = v % 100;
91 
92  int8_t rmon = mon - 2;
93  if (0 >= rmon) {
94  rmon += 12;
95  year -= 1;
96  }
97 
98  // get time in seconds since unix epoch
99  uint32_t ret = (year/4) - (GPS_LEAPSECONDS_MILLIS / 1000UL) + 367*rmon/12 + day;
100  ret += year*365 + 10501;
101  ret = ret*24 + hour;
102  ret = ret*60 + min;
103  ret = ret*60 + sec;
104 
105  // convert to time since GPS epoch
106  ret -= 272764785UL;
107 
108  // get GPS week and time
109  state.time_week = ret / AP_SEC_PER_WEEK;
110  state.time_week_ms = (ret % AP_SEC_PER_WEEK) * AP_MSEC_PER_SEC;
111  state.time_week_ms += msec;
112 }

233 void AP_GPS_Backend::check_new_itow(uint32_t itow, uint32_t msg_length)
234 {
235  if (itow != _last_itow) {
236  _last_itow = itow;
237 
238  /*
239  we need to calculate a pseudo-itow, which copes with the
240  iTow from the GPS changing in unexpected ways. We assume
241  that timestamps from the GPS are always in multiples of
242  50ms. That means we can't handle a GPS with an update rate
243  of more than 20Hz. We could do more, but we'd need the GPS
244  poll time to be higher
245  */
246  const uint32_t gps_min_period_ms = 50;
247 
248  // get the time the packet arrived on the UART
249  uint64_t uart_us = port->receive_time_constraint_us(msg_length);
250 
251  uint32_t now = AP_HAL::millis();
252  uint32_t dt_ms = now - _last_ms;
253  _last_ms = now;
254 
255  // round to nearest 50ms period
256  dt_ms = ((dt_ms + (gps_min_period_ms/2)) / gps_min_period_ms) * gps_min_period_ms;
257 
258  // work out an actual message rate. If we get 5 messages in a
259  // row with a new rate we switch rate
260  if (_last_rate_ms == dt_ms) {
261  if (_rate_counter < 5) {
262  _rate_counter++;
263  } else if (_rate_ms != dt_ms) {
264  _rate_ms = dt_ms;
265  }
266  } else {
267  _rate_counter = 0;
268  _last_rate_ms = dt_ms;
269  }
270  if (_rate_ms == 0) {
271  // only allow 5Hz to 20Hz in user config
272  _rate_ms = constrain_int16(gps.get_rate_ms(state.instance), 50, 200);
273  }
274 
275  // round to calculated message rate
276  dt_ms = ((dt_ms + (_rate_ms/2)) / _rate_ms) * _rate_ms;
277 
278  // calculate pseudo-itow
279  _pseudo_itow += dt_ms * 1000U;
280 
281  // use msg arrival time, and correct for jitter
282  uint64_t local_us = jitter_correction.correct_offboard_timestamp_usec(_pseudo_itow, uart_us);
283  state.uart_timestamp_ms = local_us / 1000U;
284  }
