31 void GCS_MAVLINK::handle_serial_control(const mavlink_message_t *msg)
32 {
33  mavlink_serial_control_t packet;
34  mavlink_msg_serial_control_decode(msg, &packet);
35 
36  AP_HAL::UARTDriver *port = nullptr;
37  AP_HAL::BetterStream *stream = nullptr;
38 
39  if (packet.flags & SERIAL_CONTROL_FLAG_REPLY) {
40  // how did this packet get to us?
41  return;
42  }
43 
44  bool exclusive = (packet.flags & SERIAL_CONTROL_FLAG_EXCLUSIVE) != 0;
45 
46  switch (packet.device) {
47  case SERIAL_CONTROL_DEV_TELEM1:
48  stream = port = hal.uartC;
49  lock_channel(MAVLINK_COMM_1, exclusive);
50  break;
51  case SERIAL_CONTROL_DEV_TELEM2:
52  stream = port = hal.uartD;
53  lock_channel(MAVLINK_COMM_2, exclusive);
54  break;
55  case SERIAL_CONTROL_DEV_GPS1:
56  stream = port = hal.uartB;
57  AP::gps().lock_port(0, exclusive);
58  break;
59  case SERIAL_CONTROL_DEV_GPS2:
60  stream = port = hal.uartE;
61  AP::gps().lock_port(1, exclusive);
62  break;
63  case SERIAL_CONTROL_DEV_SHELL:
64  stream = hal.util->get_shell_stream();
65  if (stream == nullptr) {
66  return;
67  }
68  break;
69  default:
70  // not supported yet
71  return;
72  }
73  if (stream == nullptr) {
74  // this is probably very bad
75 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
76  AP_HAL::panic("stream is nullptr");
77 #endif
78  return;
79  }
80 
81  if (exclusive && port != nullptr) {
82  // force flow control off for exclusive access. This protocol
83  // is used to talk to bootloaders which may not have flow
84  // control support
85  port->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
86  }
87 
88  // optionally change the baudrate
89  if (packet.baudrate != 0 && port != nullptr) {
90  port->begin(packet.baudrate);
91  }
92 
93  // write the data
94  if (packet.count != 0) {
95  if ((packet.flags & SERIAL_CONTROL_FLAG_BLOCKING) == 0) {
96  stream->write(packet.data, packet.count);
97  } else {
98  const uint8_t *data = &packet.data[0];
99  uint8_t count = packet.count;
100  while (count > 0) {
101  while (stream->txspace() <= 0) {
102  hal.scheduler->delay(5);
103  }
104  uint16_t n = stream->txspace();
105  if (n > packet.count) {
106  n = packet.count;
107  }
108  stream->write(data, n);
109  data += n;
110  count -= n;
111  }
112  }
113  }
114 
115  if ((packet.flags & SERIAL_CONTROL_FLAG_RESPOND) == 0) {
116  // no response expected
117  return;
118  }
119 
120  uint8_t flags = packet.flags;
121 
122 more_data:
123  // sleep for the timeout
124  while (packet.timeout != 0 &&
125  stream->available() < (int16_t)sizeof(packet.data)) {
126  hal.scheduler->delay(1);
127  packet.timeout--;
128  }
129 
130  packet.flags = SERIAL_CONTROL_FLAG_REPLY;
131 
132  // work out how many bytes are available
133  int16_t available = stream->available();
134  if (available < 0) {
135  available = 0;
136  }
137  if (available > (int16_t)sizeof(packet.data)) {
138  available = sizeof(packet.data);
139  }
140  if (available == 0 && (flags & SERIAL_CONTROL_FLAG_BLOCKING) == 0) {
141  return;
142  }
143 
144  if (packet.flags & SERIAL_CONTROL_FLAG_BLOCKING) {
145  while (!HAVE_PAYLOAD_SPACE(chan, SERIAL_CONTROL)) {
146  hal.scheduler->delay(1);
147  }
148  } else {
149  if (!HAVE_PAYLOAD_SPACE(chan, SERIAL_CONTROL)) {
150  // no space for reply
151  return;
152  }
153  }
154 
155  // read any reply data
156  packet.count = 0;
157  memset(packet.data, 0, sizeof(packet.data));
158  while (available > 0) {
159  packet.data[packet.count++] = (uint8_t)stream->read();
160  available--;
161  }
162 
163  // and send the reply
164  _mav_finalize_message_chan_send(chan,
165  MAVLINK_MSG_ID_SERIAL_CONTROL,
166  (const char *)&packet,
167  MAVLINK_MSG_ID_SERIAL_CONTROL_MIN_LEN,
168  MAVLINK_MSG_ID_SERIAL_CONTROL_LEN,
169  MAVLINK_MSG_ID_SERIAL_CONTROL_CRC);
170  if ((flags & SERIAL_CONTROL_FLAG_MULTI) && packet.count != 0) {
171  if (flags & SERIAL_CONTROL_FLAG_BLOCKING) {
172  hal.scheduler->delay(1);
173  }
174  goto more_data;
175  }
176 }
