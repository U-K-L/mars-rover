16 /*
17  driver for Cheerson CX-OF optical flow sensor
18 
19  CXOF serial packet description
20  byte0: header (0xFE)
21  byte1: reserved
22  byte2: x-motion low byte;
23  byte3: x-motion high byte;
24  byte4: y-motion low byte;
25  byte5: y-motion high byte;
26  byte6: t-motion
27  byte7: surface quality
28  byte8: footer (0xAA)
29 
30  sensor sends packets at 25hz
31  */

91 // read latest values from sensor and fill in x,y and totals.
92 void AP_OpticalFlow_CXOF::update(void)
93 {
94  // sanity check uart
95  if (uart == nullptr) {
96  return;
97  }
98 
99  // record gyro values as long as they are being used
100  // the sanity check of dt below ensures old gyro values are not used
101  if (gyro_sum_count < 1000) {
102  const Vector3f& gyro = AP::ahrs_navekf().get_gyro();
103  gyro_sum.x += gyro.x;
104  gyro_sum.y += gyro.y;
105  gyro_sum_count++;
106  }
107 
108  // sensor values
109  int32_t x_sum = 0;
110  int32_t y_sum = 0;
111  uint16_t qual_sum = 0;
112  uint16_t count = 0;
113 
114  // read any available characters in the serial buffer
115  int16_t nbytes = uart->available();
116  while (nbytes-- > 0) {
117  int16_t r = uart->read();
118  if (r < 0) {
119  continue;
120  }
121  uint8_t c = (uint8_t)r;
122  // if buffer is empty and this byte is header, add to buffer
123  if (buf_len == 0) {
124  if (c == CXOF_HEADER) {
125  buf[buf_len++] = c;
126  }
127  } else {
128  // add character to buffer
129  buf[buf_len++] = c;
130 
131  // if buffer has 9 items try to decode it
132  if (buf_len >= CXOF_FRAME_LENGTH) {
133  // check last character matches footer
134  if (buf[buf_len-1] != CXOF_FOOTER) {
135  buf_len = 0;
136  continue;
137  }
138 
139  // decode package
140  int16_t x_raw = (int16_t)((uint16_t)buf[3] << 8) | buf[2];
141  int16_t y_raw = (int16_t)((uint16_t)buf[5] << 8) | buf[4];
142 
143  // add to sum of all readings from sensor this iteration
144  count++;
145  x_sum += x_raw;
146  y_sum += y_raw;
147  qual_sum += buf[7];
148 
149  // clear buffer
150  buf_len = 0;
151  }
152  }
153  }
154 
155  // return without updating state if no readings
156  if (count == 0) {
157  return;
158  }
159 
160  struct OpticalFlow::OpticalFlow_state state {};
161 
162  // average surface quality scaled to be between 0 and 255
163  state.surface_quality = (constrain_int16(qual_sum / count, 64, 78) - 64) * 255 / 14;
164 
165  // calculate dt
166  uint64_t this_frame_us = uart->receive_time_constraint_us(CXOF_FRAME_LENGTH);
167  if (this_frame_us == 0) {
168  // for HAL that cannot estimate arrival time in serial buffer use current time
169  this_frame_us = AP_HAL::micros();
170  }
171  float dt = (this_frame_us - last_frame_us) * 1.0e-6;
172  last_frame_us = this_frame_us;
173 
174  // sanity check dt
175  if (is_positive(dt) && (dt < CXOF_TIMEOUT_SEC)) {
176  // calculate flow values
177  const Vector2f flowScaler = _flowScaler();
178  float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
179  float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
180 
181  // copy flow rates to state structure
182  state.flowRate = Vector2f(((float)x_sum / count) * flowScaleFactorX,
183  ((float)y_sum / count) * flowScaleFactorY);
184  state.flowRate *= CXOF_PIXEL_SCALING / dt;
185 
186  // copy average body rate to state structure
187  state.bodyRate = Vector2f(gyro_sum.x / gyro_sum_count, gyro_sum.y / gyro_sum_count);
188 
189  _applyYaw(state.flowRate);
190  _applyYaw(state.bodyRate);
191  } else {
192  // first frame received in some time so cannot calculate flow values
193  state.flowRate.zero();
194  state.bodyRate.zero();
195  }
196 
197  _update_frontend(state);
198 
199  // reset gyro sum
200  gyro_sum.zero();
201  gyro_sum_count = 0;
202 }
