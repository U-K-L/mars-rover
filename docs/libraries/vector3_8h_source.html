230  // given a position p1 and a velocity v1 produce a vector
231  // perpendicular to v1 maximising distance from p1. If p1 is the
232  // zero vector the return from the function will always be the
233  // zero vector - that should be checked for.
234  static Vector3<T> perpendicular(const Vector3<T> &p1, const Vector3<T> &v1)
235  {
236  const T d = p1 * v1;
237  if (fabsf(d) < FLT_EPSILON) {
238  return p1;
239  }
240  const Vector3<T> parallel = (v1 * d) / v1.length_squared();
241  Vector3<T> perpendicular = p1 - parallel;
242 
243  return perpendicular;
244  }
