57 /*
58  detect uLanding Firmware Version
59 */
60 bool AP_RangeFinder_uLanding::detect_version(void)
61 {
62  if (_version_known) {
63  // return true if we've already detected the uLanding version
64  return true;
65  } else if (uart == nullptr) {
66  return false;
67  }
68 
69  bool hdr_found = false;
70  uint8_t byte1 = 0;
71  uint8_t count = 0;
72 
73  // read any available data from uLanding
74  int16_t nbytes = uart->available();
75 
76  while (nbytes-- > 0) {
77  uint8_t c = uart->read();
78 
79  if (((c == ULANDING_HDR_V0) || (c == ULANDING_HDR)) && !hdr_found) {
80  byte1 = c;
81  hdr_found = true;
82  count++;
83  } else if (hdr_found) {
84  if (byte1 == ULANDING_HDR_V0) {
85  if (++count < 4) {
86  /* need to collect 4 bytes to check for recurring
87  * header byte in the old 3-byte data format
88  */
89  continue;
90  } else {
91  if (c == byte1) {
92  // if header byte is recurring, set uLanding Version
93  _version = 0;
94  _header = ULANDING_HDR_V0;
95  _version_known = true;
96  return true;
97  } else {
98  /* if V0 header byte didn't occur again on 4th byte,
99  * start the search again for a header byte
100  */
101  count = 0;
102  byte1 = 0;
103  hdr_found = false;
104  }
105  }
106  } else {
107  if ((c & 0x80) || (c == ULANDING_HDR_V0)) {
108  /* Though unlikely, it is possible we could find ULANDING_HDR
109  * in a data byte from the old 3-byte format. In this case,
110  * either the next byte is another data byte (which by default
111  * is of the form 0x1xxxxxxx), or the next byte is the old
112  * header byte (ULANDING_HDR_V0). In this case, start the search
113  * again for a header byte.
114  */
115  count = 0;
116  byte1 = 0;
117  hdr_found = false;
118  } else {
119  /* if this second byte passes the above if statement, this byte
120  * is the version number
121  */
122  _version = c;
123  _header = ULANDING_HDR;
124  _version_known = true;
125  return true;
126  }
127  }
128  }
129  }
130 
131  /* return false if we've gone through all available data
132  * and haven't detected a uLanding firmware version
133  */
134  return false;
135 }

138 // read - return last value measured by sensor
139 bool AP_RangeFinder_uLanding::get_reading(uint16_t &reading_cm)
140 {
141  if (uart == nullptr) {
142  return false;
143  }
144 
145 
146  if (!detect_version()) {
147  // return false if uLanding version check failed
148  return false;
149  }
150 
151  // read any available lines from the uLanding
152  float sum = 0;
153  uint16_t count = 0;
154  bool hdr_found = false;
155 
156  int16_t nbytes = uart->available();
157 
158  while (nbytes-- > 0) {
159  uint8_t c = uart->read();
160 
161  if ((c == _header) && !hdr_found) {
162  // located header byte
163  _linebuf_len = 0;
164  hdr_found = true;
165  }
166  // decode index information
167  if (hdr_found) {
168  _linebuf[_linebuf_len++] = c;
169 
170  if ((_linebuf_len < (sizeof(_linebuf)/sizeof(_linebuf[0]))) ||
171  (_version == 0 && _linebuf_len < 3)) {
172  /* don't process _linebuf until we've collected six bytes of data
173  * (or 3 bytes for Version 0 firmware)
174  */
175  continue;
176  } else {
177  if (_version == 0 && _header != ULANDING_HDR) {
178  // parse data for Firmware Version #0
179  sum += (_linebuf[2]&0x7F)*128 + (_linebuf[1]&0x7F);
180  count++;
181  } else {
182  // evaluate checksum
183  if (((_linebuf[1] + _linebuf[2] + _linebuf[3] + _linebuf[4]) & 0xFF) == _linebuf[5]) {
184  // if checksum passed, parse data for Firmware Version #1
185  sum += _linebuf[3]*256 + _linebuf[2];
186  count++;
187  }
188  }
189 
190  hdr_found = false;
191  _linebuf_len = 0;
192  }
193  }
194  }
195 
196  if (count == 0) {
197  return false;
198  }
199 
200  reading_cm = sum / count;
201 
202  if (_version == 0 && _header != ULANDING_HDR) {
203  reading_cm *= 2.5f;
204  }
205 
206  return true;
207 }
