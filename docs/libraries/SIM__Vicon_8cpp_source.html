29 Vicon::Vicon()
30 {
31  int tmp[2];
32  if (pipe(tmp) == -1) {
33  AP_HAL::panic("pipe() failed");
34  }
35  fd_my_end = tmp[1];
36  fd_their_end = tmp[0];
37 
38  // close file descriptors on exec:
39  fcntl(fd_my_end, F_SETFD, FD_CLOEXEC);
40  fcntl(fd_their_end, F_SETFD, FD_CLOEXEC);
41 
42  // make sure we don't screw the simulation up by blocking:
43  fcntl(fd_my_end, F_SETFL, fcntl(fd_my_end, F_GETFL, 0) | O_NONBLOCK);
44  fcntl(fd_their_end, F_SETFL, fcntl(fd_their_end, F_GETFL, 0) | O_NONBLOCK);
45 
46  if (!valid_channel(mavlink_ch)) {
47  AP_HAL::panic("Invalid mavlink channel");
48  }
49 }

72 void Vicon::update_vicon_position_estimate(const Location &loc,
73  const Vector3f &position,
74  const Quaternion &attitude)
75 {
76  const uint64_t now_us = AP_HAL::micros64();
77 
78  if (time_offset_us == 0) {
79  time_offset_us = (unsigned(random()) % 7000) * 1000000ULL;
80  printf("time_offset_us %llu\n", (long long unsigned)time_offset_us);
81  }
82 
83  if (time_send_us && now_us >= time_send_us) {
84  uint8_t msgbuf[300];
85  uint16_t msgbuf_len = mavlink_msg_to_send_buffer(msgbuf, &obs_msg);
86 
87  if (::write(fd_my_end, (void*)&msgbuf, msgbuf_len) != msgbuf_len) {
88  ::fprintf(stderr, "Vicon: write failure\n");
89  }
90  time_send_us = 0;
91  }
92  if (time_send_us != 0) {
93  // waiting for the last msg to go out
94  return;
95  }
96 
97  if (now_us - last_observation_usec < 10000) {
98  // create observations at 10ms
99  return;
100  }
101 
102  float roll;
103  float pitch;
104  float yaw;
105  attitude.to_euler(roll, pitch, yaw);
106 
107  mavlink_msg_vicon_position_estimate_pack_chan(
108  system_id,
109  component_id,
110  mavlink_ch,
111  &obs_msg,
112  now_us + time_offset_us,
113  position.x,
114  position.y,
115  position.z,
116  roll,
117  pitch,
118  yaw,
119  NULL);
120 
121  uint32_t delay_ms = 25 + unsigned(random()) % 300;
122  time_send_us = now_us + delay_ms * 1000UL;
123 }
