121 bool AP_Compass_AK8963::init()
122 {
123  AP_HAL::Semaphore *bus_sem = _bus->get_semaphore();
124 
125  if (!bus_sem || !_bus->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
126  hal.console->printf("AK8963: Unable to get bus semaphore\n");
127  return false;
128  }
129 
130  if (!_bus->configure()) {
131  hal.console->printf("AK8963: Could not configure the bus\n");
132  goto fail;
133  }
134 
135  if (!_check_id()) {
136  hal.console->printf("AK8963: Wrong id\n");
137  goto fail;
138  }
139 
140  if (!_calibrate()) {
141  hal.console->printf("AK8963: Could not read calibration data\n");
142  goto fail;
143  }
144 
145  if (!_setup_mode()) {
146  hal.console->printf("AK8963: Could not setup mode\n");
147  goto fail;
148  }
149 
150  if (!_bus->start_measurements()) {
151  hal.console->printf("AK8963: Could not start measurements\n");
152  goto fail;
153  }
154 
155  _initialized = true;
156 
157  /* register the compass instance in the frontend */
158  _compass_instance = register_compass();
159 
160  set_rotation(_compass_instance, _rotation);
161 
162  _bus->set_device_type(DEVTYPE_AK8963);
163  set_dev_id(_compass_instance, _bus->get_bus_id());
164 
165  bus_sem->give();
166 
167  _bus->register_periodic_callback(10000, FUNCTOR_BIND_MEMBER(&AP_Compass_AK8963::_update, void));
168 
169  return true;
170 
171 fail:
172  bus_sem->give();
173  return false;
174 }

199 void AP_Compass_AK8963::_update()
200 {
201  struct sample_regs regs;
202  Vector3f raw_field;
203 
204  if (!_bus->block_read(AK8963_HXL, (uint8_t *) &regs, sizeof(regs))) {
205  return;
206  }
207 
208  /* Check for overflow. See AK8963's datasheet, section
209  * 6.4.3.6 - Magnetic Sensor Overflow. */
210  if ((regs.st2 & 0x08)) {
211  return;
212  }
213 
214  raw_field = Vector3f(regs.val[0], regs.val[1], regs.val[2]);
215 
216  if (is_zero(raw_field.x) && is_zero(raw_field.y) && is_zero(raw_field.z)) {
217  return;
218  }
219 
220  _make_factory_sensitivity_adjustment(raw_field);
221  _make_adc_sensitivity_adjustment(raw_field);
222  raw_field *= AK8963_MILLIGAUSS_SCALE;
223 
224  accumulate_sample(raw_field, _compass_instance, 10);
225 }

252 bool AP_Compass_AK8963::_calibrate()
253 {
254  /* Enable FUSE-mode in order to be able to read calibration data */
255  _bus->register_write(AK8963_CNTL1, AK8963_FUSE_MODE | AK8963_16BIT_ADC);
256 
257  uint8_t response[3];
258 
259  _bus->block_read(AK8963_ASAX, response, 3);
260 
261  for (int i = 0; i < 3; i++) {
262  float data = response[i];
263  _magnetometer_ASA[i] = ((data - 128) / 256 + 1);
264  }
265 
266  return true;
267 }

324 bool AP_AK8963_BusDriver_Auxiliary::block_read(uint8_t reg, uint8_t *buf, uint32_t size)
325 {
326  if (_started) {
327  /*
328  * We can only read a block when reading the block of sample values -
329  * calling with any other value is a mistake
330  */
331  assert(reg == AK8963_HXL);
332 
333  int n = _slave->read(buf);
334  return n == static_cast<int>(size);
335  }
336 
337  int r = _slave->passthrough_read(reg, buf, size);
338 
339  return r > 0 && static_cast<uint32_t>(r) == size;
340 }
