91 /*
92  parse a home string into a location and yaw
93  */
94 bool Aircraft::parse_home(const char *home_str, Location &loc, float &yaw_degrees)
95 {
96  char *saveptr = nullptr;
97  char *s = strdup(home_str);
98  if (!s) {
99  free(s);
100  ::printf("No home string supplied\n");
101  return false;
102  }
103  char *lat_s = strtok_r(s, ",", &saveptr);
104  if (!lat_s) {
105  free(s);
106  ::printf("Failed to parse latitude\n");
107  return false;
108  }
109  char *lon_s = strtok_r(nullptr, ",", &saveptr);
110  if (!lon_s) {
111  free(s);
112  ::printf("Failed to parse longitude\n");
113  return false;
114  }
115  char *alt_s = strtok_r(nullptr, ",", &saveptr);
116  if (!alt_s) {
117  free(s);
118  ::printf("Failed to parse altitude\n");
119  return false;
120  }
121  char *yaw_s = strtok_r(nullptr, ",", &saveptr);
122  if (!yaw_s) {
123  free(s);
124  ::printf("Failed to parse yaw\n");
125  return false;
126  }
127 
128  loc = {};
129  loc.lat = static_cast<int32_t>(strtod(lat_s, nullptr) * 1.0e7);
130  loc.lng = static_cast<int32_t>(strtod(lon_s, nullptr) * 1.0e7);
131  loc.alt = static_cast<int32_t>(strtod(alt_s, nullptr) * 1.0e2);
132 
133  if (loc.lat == 0 && loc.lng == 0) {
134  // default to CMAC instead of middle of the ocean. This makes
135  // SITL in MissionPlanner a bit more useful
136  loc.lat = -35.363261*1e7;
137  loc.lng = 149.165230*1e7;
138  loc.alt = 584*100;
139  }
140 
141  yaw_degrees = strtof(yaw_s, nullptr);
142  free(s);
143 
144  return true;
145 }

203 /*
204  update body magnetic field from position and rotation
205 */
206 void Aircraft::update_mag_field_bf()
207 {
208  // get the magnetic field intensity and orientation
209  float intensity;
210  float declination;
211  float inclination;
212  AP_Declination::get_mag_field_ef(location.lat * 1e-7f, location.lng * 1e-7f, intensity, declination, inclination);
213 
214  // create a field vector and rotate to the required orientation
215  Vector3f mag_ef(1e3f * intensity, 0.0f, 0.0f);
216  Matrix3f R;
217  R.from_euler(0.0f, -ToRad(inclination), ToRad(declination));
218  mag_ef = R * mag_ef;
219 
220  // calculate frame height above ground
221  const float frame_height_agl = fmaxf((-position.z) + home.alt * 0.01f - ground_level, 0.0f);
222 
223  if (!sitl) {
224  // running example program
225  return;
226  }
227 
228  // calculate scaling factor that varies from 1 at ground level to 1/8 at sitl->mag_anomaly_hgt
229  // Assume magnetic anomaly strength scales with 1/R**3
230  float anomaly_scaler = (sitl->mag_anomaly_hgt / (frame_height_agl + sitl->mag_anomaly_hgt));
231  anomaly_scaler = anomaly_scaler * anomaly_scaler * anomaly_scaler;
232 
233  // add scaled anomaly to earth field
234  mag_ef += sitl->mag_anomaly_ned.get() * anomaly_scaler;
235 
236  // Rotate into body frame
237  mag_bf = dcm.transposed() * mag_ef;
238 
239  // add motor interference
240  mag_bf += sitl->mag_mot.get() * battery_current;
241 }

279 /*
280  try to synchronise simulation time with wall clock time, taking
281  into account desired speedup
282  This tries to take account of possible granularity of
283  get_wall_time_us() so it works reasonably well on windows
284 */
285 void Aircraft::sync_frame_time(void)
286 {
287  frame_counter++;
288  uint64_t now = get_wall_time_us();
289  if (frame_counter >= 40 &&
290  now > last_wall_time_us) {
291  const float rate = frame_counter * 1.0e6f/(now - last_wall_time_us);
292  achieved_rate_hz = (0.99f*achieved_rate_hz) + (0.01f * rate);
293  if (achieved_rate_hz < rate_hz * target_speedup) {
294  scaled_frame_time_us *= 0.999f;
295  } else {
296  scaled_frame_time_us /= 0.999f;
297  }
298 #if 0
299  ::printf("achieved_rate_hz=%.3f rate=%.2f rate_hz=%.3f sft=%.1f\n",
300  static_cast<double>(achieved_rate_hz),
301  static_cast<double>(rate),
302  static_cast<double>(rate_hz),
303  static_cast<double>(scaled_frame_time_us));
304 #endif
305  const uint32_t sleep_time = static_cast<uint32_t>(scaled_frame_time_us * frame_counter);
306  if (sleep_time > min_sleep_time) {
307  usleep(sleep_time);
308  }
309  last_wall_time_us = now;
310  frame_counter = 0;
311  }
312 }

314 /* add noise based on throttle level (from 0..1) */
315 void Aircraft::add_noise(float throttle)
316 {
317  gyro += Vector3f(rand_normal(0, 1),
318  rand_normal(0, 1),
319  rand_normal(0, 1)) * gyro_noise * fabsf(throttle);
320  accel_body += Vector3f(rand_normal(0, 1),
321  rand_normal(0, 1),
322  rand_normal(0, 1)) * accel_noise * fabsf(throttle);
323 }
324 
325 /*
326  normal distribution random numbers
327  See
328  http://en.literateprograms.org/index.php?title=Special:DownloadCode/Box-Muller_transform_%28C%29&oldid=7011
329 */
330 double Aircraft::rand_normal(double mean, double stddev)
331 {
332  static double n2 = 0.0;
333  static int n2_cached = 0;
334  if (!n2_cached) {
335  double x, y, r;
336  do
337  {
338  x = 2.0 * rand()/RAND_MAX - 1;
339  y = 2.0 * rand()/RAND_MAX - 1;
340  r = x*x + y*y;
341  } while (is_zero(r) || r > 1.0);
342  const double d = sqrt(-2.0 * log(r)/r);
343  const double n1 = x * d;
344  n2 = y * d;
345  const double result = n1 * stddev + mean;
346  n2_cached = 1;
347  return result;
348  } else {
349  n2_cached = 0;
350  return n2 * stddev + mean;
351  }
352 }

408  fdm.yAccel = smoothing.accel_body.y;
409  fdm.zAccel = smoothing.accel_body.z;

472 /*
473  update the simulation attitude and relative position
474  */
475 void Aircraft::update_dynamics(const Vector3f &rot_accel)
476 {
477  const float delta_time = frame_time_us * 1.0e-6f;
478 
479  // update rotational rates in body frame
480  gyro += rot_accel * delta_time;
481 
482  gyro.x = constrain_float(gyro.x, -radians(2000.0f), radians(2000.0f));
483  gyro.y = constrain_float(gyro.y, -radians(2000.0f), radians(2000.0f));
484  gyro.z = constrain_float(gyro.z, -radians(2000.0f), radians(2000.0f));
485 
486  // estimate angular acceleration using a first order difference calculation
487  // TODO the simulator interface should provide the angular acceleration
488  ang_accel = (gyro - gyro_prev) / delta_time;
489  gyro_prev = gyro;
490 
491  // update attitude
492  dcm.rotate(gyro * delta_time);
493  dcm.normalize();
494 
495  Vector3f accel_earth = dcm * accel_body;
496  accel_earth += Vector3f(0.0f, 0.0f, GRAVITY_MSS);
497 
498  // if we're on the ground, then our vertical acceleration is limited
499  // to zero. This effectively adds the force of the ground on the aircraft
500  if (on_ground() && accel_earth.z > 0) {
501  accel_earth.z = 0;
502  }
503 
504  // work out acceleration as seen by the accelerometers. It sees the kinematic
505  // acceleration (ie. real movement), plus gravity
506  accel_body = dcm.transposed() * (accel_earth + Vector3f(0.0f, 0.0f, -GRAVITY_MSS));
507 
508  // new velocity vector
509  velocity_ef += accel_earth * delta_time;
510 
511  const bool was_on_ground = on_ground();
512  // new position vector
513  position += velocity_ef * delta_time;
514 
515  // velocity relative to air mass, in earth frame
516  velocity_air_ef = velocity_ef + wind_ef;
517 
518  // velocity relative to airmass in body frame
519  velocity_air_bf = dcm.transposed() * velocity_air_ef;
520 
521  // airspeed
522  airspeed = velocity_air_ef.length();
523 
524  // airspeed as seen by a fwd pitot tube (limited to 120m/s)
525  airspeed_pitot = constrain_float(velocity_air_bf * Vector3f(1.0f, 0.0f, 0.0f), 0.0f, 120.0f);
526 
527  // constrain height to the ground
528  if (on_ground()) {
529  if (!was_on_ground && AP_HAL::millis() - last_ground_contact_ms > 1000) {
530  printf("Hit ground at %f m/s\n", velocity_ef.z);
531  last_ground_contact_ms = AP_HAL::millis();
532  }
533  position.z = -(ground_level + frame_height - home.alt * 0.01f + ground_height_difference());
534 
535  switch (ground_behavior) {
536  case GROUND_BEHAVIOR_NONE:
537  break;
538  case GROUND_BEHAVIOR_NO_MOVEMENT: {
539  // zero roll/pitch, but keep yaw
540  float r, p, y;
541  dcm.to_euler(&r, &p, &y);
542  dcm.from_euler(0.0f, 0.0f, y);
543  // no X or Y movement
544  velocity_ef.x = 0.0f;
545  velocity_ef.y = 0.0f;
546  if (velocity_ef.z > 0.0f) {
547  velocity_ef.z = 0.0f;
548  }
549  gyro.zero();
550  use_smoothing = true;
551  break;
552  }
553  case GROUND_BEHAVIOR_FWD_ONLY: {
554  // zero roll/pitch, but keep yaw
555  float r, p, y;
556  dcm.to_euler(&r, &p, &y);
557  if (velocity_ef.length() < 5) {
558  // at high speeds don't constrain pitch, otherwise we
559  // can get stuck in takeoff
560  p = 0;
561  } else {
562  p = MAX(p, 0);
563  }
564  dcm.from_euler(0.0f, p, y);
565  // only fwd movement
566  Vector3f v_bf = dcm.transposed() * velocity_ef;
567  v_bf.y = 0.0f;
568  if (v_bf.x < 0.0f) {
569  v_bf.x = 0.0f;
570  }
571  velocity_ef = dcm * v_bf;
572  if (velocity_ef.z > 0.0f) {
573  velocity_ef.z = 0.0f;
574  }
575  gyro.zero();
576  use_smoothing = true;
577  break;
578  }
579  case GROUND_BEHAVIOR_TAILSITTER: {
580  // point straight up
581  float r, p, y;
582  dcm.to_euler(&r, &p, &y);
583  dcm.from_euler(0.0f, radians(90), y);
584  // no movement
585  if (accel_earth.z > -1.1*GRAVITY_MSS) {
586  velocity_ef.zero();
587  }
588  gyro.zero();
589  use_smoothing = true;
590  break;
591  }
592  }
593  }
594 }

596 /*
597  update wind vector
598 */
599 void Aircraft::update_wind(const struct sitl_input &input)
600 {
601  // wind vector in earth frame
602  wind_ef = Vector3f(cosf(radians(input.wind.direction))*cosf(radians(input.wind.dir_z)),
603  sinf(radians(input.wind.direction))*cosf(radians(input.wind.dir_z)),
604  sinf(radians(input.wind.dir_z))) * input.wind.speed;
605 
606  const float wind_turb = input.wind.turbulence * 10.0f; // scale input.wind.turbulence to match standard deviation when using iir_coef=0.98
607  const float iir_coef = 0.98f; // filtering high frequencies from turbulence
608 
609  if (wind_turb > 0 && !on_ground()) {
610 
611  turbulence_azimuth = turbulence_azimuth + (2 * rand());
612 
613  turbulence_horizontal_speed =
614  static_cast<float>(turbulence_horizontal_speed * iir_coef+wind_turb * rand_normal(0, 1) * (1 - iir_coef));
615 
616  turbulence_vertical_speed = static_cast<float>((turbulence_vertical_speed * iir_coef) + (wind_turb * rand_normal(0, 1) * (1 - iir_coef)));
617 
618  wind_ef += Vector3f(
619  cosf(radians(turbulence_azimuth)) * turbulence_horizontal_speed,
620  sinf(radians(turbulence_azimuth)) * turbulence_horizontal_speed,
621  turbulence_vertical_speed);
622  }
623 }

625 /*
626  smooth sensors for kinematic consistancy when we interact with the ground
627  */
628 void Aircraft::smooth_sensors(void)
629 {

631  Vector3f delta_pos = position - smoothing.position;
632  if (smoothing.last_update_us == 0 || delta_pos.length() > 10) {
633  smoothing.position = position;
634  smoothing.rotation_b2e = dcm;
635  smoothing.accel_body = accel_body;
636  smoothing.velocity_ef = velocity_ef;
637  smoothing.gyro = gyro;
638  smoothing.last_update_us = now;
639  smoothing.location = location;

641  return;
642  }
643  const float delta_time = (now - smoothing.last_update_us) * 1.0e-6f;
644  if (delta_time < 0 || delta_time > 0.1) {
645  return;
646  }
647 
648  // calculate required accel to get us to desired position and velocity in the time_constant
649  const float time_constant = 0.1f;
650  Vector3f dvel = (velocity_ef - smoothing.velocity_ef) + (delta_pos / time_constant);
651  Vector3f accel_e = dvel / time_constant + (dcm * accel_body + Vector3f(0.0f, 0.0f, GRAVITY_MSS));
652  const float accel_limit = 14 * GRAVITY_MSS;
653  accel_e.x = constrain_float(accel_e.x, -accel_limit, accel_limit);
654  accel_e.y = constrain_float(accel_e.y, -accel_limit, accel_limit);
655  accel_e.z = constrain_float(accel_e.z, -accel_limit, accel_limit);
656  smoothing.accel_body = smoothing.rotation_b2e.transposed() * (accel_e + Vector3f(0.0f, 0.0f, -GRAVITY_MSS));
657 
658  // calculate rotational rate to get us to desired attitude in time constant
659  Quaternion desired_q, current_q, error_q;
660  desired_q.from_rotation_matrix(dcm);
661  desired_q.normalize();
662  current_q.from_rotation_matrix(smoothing.rotation_b2e);
663  current_q.normalize();
664  error_q = desired_q / current_q;
665  error_q.normalize();
666 
667  Vector3f angle_differential;
668  error_q.to_axis_angle(angle_differential);
669  smoothing.gyro = gyro + angle_differential / time_constant;
670 
671  float R, P, Y;
672  smoothing.rotation_b2e.to_euler(&R, &P, &Y);
673  float R2, P2, Y2;
674  dcm.to_euler(&R2, &P2, &Y2);
675

677  AP::logger().Write("SMOO", "TimeUS,AEx,AEy,AEz,DPx,DPy,DPz,R,P,Y,R2,P2,Y2",

687

690  smoothing.rotation_b2e.rotate(smoothing.gyro * delta_time);
691  smoothing.rotation_b2e.normalize();

694  smoothing.velocity_ef += accel_e * delta_time;
