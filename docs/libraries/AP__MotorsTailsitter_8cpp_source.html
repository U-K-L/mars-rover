31 void AP_MotorsTailsitter::init(motor_frame_class frame_class, motor_frame_type frame_type)
32 {
33  // setup default motor and servo mappings
34  uint8_t chan;
35 
36  // right throttle defaults to servo output 1
37  SRV_Channels::set_aux_channel_default(SRV_Channel::k_throttleRight, CH_1);
38  if (SRV_Channels::find_channel(SRV_Channel::k_throttleRight, chan)) {
39  motor_enabled[chan] = true;
40  }
41 
42  // left throttle defaults to servo output 2
43  SRV_Channels::set_aux_channel_default(SRV_Channel::k_throttleLeft, CH_2);
44  if (SRV_Channels::find_channel(SRV_Channel::k_throttleLeft, chan)) {
45  motor_enabled[chan] = true;
46  }
47 
48  // right servo defaults to servo output 3
49  SRV_Channels::set_aux_channel_default(SRV_Channel::k_tiltMotorRight, CH_3);
50  SRV_Channels::set_angle(SRV_Channel::k_tiltMotorRight, SERVO_OUTPUT_RANGE);
51 
52  // left servo defaults to servo output 4
53  SRV_Channels::set_aux_channel_default(SRV_Channel::k_tiltMotorLeft, CH_4);
54  SRV_Channels::set_angle(SRV_Channel::k_tiltMotorLeft, SERVO_OUTPUT_RANGE);
55 
56  // record successful initialisation if what we setup was the desired frame_class
57  _flags.initialised_ok = (frame_class == MOTOR_FRAME_TAILSITTER);
58 }

79 void AP_MotorsTailsitter::output_to_motors()
80 {
81  if (!_flags.initialised_ok) {
82  return;
83  }
84 
85  switch (_spool_state) {
86  case SpoolState::SHUT_DOWN:
87  SRV_Channels::set_output_pwm(SRV_Channel::k_throttleLeft, get_pwm_output_min());
88  SRV_Channels::set_output_pwm(SRV_Channel::k_throttleRight, get_pwm_output_min());
89  break;
90  case SpoolState::GROUND_IDLE:
91  set_actuator_with_slew(_actuator[1], actuator_spin_up_to_ground_idle());
92  SRV_Channels::set_output_pwm(SRV_Channel::k_throttleLeft, output_to_pwm(actuator_spin_up_to_ground_idle()));
93  SRV_Channels::set_output_pwm(SRV_Channel::k_throttleRight, output_to_pwm(actuator_spin_up_to_ground_idle()));
94  break;
95  case SpoolState::SPOOLING_UP:
96  case SpoolState::THROTTLE_UNLIMITED:
97  case SpoolState::SPOOLING_DOWN:
98  SRV_Channels::set_output_pwm(SRV_Channel::k_throttleLeft, output_to_pwm(thrust_to_actuator(_thrust_left)));
99  SRV_Channels::set_output_pwm(SRV_Channel::k_throttleRight, output_to_pwm(thrust_to_actuator(_thrust_right)));
100  break;
101  }
102 
103  // Always output to tilt servos
104  SRV_Channels::set_output_scaled(SRV_Channel::k_tiltMotorLeft, _tilt_left*SERVO_OUTPUT_RANGE);
105  SRV_Channels::set_output_scaled(SRV_Channel::k_tiltMotorRight, _tilt_right*SERVO_OUTPUT_RANGE);
106 
107 }

109 // get_motor_mask - returns a bitmask of which outputs are being used for motors (1 means being used)
110 // this can be used to ensure other pwm outputs (i.e. for servos) do not conflict
111 uint16_t AP_MotorsTailsitter::get_motor_mask()
112 {
113  uint32_t motor_mask = 0;
114  uint8_t chan;
115  if (SRV_Channels::find_channel(SRV_Channel::k_throttleLeft, chan)) {
116  motor_mask |= 1U << chan;
117  }
118  if (SRV_Channels::find_channel(SRV_Channel::k_throttleRight, chan)) {
119  motor_mask |= 1U << chan;
120  }
121 
122  // add parent's mask
123  motor_mask |= AP_MotorsMulticopter::get_motor_mask();
124 
125  return motor_mask;
126 }

128 // calculate outputs to the motors
129 void AP_MotorsTailsitter::output_armed_stabilizing()
130 {
131  float roll_thrust; // roll thrust input value, +/- 1.0
132  float pitch_thrust; // pitch thrust input value, +/- 1.0
133  float yaw_thrust; // yaw thrust input value, +/- 1.0
134  float throttle_thrust; // throttle thrust input value, 0.0 - 1.0
135  float thrust_max; // highest motor value
136  float thr_adj = 0.0f; // the difference between the pilot's desired throttle and throttle_thrust_best_rpy
137 
138  // apply voltage and air pressure compensation
139  const float compensation_gain = get_compensation_gain();
140  roll_thrust = _roll_in * compensation_gain;
141  pitch_thrust = _pitch_in * compensation_gain;
142  yaw_thrust = _yaw_in * compensation_gain;
143  throttle_thrust = get_throttle() * compensation_gain;
144 
145  // sanity check throttle is above zero and below current limited throttle
146  if (throttle_thrust <= 0.0f) {
147  throttle_thrust = 0.0f;
148  limit.throttle_lower = true;
149  }
150  if (throttle_thrust >= _throttle_thrust_max) {
151  throttle_thrust = _throttle_thrust_max;
152  limit.throttle_upper = true;
153  }
154 
155  // calculate left and right throttle outputs
156  _thrust_left = throttle_thrust + roll_thrust*0.5f;
157  _thrust_right = throttle_thrust - roll_thrust*0.5f;
158 
159  // if max thrust is more than one reduce average throttle
160  thrust_max = MAX(_thrust_right,_thrust_left);
161  if (thrust_max > 1.0f) {
162  thr_adj = 1.0f - thrust_max;
163  limit.throttle_upper = true;
164  limit.roll_pitch = true;
165  }
166 
167  // Add adjustment to reduce average throttle
168  _thrust_left = constrain_float(_thrust_left + thr_adj, 0.0f, 1.0f);
169  _thrust_right = constrain_float(_thrust_right + thr_adj, 0.0f, 1.0f);
170  _throttle = throttle_thrust + thr_adj;
171 
172  // thrust vectoring
173  _tilt_left = pitch_thrust - yaw_thrust;
174  _tilt_right = pitch_thrust + yaw_thrust;
175 }

177 // output_test_seq - spin a motor at the pwm value specified
178 // motor_seq is the motor's sequence number from 1 to the number of motors on the frame
179 // pwm value is an actual pwm value that will be output, normally in the range of 1000 ~ 2000
180 void AP_MotorsTailsitter::output_test_seq(uint8_t motor_seq, int16_t pwm)
181 {
182  // exit immediately if not armed
183  if (!armed()) {
184  return;
185  }
186 
187  // output to motors and servos
188  switch (motor_seq) {
189  case 1:
190  // right throttle
191  SRV_Channels::set_output_pwm(SRV_Channel::k_throttleRight, pwm);
192  break;
193  case 2:
194  // right tilt servo
195  SRV_Channels::set_output_pwm(SRV_Channel::k_tiltMotorRight, pwm);
196  break;
197  case 3:
198  // left throttle
199  SRV_Channels::set_output_pwm(SRV_Channel::k_throttleLeft, pwm);
200  break;
201  case 4:
202  // left tilt servo
203  SRV_Channels::set_output_pwm(SRV_Channel::k_tiltMotorLeft, pwm);
204  break;
205  default:
206  // do nothing
207  break;
208  }
209 }
