254 {
255  if (cc2500.ReadReg(CC2500_30_PARTNUM | CC2500_READ_BURST) != 0x80 ||

260

262 
263  cc2500.Reset();
264  hal.scheduler->delay_microseconds(100);
265  for (uint8_t i=0; i<ARRAY_SIZE(radio_config); i++) {
266  // write twice to cope with possible SPI errors
267  cc2500.WriteRegCheck(radio_config[i].reg, radio_config[i].value);
268  }
269  cc2500.Strobe(CC2500_SIDLE); // Go to idle...
270 
271  for (uint8_t c=0;c<0xFF;c++) {
272  //calibrate all channels
273  cc2500.Strobe(CC2500_SIDLE);
274  cc2500.WriteRegCheck(CC2500_0A_CHANNR, c);
275  cc2500.Strobe(CC2500_SCAL);
276  hal.scheduler->delay_microseconds(900);
277  calData[c][0] = cc2500.ReadReg(CC2500_23_FSCAL3);
278  calData[c][1] = cc2500.ReadReg(CC2500_24_FSCAL2);
279  calData[c][2] = cc2500.ReadReg(CC2500_25_FSCAL1);
280  }

406  handle a SRT packet
407  */
408 bool AP_Radio_cc2500::handle_SRT_packet(const uint8_t *packet)
409 {
410  const struct srt_packet *pkt = (const struct srt_packet *)packet;
411  if (pkt->length != sizeof(struct srt_packet)-1 ||
412  pkt->txid[0] != bindTxId[0] ||
413  pkt->txid[1] != bindTxId[1]) {
414  Debug(3, "len=%u p1=%02x p2=%02x\n", pkt->length, pkt->txid[0], pkt->txid[1]);
415  // not for us
416  return false;
417  }
418  if (pkt->version != 1) {
419  // only support version 1 so far
420  return false;
421  }
422  pwm_channels[0] = pkt->chan1 + 1000 + ((pkt->chan_high&0xC0)<<2);
423  pwm_channels[1] = pkt->chan2 + 1000 + ((pkt->chan_high&0x30)<<4);
424  pwm_channels[2] = pkt->chan3 + 1000 + ((pkt->chan_high&0x0C)<<6);
425  pwm_channels[3] = pkt->chan4 + 1000 + ((pkt->chan_high&0x03)<<8);
426  // we map the buttons onto two PWM channels for ease of integration with ArduPilot
427  pwm_channels[4] = 1000 + (pkt->buttons & 0x7) * 100;
428  pwm_channels[5] = 1000 + (pkt->buttons >> 3) * 100;
429 
430  uint8_t data = pkt->data;
431  /*
432  decode special data field
433  */
434  switch (pkt->pkt_type) {
435  case PKTYPE_VOLTAGE:
436  // voltage from TX is in 0.025 volt units. Convert to 0.01 volt units for easier display
437  pwm_channels[6] = data * 4;
438  break;
439  case PKTYPE_YEAR:
440  tx_date.firmware_year = data;
441  break;
442  case PKTYPE_MONTH:
443  tx_date.firmware_month = data;
444  break;
445  case PKTYPE_DAY:
446  tx_date.firmware_day = data;
447  break;
448  case PKTYPE_TELEM_RSSI:
449  tx_rssi = data;
450  break;
451  case PKTYPE_TELEM_PPS:
452  tx_pps = data;
453  break;
454  case PKTYPE_BL_VERSION:
455  // unused so far for cc2500
456  break;
457  case PKTYPE_FW_ACK: {
458  // got an fw upload ack
459  Debug(4, "ack %u seq=%u acked=%u length=%u len=%u\n",
460  data, fwupload.sequence, unsigned(fwupload.acked), unsigned(fwupload.length), fwupload.len);
461  if (fwupload.sequence == data && sem.take_nonblocking()) {
462  fwupload.sequence++;
463  fwupload.acked += fwupload.len;
464  if (fwupload.acked == fwupload.length) {
465  // trigger send of DATA16 ack to client
466  fwupload.need_ack = true;
467  }
468  sem.give();
469  }
470  break;
471  }
472  }
473 
474  if (chan_count < 7) {
475  chan_count = 7;
476  }
477 
478  if (pkt->channr != channr) {
479  Debug(2, "channr=%u hop_chan=%u\n", channr, pkt->channr);
480  channr = pkt->channr;
481  }
482  if (pkt->chanskip != chanskip) {
483  Debug(2, "chanskip=%u skip=%u\n", chanskip, pkt->chanskip);
484  chanskip = pkt->chanskip;
485  }
486  return true;
487 }
