169 static void stm32_gpio_init(void) {
170 
171  /* Enabling GPIO-related clocks, the mask comes from the
172  registry header file.*/
173 #if defined(STM32H7)
174  rccResetAHB4(STM32_GPIO_EN_MASK);
175  rccEnableAHB4(STM32_GPIO_EN_MASK, true);
176 #else
177  rccResetAHB1(STM32_GPIO_EN_MASK);
178  rccEnableAHB1(STM32_GPIO_EN_MASK, true);
179 #endif
180 
181  /* Initializing all the defined GPIO ports.*/
182 #if STM32_HAS_GPIOA
183  gpio_init(GPIOA, &gpio_default_config.PAData);
184 #endif
185 #if STM32_HAS_GPIOB
186  gpio_init(GPIOB, &gpio_default_config.PBData);
187 #endif
188 #if STM32_HAS_GPIOC
189  gpio_init(GPIOC, &gpio_default_config.PCData);
190 #endif
191 #if STM32_HAS_GPIOD
192  gpio_init(GPIOD, &gpio_default_config.PDData);
193 #endif
194 #if STM32_HAS_GPIOE
195  gpio_init(GPIOE, &gpio_default_config.PEData);
196 #endif
