18 /*
19  * This comment section explains the basic idea behind the implementation.
20  *
21  * Vectors difference notation
22  * ===========================
23  * Let v and w be vectors. For readability purposes, unless explicitly
24  * otherwise noted, the notation vw will be used to represent w - v.
25  *
26  * Relationship between a vector and a triangle in 3d space
27  * ========================================================
28  * Vector in the area of a triangle
29  * --------------------------------
30  * Let T = (a, b, c) be a triangle, where a, b and c are also vectors and
31  * linearly independent. A vector inside that triangle can be written as one of
32  * its vertices plus the sum of the positively scaled vectors from that vertex
33  * to the other ones. Taking a as the first vertex, a vector p in the area
34  * formed by T can be written as:
35  *
36  * p = a + w_ab * ab + w_ac * ac
37  *
38  * It's fairly easy to see that if p is in the area formed by T, then w_ab >= 0
39  * and w_ac >= 0. That vector p can also be written as:
40  *
41  * p = b + w_ba * ba + w_bc * bc
42  *
43  * It's easy to check that the triangle formed by (a + w_ab * ab, b + w_ba *
44  * ba, p) is similar to T and, with the correct algebraic manipulations, we can
45  * come to the conclusion that:
46  *
47  * w_ba = 1 - w_ab - w_ac
48  *
49  * Since we know that w_ba >= 0, then w_ab + w_ac <= 1. Thus:
50  *
51  * ----------------------------------------------------------
52  * | p = a + w_ab * ab + w_ac * ac is in the area of T iff: |
53  * | w_ab >= 0 and w_ac >= 0 and w_ab + w_ac <= 1 |
54  * ----------------------------------------------------------
55  *
56  * Proving backwards shouldn't be difficult.
57  *
58  * Vector p can also be written as:
59  *
60  * p = (1 - w_ab - w_ba) * a + w_ab * b + w_ba * c
61  *
62  *
63  * Vector that crosses a triangle
64  * ------------------------------
65  * Let T be the same triangle discussed above and let v be a vector such that:
66  *
67  * v = x * a + y * b + z * c
68  * where x >= 0, y >= 0, z >= 0, and x + y + z > 0.
69  *
70  * It's geometrically easy to see that v crosses the triangle T. But that can
71  * also be verified analytically.
72  *
73  * The vector v crosses the triangle T iff there's a positive alpha such that
74  * alpha * v is in the area formed by T, so we need to prove that such value
75  * exists. To find alpha, we solve the equation alpha * v = p, which will lead
76  * us to the system, for the variables alpha, w_ab and w_ac:
77  *
78  * alpha * x = 1 - w_ab - w_ac
79  * alpha * y = w_ab
80  * alpha * z = w_ac,
81  * where w_ab >= 0 and w_ac >= 0 and w_ab + w_ac <= 1
82  *
83  * That will lead to alpha = 1 / (x + y + z), w_ab = y / (x + y + b) and
84  * w_ac = z / (x + y + z) and the following holds:
85  * - alpha does exist because x + y + z > 0.
86  * - w_ab >= 0 and w_ac >= 0 because y >= 0 and z >= 0 and x + y + z > 0.
87  * - 0 <= 1 - w_ab - w_ac <= 1 because 0 <= (y + z) / (x + y + z) <= 1.
88  *
89  * Thus:
90  *
91  * ----------------------------------------------------------
92  * | v = x * a + y * b + z * c crosses T = (a, b, c), where |
93  * | a, b and c are linearly independent, iff: |
94  * | x >= 0, y >= 0, z >= 0 and x + y + z > 0 |
95  * ----------------------------------------------------------
96  *
97  * Moreover:
98  * - if one of the coefficients is zero, then v crosses the edge formed by the
99  * vertices multiplied by the non-zero coefficients.
100  * - if two of the coefficients are zero, then v crosses the vertex multiplied
101  * by the non-zero coefficient.
102  */

187 int AP_GeodesicGrid::section(const Vector3f &v, bool inclusive)
188 {
189  int i = _triangle_index(v, inclusive);
190  if (i < 0) {
191  return -1;
192  }
193 
194  int j = _subtriangle_index(i, v, inclusive);
195  if (j < 0) {
196  return -1;
197  }
198 
199  return 4 * i + j;

210 int AP_GeodesicGrid::_from_neighbor_umbrella(int idx,
211  const Vector3f &v,
212  const Vector3f &u,
213  bool inclusive)
214 {
215  /* The following comparisons between the umbrella's first and second
216  * vertices' coefficients work for this algorithm because all vertices'
217  * vectors are of the same length. */
218 
219  if (is_equal(u.x, u.y)) {
220  /* If the coefficients of the first and second vertices are equal, then
221  * v crosses the first component or the edge formed by the umbrella's
222  * pivot and forth vertex. */
223  int comp = _neighbor_umbrella_component(idx, 0);
224  auto w = _inverses[comp % 10] * v;
225  if (comp > 9) {
226  w = -w;
227  }
228  float x0 = w[_neighbor_umbrellas[idx % 3].v0_c0];
229  if (is_zero(x0)) {
230  if (!inclusive) {
231  return -1;
232  }
233  return comp;
234  } else if (x0 < 0) {
235  if (!inclusive) {
236  return -1;
237  }
238  return _neighbor_umbrella_component(idx, u.x < u.y ? 3 : 2);
239  }
240 
241  return comp;
242  }
243 
244  if (u.y > u.x) {
245  /* If the coefficient of the second vertex is greater than the first
246  * one's, then v crosses the first, second or third component. */
247  int comp = _neighbor_umbrella_component(idx, 1);
248  auto w = _inverses[comp % 10] * v;
249  if (comp > 9) {
250  w = -w;
251  }
252  float x1 = w[_neighbor_umbrellas[idx % 3].v1_c1];
253  float x2 = w[_neighbor_umbrellas[idx % 3].v2_c1];
254 
255  if (is_zero(x1)) {
256  if (!inclusive) {
257  return -1;
258  }
259  return _neighbor_umbrella_component(idx, x1 < 0 ? 2 : 1);
260  } else if (x1 < 0) {
261  return _neighbor_umbrella_component(idx, 2);
262  }
263 
264  if (is_zero(x2)) {
265  if (!inclusive) {
266  return -1;
267  }
268  return _neighbor_umbrella_component(idx, x2 > 0 ? 1 : 0);
269  } else if (x2 < 0) {
270  return _neighbor_umbrella_component(idx, 0);
271  }
272 
273  return comp;
274  } else {
275  /* If the coefficient of the second vertex is lesser than the first
276  * one's, then v crosses the first, fourth or fifth component. */
277  int comp = _neighbor_umbrella_component(idx, 4);
278  auto w = _inverses[comp % 10] * v;
279  if (comp > 9) {
280  w = -w;
281  }
282  float x4 = w[_neighbor_umbrellas[idx % 3].v4_c4];
283  float x0 = w[_neighbor_umbrellas[idx % 3].v0_c4];
284 
285  if (is_zero(x4)) {
286  if (!inclusive) {
287  return -1;
288  }
289  return _neighbor_umbrella_component(idx, x4 < 0 ? 0 : 4);
290  } else if (x4 < 0) {
291  return _neighbor_umbrella_component(idx, 0);
292  }
293 
294  if (is_zero(x0)) {
295  if (!inclusive) {
296  return -1;
297  }
298  return _neighbor_umbrella_component(idx, x0 > 0 ? 4 : 3);
299  } else if (x0 < 0) {
300  return _neighbor_umbrella_component(idx, 3);
301  }
302 
303  return comp;
304  }

307 int AP_GeodesicGrid::_triangle_index(const Vector3f &v, bool inclusive)
308 {
309  /* w holds the coordinates of v with respect to the basis comprised by the
310  * vectors of T_i */
311  auto w = _inverses[0] * v;
312  int zero_count = 0;
313  int balance = 0;
314  int umbrella = -1;
315 
316  if (is_zero(w.x)) {
317  zero_count++;
318  } else if (w.x > 0) {
319  balance++;
320  } else {
321  balance--;
322  }
323 
324  if (is_zero(w.y)) {
325  zero_count++;
326  } else if (w.y > 0) {
327  balance++;
328  } else {
329  balance--;
330  }
331 
332  if (is_zero(w.z)) {
333  zero_count++;
334  } else if (w.z > 0) {
335  balance++;
336  } else {
337  balance--;
338  }
339 
340  switch (balance) {
341  case 3:
342  /* All coefficients are positive, thus return the first triangle. */
343  return 0;
344  case -3:
345  /* All coefficients are negative, which means that the coefficients for
346  * -w are positive, thus return the first triangle's opposite. */
347  return 10;
348  case 2:
349  /* Two coefficients are positive and one is zero, thus v crosses one of
350  * the edges of the first triangle. */
351  return inclusive ? 0 : -1;
352  case -2:
353  /* Analogous to the previous case, but for the opposite of the first
354  * triangle. */
355  return inclusive ? 10 : -1;
356  case 1:
357  /* There are two possible cases when balance is 1:
358  *
359  * 1) Two coefficients are zero, which means v crosses one of the
360  * vertices of the first triangle.
361  *
362  * 2) Two coefficients are positive and one is negative. Let a and b be
363  * vertices with positive coefficients and c the one with the negative
364  * coefficient. That means that v crosses the triangle formed by a, b
365  * and -c. The vector -c happens to be the 3-th vertex, with respect to
366  * (a, b), of the first triangle's neighbor umbrella with respect to a
367  * and b. Thus, v crosses one of the components of that umbrella. */
368  if (zero_count == 2) {
369  return inclusive ? 0 : -1;
370  }
371 
372  if (!is_zero(w.x) && w.x < 0) {
373  umbrella = 1;
374  } else if (!is_zero(w.y) && w.y < 0) {
375  umbrella = 2;
376  } else {
377  umbrella = 0;
378  }
379 
380  break;
381  case -1:
382  /* Analogous to the previous case, but for the opposite of the first
383  * triangle. */
384  if (zero_count == 2) {
385  return inclusive ? 10 : -1;
386  }
387 
388  if (!is_zero(w.x) && w.x > 0) {
389  umbrella = 4;
390  } else if (!is_zero(w.y) && w.y > 0) {
391  umbrella = 5;
392  } else {
393  umbrella = 3;
394  }
395  w = -w;
396 
397  break;
398  case 0:
399  /* There are two possible cases when balance is 1:
400  *
401  * 1) The vector v is the null vector, which doesn't cross any section.
402  *
403  * 2) One coefficient is zero, another is positive and yet another is
404  * negative. Let a, b and c be the respective vertices for those
405  * coefficients, then the statements in case (2) for when balance is 1
406  * are also valid here.
407  */
408  if (zero_count == 3) {
409  return -1;
410  }
411 
412  if (!is_zero(w.x) && w.x < 0) {
413  umbrella = 1;
414  } else if (!is_zero(w.y) && w.y < 0) {
415  umbrella = 2;
416  } else {
417  umbrella = 0;
418  }
419 
420  break;
421  }
422 
423  assert(umbrella >= 0);
424 
425  switch (umbrella % 3) {
426  case 0:
427  w.z = -w.z;
428  break;
429  case 1:
430  w(w.y, w.z, -w.x);
431  break;
432  case 2:
433  w(w.z, w.x, -w.y);
434  break;
435  }
436 
437  return _from_neighbor_umbrella(umbrella, v, w, inclusive);
438 }

440 int AP_GeodesicGrid::_subtriangle_index(const unsigned int triangle_index,
441  const Vector3f &v,
442  bool inclusive)
443 {
444  /* w holds the coordinates of v with respect to the basis comprised by the
445  * vectors of the middle triangle of T_i where i is triangle_index */
446  auto w = _mid_inverses[triangle_index % 10] * v;
447  if (triangle_index > 9) {
448  w = -w;
449  }
450 
451  if ((is_zero(w.x) || is_zero(w.y) || is_zero(w.z)) && !inclusive) {
452  return -1;
453  }
454 
455  /* At this point, we know that v crosses the icosahedron triangle pointed
456  * by triangle_index. Thus, we can geometrically see that if v doesn't
457  * cross its middle triangle, then one of the coefficients will be negative
458  * and the other ones positive. Let a and b be the non-negative
459  * coefficients and c the negative one. In that case, v will cross the
460  * triangle with vertices (a, b, -c). Since we know that v crosses the
461  * icosahedron triangle and the only sub-triangle that contains the set of
462  * points (seen as vectors) that cross the triangle (a, b, -c) is the
463  * middle triangle's neighbor with respect to a and b, then that
464  * sub-triangle is the one crossed by v. */
465  if (!is_zero(w.x) && w.x < 0) {
466  return 3;
467  }
468  if (!is_zero(w.y) && w.y < 0) {
469  return 1;
470  }
471  if (!is_zero(w.z) && w.z < 0) {
472  return 2;
473  }
474 
475  /* If x >= 0 and y >= 0 and z >= 0, then v crosses the middle triangle. */
476  return 0;
477 }
