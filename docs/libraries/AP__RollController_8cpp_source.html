92 int32_t AP_RollController::_get_rate_out(float desired_rate, float scaler, bool disable_integrator)
93 {
94  uint32_t tnow = AP_HAL::millis();
95  uint32_t dt = tnow - _last_t;
96  if (_last_t == 0 || dt > 1000) {
97  dt = 0;
98  }
99  _last_t = tnow;
100 
101  // Calculate equivalent gains so that values for K_P and K_I can be taken across from the old PID law
102  // No conversion is required for K_D
103  float ki_rate = gains.I * gains.tau;
104  float eas2tas = _ahrs.get_EAS2TAS();
105  float kp_ff = MAX((gains.P - gains.I * gains.tau) * gains.tau - gains.D , 0) / eas2tas;
106  float k_ff = gains.FF / eas2tas;
107  float delta_time = (float)dt * 0.001f;
108  // Get body rate vector (radians/sec)
109  float omega_x = _ahrs.get_gyro().x;
110 
111  // Calculate the roll rate error (deg/sec) and apply gain scaler
112  float achieved_rate = ToDeg(omega_x);
113  float rate_error = (desired_rate - achieved_rate) * scaler;
114 
115  // Get an airspeed estimate - default to zero if none available
116  float aspeed;
117  if (!_ahrs.airspeed_estimate(&aspeed)) {
118  aspeed = 0.0f;
119  }
120 
121  // Multiply roll rate error by _ki_rate, apply scaler and integrate
122  // Scaler is applied before integrator so that integrator state relates directly to aileron deflection
123  // This means aileron trim offset doesn't change as the value of scaler changes with airspeed
124  // Don't integrate if in stabilise mode as the integrator will wind up against the pilots inputs
125  if (!disable_integrator && ki_rate > 0) {
126  //only integrate if gain and time step are positive and airspeed above min value.
127  if (dt > 0 && aspeed > float(aparm.airspeed_min)) {
128  float integrator_delta = rate_error * ki_rate * delta_time * scaler;
129  // prevent the integrator from increasing if surface defln demand is above the upper limit
130  if (_last_out < -45) {
131  integrator_delta = MAX(integrator_delta , 0);
132  } else if (_last_out > 45) {
133  // prevent the integrator from decreasing if surface defln demand is below the lower limit
134  integrator_delta = MIN(integrator_delta, 0);
135  }
136  _pid_info.I += integrator_delta;
137  }
138  } else {
139  _pid_info.I = 0;
140  }
141 
142  // Scale the integration limit
143  float intLimScaled = gains.imax * 0.01f;
144 
145  // Constrain the integrator state
146  _pid_info.I = constrain_float(_pid_info.I, -intLimScaled, intLimScaled);
147 
148  // Calculate the demanded control surface deflection
149  // Note the scaler is applied again. We want a 1/speed scaler applied to the feed-forward
150  // path, but want a 1/speed^2 scaler applied to the rate error path.
151  // This is because acceleration scales with speed^2, but rate scales with speed.
152  _pid_info.D = rate_error * gains.D * scaler;
153  _pid_info.P = desired_rate * kp_ff * scaler;
154  _pid_info.FF = desired_rate * k_ff * scaler;
155  _pid_info.desired = desired_rate;
156  _pid_info.actual = achieved_rate;
157 
158  _last_out = _pid_info.FF + _pid_info.P + _pid_info.D;
159 
160  if (autotune.running && aspeed > aparm.airspeed_min) {
161  // let autotune have a go at the values
162  // Note that we don't pass the integrator component so we get
163  // a better idea of how much the base PD controller
164  // contributed
165  autotune.update(desired_rate, achieved_rate, _last_out);
166  }
167 
168  _last_out += _pid_info.I;
169 
170  // Convert to centi-degrees and constrain
171  return constrain_float(_last_out * 100, -4500, 4500);
172 }

187 /*
188  Function returns an equivalent aileron deflection in centi-degrees in the range from -4500 to 4500
189  A positive demand is up
190  Inputs are:
191  1) demanded bank angle in centi-degrees
192  2) control gain scaler = scaling_speed / aspeed
193  3) boolean which is true when stabilise mode is active
194  4) minimum FBW airspeed (metres/sec)
195 */
196 int32_t AP_RollController::get_servo_out(int32_t angle_err, float scaler, bool disable_integrator)
197 {
198  if (gains.tau < 0.1f) {
199  gains.tau.set(0.1f);
200  }
201 
202  // Calculate the desired roll rate (deg/sec) from the angle error
203  float desired_rate = angle_err * 0.01f / gains.tau;
204 
205  // Limit the demanded roll rate
206  if (gains.rmax && desired_rate < -gains.rmax) {
207  desired_rate = - gains.rmax;
208  } else if (gains.rmax && desired_rate > gains.rmax) {
209  desired_rate = gains.rmax;
210  }
211 
212  return _get_rate_out(desired_rate, scaler, disable_integrator);
213 }
