55 // loop
56 void loop()
57 {
58  // terminate program if console fails to initialize
59  if (!hal.console->is_initialized()) {
60  return;
61  }
62 
63  // run accumulate() at 50Hz and update() at 10Hz
64  if ((AP_HAL::micros() - timer) > 20 * 1000UL) {
65  timer = AP_HAL::micros();
66  barometer.accumulate();
67  if (counter++ < 5) {
68  return;
69  }
70  counter = 0;
71  barometer.update();
72 
73  //calculate time taken for barometer readings to update
74  uint32_t read_time = AP_HAL::micros() - timer;
75  if (!barometer.healthy()) {
76  hal.console->printf("not healthy\n");
77  return;
78  }
79 
80  //output barometer readings to console
81  hal.console->printf(" Pressure: %.2f Pa\n"
82  " Temperature: %.2f degC\n"
83  " Relative Altitude: %.2f m\n"
84  " climb=%.2f m/s\n"
85  " Read + update time: %u usec\n"
86  "\n",
87  (double)barometer.get_pressure(),
88  (double)barometer.get_temperature(),
89  (double)barometer.get_altitude(),
90  (double)barometer.get_climb_rate(),
91  (unsigned)read_time);
92  } else {
93  // if stipulated time has not passed between two distinct readings, delay the program for a millisecond
94  hal.scheduler->delay(1);
95  }
96 }
