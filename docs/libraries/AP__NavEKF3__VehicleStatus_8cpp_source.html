252 // update inflight calculaton that determines if GPS data is good enough for reliable navigation
253 void NavEKF3_core::calcGpsGoodForFlight(void)
254 {
255  // use a simple criteria based on the GPS receivers claimed speed accuracy and the EKF innovation consistency checks
256 
257  // set up varaibles and constants used by filter that is applied to GPS speed accuracy
258  const float alpha1 = 0.2f; // coefficient for first stage LPF applied to raw speed accuracy data
259  const float tau = 10.0f; // time constant (sec) of peak hold decay
260  if (lastGpsCheckTime_ms == 0) {
261  lastGpsCheckTime_ms = imuSampleTime_ms;
262  }
263  float dtLPF = (imuSampleTime_ms - lastGpsCheckTime_ms) * 1e-3f;
264  lastGpsCheckTime_ms = imuSampleTime_ms;
265  float alpha2 = constrain_float(dtLPF/tau,0.0f,1.0f);
266 
267  // get the receivers reported speed accuracy
268  float gpsSpdAccRaw;
269  if (!AP::gps().speed_accuracy(gpsSpdAccRaw)) {
270  gpsSpdAccRaw = 0.0f;
271  }
272 
273  // filter the raw speed accuracy using a LPF
274  sAccFilterState1 = constrain_float((alpha1 * gpsSpdAccRaw + (1.0f - alpha1) * sAccFilterState1),0.0f,10.0f);
275 
276  // apply a peak hold filter to the LPF output
277  sAccFilterState2 = MAX(sAccFilterState1,((1.0f - alpha2) * sAccFilterState2));
278 
279  // Apply a threshold test with hysteresis to the filtered GPS speed accuracy data
280  if (sAccFilterState2 > 1.5f ) {
281  gpsSpdAccPass = false;
282  } else if(sAccFilterState2 < 1.0f) {
283  gpsSpdAccPass = true;
284  }
285 
286  // Apply a threshold test with hysteresis to the normalised position and velocity innovations
287  // Require a fail for one second and a pass for 10 seconds to transition
288  if (lastInnovFailTime_ms == 0) {
289  lastInnovFailTime_ms = imuSampleTime_ms;
290  lastInnovPassTime_ms = imuSampleTime_ms;
291  }
292  if (velTestRatio < 1.0f && posTestRatio < 1.0f) {
293  lastInnovPassTime_ms = imuSampleTime_ms;
294  } else if (velTestRatio > 0.7f || posTestRatio > 0.7f) {
295  lastInnovFailTime_ms = imuSampleTime_ms;
296  }
297  if ((imuSampleTime_ms - lastInnovPassTime_ms) > 1000) {
298  ekfInnovationsPass = false;
299  } else if ((imuSampleTime_ms - lastInnovFailTime_ms) > 10000) {
300  ekfInnovationsPass = true;
301  }
302 
303  // both GPS speed accuracy and EKF innovations must pass
304  gpsAccuracyGood = gpsSpdAccPass && ekfInnovationsPass;
305 }

308 void NavEKF3_core::detectFlight()
309 {
310  /*
311  If we are a fly forward type vehicle (eg plane), then in-air status can be determined through a combination of speed and height criteria.
312  Because of the differing certainty requirements of algorithms that need the in-flight / on-ground status we use two booleans where
313  onGround indicates a high certainty we are not flying and inFlight indicates a high certainty we are flying. It is possible for
314  both onGround and inFlight to be false if the status is uncertain, but they cannot both be true.
315 
316  If we are a plane as indicated by the assume_zero_sideslip() status, then different logic is used
317 
318  TODO - this logic should be moved out of the EKF and into the flight vehicle code.
319  */
320 
321  if (assume_zero_sideslip()) {
322  // To be confident we are in the air we use a criteria which combines arm status, ground speed, airspeed and height change
323  float gndSpdSq = sq(gpsDataDelayed.vel.x) + sq(gpsDataDelayed.vel.y);
324  bool highGndSpd = false;
325  bool highAirSpd = false;
326  bool largeHgtChange = false;
327 
328  // trigger at 8 m/s airspeed
329  if (_ahrs->airspeed_sensor_enabled()) {
330  const AP_Airspeed *airspeed = _ahrs->get_airspeed();
331  if (airspeed->get_airspeed() * AP::ahrs().get_EAS2TAS() > 10.0f) {
332  highAirSpd = true;
333  }
334  }
335 
336  // trigger at 10 m/s GPS velocity, but not if GPS is reporting bad velocity errors
337  if (gndSpdSq > 100.0f && gpsSpdAccuracy < 1.0f) {
338  highGndSpd = true;
339  }
340 
341  // trigger if more than 10m away from initial height
342  if (fabsf(hgtMea) > 10.0f) {
343  largeHgtChange = true;
344  }
345 
346  // Determine to a high certainty we are flying
347  if (motorsArmed && highGndSpd && (highAirSpd || largeHgtChange)) {
348  onGround = false;
349  inFlight = true;
350  }
351 
352  // if is possible we are in flight, set the time this condition was last detected
353  if (motorsArmed && (highGndSpd || highAirSpd || largeHgtChange)) {
354  airborneDetectTime_ms = imuSampleTime_ms;
355  onGround = false;
356  }
357 
358  // Determine to a high certainty we are not flying
359  // after 5 seconds of not detecting a possible flight condition or we are disarmed, we transition to on-ground mode
360  if(!motorsArmed || ((imuSampleTime_ms - airborneDetectTime_ms) > 5000)) {
361  onGround = true;
362  inFlight = false;
363  }
364  } else {
365  // Non fly forward vehicle, so can only use height and motor arm status
366 
367  // If the motors are armed then we could be flying and if they are not armed then we are definitely not flying
368  if (motorsArmed) {
369  onGround = false;
370  } else {
371  inFlight = false;
372  onGround = true;
373  }
374 
375  if (!onGround) {
376  // If height has increased since exiting on-ground, then we definitely are flying
377  if ((stateStruct.position.z - posDownAtTakeoff) < -1.5f) {
378  inFlight = true;
379  }
380 
381  // If rangefinder has increased since exiting on-ground, then we definitely are flying
382  if ((rangeDataNew.rng - rngAtStartOfFlight) > 0.5f) {
383  inFlight = true;
384  }
385 
386  // If more than 5 seconds since likely_flying was set
387  // true, then set inFlight true
388  if (_ahrs->get_time_flying_ms() > 5000) {
389  inFlight = true;
390  }
391  }
392 
393  }
394 
395  // store current on-ground and in-air status for next time
396  prevOnGround = onGround;
397  prevInFlight = inFlight;
398
