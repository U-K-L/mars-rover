72 {

74  switch (_type) {
75  case OA_PATHPLAN_DISABLED:

93 {

98  break;

101  hal.util->snprintf(failure_msg, failure_msg_len, "BendyRuler OA requires reboot");
102  return false;
103  }
104  break;
105  case OA_PATHPLAN_DIJKSTRA:
106  if (_oadijkstra == nullptr) {
107  hal.util->snprintf(failure_msg, failure_msg_len, "Dijkstra OA requires reboot");
108  return false;
109  }
110  break;
111  }
112  return true;

115 // provides an alternative target location if path planning around obstacles is required
116 // returns true and updates result_loc with an intermediate location
117 AP_OAPathPlanner::OA_RetState AP_OAPathPlanner::mission_avoidance(const Location &current_loc,
118  const Location &origin,
119  const Location &destination,
120  Location &result_origin,
121  Location &result_destination)
122 {
123  // exit immediately if disabled
124  if (_type == OA_PATHPLAN_DISABLED) {
125  return OA_NOT_REQUIRED;
126  }
127 
128  WITH_SEMAPHORE(_rsem);
129 
130  if (!_thread_created) {
131  // create the avoidance thread as low priority. It should soak
132  // up spare CPU cycles to fill in the avoidance_result structure based
133  // on requests in avoidance_request
134  if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_OAPathPlanner::avoidance_thread, void),
135  "avoidance",
136  8192, AP_HAL::Scheduler::PRIORITY_IO, -1)) {
137  return OA_ERROR;
138  }
139  _thread_created = true;
140  }
141 
142  const uint32_t now = AP_HAL::millis();
143 
144  // place new request for the thread to work on
145  avoidance_request.current_loc = current_loc;
146  avoidance_request.origin = origin;
147  avoidance_request.destination = destination;
148  avoidance_request.ground_speed_vec = AP::ahrs().groundspeed_vector();
149  avoidance_request.request_time_ms = now;
150 
151  // check result's destination matches our request
152  const bool destination_matches = (destination.lat == avoidance_result.destination.lat) && (destination.lng == avoidance_result.destination.lng);
153 
154  // check results have not timed out
155  const bool timed_out = now - avoidance_result.result_time_ms > OA_TIMEOUT_MS;
156 
157  // return results from background thread's latest checks
158  if (destination_matches && !timed_out) {
159  // we have a result from the thread
160  result_origin = avoidance_result.origin_new;
161  result_destination = avoidance_result.destination_new;
162  // log result
163  if (avoidance_result.result_time_ms != _logged_time_ms) {
164  _logged_time_ms = avoidance_result.result_time_ms;
165  AP::logger().Write_OA(_type, destination, result_destination);
166  }
167  return avoidance_result.ret_state;
168  }
169 
170  // if timeout then path planner is taking too long to respond
171  if (timed_out) {
172  return OA_ERROR;
173  }
174 
175  // background thread is working on a new destination
176  return OA_PROCESSING;
177 }

179 // avoidance thread that continually updates the avoidance_result structure based on avoidance_request
180 void AP_OAPathPlanner::avoidance_thread()
181 {
182  while (true) {
183 
184  // run at 10hz or less
185  hal.scheduler->delay(100);
186 
187  Location origin_new;
188  Location destination_new;
189  {
190  WITH_SEMAPHORE(_rsem);
191  uint32_t now = AP_HAL::millis();
192  if (now - avoidance_request.request_time_ms > OA_TIMEOUT_MS) {
193  // this is a very old request, don't process it
194  continue;
195  }
196 
197  // copy request to avoid conflict with main thread
198  avoidance_request2 = avoidance_request;
199 
200  // store passed in origin and destination so we can return it if object avoidance is not required
201  origin_new = avoidance_request.origin;
202  destination_new = avoidance_request.destination;
203  }
204 
205  // run background task looking for best alternative destination
206  OA_RetState res = OA_NOT_REQUIRED;
207  switch (_type) {
208  case OA_PATHPLAN_DISABLED:
209  continue;
210  case OA_PATHPLAN_BENDYRULER:
211  if (_oabendyruler == nullptr) {
212  continue;
213  }
214  _oabendyruler->set_config(_lookahead, _margin_max);
215  if (_oabendyruler->update(avoidance_request2.current_loc, avoidance_request2.destination, avoidance_request2.ground_speed_vec, origin_new, destination_new)) {
216  res = OA_SUCCESS;
217  }
218  break;
219  case OA_PATHPLAN_DIJKSTRA:
220  if (_oadijkstra == nullptr) {
221  continue;
222  }
223  _oadijkstra->set_fence_margin(_margin_max);
224  const AP_OADijkstra::AP_OADijkstra_State dijkstra_state = _oadijkstra->update(avoidance_request2.current_loc, avoidance_request2.destination, origin_new, destination_new);
225  switch (dijkstra_state) {
226  case AP_OADijkstra::DIJKSTRA_STATE_NOT_REQUIRED:
227  res = OA_NOT_REQUIRED;
228  break;
229  case AP_OADijkstra::DIJKSTRA_STATE_ERROR:
230  res = OA_ERROR;
231  break;
232  case AP_OADijkstra::DIJKSTRA_STATE_SUCCESS:
233  res = OA_SUCCESS;
234  break;
235  }
236  break;
237  }
238 
239  {
240  // give the main thread the avoidance result
241  WITH_SEMAPHORE(_rsem);
242  avoidance_result.destination = avoidance_request2.destination;
243  avoidance_result.origin_new = res ? origin_new : avoidance_result.origin_new;
244  avoidance_result.destination_new = res ? destination_new : avoidance_result.destination;
245  avoidance_result.result_time_ms = AP_HAL::millis();
246  avoidance_result.ret_state = res;
247  }
248  }
249 }
