255 // update inflight calculaton that determines if GPS data is good enough for reliable navigation
256 void NavEKF2_core::calcGpsGoodForFlight(void)
257 {
258  // use a simple criteria based on the GPS receivers claimed speed accuracy and the EKF innovation consistency checks
259 
260  // set up varaibles and constants used by filter that is applied to GPS speed accuracy
261  const float alpha1 = 0.2f; // coefficient for first stage LPF applied to raw speed accuracy data
262  const float tau = 10.0f; // time constant (sec) of peak hold decay
263  if (lastGpsCheckTime_ms == 0) {
264  lastGpsCheckTime_ms = imuSampleTime_ms;
265  }
266  float dtLPF = (imuSampleTime_ms - lastGpsCheckTime_ms) * 1e-3f;
267  lastGpsCheckTime_ms = imuSampleTime_ms;
268  float alpha2 = constrain_float(dtLPF/tau,0.0f,1.0f);
269 
270  // get the receivers reported speed accuracy
271  float gpsSpdAccRaw;
272  if (!AP::gps().speed_accuracy(gpsSpdAccRaw)) {
273  gpsSpdAccRaw = 0.0f;
274  }
275 
276  // filter the raw speed accuracy using a LPF
277  sAccFilterState1 = constrain_float((alpha1 * gpsSpdAccRaw + (1.0f - alpha1) * sAccFilterState1),0.0f,10.0f);
278 
279  // apply a peak hold filter to the LPF output
280  sAccFilterState2 = MAX(sAccFilterState1,((1.0f - alpha2) * sAccFilterState2));
281 
282  // Apply a threshold test with hysteresis to the filtered GPS speed accuracy data
283  if (sAccFilterState2 > 1.5f ) {
284  gpsSpdAccPass = false;
285  } else if(sAccFilterState2 < 1.0f) {
286  gpsSpdAccPass = true;
287  }
288 
289  // Apply a threshold test with hysteresis to the normalised position and velocity innovations
290  // Require a fail for one second and a pass for 10 seconds to transition
291  if (lastInnovFailTime_ms == 0) {
292  lastInnovFailTime_ms = imuSampleTime_ms;
293  lastInnovPassTime_ms = imuSampleTime_ms;
294  }
295  if (velTestRatio < 1.0f && posTestRatio < 1.0f) {
296  lastInnovPassTime_ms = imuSampleTime_ms;
297  } else if (velTestRatio > 0.7f || posTestRatio > 0.7f) {
298  lastInnovFailTime_ms = imuSampleTime_ms;
299  }
300  if ((imuSampleTime_ms - lastInnovPassTime_ms) > 1000) {
301  ekfInnovationsPass = false;
302  } else if ((imuSampleTime_ms - lastInnovFailTime_ms) > 10000) {
303  ekfInnovationsPass = true;
304  }
305 
306  // both GPS speed accuracy and EKF innovations must pass
307  gpsAccuracyGood = gpsSpdAccPass && ekfInnovationsPass;
308 }

311 void NavEKF2_core::detectFlight()
312 {
313  /*
314  If we are a fly forward type vehicle (eg plane), then in-air status can be determined through a combination of speed and height criteria.
315  Because of the differing certainty requirements of algorithms that need the in-flight / on-ground status we use two booleans where
316  onGround indicates a high certainty we are not flying and inFlight indicates a high certainty we are flying. It is possible for
317  both onGround and inFlight to be false if the status is uncertain, but they cannot both be true.
318 
319  If we are a plane as indicated by the assume_zero_sideslip() status, then different logic is used
320 
321  TODO - this logic should be moved out of the EKF and into the flight vehicle code.
322  */
323 
324  if (assume_zero_sideslip()) {
325  // To be confident we are in the air we use a criteria which combines arm status, ground speed, airspeed and height change
326  float gndSpdSq = sq(gpsDataDelayed.vel.x) + sq(gpsDataDelayed.vel.y);
327  bool highGndSpd = false;
328  bool highAirSpd = false;
329  bool largeHgtChange = false;
330 
331  // trigger at 8 m/s airspeed
332  if (_ahrs->airspeed_sensor_enabled()) {
333  const AP_Airspeed *airspeed = _ahrs->get_airspeed();
334  if (airspeed->get_airspeed() * AP::ahrs().get_EAS2TAS() > 10.0f) {
335  highAirSpd = true;
336  }
337  }
338 
339  // trigger at 10 m/s GPS velocity, but not if GPS is reporting bad velocity errors
340  if (gndSpdSq > 100.0f && gpsSpdAccuracy < 1.0f) {
341  highGndSpd = true;
342  }
343 
344  // trigger if more than 10m away from initial height
345  if (fabsf(hgtMea) > 10.0f) {
346  largeHgtChange = true;
347  }
348 
349  // Determine to a high certainty we are flying
350  if (motorsArmed && highGndSpd && (highAirSpd || largeHgtChange)) {
351  onGround = false;
352  inFlight = true;
353  }
354 
355  // if is possible we are in flight, set the time this condition was last detected
356  if (motorsArmed && (highGndSpd || highAirSpd || largeHgtChange)) {
357  airborneDetectTime_ms = imuSampleTime_ms;
358  onGround = false;
359  }
360 
361  // Determine to a high certainty we are not flying
362  // after 5 seconds of not detecting a possible flight condition or we are disarmed, we transition to on-ground mode
363  if(!motorsArmed || ((imuSampleTime_ms - airborneDetectTime_ms) > 5000)) {
364  onGround = true;
365  inFlight = false;
366  }
367  } else {
368  // Non fly forward vehicle, so can only use height and motor arm status
369 
370  // If the motors are armed then we could be flying and if they are not armed then we are definitely not flying
371  if (motorsArmed) {
372  onGround = false;
373  } else {
374  inFlight = false;
375  onGround = true;
376  }
377 
378  if (!onGround) {

382  }
383

386  inFlight = true;
387  }
388

390  // true, then set inFlight true
391  if (_ahrs->get_time_flying_ms() > 5000) {
392  inFlight = true;
393  }
394  }
395 
396  }

399  prevOnGround = onGround;
