86 // update navigation
87 void AR_WPNav::update(float dt)
88 {
89  // exit immediately if no current location, origin or destination
90  Location current_loc;
91  float speed;
92  if (!hal.util->get_soft_armed() || !_orig_and_dest_valid || !AP::ahrs().get_position(current_loc) || !_atc.get_forward_speed(speed)) {
93  _desired_speed_limited = _atc.get_desired_speed_accel_limited(0.0f, dt);
94  _desired_turn_rate_rads = 0.0f;
95  return;
96  }
97 
98  // if no recent calls initialise desired_speed_limited to current speed
99  if (!is_active()) {
100  _desired_speed_limited = speed;
101  }
102  _last_update_ms = AP_HAL::millis();
103 
104  // run path planning around obstacles
105  bool stop_vehicle = false;
106  AP_OAPathPlanner *oa = AP_OAPathPlanner::get_singleton();
107  if (oa != nullptr) {
108  const AP_OAPathPlanner::OA_RetState oa_retstate = oa->mission_avoidance(current_loc, _origin, _destination, _oa_origin, _oa_destination);
109  switch (oa_retstate) {
110  case AP_OAPathPlanner::OA_NOT_REQUIRED:
111  _oa_active = false;
112  break;
113  case AP_OAPathPlanner::OA_PROCESSING:
114  case AP_OAPathPlanner::OA_ERROR:
115  // during processing or in case of error, slow vehicle to a stop
116  stop_vehicle = true;
117  _oa_active = false;
118  break;
119  case AP_OAPathPlanner::OA_SUCCESS:
120  _oa_active = true;
121  break;
122  }
123  }
124  if (!_oa_active) {
125  _oa_origin = _origin;
126  _oa_destination = _destination;
127  }
128 
129  update_distance_and_bearing_to_destination();
130 
131  // check if vehicle has reached the destination
132  const bool near_wp = _distance_to_destination <= _radius;
133  const bool past_wp = !_oa_active && current_loc.past_interval_finish_line(_origin, _destination);
134  if (!_reached_destination && (near_wp || past_wp)) {
135  _reached_destination = true;
136  }
137 
138  // handle stopping vehicle if avoidance has failed
139  if (stop_vehicle) {
140  // decelerate to speed to zero and set turn rate to zero
141  _desired_speed_limited = _atc.get_desired_speed_accel_limited(0.0f, dt);
142  _desired_lat_accel = 0.0f;
143  _desired_turn_rate_rads = 0.0f;
144  return;
145  }
146 
147  // calculate the required turn of the wheels
148  update_steering(current_loc, speed);
149 
150  // calculate desired speed
151  update_desired_speed(dt);
152 }

154 // set desired location
155 bool AR_WPNav::set_desired_location(const struct Location& destination, float next_leg_bearing_cd)
156 {
157  // set origin to last destination if waypoint controller active
158  if (is_active() && _orig_and_dest_valid && _reached_destination) {
159  _origin = _destination;
160  } else {
161  // otherwise use reasonable stopping point
162  if (!get_stopping_location(_origin)) {
163  return false;
164  }
165  }
166 
167  // initialise some variables
168  _oa_origin = _origin;
169  _oa_destination = _destination = destination;
170  _orig_and_dest_valid = true;
171  _reached_destination = false;
172  update_distance_and_bearing_to_destination();
173 
174  // set final desired speed
175  _desired_speed_final = 0.0f;
176  if (!is_equal(next_leg_bearing_cd, AR_WPNAV_HEADING_UNKNOWN)) {
177  const float curr_leg_bearing_cd = _origin.get_bearing_to(_destination);
178  const float turn_angle_cd = wrap_180_cd(next_leg_bearing_cd - curr_leg_bearing_cd);
179  if (fabsf(turn_angle_cd) < 10.0f) {
180  // if turning less than 0.1 degrees vehicle can continue at full speed
181  // we use 0.1 degrees instead of zero to avoid divide by zero in calcs below
182  _desired_speed_final = _desired_speed;
183  } else if (use_pivot_steering_at_next_WP(turn_angle_cd)) {
184  // pivoting so we will stop
185  _desired_speed_final = 0.0f;
186  } else {
187  // calculate maximum speed that keeps overshoot within bounds
188  const float radius_m = fabsf(_overshoot / (cosf(radians(turn_angle_cd * 0.01f)) - 1.0f));
189  _desired_speed_final = MIN(_desired_speed, safe_sqrt(_turn_max_mss * radius_m));
190  }
191  }
192 
193  return true;
194 }

220 // calculate vehicle stopping point using current location, velocity and maximum acceleration
221 bool AR_WPNav::get_stopping_location(Location& stopping_loc)
222 {
223  Location current_loc;
224  if (!AP::ahrs().get_position(current_loc)) {
225  return false;
226  }
227 
228  // get current velocity vector and speed
229  const Vector2f velocity = AP::ahrs().groundspeed_vector();
230  const float speed = velocity.length();
231 
232  // avoid divide by zero
233  if (!is_positive(speed)) {
234  stopping_loc = current_loc;
235  return true;
236  }
237 
238  // get stopping distance in meters
239  const float stopping_dist = _atc.get_stopping_distance(speed);
240 
241  // calculate stopping position from current location in meters
242  const Vector2f stopping_offset = velocity.normalized() * stopping_dist;
243  stopping_loc = current_loc;
244  stopping_loc.offset(stopping_offset.x, stopping_offset.y);
245 
246  return true;
247 }

249 // returns true if vehicle should pivot turn at next waypoint
250 bool AR_WPNav::use_pivot_steering_at_next_WP(float yaw_error_cd) const
251 {
252  // check cases where we clearly cannot use pivot steering
253  if (!_pivot_possible || _pivot_angle <= 0) {
254  return false;
255  }
256 
257  // if error is larger than _pivot_angle then use pivot steering at next WP
258  if (fabsf(yaw_error_cd) * 0.01f > _pivot_angle) {
259  return true;
260  }
261 
262  return false;
263 }

265 // returns true if vehicle should pivot immediately (because heading error is too large)
266 bool AR_WPNav::use_pivot_steering(float yaw_error_cd)
267 {
268  // check cases where we clearly cannot use pivot steering
269  if (!_pivot_possible || (_pivot_angle <= 0)) {
270  _pivot_active = false;
271  return false;
272  }
273 
274  // calc bearing error
275  const float yaw_error = fabsf(yaw_error_cd) * 0.01f;
276 
277  // if error is larger than _pivot_angle start pivot steering
278  if (yaw_error > _pivot_angle) {
279  _pivot_active = true;
280  return true;
281  }
282 
283  // if within 10 degrees of the target heading, exit pivot steering
284  if (yaw_error < 10.0f) {
285  _pivot_active = false;
286  return false;
287  }
288 
289  // by default stay in
290  return _pivot_active;
291 }

299 // update distance from vehicle's current position to destination
300 void AR_WPNav::update_distance_and_bearing_to_destination()
301 {
302  // if no current location leave distance unchanged
303  Location current_loc;
304  if (!_orig_and_dest_valid || !AP::ahrs().get_position(current_loc)) {
305  _distance_to_destination = 0.0f;
306  _wp_bearing_cd = 0.0f;
307 
308  // update OA adjusted values
309  _oa_distance_to_destination = 0.0f;
310  _oa_wp_bearing_cd = 0.0f;
311  return;
312  }
313  _distance_to_destination = current_loc.get_distance(_destination);
314  _wp_bearing_cd = current_loc.get_bearing_to(_destination);
315 
316  // update OA adjusted values
317  if (_oa_active) {
318  _oa_distance_to_destination = current_loc.get_distance(_oa_destination);
319  _oa_wp_bearing_cd = _oa_origin.get_bearing_to(_oa_destination);
320  } else {
321  _oa_distance_to_destination = _distance_to_destination;
322  _oa_wp_bearing_cd = _wp_bearing_cd;
323  }
324 }

326 // calculate steering output to drive along line from origin to destination waypoint
327 // relies on update_distance_and_bearing_to_destination being called first so _wp_bearing_cd has been updated
328 void AR_WPNav::update_steering(const Location& current_loc, float current_speed)
329 {
330  // calculate yaw error for determining if vehicle should pivot towards waypoint
331  float yaw_cd = _reversed ? wrap_360_cd(_oa_wp_bearing_cd + 18000) : _oa_wp_bearing_cd;
332  float yaw_error_cd = wrap_180_cd(yaw_cd - AP::ahrs().yaw_sensor);
333 
334  // calculate desired turn rate and update desired heading
335  if (use_pivot_steering(yaw_error_cd)) {
336  _cross_track_error = 0.0f;
337  _desired_heading_cd = yaw_cd;
338  _desired_lat_accel = 0.0f;
339  _desired_turn_rate_rads = _atc.get_turn_rate_from_heading(radians(yaw_cd * 0.01f), radians(_pivot_rate));
340  } else {
341  // run L1 controller
342  _nav_controller.set_reverse(_reversed);
343  _nav_controller.update_waypoint(_reached_destination ? current_loc : _oa_origin, _oa_destination, _radius);
344 
345  // retrieve lateral acceleration, heading back towards line and crosstrack error
346  _desired_lat_accel = constrain_float(_nav_controller.lateral_acceleration(), -_turn_max_mss, _turn_max_mss);
347  _desired_heading_cd = wrap_360_cd(_nav_controller.nav_bearing_cd());
348  if (_reversed) {
349  _desired_lat_accel *= -1.0f;
350  _desired_heading_cd = wrap_360_cd(_desired_heading_cd + 18000);
351  }
352  _cross_track_error = _nav_controller.crosstrack_error();
353  _desired_turn_rate_rads = _atc.get_turn_rate_from_lat_accel(_desired_lat_accel, current_speed);
354  }
355 }

357 // calculated desired speed(in m/s) based on yaw error and lateral acceleration and/or distance to a waypoint
358 // relies on update_distance_and_bearing_to_destination and update_steering being run so these internal members
359 // have been updated: _oa_wp_bearing_cd, _cross_track_error, _oa_distance_to_destination
360 void AR_WPNav::update_desired_speed(float dt)
361 {
362  // reduce speed to zero during pivot turns
363  if (_pivot_active) {
364  // decelerate to zero
365  _desired_speed_limited = _atc.get_desired_speed_accel_limited(0.0f, dt);
366  return;
367  }
368 
369  // accelerate desired speed towards max
370  float des_speed_lim = _atc.get_desired_speed_accel_limited(_reversed ? -_desired_speed : _desired_speed, dt);
371 
372  // reduce speed to limit overshoot from line between origin and destination
373  // calculate number of degrees vehicle must turn to face waypoint
374  float ahrs_yaw_sensor = AP::ahrs().yaw_sensor;
375  const float heading_cd = is_negative(des_speed_lim) ? wrap_180_cd(ahrs_yaw_sensor + 18000) : ahrs_yaw_sensor;
376  const float wp_yaw_diff_cd = wrap_180_cd(_oa_wp_bearing_cd - heading_cd);
377  const float turn_angle_rad = fabsf(radians(wp_yaw_diff_cd * 0.01f));
378 
379  // calculate distance from vehicle to line + wp_overshoot
380  const float line_yaw_diff = wrap_180_cd(_oa_origin.get_bearing_to(_oa_destination) - heading_cd);
381  const float dist_from_line = fabsf(_cross_track_error);
382  const bool heading_away = is_positive(line_yaw_diff) == is_positive(_cross_track_error);
383  const float wp_overshoot_adj = heading_away ? -dist_from_line : dist_from_line;
384 
385  // calculate radius of circle that touches vehicle's current position and heading and target position and heading
386  float radius_m = 999.0f;
387  const float radius_calc_denom = fabsf(1.0f - cosf(turn_angle_rad));
388  if (!is_zero(radius_calc_denom)) {
389  radius_m = MAX(0.0f, _overshoot + wp_overshoot_adj) / radius_calc_denom;
390  }
391 
392  // calculate and limit speed to allow vehicle to stay on circle
393  const float overshoot_speed_max = safe_sqrt(_turn_max_mss * MAX(_turn_radius, radius_m));
394  des_speed_lim = constrain_float(des_speed_lim, -overshoot_speed_max, overshoot_speed_max);
395 
396  // limit speed based on distance to waypoint and max acceleration/deceleration
397  if (is_positive(_oa_distance_to_destination) && is_positive(_atc.get_decel_max())) {
398  const float dist_speed_max = safe_sqrt(2.0f * _oa_distance_to_destination * _atc.get_decel_max() + sq(_desired_speed_final));
399  des_speed_lim = constrain_float(des_speed_lim, -dist_speed_max, dist_speed_max);
400  }
401 
402  _desired_speed_limited = des_speed_lim;
403 }
