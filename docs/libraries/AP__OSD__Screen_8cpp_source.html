494 //Autoscroll message is the same as in minimosd-extra.
495 //Thanks to night-ghost for the approach.
496 void AP_OSD_Screen::draw_message(uint8_t x, uint8_t y)
497 {
498  AP_Notify * notify = AP_Notify::get_singleton();
499  if (notify) {
500  int32_t visible_time = AP_HAL::millis() - notify->get_text_updated_millis();
501  if (visible_time < osd->msgtime_s *1000) {
502  char buffer[NOTIFY_TEXT_BUFFER_SIZE];
503  strncpy(buffer, notify->get_text(), sizeof(buffer));
504  int16_t len = strnlen(buffer, sizeof(buffer));
505 
506  for (int16_t i=0; i<len; i++) {
507  //converted to uppercase,
508  //because we do not have small letter chars inside used font
509  buffer[i] = toupper(buffer[i]);
510  //normalize whitespace
511  if (isspace(buffer[i])) {
512  buffer[i] = ' ';
513  }
514  }
515 
516  int16_t start_position = 0;
517  //scroll if required
518  //scroll pattern: wait, scroll to the left, wait, scroll to the right
519  if (len > message_visible_width) {
520  int16_t chars_to_scroll = len - message_visible_width;
521  int16_t total_cycles = 2*message_scroll_delay + 2*chars_to_scroll;
522  int16_t current_cycle = (visible_time / message_scroll_time_ms) % total_cycles;
523 
524  //calculate scroll start_position
525  if (current_cycle < total_cycles/2) {
526  //move to the left
527  start_position = current_cycle - message_scroll_delay;
528  } else {
529  //move to the right
530  start_position = total_cycles - current_cycle;
531  }
532  start_position = constrain_int16(start_position, 0, chars_to_scroll);
533  int16_t end_position = start_position + message_visible_width;
534 
535  //ensure array boundaries
536  start_position = MIN(start_position, int(sizeof(buffer)-1));
537  end_position = MIN(end_position, int(sizeof(buffer)-1));
538 
539  //trim invisible part
540  buffer[end_position] = 0;
541  }
542 
543  backend->write(x, y, buffer + start_position);
544  }
545  }
