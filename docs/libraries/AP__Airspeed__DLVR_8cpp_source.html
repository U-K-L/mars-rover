60 // 50Hz timer
61 void AP_Airspeed_DLVR::timer()
62 {
63  uint8_t raw_bytes[4];
64  if (!dev->read((uint8_t *)&raw_bytes, sizeof(raw_bytes))) {
65  return;
66  }
67 
68  uint32_t data = (raw_bytes[0] << 24) |
69  (raw_bytes[1] << 16) |
70  (raw_bytes[2] << 8) |
71  raw_bytes[3];
72 
73  if ((data >> STATUS_SHIFT)) {
74  // anything other then 00 in the status bits is an error
75  Debug("DLVR: Bad status read %d", data >> STATUS_SHIFT);
76  return;
77  }
78 
79  uint32_t pres_raw = (data >> PRESSURE_SHIFT) & PRESSURE_MASK;
80  uint32_t temp_raw = (data >> TEMPERATURE_SHIFT) & TEMPERATURE_MASK;
81 
82  float press_h2o = 1.25f * 2.0f * DLVR_FSS * ((pres_raw - DLVR_OFFSET) / DLVR_SCALE);
83  float temp = temp_raw * (200.0f / 2047.0f) - 50.0f;
84 
85  WITH_SEMAPHORE(sem);
86 
87  pressure_sum += INCH_OF_H2O_TO_PASCAL * press_h2o;
88  temperature_sum += temp;
89  press_count++;
90  temp_count++;
91  last_sample_time_ms = AP_HAL::millis();
92 }
93 
94 // return the current differential_pressure in Pascal
95 bool AP_Airspeed_DLVR::get_differential_pressure(float &_pressure)
96 {
97  if ((AP_HAL::millis() - last_sample_time_ms) > 100) {
98  return false;
99  }
100 
101  {
102  WITH_SEMAPHORE(sem);
103  if (press_count > 0) {
104  pressure = pressure_sum / press_count;
105  press_count = 0;
106  pressure_sum = 0;
107  }
108  }
109 
110  _pressure = pressure;
111  return true;
112 }
