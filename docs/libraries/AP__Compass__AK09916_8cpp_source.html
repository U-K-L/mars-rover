96 AP_Compass_Backend *AP_Compass_AK09916::probe_ICM20948(AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev,
97  AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev_icm,
98  bool force_external,
99  enum Rotation rotation)
100 {
101  if (!dev || !dev_icm) {
102  return nullptr;
103  }
104 
105  if (!dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
106  return nullptr;
107  }
108 
109  /* Allow ICM20x48 to shortcut auxiliary bus and host bus */
110  uint8_t rval;
111  uint16_t whoami;
112  uint8_t retries = 5;
113  if (!dev_icm->read_registers(REG_ICM_WHOAMI, &rval, 1) ||
114  rval != ICM_WHOAMI_VAL) {
115  // not an ICM_WHOAMI
116  goto fail;
117  }
118  do {
119  // reset then bring sensor out of sleep mode
120  if (!dev_icm->write_register(REG_ICM_PWR_MGMT_1, 0x80)) {
121  goto fail;
122  }
123  hal.scheduler->delay(10);
124 
125  if (!dev_icm->write_register(REG_ICM_PWR_MGMT_1, 0x00)) {
126  goto fail;
127  }
128  hal.scheduler->delay(10);
129 
130  // see if ICM20948 is sleeping
131  if (!dev_icm->read_registers(REG_ICM_PWR_MGMT_1, &rval, 1)) {
132  goto fail;
133  }
134  if ((rval & 0x40) == 0) {
135  break;
136  }
137  } while (retries--);
138 
139  if (rval & 0x40) {
140  // it didn't come out of sleep
141  goto fail;
142  }
143 
144  // initially force i2c bypass off
145  dev_icm->write_register(REG_ICM_INT_PIN_CFG, 0x00);
146  hal.scheduler->delay(1);
147 
148  // now check if a AK09916 shows up on the bus. If it does then
149  // we have both a real AK09916 and a ICM20948 with an embedded
150  // AK09916. In that case we will fail the driver load and use
151  // the main AK09916 driver
152  if (dev->read_registers(REG_COMPANY_ID, (uint8_t *)&whoami, 2)) {
153  // a device is replying on the AK09916 I2C address, don't
154  // load the ICM20948
155  gcs().send_text(MAV_SEVERITY_INFO, "ICM20948: AK09916 bus conflict\n");
156  goto fail;
157  }
158 
159  // now force bypass on
160  dev_icm->write_register(REG_ICM_INT_PIN_CFG, 0x02);
161  hal.scheduler->delay(1);
162  dev->get_semaphore()->give();
163  return probe(std::move(dev), force_external, rotation);
164 fail:
165  dev->get_semaphore()->give();
166  return nullptr;
167 }

189 bool AP_Compass_AK09916::init()
190 {
191  AP_HAL::Semaphore *bus_sem = _bus->get_semaphore();
192 
193  if (!bus_sem || !_bus->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
194  gcs().send_text(MAV_SEVERITY_INFO,"AK09916: Unable to get bus semaphore\n");
195  return false;
196  }
197 
198  if (!_bus->configure()) {
199  gcs().send_text(MAV_SEVERITY_INFO,"AK09916: Could not configure the bus\n");
200  goto fail;
201  }
202 
203  if (!_reset()) {
204  gcs().send_text(MAV_SEVERITY_INFO,"AK09916: Reset Failed\n");
205  goto fail;
206  }
207 
208  if (!_check_id()) {
209  gcs().send_text(MAV_SEVERITY_INFO,"AK09916: Wrong id\n");
210  goto fail;
211  }
212 
213  if (!_setup_mode()) {
214  gcs().send_text(MAV_SEVERITY_INFO,"AK09916: Could not setup mode\n");
215  goto fail;
216  }
217 
218  if (!_bus->start_measurements()) {
219  gcs().send_text(MAV_SEVERITY_INFO,"AK09916: Could not start measurements\n");
220  goto fail;
221  }
222 
223  _initialized = true;
224 
225  /* register the compass instance in the frontend */
226  _compass_instance = register_compass();
227 
228  if (_force_external) {
229  set_external(_compass_instance, true);
230  }
231 
232  set_rotation(_compass_instance, _rotation);
233 
234  _bus->set_device_type(DEVTYPE_AK09916);
235  set_dev_id(_compass_instance, _bus->get_bus_id());
236 
237  bus_sem->give();
238 
239  _bus->register_periodic_callback(10000, FUNCTOR_BIND_MEMBER(&AP_Compass_AK09916::_update, void));
240 
241  return true;
242 
243 fail:
244  bus_sem->give();
245  return false;
246 }

264 void AP_Compass_AK09916::_update()
265 {
266  struct sample_regs regs = {0};
267  Vector3f raw_field;
268 
269  if (!_bus->block_read(REG_ST1, (uint8_t *) &regs, sizeof(regs))) {
270  return;
271  }
272 
273  if (!(regs.st1 & 0x01)) {
274  return;
275  }
276 
277  /* Check for overflow. See AK09916's datasheet*/
278  if ((regs.st2 & 0x08)) {
279  return;
280  }
281 
282  raw_field = Vector3f(regs.val[0], regs.val[1], regs.val[2]);
283 
284  if (is_zero(raw_field.x) && is_zero(raw_field.y) && is_zero(raw_field.z)) {
285  return;
286  }
287 
288  _make_adc_sensitivity_adjustment(raw_field);
289  raw_field *= AK09916_MILLIGAUSS_SCALE;
290 
291  accumulate_sample(raw_field, _compass_instance, 10);
292 }

373 bool AP_AK09916_BusDriver_Auxiliary::block_read(uint8_t reg, uint8_t *buf, uint32_t size)
374 {
375  if (_started) {
376  /*
377  * We can only read a block when reading the block of sample values -
378  * calling with any other value is a mistake
379  */
380  assert(reg == REG_ST1);
381 
382  int n = _slave->read(buf);
383  return n == static_cast<int>(size);
384  }
385 
386  int r = _slave->passthrough_read(reg, buf, size);
387 
388  return r > 0 && static_cast<uint32_t>(r) == size;
389 }
