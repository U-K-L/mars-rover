192 // initialise the Proximity class. We do detection of attached sensors here
193 // we don't allow for hot-plugging of sensors (i.e. reboot required)
194 void AP_Proximity::init(void)
195 {
196  if (num_instances != 0) {
197  // init called a 2nd time?
198  return;
199  }
200  for (uint8_t i=0; i<PROXIMITY_MAX_INSTANCES; i++) {
201  detect_instance(i);
202  if (drivers[i] != nullptr) {
203  // we loaded a driver for this instance, so it must be
204  // present (although it may not be healthy)
205  num_instances = i+1;
206  }
207 
208  // initialise status
209  state[i].status = Proximity_NotConnected;
210  }
211 }
212 
213 // update Proximity state for all instances. This should be called at a high rate by the main loop
214 void AP_Proximity::update(void)
215 {
216  for (uint8_t i=0; i<num_instances; i++) {
217  if (drivers[i] != nullptr) {
218  if (_type[i] == Proximity_Type_None) {
219  // allow user to disable a proximity sensor at runtime
220  state[i].status = Proximity_NotConnected;
221  continue;
222  }
223  drivers[i]->update();
224  }
225  }
226 
227  // work out primary instance - first sensor returning good data
228  for (int8_t i=num_instances-1; i>=0; i--) {
229  if (drivers[i] != nullptr && (state[i].status == Proximity_Good)) {
230  primary_instance = i;
231  }
232  }
233 }

281 // detect if an instance of a proximity sensor is connected.
282 void AP_Proximity::detect_instance(uint8_t instance)
283 {
284  uint8_t type = _type[instance];
285  if (type == Proximity_Type_SF40C) {
286  if (AP_Proximity_LightWareSF40C::detect(serial_manager)) {
287  state[instance].instance = instance;
288  drivers[instance] = new AP_Proximity_LightWareSF40C(*this, state[instance], serial_manager);
289  return;
290  }
291  }
292  if (type == Proximity_Type_RPLidarA2) {
293  if (AP_Proximity_RPLidarA2::detect(serial_manager)) {
294  state[instance].instance = instance;
295  drivers[instance] = new AP_Proximity_RPLidarA2(*this, state[instance], serial_manager);
296  return;
297  }
298  }
299  if (type == Proximity_Type_MAV) {
300  state[instance].instance = instance;
301  drivers[instance] = new AP_Proximity_MAV(*this, state[instance]);
302  return;
303  }
304  if (type == Proximity_Type_TRTOWER) {
305  if (AP_Proximity_TeraRangerTower::detect(serial_manager)) {
306  state[instance].instance = instance;
307  drivers[instance] = new AP_Proximity_TeraRangerTower(*this, state[instance], serial_manager);
308  return;
309  }
310  }
311  if (type == Proximity_Type_TRTOWEREVO) {
312  if (AP_Proximity_TeraRangerTowerEvo::detect(serial_manager)) {
313  state[instance].instance = instance;
314  drivers[instance] = new AP_Proximity_TeraRangerTowerEvo(*this, state[instance], serial_manager);
315  return;
316  }
317  }
318  if (type == Proximity_Type_RangeFinder) {
319  state[instance].instance = instance;
320  drivers[instance] = new AP_Proximity_RangeFinder(*this, state[instance]);
321  return;
322  }
323 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
324  if (type == Proximity_Type_SITL) {
325  state[instance].instance = instance;
326  drivers[instance] = new AP_Proximity_SITL(*this, state[instance]);
327  return;
328  }
329  if (type == Proximity_Type_MorseSITL) {
330  state[instance].instance = instance;
331  drivers[instance] = new AP_Proximity_MorseSITL(*this, state[instance]);
332  return;
333  }
334 #endif
335 }
