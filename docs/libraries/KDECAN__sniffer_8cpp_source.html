128 void KDECAN_sniffer::loop(void)
129 {
130  if (_can_driver == nullptr) {
131  return;
132  }
133 
134  uavcan::CanFrame empty_frame { (0 | uavcan::CanFrame::FlagEFF), nullptr, 0 };
135  const uavcan::CanFrame* select_frames[uavcan::MaxCanIfaces] { &empty_frame };
136  uavcan::MonotonicTime timeout = uavcan::MonotonicTime::fromMSec(AP_HAL::millis() + 1);
137 
138  uavcan::CanSelectMasks inout_mask;
139  inout_mask.read = 1 << _interface;
140 
141  uavcan::CanSelectMasks in_mask = inout_mask;
142  _can_driver->select(inout_mask, select_frames, timeout);
143 
144  if (in_mask.read & inout_mask.read) {
145  uavcan::CanFrame frame;
146  uavcan::MonotonicTime time;
147  uavcan::UtcTime utc_time;
148  uavcan::CanIOFlags flags {};
149 
150  int16_t res = _can_driver->getIface(_interface)->receive(frame, time, utc_time, flags);
151 
152  if (res == 1) {
153  uint32_t id = frame.id & uavcan::CanFrame::MaskExtID;
154  uint8_t object_address = id & 0xFF;
155  uint8_t esc_num = uint8_t((id >> 8) & 0xFF);
156 
157  count_msg(id);
158 
159  uint8_t i = 0;
160  uint8_t n = NUM_ESCS;
161 
162  if (esc_num != BROADCAST_NODE_ID) {
163  for (; i < NUM_ESCS; i++) {
164  if (object_address == UPDATE_NODE_ID_OBJ_ADDR) {
165  if (_esc_info[i].mcu_id == be64toh(*((be64_t*) &(frame.data[0])))) {
166  n = i + 1;
167  break;
168  }
169  } else if (_esc_info[i].node_id == esc_num) {
170  n = i + 1;
171  break;
172  }
173  }
174  }
175 
176  while (i < n) {
177  uavcan::CanFrame res_frame;
178 
179  switch (object_address) {
180  case ESC_INFO_OBJ_ADDR: {
181  uint8_t info[5] { 1, 2, 3, 4, 0 };
182 
183  res_frame.dlc = 5;
184  memcpy(res_frame.data, info, 5);
185 
186  break;
187  }
188  case SET_PWM_OBJ_ADDR: {
189  if ((1 << (esc_num - 2) & _mask_received_pwm) && _mask_received_pwm != ((1 << _max_node_id) - 1)) {
190  count_msg(0xFFFFFFF0);
191  _mask_received_pwm = 0;
192  }
193 
194  _mask_received_pwm |= 1 << (esc_num - 2);
195 
196  if (_mask_received_pwm == ((1 << _max_node_id) - 1)) {
197  count_msg(0xFFFFFFFF);
198  _mask_received_pwm = 0;
199  }
200 
201  res_frame.dlc = 0;
202 
203  break;
204  }
205  case UPDATE_NODE_ID_OBJ_ADDR: {
206  if (_esc_info[i].enum_timeout != 0 && _esc_info[i].enum_timeout >= AP_HAL::micros64()) {
207  _esc_info[i].node_id = esc_num;
208  _max_node_id = MAX(_max_node_id, esc_num - 2 + 1);
209  hal.console->printf("Set node ID %d for ESC %d\n", esc_num, i);
210  }
211 
212  _esc_info[i].enum_timeout = 0;
213 
214  res_frame.dlc = 1;
215  memcpy(res_frame.data, &(_esc_info[i].node_id), 1);
216 
217  break;
218  }
219  case START_ENUM_OBJ_ADDR: {
220  _esc_info[i].enum_timeout = AP_HAL::micros64() + be16toh(*((be16_t*) &(frame.data[0]))) * 1000;
221  hal.console->printf("Starting enumeration for ESC %d, timeout %" PRIu64 "\n", i, _esc_info[i].enum_timeout);
222  i++;
223  continue;
224  }
225  case TELEMETRY_OBJ_ADDR: {
226  uint8_t data[7] {};
227  *((be16_t*) &data[0]) = htobe16(get_random16());
228  *((be16_t*) &data[2]) = htobe16(get_random16());
229  *((be16_t*) &data[4]) = htobe16(get_random16());
230  data[6] = uint8_t(float(rand()) / RAND_MAX * 40.0f + 15);
231 
232  res_frame.dlc = 7;
233  memcpy(res_frame.data, data, 7);
234  break;
235  }
236  case VOLTAGE_OBJ_ADDR:
237  case CURRENT_OBJ_ADDR:
238  case RPM_OBJ_ADDR:
239  case TEMPERATURE_OBJ_ADDR:
240  case GET_PWM_INPUT_OBJ_ADDR:
241  case GET_PWM_OUTPUT_OBJ_ADDR:
242  case MCU_ID_OBJ_ADDR:
243  default:
244  // discard frame
245  return;
246  }
247 
248  res_frame.id = (_esc_info[i].node_id << 16) | object_address | uavcan::CanFrame::FlagEFF;
249  timeout = uavcan::MonotonicTime::fromUSec(AP_HAL::millis() + 500);
250  int16_t res2 = _can_driver->getIface(_interface)->send(res_frame, timeout, 0);
251 
252  if (res2 == 1) {
253  i++;
254  }
255  }
256  }
257  }
