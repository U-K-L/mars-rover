82 bool AP_Scripting::init(void) {
83  if (!_enable) {
84  return true;
85  }
86 
87  if (!hal.scheduler->thread_create(FUNCTOR_BIND_MEMBER(&AP_Scripting::thread, void),
88  "Scripting", SCRIPTING_STACK_SIZE, AP_HAL::Scheduler::PRIORITY_SCRIPTING, 0)) {
89  gcs().send_text(MAV_SEVERITY_CRITICAL, "Could not create scripting stack (%d)", SCRIPTING_STACK_SIZE);
90  return false;
91  }
92 
93  return true;
94 }

96 void AP_Scripting::thread(void) {
97  lua_scripts *lua = new lua_scripts(_script_vm_exec_count, _script_heap_size, _debug_level);
98  if (lua == nullptr) {
99  gcs().send_text(MAV_SEVERITY_CRITICAL, "Unable to allocate scripting memory");
100  return;
101  }
102  lua->run();
103 
104  // only reachable if the lua backend has died for any reason
105  gcs().send_text(MAV_SEVERITY_CRITICAL, "Scripting has died");
106 }
