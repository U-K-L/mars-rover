99 /*
100  setup invensense IMU to enable barometer, assuming both IMU and baro
101  on the same i2c bus
102 */
103 bool AP_Baro_LPS2XH::_imu_i2c_init(uint8_t imu_address)
104 {
105  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
106  return false;
107  }
108  // as the baro device is already locked we need to re-use it,
109  // changing its address to match the IMU address
110  uint8_t old_address = _dev->get_bus_address();
111  _dev->set_address(imu_address);
112 
113  _dev->set_retries(4);
114 
115  uint8_t whoami=0;
116  _dev->read_registers(MPUREG_WHOAMI, &whoami, 1);
117  hal.console->printf("IMU: whoami 0x%02x old_address=%02x\n", whoami, old_address);
118 
119  _dev->write_register(MPUREG_FIFO_EN, 0x00);
120  _dev->write_register(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_XGYRO);
121 
122  // wait for sensor to settle
123  hal.scheduler->delay(10);
124 
125  _dev->write_register(MPUREG_INT_PIN_CFG, BIT_BYPASS_EN);
126 
127  _dev->set_address(old_address);
128 
129  _dev->get_semaphore()->give();
130 
131  return true;
132 }

134 bool AP_Baro_LPS2XH::_init()
135 {
136  if (!_dev || !_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
137  return false;
138  }
139  _has_sample = false;
140 
141  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
142 
143  // top bit is for read on SPI
144  _dev->set_read_flag(0x80);
145 
146  if(!_check_whoami()){
147  _dev->get_semaphore()->give();
148  return false;
149  }
150 
151  //init control registers.
152  if(_lps2xh_type == BARO_LPS25H){
153  _dev->write_register(LPS25H_CTRL_REG1_ADDR,0x00); // turn off for config
154  _dev->write_register(LPS25H_CTRL_REG2_ADDR,0x00); //FIFO Disabled
155  _dev->write_register(LPS25H_FIFO_CTRL, 0x01);
156  _dev->write_register(LPS25H_CTRL_REG1_ADDR,0xc0);
157 
158  // request 25Hz update (maximum refresh Rate according to datasheet)
159  CallTime = 40 * AP_USEC_PER_MSEC;
160  }
161  if(_lps2xh_type == BARO_LPS22H){
162  _dev->write_register(LPS22H_CTRL_REG1, 0x00); // turn off for config
163  _dev->write_register(LPS22H_CTRL_REG1, LPS22H_CTRL_REG1_ODR_75HZ|LPS22H_CTRL_REG1_BDU|LPS22H_CTRL_REG1_EN_LPFP|LPS22H_CTRL_REG1_LPFP_CFG);
164  if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI) {
165  _dev->write_register(LPS22H_CTRL_REG2, 0x18); // disable i2c
166  } else {
167  _dev->write_register(LPS22H_CTRL_REG2, 0x10);
168  }
169 
170  // request 75Hz update
171  CallTime = 1000000/75;
172  }
173 
174  _instance = _frontend.register_sensor();
175 
176  _dev->get_semaphore()->give();
177 
178  _dev->register_periodic_callback(CallTime, FUNCTOR_BIND_MEMBER(&AP_Baro_LPS2XH::_timer, void));
179 
180  return true;
181 }

183 //check ID
184 bool AP_Baro_LPS2XH::_check_whoami(void)
185 {
186  uint8_t whoami;
187  if (! _dev->read_registers(REG_ID, &whoami, 1)) {
188  return false;
189  }
190  hal.console->printf("LPS2XH whoami 0x%02x\n", whoami);
191 
192  switch(whoami){
193  case LPS22HB_WHOAMI:
194  _lps2xh_type = BARO_LPS22H;
195  return true;
196  case LPS25HB_WHOAMI:
197  _lps2xh_type = BARO_LPS25H;
198  return true;
199  }
200  return false;
201 }

204 // acumulate a new sensor reading
205 void AP_Baro_LPS2XH::_timer(void)
206 {
207  uint8_t status;
208  // use status to check if data is available
209  if (!_dev->read_registers(STATUS_ADDR, &status, 1)) {
210  return;
211  }
212  if (status & 0x02) {
213  _update_temperature();
214  }
215  if (status & 0x01) {
216  _update_pressure();
217  }
218  _has_sample = true;
219 }

233 // calculate temperature
234 void AP_Baro_LPS2XH::_update_temperature(void)
235 {
236  uint8_t pu8[2];
237  if (!_dev->read_registers(TEMP_OUT_ADDR, pu8, 2)) {
238  return;
239  }
240  int16_t Temp_Reg_s16 = (uint16_t)(pu8[1]<<8) | pu8[0];
241 
242  WITH_SEMAPHORE(_sem);
243 
244  if (_lps2xh_type == BARO_LPS25H) {
245  _temperature = (Temp_Reg_s16 * (1.0/480)) + 42.5;
246  }
247  if (_lps2xh_type == BARO_LPS22H) {
248  _temperature = Temp_Reg_s16 * 0.01;
249  }
250 }

260  int32_t Pressure_mb = Pressure_Reg_s32 * (100.0f / 4096); // scale for pa
261 
262  WITH_SEMAPHORE(_sem);
263  _pressure = Pressure_mb;
