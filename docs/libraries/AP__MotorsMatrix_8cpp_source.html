72 void AP_MotorsMatrix::output_to_motors()
73 {
74  int8_t i;
75 
76  switch (_spool_state) {
77  case SpoolState::SHUT_DOWN: {
78  // no output
79  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
80  if (motor_enabled[i]) {
81  _actuator[i] = 0.0f;
82  }
83  }
84  break;
85  }
86  case SpoolState::GROUND_IDLE:
87  // sends output to motors when armed but not flying
88  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
89  if (motor_enabled[i]) {
90  set_actuator_with_slew(_actuator[i], actuator_spin_up_to_ground_idle());
91  }
92  }
93  break;
94  case SpoolState::SPOOLING_UP:
95  case SpoolState::THROTTLE_UNLIMITED:
96  case SpoolState::SPOOLING_DOWN:
97  // set motor output based on thrust requests
98  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
99  if (motor_enabled[i]) {
100  set_actuator_with_slew(_actuator[i], thrust_to_actuator(_thrust_rpyt_out[i]));
101  }
102  }
103  break;
104  }
105 
106  // convert output to PWM and send to each motor
107  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
108  if (motor_enabled[i]) {
109  rc_write(i, output_to_pwm(_actuator[i]));
110  }
111  }
112 }

133 // output_armed - sends commands to the motors
134 // includes new scaling stability patch
135 void AP_MotorsMatrix::output_armed_stabilizing()
136 {
137  uint8_t i; // general purpose counter
138  float roll_thrust; // roll thrust input value, +/- 1.0
139  float pitch_thrust; // pitch thrust input value, +/- 1.0
140  float yaw_thrust; // yaw thrust input value, +/- 1.0
141  float throttle_thrust; // throttle thrust input value, 0.0 - 1.0
142  float throttle_avg_max; // throttle thrust average maximum value, 0.0 - 1.0
143  float throttle_thrust_max; // throttle thrust maximum value, 0.0 - 1.0
144  float throttle_thrust_best_rpy; // throttle providing maximum roll, pitch and yaw range without climbing
145  float rpy_scale = 1.0f; // this is used to scale the roll, pitch and yaw to fit within the motor limits
146  float yaw_allowed = 1.0f; // amount of yaw we can fit in
147  float thr_adj; // the difference between the pilot's desired throttle and throttle_thrust_best_rpy
148 
149  // apply voltage and air pressure compensation
150  const float compensation_gain = get_compensation_gain(); // compensation for battery voltage and altitude
151  roll_thrust = _roll_in * compensation_gain;
152  pitch_thrust = _pitch_in * compensation_gain;
153  yaw_thrust = _yaw_in * compensation_gain;
154  throttle_thrust = get_throttle() * compensation_gain;
155  throttle_avg_max = _throttle_avg_max * compensation_gain;
156  throttle_thrust_max = _thrust_boost_ratio + (1.0f - _thrust_boost_ratio) * _throttle_thrust_max;
157 
158  // sanity check throttle is above zero and below current limited throttle
159  if (throttle_thrust <= 0.0f) {
160  throttle_thrust = 0.0f;
161  limit.throttle_lower = true;
162  }
163  if (throttle_thrust >= throttle_thrust_max) {
164  throttle_thrust = throttle_thrust_max;
165  limit.throttle_upper = true;
166  }
167 
168  // ensure that throttle_avg_max is between the input throttle and the maximum throttle
169  throttle_avg_max = constrain_float(throttle_avg_max, throttle_thrust, throttle_thrust_max);
170 
171  // calculate throttle that gives most possible room for yaw which is the lower of:
172  // 1. 0.5f - (rpy_low+rpy_high)/2.0 - this would give the maximum possible margin above the highest motor and below the lowest
173  // 2. the higher of:
174  // a) the pilot's throttle input
175  // b) the point _throttle_rpy_mix between the pilot's input throttle and hover-throttle
176  // Situation #2 ensure we never increase the throttle above hover throttle unless the pilot has commanded this.
177  // Situation #2b allows us to raise the throttle above what the pilot commanded but not so far that it would actually cause the copter to rise.
178  // We will choose #1 (the best throttle for yaw control) if that means reducing throttle to the motors (i.e. we favor reducing throttle *because* it provides better yaw control)
179  // We will choose #2 (a mix of pilot and hover throttle) only when the throttle is quite low. We favor reducing throttle instead of better yaw control because the pilot has commanded it
180 
181  // Under the motor lost condition we remove the highest motor output from our calculations and let that motor go greater than 1.0
182  // To ensure control and maximum righting performance Hex and Octo have some optimal settings that should be used
183  // Y6 : MOT_YAW_HEADROOM = 350, ATC_RAT_RLL_IMAX = 1.0, ATC_RAT_PIT_IMAX = 1.0, ATC_RAT_YAW_IMAX = 0.5
184  // Octo-Quad (x8) x : MOT_YAW_HEADROOM = 300, ATC_RAT_RLL_IMAX = 0.375, ATC_RAT_PIT_IMAX = 0.375, ATC_RAT_YAW_IMAX = 0.375
185  // Octo-Quad (x8) + : MOT_YAW_HEADROOM = 300, ATC_RAT_RLL_IMAX = 0.75, ATC_RAT_PIT_IMAX = 0.75, ATC_RAT_YAW_IMAX = 0.375
186  // Usable minimums below may result in attitude offsets when motors are lost. Hex aircraft are only marginal and must be handles with care
187  // Hex : MOT_YAW_HEADROOM = 0, ATC_RAT_RLL_IMAX = 1.0, ATC_RAT_PIT_IMAX = 1.0, ATC_RAT_YAW_IMAX = 0.5
188  // Octo-Quad (x8) x : MOT_YAW_HEADROOM = 300, ATC_RAT_RLL_IMAX = 0.25, ATC_RAT_PIT_IMAX = 0.25, ATC_RAT_YAW_IMAX = 0.25
189  // Octo-Quad (x8) + : MOT_YAW_HEADROOM = 300, ATC_RAT_RLL_IMAX = 0.5, ATC_RAT_PIT_IMAX = 0.5, ATC_RAT_YAW_IMAX = 0.25
190  // Quads cannot make use of motor loss handling because it doesn't have enough degrees of freedom.
191 
192  // calculate amount of yaw we can fit into the throttle range
193  // this is always equal to or less than the requested yaw from the pilot or rate controller
194  float rp_low = 1.0f; // lowest thrust value
195  float rp_high = -1.0f; // highest thrust value
196  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
197  if (motor_enabled[i]) {
198  // calculate the thrust outputs for roll and pitch
199  _thrust_rpyt_out[i] = roll_thrust * _roll_factor[i] + pitch_thrust * _pitch_factor[i];
200  // record lowest roll+pitch command
201  if (_thrust_rpyt_out[i] < rp_low) {
202  rp_low = _thrust_rpyt_out[i];
203  }
204  // record highest roll+pitch command
205  if (_thrust_rpyt_out[i] > rp_high && (!_thrust_boost || i != _motor_lost_index)) {
206  rp_high = _thrust_rpyt_out[i];
207  }
208  }
209  }
210 
211  // include the lost motor scaled by _thrust_boost_ratio
212  if (_thrust_boost && motor_enabled[_motor_lost_index]) {
213  // record highest roll+pitch command
214  if (_thrust_rpyt_out[_motor_lost_index] > rp_high) {
215  rp_high = _thrust_boost_ratio*rp_high + (1.0f-_thrust_boost_ratio)*_thrust_rpyt_out[_motor_lost_index];
216  }
217  }
218 
219  // check for roll and pitch saturation
220  if (rp_high-rp_low > 1.0f || throttle_avg_max < -rp_low) {
221  // Full range is being used by roll and pitch.
222  limit.roll_pitch = true;
223  }
224 
225  // calculate the highest allowed average thrust that will provide maximum control range
226  throttle_thrust_best_rpy = MIN(0.5f, throttle_avg_max);
227 
228  // calculate the maximum yaw control that can be used
229  // todo: make _yaw_headroom 0 to 1
230  yaw_allowed = (float)_yaw_headroom / 1000.0f;
231  yaw_allowed = _thrust_boost_ratio*0.5f + (1.0f - _thrust_boost_ratio) * yaw_allowed;
232  yaw_allowed = MAX(MIN(throttle_thrust_best_rpy+rp_low, 1.0f - (throttle_thrust_best_rpy + rp_high)), yaw_allowed);
233  if (fabsf(yaw_thrust) > yaw_allowed) {
234  // not all commanded yaw can be used
235  yaw_thrust = constrain_float(yaw_thrust, -yaw_allowed, yaw_allowed);
236  limit.yaw = true;
237  }
238 
239  // add yaw control to thrust outputs
240  float rpy_low = 1.0f; // lowest thrust value
241  float rpy_high = -1.0f; // highest thrust value
242  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
243  if (motor_enabled[i]) {
244  _thrust_rpyt_out[i] = _thrust_rpyt_out[i] + yaw_thrust * _yaw_factor[i];
245 
246  // record lowest roll+pitch+yaw command
247  if (_thrust_rpyt_out[i] < rpy_low) {
248  rpy_low = _thrust_rpyt_out[i];
249  }
250  // record highest roll+pitch+yaw command
251  if (_thrust_rpyt_out[i] > rpy_high && (!_thrust_boost || i != _motor_lost_index)) {
252  rpy_high = _thrust_rpyt_out[i];
253  }
254  }
255  }
256  // include the lost motor scaled by _thrust_boost_ratio
257  if (_thrust_boost) {
258  // record highest roll+pitch+yaw command
259  if (_thrust_rpyt_out[_motor_lost_index] > rpy_high && motor_enabled[_motor_lost_index]) {
260  rpy_high = _thrust_boost_ratio*rpy_high + (1.0f-_thrust_boost_ratio)*_thrust_rpyt_out[_motor_lost_index];
261  }
262  }
263 
264  // calculate any scaling needed to make the combined thrust outputs fit within the output range
265  if (rpy_high-rpy_low > 1.0f) {
266  rpy_scale = 1.0f / (rpy_high-rpy_low);
267  }
268  if (is_negative(rpy_low)) {
269  rpy_scale = MIN(rpy_scale, -throttle_avg_max / rpy_low);
270  }
271 
272  // calculate how close the motors can come to the desired throttle
273  rpy_high *= rpy_scale;
274  rpy_low *= rpy_scale;
275  throttle_thrust_best_rpy = -rpy_low;
276  thr_adj = throttle_thrust - throttle_thrust_best_rpy;
277  if (rpy_scale < 1.0f) {
278  // Full range is being used by roll, pitch, and yaw.
279  limit.roll_pitch = true;
280  limit.yaw = true;
281  if (thr_adj > 0.0f) {
282  limit.throttle_upper = true;
283  }
284  thr_adj = 0.0f;
285  } else {
286  if (thr_adj < 0.0f) {
287  // Throttle can't be reduced to desired value
288  // todo: add lower limit flag and ensure it is handled correctly in altitude controller
289  thr_adj = 0.0f;
290  } else if (thr_adj > 1.0f - (throttle_thrust_best_rpy + rpy_high)) {
291  // Throttle can't be increased to desired value
292  thr_adj = 1.0f - (throttle_thrust_best_rpy + rpy_high);
293  limit.throttle_upper = true;
294  }
295  }
296 
297  // add scaled roll, pitch, constrained yaw and throttle for each motor
298  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
299  if (motor_enabled[i]) {
300  _thrust_rpyt_out[i] = throttle_thrust_best_rpy + thr_adj + (rpy_scale * _thrust_rpyt_out[i]);
301  }
302  }
303 
304  // check for failed motor
305  check_for_failed_motor(throttle_thrust_best_rpy + thr_adj);
306 }

308 // check for failed motor
309 // should be run immediately after output_armed_stabilizing
310 // first argument is the sum of:
311 // a) throttle_thrust_best_rpy : throttle level (from 0 to 1) providing maximum roll, pitch and yaw range without climbing
312 // b) thr_adj: the difference between the pilot's desired throttle and throttle_thrust_best_rpy
313 // records filtered motor output values in _thrust_rpyt_out_filt array
314 // sets thrust_balanced to true if motors are balanced, false if a motor failure is detected
315 // sets _motor_lost_index to index of failed motor
316 void AP_MotorsMatrix::check_for_failed_motor(float throttle_thrust_best_plus_adj)
317 {
318  // record filtered and scaled thrust output for motor loss monitoring purposes
319  float alpha = 1.0f / (1.0f + _loop_rate * 0.5f);
320  for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
321  if (motor_enabled[i]) {
322  _thrust_rpyt_out_filt[i] += alpha * (_thrust_rpyt_out[i] - _thrust_rpyt_out_filt[i]);
323  }
324  }
325 
326  float rpyt_high = 0.0f;
327  float rpyt_sum = 0.0f;
328  uint8_t number_motors = 0.0f;
329  for (uint8_t i = 0; i < AP_MOTORS_MAX_NUM_MOTORS; i++) {
330  if (motor_enabled[i]) {
331  number_motors += 1;
332  rpyt_sum += _thrust_rpyt_out_filt[i];
333  // record highest thrust command
334  if (_thrust_rpyt_out_filt[i] > rpyt_high) {
335  rpyt_high = _thrust_rpyt_out_filt[i];
336  // hold motor lost index constant while thrust balance is true
337  if (_thrust_balanced) {
338  _motor_lost_index = i;
339  }
340  }
341  }
342  }
343 
344  float thrust_balance = 1.0f;
345  if (rpyt_sum > 0.1f) {
346  thrust_balance = rpyt_high * number_motors / rpyt_sum;
347  }
348  // ensure thrust balance does not activate for multirotors with less than 6 motors
349  if (number_motors >= 6 && thrust_balance >= 1.5f && _thrust_balanced) {
350  _thrust_balanced = false;
351  }
352  if (thrust_balance <= 1.25f && !_thrust_balanced) {
353  _thrust_balanced = true;
354  }
355 
356  // check to see if thrust boost is using more throttle than _throttle_thrust_max
357  if (_throttle_thrust_max > throttle_thrust_best_plus_adj && rpyt_high < 0.9f && _thrust_balanced) {
358  _thrust_boost = false;
359  }
360 }

494  // DJI quad X order
495  // see https://forum44.djicdn.com/data/attachment/forum/201711/26/172348bppvtt1ot1nrtp5j.jpg

816  yaw_fac = fabsf(_yaw_factor[i]);
