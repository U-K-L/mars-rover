631 // Initialise the filter
632 bool NavEKF3::InitialiseFilter(void)
633 {
634  if (_enable == 0) {
635  return false;
636  }
637  const AP_InertialSensor &ins = AP::ins();
638 
639  imuSampleTime_us = AP_HAL::micros64();
640 
641  // remember expected frame time
642  _frameTimeUsec = 1e6 / ins.get_sample_rate();
643 
644  // expected number of IMU frames per prediction
645  _framesPerPrediction = uint8_t((EKF_TARGET_DT / (_frameTimeUsec * 1.0e-6) + 0.5));
646 
647  if (core == nullptr) {
648 
649  // see if we will be doing logging
650  AP_Logger *logger = AP_Logger::get_singleton();
651  if (logger != nullptr) {
652  logging.enabled = logger->log_replay();
653  }
654 
655  // don't run multiple filters for 1 IMU
656  uint8_t mask = (1U<<ins.get_accel_count())-1;
657  _imuMask.set(_imuMask.get() & mask);
658 
659  // initialise the setup variables
660  for (uint8_t i=0; i<7; i++) {
661  coreSetupRequired[i] = false;
662  coreImuIndex[i] = 0;
663  }
664  num_cores = 0;
665 
666  // count IMUs from mask
667  for (uint8_t i=0; i<7; i++) {
668  if (_imuMask & (1U<<i)) {
669  coreSetupRequired[num_cores] = true;
670  coreImuIndex[num_cores] = i;
671  num_cores++;
672  }
673  }
674 
675  // check if there is enough memory to create the EKF cores
676  if (hal.util->available_memory() < sizeof(NavEKF3_core)*num_cores + 4096) {
677  gcs().send_text(MAV_SEVERITY_CRITICAL, "NavEKF3: not enough memory");
678  _enable.set(0);
679  return false;
680  }
681 
682  //try to allocate from CCM RAM, fallback to Normal RAM if not available or full
683  core = (NavEKF3_core*)hal.util->malloc_type(sizeof(NavEKF3_core)*num_cores, AP_HAL::Util::MEM_FAST);
684  if (core == nullptr) {
685  _enable.set(0);
686  gcs().send_text(MAV_SEVERITY_CRITICAL, "NavEKF3: allocation failed");
687  return false;
688  }
689  for (uint8_t i = 0; i < num_cores; i++) {
690  //Call Constructors
691  new (&core[i]) NavEKF3_core();
692  }
693  }
694 
695  // Set up any cores that have been created
696  // This specifies the IMU to be used and creates the data buffers
697  // If we are unable to set up a core, return false and try again next time the function is called
698  bool core_setup_success = true;
699  for (uint8_t core_index=0; core_index<num_cores; core_index++) {
700  if (coreSetupRequired[core_index]) {
701  coreSetupRequired[core_index] = !core[core_index].setup_core(this, coreImuIndex[core_index], core_index);
702  if (coreSetupRequired[core_index]) {
703  core_setup_success = false;
704  }
705  }
706  }
707  // exit with failure if any cores could not be setup
708  if (!core_setup_success) {
709  return false;
710  }
711 
712  // Set the primary initially to be the lowest index
713  primary = 0;
714 
715  // initialise the cores. We return success only if all cores
716  // initialise successfully
717  bool ret = true;
718  for (uint8_t i=0; i<num_cores; i++) {
719  ret &= core[i].InitialiseFilterBootstrap();
720  }
721 
722  // zero the structs used capture reset events
723  memset(&yaw_reset_data, 0, sizeof(yaw_reset_data));
724  memset((void *)&pos_reset_data, 0, sizeof(pos_reset_data));
725  memset(&pos_down_reset_data, 0, sizeof(pos_down_reset_data));
726 
727  check_log_write();
728  return ret;
729 }

731 // Update Filter States - this should be called whenever new IMU data is available
732 void NavEKF3::UpdateFilter(void)
733 {
734  if (!core) {
735  return;
736  }
737 
738  imuSampleTime_us = AP_HAL::micros64();
739 
740  const AP_InertialSensor &ins = AP::ins();
741 
742  bool statePredictEnabled[num_cores];
743  for (uint8_t i=0; i<num_cores; i++) {
744  // if we have not overrun by more than 3 IMU frames, and we
745  // have already used more than 1/3 of the CPU budget for this
746  // loop then suppress the prediction step. This allows
747  // multiple EKF instances to cooperate on scheduling
748  if (core[i].getFramesSincePredict() < (_framesPerPrediction+3) &&
749  (AP_HAL::micros() - ins.get_last_update_usec()) > _frameTimeUsec/3) {
750  statePredictEnabled[i] = false;
751  } else {
752  statePredictEnabled[i] = true;
753  }
754  core[i].UpdateFilter(statePredictEnabled[i]);
755  }
756 
757  // If the current core selected has a bad error score or is unhealthy, switch to a healthy core with the lowest fault score
758  // Don't start running the check until the primary core has started returned healthy for at least 10 seconds to avoid switching
759  // due to initial alignment fluctuations and race conditions
760  if (!runCoreSelection) {
761  static uint64_t lastUnhealthyTime_us = 0;
762  if (!core[primary].healthy() || lastUnhealthyTime_us == 0) {
763  lastUnhealthyTime_us = imuSampleTime_us;
764  }
765  runCoreSelection = (imuSampleTime_us - lastUnhealthyTime_us) > 1E7;
766  }
767  float primaryErrorScore = core[primary].errorScore();
768  if ((primaryErrorScore > 1.0f || !core[primary].healthy()) && runCoreSelection) {
769  float lowestErrorScore = 0.67f * primaryErrorScore;
770  uint8_t newPrimaryIndex = primary; // index for new primary
771  for (uint8_t coreIndex=0; coreIndex<num_cores; coreIndex++) {
772 
773  if (coreIndex != primary) {
774  // an alternative core is available for selection only if healthy and if states have been updated on this time step
775  bool altCoreAvailable = core[coreIndex].healthy() && statePredictEnabled[coreIndex];
776 
777  // If the primary core is unhealthy and another core is available, then switch now
778  // If the primary core is still healthy,then switching is optional and will only be done if
779  // a core with a significantly lower error score can be found
780  float altErrorScore = core[coreIndex].errorScore();
781  if (altCoreAvailable && (!core[newPrimaryIndex].healthy() || altErrorScore < lowestErrorScore)) {
782  newPrimaryIndex = coreIndex;
783  lowestErrorScore = altErrorScore;
784  }
785  }
786  }
787  // update the yaw and position reset data to capture changes due to the lane switch
788  if (newPrimaryIndex != primary) {
789  updateLaneSwitchYawResetData(newPrimaryIndex, primary);
790  updateLaneSwitchPosResetData(newPrimaryIndex, primary);
791  updateLaneSwitchPosDownResetData(newPrimaryIndex, primary);
792  primary = newPrimaryIndex;
793  lastLaneSwitch_ms = AP_HAL::millis();
794  }
795  }
796 
797  if (primary != 0 && core[0].healthy() && !hal.util->get_soft_armed()) {
798  // when on the ground and disarmed force the first lane. This
799  // avoids us ending with with a lottery for which IMU is used
800  // in each flight. Otherwise the alignment of the timing of
801  // the lane updates with the timing of GPS updates can lead to
802  // a lane other than the first one being used as primary for
803  // some flights. As different IMUs may have quite different
804  // noise characteristics this leads to inconsistent
805  // performance
806  primary = 0;
807  }
808 
809  check_log_write();
810 }

812 /*
813  check if switching lanes will reduce the normalised
814  innovations. This is called when the vehicle code is about to
815  trigger an EKF failsafe, and it would like to avoid that by
816  using a different EKF lane
817 */
818 void NavEKF3::checkLaneSwitch(void)
819 {
820  uint32_t now = AP_HAL::millis();
821  if (lastLaneSwitch_ms != 0 && now - lastLaneSwitch_ms < 5000) {
822  // don't switch twice in 5 seconds
823  return;
824  }
825  float primaryErrorScore = core[primary].errorScore();
826  float lowestErrorScore = primaryErrorScore;
827  uint8_t newPrimaryIndex = primary;
828  for (uint8_t coreIndex=0; coreIndex<num_cores; coreIndex++) {
829  if (coreIndex != primary) {
830  // an alternative core is available for selection only if healthy and if states have been updated on this time step
831  bool altCoreAvailable = core[coreIndex].healthy();
832  float altErrorScore = core[coreIndex].errorScore();
833  if (altCoreAvailable && altErrorScore < lowestErrorScore && altErrorScore < 0.9) {
834  newPrimaryIndex = coreIndex;
835  lowestErrorScore = altErrorScore;
836  }
837  }
838  }
839 
840  // update the yaw and position reset data to capture changes due to the lane switch
841  if (newPrimaryIndex != primary) {
842  updateLaneSwitchYawResetData(newPrimaryIndex, primary);
843  updateLaneSwitchPosResetData(newPrimaryIndex, primary);
844  updateLaneSwitchPosDownResetData(newPrimaryIndex, primary);
845  primary = newPrimaryIndex;
846  lastLaneSwitch_ms = now;
847  gcs().send_text(MAV_SEVERITY_CRITICAL, "NavEKF3: lane switch %u", primary);
848  }
849 }
