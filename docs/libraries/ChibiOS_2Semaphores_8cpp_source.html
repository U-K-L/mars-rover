42 bool Semaphore::take(uint32_t timeout_ms)
43 {
44  mutex_t *mtx = (mutex_t *)_lock;
45  if (timeout_ms == HAL_SEMAPHORE_BLOCK_FOREVER) {
46  chMtxLock(mtx);
47  return true;
48  }
49  if (take_nonblocking()) {
50  return true;
51  }
52  uint64_t start = AP_HAL::micros64();

54  hal.scheduler->delay_microseconds(200);

56  return true;
57  }
58  } while ((AP_HAL::micros64() - start) < timeout_ms*1000);

85 bool Semaphore_Recursive::give()
86 {
87  chSysLock();
88  mutex_t *mtx = (mutex_t *)_lock;
89  osalDbgAssert(count>0, "recursive semaphore");
90  if (count != 0) {

92  if (count == 0) {

96  // priority inheritance

99  }
100  chSysUnlock();
101  return true;
102 }
