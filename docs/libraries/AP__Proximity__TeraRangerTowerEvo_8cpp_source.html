50 // update the state of the sensor
51 void AP_Proximity_TeraRangerTowerEvo::update(void)
52 {
53  if (uart == nullptr) {
54  return;
55  }
56 
57  //initialize the sensor
58  if(_current_init_state != InitState::InitState_Finished)
59  {
60  initialise_modes();

62 
63  // process incoming messages
64  read_sensor_data();
65 
66  // check for timeout and set health status

68  set_status(AP_Proximity::Proximity_NoData);
69  } else {
70  set_status(AP_Proximity::Proximity_Good);

84 void AP_Proximity_TeraRangerTowerEvo::initialise_modes()
85 {
86  if((AP_HAL::millis() - _last_request_sent_ms) < _mode_request_delay) {
87  return;
88  }
89 
90  if (_current_init_state == InitState_Printout) {
91  set_mode(BINARY_MODE, 4);
92  } else if (_current_init_state == InitState_Sequence) {
93  //set tower mode - 4 sensors are triggered at once with 90 deg angle between each sensor
94  set_mode(TOWER_MODE, 4);
95  } else if (_current_init_state == InitState_Rate) {
96  //set update rate of the sensor.
97  set_mode(REFRESH_100_HZ, 5);
98  } else if (_current_init_state == InitState_StreamStart) {
99  set_mode(ACTIVATE_STREAM, 5);
100  }
101 }

109 // check for replies from sensor, returns true if at least one message was processed
110 bool AP_Proximity_TeraRangerTowerEvo::read_sensor_data()
111 {
112  if (uart == nullptr) {
113  return false;
114  }
115 
116  uint16_t message_count = 0;
117  int16_t nbytes = uart->available();
118 
119  if(_current_init_state != InitState_Finished && nbytes == 4) {
120 
121  //Increment _current_init_state only when we receive 4 ack bytes
122  switch (_current_init_state) {
123  case InitState_Printout:
124  _current_init_state = InitState_Sequence;
125  break;
126  case InitState_Sequence:
127  _current_init_state = InitState_Rate;
128  break;
129  case InitState_Rate:
130  _current_init_state = InitState_StreamStart;
131  break;
132  case InitState_StreamStart:
133  _current_init_state = InitState_Finished;
134  break;
135  case InitState_Finished:
136  break;
137  }
138  }
139 
140  while (nbytes-- > 0) {
141  char c = uart->read();
142  if (c == 'T' ) {
143  buffer_count = 0;
144  }
145  buffer[buffer_count++] = c;
146 
147  // we should always read 19 bytes THxxxxxxxxxxxxxxxxMC
148  if (buffer_count >= 20){
149  buffer_count = 0;
150 
151  //check if message has right CRC
152  if (crc_crc8(buffer, 19) == buffer[19]){
153  update_sector_data(0, UINT16_VALUE(buffer[2], buffer[3])); // d1
154  update_sector_data(45, UINT16_VALUE(buffer[4], buffer[5])); // d2
155  update_sector_data(90, UINT16_VALUE(buffer[6], buffer[7])); // d3
156  update_sector_data(135, UINT16_VALUE(buffer[8], buffer[9])); // d4
157  update_sector_data(180, UINT16_VALUE(buffer[10], buffer[11])); // d5
158  update_sector_data(225, UINT16_VALUE(buffer[12], buffer[13])); // d6
159  update_sector_data(270, UINT16_VALUE(buffer[14], buffer[15])); // d7
160  update_sector_data(315, UINT16_VALUE(buffer[16], buffer[17])); // d8
161 
162  message_count++;
163  }
164  }
165  }
166  return (message_count > 0);
167 }

170 void AP_Proximity_TeraRangerTowerEvo::update_sector_data(int16_t angle_deg, uint16_t distance_cm)
171 {
172  uint8_t sector;
173  if (convert_angle_to_sector(angle_deg, sector)) {
174  _angle[sector] = angle_deg;
175  _distance[sector] = ((float) distance_cm) / 1000;
176 
177  //check for target too far, target too close and sensor not connected
178  _distance_valid[sector] = distance_cm != 0xffff && distance_cm != 0x0000 && distance_cm != 0x0001;
179  _last_distance_received_ms = AP_HAL::millis();
180  // update boundary used for avoidance
181  update_boundary_for_sector(sector);
182  }
