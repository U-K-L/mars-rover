178 bool OpticalFlow_Onboard::read(AP_HAL::OpticalFlow::Data_Frame& frame)
179 {
180  bool ret;
181 
182  pthread_mutex_lock(&_mutex);
183  if (!_data_available) {
184  ret = false;
185  goto end;
186  }
187  frame.pixel_flow_x_integral = _pixel_flow_x_integral;
188  frame.pixel_flow_y_integral = _pixel_flow_y_integral;
189  frame.gyro_x_integral = _gyro_x_integral;
190  frame.gyro_y_integral = _gyro_y_integral;
191  frame.delta_time = _integration_timespan;
192  frame.quality = _surface_quality;
193  _integration_timespan = 0;
194  _pixel_flow_x_integral = 0;
195  _pixel_flow_y_integral = 0;
196  _gyro_x_integral = 0;
197  _gyro_y_integral = 0;
198  _data_available = false;
199  ret = true;
200 end:
201  pthread_mutex_unlock(&_mutex);
202  return ret;
203 }

205 void OpticalFlow_Onboard::push_gyro(float gyro_x, float gyro_y, float dt)
206 {
207  GyroSample sample;
208  struct timespec ts;
209 
210  if (!_gyro_ring_buffer) {
211  return;
212  }
213 
214  clock_gettime(CLOCK_MONOTONIC, &ts);
215  _integrated_gyro.x += (gyro_x - _gyro_bias.x) * dt;
216  _integrated_gyro.y += (gyro_y - _gyro_bias.y) * dt;
217  sample.gyro = _integrated_gyro;
218  sample.time_us = 1.0e6 * (ts.tv_sec + (ts.tv_nsec*1.0e-9));
219 
220  _gyro_ring_buffer->push(sample);
221 }

223 void OpticalFlow_Onboard::_get_integrated_gyros(uint64_t timestamp, GyroSample &gyro)
224 {
225  GyroSample integrated_gyro_at_time = {};
226  unsigned int retries = 0;
227 
228  // pop all samples prior to frame time
229  while (_gyro_ring_buffer->pop(integrated_gyro_at_time) &&
230  integrated_gyro_at_time.time_us < timestamp &&
231  retries++ < OPTICAL_FLOW_GYRO_BUFFER_LEN);
232  gyro = integrated_gyro_at_time;
233 }

249 void OpticalFlow_Onboard::_run_optflow()
250 {
251  GyroSample gyro_sample;
252  Vector2f flow_rate;
253  VideoIn::Frame video_frame;
254  uint32_t convert_buffer_size = 0, output_buffer_size = 0;
255  uint32_t crop_left = 0, crop_top = 0;
256  uint32_t shrink_scale = 0, shrink_width = 0, shrink_height = 0;
257  uint32_t shrink_width_offset = 0, shrink_height_offset = 0;
258  uint8_t *convert_buffer = nullptr, *output_buffer = nullptr;
259  uint8_t qual;
260 
261  if (_format == V4L2_PIX_FMT_YUYV) {
262  if (_shrink_by_software || _crop_by_software) {
263  convert_buffer_size = _camera_output_width * _camera_output_height;
264  } else {
265  convert_buffer_size = _width * _height;
266  }
267 
268  convert_buffer = (uint8_t *)calloc(1, convert_buffer_size);
269  if (!convert_buffer) {
270  AP_HAL::panic("OpticalFlow_Onboard: couldn't allocate conversion buffer\n");
271  }
272  }
273 
274  if (_shrink_by_software || _crop_by_software) {
275  output_buffer_size = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH *
276  HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT;
277 
278  output_buffer = (uint8_t *)calloc(1, output_buffer_size);
279  if (!output_buffer) {
280  if (convert_buffer) {
281  free(convert_buffer);
282  }
283 
284  AP_HAL::panic("OpticalFlow_Onboard: couldn't allocate crop buffer\n");
285  }
286  }
287 
288  if (_shrink_by_software) {
289  if (_camera_output_width > _camera_output_height) {
290  shrink_scale = (uint32_t) _camera_output_height /
291  HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT;
292  } else {
293  shrink_scale = (uint32_t) _camera_output_width /
294  HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH;
295  }
296 
297  shrink_width = HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH * shrink_scale;
298  shrink_height = HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT * shrink_scale;
299 
300  shrink_width_offset = (_camera_output_width - shrink_width) / 2;
301  shrink_height_offset = (_camera_output_height - shrink_height) / 2;
302  } else if (_crop_by_software) {
303  crop_left = _camera_output_width / 2 -
304  HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH / 2;
305  crop_top = _camera_output_height / 2 -
306  HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT / 2;
307  }
308 
309  while(true) {
310  /* wait for next frame to come */
311  if (!_videoin->get_frame(video_frame)) {
312  if (convert_buffer) {
313  free(convert_buffer);
314  }
315 
316  if (output_buffer) {
317  free(output_buffer);
318  }
319 
320  AP_HAL::panic("OpticalFlow_Onboard: couldn't get frame\n");
321  }
322 
323  if (_format == V4L2_PIX_FMT_YUYV) {
324  VideoIn::yuyv_to_grey((uint8_t *)video_frame.data,
325  convert_buffer_size * 2, convert_buffer);
326 
327  memset(video_frame.data, 0, convert_buffer_size * 2);
328  memcpy(video_frame.data, convert_buffer, convert_buffer_size);
329  }
330 
331  if (_shrink_by_software) {
332  /* shrink_8bpp() will shrink a selected area using the offsets,
333  * therefore, we don't need the crop. */
334  VideoIn::shrink_8bpp((uint8_t *)video_frame.data, output_buffer,
335  _camera_output_width, _camera_output_height,
336  shrink_width_offset, shrink_width,
337  shrink_height_offset, shrink_height,
338  shrink_scale, shrink_scale);
339  memset(video_frame.data, 0, _camera_output_width * _camera_output_height);
340  memcpy(video_frame.data, output_buffer, output_buffer_size);
341  } else if (_crop_by_software) {
342  VideoIn::crop_8bpp((uint8_t *)video_frame.data, output_buffer,
343  _camera_output_width,
344  crop_left, HAL_OPTFLOW_ONBOARD_OUTPUT_WIDTH,
345  crop_top, HAL_OPTFLOW_ONBOARD_OUTPUT_HEIGHT);
346 
347  memset(video_frame.data, 0, _camera_output_width * _camera_output_height);
348  memcpy(video_frame.data, output_buffer, output_buffer_size);
349  }
350 
351  /* if it is at least the second frame we receive
352  * since we have to compare 2 frames */
353  if (_last_video_frame.data == nullptr) {
354  _last_video_frame = video_frame;
355  continue;
356  }
357 
358  /* read the integrated gyro data */
359  _get_integrated_gyros(video_frame.timestamp, gyro_sample);
360 
361 #ifdef OPTICALFLOW_ONBOARD_RECORD_VIDEO
362  int fd = open(OPTICALFLOW_ONBOARD_VIDEO_FILE, O_CLOEXEC | O_CREAT | O_WRONLY
363  | O_APPEND, S_IRUSR | S_IWUSR | S_IRGRP |
364  S_IWGRP | S_IROTH | S_IWOTH);
365  if (fd != -1) {
366  write(fd, video_frame.data, _sizeimage);
367 #ifdef OPTICALFLOW_ONBOARD_RECORD_METADATAS
368  struct PACKED {
369  uint32_t timestamp;
370  float x;
371  float y;
372  float z;
373  } metas = { video_frame.timestamp, rate_x, rate_y, rate_z};
374  write(fd, &metas, sizeof(metas));
375 #endif
376  close(fd);
377  }
378 #endif
379 
380  /* compute gyro data and video frames
381  * get flow rate to send it to the opticalflow driver
382  */
383  qual = _flow->compute_flow((uint8_t*)_last_video_frame.data,
384  (uint8_t *)video_frame.data,
385  video_frame.timestamp -
386  _last_video_frame.timestamp,
387  &flow_rate.x, &flow_rate.y);
388 
389  /* fill data frame for upper layers */
390  pthread_mutex_lock(&_mutex);
391  _pixel_flow_x_integral += flow_rate.x /
392  HAL_FLOW_PX4_FOCAL_LENGTH_MILLIPX;
393  _pixel_flow_y_integral += flow_rate.y /
394  HAL_FLOW_PX4_FOCAL_LENGTH_MILLIPX;
395  _integration_timespan += video_frame.timestamp -
396  _last_video_frame.timestamp;
397  _gyro_x_integral += (gyro_sample.gyro.x - _last_gyro_rate.x) *
398  (video_frame.timestamp - _last_video_frame.timestamp) /
399  (gyro_sample.time_us - _last_integration_time);
400  _gyro_y_integral += (gyro_sample.gyro.y - _last_gyro_rate.y) /
401  (gyro_sample.time_us - _last_integration_time) *
402  (video_frame.timestamp - _last_video_frame.timestamp);
403  _surface_quality = qual;
404  _data_available = true;
405  pthread_mutex_unlock(&_mutex);
406 
407  /* give the last frame back to the video input driver */
408  _videoin->put_frame(_last_video_frame);
409  _last_integration_time = gyro_sample.time_us;
410  _last_video_frame = video_frame;
411  _last_gyro_rate = gyro_sample.gyro;
412  }
413 
414  if (convert_buffer) {
415  free(convert_buffer);
416  }
417 
418  if (output_buffer) {
419  free(output_buffer);
420  }
421 }
422 #endif
