69 /*
70  update gripper state
71  */
72 void Gripper_Servo::update(const struct sitl_input &input)
73 {
74  const int16_t gripper_pwm = gripper_servo_pin >= 1 ? input.servos[gripper_servo_pin-1] : -1;
75 
76  const uint64_t now = AP_HAL::micros64();
77  const float dt = (now - last_update_us) * 1.0e-6f;
78 
79  // update gripper position
80  if (gripper_pwm < 0) {
81  last_update_us = now;
82  return;
83  }
84  const int16_t diff_pwm = abs(grab_pwm - release_pwm);
85  float position_demand = (gripper_pwm - diff_pwm) * 0.001f;
86  if (gripper_pwm < MIN(grab_pwm, release_pwm) || position_demand > 1.0f) { // never updated
87  position_demand = position;
88  }
89 
90  const float position_max_change = position_slew_rate / 100.0f * dt;
91  position = constrain_float(position_demand, position - position_max_change, position + position_max_change);
92  float jaw_gap;
93  if ((release_pwm < grab_pwm && reverse) || (release_pwm > grab_pwm && !reverse)) {
94  jaw_gap = gap * position;
95  } else {
96  jaw_gap = gap * (1.0f - position);
97  }
98  if (should_report()) {
99  ::fprintf(stderr, "position_demand=%f jaw_gap=%f load=%f\n", position_demand, jaw_gap, load_mass);
100  last_report_us = now;
101  reported_position = position;
102  }
103 
104  if (jaw_gap < 5) {
105  if (altitude <= 0.0f) {
106  load_mass = 1.0f; // attach the load
107  jaw_open = false;
108  }
109  } else if (jaw_gap > 10) {
110  load_mass = 0.0f; // detach the load
111  jaw_open = true;
112  }
113 
114  last_update_us = now;
115 }

125  }
126 
127  return false;
