36 void
37 GCS_MAVLINK::queued_param_send()
38 {
39  if (!initialised) {
40  return;
41  }
42 
43  // send parameter async replies
44  uint8_t async_replies_sent_count = send_parameter_async_replies();
45 
46  const uint32_t tnow = AP_HAL::millis();
47  const uint32_t tstart = AP_HAL::micros();
48 
49  // use at most 30% of bandwidth on parameters. The constant 26 is
50  // 1/(1000 * 1/8 * 0.001 * 0.3)
51  const uint32_t link_bw = _port->bw_in_kilobytes_per_second();
52 
53  uint32_t bytes_allowed = link_bw * (tnow - _queued_parameter_send_time_ms) * 26;
54  const uint16_t size_for_one_param_value_msg = MAVLINK_MSG_ID_PARAM_VALUE_LEN + packet_overhead();
55  if (bytes_allowed < size_for_one_param_value_msg) {
56  bytes_allowed = size_for_one_param_value_msg;
57  }
58  if (bytes_allowed > comm_get_txspace(chan)) {
59  bytes_allowed = comm_get_txspace(chan);
60  }
61  uint32_t count = bytes_allowed / size_for_one_param_value_msg;
62 
63  // when we don't have flow control we really need to keep the
64  // param download very slow, or it tends to stall
65  if (!have_flow_control() && count > 5) {
66  count = 5;
67  }
68  if (async_replies_sent_count >= count) {
69  return;
70  }
71  count -= async_replies_sent_count;
72 
73  if (_queued_parameter == nullptr) {
74  return;
75  }
76 
77  while (count && _queued_parameter != nullptr) {
78  char param_name[AP_MAX_NAME_SIZE];
79  _queued_parameter->copy_name_token(_queued_parameter_token, param_name, sizeof(param_name), true);
80 
81  mavlink_msg_param_value_send(
82  chan,
83  param_name,
84  _queued_parameter->cast_to_float(_queued_parameter_type),
85  mav_param_type(_queued_parameter_type),
86  _queued_parameter_count,
87  _queued_parameter_index);
88 
89  _queued_parameter = AP_Param::next_scalar(&_queued_parameter_token, &_queued_parameter_type);
90  _queued_parameter_index++;
91 
92  if (AP_HAL::micros() - tstart > 1000) {
93  // don't use more than 1ms sending blocks of parameters
94  break;
95  }
96  count--;
97  }
98  _queued_parameter_send_time_ms = tnow;
99 }

123 /*
124  handle a request to change stream rate. Note that copter passes in
125  save==false so we don't want the save to happen when the user connects the
126  ground station.
127  */
128 void GCS_MAVLINK::handle_request_data_stream(mavlink_message_t *msg)
129 {
130  mavlink_request_data_stream_t packet;
131  mavlink_msg_request_data_stream_decode(msg, &packet);
132 
133  int16_t freq = 0; // packet frequency
134 
135  if (packet.start_stop == 0)
136  freq = 0; // stop sending
137  else if (packet.start_stop == 1)
138  freq = packet.req_message_rate; // start sending
139  else
140  return;
141 
142  // if stream_id is still NUM_STREAMS at the end of this switch
143  // block then either we set stream rates for all streams, or we
144  // were asked to set the streamrate for an unrecognised stream
145  streams stream_id = NUM_STREAMS;
146  switch (packet.req_stream_id) {
147  case MAV_DATA_STREAM_ALL:
148  for (uint8_t i=0; i<NUM_STREAMS; i++) {
149  if (i == STREAM_PARAMS) {
150  // don't touch parameter streaming rate; it is
151  // considered "internal".
152  continue;
153  }
154  if (persist_streamrates()) {
155  streamRates[i].set_and_save_ifchanged(freq);
156  } else {
157  streamRates[i].set(freq);
158  }
159  initialise_message_intervals_for_stream((streams)i);
160  }
161  break;
162  case MAV_DATA_STREAM_RAW_SENSORS:
163  stream_id = STREAM_RAW_SENSORS;
164  break;
165  case MAV_DATA_STREAM_EXTENDED_STATUS:
166  stream_id = STREAM_EXTENDED_STATUS;
167  break;
168  case MAV_DATA_STREAM_RC_CHANNELS:
169  stream_id = STREAM_RC_CHANNELS;
170  break;
171  case MAV_DATA_STREAM_RAW_CONTROLLER:
172  stream_id = STREAM_RAW_CONTROLLER;
173  break;
174  case MAV_DATA_STREAM_POSITION:
175  stream_id = STREAM_POSITION;
176  break;
177  case MAV_DATA_STREAM_EXTRA1:
178  stream_id = STREAM_EXTRA1;
179  break;
180  case MAV_DATA_STREAM_EXTRA2:
181  stream_id = STREAM_EXTRA2;
182  break;
183  case MAV_DATA_STREAM_EXTRA3:
184  stream_id = STREAM_EXTRA3;
185  break;
186  }
187 
188  if (stream_id == NUM_STREAMS) {
189  // asked to set rate on unknown stream (or all were set already)
190  return;
191  }
192 
193  AP_Int16 *rate = &streamRates[stream_id];
194 
195  if (rate != nullptr) {
196  if (persist_streamrates()) {
197  rate->set_and_save_ifchanged(freq);
198  } else {
199  rate->set(freq);
200  }
201  initialise_message_intervals_for_stream(stream_id);
202  }
203 }

224 void GCS_MAVLINK::handle_param_request_read(mavlink_message_t *msg)
225 {
226  if (param_requests.space() == 0) {
227  // we can't process this right now, drop it
228  return;
229  }
230 
231  mavlink_param_request_read_t packet;
232  mavlink_msg_param_request_read_decode(msg, &packet);
233 
234  /*
235  we reserve some space for sending parameters if the client ever
236  fails to get a parameter due to lack of space
237  */
238  uint32_t saved_reserve_param_space_start_ms = reserve_param_space_start_ms;
239  reserve_param_space_start_ms = 0; // bypass packet_overhead_chan reservation checking
240  if (!HAVE_PAYLOAD_SPACE(chan, PARAM_VALUE)) {
241  reserve_param_space_start_ms = AP_HAL::millis();
242  } else {
243  reserve_param_space_start_ms = saved_reserve_param_space_start_ms;
244  }
245 
246  struct pending_param_request req;
247  req.chan = chan;
248  req.param_index = packet.param_index;
249  memcpy(req.param_name, packet.param_id, MIN(sizeof(packet.param_id), sizeof(req.param_name)));
250  req.param_name[AP_MAX_NAME_SIZE] = 0;
251 
252  // queue it for processing by io timer
253  param_requests.push(req);
254 
255  // speaking of which, we'd best make sure it is running:
256  if (!param_timer_registered) {
257  param_timer_registered = true;
258  hal.scheduler->register_io_process(FUNCTOR_BIND_MEMBER(&GCS_MAVLINK::param_io_timer, void));
259  }

262 void GCS_MAVLINK::handle_param_set(mavlink_message_t *msg)
263 {
264  mavlink_param_set_t packet;
265  mavlink_msg_param_set_decode(msg, &packet);
266  enum ap_var_type var_type;
267 
268  // set parameter
269  AP_Param *vp;
270  char key[AP_MAX_NAME_SIZE+1];
271  strncpy(key, (char *)packet.param_id, AP_MAX_NAME_SIZE);
272  key[AP_MAX_NAME_SIZE] = 0;
273 
274  // find existing param so we can get the old value
275  vp = AP_Param::find(key, &var_type);
276  if (vp == nullptr) {
277  return;
278  }
279  float old_value = vp->cast_to_float(var_type);
280 
281  // set the value
282  vp->set_float(packet.param_value, var_type);
283 
284  /*
285  we force the save if the value is not equal to the old
286  value. This copes with the use of override values in
287  constructors, such as PID elements. Otherwise a set to the
288  default value which differs from the constructor value doesn't
289  save the change
290  */
291  bool force_save = !is_equal(packet.param_value, old_value);
292 
293  // save the change
294  vp->save(force_save);
295 
296  AP_Logger *logger = AP_Logger::get_singleton();
297  if (logger != nullptr) {
298  logger->Write_Parameter(key, vp->cast_to_float(var_type));
299  }
300 }

327  timer callback for async parameter requests
328  */
329 void GCS_MAVLINK::param_io_timer(void)
330 {
331  struct pending_param_request req;
332 
333  // this is mostly a no-op, but doing this here means we won't
334  // block the main thread counting parameters (~30ms on PH)
335  AP_Param::count_parameters();
336 
337  if (param_replies.space() == 0) {
338  // no room
339  return;
340  }
341 
342  if (!param_requests.pop(req)) {
343  // nothing to do
344  return;
345  }
346 
347  struct pending_param_reply reply;
348  AP_Param *vp;
349 
350  if (req.param_index != -1) {
351  AP_Param::ParamToken token;
352  vp = AP_Param::find_by_index(req.param_index, &reply.p_type, &token);
353  if (vp == nullptr) {
354  return;
355  }
356  vp->copy_name_token(token, reply.param_name, AP_MAX_NAME_SIZE, true);
357  } else {
358  strncpy(reply.param_name, req.param_name, AP_MAX_NAME_SIZE+1);
359  vp = AP_Param::find(req.param_name, &reply.p_type);
360  if (vp == nullptr) {
361  return;
362  }
363  }
364 
365  reply.chan = req.chan;
366  reply.param_name[AP_MAX_NAME_SIZE] = 0;
367  reply.value = vp->cast_to_float(reply.p_type);
368  reply.param_index = req.param_index;
369  reply.count = AP_Param::count_parameters();
370 
371  // queue for transmission
372  param_replies.push(reply);
373 }

375 /*
376  send replies to PARAM_REQUEST_READ
377  */
378 uint8_t GCS_MAVLINK::send_parameter_async_replies()
379 {
380  uint8_t async_replies_sent_count = 0;
381 
382  while (async_replies_sent_count < 5) {
383  if (param_replies.empty()) {
384  // nothing to do
385  return async_replies_sent_count;
386  }
387 
388  /*
389  we reserve some space for sending parameters if the client ever
390  fails to get a parameter due to lack of space
391  */
392  uint32_t saved_reserve_param_space_start_ms = reserve_param_space_start_ms;
393  reserve_param_space_start_ms = 0; // bypass packet_overhead_chan reservation checking
394  if (!HAVE_PAYLOAD_SPACE(chan, PARAM_VALUE)) {
395  reserve_param_space_start_ms = AP_HAL::millis();
396  return async_replies_sent_count;
397  }
398  reserve_param_space_start_ms = saved_reserve_param_space_start_ms;
399 
400  struct pending_param_reply reply;
401  if (!param_replies.pop(reply)) {
402  // internal error
403  return async_replies_sent_count;
404  }
405 
406  mavlink_msg_param_value_send(
407  reply.chan,
408  reply.param_name,
409  reply.value,
410  mav_param_type(reply.p_type),
411  reply.count,
412  reply.param_index);
413 
414  _queued_parameter_send_time_ms = AP_HAL::millis();
415  async_replies_sent_count++;
416  }
417  return async_replies_sent_count;
418 }
