26 template<uint16_t num_bits>
27 class Bitmask {
28 public:
29  Bitmask() :
30  numbits(num_bits),
31  numwords((num_bits+31)/32) {
32  clearall();
33  }
34 
35  Bitmask &operator=(const Bitmask&other) {
36  memcpy(bits, other.bits, sizeof(bits[0])*other.numwords);
37  return *this;
38  }
39 
40  Bitmask(const Bitmask &other) = delete;

42  // set given bitnumber
43  void set(uint16_t bit) {
44  // ignore an invalid bit number
45  if (bit >= numbits) {
46  AP::internalerror().error(AP_InternalError::error_t::bitmask_range);
47  return;
48  }
49  uint16_t word = bit/32;
50  uint8_t ofs = bit & 0x1f;
51  bits[word] |= (1U << ofs);
52  }
53 
54  // set all bits
55  void setall(void) {
56  // set all words to 111... except the last one.
57  for (uint16_t i=0; i<numwords-1; i++) {
58  bits[i] = 0xffffffff;
59  }
60  // set most of the last word to 111.., leaving out-of-range bits to be 0
61  uint16_t num_valid_bits = numbits % 32;
62  bits[numwords-1] = (1 << num_valid_bits) - 1;
63  }
64 
65  // clear given bitnumber
66  void clear(uint16_t bit) {
67  uint16_t word = bit/32;
68  uint8_t ofs = bit & 0x1f;
69  bits[word] &= ~(1U << ofs);
70  }
71 
72  // clear all bits
73  void clearall(void) {
74  memset(bits, 0, numwords*sizeof(bits[0]));
75  }
76 
77  // return true if given bitnumber is set
78  bool get(uint16_t bit) const {
79  uint16_t word = bit/32;
80  uint8_t ofs = bit & 0x1f;
81 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
82  if (bit >= numbits) {
83  AP::internalerror().error(AP_InternalError::error_t::bitmask_range);
84  return false;
85  }
86 #endif
87  return (bits[word] & (1U << ofs)) != 0;
88  }
89 
90  // return true if all bits are clear
91  bool empty(void) const {
92  for (uint16_t i=0; i<numwords; i++) {
93  if (bits[i] != 0) {
94  return false;
95  }
96  }
97  return true;
98  }
99 
100  // return number of bits set
101  uint16_t count() const {
102  uint16_t sum = 0;
103  for (uint16_t i=0; i<numwords; i++) {
104  if (sizeof(bits[i]) <= sizeof(int)) {
105  sum += __builtin_popcount(bits[i]);
106  } else if (sizeof(bits[i]) <= sizeof(long)) {
107  sum += __builtin_popcountl(bits[i]);
108  } else {
109  sum += __builtin_popcountll(bits[i]);
110  }
111  }
112  return sum;
113  }
114 
115  // return first bit set, or -1 if none set
116  int16_t first_set() const {
117  for (uint16_t i=0; i<numwords; i++) {
118  if (bits[i] == 0) {
119  continue;
120  }
121  int fs;
122  if (sizeof(bits[i]) <= sizeof(int)) {
123  fs = __builtin_ffs(bits[i]);
124  } else if (sizeof(bits[i]) <= sizeof(long)) {
125  fs = __builtin_ffsl(bits[i]);
126  } else {
127  fs = __builtin_ffsll(bits[i]);
128  }
129  return i*32 + fs - 1;
130  }
131  return -1;
132  }
133 
134  // return number of bits available
135  uint16_t size() const {
136  return numbits;
137  }
138 
139 private:
140  uint16_t numbits;

142  uint32_t bits[(num_bits+31)/32];
143 };
