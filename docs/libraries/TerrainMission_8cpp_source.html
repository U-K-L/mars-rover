32 /*
33  check that we have fetched all mission terrain data
34  */
35 void AP_Terrain::update_mission_data(void)
36 {
37  if (last_mission_change_ms != mission.last_change_time_ms() ||
38  last_mission_spacing != grid_spacing) {
39  // the mission has changed - start again
40  next_mission_index = 1;
41  next_mission_pos = 0;
42  last_mission_change_ms = mission.last_change_time_ms();
43  last_mission_spacing = grid_spacing;
44  }
45  if (next_mission_index == 0) {
46  // nothing to do
47  return;
48  }
49 
50  uint16_t pending, loaded;
51  get_statistics(pending, loaded);
52  if (pending && AP::gps().status() >= AP_GPS::GPS_OK_FIX_3D) {
53  // wait till we have fully filled the current set of grids
54  return;
55  }
56 
57  // don't do more than 20 waypoints at a time, to prevent too much
58  // CPU usage
59  for (uint8_t i=0; i<20; i++) {
60  // get next mission command
61  AP_Mission::Mission_Command cmd;
62  if (!mission.read_cmd_from_storage(next_mission_index, cmd)) {
63  // nothing more to do
64  next_mission_index = 0;
65  return;
66  }
67 
68  // we only want nav waypoint commands. That should be enough to
69  // prefill the terrain data and makes many things much simpler
70  while ((cmd.id != MAV_CMD_NAV_WAYPOINT &&
71  cmd.id != MAV_CMD_NAV_SPLINE_WAYPOINT) ||
72  (cmd.content.location.lat == 0 && cmd.content.location.lng == 0)) {
73  next_mission_index++;
74  if (!mission.read_cmd_from_storage(next_mission_index, cmd)) {
75  // nothing more to do
76  next_mission_index = 0;
77  next_mission_pos = 0;
78  return;
79  }
80  }
81 
82  // we will fetch 5 points around the waypoint. Four at 10 grid
83  // spacings away at 45, 135, 225 and 315 degrees, and the
84  // point itself
85  if (next_mission_pos != 4) {
86  cmd.content.location.offset_bearing(45+90*next_mission_pos, grid_spacing.get() * 10);
87  }
88 
89  // we have a mission command to check
90  float height;
91  if (!height_amsl(cmd.content.location, height, false)) {
92  // if we can't get data for a mission item then return and
93  // check again next time
94  return;
95  }
96 
97  next_mission_pos++;
98  if (next_mission_pos == 5) {
99 #if TERRAIN_DEBUG
100  hal.console->printf("checked waypoint %u\n", (unsigned)next_mission_index);
101 #endif
102 
103  // move to next waypoint
104  next_mission_index++;
105  next_mission_pos = 0;
106  }
107  }
108 }

110 /*
111  check that we have fetched all rally terrain data
112  */
113 void AP_Terrain::update_rally_data(void)
114 {
115  const AP_Rally *rally = AP::rally();
116  if (rally == nullptr) {
117  return;
118  }
119 
120  if (last_rally_change_ms != rally->last_change_time_ms() ||
121  last_rally_spacing != grid_spacing) {
122  // a rally point has changed - start again
123  next_rally_index = 1;
124  last_rally_change_ms = rally->last_change_time_ms();
125  last_rally_spacing = grid_spacing;
126  }
127  if (next_rally_index == 0) {
128  // nothing to do
129  return;
130  }
131 
132  uint16_t pending, loaded;
133  get_statistics(pending, loaded);
134  if (pending && AP::gps().status() >= AP_GPS::GPS_OK_FIX_3D) {
135  // wait till we have fully filled the current set of grids
136  return;
137  }
138 
139  while (true) {
140  // get next rally point
141  struct RallyLocation rp;
142  if (!rally->get_rally_point_with_index(next_rally_index, rp)) {
143  // nothing more to do
144  next_rally_index = 0;
145  return;
146  }
147 
148  Location loc;
149  loc.lat = rp.lat;
150  loc.lng = rp.lng;
151  float height;
152  if (!height_amsl(loc, height, false)) {
153  // if we can't get data for a rally item then return and
154  // check again next time
155  return;
156  }
157 
158 #if TERRAIN_DEBUG
159  hal.console->printf("checked rally point %u\n", (unsigned)next_rally_index);
160 #endif
161 
162  // move to next rally point
163  next_rally_index++;
164  }
165 }
