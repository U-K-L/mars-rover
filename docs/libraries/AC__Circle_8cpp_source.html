166 // circle's center should already have been set
167 // closest point on the circle will be placed in result
168 // result's altitude (i.e. z) will be set to the circle_center's altitude
169 // if vehicle is at the center of the circle, the edge directly behind vehicle will be returned
170 void AC_Circle::get_closest_point_on_circle(Vector3f &result)
171 {
172  // return center if radius is zero
173  if (_radius <= 0) {
174  result = _center;
175  return;
176  }
177 
178  // get current position
179  Vector3f stopping_point;
180  _pos_control.get_stopping_point_xy(stopping_point);
181 
182  // calc vector from stopping point to circle center
183  Vector2f vec; // vector from circle center to current location
184  vec.x = (stopping_point.x - _center.x);
185  vec.y = (stopping_point.y - _center.y);
186  float dist = norm(vec.x, vec.y);
187 
188  // if current location is exactly at the center of the circle return edge directly behind vehicle
189  if (is_zero(dist)) {
190  result.x = _center.x - _radius * _ahrs.cos_yaw();
191  result.y = _center.y - _radius * _ahrs.sin_yaw();
192  result.z = _center.z;
193  return;
194  }
195 
196  // calculate closest point on edge of circle
197  result.x = _center.x + vec.x / dist * _radius;
198  result.y = _center.y + vec.y / dist * _radius;
199  result.z = _center.z;
200 }

205 void AC_Circle::calc_velocities(bool init_velocity)
206 {
207  // if we are doing a panorama set the circle_angle to the current heading
208  if (_radius <= 0) {
209  _angular_vel_max = ToRad(_rate);
210  _angular_accel = MAX(fabsf(_angular_vel_max),ToRad(AC_CIRCLE_ANGULAR_ACCEL_MIN)); // reach maximum yaw velocity in 1 second
211  }else{
212  // calculate max velocity based on waypoint speed ensuring we do not use more than half our max acceleration for accelerating towards the center of the circle
213  float velocity_max = MIN(_pos_control.get_max_speed_xy(), safe_sqrt(0.5f*_pos_control.get_max_accel_xy()*_radius));
214 
215  // angular_velocity in radians per second
216  _angular_vel_max = velocity_max/_radius;
217  _angular_vel_max = constrain_float(ToRad(_rate),-_angular_vel_max,_angular_vel_max);
