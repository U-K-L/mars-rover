129 {
130  if (_initialized) {
131  debug_uavcan(2, "UAVCAN: init called more than once\n\r");
132  return;
133  }
134 
135  _driver_index = driver_index;
136 
137  AP_HAL::CANManager* can_mgr = hal.can_mgr[driver_index];
138  if (can_mgr == nullptr) {
139  debug_uavcan(2, "UAVCAN: init called for inexisting CAN driver\n\r");
140  return;
141  }
142 
143  if (!can_mgr->is_initialized()) {
144  debug_uavcan(1, "UAVCAN: CAN driver not initialized\n\r");
145  return;
146  }
147 
148  uavcan::ICanDriver* driver = can_mgr->get_driver();
149  if (driver == nullptr) {
150  debug_uavcan(2, "UAVCAN: can't get UAVCAN interface driver\n\r");
151  return;
152  }
153 
154  _node = new uavcan::Node<0>(*driver, SystemClock::instance(), _node_allocator);
155 
156  if (_node == nullptr) {
157  debug_uavcan(1, "UAVCAN: couldn't allocate node\n\r");
158  return;
159  }
160 
161  if (_node->isStarted()) {
162  debug_uavcan(2, "UAVCAN: node was already started?\n\r");
163  return;
164  }
165 
166  uavcan::NodeID self_node_id(_uavcan_node);
167  _node->setNodeID(self_node_id);
168 
169  char ndname[20];
170  snprintf(ndname, sizeof(ndname), "org.ardupilot:%u", driver_index);
171 
172  uavcan::NodeStatusProvider::NodeName name(ndname);
173  _node->setName(name);
174 
175  uavcan::protocol::SoftwareVersion sw_version; // Standard type uavcan.protocol.SoftwareVersion
176  sw_version.major = AP_UAVCAN_SW_VERS_MAJOR;
177  sw_version.minor = AP_UAVCAN_SW_VERS_MINOR;
178  _node->setSoftwareVersion(sw_version);
179 
180  uavcan::protocol::HardwareVersion hw_version; // Standard type uavcan.protocol.HardwareVersion
181 
182  hw_version.major = AP_UAVCAN_HW_VERS_MAJOR;
183  hw_version.minor = AP_UAVCAN_HW_VERS_MINOR;
184 
185  const uint8_t uid_buf_len = hw_version.unique_id.capacity();
186  uint8_t uid_len = uid_buf_len;
187  uint8_t unique_id[uid_buf_len];
188 
189  if (hal.util->get_system_id_unformatted(unique_id, uid_len)) {
190  uavcan::copy(unique_id, unique_id + uid_len, hw_version.unique_id.begin());
191  }
192  _node->setHardwareVersion(hw_version);
193 
194  int start_res = _node->start();
195  if (start_res < 0) {
196  debug_uavcan(1, "UAVCAN: node start problem, error %d\n\r", start_res);

203 #endif
204  // Roundup all subscribers from supported drivers
205  AP_GPS_UAVCAN::subscribe_msgs(this);
206  AP_Compass_UAVCAN::subscribe_msgs(this);
207  AP_Baro_UAVCAN::subscribe_msgs(this);
208  AP_BattMonitor_UAVCAN::subscribe_msgs(this);
209  AP_Airspeed_UAVCAN::subscribe_msgs(this);
210  AP_OpticalFlow_HereFlow::subscribe_msgs(this);
211  AP_RangeFinder_UAVCAN::subscribe_msgs(this);
212

251 void AP_UAVCAN::loop(void)
252 {
253  while (true) {
254  if (!_initialized) {
255  hal.scheduler->delay_microseconds(1000);
256  continue;
257  }
258 
259  const int error = _node->spin(uavcan::MonotonicDuration::fromMSec(1));
260 
261  if (error < 0) {
262  hal.scheduler->delay_microseconds(100);
263  continue;
264  }
265 
266  if (_SRV_armed) {
267  bool sent_servos = false;
268 
269  if (_servo_bm > 0) {
270  // if we have any Servos in bitmask
271  uint32_t now = AP_HAL::micros();
272  const uint32_t servo_period_us = 1000000UL / unsigned(_servo_rate_hz.get());
273  if (now - _SRV_last_send_us >= servo_period_us) {
274  _SRV_last_send_us = now;
275  SRV_send_actuator();
276  sent_servos = true;
277  for (uint8_t i = 0; i < UAVCAN_SRV_NUMBER; i++) {
278  _SRV_conf[i].servo_pending = false;
279  }
280  }
281  }
282 
283  // if we have any ESC's in bitmask
284  if (_esc_bm > 0 && !sent_servos) {
285  SRV_send_esc();
286  }
287 
288  for (uint8_t i = 0; i < UAVCAN_SRV_NUMBER; i++) {
289  _SRV_conf[i].esc_pending = false;
290  }
291  }
292 
293  led_out_send();
294  }
295 }

300 void AP_UAVCAN::SRV_send_actuator(void)
301 {
302  uint8_t starting_servo = 0;
303  bool repeat_send;
304 
305  WITH_SEMAPHORE(SRV_sem);
306 
307  do {
308  repeat_send = false;
309  uavcan::equipment::actuator::ArrayCommand msg;
310 
311  uint8_t i;
312  // UAVCAN can hold maximum of 15 commands in one frame
313  for (i = 0; starting_servo < UAVCAN_SRV_NUMBER && i < 15; starting_servo++) {
314  uavcan::equipment::actuator::Command cmd;
315 
316  /*
317  * Servo output uses a range of 1000-2000 PWM for scaling.
318  * This converts output PWM from [1000:2000] range to [-1:1] range that
319  * is passed to servo as unitless type via UAVCAN.
320  * This approach allows for MIN/TRIM/MAX values to be used fully on
321  * autopilot side and for servo it should have the setup to provide maximum
322  * physically possible throws at [-1:1] limits.
323  */
324 
325  if (_SRV_conf[starting_servo].servo_pending && ((((uint32_t) 1) << starting_servo) & _servo_bm)) {
326  cmd.actuator_id = starting_servo + 1;
327 
328  // TODO: other types
329  cmd.command_type = uavcan::equipment::actuator::Command::COMMAND_TYPE_UNITLESS;
330 
331  // TODO: failsafe, safety
332  cmd.command_value = constrain_float(((float) _SRV_conf[starting_servo].pulse - 1000.0) / 500.0 - 1.0, -1.0, 1.0);
333 
334  msg.commands.push_back(cmd);
335 
336  i++;
337  }
338  }
339 
340  if (i > 0) {
341  act_out_array[_driver_index]->broadcast(msg);
342 
343  if (i == 15) {
344  repeat_send = true;
345  }
346  }
347  } while (repeat_send);
348 }

350 void AP_UAVCAN::SRV_send_esc(void)
351 {
352  static const int cmd_max = uavcan::equipment::esc::RawCommand::FieldTypes::cmd::RawValueType::max();
353  uavcan::equipment::esc::RawCommand esc_msg;
354 
355  uint8_t active_esc_num = 0, max_esc_num = 0;
356  uint8_t k = 0;
357 
358  WITH_SEMAPHORE(SRV_sem);
359 
360  // find out how many esc we have enabled and if they are active at all
361  for (uint8_t i = 0; i < UAVCAN_SRV_NUMBER; i++) {
362  if ((((uint32_t) 1) << i) & _esc_bm) {
363  max_esc_num = i + 1;
364  if (_SRV_conf[i].esc_pending) {
365  active_esc_num++;
366  }
367  }
368  }
369 
370  // if at least one is active (update) we need to send to all
371  if (active_esc_num > 0) {
372  k = 0;
373 
374  for (uint8_t i = 0; i < max_esc_num && k < 20; i++) {
375  if ((((uint32_t) 1) << i) & _esc_bm) {
376  // TODO: ESC negative scaling for reverse thrust and reverse rotation
377  float scaled = cmd_max * (hal.rcout->scale_esc_to_unity(_SRV_conf[i].pulse) + 1.0) / 2.0;
378 
379  scaled = constrain_float(scaled, 0, cmd_max);
380 
381  esc_msg.cmd.push_back(static_cast<int>(scaled));
382  } else {
383  esc_msg.cmd.push_back(static_cast<unsigned>(0));
384  }
385 
386  k++;
387  }
388 
389  esc_raw[_driver_index]->broadcast(esc_msg);
390  }
391 }

412 void AP_UAVCAN::led_out_send()
413 {
414  uint64_t now = AP_HAL::micros64();
415 
416  if ((now - _led_conf.last_update) < LED_DELAY_US) {
417  return;
418  }
419 
420  uavcan::equipment::indication::LightsCommand msg;
421  {
422  WITH_SEMAPHORE(_led_out_sem);
423 
424  if (_led_conf.devices_count == 0) {
425  return;
426  }
427 
428  uavcan::equipment::indication::SingleLightCommand cmd;
429 
430  for (uint8_t i = 0; i < _led_conf.devices_count; i++) {
431  cmd.light_id =_led_conf.devices[i].led_index;
432  cmd.color.red = _led_conf.devices[i].red >> 3;
433  cmd.color.green = _led_conf.devices[i].green >> 2;
434  cmd.color.blue = _led_conf.devices[i].blue >> 3;
435 
436  msg.commands.push_back(cmd);
437  }
438  }
439 
440  rgb_led[_driver_index]->broadcast(msg);
441  _led_conf.last_update = now;
442 }

444 bool AP_UAVCAN::led_write(uint8_t led_index, uint8_t red, uint8_t green, uint8_t blue)
445 {
446  if (_led_conf.devices_count >= AP_UAVCAN_MAX_LED_DEVICES) {
447  return false;
448  }
449 
450  WITH_SEMAPHORE(_led_out_sem);
451 
452  // check if a device instance exists. if so, break so the instance index is remembered
453  uint8_t instance = 0;
454  for (; instance < _led_conf.devices_count; instance++) {
455  if (_led_conf.devices[instance].led_index == led_index) {
456  break;
457  }
458  }
459 
460  // load into the correct instance.
461  // if an existing instance was found in above for loop search,
462  // then instance value is < _led_conf.devices_count.
463  // otherwise a new one was just found so we increment the count.
464  // Either way, the correct instance is the current value of instance
465  _led_conf.devices[instance].led_index = led_index;
466  _led_conf.devices[instance].red = red;
467  _led_conf.devices[instance].green = green;
468  _led_conf.devices[instance].blue = blue;
469 
470  if (instance == _led_conf.devices_count) {
471  _led_conf.devices_count++;
472  }
473 
474  return true;
475 }
