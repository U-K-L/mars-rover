375  return true if progress is made

377 bool UARTDriver::_write_pending_bytes(void)
378 {
379  // write any pending bytes
380  uint32_t available_bytes = _writebuf.available();
381  uint16_t n = available_bytes;
382 
383  if (_packetise && n > 0) {
384  // send on MAVLink packet boundaries if possible
385  n = mavlink_packetise(_writebuf, n);
386  }
387 
388  if (n > 0) {
389  int ret;
390 
391  if (_packetise) {
392  // keep as a single UDP packet
393  uint8_t tmpbuf[n];
394  _writebuf.peekbytes(tmpbuf, n);
395  ret = _write_fd(tmpbuf, n);
396  if (ret > 0)
397  _writebuf.advance(ret);
398  } else {
399  ByteBuffer::IoVec vec[2];
400  const auto n_vec = _writebuf.peekiovec(vec, n);
401  for (int i = 0; i < n_vec; i++) {
402  ret = _write_fd(vec[i].data, (uint16_t)vec[i].len);
403  if (ret < 0) {
404  break;
405  }
406  _writebuf.advance(ret);
407 
408  /* We wrote less than we asked for, stop */
409  if ((unsigned)ret != vec[i].len) {
410  break;
411  }
412  }
413  }
414  }
415

417 }

464 /*
465  return timestamp estimate in microseconds for when the start of
466  a nbytes packet arrived on the uart. This should be treated as a
467  time constraint, not an exact time. It is guaranteed that the
468  packet did not start being received after this time, but it
469  could have been in a system buffer before the returned time.
470 
471  This takes account of the baudrate of the link. For transports
472  that have no baudrate (such as USB) the time estimate may be
473  less accurate.
474 
475  A return value of zero means the HAL does not support this API
476 */
477 uint64_t UARTDriver::receive_time_constraint_us(uint16_t nbytes)
478 {
