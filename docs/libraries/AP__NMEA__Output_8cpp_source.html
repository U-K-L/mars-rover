56 uint8_t AP_NMEA_Output::_nmea_checksum(const char *str)
57 {
58  uint8_t checksum = 0;
59  const uint8_t* bytes = (const uint8_t*) str;
60 
61  for (uint16_t i = 1; str[i]; i++) {
62  checksum ^= bytes[i];
63  }
64 
65  return checksum;

68 void AP_NMEA_Output::update()
69 {
70  uint16_t now = AP_HAL::millis16();
71  uint16_t time_diff = now - _last_sent;
72 
73  // only send at 10Hz
74  if (time_diff < 100) {
75  return;
76  }
77 
78  // get time and date
79  uint64_t time_usec;
80  bool time_valid = AP::rtc().get_utc_usec(time_usec);
81 
82  if (!time_valid) {
83  return;
84  }
85 
86  // not completely accurate, our time includes leap seconds and time_t should be without
87  time_t time_sec = time_usec / 1000000;
88  struct tm* tm = gmtime(&time_sec);
89 
90  // format time string
91  char tstring[11];
92  snprintf(tstring, sizeof(tstring), "%02u%02u%06.3f", tm->tm_hour, tm->tm_min, tm->tm_sec + (time_usec % 1000000) * 1.0e-6);
93 
94  // format date string
95  char dstring[7];
96  snprintf(dstring, sizeof(dstring), "%02u%02u%02u", tm->tm_mday, tm->tm_mon+1, tm->tm_year % 100);
97 
98  AP_AHRS_NavEKF& ahrs = AP::ahrs_navekf();
99 
100  // get location (note: get_position from AHRS always returns true after having GPS position once)
101  Location loc;
102  bool pos_valid = ahrs.get_location(loc);
103 
104  // format latitude
105  char lat_string[13];
106  float deg = fabsf(loc.lat * 1.0e-7f);
107  snprintf(lat_string,
108  sizeof(lat_string),
109  "%02u%08.5f,%c",
110  (unsigned) deg,
111  double((deg - int(deg)) * 60),
112  loc.lat < 0 ? 'S' : 'N');
113 
114  // format longitude
115  char lng_string[14];
116  deg = fabsf(loc.lng * 1.0e-7f);
117  snprintf(lng_string,
118  sizeof(lng_string),
119  "%03u%08.5f,%c",
120  (unsigned) deg,
121  double((deg - int(deg)) * 60),
122  loc.lng < 0 ? 'W' : 'E');
123 
124  // format GGA message
125  char* gga = nullptr;
126  int16_t gga_res = asprintf(&gga,
127  "$GPGGA,%s,%s,%s,%01d,%02d,%04.1f,%07.2f,M,0.0,M,,",
128  tstring,
129  lat_string,
130  lng_string,
131  pos_valid ? 1 : 0,
132  pos_valid ? 6 : 3,
133  2.0,
134  loc.alt * 0.01f);
135  char gga_end[6];
136  snprintf(gga_end, sizeof(gga_end), "*%02X\r\n", (unsigned) _nmea_checksum(gga));
137 
138  // get speed
139  Vector2f speed = ahrs.groundspeed_vector();
140  float speed_knots = norm(speed.x, speed.y) * M_PER_SEC_TO_KNOTS;
141  float heading = wrap_360(degrees(atan2f(speed.x, speed.y)));
142 
143  // format RMC message
144  char* rmc = nullptr;
145  int16_t rmc_res = asprintf(&rmc,
146  "$GPRMC,%s,%c,%s,%s,%.2f,%.2f,%s,,",
147  tstring,
148  pos_valid ? 'A' : 'V',
149  lat_string,
150  lng_string,
151  speed_knots,
152  heading,
153  dstring);
154  char rmc_end[6];
155  snprintf(rmc_end, sizeof(rmc_end), "*%02X\r\n", (unsigned) _nmea_checksum(rmc));
156 
157  // send to all NMEA output ports
158  for (uint8_t i = 0; i < _num_outputs; i++) {
159  if (gga_res != -1) {
160  _uart[i]->write(gga);
161  _uart[i]->write(gga_end);
162  }
163 
164  if (rmc_res != -1) {
165  _uart[i]->write(rmc);
166  _uart[i]->write(rmc_end);
167  }
168  }
169 
170  if (gga_res != -1) {
171  free(gga);
172  }
173 
174  if (rmc_res != -1) {
175  free(rmc);
176  }
177 
178  _last_sent = now;
179 }
