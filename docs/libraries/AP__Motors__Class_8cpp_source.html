101 /*
102  set frequency of a set of channels
103  */
104 void AP_Motors::rc_set_freq(uint32_t mask, uint16_t freq_hz)
105 {
106  if (freq_hz > 50) {
107  _motor_fast_mask |= mask;
108  }
109 
110  mask = rc_map_mask(mask);
111  hal.rcout->set_freq(mask, freq_hz);
112 
113  switch (pwm_type(_pwm_type.get())) {
114  case PWM_TYPE_ONESHOT:
115  if (freq_hz > 50 && mask != 0) {
116  hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_ONESHOT);
117  }
118  break;
119  case PWM_TYPE_ONESHOT125:
120  if (freq_hz > 50 && mask != 0) {
121  hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_ONESHOT125);
122  }
123  break;
124  case PWM_TYPE_BRUSHED:
125  hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_BRUSHED);
126  break;
127  case PWM_TYPE_DSHOT150:
128  hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_DSHOT150);
129  break;
130  case PWM_TYPE_DSHOT300:
131  hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_DSHOT300);
132  break;
133  case PWM_TYPE_DSHOT600:
134  hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_DSHOT600);
135  break;
136  case PWM_TYPE_DSHOT1200:
137  hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_DSHOT1200);
138  break;
139  default:
140  hal.rcout->set_output_mode(mask, AP_HAL::RCOutput::MODE_PWM_NORMAL);
141  break;
142  }
143 }

145 /*
146  map an internal motor mask to real motor mask, accounting for
147  SERVOn_FUNCTION mappings, and allowing for multiple outputs per
148  motor number
149  */
150 uint32_t AP_Motors::rc_map_mask(uint32_t mask) const
151 {
152  uint32_t mask2 = 0;
153  for (uint8_t i=0; i<32; i++) {
154  uint32_t bit = 1UL<<i;
155  if (mask & bit) {
156  SRV_Channel::Aux_servo_function_t function = SRV_Channels::get_motor_function(i);
157  mask2 |= SRV_Channels::get_output_channel_mask(function);
158  }
159  }
160  return mask2;
161 }

173  if (!SRV_Channels::find_channel(function, chan)) {
