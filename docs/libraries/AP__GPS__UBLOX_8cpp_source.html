68 AP_GPS_UBLOX::_request_next_config(void)
69 {
70  // don't request config if we shouldn't configure the GPS
71  if (gps._auto_config == AP_GPS::GPS_AUTO_CONFIG_DISABLE) {
72  return;
73  }
74 
75  // Ensure there is enough space for the largest possible outgoing message
76  if (port->txspace() < (int16_t)(sizeof(struct ubx_header)+sizeof(struct ubx_cfg_nav_rate)+2)) {
77  // not enough space - do it next time
78  return;
79  }
80 
81  if (_unconfigured_messages == CONFIG_RATE_SOL && havePvtMsg) {
82  /*
83  we don't need SOL if we have PVT and TIMEGPS. This is needed
84  as F9P doesn't support the SOL message
85  */
86  _unconfigured_messages &= ~CONFIG_RATE_SOL;
87  }
88 
89  Debug("Unconfigured messages: 0x%x Current message: %u\n", (unsigned)_unconfigured_messages, (unsigned)_next_message);
90 
91  // check AP_GPS_UBLOX.h for the enum that controls the order.
92  // This switch statement isn't maintained against the enum in order to reduce code churn
93  switch (_next_message++) {
94  case STEP_PVT:
95  if(!_request_message_rate(CLASS_NAV, MSG_PVT)) {
96  _next_message--;
97  }
98  break;
99  case STEP_TIMEGPS:
100  if(!_request_message_rate(CLASS_NAV, MSG_TIMEGPS)) {
101  _next_message--;
102  }
103  break;
104  case STEP_PORT:
105  _request_port();
106  break;
107  case STEP_POLL_SVINFO:
108  // not required once we know what generation we are on
109  if(_hardware_generation == UBLOX_UNKNOWN_HARDWARE_GENERATION) {
110  if (!_send_message(CLASS_NAV, MSG_NAV_SVINFO, 0, 0)) {
111  _next_message--;
112  }
113  }
114  break;
115  case STEP_POLL_SBAS:
116  if (gps._sbas_mode != 2) {
117  _send_message(CLASS_CFG, MSG_CFG_SBAS, nullptr, 0);
118  } else {
119  _unconfigured_messages &= ~CONFIG_SBAS;
120  }
121  break;
122  case STEP_POLL_NAV:

124  _next_message--;
125  }
126  break;
127  case STEP_POLL_GNSS:

129  _next_message--;
130  }
131  break;
132  case STEP_POLL_TP5:

135  _next_message--;
136  }
137 #endif
138  break;
139  case STEP_NAV_RATE:

181  if(gps._raw_data == 0) {
182  _unconfigured_messages &= ~CONFIG_RATE_RAW;
183  } else if(!_request_message_rate(CLASS_RXM, MSG_RXM_RAW)) {
184  _next_message--;
185  }
186 #else
187  _unconfigured_messages & = ~CONFIG_RATE_RAW;
188 #endif

353 // Ensure there is enough space for the largest possible outgoing message
354 // Process bytes available from the stream
355 //
356 // The stream is assumed to contain only messages we recognise. If it
357 // contains other messages, and those messages contain the preamble
358 // bytes, it is possible for this code to fail to synchronise to the
359 // stream immediately. Without buffering the entire message and
360 // re-processing it from the top, this is unavoidable. The parser
361 // attempts to avoid this when possible.
362 //
363 bool
364 AP_GPS_UBLOX::read(void)
365 {
366  uint8_t data;
367  int16_t numc;
368  bool parsed = false;
369  uint32_t millis_now = AP_HAL::millis();
370 
371  // walk through the gps configuration at 1 message per second
372  if (millis_now - _last_config_time >= _delay_time) {
373  _request_next_config();
374  _last_config_time = millis_now;
375  if (_unconfigured_messages) { // send the updates faster until fully configured
376  if (!havePvtMsg && (_unconfigured_messages & CONFIG_REQUIRED_INITIAL)) {
377  _delay_time = 300;
378  } else {
379  _delay_time = 750;
380  }
381  } else {
382  _delay_time = 2000;
383  }
384  }
385 
386  if(!_unconfigured_messages && gps._save_config && !_cfg_saved &&
387  _num_cfg_save_tries < 5 && (millis_now - _last_cfg_sent_time) > 5000 &&
388  !hal.util->get_soft_armed()) {
389  //save the configuration sent until now
390  if (gps._save_config == 1 ||
391  (gps._save_config == 2 && _cfg_needs_save)) {
392  _save_cfg();
393  }
394  }
395 
396  numc = port->available();
397  for (int16_t i = 0; i < numc; i++) { // Process bytes received
398 
399  // read the next byte
400  data = port->read();
401 
402  reset:
403  switch(_step) {
404 
405  // Message preamble detection
406  //
407  // If we fail to match any of the expected bytes, we reset
408  // the state machine and re-consider the failed byte as
409  // the first byte of the preamble. This improves our
410  // chances of recovering from a mismatch and makes it less
411  // likely that we will be fooled by the preamble appearing
412  // as data in some other message.
413  //
414  case 1:
415  if (PREAMBLE2 == data) {
416  _step++;
417  break;
418  }
419  _step = 0;
420  Debug("reset %u", __LINE__);
421  FALLTHROUGH;
422  case 0:
423  if(PREAMBLE1 == data)
424  _step++;
425  break;
426 
427  // Message header processing
428  //
429  // We sniff the class and message ID to decide whether we
430  // are going to gather the message bytes or just discard
431  // them.
432  //
433  // We always collect the length so that we can avoid being
434  // fooled by preamble bytes in messages.
435  //
436  case 2:
437  _step++;
438  _class = data;
439  _ck_b = _ck_a = data; // reset the checksum accumulators
440  break;
441  case 3:
442  _step++;
443  _ck_b += (_ck_a += data); // checksum byte
444  _msg_id = data;
445  break;
446  case 4:
447  _step++;
448  _ck_b += (_ck_a += data); // checksum byte
449  _payload_length = data; // payload length low byte
450  break;
451  case 5:
452  _step++;
453  _ck_b += (_ck_a += data); // checksum byte
454 
455  _payload_length += (uint16_t)(data<<8);
456  if (_payload_length > sizeof(_buffer)) {
457  Debug("large payload %u", (unsigned)_payload_length);
458  // assume any payload bigger then what we know about is noise
459  _payload_length = 0;
460  _step = 0;
461  goto reset;
462  }
463  _payload_counter = 0; // prepare to receive payload
464  break;
465 
466  // Receive message data
467  //
468  case 6:
469  _ck_b += (_ck_a += data); // checksum byte
470  if (_payload_counter < sizeof(_buffer)) {
471  _buffer[_payload_counter] = data;
472  }
473  if (++_payload_counter == _payload_length)
474  _step++;
475  break;
476 
477  // Checksum and message processing
478  //
479  case 7:
480  _step++;
481  if (_ck_a != data) {
482  Debug("bad cka %x should be %x", data, _ck_a);
483  _step = 0;
484  goto reset;
485  }
486  break;
487  case 8:
488  _step = 0;
489  if (_ck_b != data) {
490  Debug("bad ckb %x should be %x", data, _ck_b);
491  break; // bad checksum
492  }
493 
494  if (_parse_gps()) {
495  parsed = true;
496  }
497  break;
498  }
499  }
500  return parsed;
501 }

627 bool
628 AP_GPS_UBLOX::_parse_gps(void)
629 {
630  if (_class == CLASS_ACK) {
631  Debug("ACK %u", (unsigned)_msg_id);
632 
633  if(_msg_id == MSG_ACK_ACK) {
634  switch(_buffer.ack.clsID) {
635  case CLASS_CFG:
636  switch(_buffer.ack.msgID) {
637  case MSG_CFG_CFG:
638  _cfg_saved = true;
639  _cfg_needs_save = false;
640  break;
641  case MSG_CFG_GNSS:
642  _unconfigured_messages &= ~CONFIG_GNSS;
643  break;
644  case MSG_CFG_MSG:
645  // There is no way to know what MSG config was ack'ed, assume it was the last
646  // one requested. To verify it rerequest the last config we sent. If we miss
647  // the actual ack we will catch it next time through the poll loop, but that
648  // will be a good chunk of time later.
649  break;
650  case MSG_CFG_NAV_SETTINGS:
