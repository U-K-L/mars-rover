63 void AP_Tuning::check_selector_switch(void)
64 {
65  if (selector == 0) {
66  // no selector switch enabled
67  return;
68  }
69  if (!rc().has_valid_input()) {
70  selector_start_ms = 0;
71  return;
72  }
73  RC_Channel *selchan = rc().channel(selector-1);
74  if (selchan == nullptr) {
75  return;
76  }
77  uint16_t selector_in = selchan->get_radio_in();
78  if (selector_in >= 1700) {
79  // high selector
80  if (selector_start_ms == 0) {
81  selector_start_ms = AP_HAL::millis();
82  }
83  uint32_t hold_time = AP_HAL::millis() - selector_start_ms;
84  if (hold_time > 5000 && changed) {
85  // save tune
86  save_parameters();
87  re_center();
88  gcs().send_text(MAV_SEVERITY_INFO, "Tuning: Saved");
89  AP_Notify::events.tune_save = 1;
90  changed = false;
91  need_revert = 0;
92  }
93  } else if (selector_in <= 1300) {
94  // low selector
95  if (selector_start_ms != 0) {
96  uint32_t hold_time = AP_HAL::millis() - selector_start_ms;
97  if (hold_time < 200) {
98  // debounce!
99  } else if (hold_time < 2000) {
100  // re-center the value
101  re_center();
102  gcs().send_text(MAV_SEVERITY_INFO, "Tuning: recentered %s", get_tuning_name(current_parm));
103  } else if (hold_time < 5000) {
104  // change parameter
105  next_parameter();
106  }
107  selector_start_ms = 0;
108  }
109  }
