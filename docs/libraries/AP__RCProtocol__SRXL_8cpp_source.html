75 int AP_RCProtocol_SRXL::srxl_channels_get_v1v2(uint16_t max_values, uint8_t *num_values, uint16_t *values, bool *failsafe_state)
76 {
77  uint8_t loop;
78  uint32_t channel_raw_value;
79 
80  *num_values = (uint8_t)((frame_len_full - 3U)/2U);
81  *failsafe_state = 0U; /* this protocol version does not support failsafe information */
82 
83  /* get data channel data from frame */
84  for (loop=0U; loop < *num_values; loop++) {
85  channel_raw_value = ((((uint32_t)buffer[loop*2U+1U])& 0x0000000FU) << 8U) | ((uint32_t)(buffer[loop*2U+2U])); /* get 12bit channel raw value from srxl datastream (mask out unused bits with 0x0000000F) */
86  channels[loop] = (uint16_t)(((channel_raw_value * (uint32_t)1400U) >> 12U) + (uint32_t)800U); /* convert raw value to servo/esc signal pulsewidth in us */
87  }
88 
89  /* provide channel data to FMU */
90  if ((uint16_t)*num_values > max_values) {
91  *num_values = (uint8_t)max_values;
92  }
93  memcpy(values, channels, (*num_values)*2);
94 
95  return 0; /* for srxl protocol version 1 and 2 it is not expected, that any error happen during decode process */
96 }

120 int AP_RCProtocol_SRXL::srxl_channels_get_v5(uint16_t max_values, uint8_t *num_values, uint16_t *values, bool *failsafe_state)
121 {
122  // up to 7 channel values per packet. Each channel value is 16
123  // bits, with 11 bits of data and 4 bits of channel number. The
124  // top bit indicates a special X-Plus channel
125  for (uint8_t i=0; i<7; i++) {
126  uint16_t b = buffer[i*2+2] << 8 | buffer[i*2+3];
127  uint16_t c = b >> 11; // channel number
128  int32_t v = b & 0x7FF;
129  if (b & 0x8000) {
130  continue;
131  }
132  if (c == 12) {
133  // special handling for channel 12
134  // see http://www.deviationtx.com/forum/protocol-development/2088-18-channels-for-dsm2-dsmx?start=40
135  //printf("c12: 0x%x %02x %02x\n", (unsigned)(b>>9), (unsigned)buffer[0], (unsigned)buffer[1]);
136  v = (b & 0x1FF) << 2;
137  c = 10 + ((b >> 9) & 0x7);
138  if (buffer[1] & 1) {
139  c += 4;
140  }
141  } else if (c > 12) {
142  // invalid
143  v = 0;
144  }
145 
146  // if channel number if greater than 16 then it is a X-Plus
147  // channel. We don't yet know how to decode those. There is some information here:
148  // http://www.deviationtx.com/forum/protocol-development/2088-18-channels-for-dsm2-dsmx?start=40
149  // but we really need some sample data to confirm
150  if (c < SRXL_MAX_CHANNELS) {
151  v = (((v - 0x400) * 500) / 876) + 1500;
152  channels[c] = v;
153  if (c >= max_channels) {
154  max_channels = c+1;
155  }
156  }
157 
158  //printf("%u:%u ", (unsigned)c, (unsigned)v);
159  }
160  //printf("\n");
161 
162  *num_values = max_channels;
163  if (*num_values > max_values) {
164  *num_values = max_values;
165  }
166  memcpy(values, channels, (*num_values)*2);
167 
168  // check failsafe bit, this goes low when connection to the
169  // transmitter is lost
170  *failsafe_state = ((buffer[1] & 2) == 0);
171 
172  // success
173  return 0;
174 }

176 void AP_RCProtocol_SRXL::_process_byte(uint32_t timestamp_us, uint8_t byte)
177 {
178  /*----------------------------------------distinguish different srxl variants at the beginning of each frame---------------------------------------------- */
179  /* Check if we have a new begin of a frame --> indicators: Time gap in datastream + SRXL header 0xA<VARIANT>*/
180  if ((timestamp_us - last_data_us) >= SRXL_MIN_FRAMESPACE_US) {
181  /* Now detect SRXL variant based on header */
182  switch (byte) {
183  case SRXL_HEADER_V1:
184  frame_len_full = SRXL_FRAMELEN_V1;
185  frame_header = SRXL_HEADER_V1;
186  decode_state = STATE_NEW;
187  break;
188  case SRXL_HEADER_V2:
189  frame_len_full = SRXL_FRAMELEN_V2;
190  frame_header = SRXL_HEADER_V2;
191  decode_state = STATE_NEW;
192  break;
193  case SRXL_HEADER_V5:
194  frame_len_full = SRXL_FRAMELEN_V5;
195  frame_header = SRXL_HEADER_V5;
196  decode_state = STATE_NEW;
197  break;
198  default:
199  frame_len_full = 0U;
200  frame_header = SRXL_HEADER_NOT_IMPL;
201  decode_state = STATE_IDLE;
202  buflen = 0;
203  return; /* protocol version not implemented --> no channel data --> unknown packet */
204  }
205  }
206 
207 
208  /*--------------------------------------------collect all data from stream and decode-------------------------------------------------------*/
209  switch (decode_state) {
210  case STATE_NEW: /* buffer header byte and prepare for frame reception and decoding */
211  buffer[0U]=byte;
212  crc_fmu = crc_xmodem_update(0U,byte);
213  buflen = 1U;
214  decode_state_next = STATE_COLLECT;
215  break;
216 
217  case STATE_COLLECT: /* receive all bytes. After reception decode frame and provide rc channel information to FMU */
218  if (buflen >= frame_len_full) {
219  // a logic bug in the state machine, this shouldn't happen
220  decode_state = STATE_IDLE;
221  buflen = 0;
222  frame_len_full = 0;
223  frame_header = SRXL_HEADER_NOT_IMPL;
224  return;
225  }
226  buffer[buflen] = byte;
227  buflen++;
228  /* CRC not over last 2 frame bytes as these bytes inhabitate the crc */
229  if (buflen <= (frame_len_full-2)) {
230  crc_fmu = crc_xmodem_update(crc_fmu,byte);
231  }
232  if (buflen == frame_len_full) {
233  /* CRC check here */
234  crc_receiver = ((uint16_t)buffer[buflen-2] << 8U) | ((uint16_t)buffer[buflen-1]);
235  if (crc_receiver == crc_fmu) {
236  /* at this point buffer contains all frame data and crc is valid --> extract channel info according to SRXL variant */
237  const uint8_t max_values = MIN((unsigned)SRXL_MAX_CHANNELS,(unsigned)MAX_RCIN_CHANNELS);
238  uint16_t values[max_values];
239  uint8_t num_values;
240  bool failsafe_state;
241  switch (frame_header) {
242  case SRXL_HEADER_V1:
243  srxl_channels_get_v1v2(max_values, &num_values, values, &failsafe_state);
244  add_input(num_values, values, failsafe_state);
245  break;
246  case SRXL_HEADER_V2:
247  srxl_channels_get_v1v2(max_values, &num_values, values, &failsafe_state);
248  add_input(num_values, values, failsafe_state);
249  break;
250  case SRXL_HEADER_V5:
251  srxl_channels_get_v5(max_values, &num_values, values, &failsafe_state);
252  add_input(num_values, values, failsafe_state);
253  break;
254  default:
255  break;
256  }
257  }
258  decode_state_next = STATE_IDLE; /* frame data buffering and decoding finished --> statemachine not in use until new header drops is */
259  } else {
260  /* frame not completely received --> frame data buffering still ongoing */
261  decode_state_next = STATE_COLLECT;
262  }
263  break;
264 
265  default:
266  break;
267  } /* switch (decode_state) */
268 
269  decode_state = decode_state_next;
270  last_data_us = timestamp_us;
271 }
