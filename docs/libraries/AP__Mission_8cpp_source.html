399  _do_cmd.index = AP_MISSION_CMD_INDEX_NONE;

401  _flags.do_cmd_all_done = false;
402

504 bool AP_Mission::read_cmd_from_storage(uint16_t index, Mission_Command& cmd) const
505 {
506  WITH_SEMAPHORE(_rsem);
507 
508  // exit immediately if index is beyond last command but we always let cmd #0 (i.e. home) be read
509  if (index >= (unsigned)_cmd_total && index != 0) {
510  return false;
511  }
512 
513  // special handling for command #0 which is home
514  if (index == 0) {
515  cmd.index = 0;
516  cmd.id = MAV_CMD_NAV_WAYPOINT;
517  cmd.p1 = 0;
518  cmd.content.location = AP::ahrs().get_home();
519  }else{
520  // Find out proper location in memory by using the start_byte position + the index
521  // we can load a command, we don't process it yet
522  // read WP position
523  uint16_t pos_in_storage = 4 + (index * AP_MISSION_EEPROM_COMMAND_SIZE);
524 
525  PackedContent packed_content {};
526 
527  uint8_t b1 = _storage.read_byte(pos_in_storage);
528  if (b1 == 0) {
529  cmd.id = _storage.read_uint16(pos_in_storage+1);
530  cmd.p1 = _storage.read_uint16(pos_in_storage+3);
531  _storage.read_block(packed_content.bytes, pos_in_storage+5, 10);
532  } else {
533  cmd.id = b1;
534  cmd.p1 = _storage.read_uint16(pos_in_storage+1);
535  _storage.read_block(packed_content.bytes, pos_in_storage+3, 12);
536  }
537 
538  if (stored_in_location(cmd.id)) {
539  // Location is not PACKED; field-wise copy it:
540  cmd.content.location.relative_alt = packed_content.location.flags.relative_alt;
541  cmd.content.location.loiter_ccw = packed_content.location.flags.loiter_ccw;
542  cmd.content.location.terrain_alt = packed_content.location.flags.terrain_alt;
543  cmd.content.location.origin_alt = packed_content.location.flags.origin_alt;
544  cmd.content.location.loiter_xtrack = packed_content.location.flags.loiter_xtrack;
545  cmd.content.location.alt = packed_content.location.alt;
546  cmd.content.location.lat = packed_content.location.lat;
547  cmd.content.location.lng = packed_content.location.lng;

550  static_assert(sizeof(cmd.content) >= 12,

552  // (void *) cast to specify gcc that we know that we are copy byte into a non trivial type and leaving 4 bytes untouched

554  }
555 
556  // set command's index to it's position in eeprom
557  cmd.index = index;
558  }
