53 void AP_RCProtocol::process_pulse(uint32_t width_s0, uint32_t width_s1)
54 {
55  uint32_t now = AP_HAL::millis();
56  bool searching = (now - _last_input_ms >= 200);
57  if (_detected_protocol != AP_RCProtocol::NONE && _detected_with_bytes && !searching) {
58  // we're using byte inputs, discard pulses
59  return;
60  }
61  // first try current protocol
62  if (_detected_protocol != AP_RCProtocol::NONE && !searching) {
63  backend[_detected_protocol]->process_pulse(width_s0, width_s1);
64  if (backend[_detected_protocol]->new_input()) {
65  _new_input = true;
66  _last_input_ms = now;
67  }
68  return;
69  }
70 
71  // otherwise scan all protocols
72  for (uint8_t i = 0; i < AP_RCProtocol::NONE; i++) {
73  if (_disabled_for_pulses & (1U << i)) {
74  // this protocol is disabled for pulse input
75  continue;
76  }
77  if (backend[i] != nullptr) {
78  uint32_t frame_count = backend[i]->get_rc_frame_count();
79  uint32_t input_count = backend[i]->get_rc_input_count();
80  backend[i]->process_pulse(width_s0, width_s1);
81  if (frame_count != backend[i]->get_rc_frame_count()) {
82  _good_frames[i]++;
83  if (requires_3_frames((rcprotocol_t)i) && _good_frames[i] < 3) {
84  continue;
85  }
86  _new_input = (input_count != backend[i]->get_rc_input_count());
87  _detected_protocol = (enum AP_RCProtocol::rcprotocol_t)i;
88  memset(_good_frames, 0, sizeof(_good_frames));
89  _last_input_ms = now;
90  _detected_with_bytes = false;
91  break;
92  }
93  }
94  }
95 }

97 /*
98  process an array of pulses. n must be even
99  */
100 void AP_RCProtocol::process_pulse_list(const uint32_t *widths, uint16_t n, bool need_swap)
101 {
102  if (n & 1) {
103  return;
104  }
105  while (n) {
106  uint32_t widths0 = widths[0];
107  uint32_t widths1 = widths[1];
108  if (need_swap) {
109  uint32_t tmp = widths1;
110  widths1 = widths0;
111  widths0 = tmp;
112  }
113  widths1 -= widths0;
114  process_pulse(widths0, widths1);
115  widths += 2;
116  n -= 2;
117  }
118 }
