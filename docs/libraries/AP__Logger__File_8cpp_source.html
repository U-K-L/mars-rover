242 // find_oldest_log - find oldest log in _log_directory
243 // returns 0 if no log was found
244 uint16_t AP_Logger_File::find_oldest_log()
245 {
246  if (_cached_oldest_log != 0) {
247  return _cached_oldest_log;
248  }
249 
250  uint16_t last_log_num = find_last_log();
251  if (last_log_num == 0) {
252  return 0;
253  }
254 
255  uint16_t current_oldest_log = 0; // 0 is invalid
256 
257  // We could count up to find_last_log(), but if people start
258  // relying on the min_avail_space_percent feature we could end up
259  // doing a *lot* of asprintf()s and stat()s
260  EXPECT_DELAY_MS(3000);
261  DIR *d = opendir(_log_directory);
262  if (d == nullptr) {
263  // SD card may have died? On linux someone may have rm-rf-d
264  return 0;
265  }
266 
267  // we only remove files which look like xxx.BIN
268  EXPECT_DELAY_MS(3000);
269  for (struct dirent *de=readdir(d); de; de=readdir(d)) {
270  EXPECT_DELAY_MS(3000);
271  uint8_t length = strlen(de->d_name);
272  if (length < 5) {
273  // not long enough for \d+[.]BIN
274  continue;
275  }
276  if (strncmp(&de->d_name[length-4], ".BIN", 4)) {
277  // doesn't end in .BIN
278  continue;
279  }
280 
281  uint16_t thisnum = strtoul(de->d_name, nullptr, 10);
282  if (thisnum > MAX_LOG_FILES) {
283  // ignore files above our official maximum...
284  continue;
285  }
286  if (current_oldest_log == 0) {
287  current_oldest_log = thisnum;
288  } else {
289  if (current_oldest_log <= last_log_num) {
290  if (thisnum > last_log_num) {
291  current_oldest_log = thisnum;
292  } else if (thisnum < current_oldest_log) {
293  current_oldest_log = thisnum;
294  }
295  } else { // current_oldest_log > last_log_num
296  if (thisnum > last_log_num) {
297  if (thisnum < current_oldest_log) {
298  current_oldest_log = thisnum;
299  }
300  }
301  }
302  }
303  }
304  closedir(d);
305  _cached_oldest_log = current_oldest_log;
306 
307  return current_oldest_log;
308 }

310 void AP_Logger_File::Prep_MinSpace()
311 {
312  if (hal.util->was_watchdog_reset()) {
313  // don't clear space if watchdog reset, it takes too long
314  return;
315  }
316  const uint16_t first_log_to_remove = find_oldest_log();
317  if (first_log_to_remove == 0) {
318  // no files to remove
319  return;
320  }
321 
322  _cached_oldest_log = 0;
323 
324  uint16_t log_to_remove = first_log_to_remove;
325 
326  uint16_t count = 0;
327  do {
328  float avail = avail_space_percent();
329  if (is_equal(avail, -1.0f)) {
330  break;
331  }
332  if (avail >= min_avail_space_percent) {
333  break;
334  }
335  if (count++ > MAX_LOG_FILES+10) {
336  // *way* too many deletions going on here. Possible internal error.
337  AP::internalerror().error(AP_InternalError::error_t::logger_too_many_deletions);
338  break;
339  }
340  char *filename_to_remove = _log_file_name(log_to_remove);
341  if (filename_to_remove == nullptr) {
342  AP::internalerror().error(AP_InternalError::error_t::logger_bad_getfilename);
343  break;
344  }
345  if (file_exists(filename_to_remove)) {
346  hal.console->printf("Removing (%s) for minimum-space requirements (%.2f%% < %.0f%%)\n",
347  filename_to_remove, (double)avail, (double)min_avail_space_percent);
348  EXPECT_DELAY_MS(2000);
349  if (unlink(filename_to_remove) == -1) {
350  hal.console->printf("Failed to remove %s: %s\n", filename_to_remove, strerror(errno));
351  free(filename_to_remove);
352  if (errno == ENOENT) {
353  // corruption - should always have a continuous
354  // sequence of files... however, there may be still
355  // files out there, so keep going.
356  } else {
357  break;
358  }
359  } else {
360  free(filename_to_remove);
361  }
362  }
363  log_to_remove++;
364  if (log_to_remove > MAX_LOG_FILES) {
365  log_to_remove = 1;
366  }
367  } while (log_to_remove != first_log_to_remove);
368 }

512 /* Write a block of data at current offset */
513 bool AP_Logger_File::_WritePrioritisedBlock(const void *pBuffer, uint16_t size, bool is_critical)
514 {
515  if (! WriteBlockCheckStartupMessages()) {
516  _dropped++;
517  return false;
518  }
519 
520  if (!semaphore.take(1)) {
521  return false;
522  }
523 
524  uint32_t space = _writebuf.space();
525 
526  if (_writing_startup_messages &&
527  _startup_messagewriter->fmt_done()) {
528  // the state machine has called us, and it has finished
529  // writing format messages out. It can always get back to us
530  // with more messages later, so let's leave room for other
531  // things:
532  const uint32_t now = AP_HAL::millis();
533  const bool must_dribble = (now - last_messagewrite_message_sent) > 100;
534  if (!must_dribble &&
535  space < non_messagewriter_message_reserved_space()) {
536  // this message isn't dropped, it will be sent again...
537  semaphore.give();
538  return false;
539  }
540  last_messagewrite_message_sent = now;
541  } else {
542  // we reserve some amount of space for critical messages:
543  if (!is_critical && space < critical_message_reserved_space()) {
544  _dropped++;
545  semaphore.give();
546  return false;
547  }
548  }
549 
550  // if no room for entire message - drop it:
551  if (space < size) {
552  hal.util->perf_count(_perf_overruns);
553  _dropped++;
554  semaphore.give();
555  return false;
556  }
557 
558  _writebuf.write((uint8_t*)pBuffer, size);
559  df_stats_gather(size);
560  semaphore.give();
561  return true;
562 }

732  (starting at around 350k into a file) NuttX will get the
733  wrong offset for sequential reads. The offset it gets is
734  typically 128k earlier than it should be. It turns out that
735  calling lseek() with 0 offset and SEEK_CUR works around the
736  bug. We can remove this once we find the real bug.
737  */

740  if (seek_current == (off_t)-1) {
741  close(_read_fd);
742  _read_fd = -1;
743  return -1;
744  }
745  if (seek_current != (off_t)_read_offset) {
746  if (::lseek(_read_fd, _read_offset, SEEK_SET) == (off_t)-1) {
747  close(_read_fd);
748  _read_fd = -1;
749  return -1;
750  }
751  }
752  }
753 
754  if (ofs != _read_offset) {
755  if (::lseek(_read_fd, ofs, SEEK_SET) == (off_t)-1) {
756  close(_read_fd);
757  _read_fd = -1;
758  return -1;
759  }
760  _read_offset = ofs;
761  }
762  int16_t ret = (int16_t)::read(_read_fd, data, len);
763  if (ret > 0) {
764  _read_offset += ret;
765  }
766  return ret;
767 }

972 void AP_Logger_File::_io_timer(void)
973 {
974  uint32_t tnow = AP_HAL::millis();
975  _io_timer_heartbeat = tnow;
976  if (_write_fd == -1 || !_initialised || _open_error) {
977  return;
978  }
979 
980  uint32_t nbytes = _writebuf.available();
981  if (nbytes == 0) {
982  return;
983  }
984  if (nbytes < _writebuf_chunk &&
985  tnow - _last_write_time < 2000UL) {
986  // write in _writebuf_chunk-sized chunks, but always write at
987  // least once per 2 seconds if data is available
988  return;
989  }
990  if (tnow - _free_space_last_check_time > _free_space_check_interval) {
991  _free_space_last_check_time = tnow;
992  last_io_operation = "disk_space_avail";
993  if (disk_space_avail() < _free_space_min_avail) {
994  hal.console->printf("Out of space for logging\n");
995  stop_logging();
996  _open_error = true; // prevent logging starting again
997  last_io_operation = "";
998  return;
999  }
1000  last_io_operation = "";
1001  }
1002 
1003  hal.util->perf_begin(_perf_write);
1004 
1005  _last_write_time = tnow;
1006  if (nbytes > _writebuf_chunk) {
1007  // be kind to the filesystem layer
1008  nbytes = _writebuf_chunk;
1009  }
1010 
1011  uint32_t size;
1012  const uint8_t *head = _writebuf.readptr(size);
1013  nbytes = MIN(nbytes, size);
1014 
1015  // try to align writes on a 512 byte boundary to avoid filesystem reads
1016  if ((nbytes + _write_offset) % 512 != 0) {
1017  uint32_t ofs = (nbytes + _write_offset) % 512;
1018  if (ofs < nbytes) {
1019  nbytes -= ofs;
1020  }
1021  }
1022 
1023  last_io_operation = "write";
1024  if (!write_fd_semaphore.take(1)) {
1025  return;
1026  }
1027  if (_write_fd == -1) {
1028  write_fd_semaphore.give();
1029  return;
1030  }
1031  ssize_t nwritten = ::write(_write_fd, head, nbytes);
1032  last_io_operation = "";
1033  if (nwritten <= 0) {
1034  if (tnow - _last_write_ms > 2000) {
1035  // if we can't write for 2 seconds we give up and close
1036  // the file. This allows us to cope with temporary write
1037  // failures caused by directory listing
1038  hal.util->perf_count(_perf_errors);
1039  last_io_operation = "close";
1040  close(_write_fd);
1041  last_io_operation = "";
1042  _write_fd = -1;
1043  _initialised = false;
1044  printf("Failed to write to File: %s\n", strerror(errno));
1045  }
1046  } else {
1047  _last_write_ms = tnow;
1048  _write_offset += nwritten;
1049  _writebuf.advance(nwritten);
1050  /*
1051  the best strategy for minimizing corruption on microSD cards
1052  seems to be to write in 4k chunks and fsync the file on each
1053  chunk, ensuring the directory entry is updated after each
1054  write.
1055  */
1056 #if CONFIG_HAL_BOARD != HAL_BOARD_SITL && CONFIG_HAL_BOARD_SUBTYPE != HAL_BOARD_SUBTYPE_LINUX_NONE
1057  last_io_operation = "fsync";
1058  ::fsync(_write_fd);
1059  last_io_operation = "";
1060 #endif
1061  }
1062  write_fd_semaphore.give();
1063  hal.util->perf_end(_perf_write);
1064 }
