26 MultiCopter::MultiCopter(const char *home_str, const char *frame_str) :
27  Aircraft(home_str, frame_str),
28  frame(nullptr)
29 {
30  mass = 1.5f;

32  frame = Frame::find_frame(frame_str);
33  if (frame == nullptr) {
34  printf("Frame '%s' not found", frame_str);

36  }
37  // initial mass is passed through to Frame for it to calculate a
