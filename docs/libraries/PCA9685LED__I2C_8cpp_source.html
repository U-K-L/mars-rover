40 bool PCA9685LED_I2C::hw_init()
41 {
42  _dev = hal.i2c_mgr->get_device(1, PCA9685_ADDRESS);
43 
44  if (!_dev) {
45  return false;
46  }
47 
48  _dev->get_semaphore()->take_blocking();
49 
50  _dev->set_retries(5);
51 
52  // read the current mode1 configuration
53  uint8_t mode1 = 0;
54  if (!_dev->read_registers(PCA9685_MODE1, &mode1, sizeof(mode1))) {
55  _dev->get_semaphore()->give();
56  return false;
57  }
58 
59  // bring the device out of sleep, and enable auto register increment
60  uint8_t new_mode1 = (mode1 | PCA9685_MODE_AUTO_INCREMENT) & ~PCA9685_MODE_SLEEP;
61  const uint8_t config[2] = {PCA9685_MODE1, new_mode1};
62  if (!_dev->transfer(config, sizeof(config), nullptr, 0)) {
63  _dev->get_semaphore()->give();
64  return false;
65  }
66 
67  _dev->set_retries(1);
68 
69  _dev->get_semaphore()->give();
70 
71  _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&PCA9685LED_I2C::_timer, void));
72 
73  return true;
74 }
