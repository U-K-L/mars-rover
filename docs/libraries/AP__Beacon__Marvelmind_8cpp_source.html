60 // Calculate Modbus CRC16 for array of bytes
61 // buf: input buffer
62 // len: size of buffer
63 // returncode: CRC value
65 uint16_t AP_Beacon_Marvelmind::calc_crc_modbus(uint8_t *buf, uint16_t len)
66 {
67  uint16_t crc = 0xFFFF;
68  for (uint16_t pos = 0; pos < len; pos++) {
69  crc ^= (uint16_t) buf[pos]; // XOR byte into least sig. byte of crc
70  for (uint8_t i = 8; i != 0; i--) { // Loop over each bit
71  if ((crc & 0x0001) != 0) { // If the LSB is set
72  crc >>= 1; // Shift right and XOR 0xA001
73  crc ^= 0xA001;
74  } else {
75  // Else LSB is not set
76  crc >>= 1; // Just shift right
77  }
78  }
79  }
80  return crc;
81 }

83 void AP_Beacon_Marvelmind::process_position_datagram()
84 {
85  hedge.cur_position.address = input_buffer[16];
86  hedge.cur_position.timestamp = input_buffer[5]
87  | (((uint32_t) input_buffer[6]) << 8)
88  | (((uint32_t) input_buffer[7]) << 16)
89  | (((uint32_t) input_buffer[8]) << 24);
90  const int16_t vx = input_buffer[9] | (((uint16_t) input_buffer[10]) << 8);
91  hedge.cur_position.x__mm = vx * 10; // centimeters -> millimeters
92  const int16_t vy = input_buffer[11] | (((uint16_t) input_buffer[12]) << 8);
93  hedge.cur_position.y__mm = vy * 10; // centimeters -> millimeters
94  const int16_t vz = input_buffer[13] | (((uint16_t) input_buffer[14]) << 8);
95  hedge.cur_position.z__mm = vz * 10; // centimeters -> millimeters
96  hedge.cur_position.high_resolution = false;

135 void AP_Beacon_Marvelmind::process_beacons_positions_datagram()
136 {
137  const uint8_t n = input_buffer[5]; // number of beacons in packet
138  StationaryBeaconPosition *stationary_beacon;
139  if ((1 + n * 8) != input_buffer[4]) {
140  Debug(1, "beacon pos lo pkt size %d != %d", input_buffer[4], (1 + n * 8));
141  return; // incorrect size
142  }
143  for (uint8_t i = 0; i < n; i++) {
144  const uint8_t ofs = 6 + i * 8;
145  const uint8_t address = input_buffer[ofs];
146  const int16_t x = input_buffer[ofs + 1]
147  | (((uint16_t) input_buffer[ofs + 2]) << 8);
148  const int16_t y = input_buffer[ofs + 3]
149  | (((uint16_t) input_buffer[ofs + 4]) << 8);
150  const int16_t z = input_buffer[ofs + 5]
151  | (((uint16_t) input_buffer[ofs + 6]) << 8);
152  stationary_beacon = get_or_alloc_beacon(address);
153  if (stationary_beacon != nullptr) {
154  stationary_beacon->address = address; //The instance and the address are the same
155  stationary_beacon->x__mm = x * 10; // centimeters -> millimeters
156  stationary_beacon->y__mm = y * 10; // centimeters -> millimeters
157  stationary_beacon->z__mm = z * 10; // centimeters -> millimeters
158  stationary_beacon->high_resolution = false;
159  hedge.positions_beacons.updated = true;
160  }
161  }
162  order_stationary_beacons();
163 }

201 void AP_Beacon_Marvelmind::process_beacons_distances_datagram()
202 {
203  if (32 != input_buffer[4]) {
204  Debug(1, "beacon dist pkt size %d != 32", input_buffer[4]);
205  return; // incorrect size
206  }
207  bool set = false;
208  for (uint8_t i = 0; i < hedge.positions_beacons.num_beacons; i++) {
209  const uint8_t ofs = 6 + i * 6;
210  const uint8_t address = input_buffer[ofs];
211  const int8_t instance = find_beacon_instance(address);
212  if (instance != -1) {
213  const uint32_t distance = input_buffer[ofs + 1]
214  | (((uint32_t) input_buffer[ofs + 2]) << 8)
215  | (((uint32_t) input_buffer[ofs + 3]) << 16)
216  | (((uint32_t) input_buffer[ofs + 4]) << 24);
217  hedge.positions_beacons.beacons[instance].distance__m = distance * 0.001f; // millimeters -> meters
218  set_beacon_distance(instance, hedge.positions_beacons.beacons[instance].distance__m);
219  set = true;
220  Debug(2, "Beacon %d is %.2fm", instance, hedge.positions_beacons.beacons[instance].distance__m);
221  }
222  }
223  if (set) {
224  last_update_ms = AP_HAL::millis();
225  }

238 void AP_Beacon_Marvelmind::update(void)
239 {
240  if (uart == nullptr) {
241  return;
242  }
243  // read any available characters
244  int32_t num_bytes_read = uart->available();
245  uint8_t received_char = 0;
246  if (num_bytes_read < 0) {
247  return;
248  }
249  while (num_bytes_read-- > 0) {
250  bool good_byte = false;
251  received_char = uart->read();
252  input_buffer[num_bytes_in_block_received] = received_char;
253  switch (parse_state) {
254  case RECV_HDR:
255  switch (num_bytes_in_block_received) {
256  case 0:
257  good_byte = (received_char == 0xff);
258  break;
259  case 1:
260  good_byte = (received_char == 0x47);
261  break;
262  case 2:
263  good_byte = true;
264  break;
265  case 3:
266  data_id = (((uint16_t)received_char) << 8) + input_buffer[2];
267  good_byte = (data_id == AP_BEACON_MARVELMIND_POSITION_DATAGRAM_ID)
268  || (data_id == AP_BEACON_MARVELMIND_POSITIONS_DATAGRAM_ID)
269  || (data_id == AP_BEACON_MARVELMIND_DISTANCES_DATAGRAM_ID)
270  || (data_id == AP_BEACON_MARVELMIND_POSITION_DATAGRAM_HIGHRES_ID)
271  || (data_id == AP_BEACON_MARVELMIND_POSITIONS_DATAGRAM_HIGHRES_ID);
272  break;
273  case 4: {
274  switch (data_id) {
275  case AP_BEACON_MARVELMIND_POSITION_DATAGRAM_ID: {
276  good_byte = (received_char == 0x10);
277  break;
278  }
279  case AP_BEACON_MARVELMIND_DISTANCES_DATAGRAM_ID:
280  case AP_BEACON_MARVELMIND_POSITIONS_DATAGRAM_ID:
281  case AP_BEACON_MARVELMIND_POSITIONS_DATAGRAM_HIGHRES_ID:
282  good_byte = true;
283  break;
284  case AP_BEACON_MARVELMIND_POSITION_DATAGRAM_HIGHRES_ID: {
285  good_byte = (received_char == 0x16);
286  break;
287  }
288  }
289  if (good_byte) {
290  parse_state = RECV_DGRAM;
291  }
292  break;
293  }
294  }
295  if (good_byte) {
296  // correct header byte
297  num_bytes_in_block_received++;
298  } else {
299  // ...or incorrect
300  parse_state = RECV_HDR;
301  num_bytes_in_block_received = 0;
302  }
303  break;
304 
305  case RECV_DGRAM:
306  num_bytes_in_block_received++;
307  if (num_bytes_in_block_received >= 7 + input_buffer[4]) {
308  // parse dgram
309  uint16_t block_crc = calc_crc_modbus(input_buffer, num_bytes_in_block_received);
310  if (block_crc == 0) {
311  switch (data_id) {
312  case AP_BEACON_MARVELMIND_POSITION_DATAGRAM_ID:
313  {
314  // add to position_buffer
315  process_position_datagram();
316  vehicle_position_initialized = true;
317  set_stationary_beacons_positions();
318  break;
319  }
320 
321  case AP_BEACON_MARVELMIND_POSITIONS_DATAGRAM_ID:
322  {
323  process_beacons_positions_datagram();
324  beacon_position_initialized = true;
325  set_stationary_beacons_positions();
326  break;
327  }
328 
329  case AP_BEACON_MARVELMIND_DISTANCES_DATAGRAM_ID:
330  {
331  process_beacons_distances_datagram();
332  break;
333  }
334 
335  case AP_BEACON_MARVELMIND_POSITION_DATAGRAM_HIGHRES_ID:
336  {
337  process_position_highres_datagram();
338  vehicle_position_initialized = true;
339  set_stationary_beacons_positions();
340  break;
341  }
342 
343  case AP_BEACON_MARVELMIND_POSITIONS_DATAGRAM_HIGHRES_ID:
344  {
345  process_beacons_positions_highres_datagram();
346  beacon_position_initialized = true;
347  set_stationary_beacons_positions();
348  break;
349  }
350  }
351  }
352  // and repeat
353  parse_state = RECV_HDR;
354  num_bytes_in_block_received = 0;
355  }
356  break;
357  }
358  }

372  vehicle_position_NED__m = Vector3f(hedge.cur_position.y__mm * 0.001f,

409 void AP_Beacon_Marvelmind::order_stationary_beacons()
410 {
411  if (hedge.positions_beacons.updated) {
412  bool swapped = false;
413  uint8_t j = hedge.positions_beacons.num_beacons;
414  do
415  {
416  swapped = false;
417  StationaryBeaconPosition beacon_to_swap;
418  for (uint8_t i = 1; i < j; i++) {
419  if (hedge.positions_beacons.beacons[i-1].address > hedge.positions_beacons.beacons[i].address) {
420  beacon_to_swap = hedge.positions_beacons.beacons[i];
421  hedge.positions_beacons.beacons[i] = hedge.positions_beacons.beacons[i-1];
422  hedge.positions_beacons.beacons[i-1] = beacon_to_swap;
423  swapped = true;
424  }
425  }
426  j--;
427  } while(swapped);
428  }
