302 // returns the closest these objects will get in the body z axis (in metres)
303 float closest_approach_z(const Location &my_loc,
304  const Vector3f &my_vel,
305  const Location &obstacle_loc,
306  const Vector3f &obstacle_vel,
307  const uint8_t time_horizon)
308 {
309 
310  float delta_vel_d = obstacle_vel[2] - my_vel[2];
311  float delta_pos_d = obstacle_loc.alt - my_loc.alt;
312 
313  float ret;
314  if (delta_pos_d >= 0 && delta_vel_d >= 0) {

317  ret = fabsf(delta_pos_d);
318  } else {
319  ret = fabsf(delta_pos_d - delta_vel_d * time_horizon);
320  }
321 
322  debug(" time_horizon: (%d)", time_horizon);
323  debug(" delta pos: (%f) metres", delta_pos_d/100.0f);

330 void AP_Avoidance::update_threat_level(const Location &my_loc,
331  const Vector3f &my_vel,
332  AP_Avoidance::Obstacle &obstacle)
333 {
334 
335  Location &obstacle_loc = obstacle._location;
336  Vector3f &obstacle_vel = obstacle._velocity;
337 
338  obstacle.threat_level = MAV_COLLISION_THREAT_LEVEL_NONE;
339 
340  const uint32_t obstacle_age = AP_HAL::millis() - obstacle.timestamp_ms;
341  float closest_xy = closest_approach_xy(my_loc, my_vel, obstacle_loc, obstacle_vel, _fail_time_horizon + obstacle_age/1000);
342  if (closest_xy < _fail_distance_xy) {
343  obstacle.threat_level = MAV_COLLISION_THREAT_LEVEL_HIGH;
344  } else {
345  closest_xy = closest_approach_xy(my_loc, my_vel, obstacle_loc, obstacle_vel, _warn_time_horizon + obstacle_age/1000);
346  if (closest_xy < _warn_distance_xy) {
347  obstacle.threat_level = MAV_COLLISION_THREAT_LEVEL_LOW;
348  }
349  }
350 
351  // check for vertical separation; our threat level is the minimum
352  // of vertical and horizontal threat levels
353  float closest_z = closest_approach_z(my_loc, my_vel, obstacle_loc, obstacle_vel, _warn_time_horizon + obstacle_age/1000);
354  if (obstacle.threat_level != MAV_COLLISION_THREAT_LEVEL_NONE) {
355  if (closest_z > _warn_distance_z) {
356  obstacle.threat_level = MAV_COLLISION_THREAT_LEVEL_NONE;
357  } else {
358  closest_z = closest_approach_z(my_loc, my_vel, obstacle_loc, obstacle_vel, _fail_time_horizon + obstacle_age/1000);
359  if (closest_z > _fail_distance_z) {
360  obstacle.threat_level = MAV_COLLISION_THREAT_LEVEL_LOW;
361  }
362  }
363  }
364 
365  // If we haven't heard from a vehicle then assume it is no threat
366  if (obstacle_age > MAX_OBSTACLE_AGE_MS) {
367  obstacle.threat_level = MAV_COLLISION_THREAT_LEVEL_NONE;
368  }
369 
370  // could optimise this to not calculate a lot of this if threat
371  // level is none - but only *once the GCS has been informed*!
372  obstacle.closest_approach_xy = closest_xy;
373  obstacle.closest_approach_z = closest_z;
374  float current_distance = my_loc.get_distance(obstacle_loc);
375  obstacle.distance_to_closest_approach = current_distance - closest_xy;
376  Vector2f net_velocity_ne = Vector2f(my_vel[0] - obstacle_vel[0], my_vel[1] - obstacle_vel[1]);
377  obstacle.time_to_closest_approach = 0.0f;
378  if (!is_zero(obstacle.distance_to_closest_approach) &&
379  ! is_zero(net_velocity_ne.length())) {
380  obstacle.time_to_closest_approach = obstacle.distance_to_closest_approach / net_velocity_ne.length();
381  }
382 }

437 void AP_Avoidance::check_for_threats()
438 {
439  Location my_loc;
440  if (!_ahrs.get_position(my_loc)) {
441  // if we don't know our own location we can't determine any threat level
442  return;
443  }
444 
445  Vector3f my_vel;
446  if (!_ahrs.get_velocity_NED(my_vel)) {
447  // assuming our own velocity to be zero here may cause us to
448  // fly into something. Better not to attempt to avoid in this
449  // case.
450  return;
451  }
452 
453  // we always check all obstacles to see if they are threats since it
454  // is most likely our own position and/or velocity have changed
455  // determine the current most-serious-threat
456  _current_most_serious_threat = -1;
457  for (uint8_t i=0; i<_obstacle_count; i++) {
458 
459  AP_Avoidance::Obstacle &obstacle = _obstacles[i];
460  const uint32_t obstacle_age = AP_HAL::millis() - obstacle.timestamp_ms;
461  debug("i=%d src_id=%d timestamp=%u age=%d", i, obstacle.src_id, obstacle.timestamp_ms, obstacle_age);
462 
463  update_threat_level(my_loc, my_vel, obstacle);
464  debug(" threat-level=%d", obstacle.threat_level);
465 
466  // ignore any really old data:
467  if (obstacle_age > MAX_OBSTACLE_AGE_MS) {
468  // shrink list if this is the last entry:
469  if (i == _obstacle_count-1) {
470  _obstacle_count -= 1;
471  }
472  continue;
473  }
474 
475  if (obstacle_is_more_serious_threat(obstacle)) {
476  _current_most_serious_threat = i;
477  }
478  }
479  if (_current_most_serious_threat != -1) {
480  debug("Current most serious threat: %d level=%d", _current_most_serious_threat, _obstacles[_current_most_serious_threat].threat_level);
481  }
482 }

514 void AP_Avoidance::handle_avoidance_local(AP_Avoidance::Obstacle *threat)
515 {
516  MAV_COLLISION_THREAT_LEVEL new_threat_level = MAV_COLLISION_THREAT_LEVEL_NONE;
517  MAV_COLLISION_ACTION action = MAV_COLLISION_ACTION_NONE;
518 
519  if (threat != nullptr) {
520  new_threat_level = threat->threat_level;
521  if (new_threat_level == MAV_COLLISION_THREAT_LEVEL_HIGH) {
522  action = (MAV_COLLISION_ACTION)_fail_action.get();
523  Location my_loc;
524  if (action != MAV_COLLISION_ACTION_NONE && _fail_altitude_minimum > 0 &&
525  _ahrs.get_position(my_loc) && ((my_loc.alt*0.01f) < _fail_altitude_minimum)) {
526  // disable avoidance when close to ground, report only
527  action = MAV_COLLISION_ACTION_REPORT;
528  }
529  }
530  }
531 
532  uint32_t now = AP_HAL::millis();
533 
534  if (new_threat_level != _threat_level) {
535  // transition to higher states immediately, recovery to lower states more slowly
536  if (((now - _last_state_change_ms) > AP_AVOIDANCE_STATE_RECOVERY_TIME_MS) || (new_threat_level > _threat_level)) {
537  // handle recovery from high threat level
538  if (_threat_level == MAV_COLLISION_THREAT_LEVEL_HIGH) {
539  handle_recovery(_fail_recovery);
540  _latest_action = MAV_COLLISION_ACTION_NONE;
541  }
542 
543  // update state
544  _last_state_change_ms = now;
545  _threat_level = new_threat_level;
546  }
547  }
548 
549  // handle ongoing threat by calling vehicle specific handler
550  if ((threat != nullptr) && (_threat_level == MAV_COLLISION_THREAT_LEVEL_HIGH) && (action > MAV_COLLISION_ACTION_REPORT)) {
551  _latest_action = handle_avoidance(threat, action);
552  }
553 }

591 // get unit vector away from the nearest obstacle
592 bool AP_Avoidance::get_vector_perpendicular(const AP_Avoidance::Obstacle *obstacle, Vector3f &vec_neu)
593 {
594  if (obstacle == nullptr) {
595  // why where we called?!
596  return false;
597  }
598 
599  Location my_abs_pos;
600  if (!_ahrs.get_position(my_abs_pos)) {
601  // we should not get to here! If we don't know our position
602  // we can't know if there are any threats, for starters!
603  return false;
604  }
605 
606  // if their velocity is moving around close to zero then flying
607  // perpendicular to that velocity may mean we do weird things.
608  // Instead, we will fly directly away from them
609  if (obstacle->_velocity.length() < _low_velocity_threshold) {
610  const Vector2f delta_pos_xy = obstacle->_location.get_distance_NE(my_abs_pos);
611  const float delta_pos_z = my_abs_pos.alt - obstacle->_location.alt;
612  Vector3f delta_pos_xyz = Vector3f(delta_pos_xy.x, delta_pos_xy.y, delta_pos_z);
613  // avoid div by zero
614  if (delta_pos_xyz.is_zero()) {
615  return false;
616  }
617  delta_pos_xyz.normalize();
618  vec_neu = delta_pos_xyz;
619  return true;
620  } else {
621  vec_neu = perpendicular_xyz(obstacle->_location, obstacle->_velocity, my_abs_pos);
622  // avoid div by zero
623  if (vec_neu.is_zero()) {
624  return false;
625  }
626  vec_neu.normalize();
627  return true;
628  }
629 }
