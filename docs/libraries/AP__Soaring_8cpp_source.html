148 bool SoaringController::suppress_throttle()
149 {
150  float alt = 0;
151  get_altitude_wrt_home(&alt);
152 
153  if (_throttle_suppressed && (alt < alt_min)) {
154  // Time to throttle up
155  _throttle_suppressed = false;
156  } else if ((!_throttle_suppressed) && (alt > alt_cutoff)) {
157  // Start glide
158  _throttle_suppressed = true;
159  // Zero the pitch integrator - the nose is currently raised to climb, we need to go back to glide.
160  _spdHgt.reset_pitch_I();
161  _cruise_start_time_us = AP_HAL::micros64();
162  // Reset the filtered vario rate - it is currently elevated due to the climb rate and would otherwise take a while to fall again,
163  // leading to false positives.
164  _vario.filtered_reading = 0;
165  }
166 
167  return _throttle_suppressed;
168 }

179 bool SoaringController::check_cruise_criteria()
180 {
181  float thermalability = (_ekf.X[0]*expf(-powf(_loiter_rad / _ekf.X[1], 2))) - EXPECTED_THERMALLING_SINK;
182  float alt = _vario.alt;
183

185  gcs().send_text(MAV_SEVERITY_INFO, "Thermal weak, recommend quitting: W %f R %f th %f alt %f Mc %f", (double)_ekf.X[0], (double)_ekf.X[1], (double)thermalability, (double)alt, (double)McCready(alt));
186  return true;

205 void SoaringController::init_thermalling()
206 {
207  // Calc filter matrices - so that changes to parameters can be updated by switching in and out of thermal mode
208  float r = powf(thermal_r, 2);
209  float cov_q1 = powf(thermal_q1, 2); // State covariance
210  float cov_q2 = powf(thermal_q2, 2); // State covariance
211  const float init_q[4] = {cov_q1, cov_q2, cov_q2, cov_q2};
212  const MatrixN<float,4> q{init_q};
213  const float init_p[4] = {INITIAL_STRENGTH_COVARIANCE, INITIAL_RADIUS_COVARIANCE, INITIAL_POSITION_COVARIANCE, INITIAL_POSITION_COVARIANCE};
214  const MatrixN<float,4> p{init_p};
215 
216  // New state vector filter will be reset. Thermal location is placed in front of a/c
217  const float init_xr[4] = {INITIAL_THERMAL_STRENGTH,
218  INITIAL_THERMAL_RADIUS,
219  thermal_distance_ahead * cosf(_ahrs.yaw),
220  thermal_distance_ahead * sinf(_ahrs.yaw)};
221  const VectorN<float,4> xr{init_xr};
222 
223  // Also reset covariance matrix p so filter is not affected by previous data
224  _ekf.reset(xr, p, q, r);
225 
226  _ahrs.get_position(_prev_update_location);
227  _prev_update_time = AP_HAL::micros64();
228  _thermal_start_time_us = AP_HAL::micros64();
229 }
