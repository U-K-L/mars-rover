47 bool ToshibaLED_I2C::hw_init(void)
48 {
49  // first look for led on external bus
50  _dev = std::move(hal.i2c_mgr->get_device(_bus, TOSHIBA_LED_I2C_ADDR));
51  if (!_dev) {
52  return false;
53  }
54  WITH_SEMAPHORE(_dev->get_semaphore());
55 
56  _dev->set_retries(10);
57 
58  // enable the led
59  bool ret = _dev->write_register(TOSHIBA_LED_ENABLE, 0x03);
60  if (!ret) {
61  return false;
62  }
63 
64  // update the red, green and blue values to zero
65  uint8_t val[4] = { TOSHIBA_LED_PWM0, _led_off, _led_off, _led_off };
66  ret = _dev->transfer(val, sizeof(val), nullptr, 0);
67 
68  _dev->set_retries(1);
69 
70  if (ret) {
71  _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&ToshibaLED_I2C::_timer, void));
72  }
73 
74  return ret;
75 }
