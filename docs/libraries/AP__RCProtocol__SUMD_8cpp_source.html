82 void AP_RCProtocol_SUMD::_process_byte(uint32_t timestamp_us, uint8_t byte)
83 {
84  if (timestamp_us - last_packet_us > 3000U) {
85  _decode_state = SUMD_DECODE_STATE_UNSYNCED;
86  }
87  switch (_decode_state) {
88  case SUMD_DECODE_STATE_UNSYNCED:
89 #ifdef SUMD_DEBUG
90  hal.console->printf(" SUMD_DECODE_STATE_UNSYNCED \n") ;
91 #endif
92  if (byte == SUMD_HEADER_ID) {
93  _rxpacket.header = byte;
94  _sumd = true;
95  _rxlen = 0;
96  _crc16 = 0x0000;
97  _crc8 = 0x00;
98  _crcOK = false;
99  _crc16 = crc_xmodem_update(_crc16, byte);
100  _crc8 = sumd_crc8(_crc8, byte);
101  _decode_state = SUMD_DECODE_STATE_GOT_HEADER;
102 
103 #ifdef SUMD_DEBUG
104  hal.console->printf(" SUMD_DECODE_STATE_GOT_HEADER: %x \n", byte) ;
105 #endif
106  last_packet_us = timestamp_us;
107  }
108  break;
109 
110  case SUMD_DECODE_STATE_GOT_HEADER:
111  if (byte == SUMD_ID_SUMD || byte == SUMD_ID_SUMH) {
112  _rxpacket.status = byte;
113 
114  if (byte == SUMD_ID_SUMH) {
115  _sumd = false;
116  }
117 
118  if (_sumd) {
119  _crc16 = crc_xmodem_update(_crc16, byte);
120 
121  } else {
122  _crc8 = sumd_crc8(_crc8, byte);
123  }
124 
125  _decode_state = SUMD_DECODE_STATE_GOT_STATE;
126 
127 #ifdef SUMD_DEBUG
128  hal.console->printf(" SUMD_DECODE_STATE_GOT_STATE: %x \n", byte) ;
129 #endif
130 
131  } else {
132  _decode_state = SUMD_DECODE_STATE_UNSYNCED;
133  }
134 
135  break;
136 
137  case SUMD_DECODE_STATE_GOT_STATE:
138  if (byte >= 2 && byte <= SUMD_MAX_CHANNELS) {
139  _rxpacket.length = byte;
140 
141  if (_sumd) {
142  _crc16 = crc_xmodem_update(_crc16, byte);
143 
144  } else {
145  _crc8 = sumd_crc8(_crc8, byte);
146  }
147 
148  _rxlen++;
149  _decode_state = SUMD_DECODE_STATE_GOT_LEN;
150 
151 #ifdef SUMD_DEBUG
152  hal.console->printf(" SUMD_DECODE_STATE_GOT_LEN: %x (%d) \n", byte, byte) ;
153 #endif
154 
155  } else {
156  _decode_state = SUMD_DECODE_STATE_UNSYNCED;
157  }
158 
159  break;
160 
161  case SUMD_DECODE_STATE_GOT_LEN:
162  _rxpacket.sumd_data[_rxlen] = byte;
163 
164  if (_sumd) {
165  _crc16 = crc_xmodem_update(_crc16, byte);
166 
167  } else {
168  _crc8 = sumd_crc8(_crc8, byte);
169  }
170 
171  _rxlen++;
172 
173  if (_rxlen <= ((_rxpacket.length * 2))) {
174 #ifdef SUMD_DEBUG
175  hal.console->printf(" SUMD_DECODE_STATE_GOT_DATA[%d]: %x\n", _rxlen - 2, byte) ;
176 #endif
177 
178  } else {
179  _decode_state = SUMD_DECODE_STATE_GOT_DATA;
180 
181 #ifdef SUMD_DEBUG
182  hal.console->printf(" SUMD_DECODE_STATE_GOT_DATA -- finish --\n") ;
183 #endif
184 
185  }
186 
187  break;
188 
189  case SUMD_DECODE_STATE_GOT_DATA:
190  _rxpacket.crc16_high = byte;
191 
192 #ifdef SUMD_DEBUG
193  hal.console->printf(" SUMD_DECODE_STATE_GOT_CRC16[1]: %x [%x]\n", byte, ((_crc16 >> 8) & 0xff)) ;
194 #endif
195 
196  if (_sumd) {
197  _decode_state = SUMD_DECODE_STATE_GOT_CRC;
198 
199  } else {
200  _decode_state = SUMD_DECODE_STATE_GOT_CRC16_BYTE_1;
201  }
202 
203  break;
204 
205  case SUMD_DECODE_STATE_GOT_CRC16_BYTE_1:
206  _rxpacket.crc16_low = byte;
207 
208 #ifdef SUMD_DEBUG
209  hal.console->printf(" SUMD_DECODE_STATE_GOT_CRC16[2]: %x [%x]\n", byte, (_crc16 & 0xff)) ;
210 #endif
211 
212  _decode_state = SUMD_DECODE_STATE_GOT_CRC16_BYTE_2;
213 
214  break;
215 
216  case SUMD_DECODE_STATE_GOT_CRC16_BYTE_2:
217  _rxpacket.telemetry = byte;
218 
219 #ifdef SUMD_DEBUG
220  hal.console->printf(" SUMD_DECODE_STATE_GOT_SUMH_TELEMETRY: %x\n", byte) ;
221 #endif
222 
223  _decode_state = SUMD_DECODE_STATE_GOT_CRC;
224 
225  break;
226 
227  case SUMD_DECODE_STATE_GOT_CRC:
228  if (_sumd) {
229  _rxpacket.crc16_low = byte;
230 
231 #ifdef SUMD_DEBUG
232  hal.console->printf(" SUMD_DECODE_STATE_GOT_CRC[2]: %x [%x]\n\n", byte, (_crc16 & 0xff)) ;
233 #endif
234 
235  if (_crc16 == (uint16_t)(_rxpacket.crc16_high << 8) + _rxpacket.crc16_low) {
236  _crcOK = true;
237  }
238 
239  } else {
240  _rxpacket.crc8 = byte;
241 
242 #ifdef SUMD_DEBUG
243  hal.console->printf(" SUMD_DECODE_STATE_GOT_CRC8_SUMH: %x [%x]\n\n", byte, _crc8) ;
244 #endif
245 
246  if (_crc8 == _rxpacket.crc8) {
247  _crcOK = true;
248  }
249  }
250 
251  if (_crcOK) {
252 #ifdef SUMD_DEBUG
253  hal.console->printf(" CRC - OK \n") ;
254 #endif
255  if (_sumd) {
256 #ifdef SUMD_DEBUG
257  hal.console->printf(" Got valid SUMD Packet\n") ;
258 #endif
259  } else {
260 #ifdef SUMD_DEBUG
261  hal.console->printf(" Got valid SUMH Packet\n") ;
262 #endif
263 
264  }
265 
266 #ifdef SUMD_DEBUG
267  hal.console->printf(" RXLEN: %d [Chans: %d] \n\n", _rxlen - 1, (_rxlen - 1) / 2) ;
268 #endif
269 
270 
271  unsigned i;
272  uint8_t num_values;
273  uint16_t values[SUMD_MAX_CHANNELS];
274 
275  /* received Channels */
276  if ((uint16_t)_rxpacket.length > SUMD_MAX_CHANNELS) {
277  _rxpacket.length = (uint8_t) SUMD_MAX_CHANNELS;
278  }
279 
280  num_values = (uint16_t)_rxpacket.length;
281 
282  /* decode the actual packet */
283  /* reorder first 4 channels */
284 
285  /* ch1 = roll -> sumd = ch2 */
286  values[0] = (uint16_t)((_rxpacket.sumd_data[1 * 2 + 1] << 8) | _rxpacket.sumd_data[1 * 2 + 2]) >> 3;
287  /* ch2 = pitch -> sumd = ch2 */
288  values[1] = (uint16_t)((_rxpacket.sumd_data[2 * 2 + 1] << 8) | _rxpacket.sumd_data[2 * 2 + 2]) >> 3;
289  /* ch3 = throttle -> sumd = ch2 */
290  values[2] = (uint16_t)((_rxpacket.sumd_data[0 * 2 + 1] << 8) | _rxpacket.sumd_data[0 * 2 + 2]) >> 3;
291  /* ch4 = yaw -> sumd = ch2 */
292  values[3] = (uint16_t)((_rxpacket.sumd_data[3 * 2 + 1] << 8) | _rxpacket.sumd_data[3 * 2 + 2]) >> 3;
293 
294  /* we start at channel 5(index 4) */
295  unsigned chan_index = 4;
296 
297  for (i = 4; i < _rxpacket.length; i++) {
298 #ifdef SUMD_DEBUG
299  hal.console->printf("ch[%d] : %x %x [ %x %d ]\n", i + 1, _rxpacket.sumd_data[i * 2 + 1], _rxpacket.sumd_data[i * 2 + 2],
300  ((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3,
301  ((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3);
302 #endif
303 
304  values[chan_index] = (uint16_t)((_rxpacket.sumd_data[i * 2 + 1] << 8) | _rxpacket.sumd_data[i * 2 + 2]) >> 3;
305  /* convert values to 1000-2000 ppm encoding in a not too sloppy fashion */
306  //channels[chan_index] = (uint16_t)(channels[chan_index] * SUMD_SCALE_FACTOR + .5f) + SUMD_SCALE_OFFSET;
307 
308  chan_index++;
309  }
310  if (_rxpacket.status == 0x01) {
311  add_input(num_values, values, false);
312  } else if (_rxpacket.status == 0x81) {
313  add_input(num_values, values, true);
314  }
315  } else {
316 #ifdef SUMD_DEBUG
317  hal.console->printf(" CRC - fail 0x%X 0x%X\n", _crc16, (uint16_t)(_rxpacket.crc16_high << 8) + _rxpacket.crc16_low);
318 #endif
319  }
320 
321  _decode_state = SUMD_DECODE_STATE_UNSYNCED;
322  break;
323  }
324 }
