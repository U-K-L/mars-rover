85 bool AP_Compass_BMM150::_load_trim_values()
86 {
87  struct {
88  int8_t dig_x1;
89  int8_t dig_y1;
90  uint8_t rsv[3];
91  le16_t dig_z4;
92  int8_t dig_x2;
93  int8_t dig_y2;
94  uint8_t rsv2[2];
95  le16_t dig_z2;
96  le16_t dig_z1;
97  le16_t dig_xyz1;
98  le16_t dig_z3;
99  int8_t dig_xy2;
100  uint8_t dig_xy1;
101  } PACKED trim_registers, trim_registers2;
102 
103  // read the registers twice to confirm we have the right
104  // values. There is no CRC in the registers and these values are
105  // vital to correct operation
106  int8_t tries = 4;
107  while (tries--) {
108  if (!_dev->read_registers(DIG_X1_REG, (uint8_t *)&trim_registers,
109  sizeof(trim_registers))) {
110  continue;
111  }
112  if (!_dev->read_registers(DIG_X1_REG, (uint8_t *)&trim_registers2,
113  sizeof(trim_registers))) {
114  continue;
115  }
116  if (memcmp(&trim_registers, &trim_registers2, sizeof(trim_registers)) == 0) {
117  break;
118  }
119  }
120  if (-1 == tries) {
121  hal.console->printf("BMM150: Failed to load trim registers\n");
122  return false;
123  }
124 
125  _dig.x1 = trim_registers.dig_x1;
126  _dig.x2 = trim_registers.dig_x2;
127  _dig.xy1 = trim_registers.dig_xy1;
128  _dig.xy2 = trim_registers.dig_xy2;
129  _dig.xyz1 = le16toh(trim_registers.dig_xyz1);
130  _dig.y1 = trim_registers.dig_y1;
131  _dig.y2 = trim_registers.dig_y2;
132  _dig.z1 = le16toh(trim_registers.dig_z1);
133  _dig.z2 = le16toh(trim_registers.dig_z2);
134  _dig.z3 = le16toh(trim_registers.dig_z3);
135  _dig.z4 = le16toh(trim_registers.dig_z4);
136 
137  return true;
138 }

140 bool AP_Compass_BMM150::init()
141 {
142  uint8_t val = 0;
143  bool ret;
144 
145  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
146  hal.console->printf("BMM150: Unable to get bus semaphore\n");
147  return false;
148  }
149 
150  // 10 retries for init
151  _dev->set_retries(10);
152 
153  // use checked registers to cope with bus errors
154  _dev->setup_checked_registers(4);
155 
156  int8_t boot_tries = 4;
157  while (boot_tries--) {
158  /* Do a soft reset */
159  ret = _dev->write_register(POWER_AND_OPERATIONS_REG, SOFT_RESET);
160  hal.scheduler->delay(2);
161  if (!ret) {
162  continue;
163  }
164 
165  /* Change power state from suspend mode to sleep mode */
166  ret = _dev->write_register(POWER_AND_OPERATIONS_REG, POWER_CONTROL_VAL, true);
167  hal.scheduler->delay(2);
168  if (!ret) {
169  continue;
170  }
171 
172  ret = _dev->read_registers(CHIP_ID_REG, &val, 1);
173  if (!ret) {
174  continue;
175  }
176  if (val == CHIP_ID_VAL) {
177  // found it
178  break;
179  }
180  if (boot_tries == 0) {
181  hal.console->printf("BMM150: Wrong chip ID 0x%02x should be 0x%02x\n", val, CHIP_ID_VAL);
182  }
183  }
184  if (-1 == boot_tries) {
185  goto bus_error;
186  }
187 
188  ret = _load_trim_values();
189  if (!ret) {
190  goto bus_error;
191  }
192 
193  /*
194  * Recommended preset for high accuracy:
195  * - Rep X/Y = 47
196  * - Rep Z = 83
197  * - ODR = 20
198  * But we are going to use 30Hz of ODR
199  */
200  ret = _dev->write_register(REPETITIONS_XY_REG, (47 - 1) / 2, true);
201  if (!ret) {
202  goto bus_error;
203  }
204  ret = _dev->write_register(REPETITIONS_Z_REG, 83 - 1, true);
205  if (!ret) {
206  goto bus_error;
207  }
208  /* Change operation mode from sleep to normal and set ODR */
209  ret = _dev->write_register(OP_MODE_SELF_TEST_ODR_REG, NORMAL_MODE | ODR_30HZ, true);
210  if (!ret) {
211  goto bus_error;
212  }
213 
214  _dev->get_semaphore()->give();
215 
216  /* register the compass instance in the frontend */
217  _compass_instance = register_compass();
218 
219  _dev->set_device_type(DEVTYPE_BMM150);
220  set_dev_id(_compass_instance, _dev->get_bus_id());
221 
222  _perf_err = hal.util->perf_alloc(AP_HAL::Util::PC_COUNT, "BMM150_err");
223 
224  // 2 retries for run
225  _dev->set_retries(2);
226 
227  _dev->register_periodic_callback(MEASURE_TIME_USEC,
228  FUNCTOR_BIND_MEMBER(&AP_Compass_BMM150::_update, void));
229 
230  _last_read_ms = AP_HAL::millis();
231 
232  return true;
233 
234 bus_error:
235  hal.console->printf("BMM150: Bus communication error\n");
236  _dev->get_semaphore()->give();
237  return false;
238 }

244 int16_t AP_Compass_BMM150::_compensate_xy(int16_t xy, uint32_t rhall, int32_t txy1, int32_t txy2)
245 {
246  int32_t inter = ((int32_t)_dig.xyz1) << 14;
247  inter /= rhall;
248  inter -= 0x4000;
249 
250  int32_t val = _dig.xy2 * ((inter * inter) >> 7);
251  val += (inter * (((uint32_t)_dig.xy1) << 7));
252  val >>= 9;
253  val += 0x100000;
254  val *= (txy2 + 0xA0);
255  val >>= 12;
256  val *= xy;
257  val >>= 13;
258  val += (txy1 << 3);
259 
260  return val;
261 }

263 int16_t AP_Compass_BMM150::_compensate_z(int16_t z, uint32_t rhall)
264 {
265  int32_t dividend = int32_t(z - _dig.z4) << 15;
266  int32_t dividend2 = dividend - ((_dig.z3 * (int32_t(rhall) - int32_t(_dig.xyz1))) >> 2);
267 
268  int32_t divisor = int32_t(_dig.z1) * (rhall << 1);
269  divisor += 0x8000;
270  divisor >>= 16;
271  divisor += _dig.z2;
272 
273  int16_t ret = constrain_int32(dividend2 / divisor, -0x8000, 0x8000);
274 #if 0

284 void AP_Compass_BMM150::_update()
285 {
286  le16_t data[4];
287  bool ret = _dev->read_registers(DATA_X_LSB_REG, (uint8_t *) &data, sizeof(data));
288 
289  /* Checking data ready status */
290  if (!ret || !(data[3] & 0x1)) {
291  _dev->check_next_register();
292  uint32_t now = AP_HAL::millis();
293  if (now - _last_read_ms > 250) {
294  // cope with power cycle to sensor
295  _last_read_ms = now;
296  _dev->write_register(POWER_AND_OPERATIONS_REG, SOFT_RESET);
297  _dev->write_register(POWER_AND_OPERATIONS_REG, POWER_CONTROL_VAL, true);
298  hal.util->perf_count(_perf_err);
299  }
300  return;
301  }
302 
303  const uint16_t rhall = le16toh(data[3]) >> 2;
304 
305  Vector3f raw_field = Vector3f{
306  (float) _compensate_xy(((int16_t)le16toh(data[0])) >> 3,
307  rhall, _dig.x1, _dig.x2),
308  (float) _compensate_xy(((int16_t)le16toh(data[1])) >> 3,
309  rhall, _dig.y1, _dig.y2),
310  (float) _compensate_z(((int16_t)le16toh(data[2])) >> 1, rhall)};
311 
312  /* apply sensitivity scale 16 LSB/uT */
313  raw_field /= 16;
314  /* convert uT to milligauss */
315  raw_field *= 10;
316 
317  _last_read_ms = AP_HAL::millis();
318 
319  accumulate_sample(raw_field, _compass_instance);
320  _dev->check_next_register();
321 }
