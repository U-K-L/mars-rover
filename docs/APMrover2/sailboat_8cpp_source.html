102 void Sailboat::update_mainsail(float desired_speed)
103 {
104  if (!enabled()) {
105  return;
106  }
107 
108  // relax sail if desired speed is zero
109  if (!is_positive(desired_speed)) {
110  rover.g2.motors.set_mainsail(100.0f);
111  return;
112  }
113 
114  // + is wind over starboard side, - is wind over port side, but as the sails are sheeted the same on each side it makes no difference so take abs
115  float wind_dir_apparent = fabsf(rover.g2.windvane.get_apparent_wind_direction_rad());
116  wind_dir_apparent = degrees(wind_dir_apparent);
117 
118  // set the main sail to the ideal angle to the wind
119  float mainsail_angle = wind_dir_apparent -sail_angle_ideal;
120 
121  // make sure between allowable range
122  mainsail_angle = constrain_float(mainsail_angle,sail_angle_min, sail_angle_max);
123 
124  // linear interpolate mainsail value (0 to 100) from wind angle mainsail_angle
125  float mainsail = linear_interpolate(0.0f, 100.0f, mainsail_angle,sail_angle_min,sail_angle_max);
126 
127  // use PID controller to sheet out
128  const float pid_offset = rover.g2.attitude_control.get_sail_out_from_heel(radians(sail_heel_angle_max), rover.G_Dt) * 100.0f;
129 
130  mainsail = constrain_float((mainsail+pid_offset), 0.0f ,100.0f);
131  rover.g2.motors.set_mainsail(mainsail);

202 // this function assumes the caller has already checked sailboat_use_indirect_route(desired_heading_cd) returned true
203 float Sailboat::calc_heading(float desired_heading_cd)
204 {
205  if (!enabled()) {
206  return desired_heading_cd;
207  }
208 
209  bool should_tack = false;
210 
211  // check for user requested tack
212  uint32_t now = AP_HAL::millis();
213  if (auto_tack_request_ms != 0) {
214  // set should_tack flag is user requested tack within last 0.5 sec
215  should_tack = ((now - auto_tack_request_ms) < 500);
216  auto_tack_request_ms = 0;
217  }
218 
219  // calculate left and right no go headings looking upwind
220  const float left_no_go_heading_rad = wrap_2PI(rover.g2.windvane.get_absolute_wind_direction_rad() + radians(sail_no_go));
221  const float right_no_go_heading_rad = wrap_2PI(rover.g2.windvane.get_absolute_wind_direction_rad() - radians(sail_no_go));
222 
223  // calculate current tack, Port if heading is left of no-go, STBD if right of no-go
224  Sailboat_Tack current_tack;
225  if (is_negative(rover.g2.windvane.get_apparent_wind_direction_rad())) {
226  current_tack = TACK_PORT;
227  } else {
228  current_tack = TACK_STARBOARD;
229  }
230 
231  // trigger tack if cross track error larger than waypoint_overshoot parameter
232  // this effectively defines a 'corridor' of width 2*waypoint_overshoot that the boat will stay within
233  const float cross_track_error = rover.g2.wp_nav.crosstrack_error();
234  if ((fabsf(cross_track_error) >= rover.g2.wp_nav.get_overshoot()) && !is_zero(rover.g2.wp_nav.get_overshoot()) && !currently_tacking) {
235  // make sure the new tack will reduce the cross track error
236  // if were on starboard tack we are traveling towards the left hand boundary
237  if (is_positive(cross_track_error) && (current_tack == TACK_STARBOARD)) {
238  should_tack = true;
239  }
240  // if were on port tack we are traveling towards the right hand boundary
241  if (is_negative(cross_track_error) && (current_tack == TACK_PORT)) {
242  should_tack = true;
243  }
244  }
245 
246  // if tack triggered, calculate target heading
247  if (should_tack) {
248  gcs().send_text(MAV_SEVERITY_INFO, "Sailboat: Tacking");
249  // calculate target heading for the new tack
250  switch (current_tack) {
251  case TACK_PORT:
252  tack_heading_rad = right_no_go_heading_rad;
253  break;
254  case TACK_STARBOARD:
255  tack_heading_rad = left_no_go_heading_rad;
256  break;
257  }
258  currently_tacking = true;
259  auto_tack_start_ms = AP_HAL::millis();
260  }
261 
262  // if we are tacking we maintain the target heading until the tack completes or times out
263  if (currently_tacking) {
264  // check if we have reached target
265  if (fabsf(wrap_PI(tack_heading_rad - rover.ahrs.yaw)) <= radians(SAILBOAT_TACKING_ACCURACY_DEG)) {
266  clear_tack();
267  } else if ((now - auto_tack_start_ms) > SAILBOAT_AUTO_TACKING_TIMEOUT_MS) {
268  // tack has taken too long
269  gcs().send_text(MAV_SEVERITY_INFO, "Sailboat: Tacking timed out");
270  clear_tack();
271  }
272  // return tack target heading
273  return degrees(tack_heading_rad) * 100.0f;
274  }
275 
276  // return closest possible heading to wind
277  if (current_tack == TACK_PORT) {
278  return degrees(left_no_go_heading_rad) * 100.0f;
279  } else {
280  return degrees(right_no_go_heading_rad) * 100.0f;
281  }
