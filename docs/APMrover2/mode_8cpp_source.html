343 // calculate pilot input to nudge speed up or down
344 // target_speed should be in meters/sec
345 // reversed should be true if the vehicle is intentionally backing up which allows the pilot to increase the backing up speed by pulling the throttle stick down
346 float Mode::calc_speed_nudge(float target_speed, bool reversed)
347 {
348  // return immediately during RC/GCS failsafe
349  if (rover.failsafe.bits & FAILSAFE_EVENT_THROTTLE) {
350  return target_speed;
351  }
352 
353  // return immediately if pilot is not attempting to nudge speed
354  // pilot can nudge up speed if throttle (in range -100 to +100) is above 50% of center in direction of travel
355  const int16_t pilot_throttle = constrain_int16(rover.channel_throttle->get_control_in(), -100, 100);
356  if (((pilot_throttle <= 50) && !reversed) ||
357  ((pilot_throttle >= -50) && reversed)) {
358  return target_speed;
359  }
360 
361  // sanity checks
362  if (g.throttle_cruise > 100 || g.throttle_cruise < 5) {
363  return target_speed;
364  }
365 
366  // project vehicle's maximum speed
367  const float vehicle_speed_max = calc_speed_max(g.speed_cruise, g.throttle_cruise * 0.01f);
368 
369  // return unadjusted target if already over vehicle's projected maximum speed
370  if (fabsf(target_speed) >= vehicle_speed_max) {
371  return target_speed;
372  }
373 
374  const float speed_increase_max = vehicle_speed_max - fabsf(target_speed);
375  float speed_nudge = ((static_cast<float>(abs(pilot_throttle)) - 50.0f) * 0.02f) * speed_increase_max;
376  if (pilot_throttle < 0) {
377  speed_nudge = -speed_nudge;
378  }
379 
380  return target_speed + speed_nudge;
381 }
