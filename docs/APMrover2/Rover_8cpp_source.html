158 // update AHRS system
159 void Rover::ahrs_update()
160 {
161  arming.update_soft_armed();
162 
163 #if HIL_MODE != HIL_MODE_DISABLED
164  // update hil before AHRS update
165  gcs().update();
166 #endif
167 
168  // AHRS may use movement to calculate heading
169  update_ahrs_flyforward();
170 
171  ahrs.update();
172 
173  // update position
174  have_position = ahrs.get_position(current_loc);
175 
176  // set home from EKF if necessary and possible
177  if (!ahrs.home_is_set()) {
178  if (!set_home_to_current_location(false)) {
179  // ignore this failure
180  }
181  }
182 
183  // if using the EKF get a speed update now (from accelerometers)
184  Vector3f velocity;
185  if (ahrs.get_velocity_NED(velocity)) {
186  ground_speed = norm(velocity.x, velocity.y);
187  } else if (gps.status() >= AP_GPS::GPS_OK_FIX_3D) {
188  ground_speed = ahrs.groundspeed();
189  }
190 
191  if (should_log(MASK_LOG_ATTITUDE_FAST)) {
192  Log_Write_Attitude();
193  Log_Write_Sail();
194  }
195 
196  if (should_log(MASK_LOG_IMU)) {
197  logger.Write_IMU();
198  }
199 }

201 /*
202  check for GCS failsafe - 10Hz
203  */
204 void Rover::gcs_failsafe_check(void)
205 {
206  if (!g.fs_gcs_enabled) {
207  // gcs failsafe disabled
208  return;
209  }
210 
211  // check for updates from GCS within 2 seconds
212  failsafe_trigger(FAILSAFE_EVENT_GCS, failsafe.last_heartbeat_ms != 0 && (millis() - failsafe.last_heartbeat_ms) > 2000);
213 }

259 /*
260  once a second events
261  */
262 void Rover::one_second_loop(void)
263 {
264  // allow orientation change at runtime to aid config
265  ahrs.update_orientation();
266 
267  set_control_channels();
268 
269  // cope with changes to aux functions
270  SRV_Channels::enable_aux_servos();
271 
272  // update notify flags
273  AP_Notify::flags.pre_arm_check = arming.pre_arm_checks(false);
274  AP_Notify::flags.pre_arm_gps_check = true;
275  AP_Notify::flags.armed = arming.is_armed() || arming.arming_required() == AP_Arming::Required::NO;
276  AP_Notify::flags.flying = hal.util->get_soft_armed();
277 
278  // cope with changes to mavlink system ID
279  mavlink_system.sysid = g.sysid_this_mav;
280 
281  // attempt to update home position and baro calibration if not armed:
282  if (!hal.util->get_soft_armed()) {
283  update_home();
284  }
285 
286  // init compass location for declination
287  init_compass_location();
288 
289  // update error mask of sensors and subsystems. The mask uses the
290  // MAV_SYS_STATUS_* values from mavlink. If a bit is set then it
291  // indicates that the sensor or subsystem is present but not
292  // functioning correctly
293  gcs().update_sensor_status_flags();
294 
295  // need to set "likely flying" when armed to allow for compass
296  // learning to run
297  ahrs.set_likely_flying(hal.util->get_soft_armed());
298 
299  // send latest param values to wp_nav
300  g2.wp_nav.set_turn_params(g.turn_max_g, g2.turn_radius, g2.motors.have_skid_steering());
301 }

310  camera.update();
