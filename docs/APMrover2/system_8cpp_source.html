208 // update the ahrs flyforward setting which can allow
209 // the vehicle's movements to be used to estimate heading
210 void Rover::update_ahrs_flyforward()
211 {
212  bool flyforward = false;
213 
214  // boats never use movement to estimate heading
215  if (!is_boat()) {
216  // throttle threshold is 15% or 1/2 cruise throttle
217  bool throttle_over_thresh = g2.motors.get_throttle() > MIN(g.throttle_cruise * 0.50f, 15.0f);
218  // desired speed threshold of 1m/s
219  bool desired_speed_over_thresh = g2.attitude_control.speed_control_active() && (g2.attitude_control.get_desired_speed() > 0.5f);
220  if (throttle_over_thresh || (is_positive(g2.motors.get_throttle()) && desired_speed_over_thresh)) {
221  uint32_t now = AP_HAL::millis();
222  // if throttle over threshold start timer
223  if (flyforward_start_ms == 0) {
224  flyforward_start_ms = now;
225  }
226  // if throttle over threshold for 2 seconds set flyforward to true
227  flyforward = (now - flyforward_start_ms > 2000);
228  } else {
229  // reset timer
230  flyforward_start_ms = 0;
231  }
232  }
233 
234  ahrs.set_fly_forward(flyforward);
235 }

237 bool Rover::set_mode(Mode &new_mode, mode_reason_t reason)
238 {
239  if (control_mode == &new_mode) {
240  // don't switch modes if we are already in the correct mode.
241  return true;
242  }
243 
244  Mode &old_mode = *control_mode;
245  if (!new_mode.enter()) {
246  // Log error that we failed to enter desired flight mode
247  AP::logger().Write_Error(LogErrorSubsystem::FLIGHT_MODE,
248  LogErrorCode(new_mode.mode_number()));
249  gcs().send_text(MAV_SEVERITY_WARNING, "Flight mode change failed");
250  return false;
251  }
252 
253  control_mode = &new_mode;
254 
255  // pilot requested flight mode change during a fence breach indicates pilot is attempting to manually recover
256  // this flight mode change could be automatic (i.e. fence, battery, GPS or GCS failsafe)
257  // but it should be harmless to disable the fence temporarily in these situations as well
258  g2.fence.manual_recovery_start();
259 
260 #if CAMERA == ENABLED
261  camera.set_is_auto_mode(control_mode->mode_number() == Mode::Number::AUTO);
262 #endif
263 
264  old_mode.exit();
265 
266  control_mode_reason = reason;
267  logger.Write_Mode(control_mode->mode_number(), control_mode_reason);
268  gcs().send_message(MSG_HEARTBEAT);
269 
270  notify_mode(control_mode);
271  return true;
272 }
