17  uint8_t _base_mode = MAV_MODE_FLAG_CUSTOM_MODE_ENABLED;
18 
19  // work out the base_mode. This value is not very useful
20  // for APM, but we calculate it as best we can so a generic
21  // MAVLink enabled ground station can work out something about
22  // what the MAV is up to. The actual bit values are highly
23  // ambiguous for most of the APM flight modes. In practice, you
24  // only get useful information from the custom_mode, which maps to
25  // the APM flight mode and has a well defined meaning in the
26  // ArduPlane documentation
27  if (rover.control_mode->has_manual_input()) {
28  _base_mode |= MAV_MODE_FLAG_MANUAL_INPUT_ENABLED;
29  }
30 
31  if (rover.control_mode->is_autopilot_mode()) {
32  _base_mode |= MAV_MODE_FLAG_GUIDED_ENABLED;
33  }
34 
35  if (rover.g2.stick_mixing > 0 && rover.control_mode != &rover.mode_initializing) {
36  // all modes except INITIALISING have some form of manual
37  // override if stick mixing is enabled
38  _base_mode |= MAV_MODE_FLAG_MANUAL_INPUT_ENABLED;
39  }
40 
41 #if HIL_MODE != HIL_MODE_DISABLED
42  _base_mode |= MAV_MODE_FLAG_HIL_ENABLED;
43 #endif
44 
45  // we are armed if we are not initialising
46  if (rover.control_mode != &rover.mode_initializing && rover.arming.is_armed()) {
47  _base_mode |= MAV_MODE_FLAG_SAFETY_ARMED;
48  }
