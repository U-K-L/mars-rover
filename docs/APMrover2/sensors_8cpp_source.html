50 // update wheel encoders
51 void Rover::update_wheel_encoder()
52 {
53  // exit immediately if not enabled
54  if (g2.wheel_encoder.num_sensors() == 0) {
55  return;
56  }
57 
58  // update encoders
59  g2.wheel_encoder.update();
60 
61  // initialise on first iteration
62  const uint32_t now = AP_HAL::millis();
63  if (wheel_encoder_last_ekf_update_ms == 0) {
64  wheel_encoder_last_ekf_update_ms = now;
65  for (uint8_t i = 0; i < g2.wheel_encoder.num_sensors(); i++) {
66  wheel_encoder_last_angle_rad[i] = g2.wheel_encoder.get_delta_angle(i);
67  wheel_encoder_last_update_ms[i] = g2.wheel_encoder.get_last_reading_ms(i);
68  }
69  return;
70  }
71 
72  // calculate delta angle and delta time and send to EKF
73  // use time of last ping from wheel encoder
74  // send delta time (time between this wheel encoder time and previous wheel encoder time)
75  // in case where wheel hasn't moved, count = 0 (cap the delta time at 50ms - or system time)
76  // use system clock of last update instead of time of last ping
77  const float system_dt = (now - wheel_encoder_last_ekf_update_ms) / 1000.0f;
78  for (uint8_t i = 0; i < g2.wheel_encoder.num_sensors(); i++) {
79  // calculate angular change (in radians)
80  const float curr_angle_rad = g2.wheel_encoder.get_delta_angle(i);
81  const float delta_angle = curr_angle_rad - wheel_encoder_last_angle_rad[i];
82  wheel_encoder_last_angle_rad[i] = curr_angle_rad;
83 
84  // save cumulative distances at current time (in meters)
85  wheel_encoder_last_distance_m[i] = g2.wheel_encoder.get_distance(i);
86 
87  // calculate delta time
88  float delta_time;
89  const uint32_t latest_sensor_update_ms = g2.wheel_encoder.get_last_reading_ms(i);
90  const uint32_t sensor_diff_ms = latest_sensor_update_ms - wheel_encoder_last_update_ms[i];
91 
92  // if we have not received any sensor updates, or time difference is too high then use time since last update to the ekf
93  // check for old or insane sensor update times
94  if (sensor_diff_ms == 0 || sensor_diff_ms > 100) {
95  delta_time = system_dt;
96  wheel_encoder_last_update_ms[i] = wheel_encoder_last_ekf_update_ms;
97  } else {
98  delta_time = sensor_diff_ms / 1000.0f;
99  wheel_encoder_last_update_ms[i] = latest_sensor_update_ms;
100  }
101 
102  /* delAng is the measured change in angular position from the previous measurement where a positive rotation is produced by forward motion of the vehicle (rad)
103  * delTime is the time interval for the measurement of delAng (sec)
104  * timeStamp_ms is the time when the rotation was last measured (msec)
105  * posOffset is the XYZ body frame position of the wheel hub (m)
106  */
107  EKF3.writeWheelOdom(delta_angle, delta_time, wheel_encoder_last_update_ms[i], g2.wheel_encoder.get_pos_offset(i), g2.wheel_encoder.get_wheel_radius(i));
108  }
109 
110  // record system time update for next iteration
111  wheel_encoder_last_ekf_update_ms = now;
112 }

134  AP_RangeFinder_Backend *s1 = rangefinder.get_backend(1);
135 
136  if (s0 == nullptr || s0->status() == RangeFinder::RangeFinder_NotConnected) {
137  // this makes it possible to disable rangefinder at runtime
138  return;
139  }
140 
141  if (s1 != nullptr && s1->has_data()) {
142  // we have two rangefinders
143  obstacle.rangefinder1_distance_cm = s0->distance_cm();
144  obstacle.rangefinder2_distance_cm = s1->distance_cm();
145  if (obstacle.rangefinder1_distance_cm < static_cast<uint16_t>(g.rangefinder_trigger_cm) &&
146  obstacle.rangefinder1_distance_cm < static_cast<uint16_t>(obstacle.rangefinder2_distance_cm)) {
147  // we have an object on the left
148  if (obstacle.detected_count < 127) {
149  obstacle.detected_count++;
150  }
151  if (obstacle.detected_count == g.rangefinder_debounce) {
152  gcs().send_text(MAV_SEVERITY_INFO, "Rangefinder1 obstacle %u cm",
153  static_cast<uint32_t>(obstacle.rangefinder1_distance_cm));
154  }
155  obstacle.detected_time_ms = AP_HAL::millis();
156  obstacle.turn_angle = g.rangefinder_turn_angle;
157  } else if (obstacle.rangefinder2_distance_cm < static_cast<uint16_t>(g.rangefinder_trigger_cm)) {
158  // we have an object on the right
159  if (obstacle.detected_count < 127) {
160  obstacle.detected_count++;
161  }
162  if (obstacle.detected_count == g.rangefinder_debounce) {
163  gcs().send_text(MAV_SEVERITY_INFO, "Rangefinder2 obstacle %u cm",
164  static_cast<uint32_t>(obstacle.rangefinder2_distance_cm));
165  }
166  obstacle.detected_time_ms = AP_HAL::millis();
167  obstacle.turn_angle = -g.rangefinder_turn_angle;
168  }
169  } else {
170  // we have a single rangefinder
171  obstacle.rangefinder1_distance_cm = s0->distance_cm();
172  obstacle.rangefinder2_distance_cm = 0;
173  if (obstacle.rangefinder1_distance_cm < static_cast<uint16_t>(g.rangefinder_trigger_cm)) {
174  // obstacle detected in front
175  if (obstacle.detected_count < 127) {
176  obstacle.detected_count++;
177  }
178  if (obstacle.detected_count == g.rangefinder_debounce) {
179  gcs().send_text(MAV_SEVERITY_INFO, "Rangefinder obstacle %u cm",
180  static_cast<uint32_t>(obstacle.rangefinder1_distance_cm));
181  }
182  obstacle.detected_time_ms = AP_HAL::millis();
183  obstacle.turn_angle = g.rangefinder_turn_angle;
184  }
185  }
186 
187  Log_Write_Rangefinder();
188  Log_Write_Depth();
189 
190  // no object detected - reset after the turn time
191  if (obstacle.detected_count >= g.rangefinder_debounce &&
192  AP_HAL::millis() > obstacle.detected_time_ms + g.rangefinder_turn_time*1000) {
193  gcs().send_text(MAV_SEVERITY_INFO, "Obstacle passed");
194  obstacle.detected_count = 0;
195  obstacle.turn_angle = 0;
196  }
197 }
